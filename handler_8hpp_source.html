331  template <typename KernelName, typename KernelType, int Dims,
332  typename LambdaArgType = sycl::detail::lambda_arg_type<KernelType>>
333  void StoreLambda(KernelType KernelFunc) {
334  MHostKernel.reset(
335  new detail::HostKernel<KernelType, LambdaArgType, Dims>(KernelFunc));
336 
337  using KI = sycl::detail::KernelInfo<KernelName>;
338  // Empty name indicates that the compilation happens without integration
339  // header, so don't perform things that require it.
340  if (KI::getName() != "") {
341  MArgs.clear();
342  extractArgsAndReqsFromLambda(MHostKernel->getPtr(), KI::getNumParams(),
343  &KI::getParamDesc(0));
344  MKernelName = KI::getName();
345  MOSModuleHandle = detail::OSUtil::getOSModuleHandle(KI::getName());
346  } else {
347  // In case w/o the integration header it is necessary to process
348  // accessors from the list(which are associated with this handler) as
349  // arguments.
350  MArgs = std::move(MAssociatedAccesors);
351  }
352  }
