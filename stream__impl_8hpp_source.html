182 template <typename T>
183 EnableIfFP<T, unsigned> floatingPointToDecStr(T AbsVal, char *Digits,
184  int Precision, bool IsSci) {
185  int Exp = 0;
186 
187  // For the case that the value is larger than 10.0
188  while (AbsVal >= 10.0) {
189  ++Exp;
190  AbsVal /= 10.0;
191  }
192  // For the case that the value is less than 1.0
193  while (AbsVal > 0.0 && AbsVal < 1.0) {
194  --Exp;
195  AbsVal *= 10.0;
196  }
197 
198  auto IntegralPart = static_cast<int>(AbsVal);
199  auto FractionPart = AbsVal - IntegralPart;
200 
201  int FractionDigits[MAX_FLOATING_POINT_DIGITS] = {0};
202 
203  // Exponent
204  int P = Precision > 0 ? Precision : 4;
205  size_t FractionLength = Exp + P;
206 
207  // After normalization integral part contains 1 symbol, also there could be
208  // '.', 'e', sign of the exponent and sign of the number, overall 5 symbols.
209  // So, clamp fraction length if required according to maximum size of the
210  // buffer for floating point number.
211  if (FractionLength > MAX_FLOATING_POINT_DIGITS - 5)
212  FractionLength = MAX_FLOATING_POINT_DIGITS - 5;
213 
214  for (unsigned I = 0; I < FractionLength; ++I) {
215  FractionPart *= 10.0;
216  FractionDigits[I] = static_cast<int>(FractionPart);
217  FractionPart -= static_cast<int>(FractionPart);
218  }
219 
220  int Carry = FractionPart > static_cast<T>(0.5) ? 1 : 0;
221 
222  // Propagate the Carry
223  for (int I = FractionLength - 1; I >= 0 && Carry; --I) {
224  auto Digit = FractionDigits[I] + Carry;
225  FractionDigits[I] = Digit % 10;
226  Carry = Digit / 10;
227  }
228 
229  // Carry from the fraction part is propagated to integral part
230  IntegralPart += Carry;
231  if (IntegralPart == 10) {
232  IntegralPart = 1;
233  ++Exp;
234  }
235 
236  unsigned Offset = 0;
237 
238  // Assemble the final string correspondingly
239  if (IsSci) { // scientific mode
240  // Append the integral part
241  Digits[Offset++] = digitToChar(IntegralPart);
242  Digits[Offset++] = '.';
243 
244  // Append all fraction
245  for (unsigned I = 0; I < FractionLength; ++I)
246  Digits[Offset++] = digitToChar(FractionDigits[I]);
247 
248  // Exponent part
249  Digits[Offset++] = 'e';
250  Digits[Offset++] = Exp >= 0 ? '+' : '-';
251  Digits[Offset++] = digitToChar(abs(Exp) / 10);
252  Digits[Offset++] = digitToChar(abs(Exp) % 10);
253  } else { // normal mode
254  if (Exp < 0) {
255  Digits[Offset++] = '0';
256  Digits[Offset++] = '.';
257  while (++Exp)
258  Digits[Offset++] = '0';
259 
260  // Append the integral part
261  Digits[Offset++] = digitToChar(IntegralPart);
262 
263  // Append all fraction
264  for (unsigned I = 0; I < FractionLength; ++I)
265  Digits[Offset++] = digitToChar(FractionDigits[I]);
266  } else {
267  // Append the integral part
268  Digits[Offset++] = digitToChar(IntegralPart);
269  unsigned I = 0;
270  // Append the integral part first
271  for (; I < FractionLength && Exp--; ++I)
272  Digits[Offset++] = digitToChar(FractionDigits[I]);
273 
274  // Put the dot
275  Digits[Offset++] = '.';
276 
277  // Append the rest of fraction part, or the real fraction part
278  for (; I < FractionLength; ++I)
279  Digits[Offset++] = digitToChar(FractionDigits[I]);
280  }
281  // The normal mode requires no tailing zero digit, then we need to first
282  // find the first non-zero digit
283  while (Digits[Offset - 1] == '0')
284  Offset--;
285 
286  // If dot is the last digit, it should be stripped off as well
287  if (Digits[Offset - 1] == '.')
288  Offset--;
289  }
290  return Offset;
291 }

293 // Helper method to update offset in the global buffer atomically according to
294 // the provided size of the data in the flush buffer. Return true if offset is
295 // updated and false in case of overflow.
296 inline bool updateOffset(stream_impl::GlobalOffsetAccessorT &GlobalOffset,
297  stream_impl::GlobalBufAccessorT &GlobalBuf,
298  unsigned Size, unsigned &Cur) {
299  unsigned New;
300  Cur = GlobalOffset[0].load();
301  do {
302  if (GlobalBuf.get_range().size() - Cur < Size)
303  // Overflow
304  return false;
305  New = Cur + Size;
306  } while (!GlobalOffset[0].compare_exchange_strong(Cur, New));
307  return true;
308 }

310 inline void flushBuffer(stream_impl::GlobalOffsetAccessorT &GlobalOffset,
311  stream_impl::GlobalBufAccessorT &GlobalBuf,
312  stream_impl::FlushBufAccessorT &FlushBufs,
313  unsigned &WIOffset, unsigned &Offset) {
314  // Copy data from flush buffer (local memory) to global buffer (global
315  // memory)
316  unsigned Cur = 0;
317  if (!updateOffset(GlobalOffset, GlobalBuf, Offset, Cur))
318  return;
319 
320  for (unsigned I = WIOffset; I < WIOffset + Offset; I++) {
321  GlobalBuf[Cur++] = FlushBufs[I];
322  }
323  // Reset the offset in the flush buffer
324  Offset = 0;
325 }

344 inline void reverseBuf(char *Buf, unsigned Len) {
345  int I = Len - 1;
346  int J = 0;
347  while (I > J) {
348  int Temp = Buf[I];
349  Buf[I] = Buf[J];
350  Buf[J] = Temp;
351  I--;
352  J++;
353  }
354 }

356 inline unsigned append(char *Dst, const char *Src) {
357  unsigned Len = 0;
358  for (; Src[Len] != '\0'; ++Len)
359  ;
360 
361  for (unsigned I = 0; I < Len; ++I)
362  Dst[I] = Src[I];
363  return Len;
364 }

366 template <typename T>
367 inline typename std::enable_if<std::is_same<T, half>::value, unsigned>::type
368 checkForInfNan(char *Buf, T Val) {
369  if (Val != Val)
370  return append(Buf, "nan");
371 
372  // Extract the sign from the bits
373  const uint16_t Sign = reinterpret_cast<uint16_t &>(Val) & 0x8000;
374  // Extract the exponent from the bits
375  const uint16_t Exp16 = (reinterpret_cast<uint16_t &>(Val) & 0x7c00) >> 10;
376 
377  if (Exp16 == 0x1f) {
378  if (Sign)
379  return append(Buf, "-inf");
380  return append(Buf, "inf");
381  }
382  return 0;
383 }

400 // Returns number of symbols written to the buffer
401 template <typename T>
402 inline EnableIfFP<T, unsigned> ScalarToStr(const T &Val, char *Buf,
403  unsigned Flags, int Width,
404  int Precision = -1) {
405  unsigned Offset = checkForInfNan(Buf, Val);
406  if (Offset)
407  return Offset;
408 
409  T Neg = -Val;
410  auto AbsVal = Val < 0 ? Neg : Val;
411 
412  if (Val < 0) {
413  Buf[Offset++] = '-';
414  } else if (Flags & ShowPos) {
415  Buf[Offset++] = '+';
416  }
417 
418  bool IsSci = false;
419  if (Flags & detail::Scientific)
420  IsSci = true;
421 
422  // TODO: manipulators for floating-point output - hexfloat, fixed
423  Offset += floatingPointToDecStr(AbsVal, Buf + Offset, Precision, IsSci);
424 
425  return Offset;
426 }

429 template <typename T>
430 inline typename std::enable_if<std::is_integral<T>::value, unsigned>::type
431 ScalarToStr(const T &Val, char *Buf, unsigned Flags, int Width,
432  int Precision = -1) {
433  int Base = 10;
434 
435  // append base manipulator
436  switch (Flags & BaseField) {
437  case Dec:
438  Base = 10;
439  break;
440  case Hex:
441  Base = 16;
442  break;
443  case Oct:
444  Base = 8;
445  break;
446  default:
447  // default value is 10
448  break;
449  }
450 
451  unsigned Offset = 0;
452 
453  // write '+' to the stream if the base is 10 and the value is non-negative
454  // or write '-' to stream if base is 10 and the value is negative
455  if (Base == 10) {
456  if ((Flags & ShowPos) && Val >= 0)
457  Buf[Offset++] = '+';
458  else if (Val < 0)
459  Buf[Offset++] = '-';
460  }
461 
462  // write 0 or 0x to the stream if base is not 10 and the manipulator is set
463  if (Base != 10 && (Flags & ShowBase)) {
464  Buf[Offset++] = '0';
465  if (Base == 16)
466  Buf[Offset++] = 'x';
467  }
468 
469  auto AbsVal = getAbsVal(Val, Base);
470 
471  const unsigned NumBuf = integralToBase(AbsVal, Base, Buf + Offset);
472 
473  reverseBuf(Buf + Offset, NumBuf);
474  return Offset + NumBuf;
475 }
