32 struct LineEntry {

35 
36  /// The presumed line number of this line entry: \#line 4.
37  unsigned LineNo;

41  int FilenameID;
42 
43  /// Set the 0 if no flags, 1 if a system header,

82  ///
83  /// This allows us to unique the filenames that
