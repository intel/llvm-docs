30  const InputInfo &Output,

33  const char *LinkingOutput) const {

50 void solaris::Linker::ConstructJob(Compilation &C, const JobAction &JA,
51  const InputInfo &Output,
52  const InputInfoList &Inputs,
53  const ArgList &Args,
54  const char *LinkingOutput) const {
55  ArgStringList CmdArgs;
56 
57  // Demangle C++ names in errors
58  CmdArgs.push_back("-C");
59 
60  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_shared)) {
61  CmdArgs.push_back("-e");
62  CmdArgs.push_back("_start");
63  }
64 
65  if (Args.hasArg(options::OPT_static)) {
66  CmdArgs.push_back("-Bstatic");
67  CmdArgs.push_back("-dn");
68  } else {
69  CmdArgs.push_back("-Bdynamic");
70  if (Args.hasArg(options::OPT_shared)) {
71  CmdArgs.push_back("-shared");
72  }
73 
74  // libpthread has been folded into libc since Solaris 10, no need to do
75  // anything for pthreads. Claim argument to avoid warning.
76  Args.ClaimAllArgs(options::OPT_pthread);
77  Args.ClaimAllArgs(options::OPT_pthreads);
78  }
79 
80  if (Output.isFilename()) {
81  CmdArgs.push_back("-o");
82  CmdArgs.push_back(Output.getFilename());
83  } else {
84  assert(Output.isNothing() && "Invalid output.");
85  }
86 
87  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nostartfiles)) {
88  if (!Args.hasArg(options::OPT_shared))
89  CmdArgs.push_back(
90  Args.MakeArgString(getToolChain().GetFilePath("crt1.o")));
91 
92  CmdArgs.push_back(Args.MakeArgString(getToolChain().GetFilePath("crti.o")));
93 
94  const Arg *Std = Args.getLastArg(options::OPT_std_EQ, options::OPT_ansi);
95  bool HaveAnsi = false;
96  const LangStandard *LangStd = nullptr;
97  if (Std) {
98  HaveAnsi = Std->getOption().matches(options::OPT_ansi);
99  if (!HaveAnsi)
100  LangStd = LangStandard::getLangStandardForName(Std->getValue());
101  }
102 
103  const char *values_X = "values-Xa.o";
104  // Use values-Xc.o for -ansi, -std=c*, -std=iso9899:199409.
105  if (HaveAnsi || (LangStd && !LangStd->isGNUMode()))
106  values_X = "values-Xc.o";
107  CmdArgs.push_back(Args.MakeArgString(getToolChain().GetFilePath(values_X)));
108 
109  const char *values_xpg = "values-xpg6.o";
110  // Use values-xpg4.o for -std=c90, -std=gnu90, -std=iso9899:199409.
111  if (LangStd && LangStd->getLanguage() == Language::C && !LangStd->isC99())
112  values_xpg = "values-xpg4.o";
113  CmdArgs.push_back(
114  Args.MakeArgString(getToolChain().GetFilePath(values_xpg)));
115  CmdArgs.push_back(
116  Args.MakeArgString(getToolChain().GetFilePath("crtbegin.o")));
117  }
118 
119  getToolChain().AddFilePathLibArgs(Args, CmdArgs);
120 
121  Args.AddAllArgs(CmdArgs, {options::OPT_L, options::OPT_T_Group,
122  options::OPT_e, options::OPT_r});
123 
124  bool NeedsSanitizerDeps = addSanitizerRuntimes(getToolChain(), Args, CmdArgs);
125  AddLinkerInputs(getToolChain(), Inputs, Args, CmdArgs, JA);
126 
127  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nodefaultlibs)) {
128  if (getToolChain().ShouldLinkCXXStdlib(Args))
129  getToolChain().AddCXXStdlibLibArgs(Args, CmdArgs);
130  if (Args.hasArg(options::OPT_fstack_protector) ||
131  Args.hasArg(options::OPT_fstack_protector_strong) ||
132  Args.hasArg(options::OPT_fstack_protector_all)) {
133  // Explicitly link ssp libraries, not folded into Solaris libc.
134  CmdArgs.push_back("-lssp_nonshared");
135  CmdArgs.push_back("-lssp");
136  }
137  // LLVM support for atomics on 32-bit SPARC V8+ is incomplete, so
138  // forcibly link with libatomic as a workaround.
139  if (getToolChain().getTriple().getArch() == llvm::Triple::sparc) {
140  CmdArgs.push_back(getAsNeededOption(getToolChain(), true));
141  CmdArgs.push_back("-latomic");
142  CmdArgs.push_back(getAsNeededOption(getToolChain(), false));
143  }
144  CmdArgs.push_back("-lgcc_s");
145  CmdArgs.push_back("-lc");
146  if (!Args.hasArg(options::OPT_shared)) {
147  CmdArgs.push_back("-lgcc");
148  CmdArgs.push_back("-lm");
149  }
150  if (NeedsSanitizerDeps) {
151  linkSanitizerRuntimeDeps(getToolChain(), CmdArgs);
152 
153  // Work around Solaris/amd64 ld bug when calling __tls_get_addr directly.
154  // However, ld -z relax=transtls is available since Solaris 11.2, but not
155  // in Illumos.
156  const SanitizerArgs &SA = getToolChain().getSanitizerArgs(Args);
157  if (getToolChain().getTriple().getArch() == llvm::Triple::x86_64 &&
158  (SA.needsAsanRt() || SA.needsStatsRt() ||
159  (SA.needsUbsanRt() && !SA.requiresMinimalRuntime())))
160  CmdArgs.push_back("-zrelax=transtls");
161  }
162  }
163 
164  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nostartfiles)) {
165  CmdArgs.push_back(
166  Args.MakeArgString(getToolChain().GetFilePath("crtend.o")));
167  }
168  CmdArgs.push_back(Args.MakeArgString(getToolChain().GetFilePath("crtn.o")));
169 
170  getToolChain().addProfileRTLibs(Args, CmdArgs);
171 
172  const char *Exec = Args.MakeArgString(getToolChain().GetLinkerPath());
173  C.addCommand(std::make_unique<Command>(JA, *this, ResponseFileSupport::None(),
174  Exec, CmdArgs, Inputs, Output));
175 }

177 static StringRef getSolarisLibSuffix(const llvm::Triple &Triple) {
178  switch (Triple.getArch()) {
179  case llvm::Triple::x86:
180  case llvm::Triple::sparc:
181  break;
182  case llvm::Triple::x86_64:
183  return "/amd64";
184  case llvm::Triple::sparcv9:
185  return "/sparcv9";
186  default:
187  llvm_unreachable("Unsupported architecture");
188  }
189  return "";
190 }

194 Solaris::Solaris(const Driver &D, const llvm::Triple &Triple,
195  const ArgList &Args)
196  : Generic_ELF(D, Triple, Args) {
197 
198  GCCInstallation.init(Triple, Args);
199 
200  StringRef LibSuffix = getSolarisLibSuffix(Triple);
201  path_list &Paths = getFilePaths();
202  if (GCCInstallation.isValid()) {
203  // On Solaris gcc uses both an architecture-specific path with triple in it
204  // as well as a more generic lib path (+arch suffix).
205  addPathIfExists(D,
206  GCCInstallation.getInstallPath() +
207  GCCInstallation.getMultilib().gccSuffix(),
208  Paths);
209  addPathIfExists(D, GCCInstallation.getParentLibPath() + LibSuffix, Paths);
210  }
211 
212  // If we are currently running Clang inside of the requested system root,
213  // add its parent library path to those searched.
214  if (StringRef(D.Dir).startswith(D.SysRoot))
215  addPathIfExists(D, D.Dir + "/../lib", Paths);
216 
217  addPathIfExists(D, D.SysRoot + "/usr/lib" + LibSuffix, Paths);
218 }

220 SanitizerMask Solaris::getSupportedSanitizers() const {
221  const bool IsX86 = getTriple().getArch() == llvm::Triple::x86;
222  const bool IsX86_64 = getTriple().getArch() == llvm::Triple::x86_64;
223  SanitizerMask Res = ToolChain::getSupportedSanitizers();
224  // FIXME: Omit X86_64 until 64-bit support is figured out.
225  if (IsX86) {
226  Res |= SanitizerKind::Address;
227  Res |= SanitizerKind::PointerCompare;
228  Res |= SanitizerKind::PointerSubtract;
229  }
230  if (IsX86 || IsX86_64)
231  Res |= SanitizerKind::Function;
232  Res |= SanitizerKind::Vptr;
233  return Res;
234 }

242 void Solaris::AddClangSystemIncludeArgs(const ArgList &DriverArgs,
243  ArgStringList &CC1Args) const {
244  const Driver &D = getDriver();
245 
246  if (DriverArgs.hasArg(clang::driver::options::OPT_nostdinc))
247  return;
248 
249  if (!DriverArgs.hasArg(options::OPT_nostdlibinc))
250  addSystemInclude(DriverArgs, CC1Args, D.SysRoot + "/usr/local/include");
251 
252  if (!DriverArgs.hasArg(options::OPT_nobuiltininc)) {
253  SmallString<128> P(D.ResourceDir);
254  llvm::sys::path::append(P, "include");
255  addSystemInclude(DriverArgs, CC1Args, P);
256  }
257 
258  if (DriverArgs.hasArg(options::OPT_nostdlibinc))
259  return;
260 
261  // Check for configure-time C include directories.
262  StringRef CIncludeDirs(C_INCLUDE_DIRS);
263  if (CIncludeDirs != "") {
264  SmallVector<StringRef, 5> dirs;
265  CIncludeDirs.split(dirs, ":");
266  for (StringRef dir : dirs) {
267  StringRef Prefix =
268  llvm::sys::path::is_absolute(dir) ? "" : StringRef(D.SysRoot);
269  addExternCSystemInclude(DriverArgs, CC1Args, Prefix + dir);
270  }
271  return;
272  }
273 
274  // Add include directories specific to the selected multilib set and multilib.
275  if (GCCInstallation.isValid()) {
276  const MultilibSet::IncludeDirsFunc &Callback =
277  Multilibs.includeDirsCallback();
278  if (Callback) {
279  for (const auto &Path : Callback(GCCInstallation.getMultilib()))
280  addExternCSystemIncludeIfExists(
281  DriverArgs, CC1Args, GCCInstallation.getInstallPath() + Path);
282  }
283  }
284 
285  addExternCSystemInclude(DriverArgs, CC1Args, D.SysRoot + "/usr/include");
286 }

288 void Solaris::addLibStdCxxIncludePaths(
289  const llvm::opt::ArgList &DriverArgs,
290  llvm::opt::ArgStringList &CC1Args) const {
291  // We need a detected GCC installation on Solaris (similar to Linux)
292  // to provide libstdc++'s headers.
293  if (!GCCInstallation.isValid())
294  return;
295 
296  // By default, look for the C++ headers in an include directory adjacent to
297  // the lib directory of the GCC installation.
298  // On Solaris this usually looks like /usr/gcc/X.Y/include/c++/X.Y.Z
299  StringRef LibDir = GCCInstallation.getParentLibPath();
300  StringRef TripleStr = GCCInstallation.getTriple().str();
301  const Multilib &Multilib = GCCInstallation.getMultilib();
302  const GCCVersion &Version = GCCInstallation.getVersion();
303 
304  // The primary search for libstdc++ supports multiarch variants.
305  addLibStdCXXIncludePaths(LibDir.str() + "/../include/c++/" + Version.Text,
306  TripleStr, Multilib.includeSuffix(), DriverArgs,
307  CC1Args);
308 }
