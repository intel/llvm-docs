42  /// When printing type to be inserted into code in specific context, this
43  /// callback can be used to avoid printing the redundant part of the
44  /// qualifier. For example, when inserting code inside namespace foo, we
45  /// should print bar::SomeType instead of foo::bar::SomeType.

100  /// The number of spaces to use to indent each line.

103  /// Whether we should suppress printing of the actual specifiers for
104  /// the given type or declaration.
105  ///
106  /// This flag is only used when we are printing declarators beyond
107  /// the first declarator within a declaration group. For example, given:
108  ///
109  /// \code
110  /// const int *x, *y;
111  /// \endcode
112  ///
113  /// SuppressSpecifiers will be false when printing the
114  /// declaration for "x", so that we will print "int *x"; it will be
115  /// \c true when we print "y", so that we suppress printing the
116  /// "const int" type specifier and instead only print the "*y".
117  unsigned SuppressSpecifiers : 1;
118 
119  /// Whether type printing should skip printing the tag keyword.
120  ///
121  /// This is used when printing the inner type of elaborated types,
122  /// (as the tag keyword is part of the elaborated type):
123  ///
124  /// \code
125  /// struct Geometry::Point;
126  /// \endcode
127  unsigned SuppressTagKeyword : 1;
128
