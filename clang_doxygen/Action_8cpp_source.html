76 void Action::propagateDeviceOffloadInfo(OffloadKind OKind, const char *OArch,
77  const ToolChain *OToolChain) {
78  // Offload action set its own kinds on their dependences.
79  if (Kind == OffloadClass)
80  return;
81  // Unbundling actions use the host kinds.
82  if (Kind == OffloadUnbundlingJobClass)
83  return;
84  // Deps job uses the host kinds.
85  if (Kind == OffloadDepsJobClass)
86  return;
87 
88  assert((OffloadingDeviceKind == OKind || OffloadingDeviceKind == OFK_None) &&
89  "Setting device kind to a different device??");
90  assert(!ActiveOffloadKindMask && "Setting a device kind in a host action??");
91  OffloadingDeviceKind = OKind;
92  OffloadingArch = OArch;
93  OffloadingToolChain = OToolChain;
94 
95  for (auto *A : Inputs)
96  A->propagateDeviceOffloadInfo(OffloadingDeviceKind, OArch, OToolChain);
97 }

99 void Action::propagateHostOffloadInfo(unsigned OKinds, const char *OArch) {
100  // Offload action set its own kinds on their dependences.
101  if (Kind == OffloadClass)
102  return;
103 
104  assert(OffloadingDeviceKind == OFK_None &&
105  "Setting a host kind in a device action.");
106  ActiveOffloadKindMask |= OKinds;
107  OffloadingArch = OArch;
108 
109  for (auto *A : Inputs)
110  A->propagateHostOffloadInfo(ActiveOffloadKindMask, OArch);
111 }

122 std::string Action::getOffloadingKindPrefix() const {
123  switch (OffloadingDeviceKind) {
124  case OFK_None:
125  break;
126  case OFK_Host:
127  llvm_unreachable("Host kind is not an offloading device kind.");
128  break;
129  case OFK_Cuda:
130  return "device-cuda";
131  case OFK_OpenMP:
132  return "device-openmp";
133  case OFK_HIP:
134  return "device-hip";
135  case OFK_SYCL:
136  return "device-sycl";
137 
138  // TODO: Add other programming models here.
139  }
140 
141  if (!ActiveOffloadKindMask)
142  return {};
143 
144  std::string Res("host");
145  assert(!((ActiveOffloadKindMask & OFK_Cuda) &&
146  (ActiveOffloadKindMask & OFK_HIP)) &&
147  "Cannot offload CUDA and HIP at the same time");
148  if (ActiveOffloadKindMask & OFK_Cuda)
149  Res += "-cuda";
150  if (ActiveOffloadKindMask & OFK_HIP)
151  Res += "-hip";
152  if (ActiveOffloadKindMask & OFK_OpenMP)
153  Res += "-openmp";
154  if (ActiveOffloadKindMask & OFK_SYCL)
155  Res += "-sycl";
156 
157  // TODO: Add other programming models here.
158 
159  return Res;
160 }

221 OffloadAction::OffloadAction(const DeviceDependences &DDeps, types::ID Ty)
222  : Action(OffloadClass, DDeps.getActions(), Ty),
223  DevToolChains(DDeps.getToolChains()) {
224  auto &OKinds = DDeps.getOffloadKinds();
225  auto &BArchs = DDeps.getBoundArchs();
226  auto &OTCs = DDeps.getToolChains();
227 
228  // If all inputs agree on the same kind, use it also for this action.
229  if (llvm::all_equal(OKinds))
230  OffloadingDeviceKind = OKinds.front();
231 
232  // If we have a single dependency, inherit the architecture from it.
233  if (OKinds.size() == 1)
234  OffloadingArch = BArchs.front();
235 
236  // Propagate info to the dependencies.
237  for (unsigned i = 0, e = getInputs().size(); i != e; ++i)
238  getInputs()[i]->propagateDeviceOffloadInfo(OKinds[i], BArchs[i], OTCs[i]);
239 }

241 OffloadAction::OffloadAction(const HostDependence &HDep,
242  const DeviceDependences &DDeps)
243  : Action(OffloadClass, HDep.getAction()), HostTC(HDep.getToolChain()),
244  DevToolChains(DDeps.getToolChains()) {
245  // We use the kinds of the host dependence for this action.
246  OffloadingArch = HDep.getBoundArch();
247  ActiveOffloadKindMask = HDep.getOffloadKinds();
248  HDep.getAction()->propagateHostOffloadInfo(HDep.getOffloadKinds(),
249  HDep.getBoundArch());
250 
251  // Add device inputs and propagate info to the device actions. Do work only if
252  // we have dependencies.
253  for (unsigned i = 0, e = DDeps.getActions().size(); i != e; ++i) {
254  if (auto *A = DDeps.getActions()[i]) {
255  getInputs().push_back(A);
256  A->propagateDeviceOffloadInfo(DDeps.getOffloadKinds()[i],
257  DDeps.getBoundArchs()[i],
258  DDeps.getToolChains()[i]);
259  // If this action is used to forward single dependency, set the toolchain.
260  if (DDeps.getActions().size() == 1)
261  OffloadingToolChain = DDeps.getToolChains()[i];
262  }
263  }

274 void OffloadAction::doOnEachDeviceDependence(
275  const OffloadActionWorkTy &Work) const {
276  auto I = getInputs().begin();
277  auto E = getInputs().end();
278  if (I == E)
279  return;
280 
281  // We expect to have the same number of input dependences and device tool
282  // chains, except if we also have a host dependence. In that case we have one
283  // more dependence than we have device tool chains.
284  assert(getInputs().size() == DevToolChains.size() + (HostTC ? 1 : 0) &&
285  "Sizes of action dependences and toolchains are not consistent!");
286 
287  // Skip host action
288  if (HostTC)
289  ++I;
290 
291  auto TI = DevToolChains.begin();
292  for (; I != E; ++I, ++TI)
293  Work(*I, *TI, (*I)->getOffloadingArch());
294 }
