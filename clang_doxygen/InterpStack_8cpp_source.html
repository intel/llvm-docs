69 void InterpStack::shrink(size_t Size) {
70  assert(Chunk && "Chunk is empty!");
71 
72  while (Size > Chunk->size()) {
73  Size -= Chunk->size();
74  if (Chunk->Next) {
75  std::free(Chunk->Next);

77  }
78  Chunk->End = Chunk->start();
