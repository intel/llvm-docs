532 /// Floating point control options
533 class FPOptionsOverride;
534 class FPOptions {
535 public:
536  // We start by defining the layout.
537  using storage_type = uint16_t;
538 
539  using RoundingMode = llvm::RoundingMode;
540 
541  static constexpr unsigned StorageBitSize = 8 * sizeof(storage_type);
542 
543  // Define a fake option named "First" so that we have a PREVIOUS even for the
544  // real first option.
545  static constexpr storage_type FirstShift = 0, FirstWidth = 0;
546 #define OPTION(NAME, TYPE, WIDTH, PREVIOUS) \
547  static constexpr storage_type NAME##Shift = \
548  PREVIOUS##Shift + PREVIOUS##Width; \
549  static constexpr storage_type NAME##Width = WIDTH; \
550  static constexpr storage_type NAME##Mask = ((1 << NAME##Width) - 1) \
551  << NAME##Shift;
552 #include "clang/Basic/FPOptions.def"
553 
554  static constexpr storage_type TotalWidth = 0
555 #define OPTION(NAME, TYPE, WIDTH, PREVIOUS) +WIDTH
556 #include "clang/Basic/FPOptions.def"
557  ;
558  static_assert(TotalWidth <= StorageBitSize, "Too short type for FPOptions");
559 
560 private:
561  storage_type Value;
562 
563 public:
564  FPOptions() : Value(0) {
565  setFPContractMode(LangOptions::FPM_Off);
566  setRoundingMode(static_cast<RoundingMode>(LangOptions::FPR_ToNearest));
567  setFPExceptionMode(LangOptions::FPE_Ignore);
568  }
569  explicit FPOptions(const LangOptions &LO) {
570  Value = 0;
571  // The language fp contract option FPM_FastHonorPragmas has the same effect
572  // as FPM_Fast in frontend. For simplicity, use FPM_Fast uniformly in
573  // frontend.
574  auto LangOptContractMode = LO.getDefaultFPContractMode();
575  if (LangOptContractMode == LangOptions::FPM_FastHonorPragmas)
576  LangOptContractMode = LangOptions::FPM_Fast;
577  setFPContractMode(LangOptContractMode);
578  setRoundingMode(LO.getFPRoundingMode());
579  setFPExceptionMode(LO.getFPExceptionMode());
580  setAllowFPReassociate(LO.AllowFPReassoc);
581  setNoHonorNaNs(LO.NoHonorNaNs);
582  setNoHonorInfs(LO.NoHonorInfs);
583  setNoSignedZero(LO.NoSignedZero);
584  setAllowReciprocal(LO.AllowRecip);
585  setAllowApproxFunc(LO.ApproxFunc);
586  if (getFPContractMode() == LangOptions::FPM_On &&
587  getRoundingMode() == llvm::RoundingMode::Dynamic &&
588  getFPExceptionMode() == LangOptions::FPE_Strict)
589  // If the FP settings are set to the "strict" model, then
590  // FENV access is set to true. (ffp-model=strict)
591  setAllowFEnvAccess(true);
592  else
593  setAllowFEnvAccess(LangOptions::FPM_Off);
594  }
595 
596  bool allowFPContractWithinStatement() const {
597  return getFPContractMode() == LangOptions::FPM_On;
598  }
599  void setAllowFPContractWithinStatement() {
600  setFPContractMode(LangOptions::FPM_On);
601  }
602 
603  bool allowFPContractAcrossStatement() const {
604  return getFPContractMode() == LangOptions::FPM_Fast;
605  }
606  void setAllowFPContractAcrossStatement() {
607  setFPContractMode(LangOptions::FPM_Fast);
608  }
609 
610  bool isFPConstrained() const {
611  return getRoundingMode() != llvm::RoundingMode::NearestTiesToEven ||
612  getFPExceptionMode() != LangOptions::FPE_Ignore ||
613  getAllowFEnvAccess();
614  }

641 /// Represents difference between two FPOptions values.
642 ///
643 /// The effect of language constructs changing the set of floating point options
644 /// is usually a change of some FP properties while leaving others intact. This
645 /// class describes such changes by keeping information about what FP options
646 /// are overridden.
647 ///
648 /// The integral set of FP options, described by the class FPOptions, may be
649 /// represented as a default FP option set, defined by language standard and
650 /// command line options, with the overrides introduced by pragmas.
651 ///
652 /// The is implemented as a value of the new FPOptions plus a mask showing which
653 /// fields are actually set in it.
654 class FPOptionsOverride {
655  FPOptions Options = FPOptions::getFromOpaqueInt(0);
656  FPOptions::storage_type OverrideMask = 0;
657 
658 public:
659  using RoundingMode = llvm::RoundingMode;
660 
661  /// The type suitable for storing values of FPOptionsOverride. Must be twice
662  /// as wide as bit size of FPOption.
663  using storage_type = uint32_t;
664  static_assert(sizeof(storage_type) >= 2 * sizeof(FPOptions::storage_type),
665  "Too short type for FPOptionsOverride");
666 
667  /// Bit mask selecting bits of OverrideMask in serialized representation of
668  /// FPOptionsOverride.
669  static constexpr storage_type OverrideMaskBits =
670  (static_cast<storage_type>(1) << FPOptions::StorageBitSize) - 1;
671 
672  FPOptionsOverride() {}
673  FPOptionsOverride(const LangOptions &LO)
674  : Options(LO), OverrideMask(OverrideMaskBits) {}
675  FPOptionsOverride(FPOptions FPO)
676  : Options(FPO), OverrideMask(OverrideMaskBits) {}
