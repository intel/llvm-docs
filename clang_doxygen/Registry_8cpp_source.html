63  assert(Constructors.find(MatcherName) == Constructors.end());

643 std::vector<ArgKind> Registry::getAcceptedCompletionTypes(
644  ArrayRef<std::pair<MatcherCtor, unsigned>> Context) {
645  ASTNodeKind InitialTypes[] = {
646  ASTNodeKind::getFromNodeKind<Decl>(),
647  ASTNodeKind::getFromNodeKind<QualType>(),
648  ASTNodeKind::getFromNodeKind<Type>(),
649  ASTNodeKind::getFromNodeKind<Stmt>(),
650  ASTNodeKind::getFromNodeKind<NestedNameSpecifier>(),
651  ASTNodeKind::getFromNodeKind<NestedNameSpecifierLoc>(),
652  ASTNodeKind::getFromNodeKind<TypeLoc>()};
653 
654  // Starting with the above seed of acceptable top-level matcher types, compute
655  // the acceptable type set for the argument indicated by each context element.
656  std::set<ArgKind> TypeSet;
657  for (auto IT : InitialTypes) {
658  TypeSet.insert(ArgKind::MakeMatcherArg(IT));
659  }
660  for (const auto &CtxEntry : Context) {
661  MatcherCtor Ctor = CtxEntry.first;
662  unsigned ArgNumber = CtxEntry.second;
663  std::vector<ArgKind> NextTypeSet;
664  for (const ArgKind &Kind : TypeSet) {
665  if (Kind.getArgKind() == Kind.AK_Matcher &&
666  Ctor->isConvertibleTo(Kind.getMatcherKind()) &&
667  (Ctor->isVariadic() || ArgNumber < Ctor->getNumArgs()))
668  Ctor->getArgKinds(Kind.getMatcherKind(), ArgNumber, NextTypeSet);
669  }
670  TypeSet.clear();
671  TypeSet.insert(NextTypeSet.begin(), NextTypeSet.end());
672  }
673  return std::vector<ArgKind>(TypeSet.begin(), TypeSet.end());
674 }

676 std::vector<MatcherCompletion>
677 Registry::getMatcherCompletions(ArrayRef<ArgKind> AcceptedTypes) {
678  std::vector<MatcherCompletion> Completions;
679 
680  // Search the registry for acceptable matchers.
681  for (const auto &M : RegistryData->constructors()) {
682  const MatcherDescriptor& Matcher = *M.getValue();
683  StringRef Name = M.getKey();
684 
685  std::set<ASTNodeKind> RetKinds;
686  unsigned NumArgs = Matcher.isVariadic() ? 1 : Matcher.getNumArgs();
687  bool IsPolymorphic = Matcher.isPolymorphic();
688  std::vector<std::vector<ArgKind>> ArgsKinds(NumArgs);
689  unsigned MaxSpecificity = 0;
690  bool NodeArgs = false;
691  for (const ArgKind& Kind : AcceptedTypes) {
692  if (Kind.getArgKind() != Kind.AK_Matcher &&
693  Kind.getArgKind() != Kind.AK_Node) {
694  continue;
695  }
696 
697  if (Kind.getArgKind() == Kind.AK_Node) {
698  NodeArgs = true;
699  unsigned Specificity;
700  ASTNodeKind LeastDerivedKind;
701  if (Matcher.isConvertibleTo(Kind.getNodeKind(), &Specificity,
702  &LeastDerivedKind)) {
703  if (MaxSpecificity < Specificity)
704  MaxSpecificity = Specificity;
705  RetKinds.insert(LeastDerivedKind);
706  for (unsigned Arg = 0; Arg != NumArgs; ++Arg)
707  Matcher.getArgKinds(Kind.getNodeKind(), Arg, ArgsKinds[Arg]);
708  if (IsPolymorphic)
709  break;
710  }
711  } else {
712  unsigned Specificity;
713  ASTNodeKind LeastDerivedKind;
714  if (Matcher.isConvertibleTo(Kind.getMatcherKind(), &Specificity,
715  &LeastDerivedKind)) {
716  if (MaxSpecificity < Specificity)
717  MaxSpecificity = Specificity;
718  RetKinds.insert(LeastDerivedKind);
719  for (unsigned Arg = 0; Arg != NumArgs; ++Arg)
720  Matcher.getArgKinds(Kind.getMatcherKind(), Arg, ArgsKinds[Arg]);
721  if (IsPolymorphic)
722  break;
723  }
724  }
725  }
726 
727  if (!RetKinds.empty() && MaxSpecificity > 0) {
728  std::string Decl;
729  llvm::raw_string_ostream OS(Decl);
730 
731  std::string TypedText = std::string(Name);
732 
733  if (NodeArgs) {
734  OS << Name;
735  } else {
736 
737  if (IsPolymorphic) {
738  OS << "Matcher<T> " << Name << "(Matcher<T>";
739  } else {
740  OS << "Matcher<" << RetKinds << "> " << Name << "(";
741  for (const std::vector<ArgKind> &Arg : ArgsKinds) {
742  if (&Arg != &ArgsKinds[0])
743  OS << ", ";
744 
745  bool FirstArgKind = true;
746  std::set<ASTNodeKind> MatcherKinds;
747  // Two steps. First all non-matchers, then matchers only.
748  for (const ArgKind &AK : Arg) {
749  if (AK.getArgKind() == ArgKind::AK_Matcher) {
750  MatcherKinds.insert(AK.getMatcherKind());
751  } else {
752  if (!FirstArgKind)
753  OS << "|";
754  FirstArgKind = false;
755  OS << AK.asString();
756  }
757  }
758  if (!MatcherKinds.empty()) {
759  if (!FirstArgKind) OS << "|";
760  OS << "Matcher<" << MatcherKinds << ">";
761  }
762  }
763  }
764  if (Matcher.isVariadic())
765  OS << "...";
766  OS << ")";
767 
768  TypedText += "(";
769  if (ArgsKinds.empty())
770  TypedText += ")";
771  else if (ArgsKinds[0][0].getArgKind() == ArgKind::AK_String)
772  TypedText += "\"";
773  }
774 
775  Completions.emplace_back(TypedText, OS.str(), MaxSpecificity);
776  }
777  }
778 
779  return Completions;
780 }
