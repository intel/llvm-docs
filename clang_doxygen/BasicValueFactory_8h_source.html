141  /// Returns the type of the APSInt used to store values of the given QualType.
142  APSIntType getAPSIntType(QualType T) const {
143  // For the purposes of the analysis and constraints, we treat atomics
144  // as their underlying types.
145  if (const AtomicType *AT = T->getAs<AtomicType>()) {
146  T = AT->getValueType();
147  }
148 
149  assert(T->isIntegralOrEnumerationType() || Loc::isLocType(T));
150  return APSIntType(Ctx.getIntWidth(T),
151  !T->isSignedIntegerOrEnumerationType());
152  }
153 
154  /// Convert - Create a new persistent APSInt with the same value as 'From'
155  /// but with the bitwidth and signedness of 'To'.
156  const llvm::APSInt &Convert(const llvm::APSInt& To,
157  const llvm::APSInt& From) {
158  APSIntType TargetType(To);
159  if (TargetType == APSIntType(From))
160  return From;
161 
162  return getValue(TargetType.convert(From));
163  }
164 
165  const llvm::APSInt &Convert(QualType T, const llvm::APSInt &From) {
166  APSIntType TargetType = getAPSIntType(T);
167  return Convert(TargetType, From);
168  }

175  }
