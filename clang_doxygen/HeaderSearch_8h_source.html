58 /// The preprocessor keeps track of this information for each
59 /// file that is \#included.
60 struct HeaderFileInfo {
61  // TODO: Whether the file was imported is not a property of the file itself.
62  // It's a preprocessor state, move it there.
63  /// True if this is a \#import'd file.
64  unsigned isImport : 1;
65 
66  /// True if this is a \#pragma once file.
67  unsigned isPragmaOnce : 1;
68 
69  /// Keep track of whether this is a system header, and if so,
70  /// whether it is C++ clean or not. This can be set by the include paths or
71  /// by \#pragma gcc system_header. This is an instance of
72  /// SrcMgr::CharacteristicKind.
73  unsigned DirInfo : 3;
74 
75  /// Whether this header file info was supplied by an external source,
76  /// and has not changed since.
77  unsigned External : 1;
78 
79  /// Whether this header is part of a module.
80  unsigned isModuleHeader : 1;
81 
82  /// Whether this header is part of the module that we are building.
83  unsigned isCompilingModuleHeader : 1;
84 
85  /// Whether this structure is considered to already have been
86  /// "resolved", meaning that it was loaded from the external source.
87  unsigned Resolved : 1;
88 
89  /// Whether this is a header inside a framework that is currently
90  /// being built.
91  ///
92  /// When a framework is being built, the headers have not yet been placed
93  /// into the appropriate framework subdirectories, and therefore are
94  /// provided via a header map. This bit indicates when this is one of
95  /// those framework headers.
96  unsigned IndexHeaderMapHeader : 1;
97 
98  /// Whether this file has been looked up as a header.
99  unsigned IsValid : 1;
100 
101  /// The ID number of the controlling macro.
102  ///
103  /// This ID number will be non-zero when there is a controlling
104  /// macro whose IdentifierInfo may not yet have been loaded from
105  /// external storage.
106  unsigned ControllingMacroID = 0;
107 
108  /// If this file has a \#ifndef XXX (or equivalent) guard that
109  /// protects the entire contents of the file, this is the identifier
110  /// for the macro that controls whether or not it has any effect.
111  ///
112  /// Note: Most clients should use getControllingMacro() to access
113  /// the controlling macro of this header, since
114  /// getControllingMacro() is able to load a controlling macro from
115  /// external storage.
116  const IdentifierInfo *ControllingMacro = nullptr;
117 
118  /// If this header came from a framework include, this is the name
119  /// of the framework.
120  StringRef Framework;
121 
122  /// List of aliases that this header is known as.
123  /// Most headers should only have at most one alias, but a handful
124  /// have two.
125  llvm::SetVector<llvm::SmallString<32>,
126  llvm::SmallVector<llvm::SmallString<32>, 2>,
127  llvm::SmallSet<llvm::SmallString<32>, 2>>
128  Aliases;
129 
130  HeaderFileInfo()
131  : isImport(false), isPragmaOnce(false), DirInfo(SrcMgr::C_User),
132  External(false), isModuleHeader(false), isCompilingModuleHeader(false),
133  Resolved(false), IndexHeaderMapHeader(false), IsValid(false) {}
134 
135  /// Retrieve the controlling macro for this header file, if
136  /// any.
137  const IdentifierInfo *
138  getControllingMacro(ExternalPreprocessorSource *External);
139 };

170 /// Forward iterator over the search directories of \c HeaderSearch.
171 template <bool IsConst>
172 struct SearchDirIteratorImpl
173  : llvm::iterator_facade_base<SearchDirIteratorImpl<IsConst>,
174  std::forward_iterator_tag,
175  Qualified<IsConst, DirectoryLookup>> {
176  /// Const -> non-const iterator conversion.
177  template <typename Enable = std::enable_if<IsConst, bool>>
178  SearchDirIteratorImpl(const SearchDirIteratorImpl<false> &Other)
179  : HS(Other.HS), Idx(Other.Idx) {}
180 
181  SearchDirIteratorImpl(const SearchDirIteratorImpl &) = default;
182 
183  SearchDirIteratorImpl &operator=(const SearchDirIteratorImpl &) = default;
184 
185  bool operator==(const SearchDirIteratorImpl &RHS) const {
186  return HS == RHS.HS && Idx == RHS.Idx;
187  }
188 
189  SearchDirIteratorImpl &operator++() {
190  assert(*this && "Invalid iterator.");
191  ++Idx;
192  return *this;
193  }
194 
195  Qualified<IsConst, DirectoryLookup> &operator*() const {
196  assert(*this && "Invalid iterator.");
197  return HS->SearchDirs[Idx];
198  }
199 
200  /// Creates an invalid iterator.
201  SearchDirIteratorImpl(std::nullptr_t) : HS(nullptr), Idx(0) {}
202 
203  /// Checks whether the iterator is valid.
204  explicit operator bool() const { return HS != nullptr; }
205 
206 private:
207  /// The parent \c HeaderSearch. This is \c nullptr for invalid iterator.
208  Qualified<IsConst, HeaderSearch> *HS;
209 
210  /// The index of the current element.
211  size_t Idx;
212 
213  /// The constructor that creates a valid iterator.
214  SearchDirIteratorImpl(Qualified<IsConst, HeaderSearch> &HS, size_t Idx)
215  : HS(&HS), Idx(Idx) {}
216 
217  /// Only HeaderSearch is allowed to instantiate valid iterators.
218  friend HeaderSearch;
219 
220  /// Enables const -> non-const conversion.
221  friend SearchDirIteratorImpl<!IsConst>;
222 };
