63  llvm::Error ParseAndExecute(llvm::StringRef Code) {
64  auto PTU = Parse(Code);
65  if (!PTU)
66  return PTU.takeError();
67  if (PTU->TheModule)
68  return Execute(*PTU);
69  return llvm::Error::success();
70  }
