78 static std::string getHighestNumericTupleInDirectory(llvm::vfs::FileSystem &VFS,
79  StringRef Directory) {
80  std::string Highest;
81  llvm::VersionTuple HighestTuple;
82 
83  std::error_code EC;
84  for (llvm::vfs::directory_iterator DirIt = VFS.dir_begin(Directory, EC),
85  DirEnd;
86  !EC && DirIt != DirEnd; DirIt.increment(EC)) {
87  auto Status = VFS.status(DirIt->path());
88  if (!Status || !Status->isDirectory())
89  continue;
90  StringRef CandidateName = llvm::sys::path::filename(DirIt->path());
91  llvm::VersionTuple Tuple;
92  if (Tuple.tryParse(CandidateName)) // tryParse() returns true on error.
93  continue;
94  if (Tuple > HighestTuple) {
95  HighestTuple = Tuple;
96  Highest = CandidateName.str();
97  }
98  }
99 
100  return Highest;
101 }

131 // Check various environment variables to try and find a toolchain.
132 static bool
133 findVCToolChainViaEnvironment(llvm::vfs::FileSystem &VFS, std::string &Path,
134  MSVCToolChain::ToolsetLayout &VSLayout) {
135  // These variables are typically set by vcvarsall.bat
136  // when launching a developer command prompt.
137  if (llvm::Optional<std::string> VCToolsInstallDir =
138  llvm::sys::Process::GetEnv("VCToolsInstallDir")) {
139  // This is only set by newer Visual Studios, and it leads straight to
140  // the toolchain directory.
141  Path = std::move(*VCToolsInstallDir);
142  VSLayout = MSVCToolChain::ToolsetLayout::VS2017OrNewer;
143  return true;
144  }
145  if (llvm::Optional<std::string> VCInstallDir =
146  llvm::sys::Process::GetEnv("VCINSTALLDIR")) {
147  // If the previous variable isn't set but this one is, then we've found
148  // an older Visual Studio. This variable is set by newer Visual Studios too,
149  // so this check has to appear second.
150  // In older Visual Studios, the VC directory is the toolchain.
151  Path = std::move(*VCInstallDir);
152  VSLayout = MSVCToolChain::ToolsetLayout::OlderVS;
153  return true;
154  }
155 
156  // We couldn't find any VC environment variables. Let's walk through PATH and
157  // see if it leads us to a VC toolchain bin directory. If it does, pick the
158  // first one that we find.
159  if (llvm::Optional<std::string> PathEnv =
160  llvm::sys::Process::GetEnv("PATH")) {
161  llvm::SmallVector<llvm::StringRef, 8> PathEntries;
162  llvm::StringRef(*PathEnv).split(PathEntries, llvm::sys::EnvPathSeparator);
163  for (llvm::StringRef PathEntry : PathEntries) {
164  if (PathEntry.empty())
165  continue;
166 
167  llvm::SmallString<256> ExeTestPath;
168 
169  // If cl.exe doesn't exist, then this definitely isn't a VC toolchain.
170  ExeTestPath = PathEntry;
171  llvm::sys::path::append(ExeTestPath, "cl.exe");
172  if (!VFS.exists(ExeTestPath))
173  continue;
174 
175  // cl.exe existing isn't a conclusive test for a VC toolchain; clang also
176  // has a cl.exe. So let's check for link.exe too.
177  ExeTestPath = PathEntry;
178  llvm::sys::path::append(ExeTestPath, "link.exe");
179  if (!VFS.exists(ExeTestPath))
180  continue;
181 
182  // whatever/VC/bin --> old toolchain, VC dir is toolchain dir.
183  llvm::StringRef TestPath = PathEntry;
184  bool IsBin =
185  llvm::sys::path::filename(TestPath).equals_insensitive("bin");
186  if (!IsBin) {
187  // Strip any architecture subdir like "amd64".
188  TestPath = llvm::sys::path::parent_path(TestPath);
189  IsBin = llvm::sys::path::filename(TestPath).equals_insensitive("bin");
190  }
191  if (IsBin) {
192  llvm::StringRef ParentPath = llvm::sys::path::parent_path(TestPath);
193  llvm::StringRef ParentFilename = llvm::sys::path::filename(ParentPath);
194  if (ParentFilename.equals_insensitive("VC")) {
195  Path = std::string(ParentPath);
196  VSLayout = MSVCToolChain::ToolsetLayout::OlderVS;
197  return true;
198  }
199  if (ParentFilename.equals_insensitive("x86ret") ||
200  ParentFilename.equals_insensitive("x86chk") ||
201  ParentFilename.equals_insensitive("amd64ret") ||
202  ParentFilename.equals_insensitive("amd64chk")) {
203  Path = std::string(ParentPath);
204  VSLayout = MSVCToolChain::ToolsetLayout::DevDivInternal;
205  return true;
206  }
207 
208  } else {
209  // This could be a new (>=VS2017) toolchain. If it is, we should find
210  // path components with these prefixes when walking backwards through
211  // the path.
212  // Note: empty strings match anything.
213  llvm::StringRef ExpectedPrefixes[] = {"", "Host", "bin", "",
214  "MSVC", "Tools", "VC"};
215 
216  auto It = llvm::sys::path::rbegin(PathEntry);
217  auto End = llvm::sys::path::rend(PathEntry);
218  for (llvm::StringRef Prefix : ExpectedPrefixes) {
219  if (It == End)
220  goto NotAToolChain;
221  if (!It->startswith_insensitive(Prefix))
222  goto NotAToolChain;
223  ++It;
224  }
225 
226  // We've found a new toolchain!
227  // Back up 3 times (/bin/Host/arch) to get the root path.
228  llvm::StringRef ToolChainPath(PathEntry);
229  for (int i = 0; i < 3; ++i)
230  ToolChainPath = llvm::sys::path::parent_path(ToolChainPath);
231 
232  Path = std::string(ToolChainPath);
233  VSLayout = MSVCToolChain::ToolsetLayout::VS2017OrNewer;
234  return true;
235  }
236 
237  NotAToolChain:
238  continue;
239  }
240  }
241  return false;
242 }

244 // Query the Setup Config server for installs, then pick the newest version
245 // and find its default VC toolchain.
246 // This is the preferred way to discover new Visual Studios, as they're no
247 // longer listed in the registry.
248 static bool
249 findVCToolChainViaSetupConfig(llvm::vfs::FileSystem &VFS, std::string &Path,
250  MSVCToolChain::ToolsetLayout &VSLayout) {
251 #if !defined(USE_MSVC_SETUP_API)
252  return false;
253 #else
254  // FIXME: This really should be done once in the top-level program's main
255  // function, as it may have already been initialized with a different
256  // threading model otherwise.
257  llvm::sys::InitializeCOMRAII COM(llvm::sys::COMThreadingMode::SingleThreaded);
258  HRESULT HR;
259 
260  // _com_ptr_t will throw a _com_error if a COM calls fail.
261  // The LLVM coding standards forbid exception handling, so we'll have to
262  // stop them from being thrown in the first place.
263  // The destructor will put the regular error handler back when we leave
264  // this scope.
265  struct SuppressCOMErrorsRAII {
266  static void __stdcall handler(HRESULT hr, IErrorInfo *perrinfo) {}
267 
268  SuppressCOMErrorsRAII() { _set_com_error_handler(handler); }
269 
270  ~SuppressCOMErrorsRAII() { _set_com_error_handler(_com_raise_error); }
271 
272  } COMErrorSuppressor;
273 
274  ISetupConfigurationPtr Query;
275  HR = Query.CreateInstance(__uuidof(SetupConfiguration));
276  if (FAILED(HR))
277  return false;
278 
279  IEnumSetupInstancesPtr EnumInstances;
280  HR = ISetupConfiguration2Ptr(Query)->EnumAllInstances(&EnumInstances);
281  if (FAILED(HR))
282  return false;
283 
284  ISetupInstancePtr Instance;
285  HR = EnumInstances->Next(1, &Instance, nullptr);
286  if (HR != S_OK)
287  return false;
288 
289  ISetupInstancePtr NewestInstance;
290  Optional<uint64_t> NewestVersionNum;
291  do {
292  bstr_t VersionString;
293  uint64_t VersionNum;
294  HR = Instance->GetInstallationVersion(VersionString.GetAddress());
295  if (FAILED(HR))
296  continue;
297  HR = ISetupHelperPtr(Query)->ParseVersion(VersionString, &VersionNum);
298  if (FAILED(HR))
299  continue;
300  if (!NewestVersionNum || (VersionNum > NewestVersionNum)) {
301  NewestInstance = Instance;
302  NewestVersionNum = VersionNum;
303  }
304  } while ((HR = EnumInstances->Next(1, &Instance, nullptr)) == S_OK);
305 
306  if (!NewestInstance)
307  return false;
308 
309  bstr_t VCPathWide;
310  HR = NewestInstance->ResolvePath(L"VC", VCPathWide.GetAddress());
311  if (FAILED(HR))
312  return false;
313 
314  std::string VCRootPath;
315  llvm::convertWideToUTF8(std::wstring(VCPathWide), VCRootPath);
316 
317  llvm::SmallString<256> ToolsVersionFilePath(VCRootPath);
318  llvm::sys::path::append(ToolsVersionFilePath, "Auxiliary", "Build",
319  "Microsoft.VCToolsVersion.default.txt");
320 
321  auto ToolsVersionFile = llvm::MemoryBuffer::getFile(ToolsVersionFilePath);
322  if (!ToolsVersionFile)
323  return false;
324 
325  llvm::SmallString<256> ToolchainPath(VCRootPath);
326  llvm::sys::path::append(ToolchainPath, "Tools", "MSVC",
327  ToolsVersionFile->get()->getBuffer().rtrim());
328  auto Status = VFS.status(ToolchainPath);
329  if (!Status || !Status->isDirectory())
330  return false;
331 
332  Path = std::string(ToolchainPath.str());
333  VSLayout = MSVCToolChain::ToolsetLayout::VS2017OrNewer;
334  return true;
335 #endif
336 }

409 void visualstudio::Linker::ConstructJob(Compilation &C, const JobAction &JA,
410  const InputInfo &Output,
411  const InputInfoList &Inputs,
412  const ArgList &Args,
413  const char *LinkingOutput) const {
414  ArgStringList CmdArgs;
415 
416  // Create a library with -fsycl-link
417  if (Args.hasArg(options::OPT_fsycl_link_EQ) &&
418  JA.getType() == types::TY_Archive) {
419  constructMSVCLibCommand(C, JA, Output, Inputs, Args);

421  }
422 
423  auto &TC = static_cast<const toolchains::MSVCToolChain &>(getToolChain());
424 
425  assert((Output.isFilename() || Output.isNothing()) && "invalid output");
426  if (Output.isFilename())
427  CmdArgs.push_back(
428  Args.MakeArgString(std::string("-out:") + Output.getFilename()));
429 
430  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nostartfiles) &&
431  !C.getDriver().IsCLMode()) {
432  if (Args.hasArg(options::OPT_fsycl) && !Args.hasArg(options::OPT_nolibsycl))
433  CmdArgs.push_back("-defaultlib:msvcrt");
434  else
435  CmdArgs.push_back("-defaultlib:libcmt");
436  CmdArgs.push_back("-defaultlib:oldnames");
437  }
438 
439  if ((!C.getDriver().IsCLMode() && !Args.hasArg(options::OPT_nostdlib) &&
440  Args.hasArg(options::OPT_fsycl) &&
441  !Args.hasArg(options::OPT_nolibsycl)) ||
442  Args.hasArg(options::OPT_fsycl_host_compiler_EQ)) {
443  if (Args.hasArg(options::OPT__SLASH_MDd))
444  CmdArgs.push_back("-defaultlib:sycld.lib");
445  else
446  CmdArgs.push_back("-defaultlib:sycl.lib");
447  }
448 
449  for (const auto *A : Args.filtered(options::OPT_foffload_static_lib_EQ))
450  CmdArgs.push_back(
451  Args.MakeArgString(Twine("-defaultlib:") + A->getValue()));
452  for (const auto *A : Args.filtered(options::OPT_foffload_whole_static_lib_EQ))
453  CmdArgs.push_back(
454  Args.MakeArgString(Twine("-wholearchive:") + A->getValue()));
455 
456  // Suppress multiple section warning LNK4078
457  if (Args.hasFlag(options::OPT_fsycl, options::OPT_fno_sycl, false))
458  CmdArgs.push_back("/IGNORE:4078");
459 
460  // If the VC environment hasn't been configured (perhaps because the user
461  // did not run vcvarsall), try to build a consistent link environment. If
462  // the environment variable is set however, assume the user knows what
463  // they're doing. If the user passes /vctoolsdir or /winsdkdir, trust that

465  if (!llvm::sys::Process::GetEnv("LIB") ||

471  toolchains::MSVCToolChain::SubDirectoryType::Lib)));
472  CmdArgs.push_back(Args.MakeArgString(
473  Twine("-libpath:") +
474  TC.getSubDirectoryPath(toolchains::MSVCToolChain::SubDirectoryType::Lib,
475  "atlmfc")));
476  }
477  if (!llvm::sys::Process::GetEnv("LIB") ||
478  Args.getLastArg(options::OPT__SLASH_winsdkdir,
479  options::OPT__SLASH_winsysroot)) {
480  if (TC.useUniversalCRT()) {
481  std::string UniversalCRTLibPath;
482  if (TC.getUniversalCRTLibraryPath(Args, UniversalCRTLibPath))
483  CmdArgs.push_back(
484  Args.MakeArgString(Twine("-libpath:") + UniversalCRTLibPath));
485  }
486  std::string WindowsSdkLibPath;
487  if (TC.getWindowsSDKLibraryPath(Args, WindowsSdkLibPath))
488  CmdArgs.push_back(
489  Args.MakeArgString(std::string("-libpath:") + WindowsSdkLibPath));
490  }
491 
492  // Add the compiler-rt library directories to libpath if they exist to help
493  // the linker find the various sanitizer, builtin, and profiling runtimes.
494  for (const auto &LibPath : TC.getLibraryPaths()) {
495  if (TC.getVFS().exists(LibPath))
496  CmdArgs.push_back(Args.MakeArgString("-libpath:" + LibPath));
497  }
498  auto CRTPath = TC.getCompilerRTPath();
499  if (TC.getVFS().exists(CRTPath))
500  CmdArgs.push_back(Args.MakeArgString("-libpath:" + CRTPath));
501 
502  if (!C.getDriver().IsCLMode() && Args.hasArg(options::OPT_L))
503  for (const auto &LibPath : Args.getAllArgValues(options::OPT_L))
504  CmdArgs.push_back(Args.MakeArgString("-libpath:" + LibPath));
505 
506  CmdArgs.push_back("-nologo");
507 
508  if (Args.hasArg(options::OPT_g_Group, options::OPT__SLASH_Z7))
509  CmdArgs.push_back("-debug");
510 
511  // Pass on /Brepro if it was passed to the compiler.
512  // Note that /Brepro maps to -mno-incremental-linker-compatible.
513  bool DefaultIncrementalLinkerCompatible =
514  C.getDefaultToolChain().getTriple().isWindowsMSVCEnvironment();
515  if (!Args.hasFlag(options::OPT_mincremental_linker_compatible,
516  options::OPT_mno_incremental_linker_compatible,
517  DefaultIncrementalLinkerCompatible))
518  CmdArgs.push_back("-Brepro");
519 
520  bool DLL = Args.hasArg(options::OPT__SLASH_LD, options::OPT__SLASH_LDd,
521  options::OPT_shared);
522  if (DLL) {
523  CmdArgs.push_back(Args.MakeArgString("-dll"));
524 
525  SmallString<128> ImplibName(Output.getFilename());
526  llvm::sys::path::replace_extension(ImplibName, "lib");
527  CmdArgs.push_back(Args.MakeArgString(std::string("-implib:") + ImplibName));
528  }
529 
530  if (TC.getSanitizerArgs().needsFuzzer()) {
531  if (!Args.hasArg(options::OPT_shared))
532  CmdArgs.push_back(
533  Args.MakeArgString(std::string("-wholearchive:") +
534  TC.getCompilerRTArgString(Args, "fuzzer")));
535  CmdArgs.push_back(Args.MakeArgString("-debug"));
536  // Prevent the linker from padding sections we use for instrumentation
537  // arrays.
538  CmdArgs.push_back(Args.MakeArgString("-incremental:no"));
539  }
540 
541  if (TC.getSanitizerArgs().needsAsanRt()) {
542  CmdArgs.push_back(Args.MakeArgString("-debug"));
543  CmdArgs.push_back(Args.MakeArgString("-incremental:no"));
544  if (TC.getSanitizerArgs().needsSharedRt() ||
545  Args.hasArg(options::OPT__SLASH_MD, options::OPT__SLASH_MDd)) {
546  for (const auto &Lib : {"asan_dynamic", "asan_dynamic_runtime_thunk"})
547  CmdArgs.push_back(TC.getCompilerRTArgString(Args, Lib));
548  // Make sure the dynamic runtime thunk is not optimized out at link time
549  // to ensure proper SEH handling.
550  CmdArgs.push_back(Args.MakeArgString(
551  TC.getArch() == llvm::Triple::x86
552  ? "-include:___asan_seh_interceptor"
553  : "-include:__asan_seh_interceptor"));
554  // Make sure the linker consider all object files from the dynamic runtime
555  // thunk.
556  CmdArgs.push_back(Args.MakeArgString(std::string("-wholearchive:") +

975 /// Read registry string.
976 /// This also supports a means to look for high-versioned keys by use
977 /// of a $VERSION placeholder in the key path.
978 /// $VERSION in the key path is a placeholder for the version number,
979 /// causing the highest value path to be searched for and used.
980 /// I.e. "SOFTWARE\\Microsoft\\VisualStudio\\$VERSION".
981 /// There can be additional characters in the component. Only the numeric
982 /// characters are compared. This function only searches HKLM.
983 static bool getSystemRegistryString(const char *keyPath, const char *valueName,
984  std::string &value, std::string *phValue) {
985 #ifndef _WIN32
986  return false;
987 #else
988  HKEY hRootKey = HKEY_LOCAL_MACHINE;
989  HKEY hKey = NULL;
990  long lResult;
991  bool returnValue = false;
992 
993  const char *placeHolder = strstr(keyPath, "$VERSION");
994  std::string bestName;
995  // If we have a $VERSION placeholder, do the highest-version search.
996  if (placeHolder) {
997  const char *keyEnd = placeHolder - 1;
998  const char *nextKey = placeHolder;
999  // Find end of previous key.
1000  while ((keyEnd > keyPath) && (*keyEnd != '\\'))
1001  keyEnd--;
1002  // Find end of key containing $VERSION.
1003  while (*nextKey && (*nextKey != '\\'))
1004  nextKey++;
1005  size_t partialKeyLength = keyEnd - keyPath;
1006  char partialKey[256];
1007  if (partialKeyLength >= sizeof(partialKey))
1008  partialKeyLength = sizeof(partialKey) - 1;
1009  strncpy(partialKey, keyPath, partialKeyLength);
1010  partialKey[partialKeyLength] = '\0';
1011  HKEY hTopKey = NULL;
1012  lResult = RegOpenKeyExA(hRootKey, partialKey, 0, KEY_READ | KEY_WOW64_32KEY,
1013  &hTopKey);
1014  if (lResult == ERROR_SUCCESS) {
1015  char keyName[256];
1016  double bestValue = 0.0;
1017  DWORD index, size = sizeof(keyName) - 1;
1018  for (index = 0; RegEnumKeyExA(hTopKey, index, keyName, &size, NULL, NULL,
1019  NULL, NULL) == ERROR_SUCCESS;
1020  index++) {
1021  const char *sp = keyName;
1022  while (*sp && !isDigit(*sp))
1023  sp++;
1024  if (!*sp)
1025  continue;
1026  const char *ep = sp + 1;
1027  while (*ep && (isDigit(*ep) || (*ep == '.')))
1028  ep++;
1029  char numBuf[32];
1030  strncpy(numBuf, sp, sizeof(numBuf) - 1);
1031  numBuf[sizeof(numBuf) - 1] = '\0';
1032  double dvalue = strtod(numBuf, NULL);
1033  if (dvalue > bestValue) {
1034  // Test that InstallDir is indeed there before keeping this index.
1035  // Open the chosen key path remainder.
1036  bestName = keyName;
1037  // Append rest of key.
1038  bestName.append(nextKey);
1039  lResult = RegOpenKeyExA(hTopKey, bestName.c_str(), 0,
1040  KEY_READ | KEY_WOW64_32KEY, &hKey);
1041  if (lResult == ERROR_SUCCESS) {
1042  if (readFullStringValue(hKey, valueName, value)) {
1043  bestValue = dvalue;
1044  if (phValue)
1045  *phValue = bestName;
1046  returnValue = true;
1047  }
1048  RegCloseKey(hKey);
1049  }
1050  }
1051  size = sizeof(keyName) - 1;
1052  }
1053  RegCloseKey(hTopKey);
1054  }
1055  } else {
1056  lResult =
1057  RegOpenKeyExA(hRootKey, keyPath, 0, KEY_READ | KEY_WOW64_32KEY, &hKey);
1058  if (lResult == ERROR_SUCCESS) {
1059  if (readFullStringValue(hKey, valueName, value))
1060  returnValue = true;
1061  if (phValue)
1062  phValue->clear();
1063  RegCloseKey(hKey);
1064  }
1065  }
1066  return returnValue;
1067 #endif // _WIN32
1068 }

1475 static void TranslateOptArg(Arg *A, llvm::opt::DerivedArgList &DAL,
1476  bool SupportsForcingFramePointer,
1477  const char *ExpandChar, const OptTable &Opts) {
1478  assert(A->getOption().matches(options::OPT__SLASH_O));
1479 
1480  StringRef OptStr = A->getValue();
1481  for (size_t I = 0, E = OptStr.size(); I != E; ++I) {
1482  const char &OptChar = *(OptStr.data() + I);
1483  switch (OptChar) {
1484  default:
1485  break;
1486  case '1':
1487  case '2':
1488  case 'x':
1489  case 'd':
1490  // Ignore /O[12xd] flags that aren't the last one on the command line.
1491  // Only the last one gets expanded.
1492  if (&OptChar != ExpandChar) {
1493  A->claim();
1494  break;
1495  }
1496  if (OptChar == 'd') {
1497  DAL.AddFlagArg(A, Opts.getOption(options::OPT_O0));
1498  } else {
1499  if (OptChar == '1') {
1500  DAL.AddJoinedArg(A, Opts.getOption(options::OPT_O), "s");
1501  } else if (OptChar == '2' || OptChar == 'x') {
1502  DAL.AddFlagArg(A, Opts.getOption(options::OPT_fbuiltin));
1503  DAL.AddJoinedArg(A, Opts.getOption(options::OPT_O), "2");
1504  }
1505  if (SupportsForcingFramePointer &&
1506  !DAL.hasArgNoClaim(options::OPT_fno_omit_frame_pointer))
1507  DAL.AddFlagArg(A, Opts.getOption(options::OPT_fomit_frame_pointer));
1508  if (OptChar == '1' || OptChar == '2')
1509  DAL.AddFlagArg(A, Opts.getOption(options::OPT_ffunction_sections));
1510  }
1511  break;
1512  case 'b':
1513  if (I + 1 != E && isdigit(OptStr[I + 1])) {
1514  switch (OptStr[I + 1]) {
1515  case '0':
1516  DAL.AddFlagArg(A, Opts.getOption(options::OPT_fno_inline));
1517  break;
1518  case '1':
1519  DAL.AddFlagArg(A, Opts.getOption(options::OPT_finline_hint_functions));
1520  break;
1521  case '2':
1522  DAL.AddFlagArg(A, Opts.getOption(options::OPT_finline_functions));
1523  break;
1524  }
1525  ++I;
1526  }
1527  break;
1528  case 'g':
1529  A->claim();
1530  break;
1531  case 'i':
1532  if (I + 1 != E && OptStr[I + 1] == '-') {
1533  ++I;
1534  DAL.AddFlagArg(A, Opts.getOption(options::OPT_fno_builtin));
1535  } else {
1536  DAL.AddFlagArg(A, Opts.getOption(options::OPT_fbuiltin));
1537  }
1538  break;
1539  case 's':
1540  DAL.AddJoinedArg(A, Opts.getOption(options::OPT_O), "s");
1541  break;
1542  case 't':
1543  DAL.AddJoinedArg(A, Opts.getOption(options::OPT_O), "2");
1544  break;
1545  case 'y': {
1546  bool OmitFramePointer = true;
1547  if (I + 1 != E && OptStr[I + 1] == '-') {
1548  OmitFramePointer = false;
1549  ++I;
1550  }
1551  if (SupportsForcingFramePointer) {
1552  if (OmitFramePointer)
1553  DAL.AddFlagArg(A,
1554  Opts.getOption(options::OPT_fomit_frame_pointer));
1555  else

1557  A, Opts.getOption(options::OPT_fno_omit_frame_pointer));
1558  } else {

1560  // SupportsForcingFramePointer is false). The flag having no effect

1562  // to special-case their build files for x86-64 clang-cl.

1599 llvm::opt::DerivedArgList *
1600 MSVCToolChain::TranslateArgs(const llvm::opt::DerivedArgList &Args,
1601  StringRef BoundArch,
1602  Action::OffloadKind OFK) const {
1603  DerivedArgList *DAL = new DerivedArgList(Args.getBaseArgs());
1604  const OptTable &Opts = getDriver().getOpts();
1605 
1606  // /Oy and /Oy- don't have an effect on X86-64
1607  bool SupportsForcingFramePointer = getArch() != llvm::Triple::x86_64;
1608 
1609  // The -O[12xd] flag actually expands to several flags. We must desugar the
1610  // flags so that options embedded can be negated. For example, the '-O2' flag
1611  // enables '-Oy'. Expanding '-O2' into its constituent flags allows us to
1612  // correctly handle '-O2 -Oy-' where the trailing '-Oy-' disables a single
1613  // aspect of '-O2'.
1614  //
1615  // Note that this expansion logic only applies to the *last* of '[12xd]'.
1616 
1617  // First step is to search for the character we'd like to expand.
1618  const char *ExpandChar = nullptr;
1619  for (Arg *A : Args.filtered(options::OPT__SLASH_O)) {
1620  StringRef OptStr = A->getValue();
1621  for (size_t I = 0, E = OptStr.size(); I != E; ++I) {
1622  char OptChar = OptStr[I];
1623  char PrevChar = I > 0 ? OptStr[I - 1] : '0';
1624  if (PrevChar == 'b') {
1625  // OptChar does not expand; it's an argument to the previous char.
1626  continue;
1627  }
1628  if (OptChar == '1' || OptChar == '2' || OptChar == 'x' || OptChar == 'd')
1629  ExpandChar = OptStr.data() + I;
1630  }
1631  }
1632 
1633  for (Arg *A : Args) {
1634  if (A->getOption().matches(options::OPT__SLASH_O)) {
1635  // The -O flag actually takes an amalgam of other options. For example,
1636  // '/Ogyb2' is equivalent to '/Og' '/Oy' '/Ob2'.
1637  TranslateOptArg(A, *DAL, SupportsForcingFramePointer, ExpandChar, Opts);
1638  } else if (A->getOption().matches(options::OPT_D)) {
1639  // Translate -Dfoo#bar into -Dfoo=bar.
1640  TranslateDArg(A, *DAL, Opts);
1641  } else if (A->getOption().matches(options::OPT__SLASH_permissive)) {
1642  // Expand /permissive
1643  TranslatePermissive(A, *DAL, Opts);
1644  } else if (A->getOption().matches(options::OPT__SLASH_permissive_)) {
1645  // Expand /permissive-
1646  TranslatePermissiveMinus(A, *DAL, Opts);
1647  } else if (OFK != Action::OFK_HIP) {
1648  // HIP Toolchain translates input args by itself.
1649  DAL->append(A);
1650  }
1651  }
1652 
1653  return DAL;
1654 }
