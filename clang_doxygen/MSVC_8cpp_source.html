55 // Try to find Exe from a Visual Studio distribution. This first tries to find
56 // an installed copy of Visual Studio and, failing that, looks in the PATH,
57 // making sure that whatever executable that's found is not a same-named exe
58 // from clang itself to prevent clang from falling back to itself.
59 static std::string FindVisualStudioExecutable(const ToolChain &TC,
60  const char *Exe) {
61  const auto &MSVC = static_cast<const toolchains::MSVCToolChain &>(TC);
62  SmallString<128> FilePath(
63  MSVC.getSubDirectoryPath(llvm::SubDirectoryType::Bin));
64  llvm::sys::path::append(FilePath, Exe);
65  return std::string(canExecute(TC.getVFS(), FilePath) ? FilePath.str() : Exe);
66 }

460  size_t EnvCount = 0;
461  size_t EnvBlockLen = 0;
462  while (EnvBlockWide[EnvBlockLen] != L'\0') {
463  ++EnvCount;
464  EnvBlockLen += std::wcslen(&EnvBlockWide[EnvBlockLen]) +
465  1 /*string null-terminator*/;
466  }
467  ++EnvBlockLen; // add the block null-terminator
468 
469  std::string EnvBlock;

480  // find it.

482  llvm::StringRef EnvVar(Cursor);

485  Environment.push_back(Args.MakeArgString(
486  EnvVar.substr(0, PrefixLen) +
487  TC.getSubDirectoryPath(llvm::SubDirectoryType::Bin) +
488  llvm::Twine(llvm::sys::EnvPathSeparator) +
489  TC.getSubDirectoryPath(llvm::SubDirectoryType::Bin, HostArch) +
490  (EnvVar.size() > PrefixLen
491  ? llvm::Twine(llvm::sys::EnvPathSeparator) +
492  EnvVar.substr(PrefixLen)
493  : "")));
494  } else {
495  Environment.push_back(Args.MakeArgString(EnvVar));
496  }
497  Cursor += EnvVar.size() + 1 /*null-terminator*/;
498  }
499  }
500  SkipSettingEnvironment:;
501 #endif
502  } else {
503  linkPath = TC.GetProgramPath(Linker.str().c_str());
504  }
505

507  JA, *this, ResponseFileSupport::AtFileUTF16(),
508  Args.MakeArgString(linkPath), CmdArgs, Inputs, Output);
509  if (!Environment.empty())
510  LinkCmd->setEnvironment(Environment);
511  C.addCommand(std::move(LinkCmd));
512 }

923 static void TranslateOptArg(Arg *A, llvm::opt::DerivedArgList &DAL,
924  bool SupportsForcingFramePointer,
925  const char *ExpandChar, const OptTable &Opts) {
926  assert(A->getOption().matches(options::OPT__SLASH_O));
927 
928  StringRef OptStr = A->getValue();
929  for (size_t I = 0, E = OptStr.size(); I != E; ++I) {
930  const char &OptChar = *(OptStr.data() + I);
931  switch (OptChar) {
932  default:
933  break;
934  case '1':
935  case '2':
936  case '3':
937  case 'x':
938  case 'd':
939  // Ignore /O[12xd] flags that aren't the last one on the command line.
940  // Only the last one gets expanded.
941  if (&OptChar != ExpandChar) {
942  A->claim();
943  break;
944  }
945  if (OptChar == 'd') {
946  DAL.AddFlagArg(A, Opts.getOption(options::OPT_O0));
947  } else {
948  if (OptChar == '1') {
949  DAL.AddJoinedArg(A, Opts.getOption(options::OPT_O), "s");
950  } else if (OptChar == '2' || OptChar == 'x') {
951  DAL.AddFlagArg(A, Opts.getOption(options::OPT_fbuiltin));
952  DAL.AddJoinedArg(A, Opts.getOption(options::OPT_O), "2");
953  } else if (OptChar == '3') {
954  DAL.AddFlagArg(A, Opts.getOption(options::OPT_fbuiltin));
955  DAL.AddJoinedArg(A, Opts.getOption(options::OPT_O), "3");
956  }
957  if (SupportsForcingFramePointer &&
958  !DAL.hasArgNoClaim(options::OPT_fno_omit_frame_pointer))
959  DAL.AddFlagArg(A, Opts.getOption(options::OPT_fomit_frame_pointer));
960  if (OptChar == '1' || OptChar == '2' || OptChar == '3')
961  DAL.AddFlagArg(A, Opts.getOption(options::OPT_ffunction_sections));
962  }
963  break;
964  case 'b':
965  if (I + 1 != E && isdigit(OptStr[I + 1])) {
966  switch (OptStr[I + 1]) {
967  case '0':
968  DAL.AddFlagArg(A, Opts.getOption(options::OPT_fno_inline));
969  break;
970  case '1':
971  DAL.AddFlagArg(A, Opts.getOption(options::OPT_finline_hint_functions));
972  break;
973  case '2':
974  DAL.AddFlagArg(A, Opts.getOption(options::OPT_finline_functions));
975  break;
976  }
977  ++I;
978  }
979  break;
980  case 'g':
981  A->claim();
982  break;
983  case 'i':
984  if (I + 1 != E && OptStr[I + 1] == '-') {
985  ++I;
986  DAL.AddFlagArg(A, Opts.getOption(options::OPT_fno_builtin));
987  } else {
988  DAL.AddFlagArg(A, Opts.getOption(options::OPT_fbuiltin));
989  }
990  break;
991  case 's':
992  DAL.AddJoinedArg(A, Opts.getOption(options::OPT_O), "s");
993  break;
994  case 't':
995  DAL.AddJoinedArg(A, Opts.getOption(options::OPT_O), "2");
996  break;
997  case 'y': {
998  bool OmitFramePointer = true;
999  if (I + 1 != E && OptStr[I + 1] == '-') {
1000  OmitFramePointer = false;
1001  ++I;
1002  }
1003  if (SupportsForcingFramePointer) {
1004  if (OmitFramePointer)
1005  DAL.AddFlagArg(A,
1006  Opts.getOption(options::OPT_fomit_frame_pointer));
1007  else
1008  DAL.AddFlagArg(
1009  A, Opts.getOption(options::OPT_fno_omit_frame_pointer));
1010  } else {
1011  // Don't warn about /Oy- in x86-64 builds (where
1012  // SupportsForcingFramePointer is false). The flag having no effect
1013  // there is a compiler-internal optimization, and people shouldn't have
1014  // to special-case their build files for x86-64 clang-cl.
1015  A->claim();
1016  }
1017  break;
1018  }
1019  }
1020  }
1021 }

1051 llvm::opt::DerivedArgList *
1052 MSVCToolChain::TranslateArgs(const llvm::opt::DerivedArgList &Args,
1053  StringRef BoundArch,
1054  Action::OffloadKind OFK) const {
1055  DerivedArgList *DAL = new DerivedArgList(Args.getBaseArgs());
1056  const OptTable &Opts = getDriver().getOpts();
1057 
1058  // /Oy and /Oy- don't have an effect on X86-64
1059  bool SupportsForcingFramePointer = getArch() != llvm::Triple::x86_64;
1060 
1061  // The -O[12xd] flag actually expands to several flags. We must desugar the
1062  // flags so that options embedded can be negated. For example, the '-O2' flag
1063  // enables '-Oy'. Expanding '-O2' into its constituent flags allows us to
1064  // correctly handle '-O2 -Oy-' where the trailing '-Oy-' disables a single
1065  // aspect of '-O2'.
1066  //
1067  // Note that this expansion logic only applies to the *last* of '[12xd]'.
1068 
1069  // First step is to search for the character we'd like to expand.
1070  const char *ExpandChar = nullptr;
1071  for (Arg *A : Args.filtered(options::OPT__SLASH_O)) {
1072  StringRef OptStr = A->getValue();
1073  for (size_t I = 0, E = OptStr.size(); I != E; ++I) {
1074  char OptChar = OptStr[I];
1075  char PrevChar = I > 0 ? OptStr[I - 1] : '0';
1076  if (PrevChar == 'b') {
1077  // OptChar does not expand; it's an argument to the previous char.
1078  continue;
1079  }
1080  if (OptChar == '1' || OptChar == '2' || OptChar == 'x' ||
1081  OptChar == 'd' || OptChar == '3')
1082  ExpandChar = OptStr.data() + I;
1083  }
1084  }
1085 
1086  for (Arg *A : Args) {
1087  if (A->getOption().matches(options::OPT__SLASH_O)) {
1088  // The -O flag actually takes an amalgam of other options. For example,
1089  // '/Ogyb2' is equivalent to '/Og' '/Oy' '/Ob2'.
