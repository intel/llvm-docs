118  bool hasUnsupportedSplice(const SourceManager &SourceMgr) const {
119  if (!isInvalid())
120  return false;
121  StringRef Text = getRawText(SourceMgr);
122  if (Text.size() < 6 || Text[0] != '/')
123  return false;
124  if (Text[1] == '*')
125  return Text[Text.size() - 1] != '/' || Text[Text.size() - 2] != '*';
126  return Text[1] != '/';
127  }
