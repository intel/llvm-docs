141 static void AppendTypeQualList(raw_ostream &OS, unsigned TypeQuals,
142  bool HasRestrictKeyword) {
143  bool appendSpace = false;
144  if (TypeQuals & Qualifiers::Const) {
145  OS << "const";
146  appendSpace = true;
147  }
148  if (TypeQuals & Qualifiers::Volatile) {
149  if (appendSpace) OS << ' ';
150  OS << "volatile";
151  appendSpace = true;
152  }
153  if (TypeQuals & Qualifiers::Restrict) {
154  if (appendSpace) OS << ' ';
155  if (HasRestrictKeyword) {
156  OS << "restrict";
157  } else {
158  OS << "__restrict";
159  }
160  }

196 bool TypePrinter::canPrefixQualifiers(const Type *T,
197  bool &NeedARCStrongQualifier) {
198  // CanPrefixQualifiers - We prefer to print type qualifiers before the type,
199  // so that we get "const int" instead of "int const", but we can't do this if
200  // the type is complex. For example if the type is "int*", we *must* print
201  // "int * const", printing "const int *" is different. Only do this when the
202  // type expands to a simple string.
203  bool CanPrefixQualifiers = false;
204  NeedARCStrongQualifier = false;
205  const Type *UnderlyingType = T;
206  if (const auto *AT = dyn_cast<AutoType>(T))
207  UnderlyingType = AT->desugar().getTypePtr();
208  if (const auto *Subst = dyn_cast<SubstTemplateTypeParmType>(T))
209  UnderlyingType = Subst->getReplacementType().getTypePtr();
210  Type::TypeClass TC = UnderlyingType->getTypeClass();
211 
212  switch (TC) {
213  case Type::Auto:
214  case Type::Builtin:
215  case Type::Complex:
216  case Type::UnresolvedUsing:
217  case Type::Using:
218  case Type::Typedef:
219  case Type::TypeOfExpr:
220  case Type::TypeOf:
221  case Type::Decltype:
222  case Type::UnaryTransform:
223  case Type::Record:
224  case Type::Enum:
225  case Type::Elaborated:
226  case Type::TemplateTypeParm:
227  case Type::SubstTemplateTypeParmPack:
228  case Type::DeducedTemplateSpecialization:
229  case Type::TemplateSpecialization:
230  case Type::InjectedClassName:
231  case Type::DependentName:
232  case Type::DependentTemplateSpecialization:
233  case Type::ObjCObject:
234  case Type::ObjCTypeParam:
235  case Type::ObjCInterface:
236  case Type::Atomic:
237  case Type::Pipe:
238  case Type::BitInt:
239  case Type::DependentBitInt:
240  case Type::BTFTagAttributed:
241  CanPrefixQualifiers = true;
242  break;
243 
244  case Type::ObjCObjectPointer:
245  CanPrefixQualifiers = T->isObjCIdType() || T->isObjCClassType() ||
246  T->isObjCQualifiedIdType() || T->isObjCQualifiedClassType();
247  break;
248 
249  case Type::VariableArray:
250  case Type::DependentSizedArray:
251  NeedARCStrongQualifier = true;
252  LLVM_FALLTHROUGH;
253 
254  case Type::ConstantArray:
255  case Type::IncompleteArray:
256  return canPrefixQualifiers(
257  cast<ArrayType>(UnderlyingType)->getElementType().getTypePtr(),
258  NeedARCStrongQualifier);
259 
260  case Type::Adjusted:
261  case Type::Decayed:
262  case Type::Pointer:
263  case Type::BlockPointer:
264  case Type::LValueReference:
265  case Type::RValueReference:
266  case Type::MemberPointer:
267  case Type::DependentAddressSpace:
268  case Type::DependentVector:
269  case Type::DependentSizedExtVector:
270  case Type::Vector:
271  case Type::ExtVector:
272  case Type::ConstantMatrix:
273  case Type::DependentSizedMatrix:
274  case Type::FunctionProto:
275  case Type::FunctionNoProto:
276  case Type::Paren:
277  case Type::PackExpansion:
278  case Type::SubstTemplateTypeParm:
279  case Type::MacroQualified:
280  CanPrefixQualifiers = false;
281  break;
282 
283  case Type::Attributed: {
284  // We still want to print the address_space before the type if it is an
285  // address_space attribute.
286  const auto *AttrTy = cast<AttributedType>(UnderlyingType);
287  CanPrefixQualifiers = AttrTy->getAttrKind() == attr::AddressSpace;
288  }
289  }
290 
291  return CanPrefixQualifiers;
292 }

294 void TypePrinter::printBefore(QualType T, raw_ostream &OS) {
295  SplitQualType Split = splitAccordingToPolicy(T, Policy);
296 
297  // If we have cv1 T, where T is substituted for cv2 U, only print cv1 - cv2
298  // at this level.
299  Qualifiers Quals = Split.Quals;
300  if (const auto *Subst = dyn_cast<SubstTemplateTypeParmType>(Split.Ty))
301  Quals -= QualType(Subst, 0).getQualifiers();
302 
303  printBefore(Split.Ty, Quals, OS);
304 }

306 /// Prints the part of the type string before an identifier, e.g. for
307 /// "int foo[10]" it prints "int ".
308 void TypePrinter::printBefore(const Type *T,Qualifiers Quals, raw_ostream &OS) {
309  if (Policy.SuppressSpecifiers && T->isSpecifierType())
310  return;
311 
312  if (Policy.SuppressTypedefs && (T->getTypeClass() == Type::Typedef)) {
313  QualType UnderlyingType = T->getCanonicalTypeInternal();
314  SplitQualType Split = splitAccordingToPolicy(UnderlyingType, Policy);
315  Qualifiers FullQuals = Quals + Split.Quals;
316  printBefore(Split.Ty, FullQuals, OS);
317  return;
318  }
319 
320  SaveAndRestore<bool> PrevPHIsEmpty(HasEmptyPlaceHolder);
321 
322  // Print qualifiers as appropriate.
323 
324  bool CanPrefixQualifiers = false;
325  bool NeedARCStrongQualifier = false;
326  CanPrefixQualifiers = canPrefixQualifiers(T, NeedARCStrongQualifier);
327 
328  if (CanPrefixQualifiers && !Quals.empty()) {
329  if (NeedARCStrongQualifier) {
330  IncludeStrongLifetimeRAII Strong(Policy);
331  Quals.print(OS, Policy, /*appendSpaceIfNonEmpty=*/true);
332  } else {
333  Quals.print(OS, Policy, /*appendSpaceIfNonEmpty=*/true);
334  }
335  }
336 
337  bool hasAfterQuals = false;
338  if (!CanPrefixQualifiers && !Quals.empty()) {
339  hasAfterQuals = !Quals.isEmptyWhenPrinted(Policy);
340  if (hasAfterQuals)
341  HasEmptyPlaceHolder = false;
342  }
343 
344  switch (T->getTypeClass()) {
345 #define ABSTRACT_TYPE(CLASS, PARENT)
346 #define TYPE(CLASS, PARENT) case Type::CLASS: \
347  print##CLASS##Before(cast<CLASS##Type>(T), OS); \
348  break;
349 #include "clang/AST/TypeNodes.inc"
350  }
351 
352  if (hasAfterQuals) {
353  if (NeedARCStrongQualifier) {
354  IncludeStrongLifetimeRAII Strong(Policy);
355  Quals.print(OS, Policy, /*appendSpaceIfNonEmpty=*/!PrevPHIsEmpty.get());
356  } else {
357  Quals.print(OS, Policy, /*appendSpaceIfNonEmpty=*/!PrevPHIsEmpty.get());
358  }
359  }
360 }

1251 /// Appends the given scope to the end of a string.
1252 void TypePrinter::AppendScope(DeclContext *DC, raw_ostream &OS,
1253  DeclarationName NameInScope) {
1254  if (DC->isTranslationUnit())
1255  return;
1256 
1257  // FIXME: Consider replacing this with NamedDecl::printNestedNameSpecifier,
1258  // which can also print names for function and method scopes.
1259  if (DC->isFunctionOrMethod())
1260  return;
1261 
1262  if (Policy.Callbacks && Policy.Callbacks->isScopeVisible(DC))
1263  return;
1264 
1265  if (const auto *NS = dyn_cast<NamespaceDecl>(DC)) {
1266  if (Policy.SuppressUnwrittenScope && NS->isAnonymousNamespace())
1267  return AppendScope(DC->getParent(), OS, NameInScope);
1268 
1269  // Only suppress an inline namespace if the name has the same lookup
1270  // results in the enclosing namespace.
1271  if (Policy.SuppressInlineNamespace && NS->isInline() && NameInScope &&
1272  NS->isRedundantInlineQualifierFor(NameInScope))
1273  return AppendScope(DC->getParent(), OS, NameInScope);
1274 
1275  AppendScope(DC->getParent(), OS, NS->getDeclName());
1276  if (NS->getIdentifier())
1277  OS << NS->getName() << "::";
1278  else
1279  OS << "(anonymous namespace)::";

1302 void TypePrinter::printTag(TagDecl *D, raw_ostream &OS) {
1303  if (Policy.IncludeTagDefinition) {
1304  PrintingPolicy SubPolicy = Policy;
1305  SubPolicy.IncludeTagDefinition = false;
1306  D->print(OS, SubPolicy, Indentation);
1307  spaceBeforePlaceHolder(OS);
1308  return;
1309  }
1310 
1311  bool HasKindDecoration = false;
1312 
1313  // We don't print tags unless this is an elaborated type.
1314  // In C, we just assume every RecordType is an elaborated type.
1315  if (!Policy.SuppressTagKeyword && !D->getTypedefNameForAnonDecl()) {
1316  HasKindDecoration = true;
1317  OS << D->getKindName();
1318  OS << ' ';
1319  }
1320 
1321  // Compute the full nested-name-specifier for this type.
1322  // In C, this will always be empty except when the type
1323  // being printed is anonymous within other Record.
1324  if (!Policy.SuppressScope)
1325  AppendScope(D->getDeclContext(), OS, D->getDeclName());
1326 
1327  if (const IdentifierInfo *II = D->getIdentifier())
1328  OS << II->getName();
1329  else if (TypedefNameDecl *Typedef = D->getTypedefNameForAnonDecl()) {
1330  assert(Typedef->getIdentifier() && "Typedef without identifier?");
1331  OS << Typedef->getIdentifier()->getName();
1332  } else {
1333  // Make an unambiguous representation for anonymous types, e.g.
1334  // (anonymous enum at /usr/include/string.h:120:9)
1335  OS << (Policy.MSVCFormatting ? '`' : '(');
1336 
1337  if (isa<CXXRecordDecl>(D) && cast<CXXRecordDecl>(D)->isLambda()) {
1338  OS << "lambda";
1339  HasKindDecoration = true;
1340  } else if ((isa<RecordDecl>(D) && cast<RecordDecl>(D)->isAnonymousStructOrUnion())) {
1341  OS << "anonymous";
1342  } else {
1343  OS << "unnamed";
1344  }
1345 
1346  if (Policy.AnonymousTagLocations) {
1347  // Suppress the redundant tag keyword if we just printed one.
1348  // We don't have to worry about ElaboratedTypes here because you can't
1349  // refer to an anonymous type with one.
1350  if (!HasKindDecoration)
1351  OS << " " << D->getKindName();
1352 
1353  PresumedLoc PLoc = D->getASTContext().getSourceManager().getPresumedLoc(
1354  D->getLocation());

1356  OS << " at ";
1357  StringRef File = PLoc.getFilename();
1358  if (auto *Callbacks = Policy.Callbacks)
1359  OS << Callbacks->remapPath(File);
1360  else
1361  OS << File;
1362  OS << ':' << PLoc.getLine() << ':' << PLoc.getColumn();
1363  }
1364  }
1365 
1366  OS << (Policy.MSVCFormatting ? '\'' : ')');
1367  }
1368 
1369  // If this is a class template specialization, print the template
1370  // arguments.
1371  if (const auto *Spec = dyn_cast<ClassTemplateSpecializationDecl>(D)) {
1372  ArrayRef<TemplateArgument> Args;
1373  TypeSourceInfo *TAW = Spec->getTypeAsWritten();
1374  if (!Policy.PrintCanonicalTypes && TAW) {
1375  const TemplateSpecializationType *TST =
1376  cast<TemplateSpecializationType>(TAW->getType());
1377  Args = TST->template_arguments();
1378  } else {
1379  const TemplateArgumentList &TemplateArgs = Spec->getTemplateArgs();
1380  Args = TemplateArgs.asArray();
1381  }
1382  IncludeStrongLifetimeRAII Strong(Policy);
1383  printTemplateArgumentList(
1384  OS, Args, Policy,
1385  Spec->getSpecializedTemplate()->getTemplateParameters());
1386  }
1387 
1388  spaceBeforePlaceHolder(OS);
1389 }
