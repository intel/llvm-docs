152 static void AppendTypeQualList(raw_ostream &OS, unsigned TypeQuals,
153  bool HasRestrictKeyword) {
154  bool appendSpace = false;
155  if (TypeQuals & Qualifiers::Const) {
156  OS << "const";
157  appendSpace = true;
158  }
159  if (TypeQuals & Qualifiers::Volatile) {
160  if (appendSpace) OS << ' ';
161  OS << "volatile";
162  appendSpace = true;
163  }
164  if (TypeQuals & Qualifiers::Restrict) {
165  if (appendSpace) OS << ' ';
166  if (HasRestrictKeyword) {
167  OS << "restrict";
168  } else {
169  OS << "__restrict";
170  }
171  }
172 }

207 bool TypePrinter::canPrefixQualifiers(const Type *T,
208  bool &NeedARCStrongQualifier) {
209  // CanPrefixQualifiers - We prefer to print type qualifiers before the type,
210  // so that we get "const int" instead of "int const", but we can't do this if
211  // the type is complex. For example if the type is "int*", we *must* print
212  // "int * const", printing "const int *" is different. Only do this when the
213  // type expands to a simple string.
214  bool CanPrefixQualifiers = false;
215  NeedARCStrongQualifier = false;
216  const Type *UnderlyingType = T;
217  if (const auto *AT = dyn_cast<AutoType>(T))
218  UnderlyingType = AT->desugar().getTypePtr();
219  if (const auto *Subst = dyn_cast<SubstTemplateTypeParmType>(T))
220  UnderlyingType = Subst->getReplacementType().getTypePtr();
221  Type::TypeClass TC = UnderlyingType->getTypeClass();
222 
223  switch (TC) {
224  case Type::Auto:
225  case Type::Builtin:
226  case Type::Complex:
227  case Type::UnresolvedUsing:
228  case Type::Using:
229  case Type::Typedef:
230  case Type::TypeOfExpr:
231  case Type::TypeOf:
232  case Type::Decltype:
233  case Type::UnaryTransform:
234  case Type::Record:
235  case Type::Enum:
236  case Type::Elaborated:
237  case Type::TemplateTypeParm:
238  case Type::SubstTemplateTypeParmPack:
239  case Type::DeducedTemplateSpecialization:
240  case Type::TemplateSpecialization:
241  case Type::InjectedClassName:
242  case Type::DependentName:
243  case Type::DependentTemplateSpecialization:
244  case Type::ObjCObject:
245  case Type::ObjCTypeParam:
246  case Type::ObjCInterface:
247  case Type::Atomic:
248  case Type::Pipe:
249  case Type::BitInt:
250  case Type::DependentBitInt:
251  case Type::BTFTagAttributed:
252  CanPrefixQualifiers = true;
253  break;
254 
255  case Type::ObjCObjectPointer:
256  CanPrefixQualifiers = T->isObjCIdType() || T->isObjCClassType() ||
257  T->isObjCQualifiedIdType() || T->isObjCQualifiedClassType();
258  break;
259 
260  case Type::VariableArray:
261  case Type::DependentSizedArray:
262  NeedARCStrongQualifier = true;
263  [[fallthrough]];
264 
265  case Type::ConstantArray:
266  case Type::IncompleteArray:
267  return canPrefixQualifiers(
268  cast<ArrayType>(UnderlyingType)->getElementType().getTypePtr(),
269  NeedARCStrongQualifier);
270 
271  case Type::Adjusted:
272  case Type::Decayed:
273  case Type::ArrayParameter:
274  case Type::Pointer:
275  case Type::BlockPointer:
276  case Type::LValueReference:
277  case Type::RValueReference:
278  case Type::MemberPointer:
279  case Type::DependentAddressSpace:
280  case Type::DependentVector:
281  case Type::DependentSizedExtVector:
282  case Type::Vector:
283  case Type::ExtVector:
284  case Type::ConstantMatrix:
285  case Type::DependentSizedMatrix:
286  case Type::FunctionProto:
287  case Type::FunctionNoProto:
288  case Type::Paren:
289  case Type::PackExpansion:
290  case Type::SubstTemplateTypeParm:
291  case Type::MacroQualified:
292  case Type::CountAttributed:
293  CanPrefixQualifiers = false;
294  break;
295 
296  case Type::Attributed: {
297  // We still want to print the address_space before the type if it is an
298  // address_space attribute.
299  const auto *AttrTy = cast<AttributedType>(UnderlyingType);
300  CanPrefixQualifiers = AttrTy->getAttrKind() == attr::AddressSpace;
301  break;
302  }
303  case Type::PackIndexing: {
304  return canPrefixQualifiers(
305  cast<PackIndexingType>(UnderlyingType)->getPattern().getTypePtr(),
306  NeedARCStrongQualifier);
307  }
308  }
309 
310  return CanPrefixQualifiers;
311 }

313 void TypePrinter::printBefore(QualType T, raw_ostream &OS) {
314  SplitQualType Split = splitAccordingToPolicy(T, Policy);
315 
316  // If we have cv1 T, where T is substituted for cv2 U, only print cv1 - cv2
317  // at this level.
318  Qualifiers Quals = Split.Quals;
319  if (const auto *Subst = dyn_cast<SubstTemplateTypeParmType>(Split.Ty))
320  Quals -= QualType(Subst, 0).getQualifiers();
321 
322  printBefore(Split.Ty, Quals, OS);
323 }

325 /// Prints the part of the type string before an identifier, e.g. for
326 /// "int foo[10]" it prints "int ".
327 void TypePrinter::printBefore(const Type *T,Qualifiers Quals, raw_ostream &OS) {
328  if (Policy.SuppressSpecifiers && T->isSpecifierType())
329  return;
330 
331  if (Policy.SuppressTypedefs && (T->getTypeClass() == Type::Typedef)) {
332  QualType UnderlyingType = T->getCanonicalTypeInternal();
333  SplitQualType Split = splitAccordingToPolicy(UnderlyingType, Policy);
334  Qualifiers FullQuals = Quals + Split.Quals;
335  printBefore(Split.Ty, FullQuals, OS);
336  return;
337  }
338 
339  SaveAndRestore PrevPHIsEmpty(HasEmptyPlaceHolder);
340 
341  // Print qualifiers as appropriate.
342 
343  bool CanPrefixQualifiers = false;
344  bool NeedARCStrongQualifier = false;
345  CanPrefixQualifiers = canPrefixQualifiers(T, NeedARCStrongQualifier);
346 
347  if (CanPrefixQualifiers && !Quals.empty()) {
348  if (NeedARCStrongQualifier) {
349  IncludeStrongLifetimeRAII Strong(Policy);
350  Quals.print(OS, Policy, /*appendSpaceIfNonEmpty=*/true);
351  } else {
352  Quals.print(OS, Policy, /*appendSpaceIfNonEmpty=*/true);
353  }
354  }
355 
356  bool hasAfterQuals = false;
357  if (!CanPrefixQualifiers && !Quals.empty()) {
358  hasAfterQuals = !Quals.isEmptyWhenPrinted(Policy);
359  if (hasAfterQuals)
360  HasEmptyPlaceHolder = false;
361  }
362 
363  switch (T->getTypeClass()) {
364 #define ABSTRACT_TYPE(CLASS, PARENT)
365 #define TYPE(CLASS, PARENT) case Type::CLASS: \
366  print##CLASS##Before(cast<CLASS##Type>(T), OS); \
367  break;
368 #include "clang/AST/TypeNodes.inc"
369  }
370 
371  if (hasAfterQuals) {
372  if (NeedARCStrongQualifier) {
373  IncludeStrongLifetimeRAII Strong(Policy);
374  Quals.print(OS, Policy, /*appendSpaceIfNonEmpty=*/!PrevPHIsEmpty.get());
375  } else {
376  Quals.print(OS, Policy, /*appendSpaceIfNonEmpty=*/!PrevPHIsEmpty.get());
377  }
378  }
379 }

1406 void TypePrinter::printTag(TagDecl *D, raw_ostream &OS) {
1407  if (Policy.IncludeTagDefinition) {
1408  PrintingPolicy SubPolicy = Policy;
1409  SubPolicy.IncludeTagDefinition = false;
1410  D->print(OS, SubPolicy, Indentation);
1411  spaceBeforePlaceHolder(OS);
1412  return;
1413  }
1414 
1415  bool HasKindDecoration = false;
1416 
1417  // We don't print tags unless this is an elaborated type.
1418  // In C, we just assume every RecordType is an elaborated type.
1419  if (!Policy.SuppressTagKeyword && !D->getTypedefNameForAnonDecl()) {
1420  HasKindDecoration = true;
1421  OS << D->getKindName();

1423  }
1424 
1425  // Compute the full nested-name-specifier for this type.
1426  // In C, this will always be empty except when the type
1427  // being printed is anonymous within other Record.
1428  if (!Policy.SuppressScope)
1429  AppendScope(D->getDeclContext(), OS, D->getDeclName());
1430 
1431  if (const IdentifierInfo *II = D->getIdentifier())
1432  OS << II->getName();
1433  else if (TypedefNameDecl *Typedef = D->getTypedefNameForAnonDecl()) {

1435  OS << Typedef->getIdentifier()->getName();

1438  // (anonymous enum at /usr/include/string.h:120:9)
1439  OS << (Policy.MSVCFormatting ? '`' : '(');
1440 
1441  if (isa<CXXRecordDecl>(D) && cast<CXXRecordDecl>(D)->isLambda()) {
1442  OS << "lambda";
1443  HasKindDecoration = true;
1444  } else if ((isa<RecordDecl>(D) && cast<RecordDecl>(D)->isAnonymousStructOrUnion())) {
1445  OS << "anonymous";
1446  } else {
1447  OS << "unnamed";
1448  }
1449

1451  // Suppress the redundant tag keyword if we just printed one.
1452  // We don't have to worry about ElaboratedTypes here because you can't
1453  // refer to an anonymous type with one.

1458  D->getLocation());

1460  OS << " at ";
1461  StringRef File = PLoc.getFilename();

1464  WrittenFile = Callbacks->remapPath(File);
1465  // Fix inconsistent path separator created by
1466  // clang::DirectoryLookup::LookupFile when the file path is relative
1467  // path.
1468  llvm::sys::path::Style Style =
1469  llvm::sys::path::is_absolute(WrittenFile)
1470  ? llvm::sys::path::Style::native
1471  : (Policy.MSVCFormatting
1472  ? llvm::sys::path::Style::windows_backslash
1473  : llvm::sys::path::Style::posix);
1474  llvm::sys::path::native(WrittenFile, Style);
1475  OS << WrittenFile << ':' << PLoc.getLine() << ':' << PLoc.getColumn();
1476  }
1477  }
1478 
1479  OS << (Policy.MSVCFormatting ? '\'' : ')');
1480  }
1481 
1482  // If this is a class template specialization, print the template
1483  // arguments.
1484  if (const auto *Spec = dyn_cast<ClassTemplateSpecializationDecl>(D)) {
1485  ArrayRef<TemplateArgument> Args;
1486  TypeSourceInfo *TAW = Spec->getTypeAsWritten();

1493  Args = TemplateArgs.asArray();
1494  }

1496  printTemplateArgumentList(
1497  OS, Args, Policy,
1498  Spec->getSpecializedTemplate()->getTemplateParameters());
1499  }
1500 
1501  spaceBeforePlaceHolder(OS);
1502 }
