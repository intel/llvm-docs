288 // Check if ASan should use GC-friendly instrumentation for globals.
289 // First of all, there is no point if -fdata-sections is off (expect for MachO,
290 // where this is not a factor). Also, on ELF this feature requires an assembler
291 // extension that only works with -integrated-as at the moment.
292 static bool asanUseGlobalsGC(const Triple &T, const CodeGenOptions &CGOpts) {
293  if (!CGOpts.SanitizeAddressGlobalsDeadStripping)
294  return false;
295  switch (T.getObjectFormat()) {
296  case Triple::MachO:
297  case Triple::COFF:
298  return true;
299  case Triple::ELF:
300  return !CGOpts.DisableIntegratedAS;
301  case Triple::GOFF:
302  llvm::report_fatal_error("ASan not implemented for GOFF");
303  case Triple::XCOFF:
304  llvm::report_fatal_error("ASan not implemented for XCOFF.");
305  case Triple::Wasm:
306  case Triple::DXContainer:
307  case Triple::SPIRV:
308  case Triple::UnknownObjectFormat:
309  break;
310  }
311  return false;
312 }

702 static void addSanitizers(const Triple &TargetTriple,
703  const CodeGenOptions &CodeGenOpts,
704  const LangOptions &LangOpts, PassBuilder &PB) {
705  auto SanitizersCallback = [&](ModulePassManager &MPM,
706  OptimizationLevel Level) {
707  if (CodeGenOpts.hasSanitizeCoverage()) {
708  auto SancovOpts = getSancovOptsFromCGOpts(CodeGenOpts);
709  MPM.addPass(SanitizerCoveragePass(
710  SancovOpts, CodeGenOpts.SanitizeCoverageAllowlistFiles,
711  CodeGenOpts.SanitizeCoverageIgnorelistFiles));
712  }
713 
714  if (CodeGenOpts.hasSanitizeBinaryMetadata()) {
715  MPM.addPass(SanitizerBinaryMetadataPass(
716  getSanitizerBinaryMetadataOptions(CodeGenOpts),
717  CodeGenOpts.SanitizeMetadataIgnorelistFiles));
718  }
719 
720  auto MSanPass = [&](SanitizerMask Mask, bool CompileKernel) {
721  if (LangOpts.Sanitize.has(Mask)) {
722  int TrackOrigins = CodeGenOpts.SanitizeMemoryTrackOrigins;
723  bool Recover = CodeGenOpts.SanitizeRecover.has(Mask);
724 
725  MemorySanitizerOptions options(TrackOrigins, Recover, CompileKernel,
726  CodeGenOpts.SanitizeMemoryParamRetval);
727  MPM.addPass(MemorySanitizerPass(options));
728  if (Level != OptimizationLevel::O0) {
729  // MemorySanitizer inserts complex instrumentation that mostly follows
730  // the logic of the original code, but operates on "shadow" values. It
731  // can benefit from re-running some general purpose optimization
732  // passes.
733  MPM.addPass(RequireAnalysisPass<GlobalsAA, Module>());
734  FunctionPassManager FPM;
735  FPM.addPass(EarlyCSEPass(true /* Enable mem-ssa. */));
736  FPM.addPass(InstCombinePass());
737  FPM.addPass(JumpThreadingPass());
738  FPM.addPass(GVNPass());
739  FPM.addPass(InstCombinePass());
740  MPM.addPass(createModuleToFunctionPassAdaptor(std::move(FPM)));
741  }
742  }
743  };
744  MSanPass(SanitizerKind::Memory, false);
745  MSanPass(SanitizerKind::KernelMemory, true);
746 
747  if (LangOpts.Sanitize.has(SanitizerKind::Thread)) {
748  MPM.addPass(ModuleThreadSanitizerPass());
749  MPM.addPass(createModuleToFunctionPassAdaptor(ThreadSanitizerPass()));
750  }
751 
752  auto ASanPass = [&](SanitizerMask Mask, bool CompileKernel) {
753  if (LangOpts.Sanitize.has(Mask)) {
754  bool UseGlobalGC = asanUseGlobalsGC(TargetTriple, CodeGenOpts);
755  bool UseOdrIndicator = CodeGenOpts.SanitizeAddressUseOdrIndicator;
756  llvm::AsanDtorKind DestructorKind =
757  CodeGenOpts.getSanitizeAddressDtor();
758  AddressSanitizerOptions Opts;
759  Opts.CompileKernel = CompileKernel;
760  Opts.Recover = CodeGenOpts.SanitizeRecover.has(Mask);
761  Opts.UseAfterScope = CodeGenOpts.SanitizeAddressUseAfterScope;
762  Opts.UseAfterReturn = CodeGenOpts.getSanitizeAddressUseAfterReturn();
763  MPM.addPass(AddressSanitizerPass(Opts, UseGlobalGC, UseOdrIndicator,
764  DestructorKind));
765  }
766  };
767  ASanPass(SanitizerKind::Address, false);
768  ASanPass(SanitizerKind::KernelAddress, true);
769 
770  auto HWASanPass = [&](SanitizerMask Mask, bool CompileKernel) {
771  if (LangOpts.Sanitize.has(Mask)) {
772  bool Recover = CodeGenOpts.SanitizeRecover.has(Mask);
773  MPM.addPass(HWAddressSanitizerPass(
774  {CompileKernel, Recover,
775  /*DisableOptimization=*/CodeGenOpts.OptimizationLevel == 0}));
776  }
777  };
778  HWASanPass(SanitizerKind::HWAddress, false);
779  HWASanPass(SanitizerKind::KernelHWAddress, true);
780 
781  if (LangOpts.Sanitize.has(SanitizerKind::DataFlow)) {
782  MPM.addPass(DataFlowSanitizerPass(LangOpts.NoSanitizeFiles));
783  }
784  };
785  if (ClSanitizeOnOptimizerEarlyEP) {
786  PB.registerOptimizerEarlyEPCallback(
787  [SanitizersCallback](ModulePassManager &MPM, OptimizationLevel Level) {
788  ModulePassManager NewMPM;
789  SanitizersCallback(NewMPM, Level);
790  if (!NewMPM.isEmpty()) {
791  // Sanitizers can abandon<GlobalsAA>.
792  NewMPM.addPass(RequireAnalysisPass<GlobalsAA, Module>());
793  MPM.addPass(std::move(NewMPM));
794  }
795  });
796  } else {
797  // LastEP does not need GlobalsAA.
798  PB.registerOptimizerLastEPCallback(SanitizersCallback);
799  }
800 
801  if (LowerAllowCheckPass::IsRequested()) {
802  // We can optimize after inliner, and PGO profile matching. The hook below
803  // is called at the end `buildFunctionSimplificationPipeline`, which called
804  // from `buildInlinerPipeline`, which called after profile matching.
805  PB.registerScalarOptimizerLateEPCallback(
806  [](FunctionPassManager &FPM, OptimizationLevel Level) {
807  FPM.addPass(LowerAllowCheckPass());
808  });
809  }
810 }

812 void EmitAssemblyHelper::RunOptimizationPipeline(
813  BackendAction Action, std::unique_ptr<raw_pwrite_stream> &OS,
814  std::unique_ptr<llvm::ToolOutputFile> &ThinLinkOS) {
815  std::optional<PGOOptions> PGOOpt;
816

818  // -fprofile-generate.
819  PGOOpt = PGOOptions(
820  CodeGenOpts.InstrProfileOutput.empty() ? getDefaultProfileGenName()
821  : CodeGenOpts.InstrProfileOutput,
822  "", "", CodeGenOpts.MemoryProfileUsePath, nullptr, PGOOptions::IRInstr,
823  PGOOptions::NoCSAction, ClPGOColdFuncAttr,
824  CodeGenOpts.DebugInfoForProfiling,
825  /*PseudoProbeForProfiling=*/false, CodeGenOpts.AtomicProfileUpdate);
826  else if (CodeGenOpts.hasProfileIRUse()) {

829  : PGOOptions::NoCSAction;
830  PGOOpt = PGOOptions(CodeGenOpts.ProfileInstrumentUsePath, "",
831  CodeGenOpts.ProfileRemappingFile,
832  CodeGenOpts.MemoryProfileUsePath, VFS,
833  PGOOptions::IRUse, CSAction, ClPGOColdFuncAttr,
834  CodeGenOpts.DebugInfoForProfiling);
835  } else if (!CodeGenOpts.SampleProfileFile.empty())
836  // -fprofile-sample-use
837  PGOOpt = PGOOptions(
838  CodeGenOpts.SampleProfileFile, "", CodeGenOpts.ProfileRemappingFile,
839  CodeGenOpts.MemoryProfileUsePath, VFS, PGOOptions::SampleUse,
840  PGOOptions::NoCSAction, ClPGOColdFuncAttr,
841  CodeGenOpts.DebugInfoForProfiling, CodeGenOpts.PseudoProbeForProfiling);
842  else if (!CodeGenOpts.MemoryProfileUsePath.empty())
843  // -fmemory-profile-use (without any of the above options)
844  PGOOpt = PGOOptions("", "", "", CodeGenOpts.MemoryProfileUsePath, VFS,
845  PGOOptions::NoAction, PGOOptions::NoCSAction,
846  ClPGOColdFuncAttr, CodeGenOpts.DebugInfoForProfiling);
847  else if (CodeGenOpts.PseudoProbeForProfiling)
848  // -fpseudo-probe-for-profiling
849  PGOOpt =
850  PGOOptions("", "", "", /*MemoryProfile=*/"", nullptr,
851  PGOOptions::NoAction, PGOOptions::NoCSAction,
852  ClPGOColdFuncAttr, CodeGenOpts.DebugInfoForProfiling, true);
853  else if (CodeGenOpts.DebugInfoForProfiling)
854  // -fdebug-info-for-profiling
855  PGOOpt = PGOOptions("", "", "", /*MemoryProfile=*/"", nullptr,
856  PGOOptions::NoAction, PGOOptions::NoCSAction,
857  ClPGOColdFuncAttr, true);
858 
859  // Check to see if we want to generate a CS profile.
860  if (CodeGenOpts.hasProfileCSIRInstr()) {
861  assert(!CodeGenOpts.hasProfileCSIRUse() &&
862  "Cannot have both CSProfileUse pass and CSProfileGen pass at "
863  "the same time");
864  if (PGOOpt) {
865  assert(PGOOpt->Action != PGOOptions::IRInstr &&
866  PGOOpt->Action != PGOOptions::SampleUse &&
867  "Cannot run CSProfileGen pass with ProfileGen or SampleUse "
868  " pass");
869  PGOOpt->CSProfileGenFile = CodeGenOpts.InstrProfileOutput.empty()
870  ? getDefaultProfileGenName()
871  : CodeGenOpts.InstrProfileOutput;
872  PGOOpt->CSAction = PGOOptions::CSIRInstr;
873  } else
874  PGOOpt = PGOOptions("",
875  CodeGenOpts.InstrProfileOutput.empty()
876  ? getDefaultProfileGenName()
877  : CodeGenOpts.InstrProfileOutput,
878  "", /*MemoryProfile=*/"", nullptr,
879  PGOOptions::NoAction, PGOOptions::CSIRInstr,
880  ClPGOColdFuncAttr, CodeGenOpts.DebugInfoForProfiling);
881  }
882  if (TM)
883  TM->setPGOOption(PGOOpt);
884 
885  PipelineTuningOptions PTO;
886  PTO.LoopUnrolling = CodeGenOpts.UnrollLoops;
887  // For historical reasons, loop interleaving is set to mirror setting for loop
888  // unrolling.
889  PTO.LoopInterleaving = CodeGenOpts.UnrollLoops;
890  PTO.LoopVectorization = CodeGenOpts.VectorizeLoop;
891  PTO.SLPVectorization = CodeGenOpts.VectorizeSLP;
892  PTO.MergeFunctions = CodeGenOpts.MergeFunctions;
893  // Only enable CGProfilePass when using integrated assembler, since
894  // non-integrated assemblers don't recognize .cgprofile section.
895  PTO.CallGraphProfile = !CodeGenOpts.DisableIntegratedAS;
896  // Enable a custom optimization pipeline for non-user SYCL code.
897  PTO.OptimizeSYCLFramework =
898  CodeGenOpts.OptimizeSYCLFramework && !CodeGenOpts.DisableLLVMPasses;
899  PTO.UnifiedLTO = CodeGenOpts.UnifiedLTO;
900 
901  LoopAnalysisManager LAM;
902  FunctionAnalysisManager FAM;
903  CGSCCAnalysisManager CGAM;
904  ModuleAnalysisManager MAM;
905 
906  bool DebugPassStructure = CodeGenOpts.DebugPass == "Structure";
907  PassInstrumentationCallbacks PIC;
908  PrintPassOptions PrintPassOpts;
909  PrintPassOpts.Indent = DebugPassStructure;
910  PrintPassOpts.SkipAnalyses = DebugPassStructure;
911  StandardInstrumentations SI(
912  TheModule->getContext(),
913  (CodeGenOpts.DebugPassManager || DebugPassStructure),
914  CodeGenOpts.VerifyEach, PrintPassOpts);
915  SI.registerCallbacks(PIC, &MAM);
916  PassBuilder PB(TM.get(), PTO, PGOOpt, &PIC);
917 
918  // Handle the assignment tracking feature options.
919  switch (CodeGenOpts.getAssignmentTrackingMode()) {
920  case CodeGenOptions::AssignmentTrackingOpts::Forced:
921  PB.registerPipelineStartEPCallback(
922  [&](ModulePassManager &MPM, OptimizationLevel Level) {
923  MPM.addPass(AssignmentTrackingPass());
924  });
925  break;
926  case CodeGenOptions::AssignmentTrackingOpts::Enabled:
927  // Disable assignment tracking in LTO builds for now as the performance
928  // cost is too high. Disable for LLDB tuning due to llvm.org/PR43126.
929  if (!CodeGenOpts.PrepareForThinLTO && !CodeGenOpts.PrepareForLTO &&
930  CodeGenOpts.getDebuggerTuning() != llvm::DebuggerKind::LLDB) {
931  PB.registerPipelineStartEPCallback(
932  [&](ModulePassManager &MPM, OptimizationLevel Level) {
933  // Only use assignment tracking if optimisations are enabled.
934  if (Level != OptimizationLevel::O0)
935  MPM.addPass(AssignmentTrackingPass());
936  });
937  }
938  break;
939  case CodeGenOptions::AssignmentTrackingOpts::Disabled:
940  break;
941  }
942 
943  // Enable verify-debuginfo-preserve-each for new PM.
944  DebugifyEachInstrumentation Debugify;
945  DebugInfoPerPass DebugInfoBeforePass;
946  if (CodeGenOpts.EnableDIPreservationVerify) {
947  Debugify.setDebugifyMode(DebugifyMode::OriginalDebugInfo);
948  Debugify.setDebugInfoBeforePass(DebugInfoBeforePass);
949

951  Debugify.setOrigDIVerifyBugsReportFilePath(
952  CodeGenOpts.DIBugsReportFilePath);
953  Debugify.registerCallbacks(PIC, MAM);
954  }
955  // Attempt to load pass plugins and register their callbacks with PB.
956  for (auto &PluginFN : CodeGenOpts.PassPlugins) {
957  auto PassPlugin = PassPlugin::Load(PluginFN);
958  if (PassPlugin) {
959  PassPlugin->registerPassBuilderCallbacks(PB);
960  } else {
961  Diags.Report(diag::err_fe_unable_to_load_plugin)
962  << PluginFN << toString(PassPlugin.takeError());
963  }
964  }
965  for (const auto &PassCallback : CodeGenOpts.PassBuilderCallbacks)
966  PassCallback(PB);
967 #define HANDLE_EXTENSION(Ext) \
968  get##Ext##PluginInfo().RegisterPassBuilderCallbacks(PB);
969 #include "llvm/Support/Extension.def"
970

972  // preset TLI.

974  createTLII(TargetTriple, CodeGenOpts));

978  PB.registerModuleAnalyses(MAM);
979  PB.registerCGSCCAnalyses(CGAM);
980  PB.registerFunctionAnalyses(FAM);
981  PB.registerLoopAnalyses(LAM);
982  PB.crossRegisterProxies(LAM, FAM, CGAM, MAM);

984  ModulePassManager MPM;

986  if (CodeGenOpts.VerifyModule)
987  MPM.addPass(VerifierPass());
988 
989  if (!CodeGenOpts.DisableLLVMPasses) {
990  // Map our optimization levels into one of the distinct levels used to
991  // configure the pipeline.
992  OptimizationLevel Level = mapToLevel(CodeGenOpts);
993 
994  if (LangOpts.SYCLIsDevice)
995  PB.registerPipelineStartEPCallback([&](ModulePassManager &MPM,
996  OptimizationLevel Level) {
997  MPM.addPass(ESIMDVerifierPass(LangOpts.SYCLESIMDForceStatelessMem));
998  if (Level == OptimizationLevel::O0)
999  MPM.addPass(ESIMDRemoveOptnoneNoinlinePass());
1000  MPM.addPass(SYCLPropagateAspectsUsagePass(/*ExcludeAspects=*/{"fp64"}));
1001  MPM.addPass(SYCLPropagateJointMatrixUsagePass());
1002  });
1003  else if (LangOpts.SYCLIsHost && !LangOpts.SYCLESIMDBuildHostCode)
1004  PB.registerPipelineStartEPCallback(
1005  [&](ModulePassManager &MPM, OptimizationLevel Level) {
1006  MPM.addPass(ESIMDRemoveHostCodePass());
1007  });
1008

1010  if (TargetTriple.isSPIR() || TargetTriple.isSPIRV()) {
1011  PB.registerOptimizerLastEPCallback(
1012  [](ModulePassManager &MPM, OptimizationLevel Level) {
1013  MPM.addPass(createModuleToFunctionPassAdaptor(
1014  InferAddressSpacesPass(clang::targets::SPIR_GENERIC_AS)));
1015  });
1016  }

1018  const bool PrepareForThinLTO = CodeGenOpts.PrepareForThinLTO;
1019  const bool PrepareForLTO = CodeGenOpts.PrepareForLTO;
1020 
1021  if (LangOpts.ObjCAutoRefCount) {
1022  PB.registerPipelineStartEPCallback(
1023  [](ModulePassManager &MPM, OptimizationLevel Level) {
1024  if (Level != OptimizationLevel::O0)
1025  MPM.addPass(
1026  createModuleToFunctionPassAdaptor(ObjCARCExpandPass()));
1027  });
1028  PB.registerPipelineEarlySimplificationEPCallback(
1029  [](ModulePassManager &MPM, OptimizationLevel Level) {
1030  if (Level != OptimizationLevel::O0)
1031  MPM.addPass(ObjCARCAPElimPass());
1032  });
1033  PB.registerScalarOptimizerLateEPCallback(
1034  [](FunctionPassManager &FPM, OptimizationLevel Level) {
1035  if (Level != OptimizationLevel::O0)
1036  FPM.addPass(ObjCARCOptPass());
1037  });
1038  }
1039 
1040  // If we reached here with a non-empty index file name, then the index
1041  // file was empty and we are not performing ThinLTO backend compilation
1042  // (used in testing in a distributed build environment).
1043  bool IsThinLTOPostLink = !CodeGenOpts.ThinLTOIndexFile.empty();
1044  // If so drop any the type test assume sequences inserted for whole program
1045  // vtables so that codegen doesn't complain.
1046  if (IsThinLTOPostLink)
1047  PB.registerPipelineStartEPCallback(
1048  [](ModulePassManager &MPM, OptimizationLevel Level) {
1049  MPM.addPass(LowerTypeTestsPass(/*ExportSummary=*/nullptr,
1050  /*ImportSummary=*/nullptr,
1051  /*DropTypeTests=*/true));
1052  });

1054  if (CodeGenOpts.InstrumentFunctions ||
1055  CodeGenOpts.InstrumentFunctionEntryBare ||
