361  template <typename AttrTy>
362  static bool isTypeDecoratedWithDeclAttribute(QualType Ty) {
363  const CXXRecordDecl *RecTy = Ty->getAsCXXRecordDecl();
364  if (!RecTy)
365  return false;
366 
367  if (RecTy->hasAttr<AttrTy>())
368  return true;
369 
370  if (auto *CTSD = dyn_cast<ClassTemplateSpecializationDecl>(RecTy)) {
371  ClassTemplateDecl *Template = CTSD->getSpecializedTemplate();
372  if (CXXRecordDecl *RD = Template->getTemplatedDecl())
373  return RD->hasAttr<AttrTy>();
374  }
375  return false;
