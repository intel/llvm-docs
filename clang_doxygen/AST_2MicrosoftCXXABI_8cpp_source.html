195 // getNumBases() seems to only give us the number of direct bases, and not the
196 // total. This function tells us if we inherit from anybody that uses MI, or if
197 // we have a non-primary base class, which uses the multiple inheritance model.
198 static bool usesMultipleInheritanceModel(const CXXRecordDecl *RD) {
199  while (RD->getNumBases() > 0) {
200  if (RD->getNumBases() > 1)
201  return true;
202  assert(RD->getNumBases() == 1);
203  const CXXRecordDecl *Base =
204  RD->bases_begin()->getType()->getAsCXXRecordDecl();
205  if (RD->isPolymorphic() && !Base->isPolymorphic())
206  return true;
207  RD = Base;
208  }
209  return false;
210 }

212 MSInheritanceModel CXXRecordDecl::calculateInheritanceModel() const {
213  if (!hasDefinition() || isParsingBaseSpecifiers())
214  return MSInheritanceModel::Unspecified;
215  if (getNumVBases() > 0)
216  return MSInheritanceModel::Virtual;
217  if (usesMultipleInheritanceModel(this))
218  return MSInheritanceModel::Multiple;
219  return MSInheritanceModel::Single;

240 // Returns the number of pointer and integer slots used to represent a member
241 // pointer in the MS C++ ABI.
242 //
243 // Member function pointers have the following general form; however, fields
244 // are dropped as permitted (under the MSVC interpretation) by the inheritance
245 // model of the actual class.
246 //
247 // struct {
248 // // A pointer to the member function to call. If the member function is
249 // // virtual, this will be a thunk that forwards to the appropriate vftable
250 // // slot.
251 // void *FunctionPointerOrVirtualThunk;
252 //
253 // // An offset to add to the address of the vbtable pointer after
254 // // (possibly) selecting the virtual base but before resolving and calling
255 // // the function.
256 // // Only needed if the class has any virtual bases or bases at a non-zero
257 // // offset.
258 // int NonVirtualBaseAdjustment;
259 //
260 // // The offset of the vb-table pointer within the object. Only needed for
261 // // incomplete types.
262 // int VBPtrOffset;
263 //
264 // // An offset within the vb-table that selects the virtual base containing
265 // // the member. Loading from this offset produces a new offset that is
266 // // added to the address of the vb-table pointer to produce the base.
267 // int VirtualBaseAdjustmentOffset;
268 // };
269 static std::pair<unsigned, unsigned>
270 getMSMemberPointerSlots(const MemberPointerType *MPT) {
271  const CXXRecordDecl *RD = MPT->getMostRecentCXXRecordDecl();
272  MSInheritanceModel Inheritance = RD->getMSInheritanceModel();
273  unsigned Ptrs = 0;
274  unsigned Ints = 0;
275  if (MPT->isMemberFunctionPointer())
276  Ptrs = 1;
277  else
278  Ints = 1;
279  if (inheritanceModelHasNVOffsetField(MPT->isMemberFunctionPointer(),
280  Inheritance))
281  Ints++;
282  if (inheritanceModelHasVBPtrOffsetField(Inheritance))
283  Ints++;
284  if (inheritanceModelHasVBTableOffsetField(Inheritance))
285  Ints++;
286  return std::make_pair(Ptrs, Ints);
287 }

289 CXXABI::MemberPointerInfo MicrosoftCXXABI::getMemberPointerInfo(
290  const MemberPointerType *MPT) const {
291  // The nominal struct is laid out with pointers followed by ints and aligned
292  // to a pointer width if any are present and an int width otherwise.
293  const TargetInfo &Target = Context.getTargetInfo();
294  unsigned PtrSize = Target.getPointerWidth(0);
295  unsigned IntSize = Target.getIntWidth();
296 
297  unsigned Ptrs, Ints;
298  std::tie(Ptrs, Ints) = getMSMemberPointerSlots(MPT);
299  MemberPointerInfo MPI;
300  MPI.HasPadding = false;
301  MPI.Width = Ptrs * PtrSize + Ints * IntSize;
302 
303  // When MSVC does x86_32 record layout, it aligns aggregate member pointers to
304  // 8 bytes. However, __alignof usually returns 4 for data memptrs and 8 for
305  // function memptrs.
306  if (Ptrs + Ints > 1 && Target.getTriple().isArch32Bit())
307  MPI.Align = 64;
308  else if (Ptrs)
309  MPI.Align = Target.getPointerAlign(0);
310  else
311  MPI.Align = Target.getIntAlign();
312 
313  if (Target.getTriple().isArch64Bit()) {
314  MPI.Width = llvm::alignTo(MPI.Width, MPI.Align);
315  MPI.HasPadding = MPI.Width != (Ptrs * PtrSize + Ints * IntSize);
316  }
317  return MPI;
318 }
