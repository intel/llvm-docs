38 /// The AST-based call graph.
39 ///
40 /// The call graph extends itself with the given declarations by implementing
41 /// the recursive AST visitor, which constructs the graph by visiting the given
42 /// declarations.
43 class CallGraph : public RecursiveASTVisitor<CallGraph> {
44  friend class CallGraphNode;
45 
46  using FunctionMapTy =
47  llvm::DenseMap<const Decl *, std::unique_ptr<CallGraphNode>>;
48 
49  /// FunctionMap owns all CallGraphNodes.
50  FunctionMapTy FunctionMap;
51 
52  /// This is a virtual root node that has edges to all the functions.
53  CallGraphNode *Root;
54 
55  /// A setting to determine whether this should include calls that are done in
56  /// a constant expression's context. This DOES require the ASTContext object
57  /// for constexpr-if, so setting it requires a valid ASTContext.
58  bool ShouldSkipConstexpr = false;
59  ASTContext *Ctx;
60 
61 public:
62  CallGraph();
63  ~CallGraph();
64 
65  /// Populate the call graph with the functions in the given
66  /// declaration.
67  ///
68  /// Recursively walks the declaration to find all the dependent Decls as well.
69  void addToCallGraph(Decl *D) {
70  TraverseDecl(D);
71  }
72 
73  /// Determine if a declaration should be included in the graph.
74  static bool includeInGraph(const Decl *D);
75 
76  /// Determine if a declaration should be included in the graph for the
77  /// purposes of being a callee. This is similar to includeInGraph except
78  /// it permits declarations, not just definitions.
79  static bool includeCalleeInGraph(const Decl *D);
80 
81  /// Lookup the node for the given declaration.
82  CallGraphNode *getNode(const Decl *) const;
83 
84  /// Lookup the node for the given declaration. If none found, insert
85  /// one into the graph.
86  CallGraphNode *getOrInsertNode(Decl *);
87 
88  using iterator = FunctionMapTy::iterator;
89  using const_iterator = FunctionMapTy::const_iterator;
90 
91  /// Iterators through all the elements in the graph. Note, this gives
92  /// non-deterministic order.
93  iterator begin() { return FunctionMap.begin(); }
94  iterator end() { return FunctionMap.end(); }
95  const_iterator begin() const { return FunctionMap.begin(); }
96  const_iterator end() const { return FunctionMap.end(); }
97 
98  /// Get the number of nodes in the graph.
99  unsigned size() const { return FunctionMap.size(); }
100 
101  /// Get the virtual root of the graph, all the functions available externally
102  /// are represented as callees of the node.
103  CallGraphNode *getRoot() const { return Root; }
104 
105  /// Iterators through all the nodes of the graph that have no parent. These
106  /// are the unreachable nodes, which are either unused or are due to us
107  /// failing to add a call edge due to the analysis imprecision.
108  using nodes_iterator = llvm::SetVector<CallGraphNode *>::iterator;
109  using const_nodes_iterator = llvm::SetVector<CallGraphNode *>::const_iterator;
110 
111  void print(raw_ostream &os) const;
112  void dump() const;
113  void viewGraph() const;
114 
115  void addNodesForBlocks(DeclContext *D);
116 
117  /// Part of recursive declaration visitation. We recursively visit all the
118  /// declarations to collect the root functions.
119  bool VisitFunctionDecl(FunctionDecl *FD) {
120  // We skip function template definitions, as their semantics is
121  // only determined when they are instantiated.
122  if (includeInGraph(FD) && FD->isThisDeclarationADefinition()) {
123  // Add all blocks declared inside this function to the graph.
124  addNodesForBlocks(FD);
125  // If this function has external linkage, anything could call it.
126  // Note, we are not precise here. For example, the function could have
127  // its address taken.
128  addNodeForDecl(FD, FD->isGlobal());
129  }
130  return true;
131  }
132 
133  /// Part of recursive declaration visitation.
134  bool VisitObjCMethodDecl(ObjCMethodDecl *MD) {
135  if (includeInGraph(MD)) {
136  addNodesForBlocks(MD);
137  addNodeForDecl(MD, true);
138  }
139  return true;
140  }
141 
142  // We are only collecting the declarations, so do not step into the bodies.
143  bool TraverseStmt(Stmt *S) { return true; }
144 
145  bool shouldWalkTypesOfTypeLocs() const { return false; }
146  bool shouldVisitTemplateInstantiations() const { return true; }
147  bool shouldVisitImplicitCode() const { return true; }
148  bool shouldSkipConstantExpressions() const { return ShouldSkipConstexpr; }
149  void setSkipConstantExpressions(ASTContext &Context) {
150  Ctx = &Context;
151  ShouldSkipConstexpr = true;
152  }
153  ASTContext &getASTContext() {
154  assert(Ctx);
155  return *Ctx;
156  }
157 
158 private:
159  /// Add the given declaration to the call graph.
160  void addNodeForDecl(Decl *D, bool IsGlobal);
161 };

163 class CallGraphNode {
164 public:
165  struct CallRecord {
166  CallGraphNode *Callee;
167  Expr *CallExpr;
168 
169  CallRecord() = default;
170 
171  CallRecord(CallGraphNode *Callee_, Expr *CallExpr_)
172  : Callee(Callee_), CallExpr(CallExpr_) {}
173 
174  // The call destination is the only important data here,
175  // allow to transparently unwrap into it.
176  operator CallGraphNode *() const { return Callee; }
177  };
178 
179 private:
180  /// The function/method declaration.
181  Decl *FD;
182 
183  /// The list of functions called from this node.
184  SmallVector<CallRecord, 5> CalledFunctions;
185 
186 public:
187  CallGraphNode(Decl *D) : FD(D) {}
188 
189  using iterator = SmallVectorImpl<CallRecord>::iterator;
190  using const_iterator = SmallVectorImpl<CallRecord>::const_iterator;
191 
192  /// Iterators through all the callees/children of the node.
193  iterator begin() { return CalledFunctions.begin(); }
194  iterator end() { return CalledFunctions.end(); }
195  const_iterator begin() const { return CalledFunctions.begin(); }
196  const_iterator end() const { return CalledFunctions.end(); }
197 
198  /// Iterator access to callees/children of the node.
199  llvm::iterator_range<iterator> callees() {
200  return llvm::make_range(begin(), end());
201  }
202  llvm::iterator_range<const_iterator> callees() const {
203  return llvm::make_range(begin(), end());
204  }
205 
206  bool empty() const { return CalledFunctions.empty(); }
207  unsigned size() const { return CalledFunctions.size(); }
208 
209  void addCallee(CallRecord Call) { CalledFunctions.push_back(Call); }
210 
211  Decl *getDecl() const { return FD; }
212 
213  FunctionDecl *getDefinition() const {
214  return getDecl()->getAsFunction()->getDefinition();
215  }
216 
217  void print(raw_ostream &os) const;
218  void dump() const;
219 };

221 // NOTE: we are comparing based on the callee only. So different call records
222 // (with different call expressions) to the same callee will compare equal!
223 inline bool operator==(const CallGraphNode::CallRecord &LHS,
224  const CallGraphNode::CallRecord &RHS) {
225  return LHS.Callee == RHS.Callee;
226 }
