63 /// Annotates a diagnostic with some code that should be
64 /// inserted, removed, or replaced to fix the problem.
65 ///
66 /// This kind of hint should be used when we are certain that the
67 /// introduction, removal, or modification of a particular (small!)
68 /// amount of code will correct a compilation error. The compiler
69 /// should also provide full recovery from such errors, such that
70 /// suppressing the diagnostic output can still result in successful
71 /// compilation.
72 class FixItHint {
73 public:
74  /// Code that should be replaced to correct the error. Empty for an
75  /// insertion hint.
76  CharSourceRange RemoveRange;
77 
78  /// Code in the specific range that should be inserted in the insertion
79  /// location.
80  CharSourceRange InsertFromRange;
81 
82  /// The actual code to insert at the insertion location, as a
83  /// string.
84  std::string CodeToInsert;
85 
86  bool BeforePreviousInsertions = false;
87 
88  /// Empty code modification hint, indicating that no code
89  /// modification is known.
90  FixItHint() = default;
91 
92  bool isNull() const {
93  return !RemoveRange.isValid();
94  }
95 
96  /// Create a code modification hint that inserts the given
97  /// code string at a specific location.
98  static FixItHint CreateInsertion(SourceLocation InsertionLoc,
99  StringRef Code,
100  bool BeforePreviousInsertions = false) {
101  FixItHint Hint;
102  Hint.RemoveRange =
103  CharSourceRange::getCharRange(InsertionLoc, InsertionLoc);
104  Hint.CodeToInsert = std::string(Code);
105  Hint.BeforePreviousInsertions = BeforePreviousInsertions;
106  return Hint;
107  }
108 
109  /// Create a code modification hint that inserts the given
110  /// code from \p FromRange at a specific location.
111  static FixItHint CreateInsertionFromRange(SourceLocation InsertionLoc,
112  CharSourceRange FromRange,
113  bool BeforePreviousInsertions = false) {
114  FixItHint Hint;
115  Hint.RemoveRange =
116  CharSourceRange::getCharRange(InsertionLoc, InsertionLoc);
117  Hint.InsertFromRange = FromRange;
118  Hint.BeforePreviousInsertions = BeforePreviousInsertions;
119  return Hint;
120  }
121 
122  /// Create a code modification hint that removes the given
123  /// source range.
124  static FixItHint CreateRemoval(CharSourceRange RemoveRange) {
125  FixItHint Hint;
126  Hint.RemoveRange = RemoveRange;
127  return Hint;
128  }
129  static FixItHint CreateRemoval(SourceRange RemoveRange) {
130  return CreateRemoval(CharSourceRange::getTokenRange(RemoveRange));
131  }
132 
133  /// Create a code modification hint that replaces the given
134  /// source range with the given code string.
135  static FixItHint CreateReplacement(CharSourceRange RemoveRange,
136  StringRef Code) {
137  FixItHint Hint;
138  Hint.RemoveRange = RemoveRange;
139  Hint.CodeToInsert = std::string(Code);
140  return Hint;
141  }
142 
143  static FixItHint CreateReplacement(SourceRange RemoveRange,
144  StringRef Code) {
145  return CreateReplacement(CharSourceRange::getTokenRange(RemoveRange), Code);
146  }
