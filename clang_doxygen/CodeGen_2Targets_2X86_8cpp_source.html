52 /// Returns true if this type can be passed in SSE registers with the
53 /// X86_VectorCall calling convention. Shared between x86_32 and x86_64.
54 static bool isX86VectorTypeForVectorCall(ASTContext &Context, QualType Ty) {
55  if (const BuiltinType *BT = Ty->getAs<BuiltinType>()) {
56  if (BT->isFloatingPoint() && BT->getKind() != BuiltinType::Half) {
57  if (BT->getKind() == BuiltinType::LongDouble) {
58  if (&Context.getTargetInfo().getLongDoubleFormat() ==
59  &llvm::APFloat::x87DoubleExtended())
60  return false;
61  }
62  return true;
63  }
64  } else if (const VectorType *VT = Ty->getAs<VectorType>()) {
65  // vectorcall can pass XMM, YMM, and ZMM vectors. We don't pass SSE1 MMX
66  // registers specially.
67  unsigned VecSize = Context.getTypeSize(VT);
68  if (VecSize == 128 || VecSize == 256 || VecSize == 512)
69  return true;
70  }
71  return false;
72 }

254 /// Rewrite input constraint references after adding some output constraints.
255 /// In the case where there is one output and one input and we add one output,
256 /// we need to replace all operand references greater than or equal to 1:
257 /// mov $0, $1
258 /// mov eax, $1
259 /// The result will be:
260 /// mov $0, $2
261 /// mov eax, $2
262 static void rewriteInputConstraintReferences(unsigned FirstIn,
263  unsigned NumNewOuts,
264  std::string &AsmString) {
265  std::string Buf;
266  llvm::raw_string_ostream OS(Buf);
267  size_t Pos = 0;
268  while (Pos < AsmString.size()) {
269  size_t DollarStart = AsmString.find('$', Pos);
270  if (DollarStart == std::string::npos)
271  DollarStart = AsmString.size();
272  size_t DollarEnd = AsmString.find_first_not_of('$', DollarStart);
273  if (DollarEnd == std::string::npos)
274  DollarEnd = AsmString.size();
275  OS << StringRef(&AsmString[Pos], DollarEnd - Pos);
276  Pos = DollarEnd;
277  size_t NumDollars = DollarEnd - DollarStart;
278  if (NumDollars % 2 != 0 && Pos < AsmString.size()) {
279  // We have an operand reference.
280  size_t DigitStart = Pos;
281  if (AsmString[DigitStart] == '{') {
282  OS << '{';
283  ++DigitStart;
284  }
285  size_t DigitEnd = AsmString.find_first_not_of("0123456789", DigitStart);
286  if (DigitEnd == std::string::npos)
287  DigitEnd = AsmString.size();
288  StringRef OperandStr(&AsmString[DigitStart], DigitEnd - DigitStart);
289  unsigned OperandIndex;
290  if (!OperandStr.getAsInteger(10, OperandIndex)) {
291  if (OperandIndex >= FirstIn)
292  OperandIndex += NumNewOuts;
293  OS << OperandIndex;
294  } else {
295  OS << OperandStr;
296  }
297  Pos = DigitEnd;
298  }
299  }
300  AsmString = std::move(OS.str());
301 }

303 /// Add output constraints for EAX:EDX because they are return registers.
304 void X86_32TargetCodeGenInfo::addReturnRegisterOutputs(
305  CodeGenFunction &CGF, LValue ReturnSlot, std::string &Constraints,
306  std::vector<llvm::Type *> &ResultRegTypes,
307  std::vector<llvm::Type *> &ResultTruncRegTypes,
308  std::vector<LValue> &ResultRegDests, std::string &AsmString,
309  unsigned NumOutputs) const {
310  uint64_t RetWidth = CGF.getContext().getTypeSize(ReturnSlot.getType());
311 
312  // Use the EAX constraint if the width is 32 or smaller and EAX:EDX if it is
313  // larger.
314  if (!Constraints.empty())
315  Constraints += ',';
316  if (RetWidth <= 32) {
317  Constraints += "={eax}";
318  ResultRegTypes.push_back(CGF.Int32Ty);
319  } else {
320  // Use the 'A' constraint for EAX:EDX.
321  Constraints += "=A";
322  ResultRegTypes.push_back(CGF.Int64Ty);
323  }
324 
325  // Truncate EAX or EAX:EDX to an integer of the appropriate size.
326  llvm::Type *CoerceTy = llvm::IntegerType::get(CGF.getLLVMContext(), RetWidth);
327  ResultTruncRegTypes.push_back(CoerceTy);
328 
329  // Coerce the integer by bitcasting the return slot pointer.
330  ReturnSlot.setAddress(ReturnSlot.getAddress(CGF).withElementType(CoerceTy));
331  ResultRegDests.push_back(ReturnSlot);
332 
333  rewriteInputConstraintReferences(NumOutputs, 1, AsmString);
334 }

336 /// shouldReturnTypeInRegister - Determine if the given type should be
337 /// returned in a register (for the Darwin and MCU ABI).
338 bool X86_32ABIInfo::shouldReturnTypeInRegister(QualType Ty,
339  ASTContext &Context) const {
340  uint64_t Size = Context.getTypeSize(Ty);
341 
342  // For i386, type must be register sized.
343  // For the MCU ABI, it only needs to be <= 8-byte
344  if ((IsMCUABI && Size > 64) || (!IsMCUABI && !isRegisterSize(Size)))
345  return false;
346 
347  if (Ty->isVectorType()) {
348  // 64- and 128- bit vectors inside structures are not returned in
349  // registers.
350  if (Size == 64 || Size == 128)
351  return false;
352 
353  return true;
354  }
355 
356  // If this is a builtin, pointer, enum, complex type, member pointer, or
357  // member function pointer it is ok.
358  if (Ty->getAs<BuiltinType>() || Ty->hasPointerRepresentation() ||
359  Ty->isAnyComplexType() || Ty->isEnumeralType() ||
360  Ty->isBlockPointerType() || Ty->isMemberPointerType())
361  return true;
362 
363  // Arrays are treated like records.
364  if (const ConstantArrayType *AT = Context.getAsConstantArrayType(Ty))
365  return shouldReturnTypeInRegister(AT->getElementType(), Context);
366 
367  // Otherwise, it must be a record type.
368  const RecordType *RT = Ty->getAs<RecordType>();
369  if (!RT) return false;
370 
371  // FIXME: Traverse bases here too.
372 
373  // Structure types are passed in register if all fields would be
374  // passed in a register.
375  for (const auto *FD : RT->getDecl()->fields()) {
376  // Empty fields are ignored.
377  if (isEmptyField(Context, FD, true))
378  continue;
379 
380  // Check fields recursively.
381  if (!shouldReturnTypeInRegister(FD->getType(), Context))
382  return false;
383  }
384  return true;
385 }

387 static bool is32Or64BitBasicType(QualType Ty, ASTContext &Context) {
388  // Treat complex types as the element type.
389  if (const ComplexType *CTy = Ty->getAs<ComplexType>())
390  Ty = CTy->getElementType();
391 
392  // Check for a type which we know has a simple scalar argument-passing
393  // convention without any padding. (We're specifically looking for 32
394  // and 64-bit integer and integer-equivalents, float, and double.)
395  if (!Ty->getAs<BuiltinType>() && !Ty->hasPointerRepresentation() &&
396  !Ty->isEnumeralType() && !Ty->isBlockPointerType())
397  return false;
398 
399  uint64_t Size = Context.getTypeSize(Ty);
400  return Size == 32 || Size == 64;
401 }

403 static bool addFieldSizes(ASTContext &Context, const RecordDecl *RD,
404  uint64_t &Size) {
405  for (const auto *FD : RD->fields()) {
406  // Scalar arguments on the stack get 4 byte alignment on x86. If the
407  // argument is smaller than 32-bits, expanding the struct will create
408  // alignment padding.
409  if (!is32Or64BitBasicType(FD->getType(), Context))
410  return false;
411 
412  // FIXME: Reject bit-fields wholesale; there are two problems, we don't know
413  // how to expand them yet, and the predicate for telling if a bitfield still
414  // counts as "basic" is more complicated than what we were doing previously.
415  if (FD->isBitField())
416  return false;
417 
418  Size += Context.getTypeSize(FD->getType());
419  }
420  return true;
421 }

436 /// Test whether an argument type which is to be passed indirectly (on the
437 /// stack) would have the equivalent layout if it was expanded into separate
438 /// arguments. If so, we prefer to do the latter to avoid inhibiting
439 /// optimizations.
440 bool X86_32ABIInfo::canExpandIndirectArgument(QualType Ty) const {
441  // We can only expand structure types.
442  const RecordType *RT = Ty->getAs<RecordType>();
443  if (!RT)
444  return false;
445  const RecordDecl *RD = RT->getDecl();
446  uint64_t Size = 0;
447  if (const CXXRecordDecl *CXXRD = dyn_cast<CXXRecordDecl>(RD)) {
448  if (!IsWin32StructABI) {
449  // On non-Windows, we have to conservatively match our old bitcode
450  // prototypes in order to be ABI-compatible at the bitcode level.
451  if (!CXXRD->isCLike())
452  return false;
453  } else {
454  // Don't do this for dynamic classes.
455  if (CXXRD->isDynamicClass())
456  return false;
457  }
458  if (!addBaseAndFieldSizes(getContext(), CXXRD, Size))
459  return false;
460  } else {
461  if (!addFieldSizes(getContext(), RD, Size))
462  return false;
463  }
464 
465  // We can do this if there was no alignment padding.
466  return Size == getContext().getTypeSize(Ty);
467 }

480 ABIArgInfo X86_32ABIInfo::classifyReturnType(QualType RetTy,
481  CCState &State) const {
482  if (RetTy->isVoidType())
483  return ABIArgInfo::getIgnore();
484 
485  const Type *Base = nullptr;
486  uint64_t NumElts = 0;
487  if ((State.CC == llvm::CallingConv::X86_VectorCall ||
488  State.CC == llvm::CallingConv::X86_RegCall) &&
489  isHomogeneousAggregate(RetTy, Base, NumElts)) {
490  // The LLVM struct type for such an aggregate should lower properly.
491  return ABIArgInfo::getDirect();
492  }
493 
494  if (const VectorType *VT = RetTy->getAs<VectorType>()) {
495  // On Darwin, some vectors are returned in registers.
496  if (IsDarwinVectorABI) {
497  uint64_t Size = getContext().getTypeSize(RetTy);
498 
499  // 128-bit vectors are a special case; they are returned in
500  // registers and we need to make sure to pick a type the LLVM
501  // backend will like.
502  if (Size == 128)
503  return ABIArgInfo::getDirect(llvm::FixedVectorType::get(
504  llvm::Type::getInt64Ty(getVMContext()), 2));
505 
506  // Always return in register if it fits in a general purpose
507  // register, or if it is 64 bits and has a single element.
508  if ((Size == 8 || Size == 16 || Size == 32) ||
509  (Size == 64 && VT->getNumElements() == 1))
510  return ABIArgInfo::getDirect(llvm::IntegerType::get(getVMContext(),
511  Size));
512 
513  return getIndirectReturnResult(RetTy, State);
514  }
515 
516  return ABIArgInfo::getDirect();
517  }
518 
519  if (isAggregateTypeForABI(RetTy)) {
520  if (const RecordType *RT = RetTy->getAs<RecordType>()) {
521  // Structures with flexible arrays are always indirect.
522  if (RT->getDecl()->hasFlexibleArrayMember())
523  return getIndirectReturnResult(RetTy, State);
524  }
525 
526  // If specified, structs and unions are always indirect.
527  if (!IsRetSmallStructInRegABI && !RetTy->isAnyComplexType())
528  return getIndirectReturnResult(RetTy, State);
529 
530  // Ignore empty structs/unions.
531  if (isEmptyRecord(getContext(), RetTy, true))
532  return ABIArgInfo::getIgnore();
533 
534  // Return complex of _Float16 as <2 x half> so the backend will use xmm0.
535  if (const ComplexType *CT = RetTy->getAs<ComplexType>()) {
536  QualType ET = getContext().getCanonicalType(CT->getElementType());
537  if (ET->isFloat16Type())
538  return ABIArgInfo::getDirect(llvm::FixedVectorType::get(
539  llvm::Type::getHalfTy(getVMContext()), 2));
540  }
541 
542  // Small structures which are register sized are generally returned
543  // in a register.
544  if (shouldReturnTypeInRegister(RetTy, getContext())) {
545  uint64_t Size = getContext().getTypeSize(RetTy);
546 
547  // As a special-case, if the struct is a "single-element" struct, and
548  // the field is of type "float" or "double", return it in a
549  // floating-point register. (MSVC does not apply this special case.)
550  // We apply a similar transformation for pointer types to improve the
551  // quality of the generated IR.
552  if (const Type *SeltTy = isSingleElementStruct(RetTy, getContext()))
553  if ((!IsWin32StructABI && SeltTy->isRealFloatingType())
554  || SeltTy->hasPointerRepresentation())
555  return ABIArgInfo::getDirect(CGT.ConvertType(QualType(SeltTy, 0)));
556 
557  // FIXME: We should be able to narrow this integer in cases with dead
558  // padding.
559  return ABIArgInfo::getDirect(llvm::IntegerType::get(getVMContext(),Size));
560  }
561 
562  return getIndirectReturnResult(RetTy, State);
563  }
564 
565  // Treat an enum type as its underlying type.
566  if (const EnumType *EnumTy = RetTy->getAs<EnumType>())
567  RetTy = EnumTy->getDecl()->getIntegerType();
568 
569  if (const auto *EIT = RetTy->getAs<BitIntType>())
570  if (EIT->getNumBits() > 64)
571  return getIndirectReturnResult(RetTy, State);
572 
573  return (isPromotableIntegerTypeForABI(RetTy) ? ABIArgInfo::getExtend(RetTy)
574  : ABIArgInfo::getDirect());
575 }

577 unsigned X86_32ABIInfo::getTypeStackAlignInBytes(QualType Ty,
578  unsigned Align) const {
579  // Otherwise, if the alignment is less than or equal to the minimum ABI
580  // alignment, just use the default; the backend will handle this.
581  if (Align <= MinABIStackAlignInBytes)
582  return 0; // Use default alignment.
583 
584  if (IsLinuxABI) {
585  // Exclude other System V OS (e.g Darwin, PS4 and FreeBSD) since we don't
586  // want to spend any effort dealing with the ramifications of ABI breaks.
587  //
588  // If the vector type is __m128/__m256/__m512, return the default alignment.
589  if (Ty->isVectorType() && (Align == 16 || Align == 32 || Align == 64))
590  return Align;
591  }
592  // On non-Darwin, the stack type alignment is always 4.
593  if (!IsDarwinVectorABI) {
594  // Set explicit alignment, since we may need to realign the top.
595  return MinABIStackAlignInBytes;
596  }
597 
598  // Otherwise, if the type contains an SSE vector type, the alignment is 16.
599  if (Align >= 16 && (isSIMDVectorType(getContext(), Ty) ||
600  isRecordWithSIMDVectorType(getContext(), Ty)))
601  return 16;
602 
603  return MinABIStackAlignInBytes;
604 }

643 bool X86_32ABIInfo::updateFreeRegs(QualType Ty, CCState &State) const {
644  if (!IsSoftFloatABI) {
645  Class C = classify(Ty);
646  if (C == Float)
647  return false;
648  }
649 
650  unsigned Size = getContext().getTypeSize(Ty);
651  unsigned SizeInRegs = (Size + 31) / 32;
652 
653  if (SizeInRegs == 0)
654  return false;
655 
656  if (!IsMCUABI) {
657  if (SizeInRegs > State.FreeRegs) {
658  State.FreeRegs = 0;
659  return false;
660  }
661  } else {
662  // The MCU psABI allows passing parameters in-reg even if there are
663  // earlier parameters that are passed on the stack. Also,
664  // it does not allow passing >8-byte structs in-register,
665  // even if there are 3 free registers available.
666  if (SizeInRegs > State.FreeRegs || SizeInRegs > 2)
667  return false;
668  }
669 
670  State.FreeRegs -= SizeInRegs;
671  return true;

723 void X86_32ABIInfo::runVectorCallFirstPass(CGFunctionInfo &FI, CCState &State) const {
724  // Vectorcall x86 works subtly different than in x64, so the format is
725  // a bit different than the x64 version. First, all vector types (not HVAs)
726  // are assigned, with the first 6 ending up in the [XYZ]MM0-5 registers.
727  // This differs from the x64 implementation, where the first 6 by INDEX get
728  // registers.
729  // In the second pass over the arguments, HVAs are passed in the remaining
730  // vector registers if possible, or indirectly by address. The address will be
731  // passed in ECX/EDX if available. Any other arguments are passed according to
732  // the usual fastcall rules.
733  MutableArrayRef<CGFunctionInfoArgInfo> Args = FI.arguments();
734  for (int I = 0, E = Args.size(); I < E; ++I) {
735  const Type *Base = nullptr;
736  uint64_t NumElts = 0;
737  const QualType &Ty = Args[I].type;
738  if ((Ty->isVectorType() || Ty->isBuiltinType()) &&
739  isHomogeneousAggregate(Ty, Base, NumElts)) {
740  if (State.FreeSSERegs >= NumElts) {
741  State.FreeSSERegs -= NumElts;
742  Args[I].info = ABIArgInfo::getDirectInReg();
743  State.IsPreassigned.set(I);
744  }
745  }
746  }
747 }

749 ABIArgInfo X86_32ABIInfo::classifyArgumentType(QualType Ty, CCState &State,
750  unsigned ArgIndex) const {
751  // FIXME: Set alignment on indirect arguments.
752  bool IsFastCall = State.CC == llvm::CallingConv::X86_FastCall;
753  bool IsRegCall = State.CC == llvm::CallingConv::X86_RegCall;
754  bool IsVectorCall = State.CC == llvm::CallingConv::X86_VectorCall;
755 
756  Ty = useFirstFieldIfTransparentUnion(Ty);
757  TypeInfo TI = getContext().getTypeInfo(Ty);
758 
759  // Check with the C++ ABI first.
760  const RecordType *RT = Ty->getAs<RecordType>();
761  if (RT) {
762  CGCXXABI::RecordArgABI RAA = getRecordArgABI(RT, getCXXABI());
763  if (RAA == CGCXXABI::RAA_Indirect) {
764  return getIndirectResult(Ty, false, State);
765  } else if (State.IsDelegateCall) {
766  // Avoid having different alignments on delegate call args by always
767  // setting the alignment to 4, which is what we do for inallocas.
768  ABIArgInfo Res = getIndirectResult(Ty, false, State);
769  Res.setIndirectAlign(CharUnits::fromQuantity(4));
770  return Res;
771  } else if (RAA == CGCXXABI::RAA_DirectInMemory) {
772  // The field index doesn't matter, we'll fix it up later.
773  return ABIArgInfo::getInAlloca(/*FieldIndex=*/0);
774  }
775  }
776 
777  // Regcall uses the concept of a homogenous vector aggregate, similar
778  // to other targets.
779  const Type *Base = nullptr;
780  uint64_t NumElts = 0;
781  if ((IsRegCall || IsVectorCall) &&
782  isHomogeneousAggregate(Ty, Base, NumElts)) {
783  if (State.FreeSSERegs >= NumElts) {
784  State.FreeSSERegs -= NumElts;
785 
786  // Vectorcall passes HVAs directly and does not flatten them, but regcall
787  // does.
788  if (IsVectorCall)
789  return getDirectX86Hva();
790 
791  if (Ty->isBuiltinType() || Ty->isVectorType())
792  return ABIArgInfo::getDirect();
793  return ABIArgInfo::getExpand();
794  }
795  return getIndirectResult(Ty, /*ByVal=*/false, State);
796  }
797 
798  if (isAggregateTypeForABI(Ty)) {
799  // Structures with flexible arrays are always indirect.
800  // FIXME: This should not be byval!
801  if (RT && RT->getDecl()->hasFlexibleArrayMember())
802  return getIndirectResult(Ty, true, State);
803 
804  // Ignore empty structs/unions on non-Windows.
805  if (!IsWin32StructABI && isEmptyRecord(getContext(), Ty, true))
806  return ABIArgInfo::getIgnore();
807 
808  llvm::LLVMContext &LLVMContext = getVMContext();
809  llvm::IntegerType *Int32 = llvm::Type::getInt32Ty(LLVMContext);
810  bool NeedsPadding = false;
811  bool InReg;
812  if (shouldAggregateUseDirect(Ty, State, InReg, NeedsPadding)) {
813  unsigned SizeInRegs = (TI.Width + 31) / 32;
814  SmallVector<llvm::Type*, 3> Elements(SizeInRegs, Int32);
815  llvm::Type *Result = llvm::StructType::get(LLVMContext, Elements);
816  if (InReg)
817  return ABIArgInfo::getDirectInReg(Result);
818  else
819  return ABIArgInfo::getDirect(Result);
820  }
821  llvm::IntegerType *PaddingType = NeedsPadding ? Int32 : nullptr;
822 
823  // Pass over-aligned aggregates to non-variadic functions on Windows
824  // indirectly. This behavior was added in MSVC 2015. Use the required
825  // alignment from the record layout, since that may be less than the
826  // regular type alignment, and types with required alignment of less than 4
827  // bytes are not passed indirectly.
828  if (IsWin32StructABI && State.Required.isRequiredArg(ArgIndex)) {
829  unsigned AlignInBits = 0;
830  if (RT) {
831  const ASTRecordLayout &Layout =
832  getContext().getASTRecordLayout(RT->getDecl());
833  AlignInBits = getContext().toBits(Layout.getRequiredAlignment());
834  } else if (TI.isAlignRequired()) {
835  AlignInBits = TI.Align;
836  }
837  if (AlignInBits > 32)
838  return getIndirectResult(Ty, /*ByVal=*/false, State);
839  }
840 
841  // Expand small (<= 128-bit) record types when we know that the stack layout
842  // of those arguments will match the struct. This is important because the
843  // LLVM backend isn't smart enough to remove byval, which inhibits many
844  // optimizations.
845  // Don't do this for the MCU if there are still free integer registers
846  // (see X86_64 ABI for full explanation).
847  if (TI.Width <= 4 * 32 && (!IsMCUABI || State.FreeRegs == 0) &&
848  canExpandIndirectArgument(Ty))
849  return ABIArgInfo::getExpandWithPadding(
850  IsFastCall || IsVectorCall || IsRegCall, PaddingType);
851 
852  return getIndirectResult(Ty, true, State);
853  }
854 
855  if (const VectorType *VT = Ty->getAs<VectorType>()) {
856  // On Windows, vectors are passed directly if registers are available, or
857  // indirectly if not. This avoids the need to align argument memory. Pass
858  // user-defined vector types larger than 512 bits indirectly for simplicity.
859  if (IsWin32StructABI) {
860  if (TI.Width <= 512 && State.FreeSSERegs > 0) {
861  --State.FreeSSERegs;
862  return ABIArgInfo::getDirectInReg();
863  }
864  return getIndirectResult(Ty, /*ByVal=*/false, State);
865  }
866 
867  // On Darwin, some vectors are passed in memory, we handle this by passing
868  // it as an i8/i16/i32/i64.
869  if (IsDarwinVectorABI) {
870  if ((TI.Width == 8 || TI.Width == 16 || TI.Width == 32) ||
871  (TI.Width == 64 && VT->getNumElements() == 1))
872  return ABIArgInfo::getDirect(
873  llvm::IntegerType::get(getVMContext(), TI.Width));
874  }
875 
876  if (IsX86_MMXType(CGT.ConvertType(Ty)))
877  return ABIArgInfo::getDirect(llvm::IntegerType::get(getVMContext(), 64));
878 
879  return ABIArgInfo::getDirect();
880  }
881 
882 
883  if (const EnumType *EnumTy = Ty->getAs<EnumType>())
884  Ty = EnumTy->getDecl()->getIntegerType();
885 
886  bool InReg = shouldPrimitiveUseInReg(Ty, State);
887 
888  if (isPromotableIntegerTypeForABI(Ty)) {
889  if (InReg)
890  return ABIArgInfo::getExtendInReg(Ty);
891  return ABIArgInfo::getExtend(Ty);
892  }
893 
894  if (const auto *EIT = Ty->getAs<BitIntType>()) {
895  if (EIT->getNumBits() <= 64) {
896  if (InReg)
897  return ABIArgInfo::getDirectInReg();
898  return ABIArgInfo::getDirect();
899  }
900  return getIndirectResult(Ty, /*ByVal=*/false, State);
901  }
902 
903  if (InReg)
904  return ABIArgInfo::getDirectInReg();
905  return ABIArgInfo::getDirect();
906 }

944 void X86_32ABIInfo::computeInfo(CGFunctionInfo &FI) const {
945  ASTContext &Context = getContext();
946  if (doOpenCLClassification(FI, Context))
947  return;
948 
949  CCState State(FI);
950  if (IsMCUABI)
951  State.FreeRegs = 3;
952  else if (State.CC == llvm::CallingConv::X86_FastCall) {
953  State.FreeRegs = 2;
954  State.FreeSSERegs = 3;
955  } else if (State.CC == llvm::CallingConv::X86_VectorCall) {
956  State.FreeRegs = 2;
957  State.FreeSSERegs = 6;
958  } else if (FI.getHasRegParm())
959  State.FreeRegs = FI.getRegParm();
960  else if (State.CC == llvm::CallingConv::X86_RegCall) {
961  State.FreeRegs = 5;
962  State.FreeSSERegs = 8;
963  } else if (IsWin32StructABI) {
964  // Since MSVC 2015, the first three SSE vectors have been passed in
965  // registers. The rest are passed indirectly.
966  State.FreeRegs = DefaultNumRegisterParameters;
967  State.FreeSSERegs = 3;
968  } else
969  State.FreeRegs = DefaultNumRegisterParameters;
970 
971  if (!::classifyReturnType(getCXXABI(), FI, *this)) {
972  FI.getReturnInfo() = classifyReturnType(FI.getReturnType(), State);
973  } else if (FI.getReturnInfo().isIndirect()) {
974  // The C++ ABI is not aware of register usage, so we have to check if the
975  // return value was sret and put it in a register ourselves if appropriate.
976  if (State.FreeRegs) {
977  --State.FreeRegs; // The sret parameter consumes a register.
978  if (!IsMCUABI)
979  FI.getReturnInfo().setInReg(true);
980  }
981  }
982 
983  // The chain argument effectively gives us another free register.
984  if (FI.isChainCall())
985  ++State.FreeRegs;
986 
987  // For vectorcall, do a first pass over the arguments, assigning FP and vector
988  // arguments to XMM registers as available.
989  if (State.CC == llvm::CallingConv::X86_VectorCall)
990  runVectorCallFirstPass(FI, State);
991 
992  bool UsedInAlloca = false;
993  MutableArrayRef<CGFunctionInfoArgInfo> Args = FI.arguments();
994  for (unsigned I = 0, E = Args.size(); I < E; ++I) {
995  // Skip arguments that have already been assigned.
996  if (State.IsPreassigned.test(I))
997  continue;
998 
999  Args[I].info =
1000  classifyArgumentType(Args[I].type, State, I);
1001  UsedInAlloca |= (Args[I].info.getKind() == ABIArgInfo::InAlloca);
1002  }
1003 
1004  // If we needed to use inalloca for any argument, do a second pass and rewrite
1005  // all the memory arguments to use inalloca.
1006  if (UsedInAlloca)
1007  rewriteWithInAlloca(FI);
1008 }

1011 X86_32ABIInfo::addFieldToArgStruct(SmallVector<llvm::Type *, 6> &FrameFields,
1012  CharUnits &StackOffset, ABIArgInfo &Info,
1013  QualType Type) const {
1014  // Arguments are always 4-byte-aligned.
1015  CharUnits WordSize = CharUnits::fromQuantity(4);
1016  assert(StackOffset.isMultipleOf(WordSize) && "unaligned inalloca struct");
1017 
1018  // sret pointers and indirect things will require an extra pointer
1019  // indirection, unless they are byval. Most things are byval, and will not
1020  // require this indirection.
1021  bool IsIndirect = false;

1023  IsIndirect = true;
1024  Info = ABIArgInfo::getInAlloca(FrameFields.size(), IsIndirect);

1026  if (IsIndirect)
1027  LLTy = llvm::PointerType::getUnqual(getVMContext());
1028  FrameFields.push_back(LLTy);
1029  StackOffset += IsIndirect ? WordSize : getContext().getTypeSizeInChars(Type);
1030

1033  StackOffset = FieldEnd.alignTo(WordSize);
1034  if (StackOffset != FieldEnd) {
1035  CharUnits NumBytes = StackOffset - FieldEnd;
1036  llvm::Type *Ty = llvm::Type::getInt8Ty(getVMContext());
1037  Ty = llvm::ArrayType::get(Ty, NumBytes.getQuantity());

1063 void X86_32ABIInfo::rewriteWithInAlloca(CGFunctionInfo &FI) const {
1064  assert(IsWin32StructABI && "inalloca only supported on win32");
1065 
1066  // Build a packed struct type for all of the arguments in memory.
1067  SmallVector<llvm::Type *, 6> FrameFields;
1068 
1069  // The stack alignment is always 4.
1070  CharUnits StackAlign = CharUnits::fromQuantity(4);
1071 
1072  CharUnits StackOffset;
1073  CGFunctionInfo::arg_iterator I = FI.arg_begin(), E = FI.arg_end();
1074 
1075  // Put 'this' into the struct before 'sret', if necessary.
1076  bool IsThisCall =
1077  FI.getCallingConvention() == llvm::CallingConv::X86_ThisCall;
1078  ABIArgInfo &Ret = FI.getReturnInfo();
1079  if (Ret.isIndirect() && Ret.isSRetAfterThis() && !IsThisCall &&
1080  isArgInAlloca(I->info)) {
1081  addFieldToArgStruct(FrameFields, StackOffset, I->info, I->type);
1082  ++I;
1083  }
1084 
1085  // Put the sret parameter into the inalloca struct if it's in memory.
1086  if (Ret.isIndirect() && !Ret.getInReg()) {
1087  addFieldToArgStruct(FrameFields, StackOffset, Ret, FI.getReturnType());
1088  // On Windows, the hidden sret parameter is always returned in eax.
1089  Ret.setInAllocaSRet(IsWin32StructABI);
1090  }
1091 
1092  // Skip the 'this' parameter in ecx.
1093  if (IsThisCall)
1094  ++I;
1095 
1096  // Put arguments passed in memory into the struct.
1097  for (; I != E; ++I) {
1098  if (isArgInAlloca(I->info))
1099  addFieldToArgStruct(FrameFields, StackOffset, I->info, I->type);
1100  }
1101 
1102  FI.setArgStruct(llvm::StructType::get(getVMContext(), FrameFields,
1103  /*isPacked=*/true),
1104  StackAlign);
1105 }

1601 void X86_64TargetCodeGenInfo::checkFunctionCallABI(
1602  CodeGenModule &CGM, SourceLocation CallLoc, const FunctionDecl *Caller,
1603  const FunctionDecl *Callee, const CallArgList &Args) const {
1604  llvm::StringMap<bool> CallerMap;
1605  llvm::StringMap<bool> CalleeMap;
1606  unsigned ArgIndex = 0;
1607 
1608  // We need to loop through the actual call arguments rather than the
1609  // function's parameters, in case this variadic.
1610  for (const CallArg &Arg : Args) {
1611  // The "avx" feature changes how vectors >128 in size are passed. "avx512f"
1612  // additionally changes how vectors >256 in size are passed. Like GCC, we
1613  // warn when a function is called with an argument where this will change.
1614  // Unlike GCC, we also error when it is an obvious ABI mismatch, that is,
1615  // the caller and callee features are mismatched.
1616  // Unfortunately, we cannot do this diagnostic in SEMA, since the callee can
1617  // change its ABI with attribute-target after this call.
1618  if (Arg.getType()->isVectorType() &&
1619  CGM.getContext().getTypeSize(Arg.getType()) > 128) {
1620  initFeatureMaps(CGM.getContext(), CallerMap, Caller, CalleeMap, Callee);
1621  QualType Ty = Arg.getType();
1622  // The CallArg seems to have desugared the type already, so for clearer
1623  // diagnostics, replace it with the type in the FunctionDecl if possible.
1624  if (ArgIndex < Callee->getNumParams())
1625  Ty = Callee->getParamDecl(ArgIndex)->getType();
1626 
1627  if (checkAVXParam(CGM.getDiags(), CGM.getContext(), CallLoc, CallerMap,
1628  CalleeMap, Ty, /*IsArgument*/ true))
1629  return;
1630  }
1631  ++ArgIndex;
1632  }
1633 
1634  // Check return always, as we don't have a good way of knowing in codegen
1635  // whether this value is used, tail-called, etc.
1636  if (Callee->getReturnType()->isVectorType() &&
1637  CGM.getContext().getTypeSize(Callee->getReturnType()) > 128) {
1638  initFeatureMaps(CGM.getContext(), CallerMap, Caller, CalleeMap, Callee);
1639  checkAVXParam(CGM.getDiags(), CGM.getContext(), CallLoc, CallerMap,
1640  CalleeMap, Callee->getReturnType(),
1641  /*IsArgument*/ false);
1642  }
1643 }

1755  // memory.
1756  //
1757  // (b) If X87UP is not preceded by X87, the whole argument is passed in
1758  // memory.
1759  //
1760  // (c) If the size of the aggregate exceeds two eightbytes and the first
1761  // eightbyte isn't SSE or any other eightbyte isn't SSEUP, the whole
1762  // argument is passed in memory. NOTE: This is necessary to keep the
1763  // ABI working for processors that don't support the __m256 type.
1764  //
1765  // (d) If SSEUP is not preceded by SSE or SSEUP, it is converted to SSE.
1766  //
1767  // Some of these are enforced by the merging logic. Others can arise
1768  // only with unions; for example:
1769  // union { _Complex double; unsigned; }
1770  //
1771  // Note that clauses (b) and (c) were added in 0.98.
1772  //
1773  if (Hi == Memory)
1774  Lo = Memory;
1775  if (Hi == X87Up && Lo != X87 && honorsRevision0_98())
1776  Lo = Memory;
1777  if (AggregateSize > 128 && (Lo != SSE || Hi != SSEUp))
1778  Lo = Memory;
1779  if (Hi == SSEUp && Lo != SSE)
1780  Hi = SSE;
1781 }

1783 X86_64ABIInfo::Class X86_64ABIInfo::merge(Class Accum, Class Field) {
1784  // AMD64-ABI 3.2.3p2: Rule 4. Each field of an object is
1785  // classified recursively so that always two fields are
1786  // considered. The resulting class is calculated according to
1787  // the classes of the fields in the eightbyte:
1788  //
1789  // (a) If both classes are equal, this is the resulting class.
1790  //
1791  // (b) If one of the classes is NO_CLASS, the resulting class is
1792  // the other class.
1793  //
1794  // (c) If one of the classes is MEMORY, the result is the MEMORY
1795  // class.
1796  //
1797  // (d) If one of the classes is INTEGER, the result is the
1798  // INTEGER.
1799  //
1800  // (e) If one of the classes is X87, X87UP, COMPLEX_X87 class,
1801  // MEMORY is used as class.
1802  //
1803  // (f) Otherwise class SSE is used.
1804 
1805  // Accum should never be memory (we should have returned) or
1806  // ComplexX87 (because this cannot be passed in a structure).
1807  assert((Accum != Memory && Accum != ComplexX87) &&
1808  "Invalid accumulated classification during merge.");
1809  if (Accum == Field || Field == NoClass)
1810  return Accum;
1811  if (Field == Memory)
1812  return Memory;
1813  if (Accum == NoClass)
1814  return Field;
1815  if (Accum == Integer || Field == Integer)
1816  return Integer;
1817  if (Field == X87 || Field == X87Up || Field == ComplexX87 ||
1818  Accum == X87 || Accum == X87Up)
1819  return Memory;
1820  return SSE;
1821 }

2330 /// is known to either be off the end of the specified type or being in
2331 /// alignment padding. The user type specified is known to be at most 128 bits
2332 /// in size, and have passed through X86_64ABIInfo::classify with a successful
2333 /// classification that put one of the two halves in the INTEGER class.
2334 ///
2335 /// It is conservatively correct to return false.
2336 static bool BitsContainNoUserData(QualType Ty, unsigned StartBit,
2337  unsigned EndBit, ASTContext &Context) {
2338  // If the bytes being queried are off the end of the type, there is no user
2339  // data hiding here. This handles analysis of builtins, vectors and other
2340  // types that don't contain interesting padding.
2341  unsigned TySize = (unsigned)Context.getTypeSize(Ty);
2342  if (TySize <= StartBit)
2343  return true;
2344 
2345  if (const ConstantArrayType *AT = Context.getAsConstantArrayType(Ty)) {
2346  unsigned EltSize = (unsigned)Context.getTypeSize(AT->getElementType());
2347  unsigned NumElts = (unsigned)AT->getZExtSize();
2348 
2349  // Check each element to see if the element overlaps with the queried range.
2350  for (unsigned i = 0; i != NumElts; ++i) {
2351  // If the element is after the span we care about, then we're done..
2352  unsigned EltOffset = i*EltSize;
2353  if (EltOffset >= EndBit) break;
2354 
2355  unsigned EltStart = EltOffset < StartBit ? StartBit-EltOffset :0;
2356  if (!BitsContainNoUserData(AT->getElementType(), EltStart,
2357  EndBit-EltOffset, Context))
2358  return false;
2359  }
2360  // If it overlaps no elements, then it is safe to process as padding.
2361  return true;
2362  }
2363 
2364  if (const RecordType *RT = Ty->getAs<RecordType>()) {
2365  const RecordDecl *RD = RT->getDecl();
2366  const ASTRecordLayout &Layout = Context.getASTRecordLayout(RD);
2367 
2368  // If this is a C++ record, check the bases first.
2369  if (const CXXRecordDecl *CXXRD = dyn_cast<CXXRecordDecl>(RD)) {
2370  for (const auto &I : CXXRD->bases()) {
2371  assert(!I.isVirtual() && !I.getType()->isDependentType() &&
2372  "Unexpected base class!");
2373  const auto *Base =
2374  cast<CXXRecordDecl>(I.getType()->castAs<RecordType>()->getDecl());
2375 
2376  // If the base is after the span we care about, ignore it.
2377  unsigned BaseOffset = Context.toBits(Layout.getBaseClassOffset(Base));
2378  if (BaseOffset >= EndBit) continue;
2379 
2380  unsigned BaseStart = BaseOffset < StartBit ? StartBit-BaseOffset :0;
2381  if (!BitsContainNoUserData(I.getType(), BaseStart,
2382  EndBit-BaseOffset, Context))
2383  return false;
2384  }
2385  }
2386 
2387  // Verify that no field has data that overlaps the region of interest. Yes
2388  // this could be sped up a lot by being smarter about queried fields,
2389  // however we're only looking at structs up to 16 bytes, so we don't care
2390  // much.
2391  unsigned idx = 0;
2392  for (RecordDecl::field_iterator i = RD->field_begin(), e = RD->field_end();
2393  i != e; ++i, ++idx) {
2394  unsigned FieldOffset = (unsigned)Layout.getFieldOffset(idx);
2395 
2396  // If we found a field after the region we care about, then we're done.
2397  if (FieldOffset >= EndBit) break;
2398 
2399  unsigned FieldStart = FieldOffset < StartBit ? StartBit-FieldOffset :0;
2400  if (!BitsContainNoUserData(i->getType(), FieldStart, EndBit-FieldOffset,
2401  Context))
2402  return false;
2403  }
2404 
2405  // If nothing in this record overlapped the area of interest, then we're
2406  // clean.
2407  return true;
2408  }
2409 
2410  return false;
2411 }

2414 static llvm::Type *getFPTypeAtOffset(llvm::Type *IRType, unsigned IROffset,

2417  return IRType;

2434  IROffset -= IROffset / EltSize * EltSize;
2435  return getFPTypeAtOffset(EltTy, IROffset, TD);

2441 /// GetSSETypeAtOffset - Return a type that will be passed by the backend in the
2442 /// low 8 bytes of an XMM register, corresponding to the SSE class.
2443 llvm::Type *X86_64ABIInfo::
2444 GetSSETypeAtOffset(llvm::Type *IRType, unsigned IROffset,
2445  QualType SourceTy, unsigned SourceOffset) const {
2446  const llvm::DataLayout &TD = getDataLayout();
2447  unsigned SourceSize =
2448  (unsigned)getContext().getTypeSize(SourceTy) / 8 - SourceOffset;
2449  llvm::Type *T0 = getFPTypeAtOffset(IRType, IROffset, TD);
2450  if (!T0 || T0->isDoubleTy())
2451  return llvm::Type::getDoubleTy(getVMContext());
2452 
2453  // Get the adjacent FP type.
2454  llvm::Type *T1 = nullptr;
2455  unsigned T0Size = TD.getTypeAllocSize(T0);
2456  if (SourceSize > T0Size)
2457  T1 = getFPTypeAtOffset(IRType, IROffset + T0Size, TD);
2458  if (T1 == nullptr) {
2459  // Check if IRType is a half/bfloat + float. float type will be in IROffset+4 due
2460  // to its alignment.
2461  if (T0->is16bitFPTy() && SourceSize > 4)
2462  T1 = getFPTypeAtOffset(IRType, IROffset + 4, TD);
2463  // If we can't get a second FP type, return a simple half or float.
2464  // avx512fp16-abi.c:pr51813_2 shows it works to return float for
2465  // {float, i8} too.
2466  if (T1 == nullptr)
2467  return T0;
2468  }
2469 
2470  if (T0->isFloatTy() && T1->isFloatTy())
2471  return llvm::FixedVectorType::get(T0, 2);
2472 
2473  if (T0->is16bitFPTy() && T1->is16bitFPTy()) {
2474  llvm::Type *T2 = nullptr;
2475  if (SourceSize > 4)
2476  T2 = getFPTypeAtOffset(IRType, IROffset + 4, TD);
2477  if (T2 == nullptr)
2478  return llvm::FixedVectorType::get(T0, 2);
2479  return llvm::FixedVectorType::get(T0, 4);
2480  }
2481 
2482  if (T0->is16bitFPTy() || T1->is16bitFPTy())
2483  return llvm::FixedVectorType::get(llvm::Type::getHalfTy(getVMContext()), 4);
2484 
2485  return llvm::Type::getDoubleTy(getVMContext());
2486 }

2489 /// GetINTEGERTypeAtOffset - The ABI specifies that a value should be passed in
2490 /// an 8-byte GPR. This means that we either have a scalar or we are talking
2491 /// about the high or low part of an up-to-16-byte struct. This routine picks
2492 /// the best LLVM IR type to represent this, which may be i64 or may be anything
2493 /// else that the backend will pass in a GPR that works better (e.g. i8, %foo*,
2494 /// etc).
2495 ///
2496 /// PrefType is an LLVM IR type that corresponds to (part of) the IR type for
2497 /// the source type. IROffset is an offset in bytes into the LLVM IR type that
2498 /// the 8-byte value references. PrefType may be null.
2499 ///
2500 /// SourceTy is the source-level type for the entire argument. SourceOffset is
2501 /// an offset into this that we're processing (which is always either 0 or 8).
2502 ///
2503 llvm::Type *X86_64ABIInfo::
2504 GetINTEGERTypeAtOffset(llvm::Type *IRType, unsigned IROffset,
2505  QualType SourceTy, unsigned SourceOffset) const {
2506  // If we're dealing with an un-offset LLVM IR type, then it means that we're
2507  // returning an 8-byte unit starting with it. See if we can safely use it.
2508  if (IROffset == 0) {
2509  // Pointers and int64's always fill the 8-byte unit.
2510  if ((isa<llvm::PointerType>(IRType) && Has64BitPointers) ||
2511  IRType->isIntegerTy(64))
2512  return IRType;
2513 
2514  // If we have a 1/2/4-byte integer, we can use it only if the rest of the
2515  // goodness in the source type is just tail padding. This is allowed to
2516  // kick in for struct {double,int} on the int, but not on
2517  // struct{double,int,int} because we wouldn't return the second int. We
2518  // have to do this analysis on the source type because we can't depend on
2519  // unions being lowered a specific way etc.
2520  if (IRType->isIntegerTy(8) || IRType->isIntegerTy(16) ||
2521  IRType->isIntegerTy(32) ||
2522  (isa<llvm::PointerType>(IRType) && !Has64BitPointers)) {
2523  unsigned BitWidth = isa<llvm::PointerType>(IRType) ? 32 :
2524  cast<llvm::IntegerType>(IRType)->getBitWidth();
2525 
2526  if (BitsContainNoUserData(SourceTy, SourceOffset*8+BitWidth,
2527  SourceOffset*8+64, getContext()))
2528  return IRType;
2529  }
2530  }
2531 
2532  if (llvm::StructType *STy = dyn_cast<llvm::StructType>(IRType)) {
2533  // If this is a struct, recurse into the field at the specified offset.
2534  const llvm::StructLayout *SL = getDataLayout().getStructLayout(STy);
2535  if (IROffset < SL->getSizeInBytes()) {
2536  unsigned FieldIdx = SL->getElementContainingOffset(IROffset);
2537  IROffset -= SL->getElementOffset(FieldIdx);
2538 
2539  return GetINTEGERTypeAtOffset(STy->getElementType(FieldIdx), IROffset,
2540  SourceTy, SourceOffset);
2541  }
2542  }
2543 
2544  if (llvm::ArrayType *ATy = dyn_cast<llvm::ArrayType>(IRType)) {
2545  llvm::Type *EltTy = ATy->getElementType();
2546  unsigned EltSize = getDataLayout().getTypeAllocSize(EltTy);
2547  unsigned EltOffset = IROffset/EltSize*EltSize;
2548  return GetINTEGERTypeAtOffset(EltTy, IROffset-EltOffset, SourceTy,
2549  SourceOffset);
2550  }
2551 
2552  // Okay, we don't have any better idea of what to pass, so we pass this in an
2553  // integer register that isn't too big to fit the rest of the struct.
2554  unsigned TySizeInBytes =
2555  (unsigned)getContext().getTypeSizeInChars(SourceTy).getQuantity();
2556 
2557  assert(TySizeInBytes != SourceOffset && "Empty field?");
2558 
2559  // It is always safe to classify this as an integer type up to i64 that
2560  // isn't larger than the structure.
2561  return llvm::IntegerType::get(getVMContext(),
2562  std::min(TySizeInBytes-SourceOffset, 8U)*8);
2563 }

2566 /// GetX86_64ByValArgumentPair - Given a high and low type that can ideally
2567 /// be used as elements of a two register pair to pass or return, return a
2568 /// first class aggregate to represent them. For example, if the low part of
2569 /// a by-value argument should be passed as i32* and the high part as float,
2570 /// return {i32*, float}.
2571 static llvm::Type *
2572 GetX86_64ByValArgumentPair(llvm::Type *Lo, llvm::Type *Hi,
2573  const llvm::DataLayout &TD) {
2574  // In order to correctly satisfy the ABI, we need to the high part to start
2575  // at offset 8. If the high and low parts we inferred are both 4-byte types
2576  // (e.g. i32 and i32) then the resultant struct type ({i32,i32}) won't have
2577  // the second element at offset 8. Check for this:
2578  unsigned LoSize = (unsigned)TD.getTypeAllocSize(Lo);
2579  llvm::Align HiAlign = TD.getABITypeAlign(Hi);
2580  unsigned HiStart = llvm::alignTo(LoSize, HiAlign);
2581  assert(HiStart != 0 && HiStart <= 8 && "Invalid x86-64 argument pair!");
2582 
2583  // To handle this, we have to increase the size of the low part so that the
2584  // second element will start at an 8 byte offset. We can't increase the size
2585  // of the second element because it might make us access off the end of the
2586  // struct.
2587  if (HiStart != 8) {
2588  // There are usually two sorts of types the ABI generation code can produce
2589  // for the low part of a pair that aren't 8 bytes in size: half, float or
2590  // i8/i16/i32. This can also include pointers when they are 32-bit (X32 and
2591  // NaCl).
2592  // Promote these to a larger type.
2593  if (Lo->isHalfTy() || Lo->isFloatTy())
2594  Lo = llvm::Type::getDoubleTy(Lo->getContext());
2595  else {
2596  assert((Lo->isIntegerTy() || Lo->isPointerTy())
2597  && "Invalid/unknown lo type");
2598  Lo = llvm::Type::getInt64Ty(Lo->getContext());
2599  }
2600  }
2601 
2602  llvm::StructType *Result = llvm::StructType::get(Lo, Hi);
2603 
2604  // Verify that the second element is at an 8-byte offset.
2605  assert(TD.getStructLayout(Result)->getElementOffset(1) == 8 &&
2606  "Invalid x86-64 argument pair!");
2607  return Result;
2608 }

2610 ABIArgInfo X86_64ABIInfo::
2611 classifyReturnType(QualType RetTy) const {
2612  // AMD64-ABI 3.2.3p4: Rule 1. Classify the return type with the
2613  // classification algorithm.
2614  X86_64ABIInfo::Class Lo, Hi;
2615  classify(RetTy, 0, Lo, Hi, /*isNamedArg*/ true);
2616 
2617  // Check some invariants.
2618  assert((Hi != Memory || Lo == Memory) && "Invalid memory classification.");
2619  assert((Hi != SSEUp || Lo == SSE) && "Invalid SSEUp classification.");
2620 
2621  llvm::Type *ResType = nullptr;
2622  switch (Lo) {
2623  case NoClass:
2624  if (Hi == NoClass)
2625  return ABIArgInfo::getIgnore();
2626  // If the low part is just padding, it takes no register, leave ResType
2627  // null.
2628  assert((Hi == SSE || Hi == Integer || Hi == X87Up) &&
2629  "Unknown missing lo part");
2630  break;
2631 
2632  case SSEUp:
2633  case X87Up:
2634  llvm_unreachable("Invalid classification for lo word.");
2635 
2636  // AMD64-ABI 3.2.3p4: Rule 2. Types of class memory are returned via
2637  // hidden argument.
2638  case Memory:
2639  return getIndirectReturnResult(RetTy);
2640 
2641  // AMD64-ABI 3.2.3p4: Rule 3. If the class is INTEGER, the next
2642  // available register of the sequence %rax, %rdx is used.
2643  case Integer:
2644  ResType = GetINTEGERTypeAtOffset(CGT.ConvertType(RetTy), 0, RetTy, 0);
2645 
2646  // If we have a sign or zero extended integer, make sure to return Extend
2647  // so that the parameter gets the right LLVM IR attributes.
2648  if (Hi == NoClass && isa<llvm::IntegerType>(ResType)) {
2649  // Treat an enum type as its underlying type.
2650  if (const EnumType *EnumTy = RetTy->getAs<EnumType>())
2651  RetTy = EnumTy->getDecl()->getIntegerType();
2652 
2653  if (RetTy->isIntegralOrEnumerationType() &&
2654  isPromotableIntegerTypeForABI(RetTy))
2655  return ABIArgInfo::getExtend(RetTy);
2656  }
2657  break;
2658 
2659  // AMD64-ABI 3.2.3p4: Rule 4. If the class is SSE, the next
2660  // available SSE register of the sequence %xmm0, %xmm1 is used.
2661  case SSE:
2662  ResType = GetSSETypeAtOffset(CGT.ConvertType(RetTy), 0, RetTy, 0);
2663  break;
2664 
2665  // AMD64-ABI 3.2.3p4: Rule 6. If the class is X87, the value is
2666  // returned on the X87 stack in %st0 as 80-bit x87 number.
2667  case X87:
2668  ResType = llvm::Type::getX86_FP80Ty(getVMContext());
2669  break;
2670 
2671  // AMD64-ABI 3.2.3p4: Rule 8. If the class is COMPLEX_X87, the real
2672  // part of the value is returned in %st0 and the imaginary part in
2673  // %st1.
2674  case ComplexX87:
2675  assert(Hi == ComplexX87 && "Unexpected ComplexX87 classification.");
2676  ResType = llvm::StructType::get(llvm::Type::getX86_FP80Ty(getVMContext()),
2677  llvm::Type::getX86_FP80Ty(getVMContext()));
2678  break;
2679  }
2680 
2681  llvm::Type *HighPart = nullptr;
2682  switch (Hi) {
2683  // Memory was handled previously and X87 should
2684  // never occur as a hi class.
2685  case Memory:
2686  case X87:
2687  llvm_unreachable("Invalid classification for hi word.");
2688 
2689  case ComplexX87: // Previously handled.
2690  case NoClass:
2691  break;
2692 
2693  case Integer:
2694  HighPart = GetINTEGERTypeAtOffset(CGT.ConvertType(RetTy), 8, RetTy, 8);
2695  if (Lo == NoClass) // Return HighPart at offset 8 in memory.
2696  return ABIArgInfo::getDirect(HighPart, 8);
2697  break;
2698  case SSE:
2699  HighPart = GetSSETypeAtOffset(CGT.ConvertType(RetTy), 8, RetTy, 8);
2700  if (Lo == NoClass) // Return HighPart at offset 8 in memory.
2701  return ABIArgInfo::getDirect(HighPart, 8);
2702  break;
2703 
2704  // AMD64-ABI 3.2.3p4: Rule 5. If the class is SSEUP, the eightbyte
2705  // is passed in the next available eightbyte chunk if the last used
2706  // vector register.
2707  //
2708  // SSEUP should always be preceded by SSE, just widen.
2709  case SSEUp:
2710  assert(Lo == SSE && "Unexpected SSEUp classification.");
2711  ResType = GetByteVectorType(RetTy);
2712  break;
2713 
2714  // AMD64-ABI 3.2.3p4: Rule 7. If the class is X87UP, the value is
2715  // returned together with the previous X87 value in %st0.
2716  case X87Up:
2717  // If X87Up is preceded by X87, we don't need to do
2718  // anything. However, in some cases with unions it may not be
2719  // preceded by X87. In such situations we follow gcc and pass the
2720  // extra bits in an SSE reg.
2721  if (Lo != X87) {
2722  HighPart = GetSSETypeAtOffset(CGT.ConvertType(RetTy), 8, RetTy, 8);
2723  if (Lo == NoClass) // Return HighPart at offset 8 in memory.
2724  return ABIArgInfo::getDirect(HighPart, 8);
2725  }
2726  break;
2727  }
2728 
2729  // If a high part was specified, merge it together with the low part. It is
2730  // known to pass in the high eightbyte of the result. We do this by forming a
2731  // first class struct aggregate with the high and low part: {low, high}
2732  if (HighPart)
2733  ResType = GetX86_64ByValArgumentPair(ResType, HighPart, getDataLayout());
2734 
2735  return ABIArgInfo::getDirect(ResType);
2736 }

2739 X86_64ABIInfo::classifyArgumentType(QualType Ty, unsigned freeIntRegs,
2740  unsigned &neededInt, unsigned &neededSSE,
2741  bool isNamedArg, bool IsRegCall) const {

2743 
2744  X86_64ABIInfo::Class Lo, Hi;
2745  classify(Ty, 0, Lo, Hi, isNamedArg, IsRegCall);
2746 
2747  // Check some invariants.
2748  // FIXME: Enforce these by construction.
2749  assert((Hi != Memory || Lo == Memory) && "Invalid memory classification.");
2750  assert((Hi != SSEUp || Lo == SSE) && "Invalid SSEUp classification.");
2751 
2752  neededInt = 0;
2753  neededSSE = 0;
2754  llvm::Type *ResType = nullptr;
2755  switch (Lo) {
2756  case NoClass:
2757  if (Hi == NoClass)
2758  return ABIArgInfo::getIgnore();
2759  // If the low part is just padding, it takes no register, leave ResType
2760  // null.
2761  assert((Hi == SSE || Hi == Integer || Hi == X87Up) &&
2762  "Unknown missing lo part");
2763  break;
2764 
2765  // AMD64-ABI 3.2.3p3: Rule 1. If the class is MEMORY, pass the argument
2766  // on the stack.
2767  case Memory:
2768 
2769  // AMD64-ABI 3.2.3p3: Rule 5. If the class is X87, X87UP or
2770  // COMPLEX_X87, it is passed in memory.
2771  case X87:
2772  case ComplexX87:
2773  if (getRecordArgABI(Ty, getCXXABI()) == CGCXXABI::RAA_Indirect)
2774  ++neededInt;
2775  return getIndirectResult(Ty, freeIntRegs);
2776 
2777  case SSEUp:
2778  case X87Up:
2779  llvm_unreachable("Invalid classification for lo word.");
2780 
2781  // AMD64-ABI 3.2.3p3: Rule 2. If the class is INTEGER, the next
2782  // available register of the sequence %rdi, %rsi, %rdx, %rcx, %r8
2783  // and %r9 is used.
2784  case Integer:
2785  ++neededInt;
2786 
2787  // Pick an 8-byte type based on the preferred type.
2788  ResType = GetINTEGERTypeAtOffset(CGT.ConvertType(Ty), 0, Ty, 0);
2789 
2790  // If we have a sign or zero extended integer, make sure to return Extend
2791  // so that the parameter gets the right LLVM IR attributes.
2792  if (Hi == NoClass && isa<llvm::IntegerType>(ResType)) {
2793  // Treat an enum type as its underlying type.
2794  if (const EnumType *EnumTy = Ty->getAs<EnumType>())
2795  Ty = EnumTy->getDecl()->getIntegerType();
2796 
2797  if (Ty->isIntegralOrEnumerationType() &&
2798  isPromotableIntegerTypeForABI(Ty))
2799  return ABIArgInfo::getExtend(Ty);
2800  }
2801 
2802  break;
2803 
2804  // AMD64-ABI 3.2.3p3: Rule 3. If the class is SSE, the next
2805  // available SSE register is used, the registers are taken in the
2806  // order from %xmm0 to %xmm7.
2807  case SSE: {
2808  llvm::Type *IRType = CGT.ConvertType(Ty);
2809  ResType = GetSSETypeAtOffset(IRType, 0, Ty, 0);
2810  ++neededSSE;
2811  break;
2812  }
2813  }
2814 
2815  llvm::Type *HighPart = nullptr;
2816  switch (Hi) {
2817  // Memory was handled previously, ComplexX87 and X87 should
2818  // never occur as hi classes, and X87Up must be preceded by X87,
2819  // which is passed in memory.
2820  case Memory:
2821  case X87:
2822  case ComplexX87:
2823  llvm_unreachable("Invalid classification for hi word.");
2824 
2825  case NoClass: break;
2826 
2827  case Integer:
2828  ++neededInt;
2829  // Pick an 8-byte type based on the preferred type.
2830  HighPart = GetINTEGERTypeAtOffset(CGT.ConvertType(Ty), 8, Ty, 8);
2831 
2832  if (Lo == NoClass) // Pass HighPart at offset 8 in memory.
2833  return ABIArgInfo::getDirect(HighPart, 8);
2834  break;
2835 
2836  // X87Up generally doesn't occur here (long double is passed in
2837  // memory), except in situations involving unions.
2838  case X87Up:
2839  case SSE:
2840  ++neededSSE;
2841  HighPart = GetSSETypeAtOffset(CGT.ConvertType(Ty), 8, Ty, 8);
2842 
2843  if (Lo == NoClass) // Pass HighPart at offset 8 in memory.
2844  return ABIArgInfo::getDirect(HighPart, 8);
2845  break;
2846 
2847  // AMD64-ABI 3.2.3p3: Rule 4. If the class is SSEUP, the
2848  // eightbyte is passed in the upper half of the last used SSE
2849  // register. This only happens when 128-bit vectors are passed.
2850  case SSEUp:
2851  assert(Lo == SSE && "Unexpected SSEUp classification");
2852  ResType = GetByteVectorType(Ty);
2853  break;
2854  }

2857  // known to pass in the high eightbyte of the result. We do this by forming a
2858  // first class struct aggregate with the high and low part: {low, high}
2859  if (HighPart)
2860  ResType = GetX86_64ByValArgumentPair(ResType, HighPart, getDataLayout());
2861
