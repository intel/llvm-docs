363  // FIXME: Eliminate these ...Target functions and derive separate tool chains

365  void setTarget(DarwinPlatformKind Platform, DarwinEnvironmentKind Environment,
366  unsigned Major, unsigned Minor, unsigned Micro,
367  VersionTuple NativeTargetVersion) const {
368  // FIXME: For now, allow reinitialization as long as values don't
369  // change. This will go away when we move away from argument translation.
370  if (TargetInitialized && TargetPlatform == Platform &&
371  TargetEnvironment == Environment &&
372  (Environment == MacCatalyst ? OSTargetVersion : TargetVersion) ==
373  VersionTuple(Major, Minor, Micro))
374  return;
375 
376  assert(!TargetInitialized && "Target already initialized!");
377  TargetInitialized = true;
378  TargetPlatform = Platform;
379  TargetEnvironment = Environment;
380  TargetVersion = VersionTuple(Major, Minor, Micro);
381  if (Environment == Simulator)
382  const_cast<Darwin *>(this)->setTripleEnvironment(llvm::Triple::Simulator);
383  else if (Environment == MacCatalyst) {
384  const_cast<Darwin *>(this)->setTripleEnvironment(llvm::Triple::MacABI);
385  TargetVersion = NativeTargetVersion;

474  /// Returns true if the minimum supported macOS version for the slice that's
475  /// being built is less than the specified version. If there's no minimum
476  /// supported macOS version, the deployment target version is compared to the
477  /// specifed version instead.
478  bool isMacosxVersionLT(unsigned V0, unsigned V1 = 0, unsigned V2 = 0) const {
479  assert(isTargetMacOSBased() &&
480  (getTriple().isMacOSX() || getTriple().isMacCatalystEnvironment()) &&
481  "Unexpected call for non OS X target!");
482  // The effective triple might not be initialized yet, so construct a
483  // pseudo-effective triple to get the minimum supported OS version.
484  VersionTuple MinVers =
485  llvm::Triple(getTriple().getArchName(), "apple", "macos")
486  .getMinimumSupportedOSVersion();
487  return (!MinVers.empty() && MinVers > TargetVersion
488  ? MinVers
489  : TargetVersion) < VersionTuple(V0, V1, V2);
490  }
