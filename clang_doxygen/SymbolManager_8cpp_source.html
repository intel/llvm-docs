127 void SymExpr::symbol_iterator::expand() {
128  const SymExpr *SE = itr.pop_back_val();
129 
130  switch (SE->getKind()) {
131  case SymExpr::SymbolRegionValueKind:
132  case SymExpr::SymbolConjuredKind:
133  case SymExpr::SymbolDerivedKind:
134  case SymExpr::SymbolExtentKind:
135  case SymExpr::SymbolMetadataKind:
136  return;
137  case SymExpr::SymbolCastKind:
138  itr.push_back(cast<SymbolCast>(SE)->getOperand());
139  return;
140  case SymExpr::SymIntExprKind:
141  itr.push_back(cast<SymIntExpr>(SE)->getLHS());
142  return;
143  case SymExpr::IntSymExprKind:
144  itr.push_back(cast<IntSymExpr>(SE)->getRHS());
145  return;
146  case SymExpr::SymSymExprKind: {
147  const auto *x = cast<SymSymExpr>(SE);
148  itr.push_back(x->getLHS());
149  itr.push_back(x->getRHS());
150  return;
151  }
152  }
153  llvm_unreachable("unhandled expansion case");
154 }

365 void SymbolReaper::markDependentsLive(SymbolRef sym) {
366  // Do not mark dependents more then once.
367  SymbolMapTy::iterator LI = TheLiving.find(sym);
368  assert(LI != TheLiving.end() && "The primary symbol is not live.");
369  if (LI->second == HaveMarkedDependents)
370  return;
371  LI->second = HaveMarkedDependents;
372 
373  if (const SymbolRefSmallVectorTy *Deps = SymMgr.getDependentSymbols(sym)) {
374  for (const auto I : *Deps) {
375  if (TheLiving.find(I) != TheLiving.end())
376  continue;
377  markLive(I);
378  }
379  }

408 bool SymbolReaper::isLiveRegion(const MemRegion *MR) {
409  // TODO: For now, liveness of a memory region is equivalent to liveness of its
410  // base region. In fact we can do a bit better: say, if a particular FieldDecl
411  // is not used later in the path, we can diagnose a leak of a value within
412  // that field earlier than, say, the variable that contains the field dies.
413  MR = MR->getBaseRegion();
414 
415  if (RegionRoots.count(MR))
416  return true;
417 
418  if (const auto *SR = dyn_cast<SymbolicRegion>(MR))
419  return isLive(SR->getSymbol());
420 
421  if (const auto *VR = dyn_cast<VarRegion>(MR))
422  return isLive(VR, true);
423 
424  // FIXME: This is a gross over-approximation. What we really need is a way to
425  // tell if anything still refers to this region. Unlike SymbolicRegions,
426  // AllocaRegions don't have associated symbols, though, so we don't actually
427  // have a way to track their liveness.
428  return isa<AllocaRegion, CXXThisRegion, MemSpaceRegion, CodeTextRegion>(MR);
429 }

431 bool SymbolReaper::isLive(SymbolRef sym) {
432  if (TheLiving.count(sym)) {
433  markDependentsLive(sym);
434  return true;
435  }
436 
437  bool KnownLive;
438 
439  switch (sym->getKind()) {
440  case SymExpr::SymbolRegionValueKind:
441  KnownLive = isLiveRegion(cast<SymbolRegionValue>(sym)->getRegion());
442  break;
443  case SymExpr::SymbolConjuredKind:
444  KnownLive = false;
445  break;
446  case SymExpr::SymbolDerivedKind:
447  KnownLive = isLive(cast<SymbolDerived>(sym)->getParentSymbol());
448  break;
449  case SymExpr::SymbolExtentKind:
450  KnownLive = isLiveRegion(cast<SymbolExtent>(sym)->getRegion());
451  break;
452  case SymExpr::SymbolMetadataKind:
453  KnownLive = MetadataInUse.count(sym) &&
454  isLiveRegion(cast<SymbolMetadata>(sym)->getRegion());
455  if (KnownLive)
456  MetadataInUse.erase(sym);
457  break;
458  case SymExpr::SymIntExprKind:
459  KnownLive = isLive(cast<SymIntExpr>(sym)->getLHS());
460  break;
461  case SymExpr::IntSymExprKind:
462  KnownLive = isLive(cast<IntSymExpr>(sym)->getRHS());
463  break;
464  case SymExpr::SymSymExprKind:
465  KnownLive = isLive(cast<SymSymExpr>(sym)->getLHS()) &&
466  isLive(cast<SymSymExpr>(sym)->getRHS());
467  break;
468  case SymExpr::SymbolCastKind:
469  KnownLive = isLive(cast<SymbolCast>(sym)->getOperand());
470  break;
471  }
472 
473  if (KnownLive)
474  markLive(sym);
475 
476  return KnownLive;
477 }

480 SymbolReaper::isLive(const Expr *ExprVal, const LocationContext *ELCtx) const {
481  if (LCtx == nullptr)
482  return false;
483 
484  if (LCtx != ELCtx) {
485  // If the reaper's location context is a parent of the expression's
486  // location context, then the expression value is now "out of scope".

500 bool SymbolReaper::isLive(const VarRegion *VR, bool includeStoreBindings) const{
501  const StackFrameContext *VarContext = VR->getStackFrame();
502 
503  if (!VarContext)
504  return true;
505 
506  if (!LCtx)
507  return false;
508  const StackFrameContext *CurrentContext = LCtx->getStackFrame();
509 
510  if (VarContext == CurrentContext) {
511  // If no statement is provided, everything is live.
512  if (!Loc)
513  return true;
514 
515  // Anonymous parameters of an inheriting constructor are live for the entire
516  // duration of the constructor.
517  if (isa<CXXInheritedCtorInitExpr>(Loc))
518  return true;
519 
520  if (LCtx->getAnalysis<RelaxedLiveVariables>()->isLive(Loc, VR->getDecl()))
521  return true;
522 
523  if (!includeStoreBindings)
524  return false;
525 
526  unsigned &cachedQuery =
527  const_cast<SymbolReaper *>(this)->includedRegionCache[VR];
528 
529  if (cachedQuery) {
530  return cachedQuery == 1;
531  }
532 
533  // Query the store to see if the region occurs in any live bindings.
534  if (Store store = reapedStore.getStore()) {
535  bool hasRegion =
536  reapedStore.getStoreManager().includedInBindings(store, VR);
537  cachedQuery = hasRegion ? 1 : 2;
538  return hasRegion;
539  }
540 
541  return false;
542  }
543 
544  return VarContext->isParentOf(CurrentContext);
545 }
