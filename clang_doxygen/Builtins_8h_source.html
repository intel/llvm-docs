30 enum LanguageID {
31  GNU_LANG = 0x1, // builtin requires GNU mode.
32  C_LANG = 0x2, // builtin for c only.
33  CXX_LANG = 0x4, // builtin for cplusplus only.
34  OBJC_LANG = 0x8, // builtin for objective-c and objective-c++
35  MS_LANG = 0x10, // builtin requires MS mode.
36  OMP_LANG = 0x20, // builtin requires OpenMP.
37  CUDA_LANG = 0x40, // builtin requires CUDA.
38  COR_LANG = 0x80, // builtin requires use of 'fcoroutine-ts' option.
39  OCL_GAS = 0x100, // builtin requires OpenCL generic address space.
40  OCL_PIPE = 0x200, // builtin requires OpenCL pipe.
41  OCL_DSE = 0x400, // builtin requires OpenCL device side enqueue.
42  ALL_OCL_LANGUAGES = 0x800, // builtin for OCL languages.
43  ALL_LANGUAGES = C_LANG | CXX_LANG | OBJC_LANG, // builtin for all languages.
44  ALL_GNU_LANGUAGES = ALL_LANGUAGES | GNU_LANG, // builtin requires GNU mode.
45  ALL_MS_LANGUAGES = ALL_LANGUAGES | MS_LANG // builtin requires MS mode.
46 };

62 /// Holds information about both target-independent and
63 /// target-specific builtins, allowing easy queries by clients.
64 ///
65 /// Builtins from an optional auxiliary target are stored in
66 /// AuxTSRecords. Their IDs are shifted up by TSRecords.size() and need to
67 /// be translated back with getAuxBuiltinID() before use.
68 class Context {
69  llvm::ArrayRef<Info> TSRecords;
70  llvm::ArrayRef<Info> AuxTSRecords;
71 
72 public:
73  Context() = default;
74 
75  /// Perform target-specific initialization
76  /// \param AuxTarget Target info to incorporate builtins from. May be nullptr.
77  void InitializeTarget(const TargetInfo &Target, const TargetInfo *AuxTarget);
78 
79  /// Mark the identifiers for all the builtins with their
80  /// appropriate builtin ID # and mark any non-portable builtin identifiers as
81  /// such.
82  void initializeBuiltins(IdentifierTable &Table, const LangOptions& LangOpts);
83 
84  /// Return the identifier name for the specified builtin,
85  /// e.g. "__builtin_abs".
86  const char *getName(unsigned ID) const {
87  return getRecord(ID).Name;
88  }
89 
90  /// Get the type descriptor string for the specified builtin.
91  const char *getTypeString(unsigned ID) const {
92  return getRecord(ID).Type;
93  }
94 
95  /// Return true if this function is a target-specific builtin.
96  bool isTSBuiltin(unsigned ID) const {
97  return ID >= Builtin::FirstTSBuiltin;
98  }
99 
100  /// Return true if this function has no side effects.
101  bool isPure(unsigned ID) const {
102  return strchr(getRecord(ID).Attributes, 'U') != nullptr;
103  }
104 
105  /// Return true if this function has no side effects and doesn't
106  /// read memory.
107  bool isConst(unsigned ID) const {
108  return strchr(getRecord(ID).Attributes, 'c') != nullptr;
109  }
110 
111  /// Return true if we know this builtin never throws an exception.
112  bool isNoThrow(unsigned ID) const {
113  return strchr(getRecord(ID).Attributes, 'n') != nullptr;
114  }
115 
116  /// Return true if we know this builtin never returns.
117  bool isNoReturn(unsigned ID) const {
118  return strchr(getRecord(ID).Attributes, 'r') != nullptr;
119  }
120 
121  /// Return true if we know this builtin can return twice.
122  bool isReturnsTwice(unsigned ID) const {
123  return strchr(getRecord(ID).Attributes, 'j') != nullptr;
124  }
125 
126  /// Returns true if this builtin does not perform the side-effects
127  /// of its arguments.
128  bool isUnevaluated(unsigned ID) const {
129  return strchr(getRecord(ID).Attributes, 'u') != nullptr;
130  }
131 
132  /// Return true if this is a builtin for a libc/libm function,
133  /// with a "__builtin_" prefix (e.g. __builtin_abs).
134  bool isLibFunction(unsigned ID) const {
135  return strchr(getRecord(ID).Attributes, 'F') != nullptr;
136  }
137 
138  /// Determines whether this builtin is a predefined libc/libm
139  /// function, such as "malloc", where we know the signature a
140  /// priori.
141  bool isPredefinedLibFunction(unsigned ID) const {
142  return strchr(getRecord(ID).Attributes, 'f') != nullptr;
143  }
144 
145  /// Returns true if this builtin requires appropriate header in other
146  /// compilers. In Clang it will work even without including it, but we can emit
147  /// a warning about missing header.
148  bool isHeaderDependentFunction(unsigned ID) const {
149  return strchr(getRecord(ID).Attributes, 'h') != nullptr;
150  }
151 
152  /// Determines whether this builtin is a predefined compiler-rt/libgcc
153  /// function, such as "__clear_cache", where we know the signature a
154  /// priori.
155  bool isPredefinedRuntimeFunction(unsigned ID) const {
156  return strchr(getRecord(ID).Attributes, 'i') != nullptr;
157  }
158 
159  /// Determines whether this builtin has custom typechecking.
160  bool hasCustomTypechecking(unsigned ID) const {
161  return strchr(getRecord(ID).Attributes, 't') != nullptr;
162  }
163 
164  /// Determines whether a declaration of this builtin should be recognized
165  /// even if the type doesn't match the specified signature.
166  bool allowTypeMismatch(unsigned ID) const {
167  return strchr(getRecord(ID).Attributes, 'T') != nullptr ||
168  hasCustomTypechecking(ID);
169  }
170 
171  /// Determines whether this builtin has a result or any arguments which
172  /// are pointer types.
173  bool hasPtrArgsOrResult(unsigned ID) const {
174  return strchr(getRecord(ID).Type, '*') != nullptr;
175  }
176 
177  /// Return true if this builtin has a result or any arguments which are
178  /// reference types.
179  bool hasReferenceArgsOrResult(unsigned ID) const {
180  return strchr(getRecord(ID).Type, '&') != nullptr ||
181  strchr(getRecord(ID).Type, 'A') != nullptr;
182  }
183 
184  /// If this is a library function that comes from a specific
185  /// header, retrieve that header name.
186  const char *getHeaderName(unsigned ID) const {
187  return getRecord(ID).HeaderName;
188  }
189 
190  /// Determine whether this builtin is like printf in its
191  /// formatting rules and, if so, set the index to the format string
192  /// argument and whether this function as a va_list argument.
193  bool isPrintfLike(unsigned ID, unsigned &FormatIdx, bool &HasVAListArg);
194 
195  /// Determine whether this builtin is like scanf in its
196  /// formatting rules and, if so, set the index to the format string
197  /// argument and whether this function as a va_list argument.
198  bool isScanfLike(unsigned ID, unsigned &FormatIdx, bool &HasVAListArg);
199 
200  /// Determine whether this builtin has callback behavior (see
201  /// llvm::AbstractCallSites for details). If so, add the index to the
202  /// callback callee argument and the callback payload arguments.
203  bool performsCallback(unsigned ID,
204  llvm::SmallVectorImpl<int> &Encoding) const;
205 
206  /// Return true if this function has no side effects and doesn't
207  /// read memory, except for possibly errno.
208  ///
209  /// Such functions can be const when the MathErrno lang option is disabled.
210  bool isConstWithoutErrno(unsigned ID) const {
211  return strchr(getRecord(ID).Attributes, 'e') != nullptr;
212  }
213 
214  const char *getRequiredFeatures(unsigned ID) const {
215  return getRecord(ID).Features;
216  }
217 
218  unsigned getRequiredVectorWidth(unsigned ID) const;
219 
220  /// Return true if builtin ID belongs to AuxTarget.
221  bool isAuxBuiltinID(unsigned ID) const {
222  return ID >= (Builtin::FirstTSBuiltin + TSRecords.size());
223  }
224 
225  /// Return real builtin ID (i.e. ID it would have during compilation
226  /// for AuxTarget).
227  unsigned getAuxBuiltinID(unsigned ID) const { return ID - TSRecords.size(); }
228 
229  /// Returns true if this is a libc/libm function without the '__builtin_'
230  /// prefix.
231  static bool isBuiltinFunc(llvm::StringRef Name);
232 
233  /// Returns true if this is a builtin that can be redeclared. Returns true
234  /// for non-builtins.
235  bool canBeRedeclared(unsigned ID) const;
236 
237 private:
238  const Info &getRecord(unsigned ID) const;
239 
240  /// Is this builtin supported according to the given language options?
241  bool builtinIsSupported(const Builtin::Info &BuiltinInfo,
242  const LangOptions &LangOpts);
243 
244  /// Helper function for isPrintfLike and isScanfLike.
245  bool isLike(unsigned ID, unsigned &FormatIdx, bool &HasVAListArg,
246  const char *Fmt) const;
247 };
