177  DI->EmitAndRetainType(Ty);

185 /// EmitVarDecl - This method handles emission of any variable declaration
186 /// inside a function, including static vars etc.
187 void CodeGenFunction::EmitVarDecl(const VarDecl &D) {
188  if (D.hasExternalStorage())
189  // Don't emit it now, allow it to be emitted lazily on its first use.
190  return;
191 
192  // Some function-scope variable does not have static storage but still
193  // needs to be emitted like a static variable, e.g. a function-scope
194  // variable in constant address space in OpenCL.
195  if (D.getStorageDuration() != SD_Automatic) {
196  // Static sampler variables translated to function calls.
197  if (D.getType()->isSamplerT())
198  return;
199 
200  llvm::GlobalValue::LinkageTypes Linkage =
201  CGM.getLLVMLinkageVarDefinition(&D, /*IsConstant=*/false);
202 
203  // FIXME: We need to force the emission/use of a guard variable for
204  // some variables even if we can constant-evaluate them because
205  // we can't guarantee every translation unit will constant-evaluate them.
206 
207  return EmitStaticVarDecl(D, Linkage);
208  }
209 
210  if (D.getType().getAddressSpace() == LangAS::opencl_local)
211  return CGM.getOpenCLRuntime().EmitWorkGroupLocalVarDecl(*this, D);
212 
213  if (D.getAttr<SYCLScopeAttr>() && D.getAttr<SYCLScopeAttr>()->isWorkGroup())
214  return CGM.getSYCLRuntime().emitWorkGroupLocalVarDecl(*this, D);
215 
216  assert(D.hasLocalStorage());
217  return EmitAutoVarDecl(D);
218 }

220 static std::string getStaticDeclName(CodeGenModule &CGM, const VarDecl &D) {
221  if (CGM.getLangOpts().CPlusPlus)
222  return CGM.getMangledName(&D).str();
223 
224  // If this isn't C++, we don't need a mangled name, just a pretty one.
225  assert(!D.isExternallyVisible() && "name shouldn't matter");
226  std::string ContextName;
227  const DeclContext *DC = D.getDeclContext();
228  if (auto *CD = dyn_cast<CapturedDecl>(DC))
229  DC = cast<DeclContext>(CD->getNonClosureContext());
230  if (const auto *FD = dyn_cast<FunctionDecl>(DC))
231  ContextName = std::string(CGM.getMangledName(FD));
232  else if (const auto *BD = dyn_cast<BlockDecl>(DC))
233  ContextName = std::string(CGM.getBlockMangledName(GlobalDecl(), BD));
234  else if (const auto *OMD = dyn_cast<ObjCMethodDecl>(DC))
235  ContextName = OMD->getSelector().getAsString();
236  else
237  llvm_unreachable("Unknown context for static var decl");
238 
239  ContextName += "." + D.getNameAsString();
240  return ContextName;
241 }

243 llvm::Constant *CodeGenModule::getOrCreateStaticVarDecl(
244  const VarDecl &D, llvm::GlobalValue::LinkageTypes Linkage) {
245  // In general, we don't always emit static var decls once before we reference
246  // them. It is possible to reference them before emitting the function that
247  // contains them, and it is possible to emit the containing function multiple
248  // times.
249  if (llvm::Constant *ExistingGV = StaticLocalDeclMap[&D])
250  return ExistingGV;
251 
252  QualType Ty = D.getType();
253  assert(Ty->isConstantSizeType() && "VLAs can't be static");
254 
255  // Use the label if the variable is renamed with the asm-label extension.
256  std::string Name;
257  if (D.hasAttr<AsmLabelAttr>())
258  Name = std::string(getMangledName(&D));
259  else
260  Name = getStaticDeclName(*this, D);
261 
262  llvm::Type *LTy = getTypes().ConvertTypeForMem(Ty);
263  LangAS AS = GetGlobalVarAddressSpace(&D);
264  unsigned TargetAS = getContext().getTargetAddressSpace(AS);
265 
266  // OpenCL/SYCL variables in local address space and CUDA shared
267  // variables cannot have an initializer.
268  llvm::Constant *Init = nullptr;
269  if (AS == LangAS::opencl_local || AS == LangAS::sycl_local ||
270  D.hasAttr<CUDASharedAttr>() || D.hasAttr<LoaderUninitializedAttr>())
271  Init = llvm::UndefValue::get(LTy);
272  else
273  Init = EmitNullConstant(Ty);
274 
275  llvm::GlobalVariable *GV = new llvm::GlobalVariable(
276  getModule(), LTy, Ty.isConstant(getContext()), Linkage, Init, Name,
277  nullptr, llvm::GlobalVariable::NotThreadLocal, TargetAS);
278  GV->setAlignment(getContext().getDeclAlign(&D).getAsAlign());
279 
280  if (supportsCOMDAT() && GV->isWeakForLinker())
281  GV->setComdat(TheModule.getOrInsertComdat(GV->getName()));
282 
283  if (D.getTLSKind())
284  setTLSMode(GV, D);
285 
286  setGVProperties(GV, &D);
287 
288  // Make sure the result is of the correct type.
289  LangAS ExpectedAS = Ty.getAddressSpace();
290  llvm::Constant *Addr = GV;
291  if (AS != ExpectedAS) {
292  Addr = getTargetCodeGenInfo().performAddrSpaceCast(
293  *this, GV, AS, ExpectedAS,
294  LTy->getPointerTo(getContext().getTargetAddressSpace(ExpectedAS)));
295  }
296 
297  setStaticLocalDeclAddress(&D, Addr);
298 
299  // Do not force emission of the parent funtion since it can be a host function
300  // that contains illegal code for SYCL device.
301  if (getLangOpts().SYCLIsDevice)
302  return Addr;
303 
304  // Ensure that the static local gets initialized by making sure the parent
305  // function gets emitted eventually.
306  const Decl *DC = cast<Decl>(D.getDeclContext());
307 
308  // We can't name blocks or captured statements directly, so try to emit their
309  // parents.
310  if (isa<BlockDecl>(DC) || isa<CapturedDecl>(DC)) {
311  DC = DC->getNonClosureContext();
312  // FIXME: Ensure that global blocks get emitted.
313  if (!DC)
314  return Addr;
315  }
316 
317  GlobalDecl GD;
318  if (const auto *CD = dyn_cast<CXXConstructorDecl>(DC))
319  GD = GlobalDecl(CD, Ctor_Base);
320  else if (const auto *DD = dyn_cast<CXXDestructorDecl>(DC))
321  GD = GlobalDecl(DD, Dtor_Base);
322  else if (const auto *FD = dyn_cast<FunctionDecl>(DC))
323  GD = GlobalDecl(FD);
324  else {
325  // Don't do anything for Obj-C method decls or global closures. We should
326  // never defer them.
327  assert(isa<ObjCMethodDecl>(DC) && "unexpected parent code decl");
328  }
329  if (GD.getDecl()) {
330  // Disable emission of the parent function for the OpenMP device codegen.
331  CGOpenMPRuntime::DisableAutoDeclareTargetRAII NoDeclTarget(*this);
332  (void)GetAddrOfGlobal(GD);
333  }
334 
335  return Addr;
336 }

338 /// AddInitializerToStaticVarDecl - Add the initializer for 'D' to the
339 /// global variable that has already been created for it. If the initializer
340 /// has a different type than GV does, this may free GV and return a different
341 /// one. Otherwise it just returns GV.
342 llvm::GlobalVariable *
343 CodeGenFunction::AddInitializerToStaticVarDecl(const VarDecl &D,
344  llvm::GlobalVariable *GV) {
345  if (getLangOpts().SYCLIsDevice) {
346  auto *Scope = D.getAttr<SYCLScopeAttr>();
347  if (Scope && Scope->isWorkGroup()) {
348  // In SYCL device code globals which represent WG shared local variables
349  // 1) (TODO) must have initializer emitted even if it is constant, because
350  // LLVM->SPIRV translation does not generate optional initializer for
351  // WG shared local variables.
352  // 2) must not use guarded init because
353  // a) guarded init uses exceptions not supported in SYCL device code
354  // b) initialization is already safe because it happens in WG scope -
355  // once per WG - by specification, and this will be/ enforced in
356  // SYCL-specific Clang lowering (LowerWGScope.cpp) after CG.
357  EmitCXXGlobalVarDeclInit(D, GV, /*PerformInit*/ true);
358  // Remove const-ness, otherwise the initializer (a store into the
359  // variable) won't have effect in SPIRV due to "Constant" decoration.
360  GV->setConstant(false);
361  return GV;
362  }
363  }
364  ConstantEmitter emitter(*this);
365  llvm::Constant *Init = emitter.tryEmitForInitializer(D);
366 
367  // If constant emission failed, then this should be a C++ static
368  // initializer.
369  if (!Init) {
370  if (!getLangOpts().CPlusPlus)
371  CGM.ErrorUnsupported(D.getInit(), "constant l-value expression");
372  else if (HaveInsertPoint()) {
373  // Since we have a static initializer, this global variable can't
374  // be constant.
375  GV->setConstant(false);
376 
377  EmitCXXGuardedInit(D, GV, /*PerformInit*/true);
378  }
379  return GV;
380  }
381 
382  // The initializer may differ in type from the global. Rewrite
383  // the global to match the initializer. (We have to do this
384  // because some types, like unions, can't be completely represented
385  // in the LLVM type system.)
386  if (GV->getValueType() != Init->getType()) {
387  llvm::GlobalVariable *OldGV = GV;
388 
389  GV = new llvm::GlobalVariable(
390  CGM.getModule(), Init->getType(), OldGV->isConstant(),
391  OldGV->getLinkage(), Init, "",
392  /*InsertBefore*/ OldGV, OldGV->getThreadLocalMode(),
393  OldGV->getType()->getPointerAddressSpace());
394  GV->setVisibility(OldGV->getVisibility());
395  GV->setDSOLocal(OldGV->isDSOLocal());
396  GV->setComdat(OldGV->getComdat());
397 
398  // Steal the name of the old global
399  GV->takeName(OldGV);
400 
401  // Replace all uses of the old global with the new global
402  llvm::Constant *NewPtrForOldDecl =
403  llvm::ConstantExpr::getBitCast(GV, OldGV->getType());
404  OldGV->replaceAllUsesWith(NewPtrForOldDecl);
405 
406  // Erase the old global, since it is no longer used.
407  OldGV->eraseFromParent();
408  }
409 
410  GV->setConstant(CGM.isTypeConstant(D.getType(), true));
411  GV->setInitializer(Init);
412 
413  emitter.finalize(GV);
414 
415  if (D.needsDestruction(getContext()) == QualType::DK_cxx_destructor &&
416  HaveInsertPoint()) {
417  // We have a constant initializer, but a nontrivial destructor. We still
418  // need to perform a guarded "initialization" in order to register the
419  // destructor.
420  EmitCXXGuardedInit(D, GV, /*PerformInit*/false);
421  }
422 
423  return GV;
424 }

426 void CodeGenFunction::EmitStaticVarDecl(const VarDecl &D,
427  llvm::GlobalValue::LinkageTypes Linkage) {
428  // Check to see if we already have a global variable for this
429  // declaration. This can happen when double-emitting function
430  // bodies, e.g. with complete and base constructors.
431  llvm::Constant *addr = CGM.getOrCreateStaticVarDecl(D, Linkage);
432  CharUnits alignment = getContext().getDeclAlign(&D);
433 
434  // Store into LocalDeclMap before generating initializer to handle
435  // circular references.
436  llvm::Type *elemTy = ConvertTypeForMem(D.getType());
437  setAddrOfLocalVar(&D, Address(addr, elemTy, alignment));
438 
439  // We can't have a VLA here, but we can have a pointer to a VLA,
440  // even though that doesn't really make any sense.
441  // Make sure to evaluate VLA bounds now so that we have them for later.
442  if (D.getType()->isVariablyModifiedType())
443  EmitVariablyModifiedType(D.getType());
444 
445  // Save the type in case adding the initializer forces a type change.
446  llvm::Type *expectedType = addr->getType();
447 
448  llvm::GlobalVariable *var =
449  cast<llvm::GlobalVariable>(addr->stripPointerCasts());
450 
451  // CUDA's local and local static __shared__ variables should not
452  // have any non-empty initializers. This is ensured by Sema.
453  // Whatever initializer such variable may have when it gets here is
454  // a no-op and should not be emitted.
455  bool isCudaSharedVar = getLangOpts().CUDA && getLangOpts().CUDAIsDevice &&
456  D.hasAttr<CUDASharedAttr>();
457  // If this value has an initializer, emit it.
458  if (D.getInit() && !isCudaSharedVar)
459  var = AddInitializerToStaticVarDecl(D, var);
460 
461  var->setAlignment(alignment.getAsAlign());
462 
463  if (D.hasAttr<AnnotateAttr>())
464  CGM.AddGlobalAnnotations(&D, var);
465 
466  // Emit Intel FPGA attribute annotation for a local static variable.
467  if (getLangOpts().SYCLIsDevice)
468  CGM.addGlobalIntelFPGAAnnotation(&D, var);
469 
470  if (auto *SA = D.getAttr<PragmaClangBSSSectionAttr>())
471  var->addAttribute("bss-section", SA->getName());
472  if (auto *SA = D.getAttr<PragmaClangDataSectionAttr>())
473  var->addAttribute("data-section", SA->getName());
474  if (auto *SA = D.getAttr<PragmaClangRodataSectionAttr>())
475  var->addAttribute("rodata-section", SA->getName());
476  if (auto *SA = D.getAttr<PragmaClangRelroSectionAttr>())
477  var->addAttribute("relro-section", SA->getName());
478 
479  if (const SectionAttr *SA = D.getAttr<SectionAttr>())
480  var->setSection(SA->getName());
481 
482  if (D.hasAttr<RetainAttr>())
483  CGM.addUsedGlobal(var);
484  else if (D.hasAttr<UsedAttr>())
485  CGM.addUsedOrCompilerUsedGlobal(var);
486 
487  // We may have to cast the constant because of the initializer
488  // mismatch above.
489  //
490  // FIXME: It is really dangerous to store this in the map; if anyone
491  // RAUW's the GV uses of this constant will be invalid.
492  llvm::Constant *castedAddr =
493  llvm::ConstantExpr::getPointerBitCastOrAddrSpaceCast(var, expectedType);
494  LocalDeclMap.find(&D)->second = Address(castedAddr, elemTy, alignment);
495  CGM.setStaticLocalDeclAddress(&D, castedAddr);
496 
497  CGM.getSanitizerMetadata()->reportGlobalToASan(var, D);
498 
499  // Emit global variable debug descriptor for static vars.
500  CGDebugInfo *DI = getDebugInfo();
501  if (DI && CGM.getCodeGenOpts().hasReducedDebugInfo()) {
502  DI->setLocation(D.getLocation());
503  DI->EmitGlobalVariable(var, &D);
504  }
505 }

529  template <class Derived>
530  struct DestroyNRVOVariable : EHScopeStack::Cleanup {
531  DestroyNRVOVariable(Address addr, QualType type, llvm::Value *NRVOFlag)
532  : NRVOFlag(NRVOFlag), Loc(addr), Ty(type) {}
533 
534  llvm::Value *NRVOFlag;
535  Address Loc;
536  QualType Ty;
537 
538  void Emit(CodeGenFunction &CGF, Flags flags) override {
539  // Along the exceptions path we always execute the dtor.
540  bool NRVO = flags.isForNormalCleanup() && NRVOFlag;
541 
542  llvm::BasicBlock *SkipDtorBB = nullptr;
543  if (NRVO) {
544  // If we exited via NRVO, we skip the destructor call.
545  llvm::BasicBlock *RunDtorBB = CGF.createBasicBlock("nrvo.unused");
546  SkipDtorBB = CGF.createBasicBlock("nrvo.skipdtor");
547  llvm::Value *DidNRVO =
548  CGF.Builder.CreateFlagLoad(NRVOFlag, "nrvo.val");
549  CGF.Builder.CreateCondBr(DidNRVO, SkipDtorBB, RunDtorBB);
550  CGF.EmitBlock(RunDtorBB);
551  }
552 
553  static_cast<Derived *>(this)->emitDestructorCall(CGF);
554 
555  if (NRVO) CGF.EmitBlock(SkipDtorBB);
556  }
557 
558  virtual ~DestroyNRVOVariable() = default;
559  };

686 static bool isAccessedBy(const VarDecl &var, const Stmt *s) {
687  if (const Expr *e = dyn_cast<Expr>(s)) {
688  // Skip the most common kinds of expressions that make
689  // hierarchy-walking expensive.
690  s = e = e->IgnoreParenCasts();
691 
692  if (const DeclRefExpr *ref = dyn_cast<DeclRefExpr>(e))
693  return (ref->getDecl() == &var);
694  if (const BlockExpr *be = dyn_cast<BlockExpr>(e)) {
695  const BlockDecl *block = be->getBlockDecl();
696  for (const auto &I : block->captures()) {
697  if (I.getVariable() == &var)
698  return true;
699  }
700  }
701  }
702 
703  for (const Stmt *SubStmt : s->children())
704  // SubStmt might be null; as in missing decl or conditional of an if-stmt.
705  if (SubStmt && isAccessedBy(var, SubStmt))
706  return true;
707 
708  return false;
709 }

718 static bool tryEmitARCCopyWeakInit(CodeGenFunction &CGF,
719  const LValue &destLV, const Expr *init) {
720  bool needsCast = false;
721 
722  while (auto castExpr = dyn_cast<CastExpr>(init->IgnoreParens())) {
723  switch (castExpr->getCastKind()) {
724  // Look through casts that don't require representation changes.
725  case CK_NoOp:
726  case CK_BitCast:
727  case CK_BlockPointerToObjCPointerCast:
728  needsCast = true;
729  break;
730 
731  // If we find an l-value to r-value cast from a __weak variable,
732  // emit this operation as a copy or move.
733  case CK_LValueToRValue: {
734  const Expr *srcExpr = castExpr->getSubExpr();
735  if (srcExpr->getType().getObjCLifetime() != Qualifiers::OCL_Weak)
736  return false;
737 
738  // Emit the source l-value.
739  LValue srcLV = CGF.EmitLValue(srcExpr);
740 
741  // Handle a formal type change to avoid asserting.
742  auto srcAddr = srcLV.getAddress(CGF);
743  if (needsCast) {
744  srcAddr = CGF.Builder.CreateElementBitCast(
745  srcAddr, destLV.getAddress(CGF).getElementType());
746  }
747 
748  // If it was an l-value, use objc_copyWeak.
749  if (srcExpr->isLValue()) {
750  CGF.EmitARCCopyWeak(destLV.getAddress(CGF), srcAddr);
751  } else {
752  assert(srcExpr->isXValue());
753  CGF.EmitARCMoveWeak(destLV.getAddress(CGF), srcAddr);
754  }
755  return true;
756  }
757 
758  // Stop at anything else.
759  default:
760  return false;
761  }
762 
763  init = castExpr->getSubExpr();
764  }
765  return false;
766 }

795 void CodeGenFunction::EmitScalarInit(const Expr *init, const ValueDecl *D,
796  LValue lvalue, bool capturedByInit) {
797  Qualifiers::ObjCLifetime lifetime = lvalue.getObjCLifetime();
798  if (!lifetime) {
799  llvm::Value *value = EmitScalarExpr(init);
800  if (capturedByInit)
801  drillIntoBlockVariable(*this, lvalue, cast<VarDecl>(D));
802  EmitNullabilityCheck(lvalue, value, init->getExprLoc());
803  EmitStoreThroughLValue(RValue::get(value), lvalue, true);
804  return;
805  }
806 
807  if (const CXXDefaultInitExpr *DIE = dyn_cast<CXXDefaultInitExpr>(init))
808  init = DIE->getExpr();
809 
810  // If we're emitting a value with lifetime, we have to do the
811  // initialization *before* we leave the cleanup scopes.
812  if (auto *EWC = dyn_cast<ExprWithCleanups>(init)) {
813  CodeGenFunction::RunCleanupsScope Scope(*this);
814  return EmitScalarInit(EWC->getSubExpr(), D, lvalue, capturedByInit);
815  }
816 
817  // We have to maintain the illusion that the variable is
818  // zero-initialized. If the variable might be accessed in its
819  // initializer, zero-initialize before running the initializer, then
820  // actually perform the initialization with an assign.
821  bool accessedByInit = false;
822  if (lifetime != Qualifiers::OCL_ExplicitNone)
823  accessedByInit = (capturedByInit || isAccessedBy(D, init));
824  if (accessedByInit) {
825  LValue tempLV = lvalue;
826  // Drill down to the __block object if necessary.
827  if (capturedByInit) {
828  // We can use a simple GEP for this because it can't have been
829  // moved yet.
830  tempLV.setAddress(emitBlockByrefAddress(tempLV.getAddress(*this),
831  cast<VarDecl>(D),
832  /*follow*/ false));
833  }
834 
835  auto ty =
836  cast<llvm::PointerType>(tempLV.getAddress(*this).getElementType());
837  llvm::Value *zero = CGM.getNullPointer(ty, tempLV.getType());
838 
839  // If __weak, we want to use a barrier under certain conditions.
840  if (lifetime == Qualifiers::OCL_Weak)
841  EmitARCInitWeak(tempLV.getAddress(*this), zero);
842 
843  // Otherwise just do a simple store.
844  else
845  EmitStoreOfScalar(zero, tempLV, /* isInitialization */ true);
846  }
847 
848  // Emit the initializer.
849  llvm::Value *value = nullptr;
850 
851  switch (lifetime) {
852  case Qualifiers::OCL_None:
853  llvm_unreachable("present but none");
854 
855  case Qualifiers::OCL_Strong: {
856  if (!D || !isa<VarDecl>(D) || !cast<VarDecl>(D)->isARCPseudoStrong()) {
857  value = EmitARCRetainScalarExpr(init);
858  break;
859  }
860  // If D is pseudo-strong, treat it like __unsafe_unretained here. This means
861  // that we omit the retain, and causes non-autoreleased return values to be
862  // immediately released.
863  LLVM_FALLTHROUGH;
864  }
865 
866  case Qualifiers::OCL_ExplicitNone:
867  value = EmitARCUnsafeUnretainedScalarExpr(init);
868  break;
869 
870  case Qualifiers::OCL_Weak: {
871  // If it's not accessed by the initializer, try to emit the
872  // initialization with a copy or move.
873  if (!accessedByInit && tryEmitARCCopyWeakInit(*this, lvalue, init)) {
874  return;
875  }
876 
877  // No way to optimize a producing initializer into this. It's not
878  // worth optimizing for, because the value will immediately
879  // disappear in the common case.
880  value = EmitScalarExpr(init);
881 
882  if (capturedByInit) drillIntoBlockVariable(*this, lvalue, cast<VarDecl>(D));
883  if (accessedByInit)
884  EmitARCStoreWeak(lvalue.getAddress(*this), value, /*ignored*/ true);
885  else
886  EmitARCInitWeak(lvalue.getAddress(*this), value);
887  return;
888  }
889 
890  case Qualifiers::OCL_Autoreleasing:
891  value = EmitARCRetainAutoreleaseScalarExpr(init);
892  break;
893  }
894 
895  if (capturedByInit) drillIntoBlockVariable(*this, lvalue, cast<VarDecl>(D));
896 
897  EmitNullabilityCheck(lvalue, value, init->getExprLoc());
898 
899  // If the variable might have been accessed by its initializer, we
900  // might have to initialize with a barrier. We have to do this for
901  // both __weak and __strong, but __weak got filtered out above.
902  if (accessedByInit && lifetime == Qualifiers::OCL_Strong) {
903  llvm::Value *oldValue = EmitLoadOfScalar(lvalue, init->getExprLoc());
904  EmitStoreOfScalar(value, lvalue, /* isInitialization */ true);
905  EmitARCRelease(oldValue, ARCImpreciseLifetime);
906  return;
907  }
908 
909  EmitStoreOfScalar(value, lvalue, /* isInitialization */ true);
910 }

912 /// Decide whether we can emit the non-zero parts of the specified initializer
913 /// with equal or fewer than NumStores scalar stores.
914 static bool canEmitInitWithFewStoresAfterBZero(llvm::Constant *Init,
915  unsigned &NumStores) {
916  // Zero and Undef never requires any extra stores.
917  if (isa<llvm::ConstantAggregateZero>(Init) ||
918  isa<llvm::ConstantPointerNull>(Init) ||
919  isa<llvm::UndefValue>(Init))
920  return true;
921  if (isa<llvm::ConstantInt>(Init) || isa<llvm::ConstantFP>(Init) ||
922  isa<llvm::ConstantVector>(Init) || isa<llvm::BlockAddress>(Init) ||
923  isa<llvm::ConstantExpr>(Init))
924  return Init->isNullValue() || NumStores--;
925 
926  // See if we can emit each element.
927  if (isa<llvm::ConstantArray>(Init) || isa<llvm::ConstantStruct>(Init)) {
928  for (unsigned i = 0, e = Init->getNumOperands(); i != e; ++i) {
929  llvm::Constant *Elt = cast<llvm::Constant>(Init->getOperand(i));
930  if (!canEmitInitWithFewStoresAfterBZero(Elt, NumStores))
931  return false;
932  }
933  return true;
934  }
935 
936  if (llvm::ConstantDataSequential *CDS =
937  dyn_cast<llvm::ConstantDataSequential>(Init)) {
938  for (unsigned i = 0, e = CDS->getNumElements(); i != e; ++i) {
939  llvm::Constant *Elt = CDS->getElementAsConstant(i);
940  if (!canEmitInitWithFewStoresAfterBZero(Elt, NumStores))
941  return false;
942  }
943  return true;
944  }
945 
946  // Anything else is hard and scary.
947  return false;
948 }

1057 /// Helper function for constWithPadding() to deal with padding in structures.
1058 static llvm::Constant *constStructWithPadding(CodeGenModule &CGM,
1059  IsPattern isPattern,
1060  llvm::StructType *STy,
1061  llvm::Constant *constant) {
1062  const llvm::DataLayout &DL = CGM.getDataLayout();
1063  const llvm::StructLayout *Layout = DL.getStructLayout(STy);
1064  llvm::Type *Int8Ty = llvm::IntegerType::getInt8Ty(CGM.getLLVMContext());
1065  unsigned SizeSoFar = 0;
1066  SmallVector<llvm::Constant *, 8> Values;
1067  bool NestedIntact = true;
1068  for (unsigned i = 0, e = STy->getNumElements(); i != e; i++) {
1069  unsigned CurOff = Layout->getElementOffset(i);
1070  if (SizeSoFar < CurOff) {
1071  assert(!STy->isPacked());
1072  auto *PadTy = llvm::ArrayType::get(Int8Ty, CurOff - SizeSoFar);
1073  Values.push_back(patternOrZeroFor(CGM, isPattern, PadTy));
1074  }
1075  llvm::Constant *CurOp;
1076  if (constant->isZeroValue())
1077  CurOp = llvm::Constant::getNullValue(STy->getElementType(i));
1078  else
1079  CurOp = cast<llvm::Constant>(constant->getAggregateElement(i));
1080  auto *NewOp = constWithPadding(CGM, isPattern, CurOp);
1081  if (CurOp != NewOp)
1082  NestedIntact = false;
1083  Values.push_back(NewOp);
1084  SizeSoFar = CurOff + DL.getTypeAllocSize(CurOp->getType());
1085  }
1086  unsigned TotalSize = Layout->getSizeInBytes();
1087  if (SizeSoFar < TotalSize) {
1088  auto *PadTy = llvm::ArrayType::get(Int8Ty, TotalSize - SizeSoFar);
1089  Values.push_back(patternOrZeroFor(CGM, isPattern, PadTy));
1090  }
1091  if (NestedIntact && Values.size() == STy->getNumElements())
1092  return constant;
1093  return llvm::ConstantStruct::getAnon(Values, STy->isPacked());
1094 }

1097 /// 0x00.

1099  llvm::Constant *constant) {

1101  if (const auto STy = dyn_cast<llvm::StructType>(OrigTy))

1194  Address Loc, bool isVolatile,

1200  return;

1203  Ty->isPtrOrPtrVectorTy() || Ty->isFPOrFPVectorTy();

1205  auto *I = Builder.CreateStore(constant, Loc, isVolatile);

1448 CodeGenFunction::AutoVarEmission
1449 CodeGenFunction::EmitAutoVarAlloca(const VarDecl &D) {
1450  QualType Ty = D.getType();
1451  assert(
1452  Ty.getAddressSpace() == LangAS::Default ||
1453  (Ty.getAddressSpace() == LangAS::opencl_private && getLangOpts().OpenCL));
1454

1456 
1457  bool isEscapingByRef = D.isEscapingByref();
1458  emission.IsEscapingByRef = isEscapingByRef;
1459 
1460  CharUnits alignment = getContext().getDeclAlign(&D);
1461 
1462  // If the type is variably-modified, emit all the VLA sizes for it.
1463  if (Ty->isVariablyModifiedType())
1464  EmitVariablyModifiedType(Ty);
1465 
1466  auto *DI = getDebugInfo();
1467  bool EmitDebugInfo = DI && CGM.getCodeGenOpts().hasReducedDebugInfo();
1468 
1469  Address address = Address::invalid();
1470  Address AllocaAddr = Address::invalid();
1471  Address OpenMPLocalAddr = Address::invalid();
1472  if (CGM.getLangOpts().OpenMPIRBuilder)
1473  OpenMPLocalAddr = OMPBuilderCBHelpers::getAddressOfLocalVariable(*this, &D);
1474  else
1475  OpenMPLocalAddr =
1476  getLangOpts().OpenMP
1477  ? CGM.getOpenMPRuntime().getAddressOfLocalVariable(*this, &D)
1478  : Address::invalid();
1479 
1480  bool NRVO = getLangOpts().ElideConstructors && D.isNRVOVariable();
1481 
1482  if (getLangOpts().OpenMP && OpenMPLocalAddr.isValid()) {
1483  address = OpenMPLocalAddr;
1484  AllocaAddr = OpenMPLocalAddr;
1485  } else if (Ty->isConstantSizeType()) {
1486  // If this value is an array or struct with a statically determinable
1487  // constant initializer, there are optimizations we can do.
1488  //
1489  // TODO: We should constant-evaluate the initializer of any variable,
1490  // as long as it is initialized by a constant expression. Currently,
1491  // isConstantInitializer produces wrong answers for structs with
1492  // reference or bitfield members, and a few other cases, and checking
1493  // for POD-ness protects us from some of these.
1494  if (D.getInit() && (Ty->isArrayType() || Ty->isRecordType()) &&
1495  (D.isConstexpr() ||
1496  ((Ty.isPODType(getContext()) ||
1497  getContext().getBaseElementType(Ty)->isObjCObjectPointerType()) &&
1498  D.getInit()->isConstantInitializer(getContext(), false)))) {
1499 
1500  // If the variable's a const type, and it's neither an NRVO
1501  // candidate nor a __block variable and has no mutable members,
1502  // emit it as a global instead.
1503  // Exception is if a variable is located in non-constant address space
1504  // in OpenCL.
1505  if ((!getLangOpts().OpenCL ||
1506  Ty.getAddressSpace() == LangAS::opencl_constant) &&
1507  (CGM.getCodeGenOpts().MergeAllConstants && !NRVO &&
1508  !isEscapingByRef && CGM.isTypeConstant(Ty, true))) {
1509  EmitStaticVarDecl(D, llvm::GlobalValue::InternalLinkage);
1510

1512  emission.Addr = Address::invalid();
1513  assert(emission.wasEmittedAsGlobal());

1515  }
1516

1518  emission.IsConstantAggregate = true;
1519  }
1520 
1521  // A normal fixed sized variable becomes an alloca in the entry block,
1522  // unless:
1523  // - it's an NRVO variable.
1524  // - we are compiling OpenMP and it's an OpenMP local variable.
1525  if (NRVO) {

1527  // return slot, so that we can elide the copy when returning this
1528  // variable (C++0x [class.copy]p34).
1529  address = ReturnValue;
1530  AllocaAddr = ReturnValue;
1531

1533  const auto *RD = RecordTy->getDecl();

1535  if ((CXXRD && !CXXRD->hasTrivialDestructor()) ||
1536  RD->isNonTrivialToPrimitiveDestroy()) {

1538  // to this variable. Set it to zero to indicate that NRVO was not
1539  // applied.
1540  llvm::Value *Zero = Builder.getFalse();

1542  CreateTempAlloca(Zero->getType(), CharUnits::One(), "nrvo",
1543  /*ArraySize=*/nullptr, &AllocaAddr);

1546

1548  NRVOFlags[&D] = NRVOFlag.getPointer();

1551  }
1552  } else {

1555  if (isEscapingByRef) {

1557  allocaTy = byrefInfo.Type;
1558  allocaAlignment = byrefInfo.ByrefAlignment;

1561  allocaAlignment = alignment;

1565  // building the instruction so that it's there even in no-asserts
1566  // builds.

1569

1572  // insert code in those basic blocks.

1574  D.isExceptionVariable() && getTarget().getCXXABI().isMicrosoft();

1580  // gets broken up into several regions in IR, which requires more work
1581  // to handle correctly. For now, just omit the intrinsics; this is a
1582  // rare case, and it's better to just be conservatively correct.
1583  // PR28267.
1584  //
1585  // We have to do this in all language modes if there's a jump past the

1596  EmitLifetimeStart(Size, AllocaAddr.getPointer());
1597  }
1598  } else {
1599  assert(!emission.useLifetimeMarkers());
1600  }
1601  }
1602  } else {

1606  // Save the stack.
1607  Address Stack =
1608  CreateTempAlloca(Int8PtrTy, getPointerAlign(), "saved_stack");

1612  Builder.CreateStore(V, Stack);

1632  }

1635  emission.Addr = address;
1636  emission.AllocaAddr = AllocaAddr;

1639  if (EmitDebugInfo && HaveInsertPoint()) {

1660  CGM.getLLVMContext(), address.getAddressSpace());

1663  Arg = Builder.CreateAddrSpaceCast(Arg, CGM.Int8PtrTy, V->getName());

1666  }
1667  }

1669  if (D.hasAttr<AnnotateAttr>() && HaveInsertPoint())
1670  EmitVarAnnotations(&D, address.getPointer());
1671 
1672  // Make sure we call @llvm.lifetime.end.
1673  if (emission.useLifetimeMarkers())
1674  EHStack.pushCleanup<CallLifetimeEnd>(NormalEHLifetimeMarker,
1675  emission.getOriginalAllocatedAddress(),

1760  const VarDecl &D,
1761  Address Loc) {

1763  CharUnits Size = getContext().getTypeSizeInChars(type);
1764  bool isVolatile = type.isVolatileQualified();

1767  case LangOptions::TrivialAutoVarInitKind::Uninitialized:

1770  if (CGM.stopAutoInit())
1771  return;
1772  emitStoresForZeroInit(CGM, D, Loc, isVolatile, Builder);

1774  case LangOptions::TrivialAutoVarInitKind::Pattern:

1777  emitStoresForPatternInit(CGM, D, Loc, isVolatile, Builder);
1778  break;
1779  }
1780  return;
1781  }
1782 
1783  // VLAs look zero-sized to getTypeInfo. We can't emit constant stores to
1784  // them, so emit a memcpy with the VLA size to initialize each element.
1785  // Technically zero-sized or negative-sized VLAs are undefined, and UBSan
1786  // will catch that code, but there exists code which generates zero-sized
1787  // VLAs. Be nice and initialize whatever they requested.
1788  const auto *VlaType = getContext().getAsVariableArrayType(type);
1789  if (!VlaType)

1795  case LangOptions::TrivialAutoVarInitKind::Uninitialized:
1796  llvm_unreachable("Uninitialized handled by caller");
1797

1799  if (CGM.stopAutoInit())
1800  return;

1803  auto *I = Builder.CreateMemSet(Loc, llvm::ConstantInt::get(Int8Ty, 0),
1804  SizeVal, isVolatile);
1805  I->addAnnotationMetadata("auto-init");
1806  break;
1807  }
1808

1810  if (CGM.stopAutoInit())
1811  return;

1814  CGM, IsPattern::Yes, initializationPatternFor(CGM, ElTy));
1815  CharUnits ConstantAlign = getContext().getTypeAlignInChars(VlaSize.Type);

1817  llvm::BasicBlock *LoopBB = createBasicBlock("vla-init.loop");

1820  SizeVal, llvm::ConstantInt::get(SizeVal->getType(), 0),

1827  llvm::ConstantInt::get(IntPtrTy, EltSize.getQuantity());

1833  llvm::PHINode *Cur = Builder.CreatePHI(Begin.getType(), 2, "vla.cur");

1837  Builder.CreateMemCpy(Address(Cur, Int8Ty, CurAlign),

1840  BaseSizeInChars, isVolatile);

1844  llvm::Value *Done = Builder.CreateICmpEQ(Next, End, "vla-init.isdone");
