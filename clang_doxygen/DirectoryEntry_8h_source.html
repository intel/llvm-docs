44 /// A reference to a \c DirectoryEntry that includes the name of the directory
45 /// as it was accessed by the FileManager's client.
46 class DirectoryEntryRef {
47 public:
48  const DirectoryEntry &getDirEntry() const { return *ME->getValue(); }
49 
50  StringRef getName() const { return ME->getKey(); }
51 
52  /// Hash code is based on the DirectoryEntry, not the specific named
53  /// reference.
54  friend llvm::hash_code hash_value(DirectoryEntryRef Ref) {
55  return llvm::hash_value(&Ref.getDirEntry());
56  }
57 
58  using MapEntry = llvm::StringMapEntry<llvm::ErrorOr<DirectoryEntry &>>;
59 
60  const MapEntry &getMapEntry() const { return *ME; }
61 
62  /// Check if RHS referenced the file in exactly the same way.
63  bool isSameRef(DirectoryEntryRef RHS) const { return ME == RHS.ME; }
64 
65  DirectoryEntryRef() = delete;
66  DirectoryEntryRef(const MapEntry &ME) : ME(&ME) {}
67 
68  /// Allow DirectoryEntryRef to degrade into 'const DirectoryEntry*' to
69  /// facilitate incremental adoption.
70  ///
71  /// The goal is to avoid code churn due to dances like the following:
72  /// \code
73  /// // Old code.
74  /// lvalue = rvalue;
75  ///
76  /// // Temporary code from an incremental patch.
77  /// lvalue = &rvalue.getDirectoryEntry();
78  ///
79  /// // Final code.
80  /// lvalue = rvalue;
81  /// \endcode
82  ///
83  /// FIXME: Once DirectoryEntryRef is "everywhere" and DirectoryEntry::getName
84  /// has been deleted, delete this implicit conversion.
85  operator const DirectoryEntry *() const { return &getDirEntry(); }
86 
87 private:
88  friend class FileMgr::MapEntryOptionalStorage<DirectoryEntryRef>;
89  struct optional_none_tag {};
90 
91  // Private constructor for use by OptionalStorage.
92  DirectoryEntryRef(optional_none_tag) : ME(nullptr) {}
93  bool hasOptionalValue() const { return ME; }
94 
95  friend struct llvm::DenseMapInfo<DirectoryEntryRef>;
96  struct dense_map_empty_tag {};
97  struct dense_map_tombstone_tag {};
98 
99  // Private constructors for use by DenseMapInfo.
100  DirectoryEntryRef(dense_map_empty_tag)
101  : ME(llvm::DenseMapInfo<const MapEntry *>::getEmptyKey()) {}
102  DirectoryEntryRef(dense_map_tombstone_tag)
103  : ME(llvm::DenseMapInfo<const MapEntry *>::getTombstoneKey()) {}
104  bool isSpecialDenseMapKey() const {
105  return isSameRef(DirectoryEntryRef(dense_map_empty_tag())) ||
106  isSameRef(DirectoryEntryRef(dense_map_tombstone_tag()));
107  }
108 
109  const MapEntry *ME;
110 };

114 /// Customized storage for refs derived from map entires in FileManager, using
115 /// the private optional_none_tag to keep it to the size of a single pointer.
116 template <class RefTy> class MapEntryOptionalStorage {
117  using optional_none_tag = typename RefTy::optional_none_tag;
118  RefTy MaybeRef;
119 
120 public:
121  MapEntryOptionalStorage() : MaybeRef(optional_none_tag()) {}
122 
123  template <class... ArgTypes>
124  explicit MapEntryOptionalStorage(llvm::in_place_t, ArgTypes &&...Args)
125  : MaybeRef(std::forward<ArgTypes>(Args)...) {}
126 
127  void reset() { MaybeRef = optional_none_tag(); }
128 
129  bool hasValue() const { return MaybeRef.hasOptionalValue(); }
130 
131  RefTy &getValue() & {
132  assert(hasValue());
133  return MaybeRef;
134  }
135  RefTy const &getValue() const & {
136  assert(hasValue());
137  return MaybeRef;
138  }
139  RefTy &&getValue() && {
140  assert(hasValue());
141  return std::move(MaybeRef);
142  }
143 
144  template <class... Args> void emplace(Args &&...args) {
145  MaybeRef = RefTy(std::forward<Args>(args)...);
146  }
147 
148  MapEntryOptionalStorage &operator=(RefTy Ref) {
149  MaybeRef = Ref;
150  return *this;
151  }
152 };

207  static bool isEqual(clang::DirectoryEntryRef LHS,
208  clang::DirectoryEntryRef RHS) {
209  // Catch the easy cases: both empty, both tombstone, or the same ref.
210  if (LHS.isSameRef(RHS))
211  return true;
212 
213  // Confirm LHS and RHS are valid.
214  if (LHS.isSpecialDenseMapKey() || RHS.isSpecialDenseMapKey())
215  return false;
216 
217  // It's safe to use operator==.
218  return LHS == RHS;
219  }

226 /// Wrapper around Optional<DirectoryEntryRef> that degrades to 'const
227 /// DirectoryEntry*', facilitating incremental patches to propagate
228 /// DirectoryEntryRef.
229 ///
230 /// This class can be used as return value or field where it's convenient for
231 /// an Optional<DirectoryEntryRef> to degrade to a 'const DirectoryEntry*'. The
232 /// purpose is to avoid code churn due to dances like the following:
233 /// \code
234 /// // Old code.
235 /// lvalue = rvalue;
236 ///
237 /// // Temporary code from an incremental patch.
238 /// Optional<DirectoryEntryRef> MaybeF = rvalue;
239 /// lvalue = MaybeF ? &MaybeF.getDirectoryEntry() : nullptr;
240 ///
241 /// // Final code.
242 /// lvalue = rvalue;
243 /// \endcode
244 ///
245 /// FIXME: Once DirectoryEntryRef is "everywhere" and DirectoryEntry::LastRef
246 /// and DirectoryEntry::getName have been deleted, delete this class and
247 /// replace instances with Optional<DirectoryEntryRef>.

249  : public Optional<DirectoryEntryRef> {
250 public:
251  OptionalDirectoryEntryRefDegradesToDirectoryEntryPtr() = default;
252  OptionalDirectoryEntryRefDegradesToDirectoryEntryPtr(
253  OptionalDirectoryEntryRefDegradesToDirectoryEntryPtr &&) = default;

255  const OptionalDirectoryEntryRefDegradesToDirectoryEntryPtr &) = default;

257  operator=(OptionalDirectoryEntryRefDegradesToDirectoryEntryPtr &&) = default;
258  OptionalDirectoryEntryRefDegradesToDirectoryEntryPtr &
259  operator=(const OptionalDirectoryEntryRefDegradesToDirectoryEntryPtr &) = default;
260

262  OptionalDirectoryEntryRefDegradesToDirectoryEntryPtr(DirectoryEntryRef Ref)

268  Optional<DirectoryEntryRef>::operator=(None);
