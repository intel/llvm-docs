36 /// DocComment is a vector of RawComment::CommentLine.
37 ///
38 /// Each line represents one line of striped documentation comment,
39 /// with source range information. This simplifies calculating the source
40 /// location of a character in the doc comment for pointing back to the source
41 /// file.
42 /// e.g.
43 /// \code
44 /// /// This is a documentation comment
45 /// ^~~~~~~~~~~~~~~~~~~~~~~~~~~~~~' First line.
46 /// /// with multiple lines.
47 /// ^~~~~~~~~~~~~~~~~~~~~~~' Second line.
48 /// \endcode
49 using DocComment = std::vector<RawComment::CommentLine>;

51 /// The base representation of an API record. Holds common symbol information.
52 struct APIRecord {
53  StringRef Name;
54  StringRef USR;
55  PresumedLoc Location;
56  AvailabilityInfo Availability;
57  LinkageInfo Linkage;
58 
59  /// Documentation comment lines attached to this symbol declaration.
60  DocComment Comment;
61 
62  /// Declaration fragments of this symbol declaration.
63  DeclarationFragments Declaration;
64 
65  /// SubHeading provides a more detailed representation than the plain
66  /// declaration name.
67  ///
68  /// SubHeading is an array of declaration fragments of tagged declaration
69  /// name, with potentially more tokens (for example the \c +/- symbol for
70  /// Objective-C class/instance methods).
71  DeclarationFragments SubHeading;
72 
73  /// Discriminator for LLVM-style RTTI (dyn_cast<> et al.)
74  enum RecordKind {
75  RK_Global,
76  RK_EnumConstant,
77  RK_Enum,
78  RK_StructField,
79  RK_Struct,
80  };
81 
82 private:
83  const RecordKind Kind;
84 
85 public:
86  RecordKind getKind() const { return Kind; }
87 
88  APIRecord() = delete;
89 
90  APIRecord(RecordKind Kind, StringRef Name, StringRef USR,
91  PresumedLoc Location, const AvailabilityInfo &Availability,
92  LinkageInfo Linkage, const DocComment &Comment,
93  DeclarationFragments Declaration, DeclarationFragments SubHeading)
94  : Name(Name), USR(USR), Location(Location), Availability(Availability),
95  Linkage(Linkage), Comment(Comment), Declaration(Declaration),
96  SubHeading(SubHeading), Kind(Kind) {}
97 
98  // Pure virtual destructor to make APIRecord abstract
99  virtual ~APIRecord() = 0;
100 };

109 /// This holds information associated with global variables or functions.
110 struct GlobalRecord : APIRecord {
111  GVKind GlobalKind;
112 
113  /// The function signature of the record if it is a function.
114  FunctionSignature Signature;
115 
116  GlobalRecord(GVKind Kind, StringRef Name, StringRef USR, PresumedLoc Loc,
117  const AvailabilityInfo &Availability, LinkageInfo Linkage,
118  const DocComment &Comment, DeclarationFragments Declaration,
119  DeclarationFragments SubHeading, FunctionSignature Signature)
120  : APIRecord(RK_Global, Name, USR, Loc, Availability, Linkage, Comment,
121  Declaration, SubHeading),
122  GlobalKind(Kind), Signature(Signature) {}
123 
124  static bool classof(const APIRecord *Record) {
125  return Record->getKind() == RK_Global;
126  }
127 
128 private:
129  virtual void anchor();
130 };
