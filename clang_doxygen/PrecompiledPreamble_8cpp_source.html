81 // Collects files whose existence would invalidate the preamble.
82 // Collecting *all* of these would make validating it too slow though, so we
83 // just find all the candidates for 'file not found' diagnostics.
84 //
85 // A caveat that may be significant for generated files: we'll omit files under
86 // search path entries whose roots don't exist when the preamble is built.
87 // These are pruned by InitHeaderSearch and so we don't see the search path.

100  void InclusionDirective(SourceLocation HashLoc, const Token &IncludeTok,
101  StringRef FileName, bool IsAngled,
102  CharSourceRange FilenameRange, const FileEntry *File,
103  StringRef SearchPath, StringRef RelativePath,
104  const Module *Imported,
105  SrcMgr::CharacteristicKind FileType) override {
106  // File is null if it wasn't found.
107  // (We have some false negatives if PP recovered e.g. <foo> -> "foo")
108  if (File != nullptr)
109  return;
110 
111  // If it's a rare absolute include, we know the full path already.
112  if (llvm::sys::path::is_absolute(FileName)) {
113  Out.insert(FileName);
114  return;
115  }
116 
117  // Reconstruct the filenames that would satisfy this directive...
118  llvm::SmallString<256> Buf;
119  auto NotFoundRelativeTo = [&](const DirectoryEntry *DE) {
120  Buf = DE->getName();
121  llvm::sys::path::append(Buf, FileName);
122  llvm::sys::path::remove_dots(Buf, /*remove_dot_dot=*/true);
123  Out.insert(Buf);
124  };
125  // ...relative to the including file.
126  if (!IsAngled) {
127  if (const FileEntry *IncludingFile =
128  SM.getFileEntryForID(SM.getFileID(IncludeTok.getLocation())))
129  if (IncludingFile->getDir())
130  NotFoundRelativeTo(IncludingFile->getDir());
131  }
132  // ...relative to the search paths.
133  for (const auto &Dir : llvm::make_range(
134  IsAngled ? Search.angled_dir_begin() : Search.search_dir_begin(),
135  Search.search_dir_end())) {
136  // No support for frameworks or header maps yet.
137  if (Dir.isNormalDir())
138  NotFoundRelativeTo(Dir.getDir());
139  }
140  }

311 llvm::ErrorOr<PrecompiledPreamble> PrecompiledPreamble::Build(
312  const CompilerInvocation &Invocation,
313  const llvm::MemoryBuffer *MainFileBuffer, PreambleBounds Bounds,
314  DiagnosticsEngine &Diagnostics,
315  IntrusiveRefCntPtr<llvm::vfs::FileSystem> VFS,
316  std::shared_ptr<PCHContainerOperations> PCHContainerOps, bool StoreInMemory,
317  PreambleCallbacks &Callbacks) {
318  assert(VFS && "VFS is null");
319 
320  auto PreambleInvocation = std::make_shared<CompilerInvocation>(Invocation);
321  FrontendOptions &FrontendOpts = PreambleInvocation->getFrontendOpts();
322  PreprocessorOptions &PreprocessorOpts =
323  PreambleInvocation->getPreprocessorOpts();
324 
325  llvm::Optional<TempPCHFile> TempFile;
326  if (!StoreInMemory) {
327  // Create a temporary file for the precompiled preamble. In rare
328  // circumstances, this can fail.
329  llvm::ErrorOr<PrecompiledPreamble::TempPCHFile> PreamblePCHFile =
330  PrecompiledPreamble::TempPCHFile::CreateNewPreamblePCHFile();
331  if (!PreamblePCHFile)
332  return BuildPreambleError::CouldntCreateTempFile;
333  TempFile = std::move(*PreamblePCHFile);
334  }
335 
336  PCHStorage Storage = StoreInMemory ? PCHStorage(InMemoryPreamble())
337  : PCHStorage(std::move(*TempFile));
338 
339  // Save the preamble text for later; we'll need to compare against it for
340  // subsequent reparses.
341  std::vector<char> PreambleBytes(MainFileBuffer->getBufferStart(),
342  MainFileBuffer->getBufferStart() +
343  Bounds.Size);
344  bool PreambleEndsAtStartOfLine = Bounds.PreambleEndsAtStartOfLine;
345 
346  // Tell the compiler invocation to generate a temporary precompiled header.
347  FrontendOpts.ProgramAction = frontend::GeneratePCH;
348  FrontendOpts.OutputFile =
349  std::string(StoreInMemory ? getInMemoryPreamblePath()
350  : Storage.asFile().getFilePath());
351  PreprocessorOpts.PrecompiledPreambleBytes.first = 0;
352  PreprocessorOpts.PrecompiledPreambleBytes.second = false;
353  // Inform preprocessor to record conditional stack when building the preamble.
354  PreprocessorOpts.GeneratePreamble = true;
355 
356  // Create the compiler instance to use for building the precompiled preamble.
357  std::unique_ptr<CompilerInstance> Clang(
358  new CompilerInstance(std::move(PCHContainerOps)));
359 
360  // Recover resources if we crash before exiting this method.
361  llvm::CrashRecoveryContextCleanupRegistrar<CompilerInstance> CICleanup(
362  Clang.get());
363 
364  Clang->setInvocation(std::move(PreambleInvocation));
365  Clang->setDiagnostics(&Diagnostics);
366 
367  // Create the target instance.
368  if (!Clang->createTarget())
369  return BuildPreambleError::CouldntCreateTargetInfo;
370 
371  if (Clang->getFrontendOpts().Inputs.size() != 1 ||
372  Clang->getFrontendOpts().Inputs[0].getKind().getFormat() !=
373  InputKind::Source ||
374  Clang->getFrontendOpts().Inputs[0].getKind().getLanguage() ==
375  Language::LLVM_IR) {
376  return BuildPreambleError::BadInputs;
377  }
378 
379  // Clear out old caches and data.
380  Diagnostics.Reset();
381  ProcessWarningOptions(Diagnostics, Clang->getDiagnosticOpts());
382 
383  VFS =
384  createVFSFromCompilerInvocation(Clang->getInvocation(), Diagnostics, VFS);
385 
386  // Create a file manager object to provide access to and cache the filesystem.
387  Clang->setFileManager(new FileManager(Clang->getFileSystemOpts(), VFS));
388 
389  // Create the source manager.
390  Clang->setSourceManager(
391  new SourceManager(Diagnostics, Clang->getFileManager()));
392 
393  auto PreambleDepCollector = std::make_shared<PreambleDependencyCollector>();
394  Clang->addDependencyCollector(PreambleDepCollector);
395 
396  Clang->getLangOpts().CompilingPCH = true;
397 
398  // Remap the main source file to the preamble buffer.
399  StringRef MainFilePath = FrontendOpts.Inputs[0].getFile();
400  auto PreambleInputBuffer = llvm::MemoryBuffer::getMemBufferCopy(
401  MainFileBuffer->getBuffer().slice(0, Bounds.Size), MainFilePath);
402  if (PreprocessorOpts.RetainRemappedFileBuffers) {
403  // MainFileBuffer will be deleted by unique_ptr after leaving the method.
404  PreprocessorOpts.addRemappedFile(MainFilePath, PreambleInputBuffer.get());
405  } else {
406  // In that case, remapped buffer will be deleted by CompilerInstance on
407  // BeginSourceFile, so we call release() to avoid double deletion.
408  PreprocessorOpts.addRemappedFile(MainFilePath,
409  PreambleInputBuffer.release());
410  }
411 
412  std::unique_ptr<PrecompilePreambleAction> Act;
413  Act.reset(new PrecompilePreambleAction(
414  StoreInMemory ? &Storage.asMemory().Data : nullptr, Callbacks));
415  if (!Act->BeginSourceFile(*Clang.get(), Clang->getFrontendOpts().Inputs[0]))
416  return BuildPreambleError::BeginSourceFileFailed;
417 
418  // Performed after BeginSourceFile to ensure Clang->Preprocessor can be
419  // referenced in the callback.
420  Callbacks.BeforeExecute(*Clang);
421 
422  std::unique_ptr<PPCallbacks> DelegatedPPCallbacks =
423  Callbacks.createPPCallbacks();
424  if (DelegatedPPCallbacks)
425  Clang->getPreprocessor().addPPCallbacks(std::move(DelegatedPPCallbacks));
426  if (auto CommentHandler = Callbacks.getCommentHandler())
427  Clang->getPreprocessor().addCommentHandler(CommentHandler);
428  llvm::StringSet<> MissingFiles;
429  Clang->getPreprocessor().addPPCallbacks(
430  std::make_unique<MissingFileCollector>(
431  MissingFiles, Clang->getPreprocessor().getHeaderSearchInfo(),
432  Clang->getSourceManager()));
433 
434  if (llvm::Error Err = Act->Execute())
435  return errorToErrorCode(std::move(Err));
436 
437  // Run the callbacks.
438  Callbacks.AfterExecute(*Clang);
439 
440  Act->EndSourceFile();
441 
442  if (!Act->hasEmittedPreamblePCH())
443  return BuildPreambleError::CouldntEmitPCH;
444 
445  // Keep track of all of the files that the source manager knows about,
446  // so we can verify whether they have changed or not.
447  llvm::StringMap<PrecompiledPreamble::PreambleFileHash> FilesInPreamble;
448 
449  SourceManager &SourceMgr = Clang->getSourceManager();
450  for (auto &Filename : PreambleDepCollector->getDependencies()) {
451  auto FileOrErr = Clang->getFileManager().getFile(Filename);
452  if (!FileOrErr ||
453  *FileOrErr == SourceMgr.getFileEntryForID(SourceMgr.getMainFileID()))
454  continue;
455  auto File = *FileOrErr;
456  if (time_t ModTime = File->getModificationTime()) {
457  FilesInPreamble[File->getName()] =
458  PrecompiledPreamble::PreambleFileHash::createForFile(File->getSize(),
459  ModTime);
460  } else {
461  llvm::MemoryBufferRef Buffer =
462  SourceMgr.getMemoryBufferForFileOrFake(File);
463  FilesInPreamble[File->getName()] =
464  PrecompiledPreamble::PreambleFileHash::createForMemoryBuffer(Buffer);
465  }
466  }
467 
468  return PrecompiledPreamble(
469  std::move(Storage), std::move(PreambleBytes), PreambleEndsAtStartOfLine,
470  std::move(FilesInPreamble), std::move(MissingFiles));
471 }

477 std::size_t PrecompiledPreamble::getSize() const {
478  switch (Storage.getKind()) {
479  case PCHStorage::Kind::Empty:
480  assert(false && "Calling getSize() on invalid PrecompiledPreamble. "
481  "Was it std::moved?");
482  return 0;
483  case PCHStorage::Kind::InMemory:
484  return Storage.asMemory().Data.size();
485  case PCHStorage::Kind::TempFile: {
486  uint64_t Result;
487  if (llvm::sys::fs::file_size(Storage.asFile().getFilePath(), Result))
488  return 0;
489 
490  assert(Result <= std::numeric_limits<std::size_t>::max() &&
491  "file size did not fit into size_t");
492  return Result;
493  }
494  }
495  llvm_unreachable("Unhandled storage kind");
496 }

498 bool PrecompiledPreamble::CanReuse(const CompilerInvocation &Invocation,
499  const llvm::MemoryBufferRef &MainFileBuffer,
500  PreambleBounds Bounds,
501  llvm::vfs::FileSystem &VFS) const {
502 
503  assert(
504  Bounds.Size <= MainFileBuffer.getBufferSize() &&
505  "Buffer is too large. Bounds were calculated from a different buffer?");
506 
507  auto PreambleInvocation = std::make_shared<CompilerInvocation>(Invocation);
508  PreprocessorOptions &PreprocessorOpts =
509  PreambleInvocation->getPreprocessorOpts();
510 
511  // We've previously computed a preamble. Check whether we have the same
512  // preamble now that we did before, and that there's enough space in
513  // the main-file buffer within the precompiled preamble to fit the
514  // new main file.
515  if (PreambleBytes.size() != Bounds.Size ||
516  PreambleEndsAtStartOfLine != Bounds.PreambleEndsAtStartOfLine ||
517  !std::equal(PreambleBytes.begin(), PreambleBytes.end(),
518  MainFileBuffer.getBuffer().begin()))
519  return false;
520  // The preamble has not changed. We may be able to re-use the precompiled
521  // preamble.
522 
523  // Check that none of the files used by the preamble have changed.
524  // First, make a record of those files that have been overridden via
525  // remapping or unsaved_files.
526  std::map<llvm::sys::fs::UniqueID, PreambleFileHash> OverriddenFiles;
527  llvm::StringSet<> OverriddenAbsPaths; // Either by buffers or files.
528  for (const auto &R : PreprocessorOpts.RemappedFiles) {
529  llvm::vfs::Status Status;
530  if (!moveOnNoError(VFS.status(R.second), Status)) {
531  // If we can't stat the file we're remapping to, assume that something
532  // horrible happened.
533  return false;
534  }
535  // If a mapped file was previously missing, then it has changed.
536  llvm::SmallString<128> MappedPath(R.first);
537  if (!VFS.makeAbsolute(MappedPath))
538  OverriddenAbsPaths.insert(MappedPath);
539 
540  OverriddenFiles[Status.getUniqueID()] = PreambleFileHash::createForFile(
541  Status.getSize(), llvm::sys::toTimeT(Status.getLastModificationTime()));
542  }
543 
544  // OverridenFileBuffers tracks only the files not found in VFS.
545  llvm::StringMap<PreambleFileHash> OverridenFileBuffers;
546  for (const auto &RB : PreprocessorOpts.RemappedFileBuffers) {
547  const PrecompiledPreamble::PreambleFileHash PreambleHash =
548  PreambleFileHash::createForMemoryBuffer(RB.second->getMemBufferRef());
549  llvm::vfs::Status Status;
550  if (moveOnNoError(VFS.status(RB.first), Status))
551  OverriddenFiles[Status.getUniqueID()] = PreambleHash;
552  else
553  OverridenFileBuffers[RB.first] = PreambleHash;
554 
555  llvm::SmallString<128> MappedPath(RB.first);
556  if (!VFS.makeAbsolute(MappedPath))
557  OverriddenAbsPaths.insert(MappedPath);
558  }
559 
560  // Check whether anything has changed.
561  for (const auto &F : FilesInPreamble) {
562  auto OverridenFileBuffer = OverridenFileBuffers.find(F.first());
563  if (OverridenFileBuffer != OverridenFileBuffers.end()) {
564  // The file's buffer was remapped and the file was not found in VFS.
565  // Check whether it matches up with the previous mapping.
566  if (OverridenFileBuffer->second != F.second)
567  return false;
568  continue;
569  }
570 
571  llvm::vfs::Status Status;
572  if (!moveOnNoError(VFS.status(F.first()), Status)) {
573  // If the file's buffer is not remapped and we can't stat it,
574  // assume that something horrible happened.
575  return false;
576  }
577 
578  std::map<llvm::sys::fs::UniqueID, PreambleFileHash>::iterator Overridden =
579  OverriddenFiles.find(Status.getUniqueID());
580  if (Overridden != OverriddenFiles.end()) {
581  // This file was remapped; check whether the newly-mapped file
582  // matches up with the previous mapping.
583  if (Overridden->second != F.second)
584  return false;
585  continue;
586  }
587 
588  // Neither the file's buffer nor the file itself was remapped;
589  // check whether it has changed on disk.
590  if (Status.getSize() != uint64_t(F.second.Size) ||
591  llvm::sys::toTimeT(Status.getLastModificationTime()) !=
592  F.second.ModTime)
593  return false;
594  }
595  for (const auto &F : MissingFiles) {
596  // A missing file may be "provided" by an override buffer or file.
597  if (OverriddenAbsPaths.count(F.getKey()))
598  return false;
599  // If a file previously recorded as missing exists as a regular file, then
600  // consider the preamble out-of-date.
601  if (auto Status = VFS.status(F.getKey())) {
602  if (Status->isRegularFile())
603  return false;
604  }

794 void PrecompiledPreamble::configurePreamble(
795  PreambleBounds Bounds, CompilerInvocation &CI,
796  IntrusiveRefCntPtr<llvm::vfs::FileSystem> &VFS,
797  llvm::MemoryBuffer *MainFileBuffer) const {
798  assert(VFS);
799 
800  auto &PreprocessorOpts = CI.getPreprocessorOpts();
801 
802  // Remap main file to point to MainFileBuffer.
803  auto MainFilePath = CI.getFrontendOpts().Inputs[0].getFile();
804  PreprocessorOpts.addRemappedFile(MainFilePath, MainFileBuffer);
805 
806  // Configure ImpicitPCHInclude.
807  PreprocessorOpts.PrecompiledPreambleBytes.first = Bounds.Size;
808  PreprocessorOpts.PrecompiledPreambleBytes.second =
809  Bounds.PreambleEndsAtStartOfLine;
810  PreprocessorOpts.DisablePCHOrModuleValidation =
811  DisableValidationForModuleKind::PCH;
812 
813  setupPreambleStorage(Storage, PreprocessorOpts, VFS);
814 }

816 void PrecompiledPreamble::setupPreambleStorage(
817  const PCHStorage &Storage, PreprocessorOptions &PreprocessorOpts,
818  IntrusiveRefCntPtr<llvm::vfs::FileSystem> &VFS) {
819  if (Storage.getKind() == PCHStorage::Kind::TempFile) {
820  const TempPCHFile &PCHFile = Storage.asFile();
821  PreprocessorOpts.ImplicitPCHInclude = std::string(PCHFile.getFilePath());
822 
823  // Make sure we can access the PCH file even if we're using a VFS
824  IntrusiveRefCntPtr<llvm::vfs::FileSystem> RealFS =
825  llvm::vfs::getRealFileSystem();
826  auto PCHPath = PCHFile.getFilePath();
827  if (VFS == RealFS || VFS->exists(PCHPath))
828  return;
829  auto Buf = RealFS->getBufferForFile(PCHPath);
830  if (!Buf) {
831  // We can't read the file even from RealFS, this is clearly an error,
832  // but we'll just leave the current VFS as is and let clang's code
833  // figure out what to do with missing PCH.
834  return;
835  }
836 
837  // We have a slight inconsistency here -- we're using the VFS to
838  // read files, but the PCH was generated in the real file system.
839  VFS = createVFSOverlayForPreamblePCH(PCHPath, std::move(*Buf), VFS);
840  } else {
841  assert(Storage.getKind() == PCHStorage::Kind::InMemory);
842  // For in-memory preamble, we have to provide a VFS overlay that makes it
843  // accessible.
844  StringRef PCHPath = getInMemoryPreamblePath();
845  PreprocessorOpts.ImplicitPCHInclude = std::string(PCHPath);
846 
847  auto Buf = llvm::MemoryBuffer::getMemBuffer(Storage.asMemory().Data);
848  VFS = createVFSOverlayForPreamblePCH(PCHPath, std::move(Buf), VFS);
849  }
850 }
