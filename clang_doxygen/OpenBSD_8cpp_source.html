27 void openbsd::Assembler::ConstructJob(Compilation &C, const JobAction &JA,
28  const InputInfo &Output,
29  const InputInfoList &Inputs,
30  const ArgList &Args,
31  const char *LinkingOutput) const {
32  const toolchains::OpenBSD &ToolChain =
33  static_cast<const toolchains::OpenBSD &>(getToolChain());
34  const Driver &D = ToolChain.getDriver();
35  const llvm::Triple &Triple = ToolChain.getTriple();
36 
37  claimNoWarnArgs(Args);
38  ArgStringList CmdArgs;
39 
40  switch (ToolChain.getArch()) {
41  case llvm::Triple::x86:
42  // When building 32-bit code on OpenBSD/amd64, we have to explicitly
43  // instruct as in the base system to assemble 32-bit code.
44  CmdArgs.push_back("--32");
45  break;
46 
47  case llvm::Triple::arm:
48  case llvm::Triple::armeb: {
49  StringRef MArch, MCPU;
50  arm::getARMArchCPUFromArgs(Args, MArch, MCPU, /*FromAs*/ true);
51  std::string Arch = arm::getARMTargetCPU(MCPU, MArch, Triple);
52  CmdArgs.push_back(Args.MakeArgString("-mcpu=" + Arch));
53  break;
54  }
55 
56  case llvm::Triple::ppc:
57  CmdArgs.push_back("-mppc");
58  CmdArgs.push_back("-many");
59  break;
60 
61  case llvm::Triple::sparcv9: {
62  CmdArgs.push_back("-64");
63  std::string CPU = getCPUName(D, Args, Triple);
64  CmdArgs.push_back(sparc::getSparcAsmModeForCPU(CPU, Triple));
65  AddAssemblerKPIC(ToolChain, Args, CmdArgs);
66  break;
67  }
68 
69  case llvm::Triple::mips64:
70  case llvm::Triple::mips64el: {
71  StringRef CPUName;
72  StringRef ABIName;
73  mips::getMipsCPUAndABI(Args, Triple, CPUName, ABIName);
74 
75  CmdArgs.push_back("-march");
76  CmdArgs.push_back(CPUName.data());
77 
78  CmdArgs.push_back("-mabi");
79  CmdArgs.push_back(mips::getGnuCompatibleMipsABIName(ABIName).data());
80 
81  if (Triple.isLittleEndian())
82  CmdArgs.push_back("-EL");
83  else
84  CmdArgs.push_back("-EB");
85 
86  AddAssemblerKPIC(ToolChain, Args, CmdArgs);
87  break;
88  }
89 
90  default:
91  break;
92  }
93 
94  Args.AddAllArgValues(CmdArgs, options::OPT_Wa_COMMA, options::OPT_Xassembler);
95 
96  CmdArgs.push_back("-o");
97  CmdArgs.push_back(Output.getFilename());
98 
99  for (const auto &II : Inputs)
100  CmdArgs.push_back(II.getFilename());
101 
102  const char *Exec = Args.MakeArgString(ToolChain.GetProgramPath("as"));
103  C.addCommand(std::make_unique<Command>(JA, *this,
104  ResponseFileSupport::AtFileCurCP(),
105  Exec, CmdArgs, Inputs, Output));
106 }

108 void openbsd::Linker::ConstructJob(Compilation &C, const JobAction &JA,
109  const InputInfo &Output,
110  const InputInfoList &Inputs,
111  const ArgList &Args,
112  const char *LinkingOutput) const {
113  const toolchains::OpenBSD &ToolChain =
114  static_cast<const toolchains::OpenBSD &>(getToolChain());
115  const Driver &D = ToolChain.getDriver();
116  ArgStringList CmdArgs;
117 
118  // Silence warning for "clang -g foo.o -o foo"
119  Args.ClaimAllArgs(options::OPT_g_Group);
120  // and "clang -emit-llvm foo.o -o foo"
121  Args.ClaimAllArgs(options::OPT_emit_llvm);
122  // and for "clang -w foo.o -o foo". Other warning options are already
123  // handled somewhere else.
124  Args.ClaimAllArgs(options::OPT_w);
125 
126  if (!D.SysRoot.empty())
127  CmdArgs.push_back(Args.MakeArgString("--sysroot=" + D.SysRoot));
128 
129  if (ToolChain.getArch() == llvm::Triple::mips64)
130  CmdArgs.push_back("-EB");
131  else if (ToolChain.getArch() == llvm::Triple::mips64el)
132  CmdArgs.push_back("-EL");
133 
134  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_shared)) {
135  CmdArgs.push_back("-e");
136  CmdArgs.push_back("__start");
137  }
138 
139  CmdArgs.push_back("--eh-frame-hdr");
140  if (Args.hasArg(options::OPT_static)) {
141  CmdArgs.push_back("-Bstatic");
142  } else {
143  if (Args.hasArg(options::OPT_rdynamic))
144  CmdArgs.push_back("-export-dynamic");
145  CmdArgs.push_back("-Bdynamic");
146  if (Args.hasArg(options::OPT_shared)) {
147  CmdArgs.push_back("-shared");
148  } else {
149  CmdArgs.push_back("-dynamic-linker");
150  CmdArgs.push_back("/usr/libexec/ld.so");
151  }
152  }
153 
154  if (Args.hasArg(options::OPT_pie))
155  CmdArgs.push_back("-pie");
156  if (Args.hasArg(options::OPT_nopie) || Args.hasArg(options::OPT_pg))
157  CmdArgs.push_back("-nopie");
158 
159  if (Output.isFilename()) {
160  CmdArgs.push_back("-o");
161  CmdArgs.push_back(Output.getFilename());
162  } else {
163  assert(Output.isNothing() && "Invalid output.");
164  }
165 
166  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nostartfiles,
167  options::OPT_r)) {
168  const char *crt0 = nullptr;
169  const char *crtbegin = nullptr;
170  if (!Args.hasArg(options::OPT_shared)) {
171  if (Args.hasArg(options::OPT_pg))
172  crt0 = "gcrt0.o";
173  else if (Args.hasArg(options::OPT_static) &&
174  !Args.hasArg(options::OPT_nopie))
175  crt0 = "rcrt0.o";
176  else
177  crt0 = "crt0.o";
178  crtbegin = "crtbegin.o";
179  } else {
180  crtbegin = "crtbeginS.o";
181  }
182 
183  if (crt0)
184  CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath(crt0)));
185  CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath(crtbegin)));
186  }
187 
188  Args.AddAllArgs(CmdArgs, options::OPT_L);
189  ToolChain.AddFilePathLibArgs(Args, CmdArgs);
190  Args.AddAllArgs(CmdArgs, {options::OPT_T_Group, options::OPT_e,
191  options::OPT_s, options::OPT_t,
192  options::OPT_Z_Flag, options::OPT_r});
193 
194  bool NeedsSanitizerDeps = addSanitizerRuntimes(ToolChain, Args, CmdArgs);
195  bool NeedsXRayDeps = addXRayRuntime(ToolChain, Args, CmdArgs);
196  AddLinkerInputs(ToolChain, Inputs, Args, CmdArgs, JA);
197 
198  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nodefaultlibs,
199  options::OPT_r)) {
200  // Use the static OpenMP runtime with -static-openmp
201  bool StaticOpenMP = Args.hasArg(options::OPT_static_openmp) &&
202  !Args.hasArg(options::OPT_static);
203  addOpenMPRuntime(CmdArgs, ToolChain, Args, StaticOpenMP);
204 
205  if (D.CCCIsCXX()) {
206  if (ToolChain.ShouldLinkCXXStdlib(Args))
207  ToolChain.AddCXXStdlibLibArgs(Args, CmdArgs);
208  if (Args.hasArg(options::OPT_pg))
209  CmdArgs.push_back("-lm_p");
210  else
211  CmdArgs.push_back("-lm");
212  }
213  if (NeedsSanitizerDeps) {
214  CmdArgs.push_back(ToolChain.getCompilerRTArgString(Args, "builtins"));
215  linkSanitizerRuntimeDeps(ToolChain, CmdArgs);
216  }
217  if (NeedsXRayDeps) {
218  CmdArgs.push_back(ToolChain.getCompilerRTArgString(Args, "builtins"));
219  linkXRayRuntimeDeps(ToolChain, CmdArgs);
220  }
221  // FIXME: For some reason GCC passes -lgcc before adding
222  // the default system libraries. Just mimic this for now.
223  CmdArgs.push_back("-lcompiler_rt");
224 
225  if (Args.hasArg(options::OPT_pthread)) {
226  if (!Args.hasArg(options::OPT_shared) && Args.hasArg(options::OPT_pg))
227  CmdArgs.push_back("-lpthread_p");
228  else
229  CmdArgs.push_back("-lpthread");
230  }
231 
232  if (!Args.hasArg(options::OPT_shared)) {
233  if (Args.hasArg(options::OPT_pg))
234  CmdArgs.push_back("-lc_p");
235  else
236  CmdArgs.push_back("-lc");
237  }
238 
239  CmdArgs.push_back("-lcompiler_rt");
240  }
241 
242  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nostartfiles,
243  options::OPT_r)) {
244  const char *crtend = nullptr;
245  if (!Args.hasArg(options::OPT_shared))
246  crtend = "crtend.o";
247  else
248  crtend = "crtendS.o";
249 
250  CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath(crtend)));
251  }
252 
253  ToolChain.addProfileRTLibs(Args, CmdArgs);
254 
255  const char *Exec = Args.MakeArgString(ToolChain.GetLinkerPath());
256  C.addCommand(std::make_unique<Command>(JA, *this,
257  ResponseFileSupport::AtFileCurCP(),
258  Exec, CmdArgs, Inputs, Output));
259 }

285 void OpenBSD::AddClangSystemIncludeArgs(
286  const llvm::opt::ArgList &DriverArgs,
287  llvm::opt::ArgStringList &CC1Args) const {
288  const Driver &D = getDriver();
289 
290  if (DriverArgs.hasArg(clang::driver::options::OPT_nostdinc))
291  return;
292 
293  if (!DriverArgs.hasArg(options::OPT_nobuiltininc)) {
294  SmallString<128> Dir(D.ResourceDir);
295  llvm::sys::path::append(Dir, "include");
296  addSystemInclude(DriverArgs, CC1Args, Dir.str());
297  }
298 
299  if (DriverArgs.hasArg(options::OPT_nostdlibinc))
300  return;
301 
302  // Check for configure-time C include directories.
303  StringRef CIncludeDirs(C_INCLUDE_DIRS);
304  if (CIncludeDirs != "") {
305  SmallVector<StringRef, 5> dirs;
306  CIncludeDirs.split(dirs, ":");
307  for (StringRef dir : dirs) {
308  StringRef Prefix =
309  llvm::sys::path::is_absolute(dir) ? StringRef(D.SysRoot) : "";
310  addExternCSystemInclude(DriverArgs, CC1Args, Prefix + dir);
311  }
312  return;
313  }
314 
315  addExternCSystemInclude(DriverArgs, CC1Args, D.SysRoot + "/usr/include");
316 }
