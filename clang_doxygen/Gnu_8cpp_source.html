63 void tools::gcc::Common::ConstructJob(Compilation &C, const JobAction &JA,
64  const InputInfo &Output,
65  const InputInfoList &Inputs,
66  const ArgList &Args,
67  const char *LinkingOutput) const {
68  const Driver &D = getToolChain().getDriver();
69  ArgStringList CmdArgs;
70 
71  for (const auto &A : Args) {
72  if (forwardToGCC(A->getOption())) {
73  // It is unfortunate that we have to claim here, as this means
74  // we will basically never report anything interesting for
75  // platforms using a generic gcc, even if we are just using gcc
76  // to get to the assembler.
77  A->claim();
78 
79  A->render(Args, CmdArgs);
80  }
81  }
82 
83  RenderExtraToolArgs(JA, CmdArgs);
84 
85  // If using a driver driver, force the arch.
86  if (getToolChain().getTriple().isOSDarwin()) {
87  CmdArgs.push_back("-arch");
88  CmdArgs.push_back(
89  Args.MakeArgString(getToolChain().getDefaultUniversalArchName()));
90  }
91 
92  // Try to force gcc to match the tool chain we want, if we recognize
93  // the arch.
94  //
95  // FIXME: The triple class should directly provide the information we want
96  // here.
97  switch (getToolChain().getArch()) {
98  default:
99  break;
100  case llvm::Triple::x86:
101  case llvm::Triple::ppc:
102  case llvm::Triple::ppcle:
103  CmdArgs.push_back("-m32");
104  break;
105  case llvm::Triple::x86_64:
106  case llvm::Triple::ppc64:
107  case llvm::Triple::ppc64le:
108  CmdArgs.push_back("-m64");
109  break;
110  case llvm::Triple::sparcel:
111  CmdArgs.push_back("-EL");
112  break;
113  }
114 
115  if (Output.isFilename()) {
116  CmdArgs.push_back("-o");
117  CmdArgs.push_back(Output.getFilename());
118  } else {
119  assert(Output.isNothing() && "Unexpected output");
120  CmdArgs.push_back("-fsyntax-only");
121  }
122 
123  Args.AddAllArgValues(CmdArgs, options::OPT_Wa_COMMA, options::OPT_Xassembler);
124 
125  // Only pass -x if gcc will understand it; otherwise hope gcc
126  // understands the suffix correctly. The main use case this would go
127  // wrong in is for linker inputs if they happened to have an odd
128  // suffix; really the only way to get this to happen is a command
129  // like '-x foobar a.c' which will treat a.c like a linker input.
130  //
131  // FIXME: For the linker case specifically, can we safely convert
132  // inputs into '-Wl,' options?
133  for (const auto &II : Inputs) {
134  // Don't try to pass LLVM or AST inputs to a generic gcc.
135  if (types::isLLVMIR(II.getType()))
136  D.Diag(clang::diag::err_drv_no_linker_llvm_support)
137  << getToolChain().getTripleString();
138  else if (II.getType() == types::TY_AST)
139  D.Diag(diag::err_drv_no_ast_support) << getToolChain().getTripleString();
140  else if (II.getType() == types::TY_ModuleFile)
141  D.Diag(diag::err_drv_no_module_support)
142  << getToolChain().getTripleString();
143 
144  if (types::canTypeBeUserSpecified(II.getType())) {
145  CmdArgs.push_back("-x");
146  CmdArgs.push_back(types::getTypeName(II.getType()));
147  }
148 
149  if (II.isFilename())
150  CmdArgs.push_back(II.getFilename());
151  else {
152  const Arg &A = II.getInputArg();
153 
154  // Reverse translate some rewritten options.
155  if (A.getOption().matches(options::OPT_Z_reserved_lib_stdcxx)) {
156  CmdArgs.push_back("-lstdc++");
157  continue;
158  }
159 
160  // Don't render as input, we need gcc to do the translations.
161  A.render(Args, CmdArgs);
162  }
163  }
164 
165  const std::string &customGCCName = D.getCCCGenericGCCName();
166  const char *GCCName;
167  if (!customGCCName.empty())
168  GCCName = customGCCName.c_str();
169  else if (D.CCCIsCXX()) {
170  GCCName = "g++";
171  } else
172  GCCName = "gcc";
173 
174  const char *Exec = Args.MakeArgString(getToolChain().GetProgramPath(GCCName));
175  C.addCommand(std::make_unique<Command>(JA, *this,
176  ResponseFileSupport::AtFileCurCP(),
177  Exec, CmdArgs, Inputs, Output));
178 }

218 // On Arm the endianness of the output file is determined by the target and
219 // can be overridden by the pseudo-target flags '-mlittle-endian'/'-EL' and
220 // '-mbig-endian'/'-EB'. Unlike other targets the flag does not result in a
221 // normalized triple so we must handle the flag here.
222 static bool isArmBigEndian(const llvm::Triple &Triple,
223  const ArgList &Args) {
224  bool IsBigEndian = false;
225  switch (Triple.getArch()) {
226  case llvm::Triple::armeb:
227  case llvm::Triple::thumbeb:
228  IsBigEndian = true;
229  LLVM_FALLTHROUGH;
230  case llvm::Triple::arm:
231  case llvm::Triple::thumb:
232  if (Arg *A = Args.getLastArg(options::OPT_mlittle_endian,
233  options::OPT_mbig_endian))
234  IsBigEndian = !A->getOption().matches(options::OPT_mlittle_endian);
235  break;
236  default:
237  break;
238  }
239  return IsBigEndian;
240 }

414 void tools::gnutools::Linker::ConstructJob(Compilation &C, const JobAction &JA,
415  const InputInfo &Output,
416  const InputInfoList &Inputs,
417  const ArgList &Args,
418  const char *LinkingOutput) const {
419  // FIXME: The Linker class constructor takes a ToolChain and not a
420  // Generic_ELF, so the static_cast might return a reference to a invalid
421  // instance (see PR45061). Ideally, the Linker constructor needs to take a
422  // Generic_ELF instead.
423  const toolchains::Generic_ELF &ToolChain =
424  static_cast<const toolchains::Generic_ELF &>(getToolChain());
425  const Driver &D = ToolChain.getDriver();
426 
427  const llvm::Triple &Triple = getToolChain().getEffectiveTriple();
428 
429  const llvm::Triple::ArchType Arch = ToolChain.getArch();
430  const bool isAndroid = ToolChain.getTriple().isAndroid();
431  const bool IsIAMCU = ToolChain.getTriple().isOSIAMCU();
432  const bool IsVE = ToolChain.getTriple().isVE();
433  const bool IsPIE = getPIE(Args, ToolChain);
434  const bool IsStaticPIE = getStaticPIE(Args, ToolChain);
435  const bool IsStatic = getStatic(Args);
436  const bool HasCRTBeginEndFiles =
437  ToolChain.getTriple().hasEnvironment() ||
438  (ToolChain.getTriple().getVendor() != llvm::Triple::MipsTechnologies);
439 
440  // Use of -fsycl-link creates an archive.
441  if (Args.hasArg(options::OPT_fsycl_link_EQ) &&
442  JA.getType() == types::TY_Archive) {
443  constructLLVMARCommand(C, JA, Output, Inputs, Args);
444  return;
445  }
446  ArgStringList CmdArgs;
447 
448  // Silence warning for "clang -g foo.o -o foo"
449  Args.ClaimAllArgs(options::OPT_g_Group);
450  // and "clang -emit-llvm foo.o -o foo"
451  Args.ClaimAllArgs(options::OPT_emit_llvm);
452  // and for "clang -w foo.o -o foo". Other warning options are already
453  // handled somewhere else.
454  Args.ClaimAllArgs(options::OPT_w);
455 
456  if (!D.SysRoot.empty())
457  CmdArgs.push_back(Args.MakeArgString("--sysroot=" + D.SysRoot));
458 
459  if (IsPIE)
460  CmdArgs.push_back("-pie");
461 
462  if (IsStaticPIE) {
463  CmdArgs.push_back("-static");
464  CmdArgs.push_back("-pie");
465  CmdArgs.push_back("--no-dynamic-linker");
466  CmdArgs.push_back("-z");
467  CmdArgs.push_back("text");
468  }
469 
470  if (Args.hasArg(options::OPT_rdynamic))
471  CmdArgs.push_back("-export-dynamic");
472 
473  if (Args.hasArg(options::OPT_s))
474  CmdArgs.push_back("-s");
475 
476  if (Triple.isARM() || Triple.isThumb() || Triple.isAArch64()) {
477  bool IsBigEndian = isArmBigEndian(Triple, Args);
478  if (IsBigEndian)
479  arm::appendBE8LinkFlag(Args, CmdArgs, Triple);
480  IsBigEndian = IsBigEndian || Arch == llvm::Triple::aarch64_be;
481  CmdArgs.push_back(IsBigEndian ? "-EB" : "-EL");
482  }
483 
484  // Most Android ARM64 targets should enable the linker fix for erratum
485  // 843419. Only non-Cortex-A53 devices are allowed to skip this flag.
486  if (Arch == llvm::Triple::aarch64 && isAndroid) {
487  std::string CPU = getCPUName(D, Args, Triple);
488  if (CPU.empty() || CPU == "generic" || CPU == "cortex-a53")
489  CmdArgs.push_back("--fix-cortex-a53-843419");
490  }
491 
492  ToolChain.addExtraOpts(CmdArgs);
493 
494  CmdArgs.push_back("--eh-frame-hdr");
495 
496  if (const char *LDMOption = getLDMOption(ToolChain.getTriple(), Args)) {
497  CmdArgs.push_back("-m");
498  CmdArgs.push_back(LDMOption);
499  } else {
500  D.Diag(diag::err_target_unknown_triple) << Triple.str();
501  return;
502  }
503 
504  if (Args.hasArg(options::OPT_shared))
505  CmdArgs.push_back("-shared");
506 
507  if (IsStatic) {
508  CmdArgs.push_back("-static");
509  } else {
510  if (Args.hasArg(options::OPT_rdynamic))
511  CmdArgs.push_back("-export-dynamic");
512 
513  if (!Args.hasArg(options::OPT_shared) && !IsStaticPIE) {
514  CmdArgs.push_back("-dynamic-linker");
515  CmdArgs.push_back(Args.MakeArgString(Twine(D.DyldPrefix) +
516  ToolChain.getDynamicLinker(Args)));
517  }
518  }
519 
520  CmdArgs.push_back("-o");
521  CmdArgs.push_back(Output.getFilename());
522 
523  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nostartfiles,
524  options::OPT_r)) {
525  if (!isAndroid && !IsIAMCU) {
526  const char *crt1 = nullptr;
527  if (!Args.hasArg(options::OPT_shared)) {
528  if (Args.hasArg(options::OPT_pg))
529  crt1 = "gcrt1.o";
530  else if (IsPIE)
531  crt1 = "Scrt1.o";
532  else if (IsStaticPIE)
533  crt1 = "rcrt1.o";
534  else
535  crt1 = "crt1.o";
536  }
537  if (crt1)
538  CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath(crt1)));
539 
540  CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath("crti.o")));
541  }
542 
543  if (IsVE) {
544  CmdArgs.push_back("-z");
545  CmdArgs.push_back("max-page-size=0x4000000");
546  }
547 
548  if (IsIAMCU)
549  CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath("crt0.o")));
550  else if (HasCRTBeginEndFiles) {
551  std::string P;
552  if (ToolChain.GetRuntimeLibType(Args) == ToolChain::RLT_CompilerRT &&
553  !isAndroid) {
554  std::string crtbegin = ToolChain.getCompilerRT(Args, "crtbegin",
555  ToolChain::FT_Object);
556  if (ToolChain.getVFS().exists(crtbegin))
557  P = crtbegin;
558  }
559  if (P.empty()) {
560  const char *crtbegin;
561  if (Args.hasArg(options::OPT_shared))
562  crtbegin = isAndroid ? "crtbegin_so.o" : "crtbeginS.o";
563  else if (IsStatic)
564  crtbegin = isAndroid ? "crtbegin_static.o" : "crtbeginT.o";
565  else if (IsPIE || IsStaticPIE)
566  crtbegin = isAndroid ? "crtbegin_dynamic.o" : "crtbeginS.o";
567  else
568  crtbegin = isAndroid ? "crtbegin_dynamic.o" : "crtbegin.o";
569  P = ToolChain.GetFilePath(crtbegin);
570  }
571  CmdArgs.push_back(Args.MakeArgString(P));
572  }
573 
574  // Add crtfastmath.o if available and fast math is enabled.
575  ToolChain.addFastMathRuntimeIfAvailable(Args, CmdArgs);
576  }
577 
578  Args.AddAllArgs(CmdArgs, options::OPT_L);
579  Args.AddAllArgs(CmdArgs, options::OPT_u);
580 
581  ToolChain.AddFilePathLibArgs(Args, CmdArgs);
582 
583  if (D.isUsingLTO()) {
584  assert(!Inputs.empty() && "Must have at least one input.");
585  addLTOOptions(ToolChain, Args, CmdArgs, Output, Inputs[0],

590  CmdArgs.push_back("--no-demangle");

592  bool NeedsSanitizerDeps = addSanitizerRuntimes(ToolChain, Args, CmdArgs);
593  bool NeedsXRayDeps = addXRayRuntime(ToolChain, Args, CmdArgs);
594  addLinkerCompressDebugSectionsOption(ToolChain, Args, CmdArgs);
595  // When offloading, the input file(s) could be from unbundled partially
596  // linked archives. The unbundled information is a list of files and not
597  // an actual object/archive. Take that list and pass those to the linker
598  // instead of the original object.
599  if (JA.isDeviceOffloading(Action::OFK_OpenMP)) {
600  InputInfoList UpdatedInputs;
601  // Go through the Inputs to the link. When a listfile is encountered, we
602  // know it is an unbundled generated list.
603  for (const auto &II : Inputs) {
604  if (II.getType() == types::TY_Tempfilelist) {
605  // Take the unbundled list file and pass it in with '@'.
606  std::string FileName(II.getFilename());

609  UpdatedInputs.push_back(CurInput);
610  } else
611  UpdatedInputs.push_back(II);
612  }
613  AddLinkerInputs(ToolChain, UpdatedInputs, Args, CmdArgs, JA);
614  } else
615  AddLinkerInputs(ToolChain, Inputs, Args, CmdArgs, JA);
616 
617  // The profile runtime also needs access to system libraries.
618  getToolChain().addProfileRTLibs(Args, CmdArgs);
619 
620  if (D.CCCIsCXX() &&
621  !Args.hasArg(options::OPT_nostdlib, options::OPT_nodefaultlibs,
622  options::OPT_r)) {
623  if (ToolChain.ShouldLinkCXXStdlib(Args)) {
624  bool OnlyLibstdcxxStatic = Args.hasArg(options::OPT_static_libstdcxx) &&
625  !Args.hasArg(options::OPT_static);
626  if (OnlyLibstdcxxStatic)
627  CmdArgs.push_back("-Bstatic");
628  ToolChain.AddCXXStdlibLibArgs(Args, CmdArgs);
629  if (OnlyLibstdcxxStatic)
630  CmdArgs.push_back("-Bdynamic");
631  }
632  CmdArgs.push_back("-lm");
633  }
634 
635  // If we are linking for the device all symbols should be bound locally. The
636  // symbols are already protected which makes this redundant. This is only
637  // necessary to work around a problem in bfd.
638  // TODO: Remove this once 'lld' becomes the only linker for offloading.
639  if (JA.isDeviceOffloading(Action::OFK_OpenMP))
640  CmdArgs.push_back("-Bsymbolic");
641 
642  // Silence warnings when linking C code with a C++ '-stdlib' argument.
643  Args.ClaimAllArgs(options::OPT_stdlib_EQ);
644 
645  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_r)) {
646  if (!Args.hasArg(options::OPT_nodefaultlibs)) {
647  if (IsStatic || IsStaticPIE)
648  CmdArgs.push_back("--start-group");
649 
650  if (NeedsSanitizerDeps)
651  linkSanitizerRuntimeDeps(ToolChain, CmdArgs);
652 
653  if (NeedsXRayDeps)
654  linkXRayRuntimeDeps(ToolChain, CmdArgs);
655 
656  bool WantPthread = Args.hasArg(options::OPT_pthread) ||
657  Args.hasArg(options::OPT_pthreads);
658 
659  // Use the static OpenMP runtime with -static-openmp
660  bool StaticOpenMP = Args.hasArg(options::OPT_static_openmp) &&
661  !Args.hasArg(options::OPT_static);
662 
663  // FIXME: Only pass GompNeedsRT = true for platforms with libgomp that
664  // require librt. Most modern Linux platforms do, but some may not.
665  if (addOpenMPRuntime(CmdArgs, ToolChain, Args, StaticOpenMP,
666  JA.isHostOffloading(Action::OFK_OpenMP),
667  /* GompNeedsRT= */ true))
668  // OpenMP runtimes implies pthreads when using the GNU toolchain.
669  // FIXME: Does this really make sense for all GNU toolchains?
670  WantPthread = true;
671 
672  AddRunTimeLibs(ToolChain, D, CmdArgs, Args);
673 
674  if (Args.hasArg(options::OPT_fsycl) &&
675  !Args.hasArg(options::OPT_nolibsycl)) {
676  CmdArgs.push_back("-lsycl");
677  // Use of -fintelfpga implies -lOpenCL.
678  // FIXME: Adjust to use plugin interface when available.
679  if (Args.hasArg(options::OPT_fintelfpga))
680  CmdArgs.push_back("-lOpenCL");
681  }
682 
683  if (WantPthread && !isAndroid)
684  CmdArgs.push_back("-lpthread");
685 
686  if (Args.hasArg(options::OPT_fsplit_stack))
687  CmdArgs.push_back("--wrap=pthread_create");
688 
689  if (!Args.hasArg(options::OPT_nolibc))
690  CmdArgs.push_back("-lc");
691 
692  // Add IAMCU specific libs, if needed.
693  if (IsIAMCU)
694  CmdArgs.push_back("-lgloss");
695 
696  if (IsStatic || IsStaticPIE)
697  CmdArgs.push_back("--end-group");
698  else
699  AddRunTimeLibs(ToolChain, D, CmdArgs, Args);
700 
701  // Add IAMCU specific libs (outside the group), if needed.
702  if (IsIAMCU) {
703  CmdArgs.push_back("--as-needed");
704  CmdArgs.push_back("-lsoftfp");
705  CmdArgs.push_back("--no-as-needed");
706  }
707  }
708 
709  if (!Args.hasArg(options::OPT_nostartfiles) && !IsIAMCU) {
710  if (HasCRTBeginEndFiles) {
711  std::string P;
712  if (ToolChain.GetRuntimeLibType(Args) == ToolChain::RLT_CompilerRT &&
713  !isAndroid) {
714  std::string crtend = ToolChain.getCompilerRT(Args, "crtend",
715  ToolChain::FT_Object);
716  if (ToolChain.getVFS().exists(crtend))
717  P = crtend;
718  }
719  if (P.empty()) {
720  const char *crtend;
721  if (Args.hasArg(options::OPT_shared))
722  crtend = isAndroid ? "crtend_so.o" : "crtendS.o";
723  else if (IsPIE || IsStaticPIE)
724  crtend = isAndroid ? "crtend_android.o" : "crtendS.o";
725  else
726  crtend = isAndroid ? "crtend_android.o" : "crtend.o";
727  P = ToolChain.GetFilePath(crtend);
728  }
729  CmdArgs.push_back(Args.MakeArgString(P));
730  }
731  if (!isAndroid)
732  CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath("crtn.o")));
733  }
734  }
735 
736  Args.AddAllArgs(CmdArgs, options::OPT_T);
737 
738  const char *Exec = Args.MakeArgString(ToolChain.GetLinkerPath());
739  C.addCommand(std::make_unique<Command>(JA, *this,
740  ResponseFileSupport::AtFileCurCP(),
741  Exec, CmdArgs, Inputs, Output));
742 }

744 void tools::gnutools::Assembler::ConstructJob(Compilation &C,
745  const JobAction &JA,

753

755

782  // This is incomplete

801  CmdArgs.push_back("-a32");
802  CmdArgs.push_back("-mppc");
803  CmdArgs.push_back("-mlittle-endian");
804  CmdArgs.push_back(ppc::getPPCAsmModeForCPU(
805  getCPUName(D, Args, getToolChain().getTriple())));
806  break;

808  case llvm::Triple::ppc64: {
809  CmdArgs.push_back("-a64");
810  CmdArgs.push_back("-mppc64");
811  CmdArgs.push_back("-mbig-endian");
812  CmdArgs.push_back(ppc::getPPCAsmModeForCPU(
813  getCPUName(D, Args, getToolChain().getTriple())));
814  break;
815  }
816  case llvm::Triple::ppc64le: {
817  CmdArgs.push_back("-a64");
818  CmdArgs.push_back("-mppc64");
819  CmdArgs.push_back("-mlittle-endian");
820  CmdArgs.push_back(ppc::getPPCAsmModeForCPU(
821  getCPUName(D, Args, getToolChain().getTriple())));
822  break;
823  }
824  case llvm::Triple::riscv32:
825  case llvm::Triple::riscv64: {
826  StringRef ABIName = riscv::getRISCVABI(Args, getToolChain().getTriple());
827  CmdArgs.push_back("-mabi");
828  CmdArgs.push_back(ABIName.data());
829  StringRef MArchName = riscv::getRISCVArch(Args, getToolChain().getTriple());
830  CmdArgs.push_back("-march");
831  CmdArgs.push_back(MArchName.data());
832  break;
833  }
834  case llvm::Triple::sparc:
835  case llvm::Triple::sparcel: {
836  CmdArgs.push_back("-32");
837  std::string CPU = getCPUName(D, Args, getToolChain().getTriple());
838  CmdArgs.push_back(
839  sparc::getSparcAsmModeForCPU(CPU, getToolChain().getTriple()));
840  AddAssemblerKPIC(getToolChain(), Args, CmdArgs);
841  break;
842  }
843  case llvm::Triple::sparcv9: {
844  CmdArgs.push_back("-64");
845  std::string CPU = getCPUName(D, Args, getToolChain().getTriple());
846  CmdArgs.push_back(
847  sparc::getSparcAsmModeForCPU(CPU, getToolChain().getTriple()));
848  AddAssemblerKPIC(getToolChain(), Args, CmdArgs);
849  break;
850  }
851  case llvm::Triple::arm:
852  case llvm::Triple::armeb:
853  case llvm::Triple::thumb:
854  case llvm::Triple::thumbeb: {
855  const llvm::Triple &Triple2 = getToolChain().getTriple();
856  CmdArgs.push_back(isArmBigEndian(Triple2, Args) ? "-EB" : "-EL");
857  switch (Triple2.getSubArch()) {
858  case llvm::Triple::ARMSubArch_v7:

861  case llvm::Triple::ARMSubArch_v8:
862  CmdArgs.push_back("-mfpu=crypto-neon-fp-armv8");
863  break;
864  default:
865  break;
866  }

869  case arm::FloatABI::Invalid: llvm_unreachable("must have an ABI!");
870  case arm::FloatABI::Soft:
871  CmdArgs.push_back(Args.MakeArgString("-mfloat-abi=soft"));
872  break;
873  case arm::FloatABI::SoftFP:
874  CmdArgs.push_back(Args.MakeArgString("-mfloat-abi=softfp"));
875  break;
876  case arm::FloatABI::Hard:
877  CmdArgs.push_back(Args.MakeArgString("-mfloat-abi=hard"));
878  break;

886  }

888  case llvm::Triple::aarch64_be: {
889  CmdArgs.push_back(
890  getToolChain().getArch() == llvm::Triple::aarch64_be ? "-EB" : "-EL");
891  Args.AddLastArg(CmdArgs, options::OPT_march_EQ);

893

895  }

897  case llvm::Triple::mipsel:
898  case llvm::Triple::mips64:

901  StringRef ABIName;
902  mips::getMipsCPUAndABI(Args, getToolChain().getTriple(), CPUName, ABIName);
903  ABIName = mips::getGnuCompatibleMipsABIName(ABIName);
904 
905  CmdArgs.push_back("-march");
906  CmdArgs.push_back(CPUName.data());
907 
908  CmdArgs.push_back("-mabi");
909  CmdArgs.push_back(ABIName.data());
910

912  // or -mshared (not implemented) is in effect.
913  if (RelocationModel == llvm::Reloc::Static)
914  CmdArgs.push_back("-mno-shared");
915 
916  // LLVM doesn't support -mplt yet and acts as if it is always given.
917  // However, -mplt has no effect with the N64 ABI.
918  if (ABIName != "64" && !Args.hasArg(options::OPT_mno_abicalls))
919  CmdArgs.push_back("-call_nonpic");
920 
921  if (getToolChain().getTriple().isLittleEndian())
922  CmdArgs.push_back("-EL");
923  else
924  CmdArgs.push_back("-EB");
925 
926  if (Arg *A = Args.getLastArg(options::OPT_mnan_EQ)) {
927  if (StringRef(A->getValue()) == "2008")
928  CmdArgs.push_back(Args.MakeArgString("-mnan=2008"));

932  if (Arg *A = Args.getLastArg(options::OPT_mfp32, options::OPT_mfpxx,

947  A->claim();
948  A->render(Args, CmdArgs);
949  } else {
950  A->claim();
951  CmdArgs.push_back("-no-mips16");
952  }
953  }
954 
955  Args.AddLastArg(CmdArgs, options::OPT_mmicromips,
956  options::OPT_mno_micromips);
957  Args.AddLastArg(CmdArgs, options::OPT_mdsp, options::OPT_mno_dsp);
958  Args.AddLastArg(CmdArgs, options::OPT_mdspr2, options::OPT_mno_dspr2);

964  CmdArgs.push_back(Args.MakeArgString("-mmsa"));
965  }
966 
967  Args.AddLastArg(CmdArgs, options::OPT_mhard_float,
968  options::OPT_msoft_float);
969 
970  Args.AddLastArg(CmdArgs, options::OPT_mdouble_float,
971  options::OPT_msingle_float);
972 
973  Args.AddLastArg(CmdArgs, options::OPT_modd_spreg,
974  options::OPT_mno_odd_spreg);
975 
976  AddAssemblerKPIC(getToolChain(), Args, CmdArgs);
977  break;
978  }

983  CmdArgs.push_back(Args.MakeArgString("-march=" + CPUName));

985  }

987  DefaultAssembler = "nas";
988  }
989

996  else {

998  CmdArgs.push_back(Args.MakeArgString(Map));
999  }
1000  A->claim();
1001  }
1002 
1003  Args.AddAllArgs(CmdArgs, options::OPT_I);
1004  Args.AddAllArgValues(CmdArgs, options::OPT_Wa_COMMA, options::OPT_Xassembler);
1005 
1006  CmdArgs.push_back("-o");
1007  CmdArgs.push_back(Output.getFilename());
1008 
1009  for (const auto &II : Inputs)
1010  CmdArgs.push_back(II.getFilename());
1011 
1012  const char *Exec =
1013  Args.MakeArgString(getToolChain().GetProgramPath(DefaultAssembler));
1014  C.addCommand(std::make_unique<Command>(JA, *this,
1015  ResponseFileSupport::AtFileCurCP(),
1016  Exec, CmdArgs, Inputs, Output));

1762  case llvm::Triple::sparcv9:

1822  Result.Multilibs.push_back(Alt64);
1823  Result.Multilibs.push_back(Alt32);
1824  Result.Multilibs.push_back(Altx32);

1830  addMultilibFlag(TargetTriple.isArch32Bit(), "m32", Flags);

1847 /// Less-than for GCCVersion, implementing a Strict Weak Ordering.
1848 bool Generic_GCC::GCCVersion::isOlderThan(int RHSMajor, int RHSMinor,
1849  int RHSPatch,
1850  StringRef RHSPatchSuffix) const {
1851  if (Major != RHSMajor)
1852  return Major < RHSMajor;
1853  if (Minor != RHSMinor)
1854  return Minor < RHSMinor;
1855  if (Patch != RHSPatch) {
1856  // Note that versions without a specified patch sort higher than those with
1857  // a patch.
1858  if (RHSPatch == -1)
1859  return true;
1860  if (Patch == -1)
1861  return false;
1862 
1863  // Otherwise just sort on the patch itself.
1864  return Patch < RHSPatch;
1865  }
1866  if (PatchSuffix != RHSPatchSuffix) {
1867  // Sort empty suffixes higher.
1868  if (RHSPatchSuffix.empty())
1869  return true;
1870  if (PatchSuffix.empty())
1871  return false;
1872 
1873  // Provide a lexicographic sort to make this a total ordering.
1874  return PatchSuffix < RHSPatchSuffix;
1875  }
1876 
1877  // The versions are equal.
1878  return false;
1879 }

1881 /// Parse a GCCVersion object out of a string of text.
1882 ///
1883 /// This is the primary means of forming GCCVersion objects.
1884 /*static*/
1885 Generic_GCC::GCCVersion Generic_GCC::GCCVersion::Parse(StringRef VersionText) {
1886  const GCCVersion BadVersion = {VersionText.str(), -1, -1, -1, "", "", ""};
1887  std::pair<StringRef, StringRef> First = VersionText.split('.');
1888  std::pair<StringRef, StringRef> Second = First.second.split('.');
1889 
1890  GCCVersion GoodVersion = {VersionText.str(), -1, -1, -1, "", "", ""};
1891  if (First.first.getAsInteger(10, GoodVersion.Major) || GoodVersion.Major < 0)
1892  return BadVersion;
1893  GoodVersion.MajorStr = First.first.str();
1894  if (First.second.empty())
1895  return GoodVersion;
1896  StringRef MinorStr = Second.first;
1897  if (Second.second.empty()) {
1898  if (size_t EndNumber = MinorStr.find_first_not_of("0123456789")) {
1899  GoodVersion.PatchSuffix = std::string(MinorStr.substr(EndNumber));
1900  MinorStr = MinorStr.slice(0, EndNumber);
1901  }
1902  }
1903  if (MinorStr.getAsInteger(10, GoodVersion.Minor) || GoodVersion.Minor < 0)
1904  return BadVersion;
1905  GoodVersion.MinorStr = MinorStr.str();
1906 
1907  // First look for a number prefix and parse that if present. Otherwise just
1908  // stash the entire patch string in the suffix, and leave the number
1909  // unspecified. This covers versions strings such as:
1910  // 5 (handled above)
1911  // 4.4
1912  // 4.4-patched
1913  // 4.4.0
1914  // 4.4.x
1915  // 4.4.2-rc4
1916  // 4.4.x-patched
1917  // And retains any patch number it finds.
1918  StringRef PatchText = Second.second;
1919  if (!PatchText.empty()) {
1920  if (size_t EndNumber = PatchText.find_first_not_of("0123456789")) {
1921  // Try to parse the number and any suffix.
1922  if (PatchText.slice(0, EndNumber).getAsInteger(10, GoodVersion.Patch) ||
1923  GoodVersion.Patch < 0)
1924  return BadVersion;
1925  GoodVersion.PatchSuffix = std::string(PatchText.substr(EndNumber));
1926  }
1927  }
1928 
1929  return GoodVersion;
1930 }

1951 /// Initialize a GCCInstallationDetector from the driver.
1952 ///
1953 /// This performs all of the autodetection and sets up the various paths.
1954 /// Once constructed, a GCCInstallationDetector is essentially immutable.
1955 ///
1956 /// FIXME: We shouldn't need an explicit TargetTriple parameter here, and
1957 /// should instead pull the target out of the driver. This is currently
1958 /// necessary because the driver doesn't store the final version of the target
1959 /// triple.
1960 void Generic_GCC::GCCInstallationDetector::init(
1961  const llvm::Triple &TargetTriple, const ArgList &Args,
1962  ArrayRef<std::string> ExtraTripleAliases) {
1963  llvm::Triple BiarchVariantTriple = TargetTriple.isArch32Bit()
1964  ? TargetTriple.get64BitArchVariant()
1965  : TargetTriple.get32BitArchVariant();
1966  // The library directories which may contain GCC installations.
1967  SmallVector<StringRef, 4> CandidateLibDirs, CandidateBiarchLibDirs;
1968  // The compatible GCC triples for this particular architecture.
1969  SmallVector<StringRef, 16> CandidateTripleAliases;
1970  SmallVector<StringRef, 16> CandidateBiarchTripleAliases;
1971  CollectLibDirsAndTriples(TargetTriple, BiarchVariantTriple, CandidateLibDirs,
1972  CandidateTripleAliases, CandidateBiarchLibDirs,
1973  CandidateBiarchTripleAliases);
1974 
1975  // Compute the set of prefixes for our search.
1976  SmallVector<std::string, 8> Prefixes;
1977  StringRef GCCToolchainDir = getGCCToolchainDir(Args, D.SysRoot);
1978  if (GCCToolchainDir != "") {
1979  if (GCCToolchainDir.back() == '/')
1980  GCCToolchainDir = GCCToolchainDir.drop_back(); // remove the /
1981 
1982  Prefixes.push_back(std::string(GCCToolchainDir));
1983  } else {
1984  // If we have a SysRoot, try that first.
1985  if (!D.SysRoot.empty()) {
1986  Prefixes.push_back(D.SysRoot);
1987  AddDefaultGCCPrefixes(TargetTriple, Prefixes, D.SysRoot);
1988  }
1989 
1990  // Then look for gcc installed alongside clang.
1991  Prefixes.push_back(D.InstalledDir + "/..");
1992 
1993  // Next, look for prefix(es) that correspond to distribution-supplied gcc
1994  // installations.
1995  if (D.SysRoot.empty()) {
1996  // Typically /usr.
1997  AddDefaultGCCPrefixes(TargetTriple, Prefixes, D.SysRoot);
1998  }
1999 
2000  // Try to respect gcc-config on Gentoo if --gcc-toolchain is not provided.
2001  // This avoids accidentally enforcing the system GCC version when using a
2002  // custom toolchain.
2003  SmallVector<StringRef, 16> GentooTestTriples;
2004  // Try to match an exact triple as target triple first.
2005  // e.g. crossdev -S x86_64-gentoo-linux-gnu will install gcc libs for
2006  // x86_64-gentoo-linux-gnu. But "clang -target x86_64-gentoo-linux-gnu"
2007  // may pick the libraries for x86_64-pc-linux-gnu even when exact matching
2008  // triple x86_64-gentoo-linux-gnu is present.
2009  GentooTestTriples.push_back(TargetTriple.str());
2010  // Check rest of triples.
2011  GentooTestTriples.append(ExtraTripleAliases.begin(),
2012  ExtraTripleAliases.end());
2013  GentooTestTriples.append(CandidateTripleAliases.begin(),
2014  CandidateTripleAliases.end());
2015  if (ScanGentooConfigs(TargetTriple, Args, GentooTestTriples,
2016  CandidateBiarchTripleAliases))
2017  return;
2018  }
2019 
2020  // Loop over the various components which exist and select the best GCC
2021  // installation available. GCC installs are ranked by version number.
2022  const GCCVersion VersionZero = GCCVersion::Parse("0.0.0");
2023  Version = VersionZero;
2024  for (const std::string &Prefix : Prefixes) {
2025  auto &VFS = D.getVFS();
2026  if (!VFS.exists(Prefix))
2027  continue;
2028  for (StringRef Suffix : CandidateLibDirs) {
2029  const std::string LibDir = Prefix + Suffix.str();
2030  if (!VFS.exists(LibDir))
2031  continue;
2032  // Maybe filter out <libdir>/gcc and <libdir>/gcc-cross.
2033  bool GCCDirExists = VFS.exists(LibDir + "/gcc");
2034  bool GCCCrossDirExists = VFS.exists(LibDir + "/gcc-cross");
2035  // Try to match the exact target triple first.
2036  ScanLibDirForGCCTriple(TargetTriple, Args, LibDir, TargetTriple.str(),
2037  false, GCCDirExists, GCCCrossDirExists);
2038  // Try rest of possible triples.
2039  for (StringRef Candidate : ExtraTripleAliases) // Try these first.
2040  ScanLibDirForGCCTriple(TargetTriple, Args, LibDir, Candidate, false,
2041  GCCDirExists, GCCCrossDirExists);
2042  for (StringRef Candidate : CandidateTripleAliases)
2043  ScanLibDirForGCCTriple(TargetTriple, Args, LibDir, Candidate, false,
2044  GCCDirExists, GCCCrossDirExists);
2045  }
2046  for (StringRef Suffix : CandidateBiarchLibDirs) {
2047  const std::string LibDir = Prefix + Suffix.str();
2048  if (!VFS.exists(LibDir))
2049  continue;
2050  bool GCCDirExists = VFS.exists(LibDir + "/gcc");
2051  bool GCCCrossDirExists = VFS.exists(LibDir + "/gcc-cross");
2052  for (StringRef Candidate : CandidateBiarchTripleAliases)
2053  ScanLibDirForGCCTriple(TargetTriple, Args, LibDir, Candidate, true,
2054  GCCDirExists, GCCCrossDirExists);
2055  }
2056 
2057  // Skip other prefixes once a GCC installation is found.
2058  if (Version > VersionZero)
2059  break;
2060  }

2859  const std::string &MultiarchTriple,
