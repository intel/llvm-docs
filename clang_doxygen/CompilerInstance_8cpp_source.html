102 bool CompilerInstance::createTarget() {
103  // Create the target instance.
104  setTarget(TargetInfo::CreateTargetInfo(getDiagnostics(),
105  getInvocation().TargetOpts));
106  if (!hasTarget())
107  return false;
108 
109  // Check whether AuxTarget exists, if not, then create TargetInfo for the
110  // other side of CUDA/OpenMP/SYCL compilation.
111  if (!getAuxTarget() &&
112  (getLangOpts().CUDA || getLangOpts().OpenMPIsDevice ||
113  getLangOpts().SYCLIsDevice) &&
114  !getFrontendOpts().AuxTriple.empty()) {
115  auto TO = std::make_shared<TargetOptions>();
116  TO->Triple = llvm::Triple::normalize(getFrontendOpts().AuxTriple);
117  if (getFrontendOpts().AuxTargetCPU)
118  TO->CPU = getFrontendOpts().AuxTargetCPU.getValue();
119  if (getFrontendOpts().AuxTargetFeatures)
120  TO->FeaturesAsWritten = getFrontendOpts().AuxTargetFeatures.getValue();
121  TO->HostTriple = getTarget().getTriple().str();
122  setAuxTarget(TargetInfo::CreateTargetInfo(getDiagnostics(), TO));
123  }
124 
125  if (!getTarget().hasStrictFP() && !getLangOpts().ExpStrictFP) {
126  if (getLangOpts().getFPRoundingMode() !=
127  llvm::RoundingMode::NearestTiesToEven) {
128  getDiagnostics().Report(diag::warn_fe_backend_unsupported_fp_rounding);
129  getLangOpts().setFPRoundingMode(llvm::RoundingMode::NearestTiesToEven);
130  }
131  if (getLangOpts().getFPExceptionMode() != LangOptions::FPE_Ignore) {
132  getDiagnostics().Report(diag::warn_fe_backend_unsupported_fp_exceptions);
133  getLangOpts().setFPExceptionMode(LangOptions::FPE_Ignore);
134  }
135  // FIXME: can we disable FEnvAccess?
136  }
137 
138  // We should do it here because target knows nothing about
139  // language options when it's being created.
140  if (getLangOpts().OpenCL &&
141  !getTarget().validateOpenCLTarget(getLangOpts(), getDiagnostics()))
142  return false;
143 
144  // Inform the target of the language options.
145  //
146  // FIXME: We shouldn't need to do this, the target should be immutable once
147  // created. This complexity should be lifted elsewhere.
148  getTarget().adjust(getDiagnostics(), getLangOpts());
149 
150  // Adjust target options based on codegen options.
151  getTarget().adjustTargetOptions(getCodeGenOpts(), getTargetOpts());
152 
153  if (auto *Aux = getAuxTarget()) {
154  Aux->adjust(getDiagnostics(), getLangOpts());
155  getTarget().setAuxTarget(Aux);
156  }
157 
158  return true;
159 }

230 static void collectIncludePCH(CompilerInstance &CI,
231  std::shared_ptr<ModuleDependencyCollector> MDC) {
232  const PreprocessorOptions &PPOpts = CI.getPreprocessorOpts();
233  if (PPOpts.ImplicitPCHInclude.empty())
234  return;
235 
236  StringRef PCHInclude = PPOpts.ImplicitPCHInclude;
237  FileManager &FileMgr = CI.getFileManager();
238  auto PCHDir = FileMgr.getDirectory(PCHInclude);
239  if (!PCHDir) {
240  MDC->addFile(PCHInclude);
241  return;
242  }
243 
244  std::error_code EC;
245  SmallString<128> DirNative;
246  llvm::sys::path::native((*PCHDir)->getName(), DirNative);
247  llvm::vfs::FileSystem &FS = FileMgr.getVirtualFileSystem();
248  SimpleASTReaderListener Validator(CI.getPreprocessor());
249  for (llvm::vfs::directory_iterator Dir = FS.dir_begin(DirNative, EC), DirEnd;
250  Dir != DirEnd && !EC; Dir.increment(EC)) {
251  // Check whether this is an AST file. ASTReader::isAcceptableASTFile is not
252  // used here since we're not interested in validating the PCH at this time,
253  // but only to check whether this is a file containing an AST.
254  if (!ASTReader::readASTFileControlBlock(
255  Dir->path(), FileMgr, CI.getPCHContainerReader(),
256  /*FindModuleFileExtensions=*/false, Validator,
257  /*ValidateDiagnosticOptions=*/false))
258  MDC->addFile(Dir->path());
259  }
260 }

262 static void collectVFSEntries(CompilerInstance &CI,
263  std::shared_ptr<ModuleDependencyCollector> MDC) {
264  if (CI.getHeaderSearchOpts().VFSOverlayFiles.empty())
265  return;
266 
267  // Collect all VFS found.
268  SmallVector<llvm::vfs::YAMLVFSEntry, 16> VFSEntries;
269  for (const std::string &VFSFile : CI.getHeaderSearchOpts().VFSOverlayFiles) {
270  llvm::ErrorOr<std::unique_ptr<llvm::MemoryBuffer>> Buffer =
271  llvm::MemoryBuffer::getFile(VFSFile);
272  if (!Buffer)
273  return;
274  llvm::vfs::collectVFSFromYAML(std::move(Buffer.get()),
275  /*DiagHandler*/ nullptr, VFSFile, VFSEntries);
276  }
277 
278  for (auto &E : VFSEntries)
279  MDC->addFile(E.VPath, E.RPath);
280 }

283 static void SetUpDiagnosticLog(DiagnosticOptions *DiagOpts,
284  const CodeGenOptions *CodeGenOpts,
285  DiagnosticsEngine &Diags) {
286  std::error_code EC;
287  std::unique_ptr<raw_ostream> StreamOwner;
288  raw_ostream *OS = &llvm::errs();
289  if (DiagOpts->DiagnosticLogFile != "-") {
290  // Create the output stream.
291  auto FileOS = std::make_unique<llvm::raw_fd_ostream>(
292  DiagOpts->DiagnosticLogFile, EC,
293  llvm::sys::fs::OF_Append | llvm::sys::fs::OF_TextWithCRLF);
294  if (EC) {
295  Diags.Report(diag::warn_fe_cc_log_diagnostics_failure)
296  << DiagOpts->DiagnosticLogFile << EC.message();
297  } else {
298  FileOS->SetUnbuffered();
299  OS = FileOS.get();
300  StreamOwner = std::move(FileOS);
301  }
302  }
303 
304  // Chain in the diagnostic client which will log the diagnostics.
305  auto Logger = std::make_unique<LogDiagnosticPrinter>(*OS, DiagOpts,
306  std::move(StreamOwner));
307  if (CodeGenOpts)
308  Logger->setDwarfDebugFlags(CodeGenOpts->DwarfDebugFlags);
309  if (Diags.ownsClient()) {
310  Diags.setClient(
311  new ChainedDiagnosticConsumer(Diags.takeClient(), std::move(Logger)));
312  } else {
313  Diags.setClient(
314  new ChainedDiagnosticConsumer(Diags.getClient(), std::move(Logger)));
315  }
316 }

339 IntrusiveRefCntPtr<DiagnosticsEngine>
340 CompilerInstance::createDiagnostics(DiagnosticOptions *Opts,
341  DiagnosticConsumer *Client,
342  bool ShouldOwnClient,
343  const CodeGenOptions *CodeGenOpts) {
344  IntrusiveRefCntPtr<DiagnosticIDs> DiagID(new DiagnosticIDs());
345  IntrusiveRefCntPtr<DiagnosticsEngine>
346  Diags(new DiagnosticsEngine(DiagID, Opts));
347 
348  // Create the diagnostic client for reporting errors or for
349  // implementing -verify.
350  if (Client) {
351  Diags->setClient(Client, ShouldOwnClient);
352  } else
353  Diags->setClient(new TextDiagnosticPrinter(llvm::errs(), Opts));
354 
355  // Chain in -verify checker, if requested.
356  if (Opts->VerifyDiagnostics)
357  Diags->setClient(new VerifyDiagnosticConsumer(*Diags));
358 
359  // Chain in -diagnostic-log-file dumper, if requested.
360  if (!Opts->DiagnosticLogFile.empty())
361  SetUpDiagnosticLog(Opts, CodeGenOpts, *Diags);
362 
363  if (!Opts->DiagnosticSerializationFile.empty())
364  SetupSerializedDiagnostics(Opts, *Diags,
365  Opts->DiagnosticSerializationFile);
366 
367  // Configure our handling of diagnostics.
368  ProcessWarningOptions(*Diags, *Opts);
369 
370  return Diags;
371 }

392 // Initialize the remapping of files to alternative contents, e.g.,
393 // those specified through other files.
394 static void InitializeFileRemapping(DiagnosticsEngine &Diags,
395  SourceManager &SourceMgr,
396  FileManager &FileMgr,
397  const PreprocessorOptions &InitOpts) {
398  // Remap files in the source manager (with buffers).
399  for (const auto &RB : InitOpts.RemappedFileBuffers) {
400  // Create the file entry for the file that we're mapping from.
401  const FileEntry *FromFile =
402  FileMgr.getVirtualFile(RB.first, RB.second->getBufferSize(), 0);
403  if (!FromFile) {
404  Diags.Report(diag::err_fe_remap_missing_from_file) << RB.first;
405  if (!InitOpts.RetainRemappedFileBuffers)
406  delete RB.second;
407  continue;
408  }
409 
410  // Override the contents of the "from" file with the contents of the
411  // "to" file. If the caller owns the buffers, then pass a MemoryBufferRef;
412  // otherwise, pass as a std::unique_ptr<MemoryBuffer> to transfer ownership
413  // to the SourceManager.
414  if (InitOpts.RetainRemappedFileBuffers)
415  SourceMgr.overrideFileContents(FromFile, RB.second->getMemBufferRef());
416  else
417  SourceMgr.overrideFileContents(
418  FromFile, std::unique_ptr<llvm::MemoryBuffer>(
419  const_cast<llvm::MemoryBuffer *>(RB.second)));
420  }
421 
422  // Remap files in the source manager (with other files).
423  for (const auto &RF : InitOpts.RemappedFiles) {
424  // Find the file that we're mapping to.
425  auto ToFile = FileMgr.getFile(RF.second);
426  if (!ToFile) {
427  Diags.Report(diag::err_fe_remap_missing_to_file) << RF.first << RF.second;
428  continue;
429  }
430 
431  // Create the file entry for the file that we're mapping from.
432  const FileEntry *FromFile =
433  FileMgr.getVirtualFile(RF.first, (*ToFile)->getSize(), 0);
434  if (!FromFile) {
435  Diags.Report(diag::err_fe_remap_missing_from_file) << RF.first;
436  continue;
437  }
438 
439  // Override the contents of the "from" file with the contents of
440  // the "to" file.
441  SourceMgr.overrideFileContents(FromFile, *ToFile);
442  }
443 
444  SourceMgr.setOverridenFilesKeepOriginalName(
445  InitOpts.RemappedFilesKeepOriginalName);
446 }

450 void CompilerInstance::createPreprocessor(TranslationUnitKind TUKind) {
451  const PreprocessorOptions &PPOpts = getPreprocessorOpts();
452 
453  // The AST reader holds a reference to the old preprocessor (if any).
454  TheASTReader.reset();
455 
456  // Create the Preprocessor.
457  HeaderSearch *HeaderInfo =
458  new HeaderSearch(getHeaderSearchOptsPtr(), getSourceManager(),
459  getDiagnostics(), getLangOpts(), &getTarget());
460  PP = std::make_shared<Preprocessor>(Invocation->getPreprocessorOptsPtr(),
461  getDiagnostics(), getLangOpts(),
462  getSourceManager(), *HeaderInfo, *this,
463  /*IdentifierInfoLookup=*/nullptr,
464  /*OwnsHeaderSearch=*/true, TUKind);
465  getTarget().adjust(getDiagnostics(), getLangOpts());
466  PP->Initialize(getTarget(), getAuxTarget());
467 
468  if (PPOpts.DetailedRecord)
469  PP->createPreprocessingRecord();
470 
471  // Apply remappings to the source manager.
472  InitializeFileRemapping(PP->getDiagnostics(), PP->getSourceManager(),
473  PP->getFileManager(), PPOpts);
474 
475  // Predefine macros and configure the preprocessor.
476  InitializePreprocessor(*PP, PPOpts, getPCHContainerReader(),
477  getFrontendOpts());
478 
479  // Initialize the header search object. In CUDA compilations, we use the aux
480  // triple (the host triple) to initialize our header search, since we need to
481  // find the host headers in order to compile the CUDA code.
482  const llvm::Triple *HeaderSearchTriple = &PP->getTargetInfo().getTriple();
483  if (PP->getTargetInfo().getTriple().getOS() == llvm::Triple::CUDA &&
484  PP->getAuxTargetInfo())
485  HeaderSearchTriple = &PP->getAuxTargetInfo()->getTriple();
486 
487  ApplyHeaderSearchOptions(PP->getHeaderSearchInfo(), getHeaderSearchOpts(),
488  PP->getLangOpts(), *HeaderSearchTriple);
489 
490  PP->setPreprocessedOutput(getPreprocessorOutputOpts().ShowCPP);
491 
492  if (PP->getLangOpts().Modules && PP->getLangOpts().ImplicitModules) {
493  std::string ModuleHash = getInvocation().getModuleHash();
494  PP->getHeaderSearchInfo().setModuleHash(ModuleHash);
495  PP->getHeaderSearchInfo().setModuleCachePath(
496  getSpecificModuleCachePath(ModuleHash));
497  }
498 
499  // Handle generating dependencies, if requested.
500  const DependencyOutputOptions &DepOpts = getDependencyOutputOpts();
501  if (!DepOpts.OutputFile.empty())
502  addDependencyCollector(std::make_shared<DependencyFileGenerator>(DepOpts));
503  if (!DepOpts.DOTOutputFile.empty())
504  AttachDependencyGraphGen(*PP, DepOpts.DOTOutputFile,
505  getHeaderSearchOpts().Sysroot);
506 
507  // If we don't have a collector, but we are collecting module dependencies,
508  // then we're the top level compiler instance and need to create one.
509  if (!ModuleDepCollector && !DepOpts.ModuleDependencyOutputDir.empty()) {
510  ModuleDepCollector = std::make_shared<ModuleDependencyCollector>(
511  DepOpts.ModuleDependencyOutputDir);
512  }
513 
514  // If there is a module dep collector, register with other dep collectors
515  // and also (a) collect header maps and (b) TODO: input vfs overlay files.
516  if (ModuleDepCollector) {
517  addDependencyCollector(ModuleDepCollector);
518  collectHeaderMaps(PP->getHeaderSearchInfo(), ModuleDepCollector);
519  collectIncludePCH(*this, ModuleDepCollector);
520  collectVFSEntries(*this, ModuleDepCollector);
521  }
522 
523  for (auto &Listener : DependencyCollectors)
524  Listener->attachToPreprocessor(*PP);
525 
526  // Handle generating header include information, if requested.
527  if (DepOpts.ShowHeaderIncludes)
528  AttachHeaderIncludeGen(*PP, DepOpts);
529  if (!DepOpts.HeaderIncludeOutputFile.empty()) {
530  StringRef OutputPath = DepOpts.HeaderIncludeOutputFile;
531  if (OutputPath == "-")
532  OutputPath = "";
533  AttachHeaderIncludeGen(*PP, DepOpts,
534  /*ShowAllHeaders=*/true, OutputPath,
535  /*ShowDepth=*/false);
536  }
537 
538  if (DepOpts.ShowIncludesDest != ShowIncludesDestination::None) {
539  AttachHeaderIncludeGen(*PP, DepOpts,
540  /*ShowAllHeaders=*/true, /*OutputPath=*/"",
541  /*ShowDepth=*/true, /*MSStyle=*/true);
542  }
543 }

567 // Helper to recursively read the module names for all modules we're adding.
568 // We mark these as known and redirect any attempt to load that module to
569 // the files we were handed.
570 struct ReadModuleNames : ASTReaderListener {
571  Preprocessor &PP;
572  llvm::SmallVector<std::string, 8> LoadedModules;
573 
574  ReadModuleNames(Preprocessor &PP) : PP(PP) {}
575 
576  void ReadModuleName(StringRef ModuleName) override {
577  // Keep the module name as a string for now. It's not safe to create a new
578  // IdentifierInfo from an ASTReader callback.
579  LoadedModules.push_back(ModuleName.str());
580  }

582  void registerAll() {
583  ModuleMap &MM = PP.getHeaderSearchInfo().getModuleMap();
584  for (const std::string &LoadedModule : LoadedModules)
585  MM.cacheModuleLoad(*PP.getIdentifierInfo(LoadedModule),
586  MM.findModule(LoadedModule));
587  LoadedModules.clear();
588  }
589 
590  void markAllUnavailable() {
591  for (const std::string &LoadedModule : LoadedModules) {
592  if (Module *M = PP.getHeaderSearchInfo().getModuleMap().findModule(
593  LoadedModule)) {
594  M->HasIncompatibleModuleFile = true;
595 
596  // Mark module as available if the only reason it was unavailable
597  // was missing headers.
598  SmallVector<Module *, 2> Stack;
599  Stack.push_back(M);
600  while (!Stack.empty()) {
601  Module *Current = Stack.pop_back_val();
602  if (Current->IsUnimportable) continue;
603  Current->IsAvailable = true;
604  Stack.insert(Stack.end(),
605  Current->submodule_begin(), Current->submodule_end());
606  }
607  }
608  }
609  LoadedModules.clear();

628 IntrusiveRefCntPtr<ASTReader> CompilerInstance::createPCHExternalASTSource(
629  StringRef Path, StringRef Sysroot,
630  DisableValidationForModuleKind DisableValidation,
631  bool AllowPCHWithCompilerErrors, Preprocessor &PP,
632  InMemoryModuleCache &ModuleCache, ASTContext &Context,
633  const PCHContainerReader &PCHContainerRdr,
634  ArrayRef<std::shared_ptr<ModuleFileExtension>> Extensions,
635  ArrayRef<std::shared_ptr<DependencyCollector>> DependencyCollectors,
636  void *DeserializationListener, bool OwnDeserializationListener,
637  bool Preamble, bool UseGlobalModuleIndex) {
638  HeaderSearchOptions &HSOpts = PP.getHeaderSearchInfo().getHeaderSearchOpts();
639 
640  IntrusiveRefCntPtr<ASTReader> Reader(new ASTReader(
641  PP, ModuleCache, &Context, PCHContainerRdr, Extensions,

644  HSOpts.ModulesValidateSystemHeaders, HSOpts.ValidateASTInputFilesContent,
645  UseGlobalModuleIndex));
646 
647  // We need the external source to be set up before we read the AST, because
648  // eagerly-deserialized declarations may use it.
649  Context.setExternalSource(Reader.get());
650 
651  Reader->setDeserializationListener(
652  static_cast<ASTDeserializationListener *>(DeserializationListener),
653  /*TakeOwnership=*/OwnDeserializationListener);

656  Listener->attachToASTReader(*Reader);

658  auto Listener = std::make_unique<ReadModuleNames>(PP);
659  auto &ListenerRef = *Listener;
660  ASTReader::ListenerScope ReadModuleNamesListener(*Reader,
661  std::move(Listener));
662 
663  switch (Reader->ReadAST(Path,
664  Preamble ? serialization::MK_Preamble
665  : serialization::MK_PCH,
666  SourceLocation(),
667  ASTReader::ARR_None)) {
668  case ASTReader::Success:

670  // predefines buffer will be empty.

674 
675  case ASTReader::Failure:
676  // Unrecoverable failure: don't even try to process the input file.
677  break;
678 
679  case ASTReader::Missing:
680  case ASTReader::OutOfDate:
681  case ASTReader::VersionMismatch:
682  case ASTReader::ConfigurationMismatch:
683  case ASTReader::HadErrors:
684  // No suitable PCH file could be found. Return an error.
685  break;
686  }
687 
688  ListenerRef.markAllUnavailable();
689  Context.setExternalSource(nullptr);
690  return nullptr;

844 Expected<std::unique_ptr<llvm::raw_pwrite_stream>>
845 CompilerInstance::createOutputFileImpl(StringRef OutputPath, bool Binary,
846  bool RemoveFileOnSignal,
847  bool UseTemporary,
848  bool CreateMissingDirectories) {
849  assert((!CreateMissingDirectories || UseTemporary) &&
850  "CreateMissingDirectories is only allowed when using temporary files");
851 
852  std::unique_ptr<llvm::raw_fd_ostream> OS;
853  Optional<StringRef> OSFile;
854 
855  if (UseTemporary) {
856  if (OutputPath == "-")
857  UseTemporary = false;
858  else {
859  llvm::sys::fs::file_status Status;
860  llvm::sys::fs::status(OutputPath, Status);
861  if (llvm::sys::fs::exists(Status)) {
862  // Fail early if we can't write to the final destination.
863  if (!llvm::sys::fs::can_write(OutputPath))
864  return llvm::errorCodeToError(
865  make_error_code(llvm::errc::operation_not_permitted));
866 
867  // Don't use a temporary if the output is a special file. This handles
868  // things like '-o /dev/null'
869  if (!llvm::sys::fs::is_regular_file(Status))
870  UseTemporary = false;
871  }
872  }
873  }
874 
875  Optional<llvm::sys::fs::TempFile> Temp;
876  if (UseTemporary) {
877  // Create a temporary file.
878  // Insert -%%%%%%%% before the extension (if any), and because some tools
879  // (noticeable, clang's own GlobalModuleIndex.cpp) glob for build
880  // artifacts, also append .tmp.
881  StringRef OutputExtension = llvm::sys::path::extension(OutputPath);
882  SmallString<128> TempPath =
883  StringRef(OutputPath).drop_back(OutputExtension.size());
884  TempPath += "-%%%%%%%%";
885  TempPath += OutputExtension;
886  TempPath += ".tmp";
887  Expected<llvm::sys::fs::TempFile> ExpectedFile =
888  llvm::sys::fs::TempFile::create(
889  TempPath, llvm::sys::fs::all_read | llvm::sys::fs::all_write,
890  Binary ? llvm::sys::fs::OF_None : llvm::sys::fs::OF_Text);
891 
892  llvm::Error E = handleErrors(
893  ExpectedFile.takeError(), [&](const llvm::ECError &E) -> llvm::Error {
894  std::error_code EC = E.convertToErrorCode();
895  if (CreateMissingDirectories &&
896  EC == llvm::errc::no_such_file_or_directory) {
897  StringRef Parent = llvm::sys::path::parent_path(OutputPath);
898  EC = llvm::sys::fs::create_directories(Parent);
899  if (!EC) {
900  ExpectedFile = llvm::sys::fs::TempFile::create(TempPath);
901  if (!ExpectedFile)
902  return llvm::errorCodeToError(
903  llvm::errc::no_such_file_or_directory);
904  }
905  }
906  return llvm::errorCodeToError(EC);
907  });
908 
909  if (E) {
910  consumeError(std::move(E));
911  } else {
912  Temp = std::move(ExpectedFile.get());
913  OS.reset(new llvm::raw_fd_ostream(Temp->FD, /*shouldClose=*/false));
914  OSFile = Temp->TmpName;
915  }
916  // If we failed to create the temporary, fallback to writing to the file
917  // directly. This handles the corner case where we cannot write to the
918  // directory, but can write to the file.
919  }
920 
921  if (!OS) {
922  OSFile = OutputPath;
923  std::error_code EC;
924  OS.reset(new llvm::raw_fd_ostream(
925  *OSFile, EC,
926  (Binary ? llvm::sys::fs::OF_None : llvm::sys::fs::OF_TextWithCRLF)));
927  if (EC)
928  return llvm::errorCodeToError(EC);
929  }
930 
931  // Add the output file -- but don't try to remove "-", since this means we are
932  // using stdin.
933  OutputFiles.emplace_back(((OutputPath != "-") ? OutputPath : "").str(),
934  std::move(Temp));
935 
936  if (!Binary || OS->supportsSeeking())
937  return std::move(OS);
938 
939  return std::make_unique<llvm::buffer_unique_ostream>(std::move(OS));
940 }

993 bool CompilerInstance::ExecuteAction(FrontendAction &Act) {
994  assert(hasDiagnostics() && "Diagnostics engine is not initialized!");
995  assert(!getFrontendOpts().ShowHelp && "Client must handle '-help'!");
996  assert(!getFrontendOpts().ShowVersion && "Client must handle '-version'!");
997 
998  // Mark this point as the bottom of the stack if we don't have somewhere
999  // better. We generally expect frontend actions to be invoked with (nearly)
1000  // DesiredStackSpace available.
1001  noteBottomOfStack();
1002 
1003  auto FinishDiagnosticClient = llvm::make_scope_exit([&]() {
1004  // Notify the diagnostic client that all files were processed.
1005  getDiagnosticClient().finish();
1006  });
1007 
1008  raw_ostream &OS = getVerboseOutputStream();
1009 
1010  if (!Act.PrepareToExecute(*this))
1011  return false;
1012 
1013  if (!createTarget())
1014  return false;
1015 
1016  // rewriter project will change target built-in bool type from its default.
1017  if (getFrontendOpts().ProgramAction == frontend::RewriteObjC)
1018  getTarget().noSignedCharForObjCBool();
1019 
1020  // Validate/process some options.
1021  if (getHeaderSearchOpts().Verbose)
1022  OS << "clang -cc1 version " CLANG_VERSION_STRING
1023  << " based upon " << BACKEND_PACKAGE_STRING
1024  << " default target " << llvm::sys::getDefaultTargetTriple() << "\n";
1025 
1026  if (getCodeGenOpts().TimePasses)
1027  createFrontendTimer();
1028 
1029  if (getFrontendOpts().ShowStats || !getFrontendOpts().StatsFile.empty())
1030  llvm::EnableStatistics(false);
1031 
1032  for (const FrontendInputFile &FIF : getFrontendOpts().Inputs) {
1033  // Reset the ID tables if we are reusing the SourceManager and parsing
1034  // regular files.
1035  if (hasSourceManager() && !Act.isModelParsingAction())
1036  getSourceManager().clearIDTables();
1037 
1038  if (Act.BeginSourceFile(*this, FIF)) {
1039  if (llvm::Error Err = Act.Execute()) {
1040  consumeError(std::move(Err)); // FIXME this drops errors on the floor.
1041  }
1042  Act.EndSourceFile();
1043  }
1044  }
1045 
1046  if (getDiagnosticOpts().ShowCarets) {
1047  // We can have multiple diagnostics sharing one diagnostic client.
1048  // Get the total number of warnings/errors from the client.
1049  unsigned NumWarnings = getDiagnostics().getClient()->getNumWarnings();
1050  unsigned NumErrors = getDiagnostics().getClient()->getNumErrors();
1051 
1052  if (NumWarnings)
1053  OS << NumWarnings << " warning" << (NumWarnings == 1 ? "" : "s");
1054  if (NumWarnings && NumErrors)
1055  OS << " and ";
1056  if (NumErrors)
1057  OS << NumErrors << " error" << (NumErrors == 1 ? "" : "s");
1058  if (NumWarnings || NumErrors) {
1059  OS << " generated";
1060  if (getLangOpts().CUDA) {
1061  if (!getLangOpts().CUDAIsDevice) {
1062  OS << " when compiling for host";
1063  } else {
1064  OS << " when compiling for " << getTargetOpts().CPU;
1065  }
1066  }
1067  OS << ".\n";
1068  }
1069  }
1070 
1071  if (getFrontendOpts().ShowStats) {
1072  if (hasFileManager()) {
1073  getFileManager().PrintStats();
1074  OS << '\n';
1075  }
1076  llvm::PrintStatistics(OS);
1077  }
1078  StringRef StatsFile = getFrontendOpts().StatsFile;
1079  if (!StatsFile.empty()) {
1080  std::error_code EC;
1081  auto StatS = std::make_unique<llvm::raw_fd_ostream>(
1082  StatsFile, EC, llvm::sys::fs::OF_TextWithCRLF);
1083  if (EC) {
1084  getDiagnostics().Report(diag::warn_fe_unable_to_open_stats_file)
1085  << StatsFile << EC.message();
1086  } else {
1087  llvm::PrintStatisticsJSON(*StatS);
1088  }
1089  }
1090 
1091  return !getDiagnostics().getClient()->getNumErrors();
1092 }

1127 /// Compile a module file for the given module, using the options
1128 /// provided by the importing compiler instance. Returns true if the module
1129 /// was built without errors.
1130 static bool
1131 compileModuleImpl(CompilerInstance &ImportingInstance, SourceLocation ImportLoc,
1132  StringRef ModuleName, FrontendInputFile Input,
1133  StringRef OriginalModuleMapFile, StringRef ModuleFileName,
1134  llvm::function_ref<void(CompilerInstance &)> PreBuildStep =
1135  [](CompilerInstance &) {},
1136  llvm::function_ref<void(CompilerInstance &)> PostBuildStep =
1137  [](CompilerInstance &) {}) {
1138  llvm::TimeTraceScope TimeScope("Module Compile", ModuleName);
1139 
1140  // Never compile a module that's already finalized - this would cause the
1141  // existing module to be freed, causing crashes if it is later referenced
1142  if (ImportingInstance.getModuleCache().isPCMFinal(ModuleFileName)) {
1143  ImportingInstance.getDiagnostics().Report(
1144  ImportLoc, diag::err_module_rebuild_finalized)
1145  << ModuleName;
1146  return false;
1147  }
1148 
1149  // Construct a compiler invocation for creating this module.
1150  auto Invocation =
1151  std::make_shared<CompilerInvocation>(ImportingInstance.getInvocation());
1152 
1153  PreprocessorOptions &PPOpts = Invocation->getPreprocessorOpts();
1154 
1155  // For any options that aren't intended to affect how a module is built,
1156  // reset them to their default values.
1157  Invocation->getLangOpts()->resetNonModularOptions();
1158  PPOpts.resetNonModularOptions();
1159 
1160  // Remove any macro definitions that are explicitly ignored by the module.
1161  // They aren't supposed to affect how the module is built anyway.
1162  HeaderSearchOptions &HSOpts = Invocation->getHeaderSearchOpts();
1163  llvm::erase_if(PPOpts.Macros,
1164  [&HSOpts](const std::pair<std::string, bool> &def) {
1165  StringRef MacroDef = def.first;
1166  return HSOpts.ModulesIgnoreMacros.contains(
1167  llvm::CachedHashString(MacroDef.split('=').first));
1168  });
1169 
1170  // If the original compiler invocation had -fmodule-name, pass it through.
1171  Invocation->getLangOpts()->ModuleName =
1172  ImportingInstance.getInvocation().getLangOpts()->ModuleName;
1173 
1174  // Note the name of the module we're building.
1175  Invocation->getLangOpts()->CurrentModule = std::string(ModuleName);
1176 
1177  // Make sure that the failed-module structure has been allocated in
1178  // the importing instance, and propagate the pointer to the newly-created
1179  // instance.
1180  PreprocessorOptions &ImportingPPOpts
1181  = ImportingInstance.getInvocation().getPreprocessorOpts();
1182  if (!ImportingPPOpts.FailedModules)
1183  ImportingPPOpts.FailedModules =
1184  std::make_shared<PreprocessorOptions::FailedModulesSet>();
1185  PPOpts.FailedModules = ImportingPPOpts.FailedModules;
1186 
1187  // If there is a module map file, build the module using the module map.
1188  // Set up the inputs/outputs so that we build the module from its umbrella
1189  // header.
1190  FrontendOptions &FrontendOpts = Invocation->getFrontendOpts();
1191  FrontendOpts.OutputFile = ModuleFileName.str();
1192  FrontendOpts.DisableFree = false;
1193  FrontendOpts.GenerateGlobalModuleIndex = false;
1194  FrontendOpts.BuildingImplicitModule = true;
1195  FrontendOpts.OriginalModuleMap = std::string(OriginalModuleMapFile);
1196  // Force implicitly-built modules to hash the content of the module file.
1197  HSOpts.ModulesHashContent = true;
1198  FrontendOpts.Inputs = {Input};
1199 
1200  // Don't free the remapped file buffers; they are owned by our caller.
1201  PPOpts.RetainRemappedFileBuffers = true;
1202 
1203  Invocation->getDiagnosticOpts().VerifyDiagnostics = 0;
1204  assert(ImportingInstance.getInvocation().getModuleHash() ==
1205  Invocation->getModuleHash() && "Module hash mismatch!");
1206 
1207  // Construct a compiler instance that will be used to actually create the
1208  // module. Since we're sharing an in-memory module cache,
1209  // CompilerInstance::CompilerInstance is responsible for finalizing the
1210  // buffers to prevent use-after-frees.
1211  CompilerInstance Instance(ImportingInstance.getPCHContainerOperations(),
1212  &ImportingInstance.getModuleCache());
1213  auto &Inv = *Invocation;
1214  Instance.setInvocation(std::move(Invocation));
1215 
1216  Instance.createDiagnostics(new ForwardingDiagnosticConsumer(
1217  ImportingInstance.getDiagnosticClient()),
1218  /*ShouldOwnClient=*/true);
1219 
1220  // Note that this module is part of the module build stack, so that we
1221  // can detect cycles in the module graph.
1222  Instance.setFileManager(&ImportingInstance.getFileManager());

1402 /// Compile a module in a separate compiler instance and read the AST,
1403 /// returning true if the module compiles without errors, using a lock manager
1404 /// to avoid building the same module in multiple compiler instances.
1405 ///
1406 /// Uses a lock file manager and exponential backoff to reduce the chances that
1407 /// multiple instances will compete to create the same module. On timeout,
1408 /// deletes the lock file in order to avoid deadlock from crashing processes or
1409 /// bugs in the lock file manager.
1410 static bool compileModuleAndReadASTBehindLock(
1411  CompilerInstance &ImportingInstance, SourceLocation ImportLoc,
1412  SourceLocation ModuleNameLoc, Module *Module, StringRef ModuleFileName) {
1413  DiagnosticsEngine &Diags = ImportingInstance.getDiagnostics();
1414 
1415  Diags.Report(ModuleNameLoc, diag::remark_module_lock)
1416  << ModuleFileName << Module->Name;
1417 
1418  // FIXME: have LockFileManager return an error_code so that we can
1419  // avoid the mkdir when the directory already exists.
1420  StringRef Dir = llvm::sys::path::parent_path(ModuleFileName);
1421  llvm::sys::fs::create_directories(Dir);
1422 
1423  while (true) {
1424  llvm::LockFileManager Locked(ModuleFileName);
1425  switch (Locked) {
1426  case llvm::LockFileManager::LFS_Error:
1427  // ModuleCache takes care of correctness and locks are only necessary for
1428  // performance. Fallback to building the module in case of any lock
1429  // related errors.
1430  Diags.Report(ModuleNameLoc, diag::remark_module_lock_failure)
1431  << Module->Name << Locked.getErrorMessage();
1432  // Clear out any potential leftover.
1433  Locked.unsafeRemoveLockFile();
1434  LLVM_FALLTHROUGH;
1435  case llvm::LockFileManager::LFS_Owned:
1436  // We're responsible for building the module ourselves.
1437  return compileModuleAndReadASTImpl(ImportingInstance, ImportLoc,
1438  ModuleNameLoc, Module, ModuleFileName);
1439 
1440  case llvm::LockFileManager::LFS_Shared:
1441  break; // The interesting case.
1442  }
1443 
1444  // Someone else is responsible for building the module. Wait for them to
1445  // finish.
1446  switch (Locked.waitForUnlock()) {
1447  case llvm::LockFileManager::Res_Success:
1448  break; // The interesting case.
1449  case llvm::LockFileManager::Res_OwnerDied:
1450  continue; // try again to get the lock.
1451  case llvm::LockFileManager::Res_Timeout:
1452  // Since ModuleCache takes care of correctness, we try waiting for
1453  // another process to complete the build so clang does not do it done
1454  // twice. If case of timeout, build it ourselves.
1455  Diags.Report(ModuleNameLoc, diag::remark_module_lock_timeout)
1456  << Module->Name;
1457  // Clear the lock file so that future invocations can make progress.
1458  Locked.unsafeRemoveLockFile();
1459  continue;
1460  }
1461 
1462  // Read the module that was just written by someone else.
1463  bool OutOfDate = false;
1464  if (readASTAfterCompileModule(ImportingInstance, ImportLoc, ModuleNameLoc,
1465  Module, ModuleFileName, &OutOfDate))
1466  return true;
1467  if (!OutOfDate)
1468  return false;
1469 
1470  // The module may be out of date in the presence of file system races,
1471  // or if one of its imports depends on header search paths that are not
1472  // consistent with this ImportingInstance. Try again...
1473  }
1474 }

1495 /// Diagnose differences between the current definition of the given
1496 /// configuration macro and the definition provided on the command line.
1497 static void checkConfigMacro(Preprocessor &PP, StringRef ConfigMacro,
1498  Module *Mod, SourceLocation ImportLoc) {
1499  IdentifierInfo *Id = PP.getIdentifierInfo(ConfigMacro);
1500  SourceManager &SourceMgr = PP.getSourceManager();
1501 
1502  // If this identifier has never had a macro definition, then it could
1503  // not have changed.
1504  if (!Id->hadMacroDefinition())
1505  return;
1506  auto *LatestLocalMD = PP.getLocalMacroDirectiveHistory(Id);
1507 
1508  // Find the macro definition from the command line.
1509  MacroInfo *CmdLineDefinition = nullptr;
1510  for (auto *MD = LatestLocalMD; MD; MD = MD->getPrevious()) {
1511  // We only care about the predefines buffer.
1512  FileID FID = SourceMgr.getFileID(MD->getLocation());
1513  if (FID.isInvalid() || FID != PP.getPredefinesFileID())
1514  continue;
1515  if (auto *DMD = dyn_cast<DefMacroDirective>(MD))
1516  CmdLineDefinition = DMD->getMacroInfo();
1517  break;
1518  }
1519 
1520  auto *CurrentDefinition = PP.getMacroInfo(Id);
1521  if (CurrentDefinition == CmdLineDefinition) {
1522  // Macro matches. Nothing to do.
1523  } else if (!CurrentDefinition) {
1524  // This macro was defined on the command line, then #undef'd later.
1525  // Complain.
1526  PP.Diag(ImportLoc, diag::warn_module_config_macro_undef)
1527  << true << ConfigMacro << Mod->getFullModuleName();
1528  auto LatestDef = LatestLocalMD->getDefinition();
1529  assert(LatestDef.isUndefined() &&
1530  "predefined macro went away with no #undef?");
1531  PP.Diag(LatestDef.getUndefLocation(), diag::note_module_def_undef_here)
1532  << true;
1533  return;
1534  } else if (!CmdLineDefinition) {
1535  // There was no definition for this macro in the predefines buffer,
1536  // but there was a local definition. Complain.
1537  PP.Diag(ImportLoc, diag::warn_module_config_macro_undef)
1538  << false << ConfigMacro << Mod->getFullModuleName();
1539  PP.Diag(CurrentDefinition->getDefinitionLoc(),
1540  diag::note_module_def_undef_here)
1541  << false;
1542  } else if (!CurrentDefinition->isIdenticalTo(*CmdLineDefinition, PP,
1543  /*Syntactically=*/true)) {
1544  // The macro definitions differ.
1545  PP.Diag(ImportLoc, diag::warn_module_config_macro_undef)
1546  << false << ConfigMacro << Mod->getFullModuleName();
1547  PP.Diag(CurrentDefinition->getDefinitionLoc(),
1548  diag::note_module_def_undef_here)
1549  << false;
1550  }
1551 }

1559 /// Prune the module cache of modules that haven't been accessed in
1560 /// a long time.
1561 static void pruneModuleCache(const HeaderSearchOptions &HSOpts) {
1562  llvm::sys::fs::file_status StatBuf;
1563  llvm::SmallString<128> TimestampFile;
1564  TimestampFile = HSOpts.ModuleCachePath;
1565  assert(!TimestampFile.empty());
1566  llvm::sys::path::append(TimestampFile, "modules.timestamp");
1567 
1568  // Try to stat() the timestamp file.
1569  if (std::error_code EC = llvm::sys::fs::status(TimestampFile, StatBuf)) {
1570  // If the timestamp file wasn't there, create one now.
1571  if (EC == std::errc::no_such_file_or_directory) {
1572  writeTimestampFile(TimestampFile);
1573  }
1574  return;
1575  }
1576 
1577  // Check whether the time stamp is older than our pruning interval.
1578  // If not, do nothing.
1579  time_t TimeStampModTime =
1580  llvm::sys::toTimeT(StatBuf.getLastModificationTime());
1581  time_t CurrentTime = time(nullptr);
1582  if (CurrentTime - TimeStampModTime <= time_t(HSOpts.ModuleCachePruneInterval))
1583  return;
1584 
1585  // Write a new timestamp file so that nobody else attempts to prune.
1586  // There is a benign race condition here, if two Clang instances happen to
1587  // notice at the same time that the timestamp is out-of-date.
1588  writeTimestampFile(TimestampFile);
1589 
1590  // Walk the entire module cache, looking for unused module files and module
1591  // indices.
1592  std::error_code EC;
1593  SmallString<128> ModuleCachePathNative;
1594  llvm::sys::path::native(HSOpts.ModuleCachePath, ModuleCachePathNative);
1595  for (llvm::sys::fs::directory_iterator Dir(ModuleCachePathNative, EC), DirEnd;
1596  Dir != DirEnd && !EC; Dir.increment(EC)) {
1597  // If we don't have a directory, there's nothing to look into.
1598  if (!llvm::sys::fs::is_directory(Dir->path()))
1599  continue;
1600 
1601  // Walk all of the files within this directory.
1602  for (llvm::sys::fs::directory_iterator File(Dir->path(), EC), FileEnd;
1603  File != FileEnd && !EC; File.increment(EC)) {
1604  // We only care about module and global module index files.
1605  StringRef Extension = llvm::sys::path::extension(File->path());
1606  if (Extension != ".pcm" && Extension != ".timestamp" &&
1607  llvm::sys::path::filename(File->path()) != "modules.idx")
1608  continue;
1609 
1610  // Look at this file. If we can't stat it, there's nothing interesting
1611  // there.
1612  if (llvm::sys::fs::status(File->path(), StatBuf))
1613  continue;
1614 
1615  // If the file has been used recently enough, leave it there.
1616  time_t FileAccessTime = llvm::sys::toTimeT(StatBuf.getLastAccessedTime());
1617  if (CurrentTime - FileAccessTime <=
1618  time_t(HSOpts.ModuleCachePruneAfter)) {
1619  continue;
1620  }
1621 
1622  // Remove the file.
1623  llvm::sys::fs::remove(File->path());
1624 
1625  // Remove the timestamp file.
1626  std::string TimpestampFilename = File->path() + ".timestamp";
1627  llvm::sys::fs::remove(TimpestampFilename);
1628  }
1629 
1630  // If we removed all of the files in the directory, remove the directory
1631  // itself.
1632  if (llvm::sys::fs::directory_iterator(Dir->path(), EC) ==
1633  llvm::sys::fs::directory_iterator() && !EC)
1634  llvm::sys::fs::remove(Dir->path());
1635  }
1636 }

1745 /// Select a source for loading the named module and compute the filename to
1746 /// load it from.
1747 static ModuleSource selectModuleSource(
1748  Module *M, StringRef ModuleName, std::string &ModuleFilename,
1749  const std::map<std::string, std::string, std::less<>> &BuiltModules,
1750  HeaderSearch &HS) {
1751  assert(ModuleFilename.empty() && "Already has a module source?");
1752 
1753  // Check to see if the module has been built as part of this compilation
1754  // via a module build pragma.
1755  auto BuiltModuleIt = BuiltModules.find(ModuleName);
1756  if (BuiltModuleIt != BuiltModules.end()) {
1757  ModuleFilename = BuiltModuleIt->second;
1758  return MS_ModuleBuildPragma;
1759  }
1760 
1761  // Try to load the module from the prebuilt module path.
1762  const HeaderSearchOptions &HSOpts = HS.getHeaderSearchOpts();
1763  if (!HSOpts.PrebuiltModuleFiles.empty() ||
1764  !HSOpts.PrebuiltModulePaths.empty()) {
1765  ModuleFilename = HS.getPrebuiltModuleFileName(ModuleName);
1766  if (HSOpts.EnablePrebuiltImplicitModules && ModuleFilename.empty())
1767  ModuleFilename = HS.getPrebuiltImplicitModuleFileName(M);
1768  if (!ModuleFilename.empty())
1769  return MS_PrebuiltModulePath;
1770  }
1771 
1772  // Try to load the module from the module cache.
1773  if (M) {
1774  ModuleFilename = HS.getCachedModuleFileName(M);
1775  return MS_ModuleCache;
1776  }
1777 
1778  return MS_ModuleNotFound;
