50 void Sema::AddAlignmentAttributesForRecord(RecordDecl *RD) {
51  AlignPackInfo InfoVal = AlignPackStack.CurrentValue;
52  AlignPackInfo::Mode M = InfoVal.getAlignMode();
53  bool IsPackSet = InfoVal.IsPackSet();
54  bool IsXLPragma = getLangOpts().XLPragmaPack;
55 
56  // If we are not under mac68k/natural alignment mode and also there is no pack
57  // value, we don't need any attributes.
58  if (!IsPackSet && M != AlignPackInfo::Mac68k && M != AlignPackInfo::Natural)
59  return;
60 
61  if (M == AlignPackInfo::Mac68k && (IsXLPragma || InfoVal.IsAlignAttr())) {
62  RD->addAttr(AlignMac68kAttr::CreateImplicit(Context));
63  } else if (IsPackSet) {
64  // Check to see if we need a max field alignment attribute.
65  RD->addAttr(MaxFieldAlignmentAttr::CreateImplicit(
66  Context, InfoVal.getPackNumber() * 8));
67  }
68 
69  if (IsXLPragma && M == AlignPackInfo::Natural)
70  RD->addAttr(AlignNaturalAttr::CreateImplicit(Context));
71 
72  if (AlignPackIncludeStack.empty())
73  return;
74  // The #pragma align/pack affected a record in an included file, so Clang
75  // should warn when that pragma was written in a file that included the
76  // included file.
77  for (auto &AlignPackedInclude : llvm::reverse(AlignPackIncludeStack)) {
78  if (AlignPackedInclude.CurrentPragmaLocation !=
79  AlignPackStack.CurrentPragmaLocation)
80  break;
81  if (AlignPackedInclude.HasNonDefaultValue)
82  AlignPackedInclude.ShouldWarnOnInclude = true;
83  }
84 }

215 void Sema::ActOnPragmaOptionsAlign(PragmaOptionsAlignKind Kind,
216  SourceLocation PragmaLoc) {
217  PragmaMsStackAction Action = Sema::PSK_Reset;
218  AlignPackInfo::Mode ModeVal = AlignPackInfo::Native;
219 
220  switch (Kind) {
221  // For most of the platforms we support, native and natural are the same.
222  // With XL, native is the same as power, natural means something else.
223  //
224  // FIXME: This is not true on Darwin/PPC.
225  case POAK_Native:
226  case POAK_Power:
227  Action = Sema::PSK_Push_Set;
228  break;
229  case POAK_Natural:
230  Action = Sema::PSK_Push_Set;
231  ModeVal = AlignPackInfo::Natural;
232  break;
233 
234  // Note that '#pragma options align=packed' is not equivalent to attribute
235  // packed, it has a different precedence relative to attribute aligned.
236  case POAK_Packed:
237  Action = Sema::PSK_Push_Set;
238  ModeVal = AlignPackInfo::Packed;
239  break;
240 
241  case POAK_Mac68k:
242  // Check if the target supports this.
243  if (!this->Context.getTargetInfo().hasAlignMac68kSupport()) {
244  Diag(PragmaLoc, diag::err_pragma_options_align_mac68k_target_unsupported);
245  return;
246  }
247  Action = Sema::PSK_Push_Set;
248  ModeVal = AlignPackInfo::Mac68k;
249  break;
250  case POAK_Reset:
251  // Reset just pops the top of the stack, or resets the current alignment to
252  // default.
253  Action = Sema::PSK_Pop;
254  if (AlignPackStack.Stack.empty()) {
255  if (AlignPackStack.CurrentValue.getAlignMode() != AlignPackInfo::Native ||
256  AlignPackStack.CurrentValue.IsPackAttr()) {
257  Action = Sema::PSK_Reset;
258  } else {
259  Diag(PragmaLoc, diag::warn_pragma_options_align_reset_failed)
260  << "stack empty";
261  return;
262  }
263  }
264  break;
265  }
266 
267  AlignPackInfo Info(ModeVal, getLangOpts().XLPragmaPack);
268 
269  AlignPackStack.Act(PragmaLoc, Action, StringRef(), Info);
270 }

273  PragmaClangSectionAction Action,
274  PragmaClangSectionKind SecKind,
275  StringRef SecName) {
276  PragmaClangSection *CSec;
277  int SectionFlags = ASTContext::PSF_Read;
278  switch (SecKind) {
279  case PragmaClangSectionKind::PCSK_BSS:
280  CSec = &PragmaClangBSSSection;
281  SectionFlags |= ASTContext::PSF_Write | ASTContext::PSF_ZeroInit;
282  break;
283  case PragmaClangSectionKind::PCSK_Data:
284  CSec = &PragmaClangDataSection;
285  SectionFlags |= ASTContext::PSF_Write;
286  break;
287  case PragmaClangSectionKind::PCSK_Rodata:
288  CSec = &PragmaClangRodataSection;
289  break;
290  case PragmaClangSectionKind::PCSK_Relro:
291  CSec = &PragmaClangRelroSection;
292  break;
293  case PragmaClangSectionKind::PCSK_Text:
294  CSec = &PragmaClangTextSection;
295  SectionFlags |= ASTContext::PSF_Execute;
296  break;
297  default:
298  llvm_unreachable("invalid clang section kind");
299  }
300 
301  if (Action == PragmaClangSectionAction::PCSA_Clear) {
302  CSec->Valid = false;
303  return;
304  }
305 
306  if (llvm::Error E = isValidSectionSpecifier(SecName)) {
307  Diag(PragmaLoc, diag::err_pragma_section_invalid_for_target)
308  << toString(std::move(E));
309  CSec->Valid = false;

311  }
312 
313  if (UnifySection(SecName, SectionFlags, PragmaLoc))
314  return;
315 
316  CSec->Valid = true;
317  CSec->SectionName = std::string(SecName);
318  CSec->PragmaLocation = PragmaLoc;

321 void Sema::ActOnPragmaPack(SourceLocation PragmaLoc, PragmaMsStackAction Action,
322  StringRef SlotLabel, Expr *alignment) {
323  bool IsXLPragma = getLangOpts().XLPragmaPack;
324  // XL pragma pack does not support identifier syntax.
325  if (IsXLPragma && !SlotLabel.empty()) {
326  Diag(PragmaLoc, diag::err_pragma_pack_identifer_not_supported);
327  return;
328  }
329 
330  const AlignPackInfo CurVal = AlignPackStack.CurrentValue;
331  Expr *Alignment = static_cast<Expr *>(alignment);
332 
333  // If specified then alignment must be a "small" power of two.
334  unsigned AlignmentVal = 0;
335  AlignPackInfo::Mode ModeVal = CurVal.getAlignMode();
336 
337  if (Alignment) {
338  Optional<llvm::APSInt> Val;
339  Val = Alignment->getIntegerConstantExpr(Context);
340 
341  // pack(0) is like pack(), which just works out since that is what
342  // we use 0 for in PackAttr.
343  if (Alignment->isTypeDependent() || !Val ||
344  !(*Val == 0 || Val->isPowerOf2()) || Val->getZExtValue() > 16) {
345  Diag(PragmaLoc, diag::warn_pragma_pack_invalid_alignment);
346  return; // Ignore
347  }
348 
349  if (IsXLPragma && *Val == 0) {
350  // pack(0) does not work out with XL.
351  Diag(PragmaLoc, diag::err_pragma_pack_invalid_alignment);
352  return; // Ignore
353  }
354 
355  AlignmentVal = (unsigned)Val->getZExtValue();
356  }
357 
358  if (Action == Sema::PSK_Show) {
359  // Show the current alignment, making sure to show the right value
360  // for the default.
361  // FIXME: This should come from the target.
362  AlignmentVal = CurVal.IsPackSet() ? CurVal.getPackNumber() : 8;
363  if (ModeVal == AlignPackInfo::Mac68k &&
364  (IsXLPragma || CurVal.IsAlignAttr()))
365  Diag(PragmaLoc, diag::warn_pragma_pack_show) << "mac68k";
366  else
367  Diag(PragmaLoc, diag::warn_pragma_pack_show) << AlignmentVal;
368  }
369 
370  // MSDN, C/C++ Preprocessor Reference > Pragma Directives > pack:
371  // "#pragma pack(pop, identifier, n) is undefined"
372  if (Action & Sema::PSK_Pop) {
373  if (Alignment && !SlotLabel.empty())
374  Diag(PragmaLoc, diag::warn_pragma_pack_pop_identifier_and_alignment);
375  if (AlignPackStack.Stack.empty()) {
376  assert(CurVal.getAlignMode() == AlignPackInfo::Native &&
377  "Empty pack stack can only be at Native alignment mode.");
378  Diag(PragmaLoc, diag::warn_pragma_pop_failed) << "pack" << "stack empty";
379  }
380  }
381 
382  AlignPackInfo Info(ModeVal, AlignmentVal, IsXLPragma);
383 
384  AlignPackStack.Act(PragmaLoc, Action, SlotLabel, Info);
385 }

387 void Sema::DiagnoseNonDefaultPragmaAlignPack(PragmaAlignPackDiagnoseKind Kind,
388  SourceLocation IncludeLoc) {
389  if (Kind == PragmaAlignPackDiagnoseKind::NonDefaultStateAtInclude) {
390  SourceLocation PrevLocation = AlignPackStack.CurrentPragmaLocation;
391  // Warn about non-default alignment at #includes (without redundant
392  // warnings for the same directive in nested includes).
393  // The warning is delayed until the end of the file to avoid warnings
394  // for files that don't have any records that are affected by the modified
395  // alignment.
396  bool HasNonDefaultValue =
397  AlignPackStack.hasValue() &&
398  (AlignPackIncludeStack.empty() ||
399  AlignPackIncludeStack.back().CurrentPragmaLocation != PrevLocation);
400  AlignPackIncludeStack.push_back(
401  {AlignPackStack.CurrentValue,
402  AlignPackStack.hasValue() ? PrevLocation : SourceLocation(),
403  HasNonDefaultValue, /*ShouldWarnOnInclude*/ false});
404  return;
405  }
406 
407  assert(Kind == PragmaAlignPackDiagnoseKind::ChangedStateAtExit &&
408  "invalid kind");
409  AlignPackIncludeState PrevAlignPackState =
410  AlignPackIncludeStack.pop_back_val();
411  // FIXME: AlignPackStack may contain both #pragma align and #pragma pack
412  // information, diagnostics below might not be accurate if we have mixed
413  // pragmas.
414  if (PrevAlignPackState.ShouldWarnOnInclude) {
415  // Emit the delayed non-default alignment at #include warning.
416  Diag(IncludeLoc, diag::warn_pragma_pack_non_default_at_include);
417  Diag(PrevAlignPackState.CurrentPragmaLocation, diag::note_pragma_pack_here);
418  }
419  // Warn about modified alignment after #includes.
420  if (PrevAlignPackState.CurrentValue != AlignPackStack.CurrentValue) {
421  Diag(IncludeLoc, diag::warn_pragma_pack_modified_after_include);
422  Diag(AlignPackStack.CurrentPragmaLocation, diag::note_pragma_pack_here);
423  }

426 void Sema::DiagnoseUnterminatedPragmaAlignPack() {
427  if (AlignPackStack.Stack.empty())
428  return;
429  bool IsInnermost = true;
430 
431  // FIXME: AlignPackStack may contain both #pragma align and #pragma pack
432  // information, diagnostics below might not be accurate if we have mixed

435  Diag(StackSlot.PragmaPushLocation, diag::warn_pragma_pack_no_pop_eof);

494 void Sema::ActOnPragmaFloatControl(SourceLocation Loc,
495  PragmaMsStackAction Action,
496  PragmaFloatControlKind Value) {
497  FPOptionsOverride NewFPFeatures = CurFPFeatureOverrides();
498  if ((Action == PSK_Push_Set || Action == PSK_Push || Action == PSK_Pop) &&
499  !CurContext->getRedeclContext()->isFileContext()) {
500  // Push and pop can only occur at file or namespace scope, or within a
501  // language linkage declaration.
502  Diag(Loc, diag::err_pragma_fc_pp_scope);
503  return;
504  }
505  switch (Value) {
506  default:
507  llvm_unreachable("invalid pragma float_control kind");
508  case PFC_Precise:
509  NewFPFeatures.setFPPreciseEnabled(true);
510  FpPragmaStack.Act(Loc, Action, StringRef(), NewFPFeatures);
511  if (PP.getCurrentFPEvalMethod() ==
512  LangOptions::FPEvalMethodKind::FEM_Indeterminable &&
513  PP.getLastFPEvalPragmaLocation().isValid())
514  // A preceding `pragma float_control(precise,off)` has changed
515  // the value of the evaluation method.
516  // Set it back to its old value.
517  PP.setCurrentFPEvalMethod(SourceLocation(), PP.getLastFPEvalMethod());
518  break;
519  case PFC_NoPrecise:
520  if (CurFPFeatures.getFPExceptionMode() == LangOptions::FPE_Strict)
521  Diag(Loc, diag::err_pragma_fc_noprecise_requires_noexcept);
522  else if (CurFPFeatures.getAllowFEnvAccess())
523  Diag(Loc, diag::err_pragma_fc_noprecise_requires_nofenv);
524  else
525  NewFPFeatures.setFPPreciseEnabled(false);
526  FpPragmaStack.Act(Loc, Action, StringRef(), NewFPFeatures);
527  PP.setLastFPEvalMethod(PP.getCurrentFPEvalMethod());
528  // `AllowFPReassoc` or `AllowReciprocal` option is enabled.
529  PP.setCurrentFPEvalMethod(
530  Loc, LangOptions::FPEvalMethodKind::FEM_Indeterminable);
531  break;
532  case PFC_Except:
533  if (!isPreciseFPEnabled())
534  Diag(Loc, diag::err_pragma_fc_except_requires_precise);
535  else
536  NewFPFeatures.setFPExceptionModeOverride(LangOptions::FPE_Strict);
537  FpPragmaStack.Act(Loc, Action, StringRef(), NewFPFeatures);
538  break;
539  case PFC_NoExcept:
540  NewFPFeatures.setFPExceptionModeOverride(LangOptions::FPE_Ignore);
541  FpPragmaStack.Act(Loc, Action, StringRef(), NewFPFeatures);
542  break;
543  case PFC_Push:
544  FpPragmaStack.Act(Loc, Sema::PSK_Push_Set, StringRef(), NewFPFeatures);
545  break;
546  case PFC_Pop:
547  if (FpPragmaStack.Stack.empty()) {
548  Diag(Loc, diag::warn_pragma_pop_failed) << "float_control"
549  << "stack empty";
550  return;
551  }
552  FpPragmaStack.Act(Loc, Action, StringRef(), NewFPFeatures);
553  NewFPFeatures = FpPragmaStack.CurrentValue;
554  if (CurFPFeatures.getAllowFPReassociate() ||
555  CurFPFeatures.getAllowReciprocal())
556  // Since we are popping the pragma, we don't want to be passing
557  // a location here.
558  PP.setCurrentFPEvalMethod(SourceLocation(),
559  CurFPFeatures.getFPEvalMethod());
560  break;
561  }
562  CurFPFeatures = NewFPFeatures.applyOverrides(getLangOpts());
563 }

581 template <>
582 void Sema::PragmaStack<Sema::AlignPackInfo>::Act(SourceLocation PragmaLocation,
583  PragmaMsStackAction Action,
584  llvm::StringRef StackSlotLabel,
585  AlignPackInfo Value) {
586  if (Action == PSK_Reset) {
587  CurrentValue = DefaultValue;
588  CurrentPragmaLocation = PragmaLocation;
589  return;
590  }
591  if (Action & PSK_Push)
592  Stack.emplace_back(Slot(StackSlotLabel, CurrentValue, CurrentPragmaLocation,
593  PragmaLocation));
594  else if (Action & PSK_Pop) {
595  if (!StackSlotLabel.empty()) {
596  // If we've got a label, try to find it and jump there.
597  auto I = llvm::find_if(llvm::reverse(Stack), [&](const Slot &x) {
598  return x.StackSlotLabel == StackSlotLabel;
599  });
600  // We found the label, so pop from there.
601  if (I != Stack.rend()) {
602  CurrentValue = I->Value;
603  CurrentPragmaLocation = I->PragmaLocation;
604  Stack.erase(std::prev(I.base()), Stack.end());
605  }
606  } else if (Value.IsXLStack() && Value.IsAlignAttr() &&
607  CurrentValue.IsPackAttr()) {
608  // XL '#pragma align(reset)' would pop the stack until
609  // a current in effect pragma align is popped.
610  auto I = llvm::find_if(llvm::reverse(Stack), [&](const Slot &x) {
611  return x.Value.IsAlignAttr();
612  });
613  // If we found pragma align so pop from there.
614  if (I != Stack.rend()) {
615  Stack.erase(std::prev(I.base()), Stack.end());
616  if (Stack.empty()) {
617  CurrentValue = DefaultValue;
618  CurrentPragmaLocation = PragmaLocation;
619  } else {
620  CurrentValue = Stack.back().Value;
621  CurrentPragmaLocation = Stack.back().PragmaLocation;
622  Stack.pop_back();
623  }
624  }
625  } else if (!Stack.empty()) {
626  // xl '#pragma align' sets the baseline, and `#pragma pack` cannot pop
627  // over the baseline.
628  if (Value.IsXLStack() && Value.IsPackAttr() && CurrentValue.IsAlignAttr())
629  return;
630 
631  // We don't have a label, just pop the last entry.
632  CurrentValue = Stack.back().Value;
633  CurrentPragmaLocation = Stack.back().PragmaLocation;
634  Stack.pop_back();
635  }
636  }
637  if (Action & PSK_Set) {
638  CurrentValue = Value;
639  CurrentPragmaLocation = PragmaLocation;
640  }

838 void Sema::ActOnPragmaAttributeAttribute(
839  ParsedAttr &Attribute, SourceLocation PragmaLoc,
840  attr::ParsedSubjectMatchRuleSet Rules) {
841  Attribute.setIsPragmaClangAttribute();
842  SmallVector<attr::SubjectMatchRule, 4> SubjectMatchRules;
843  // Gather the subject match rules that are supported by the attribute.
844  SmallVector<std::pair<attr::SubjectMatchRule, bool>, 4>
845  StrictSubjectMatchRuleSet;
846  Attribute.getMatchRules(LangOpts, StrictSubjectMatchRuleSet);
847 
848  // Figure out which subject matching rules are valid.
849  if (StrictSubjectMatchRuleSet.empty()) {
850  // Check for contradicting match rules. Contradicting match rules are
851  // either:
852  // - a top-level rule and one of its sub-rules. E.g. variable and
853  // variable(is_parameter).
854  // - a sub-rule and a sibling that's negated. E.g.
855  // variable(is_thread_local) and variable(unless(is_parameter))
856  llvm::SmallDenseMap<int, std::pair<int, SourceRange>, 2>
857  RulesToFirstSpecifiedNegatedSubRule;
858  for (const auto &Rule : Rules) {
859  attr::SubjectMatchRule MatchRule = attr::SubjectMatchRule(Rule.first);
860  Optional<attr::SubjectMatchRule> ParentRule =
861  getParentAttrMatcherRule(MatchRule);
862  if (!ParentRule)
863  continue;
864  auto It = Rules.find(*ParentRule);
865  if (It != Rules.end()) {
866  // A sub-rule contradicts a parent rule.
867  Diag(Rule.second.getBegin(),
868  diag::err_pragma_attribute_matcher_subrule_contradicts_rule)
869  << attr::getSubjectMatchRuleSpelling(MatchRule)
870  << attr::getSubjectMatchRuleSpelling(*ParentRule) << It->second
871  << FixItHint::CreateRemoval(
872  replacementRangeForListElement(*this, Rule.second));
873  // Keep going without removing this rule as it won't change the set of
874  // declarations that receive the attribute.
875  continue;
876  }
877  if (isNegatedAttrMatcherSubRule(MatchRule))
878  RulesToFirstSpecifiedNegatedSubRule.insert(
879  std::make_pair(*ParentRule, Rule));
880  }
881  bool IgnoreNegatedSubRules = false;
882  for (const auto &Rule : Rules) {
883  attr::SubjectMatchRule MatchRule = attr::SubjectMatchRule(Rule.first);
884  Optional<attr::SubjectMatchRule> ParentRule =
885  getParentAttrMatcherRule(MatchRule);
886  if (!ParentRule)
887  continue;
888  auto It = RulesToFirstSpecifiedNegatedSubRule.find(*ParentRule);
889  if (It != RulesToFirstSpecifiedNegatedSubRule.end() &&
890  It->second != Rule) {
891  // Negated sub-rule contradicts another sub-rule.
892  Diag(
893  It->second.second.getBegin(),
894  diag::
895  err_pragma_attribute_matcher_negated_subrule_contradicts_subrule)
896  << attr::getSubjectMatchRuleSpelling(
897  attr::SubjectMatchRule(It->second.first))
898  << attr::getSubjectMatchRuleSpelling(MatchRule) << Rule.second
899  << FixItHint::CreateRemoval(
900  replacementRangeForListElement(*this, It->second.second));
901  // Keep going but ignore all of the negated sub-rules.
902  IgnoreNegatedSubRules = true;
903  RulesToFirstSpecifiedNegatedSubRule.erase(It);
904  }
905  }
906 
907  if (!IgnoreNegatedSubRules) {
908  for (const auto &Rule : Rules)
909  SubjectMatchRules.push_back(attr::SubjectMatchRule(Rule.first));
910  } else {
911  for (const auto &Rule : Rules) {
912  if (!isNegatedAttrMatcherSubRule(attr::SubjectMatchRule(Rule.first)))
913  SubjectMatchRules.push_back(attr::SubjectMatchRule(Rule.first));
914  }
915  }
916  Rules.clear();
917  } else {
918  // Each rule in Rules must be a strict subset of the attribute's
919  // SubjectMatch rules. I.e. we're allowed to use
920  // `apply_to=variables(is_global)` on an attrubute with SubjectList<[Var]>,
921  // but should not allow `apply_to=variables` on an attribute which has
922  // `SubjectList<[GlobalVar]>`.
923  for (const auto &StrictRule : StrictSubjectMatchRuleSet) {
924  // First, check for exact match.
925  if (Rules.erase(StrictRule.first)) {
926  // Add the rule to the set of attribute receivers only if it's supported
927  // in the current language mode.
928  if (StrictRule.second)
929  SubjectMatchRules.push_back(StrictRule.first);
930  }
931  }
932  // Check remaining rules for subset matches.
933  auto RulesToCheck = Rules;
934  for (const auto &Rule : RulesToCheck) {
935  attr::SubjectMatchRule MatchRule = attr::SubjectMatchRule(Rule.first);
936  if (auto ParentRule = getParentAttrMatcherRule(MatchRule)) {
937  if (llvm::any_of(StrictSubjectMatchRuleSet,
938  [ParentRule](const auto &StrictRule) {
939  return StrictRule.first == *ParentRule &&
940  StrictRule.second; // IsEnabled
941  })) {
942  SubjectMatchRules.push_back(MatchRule);
943  Rules.erase(MatchRule);
944  }
945  }
946  }
947  }
948 
949  if (!Rules.empty()) {
950  auto Diagnostic =
951  Diag(PragmaLoc, diag::err_pragma_attribute_invalid_matchers)
952  << Attribute;
953  SmallVector<attr::SubjectMatchRule, 2> ExtraRules;
954  for (const auto &Rule : Rules) {
955  ExtraRules.push_back(attr::SubjectMatchRule(Rule.first));
956  Diagnostic << FixItHint::CreateRemoval(
957  replacementRangeForListElement(*this, Rule.second));
958  }
959  Diagnostic << attrMatcherRuleListToString(ExtraRules);
960  }
961 
962  if (PragmaAttributeStack.empty()) {
963  Diag(PragmaLoc, diag::err_pragma_attr_attr_no_push);
964  return;
965  }
966 
967  PragmaAttributeStack.back().Entries.push_back(
968  {PragmaLoc, &Attribute, std::move(SubjectMatchRules), /*IsUsed=*/false});
969 }

1254 static bool checkCommonAttributeFeatures(Sema &S, const Ty *Node,
1255  const ParsedAttr &A,
1256  bool SkipArgCountCheck) {
1257  // Several attributes carry different semantics than the parsing requires, so
1258  // those are opted out of the common argument checks.
1259  //
1260  // We also bail on unknown and ignored attributes because those are handled
1261  // as part of the target-specific handling logic.
1262  if (A.getKind() == ParsedAttr::UnknownAttribute)
1263  return false;
1264  // Check whether the attribute requires specific language extensions to be
1265  // enabled.
1266  if (!A.diagnoseLangOpts(S))
1267  return true;
1268  // Check whether the attribute appertains to the given subject.
1269  if (!A.diagnoseAppertainsTo(S, Node))
1270  return true;
1271  // Check whether the attribute is mutually exclusive with other attributes
1272  // that have already been applied to the declaration.
1273  if (!A.diagnoseMutualExclusion(S, Node))
1274  return true;
1275  // Check whether the attribute exists in the target architecture.
1276  if (S.CheckAttrTarget(A))
1277  return true;
1278 
1279  if (A.hasCustomParsing())
1280  return false;
1281 
1282  if (!SkipArgCountCheck) {
1283  if (A.getMinArgs() == A.getMaxArgs()) {
1284  // If there are no optional arguments, then checking for the argument
1285  // count is trivial.
1286  if (!A.checkExactlyNumArgs(S, A.getMinArgs()))
1287  return true;
1288  } else {
1289  // There are optional arguments, so checking is slightly more involved.
1290  if (A.getMinArgs() && !A.checkAtLeastNumArgs(S, A.getMinArgs()))
1291  return true;
1292  else if (!A.hasVariadicArg() && A.getMaxArgs() &&
1293  !A.checkAtMostNumArgs(S, A.getMaxArgs()))
1294  return true;
1295  }
1296  }
1297 
1298  return false;
1299 }
