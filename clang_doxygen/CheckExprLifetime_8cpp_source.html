18 enum LifetimeKind {
19  /// The lifetime of a temporary bound to this entity ends at the end of the
20  /// full-expression, and that's (probably) fine.
21  LK_FullExpression,
22 
23  /// The lifetime of a temporary bound to this entity is extended to the
24  /// lifeitme of the entity itself.
25  LK_Extended,
26 
27  /// The lifetime of a temporary bound to this entity probably ends too soon,
28  /// because the entity is allocated in a new-expression.
29  LK_New,
30 
31  /// The lifetime of a temporary bound to this entity ends too soon, because
32  /// the entity is a return object.
33  LK_Return,
34 
35  /// The lifetime of a temporary bound to this entity ends too soon, because
36  /// the entity is the result of a statement expression.
37  LK_StmtExprResult,
38 
39  /// This is a mem-initializer: if it would extend a temporary (other than via
40  /// a default member initializer), the program is ill-formed.
41  LK_MemInitializer,
42 
43  /// The lifetime of a temporary bound to this entity probably ends too soon,
44  /// because the entity is a pointer and we assign the address of a temporary
45  /// object to it.
46  LK_Assignment,
47 };
48 using LifetimeResult =
49  llvm::PointerIntPair<const InitializedEntity *, 3, LifetimeKind>;

52 /// Determine the declaration which an initialized entity ultimately refers to,
53 /// for the purpose of lifetime-extending a temporary bound to a reference in
54 /// the initialization of \p Entity.
55 static LifetimeResult
56 getEntityLifetime(const InitializedEntity *Entity,
57  const InitializedEntity *InitField = nullptr) {
58  // C++11 [class.temporary]p5:
59  switch (Entity->getKind()) {
60  case InitializedEntity::EK_Variable:
61  // The temporary [...] persists for the lifetime of the reference
62  return {Entity, LK_Extended};
63 
64  case InitializedEntity::EK_Member:
65  // For subobjects, we look at the complete object.
66  if (Entity->getParent())
67  return getEntityLifetime(Entity->getParent(), Entity);
68 
69  // except:
70  // C++17 [class.base.init]p8:
71  // A temporary expression bound to a reference member in a
72  // mem-initializer is ill-formed.
73  // C++17 [class.base.init]p11:
74  // A temporary expression bound to a reference member from a
75  // default member initializer is ill-formed.
76  //
77  // The context of p11 and its example suggest that it's only the use of a
78  // default member initializer from a constructor that makes the program
79  // ill-formed, not its mere existence, and that it can even be used by
80  // aggregate initialization.
81  return {Entity, Entity->isDefaultMemberInitializer() ? LK_Extended
82  : LK_MemInitializer};
83 
84  case InitializedEntity::EK_Binding:
85  // Per [dcl.decomp]p3, the binding is treated as a variable of reference
86  // type.
87  return {Entity, LK_Extended};
88 
89  case InitializedEntity::EK_Parameter:
90  case InitializedEntity::EK_Parameter_CF_Audited:
91  // -- A temporary bound to a reference parameter in a function call
92  // persists until the completion of the full-expression containing
93  // the call.
94  return {nullptr, LK_FullExpression};
95 
96  case InitializedEntity::EK_TemplateParameter:
97  // FIXME: This will always be ill-formed; should we eagerly diagnose it
98  // here?
99  return {nullptr, LK_FullExpression};
100 
101  case InitializedEntity::EK_Result:
102  // -- The lifetime of a temporary bound to the returned value in a
103  // function return statement is not extended; the temporary is
104  // destroyed at the end of the full-expression in the return statement.
105  return {nullptr, LK_Return};
106 
107  case InitializedEntity::EK_StmtExprResult:
108  // FIXME: Should we lifetime-extend through the result of a statement
109  // expression?
110  return {nullptr, LK_StmtExprResult};
111 
112  case InitializedEntity::EK_New:
113  // -- A temporary bound to a reference in a new-initializer persists
114  // until the completion of the full-expression containing the
115  // new-initializer.
116  return {nullptr, LK_New};
117 
118  case InitializedEntity::EK_Temporary:
119  case InitializedEntity::EK_CompoundLiteralInit:
120  case InitializedEntity::EK_RelatedResult:
121  // We don't yet know the storage duration of the surrounding temporary.
122  // Assume it's got full-expression duration for now, it will patch up our
123  // storage duration if that's not correct.
124  return {nullptr, LK_FullExpression};
125 
126  case InitializedEntity::EK_ArrayElement:
127  // For subobjects, we look at the complete object.
128  return getEntityLifetime(Entity->getParent(), InitField);
129 
130  case InitializedEntity::EK_Base:
131  // For subobjects, we look at the complete object.
132  if (Entity->getParent())
133  return getEntityLifetime(Entity->getParent(), InitField);
134  return {InitField, LK_MemInitializer};
135 
136  case InitializedEntity::EK_Delegating:
137  // We can reach this case for aggregate initialization in a constructor:
138  // struct A { int &&r; };
139  // struct B : A { B() : A{0} {} };
140  // In this case, use the outermost field decl as the context.
141  return {InitField, LK_MemInitializer};
142 
143  case InitializedEntity::EK_BlockElement:
144  case InitializedEntity::EK_LambdaToBlockConversionBlockElement:
145  case InitializedEntity::EK_LambdaCapture:
146  case InitializedEntity::EK_VectorElement:
147  case InitializedEntity::EK_ComplexElement:
148  return {nullptr, LK_FullExpression};
149 
150  case InitializedEntity::EK_Exception:
151  // FIXME: Can we diagnose lifetime problems with exceptions?
152  return {nullptr, LK_FullExpression};
153 
154  case InitializedEntity::EK_ParenAggInitMember:
155  // -- A temporary object bound to a reference element of an aggregate of
156  // class type initialized from a parenthesized expression-list
157  // [dcl.init, 9.3] persists until the completion of the full-expression
158  // containing the expression-list.
159  return {nullptr, LK_FullExpression};
160  }
161 
162  llvm_unreachable("unknown entity kind");
163 }

252 // Decl::isInStdNamespace will return false for iterators in some STL
253 // implementations due to them being defined in a namespace outside of the std
254 // namespace.
255 static bool isInStlNamespace(const Decl *D) {
256  const DeclContext *DC = D->getDeclContext();
257  if (!DC)
258  return false;
259  if (const auto *ND = dyn_cast<NamespaceDecl>(DC))
260  if (const IdentifierInfo *II = ND->getIdentifier()) {
261  StringRef Name = II->getName();
262  if (Name.size() >= 2 && Name.front() == '_' &&
263  (Name[1] == '_' || isUppercase(Name[1])))
264  return true;
265  }
266 
267  return DC->isStdNamespace();
268 }

270 static bool shouldTrackImplicitObjectArg(const CXXMethodDecl *Callee) {
271  if (auto *Conv = dyn_cast_or_null<CXXConversionDecl>(Callee))
272  if (isRecordWithAttr<PointerAttr>(Conv->getConversionType()))
273  return true;
274  if (!isInStlNamespace(Callee->getParent()))
275  return false;
276  if (!isRecordWithAttr<PointerAttr>(
277  Callee->getFunctionObjectParameterType()) &&
278  !isRecordWithAttr<OwnerAttr>(Callee->getFunctionObjectParameterType()))
279  return false;
280  if (Callee->getReturnType()->isPointerType() ||
281  isRecordWithAttr<PointerAttr>(Callee->getReturnType())) {
282  if (!Callee->getIdentifier())
283  return false;
284  return llvm::StringSwitch<bool>(Callee->getName())
285  .Cases("begin", "rbegin", "cbegin", "crbegin", true)
286  .Cases("end", "rend", "cend", "crend", true)
287  .Cases("c_str", "data", "get", true)
288  // Map and set types.
289  .Cases("find", "equal_range", "lower_bound", "upper_bound", true)
290  .Default(false);
291  } else if (Callee->getReturnType()->isReferenceType()) {
292  if (!Callee->getIdentifier()) {
293  auto OO = Callee->getOverloadedOperator();
294  return OO == OverloadedOperatorKind::OO_Subscript ||
295  OO == OverloadedOperatorKind::OO_Star;
296  }
297  return llvm::StringSwitch<bool>(Callee->getName())
298  .Cases("front", "back", "at", "top", "value", true)
299  .Default(false);
300  }
301  return false;
302 }

304 static bool shouldTrackFirstArgument(const FunctionDecl *FD) {
305  if (!FD->getIdentifier() || FD->getNumParams() != 1)
306  return false;
307  const auto *RD = FD->getParamDecl(0)->getType()->getPointeeCXXRecordDecl();
308  if (!FD->isInStdNamespace() || !RD || !RD->isInStdNamespace())
309  return false;
310  if (!isRecordWithAttr<PointerAttr>(QualType(RD->getTypeForDecl(), 0)) &&
311  !isRecordWithAttr<OwnerAttr>(QualType(RD->getTypeForDecl(), 0)))
312  return false;
313  if (FD->getReturnType()->isPointerType() ||
314  isRecordWithAttr<PointerAttr>(FD->getReturnType())) {
315  return llvm::StringSwitch<bool>(FD->getName())
316  .Cases("begin", "rbegin", "cbegin", "crbegin", true)
317  .Cases("end", "rend", "cend", "crend", true)
318  .Case("data", true)
319  .Default(false);
320  } else if (FD->getReturnType()->isReferenceType()) {
321  return llvm::StringSwitch<bool>(FD->getName())
322  .Cases("get", "any_cast", true)
323  .Default(false);
324  }
325  return false;

328 static void handleGslAnnotatedTypes(IndirectLocalPath &Path, Expr *Call,
329  LocalVisitor Visit) {
330  auto VisitPointerArg = [&](const Decl *D, Expr *Arg, bool Value) {
331  // We are not interested in the temporary base objects of gsl Pointers:
332  // Temp().ptr; // Here ptr might not dangle.
333  if (isa<MemberExpr>(Arg->IgnoreImpCasts()))
334  return;
335  // Once we initialized a value with a reference, it can no longer dangle.
336  if (!Value) {
337  for (const IndirectLocalPathEntry &PE : llvm::reverse(Path)) {
338  if (PE.Kind == IndirectLocalPathEntry::GslReferenceInit)
339  continue;
340  if (PE.Kind == IndirectLocalPathEntry::GslPointerInit)
341  return;
342  break;
343  }
344  }
345  Path.push_back({Value ? IndirectLocalPathEntry::GslPointerInit
346  : IndirectLocalPathEntry::GslReferenceInit,
347  Arg, D});
348  if (Arg->isGLValue())
349  visitLocalsRetainedByReferenceBinding(Path, Arg, RK_ReferenceBinding,
350  Visit,
351  /*EnableLifetimeWarnings=*/true);
352  else
353  visitLocalsRetainedByInitializer(Path, Arg, Visit, true,
354  /*EnableLifetimeWarnings=*/true);
355  Path.pop_back();
356  };
357 
358  if (auto *MCE = dyn_cast<CXXMemberCallExpr>(Call)) {
359  const auto *MD = cast_or_null<CXXMethodDecl>(MCE->getDirectCallee());
360  if (MD && shouldTrackImplicitObjectArg(MD))
361  VisitPointerArg(MD, MCE->getImplicitObjectArgument(),
362  !MD->getReturnType()->isReferenceType());
363  return;
364  } else if (auto *OCE = dyn_cast<CXXOperatorCallExpr>(Call)) {
365  FunctionDecl *Callee = OCE->getDirectCallee();
366  if (Callee && Callee->isCXXInstanceMember() &&
367  shouldTrackImplicitObjectArg(cast<CXXMethodDecl>(Callee)))
368  VisitPointerArg(Callee, OCE->getArg(0),
369  !Callee->getReturnType()->isReferenceType());
370  return;
371  } else if (auto *CE = dyn_cast<CallExpr>(Call)) {
372  FunctionDecl *Callee = CE->getDirectCallee();
373  if (Callee && shouldTrackFirstArgument(Callee))
374  VisitPointerArg(Callee, CE->getArg(0),
375  !Callee->getReturnType()->isReferenceType());
376  return;
377  }
378 
379  if (auto *CCE = dyn_cast<CXXConstructExpr>(Call)) {
380  const auto *Ctor = CCE->getConstructor();
381  const CXXRecordDecl *RD = Ctor->getParent();
382  if (CCE->getNumArgs() > 0 && RD->hasAttr<PointerAttr>())
383  VisitPointerArg(Ctor->getParamDecl(0), CCE->getArgs()[0], true);
384  }
385 }

387 static bool implicitObjectParamIsLifetimeBound(const FunctionDecl *FD) {
388  const TypeSourceInfo *TSI = FD->getTypeSourceInfo();
389  if (!TSI)
390  return false;
391  // Don't declare this variable in the second operand of the for-statement;
392  // GCC miscompiles that by ending its lifetime before evaluating the
393  // third operand. See gcc.gnu.org/PR86769.
394  AttributedTypeLoc ATL;
395  for (TypeLoc TL = TSI->getTypeLoc();
396  (ATL = TL.getAsAdjusted<AttributedTypeLoc>());
397  TL = ATL.getModifiedLoc()) {
398  if (ATL.getAttrAs<LifetimeBoundAttr>())
399  return true;
400  }
401 
402  // Assume that all assignment operators with a "normal" return type return
403  // *this, that is, an lvalue reference that is the same type as the implicit
404  // object parameter (or the LHS for a non-member operator$=).
405  OverloadedOperatorKind OO = FD->getDeclName().getCXXOverloadedOperator();
406  if (OO == OO_Equal || isCompoundAssignmentOperator(OO)) {
407  QualType RetT = FD->getReturnType();
408  if (RetT->isLValueReferenceType()) {
409  ASTContext &Ctx = FD->getASTContext();
410  QualType LHST;
411  auto *MD = dyn_cast<CXXMethodDecl>(FD);
412  if (MD && MD->isCXXInstanceMember())
413  LHST = Ctx.getLValueReferenceType(MD->getFunctionObjectParameterType());
414  else
415  LHST = MD->getParamDecl(0)->getType();
416  if (Ctx.hasSameType(RetT, LHST))
417  return true;
418  }
419  }
420 
421  return false;
422 }

424 static void visitLifetimeBoundArguments(IndirectLocalPath &Path, Expr *Call,
425  LocalVisitor Visit) {
426  const FunctionDecl *Callee;
427  ArrayRef<Expr *> Args;
428 
429  if (auto *CE = dyn_cast<CallExpr>(Call)) {
430  Callee = CE->getDirectCallee();
431  Args = llvm::ArrayRef(CE->getArgs(), CE->getNumArgs());
432  } else {
433  auto *CCE = cast<CXXConstructExpr>(Call);
434  Callee = CCE->getConstructor();
435  Args = llvm::ArrayRef(CCE->getArgs(), CCE->getNumArgs());
436  }
437  if (!Callee)
438  return;
439 
440  Expr *ObjectArg = nullptr;
441  if (isa<CXXOperatorCallExpr>(Call) && Callee->isCXXInstanceMember()) {
442  ObjectArg = Args[0];
443  Args = Args.slice(1);
444  } else if (auto *MCE = dyn_cast<CXXMemberCallExpr>(Call)) {
445  ObjectArg = MCE->getImplicitObjectArgument();
446  }
447 
448  auto VisitLifetimeBoundArg = [&](const Decl *D, Expr *Arg) {
449  Path.push_back({IndirectLocalPathEntry::LifetimeBoundCall, Arg, D});
450  if (Arg->isGLValue())
451  visitLocalsRetainedByReferenceBinding(Path, Arg, RK_ReferenceBinding,
452  Visit,
453  /*EnableLifetimeWarnings=*/false);
454  else
455  visitLocalsRetainedByInitializer(Path, Arg, Visit, true,
456  /*EnableLifetimeWarnings=*/false);
457  Path.pop_back();
458  };
459 
460  bool CheckCoroCall = false;
461  if (const auto *RD = Callee->getReturnType()->getAsRecordDecl()) {
462  CheckCoroCall = RD->hasAttr<CoroLifetimeBoundAttr>() &&
463  RD->hasAttr<CoroReturnTypeAttr>() &&
464  !Callee->hasAttr<CoroDisableLifetimeBoundAttr>();
465  }
466 
467  if (ObjectArg) {
468  bool CheckCoroObjArg = CheckCoroCall;
469  // Coroutine lambda objects with empty capture list are not lifetimebound.
470  if (auto *LE = dyn_cast<LambdaExpr>(ObjectArg->IgnoreImplicit());
471  LE && LE->captures().empty())
472  CheckCoroObjArg = false;
473  // Allow `get_return_object()` as the object param (__promise) is not
474  // lifetimebound.
475  if (Sema::CanBeGetReturnObject(Callee))
476  CheckCoroObjArg = false;
477  if (implicitObjectParamIsLifetimeBound(Callee) || CheckCoroObjArg)
478  VisitLifetimeBoundArg(Callee, ObjectArg);
479  }
480 
481  for (unsigned I = 0,
482  N = std::min<unsigned>(Callee->getNumParams(), Args.size());
483  I != N; ++I) {
484  if (CheckCoroCall || Callee->getParamDecl(I)->hasAttr<LifetimeBoundAttr>())
485  VisitLifetimeBoundArg(Callee->getParamDecl(I), Args[I]);
486  }
487 }

489 /// Visit the locals that would be reachable through a reference bound to the
490 /// glvalue expression \c Init.
491 static void visitLocalsRetainedByReferenceBinding(IndirectLocalPath &Path,
492  Expr *Init, ReferenceKind RK,
493  LocalVisitor Visit,
494  bool EnableLifetimeWarnings) {
495  RevertToOldSizeRAII RAII(Path);
496 
497  // Walk past any constructs which we can lifetime-extend across.
498  Expr *Old;
499  do {
500  Old = Init;
501 
502  if (auto *FE = dyn_cast<FullExpr>(Init))
503  Init = FE->getSubExpr();
504 
505  if (InitListExpr *ILE = dyn_cast<InitListExpr>(Init)) {
506  // If this is just redundant braces around an initializer, step over it.
507  if (ILE->isTransparent())
508  Init = ILE->getInit(0);
509  }
510 
511  // Step over any subobject adjustments; we may have a materialized
512  // temporary inside them.
513  Init = const_cast<Expr *>(Init->skipRValueSubobjectAdjustments());
514 
515  // Per current approach for DR1376, look through casts to reference type
516  // when performing lifetime extension.
517  if (CastExpr *CE = dyn_cast<CastExpr>(Init))
518  if (CE->getSubExpr()->isGLValue())
519  Init = CE->getSubExpr();
520 
521  // Per the current approach for DR1299, look through array element access
522  // on array glvalues when performing lifetime extension.
523  if (auto *ASE = dyn_cast<ArraySubscriptExpr>(Init)) {
524  Init = ASE->getBase();
525  auto *ICE = dyn_cast<ImplicitCastExpr>(Init);
526  if (ICE && ICE->getCastKind() == CK_ArrayToPointerDecay)
527  Init = ICE->getSubExpr();
528  else
529  // We can't lifetime extend through this but we might still find some
530  // retained temporaries.
531  return visitLocalsRetainedByInitializer(Path, Init, Visit, true,
532  EnableLifetimeWarnings);
533  }
534 
535  // Step into CXXDefaultInitExprs so we can diagnose cases where a
536  // constructor inherits one as an implicit mem-initializer.
537  if (auto *DIE = dyn_cast<CXXDefaultInitExpr>(Init)) {
538  Path.push_back(
539  {IndirectLocalPathEntry::DefaultInit, DIE, DIE->getField()});
540  Init = DIE->getExpr();
541  }
542  } while (Init != Old);
543 
544  if (auto *MTE = dyn_cast<MaterializeTemporaryExpr>(Init)) {
545  if (Visit(Path, Local(MTE), RK))
546  visitLocalsRetainedByInitializer(Path, MTE->getSubExpr(), Visit, true,
547  EnableLifetimeWarnings);
548  }
549 
550  if (isa<CallExpr>(Init)) {
551  if (EnableLifetimeWarnings)
552  handleGslAnnotatedTypes(Path, Init, Visit);
553  return visitLifetimeBoundArguments(Path, Init, Visit);
554  }
555 
556  switch (Init->getStmtClass()) {
557  case Stmt::DeclRefExprClass: {
558  // If we find the name of a local non-reference parameter, we could have a
559  // lifetime problem.
560  auto *DRE = cast<DeclRefExpr>(Init);
561  auto *VD = dyn_cast<VarDecl>(DRE->getDecl());
562  if (VD && VD->hasLocalStorage() &&
563  !DRE->refersToEnclosingVariableOrCapture()) {
564  if (!VD->getType()->isReferenceType()) {
565  Visit(Path, Local(DRE), RK);
566  } else if (isa<ParmVarDecl>(DRE->getDecl())) {
567  // The lifetime of a reference parameter is unknown; assume it's OK
568  // for now.
569  break;
570  } else if (VD->getInit() && !isVarOnPath(Path, VD)) {
571  Path.push_back({IndirectLocalPathEntry::VarInit, DRE, VD});
572  visitLocalsRetainedByReferenceBinding(Path, VD->getInit(),
573  RK_ReferenceBinding, Visit,
574  EnableLifetimeWarnings);
575  }
576  }
577  break;
578  }
579 
580  case Stmt::UnaryOperatorClass: {
581  // The only unary operator that make sense to handle here
582  // is Deref. All others don't resolve to a "name." This includes
583  // handling all sorts of rvalues passed to a unary operator.
584  const UnaryOperator *U = cast<UnaryOperator>(Init);
585  if (U->getOpcode() == UO_Deref)
586  visitLocalsRetainedByInitializer(Path, U->getSubExpr(), Visit, true,
587  EnableLifetimeWarnings);
588  break;
589  }
590 
591  case Stmt::ArraySectionExprClass: {
592  visitLocalsRetainedByInitializer(Path,
593  cast<ArraySectionExpr>(Init)->getBase(),
594  Visit, true, EnableLifetimeWarnings);
595  break;
596  }
597 
598  case Stmt::ConditionalOperatorClass:
599  case Stmt::BinaryConditionalOperatorClass: {
600  auto *C = cast<AbstractConditionalOperator>(Init);
601  if (!C->getTrueExpr()->getType()->isVoidType())
602  visitLocalsRetainedByReferenceBinding(Path, C->getTrueExpr(), RK, Visit,
603  EnableLifetimeWarnings);
604  if (!C->getFalseExpr()->getType()->isVoidType())
605  visitLocalsRetainedByReferenceBinding(Path, C->getFalseExpr(), RK, Visit,
606  EnableLifetimeWarnings);
607  break;
608  }
609 
610  case Stmt::CompoundLiteralExprClass: {

615  break;
616  }

620  default:
621  break;

625 /// Visit the locals that would be reachable through an object initialized by
626 /// the prvalue expression \c Init.
627 static void visitLocalsRetainedByInitializer(IndirectLocalPath &Path,
628  Expr *Init, LocalVisitor Visit,
629  bool RevisitSubinits,
630  bool EnableLifetimeWarnings) {
631  RevertToOldSizeRAII RAII(Path);
632 
633  Expr *Old;
634  do {
635  Old = Init;
636 
637  // Step into CXXDefaultInitExprs so we can diagnose cases where a
638  // constructor inherits one as an implicit mem-initializer.
639  if (auto *DIE = dyn_cast<CXXDefaultInitExpr>(Init)) {
640  Path.push_back(
641  {IndirectLocalPathEntry::DefaultInit, DIE, DIE->getField()});
642  Init = DIE->getExpr();
643  }
644 
645  if (auto *FE = dyn_cast<FullExpr>(Init))
646  Init = FE->getSubExpr();
647 
648  // Dig out the expression which constructs the extended temporary.
649  Init = const_cast<Expr *>(Init->skipRValueSubobjectAdjustments());
650 
651  if (CXXBindTemporaryExpr *BTE = dyn_cast<CXXBindTemporaryExpr>(Init))
652  Init = BTE->getSubExpr();
653 
654  Init = Init->IgnoreParens();
655 
656  // Step over value-preserving rvalue casts.
657  if (auto *CE = dyn_cast<CastExpr>(Init)) {
658  switch (CE->getCastKind()) {
659  case CK_LValueToRValue:
660  // If we can match the lvalue to a const object, we can look at its
661  // initializer.
662  Path.push_back({IndirectLocalPathEntry::LValToRVal, CE});
663  return visitLocalsRetainedByReferenceBinding(
664  Path, Init, RK_ReferenceBinding,
665  [&](IndirectLocalPath &Path, Local L, ReferenceKind RK) -> bool {
666  if (auto *DRE = dyn_cast<DeclRefExpr>(L)) {
667  auto *VD = dyn_cast<VarDecl>(DRE->getDecl());
668  if (VD && VD->getType().isConstQualified() && VD->getInit() &&
669  !isVarOnPath(Path, VD)) {
670  Path.push_back({IndirectLocalPathEntry::VarInit, DRE, VD});
671  visitLocalsRetainedByInitializer(
672  Path, VD->getInit(), Visit, true, EnableLifetimeWarnings);
673  }
674  } else if (auto *MTE = dyn_cast<MaterializeTemporaryExpr>(L)) {
675  if (MTE->getType().isConstQualified())
676  visitLocalsRetainedByInitializer(Path, MTE->getSubExpr(),
677  Visit, true,
678  EnableLifetimeWarnings);
679  }
680  return false;
681  },
682  EnableLifetimeWarnings);
683 
684  // We assume that objects can be retained by pointers cast to integers,
685  // but not if the integer is cast to floating-point type or to _Complex.
686  // We assume that casts to 'bool' do not preserve enough information to
687  // retain a local object.
688  case CK_NoOp:
689  case CK_BitCast:
690  case CK_BaseToDerived:
691  case CK_DerivedToBase:
692  case CK_UncheckedDerivedToBase:
693  case CK_Dynamic:
694  case CK_ToUnion:
695  case CK_UserDefinedConversion:
696  case CK_ConstructorConversion:
697  case CK_IntegralToPointer:
698  case CK_PointerToIntegral:
699  case CK_VectorSplat:
700  case CK_IntegralCast:
701  case CK_CPointerToObjCPointerCast:
702  case CK_BlockPointerToObjCPointerCast:
703  case CK_AnyPointerToBlockPointerCast:
704  case CK_AddressSpaceConversion:
705  break;
706 
707  case CK_ArrayToPointerDecay:
708  // Model array-to-pointer decay as taking the address of the array
709  // lvalue.
710  Path.push_back({IndirectLocalPathEntry::AddressOf, CE});
711  return visitLocalsRetainedByReferenceBinding(Path, CE->getSubExpr(),
712  RK_ReferenceBinding, Visit,
713  EnableLifetimeWarnings);
714 
715  default:
716  return;
717  }
718 
719  Init = CE->getSubExpr();
720  }
721  } while (Old != Init);
722 
723  // C++17 [dcl.init.list]p6:
724  // initializing an initializer_list object from the array extends the
725  // lifetime of the array exactly like binding a reference to a temporary.
726  if (auto *ILE = dyn_cast<CXXStdInitializerListExpr>(Init))
727  return visitLocalsRetainedByReferenceBinding(Path, ILE->getSubExpr(),
728  RK_StdInitializerList, Visit,
729  EnableLifetimeWarnings);
730 
731  if (InitListExpr *ILE = dyn_cast<InitListExpr>(Init)) {
732  // We already visited the elements of this initializer list while
733  // performing the initialization. Don't visit them again unless we've
734  // changed the lifetime of the initialized entity.
735  if (!RevisitSubinits)
736  return;
737 
738  if (ILE->isTransparent())
739  return visitLocalsRetainedByInitializer(Path, ILE->getInit(0), Visit,
740  RevisitSubinits,
741  EnableLifetimeWarnings);
742 
743  if (ILE->getType()->isArrayType()) {
744  for (unsigned I = 0, N = ILE->getNumInits(); I != N; ++I)
745  visitLocalsRetainedByInitializer(Path, ILE->getInit(I), Visit,
746  RevisitSubinits,
747  EnableLifetimeWarnings);
748  return;
749  }
750 
751  if (CXXRecordDecl *RD = ILE->getType()->getAsCXXRecordDecl()) {
752  assert(RD->isAggregate() && "aggregate init on non-aggregate");
753 
754  // If we lifetime-extend a braced initializer which is initializing an
755  // aggregate, and that aggregate contains reference members which are
756  // bound to temporaries, those temporaries are also lifetime-extended.
757  if (RD->isUnion() && ILE->getInitializedFieldInUnion() &&
758  ILE->getInitializedFieldInUnion()->getType()->isReferenceType())
759  visitLocalsRetainedByReferenceBinding(Path, ILE->getInit(0),
760  RK_ReferenceBinding, Visit,
761  EnableLifetimeWarnings);
762  else {
763  unsigned Index = 0;
764  for (; Index < RD->getNumBases() && Index < ILE->getNumInits(); ++Index)
765  visitLocalsRetainedByInitializer(Path, ILE->getInit(Index), Visit,
766  RevisitSubinits,
767  EnableLifetimeWarnings);
768  for (const auto *I : RD->fields()) {
769  if (Index >= ILE->getNumInits())
770  break;
771  if (I->isUnnamedBitField())
772  continue;
773  Expr *SubInit = ILE->getInit(Index);
774  if (I->getType()->isReferenceType())
775  visitLocalsRetainedByReferenceBinding(Path, SubInit,
776  RK_ReferenceBinding, Visit,
777  EnableLifetimeWarnings);
778  else
779  // This might be either aggregate-initialization of a member or
780  // initialization of a std::initializer_list object. Regardless,
781  // we should recursively lifetime-extend that initializer.
782  visitLocalsRetainedByInitializer(
783  Path, SubInit, Visit, RevisitSubinits, EnableLifetimeWarnings);
784  ++Index;
785  }
786  }
787  }
788  return;
789  }
790 
791  // The lifetime of an init-capture is that of the closure object constructed
792  // by a lambda-expression.
793  if (auto *LE = dyn_cast<LambdaExpr>(Init)) {
794  LambdaExpr::capture_iterator CapI = LE->capture_begin();
795  for (Expr *E : LE->capture_inits()) {
796  assert(CapI != LE->capture_end());
797  const LambdaCapture &Cap = *CapI++;
798  if (!E)
799  continue;
800  if (Cap.capturesVariable())
801  Path.push_back({IndirectLocalPathEntry::LambdaCaptureInit, E, &Cap});
802  if (E->isGLValue())
803  visitLocalsRetainedByReferenceBinding(Path, E, RK_ReferenceBinding,
804  Visit, EnableLifetimeWarnings);
805  else
806  visitLocalsRetainedByInitializer(Path, E, Visit, true,
807  EnableLifetimeWarnings);
808  if (Cap.capturesVariable())
809  Path.pop_back();
810  }
811  }
812 
813  // Assume that a copy or move from a temporary references the same objects
814  // that the temporary does.
815  if (auto *CCE = dyn_cast<CXXConstructExpr>(Init)) {
816  if (CCE->getConstructor()->isCopyOrMoveConstructor()) {
817  if (auto *MTE = dyn_cast<MaterializeTemporaryExpr>(CCE->getArg(0))) {
818  // assert(false && "hit temporary copy path");
819  Expr *Arg = MTE->getSubExpr();
820  Path.push_back({IndirectLocalPathEntry::TemporaryCopy, Arg,
821  CCE->getConstructor()});
822  visitLocalsRetainedByInitializer(Path, Arg, Visit, true,
823  /*EnableLifetimeWarnings*/ false);
824  Path.pop_back();
825  }
826  }
827  }
828 
829  if (isa<CallExpr>(Init) || isa<CXXConstructExpr>(Init)) {
830  if (EnableLifetimeWarnings)
831  handleGslAnnotatedTypes(Path, Init, Visit);
832  return visitLifetimeBoundArguments(Path, Init, Visit);
833  }
834 
835  switch (Init->getStmtClass()) {
836  case Stmt::UnaryOperatorClass: {
837  auto *UO = cast<UnaryOperator>(Init);
838  // If the initializer is the address of a local, we could have a lifetime
839  // problem.
840  if (UO->getOpcode() == UO_AddrOf) {
841  // If this is &rvalue, then it's ill-formed and we have already diagnosed
842  // it. Don't produce a redundant warning about the lifetime of the
843  // temporary.
844  if (isa<MaterializeTemporaryExpr>(UO->getSubExpr()))
845  return;
846 
847  Path.push_back({IndirectLocalPathEntry::AddressOf, UO});
848  visitLocalsRetainedByReferenceBinding(Path, UO->getSubExpr(),
849  RK_ReferenceBinding, Visit,
850  EnableLifetimeWarnings);
851  }
852  break;
853  }
854 
855  case Stmt::BinaryOperatorClass: {
856  // Handle pointer arithmetic.
857  auto *BO = cast<BinaryOperator>(Init);
858  BinaryOperatorKind BOK = BO->getOpcode();
859  if (!BO->getType()->isPointerType() || (BOK != BO_Add && BOK != BO_Sub))
860  break;
861 
862  if (BO->getLHS()->getType()->isPointerType())
863  visitLocalsRetainedByInitializer(Path, BO->getLHS(), Visit, true,
864  EnableLifetimeWarnings);
865  else if (BO->getRHS()->getType()->isPointerType())
866  visitLocalsRetainedByInitializer(Path, BO->getRHS(), Visit, true,
867  EnableLifetimeWarnings);
868  break;
869  }
870 
871  case Stmt::ConditionalOperatorClass:
872  case Stmt::BinaryConditionalOperatorClass: {
873  auto *C = cast<AbstractConditionalOperator>(Init);
874  // In C++, we can have a throw-expression operand, which has 'void' type
875  // and isn't interesting from a lifetime perspective.
876  if (!C->getTrueExpr()->getType()->isVoidType())
877  visitLocalsRetainedByInitializer(Path, C->getTrueExpr(), Visit, true,
878  EnableLifetimeWarnings);
879  if (!C->getFalseExpr()->getType()->isVoidType())
880  visitLocalsRetainedByInitializer(Path, C->getFalseExpr(), Visit, true,
881  EnableLifetimeWarnings);
882  break;
883  }
884 
885  case Stmt::BlockExprClass:
886  if (cast<BlockExpr>(Init)->getBlockDecl()->hasCaptures()) {
887  // This is a local block, whose lifetime is that of the function.
888  Visit(Path, Local(cast<BlockExpr>(Init)), RK_ReferenceBinding);
889  }
890  break;
891 
892  case Stmt::AddrLabelExprClass:
893  // We want to warn if the address of a label would escape the function.
894  Visit(Path, Local(cast<AddrLabelExpr>(Init)), RK_ReferenceBinding);
895  break;
896 
897  default:
898  break;
899  }
900 }

902 /// Whether a path to an object supports lifetime extension.
903 enum PathLifetimeKind {
904  /// Lifetime-extend along this path.
905  Extend,
906  /// We should lifetime-extend, but we don't because (due to technical
907  /// limitations) we can't. This happens for default member initializers,
908  /// which we don't clone for every use, so we don't have a unique
909  /// MaterializeTemporaryExpr to update.
910  ShouldExtend,
911  /// Do not lifetime extend along this path.
912  NoExtend
913 };

915 /// Determine whether this is an indirect path to a temporary that we are
916 /// supposed to lifetime-extend along.
917 static PathLifetimeKind
918 shouldLifetimeExtendThroughPath(const IndirectLocalPath &Path) {
919  PathLifetimeKind Kind = PathLifetimeKind::Extend;
920  for (auto Elem : Path) {
921  if (Elem.Kind == IndirectLocalPathEntry::DefaultInit)
922  Kind = PathLifetimeKind::ShouldExtend;
923  else if (Elem.Kind != IndirectLocalPathEntry::LambdaCaptureInit)
924  return PathLifetimeKind::NoExtend;
925  }
926  return Kind;
927 }

929 /// Find the range for the first interesting entry in the path at or after I.
930 static SourceRange nextPathEntryRange(const IndirectLocalPath &Path, unsigned I,
931  Expr *E) {
932  for (unsigned N = Path.size(); I != N; ++I) {
933  switch (Path[I].Kind) {
934  case IndirectLocalPathEntry::AddressOf:
935  case IndirectLocalPathEntry::LValToRVal:
936  case IndirectLocalPathEntry::LifetimeBoundCall:
937  case IndirectLocalPathEntry::TemporaryCopy:
938  case IndirectLocalPathEntry::GslReferenceInit:
939  case IndirectLocalPathEntry::GslPointerInit:
940  // These exist primarily to mark the path as not permitting or
941  // supporting lifetime extension.
942  break;
943 
944  case IndirectLocalPathEntry::VarInit:
945  if (cast<VarDecl>(Path[I].D)->isImplicit())
946  return SourceRange();
947  [[fallthrough]];
948  case IndirectLocalPathEntry::DefaultInit:
949  return Path[I].E->getSourceRange();
950 
951  case IndirectLocalPathEntry::LambdaCaptureInit:
952  if (!Path[I].Capture->capturesVariable())
953  continue;
954  return Path[I].E->getSourceRange();
955  }
956  }
957  return E->getSourceRange();
958 }

960 static bool pathOnlyInitializesGslPointer(IndirectLocalPath &Path) {
961  for (const auto &It : llvm::reverse(Path)) {
962  if (It.Kind == IndirectLocalPathEntry::VarInit)
963  continue;
964  if (It.Kind == IndirectLocalPathEntry::AddressOf)
965  continue;
966  if (It.Kind == IndirectLocalPathEntry::LifetimeBoundCall)
967  continue;
968  return It.Kind == IndirectLocalPathEntry::GslPointerInit ||
969  It.Kind == IndirectLocalPathEntry::GslReferenceInit;
970  }
971  return false;

974 static void checkExprLifetimeImpl(Sema &SemaRef,
975  const InitializedEntity *InitEntity,
976  const InitializedEntity *ExtendingEntity,
977  LifetimeKind LK,
978  const AssignedEntity *AEntity, Expr *Init) {
979  assert((AEntity && LK == LK_Assignment) ||
980  (InitEntity && LK != LK_Assignment));
981  // If this entity doesn't have an interesting lifetime, don't bother looking
982  // for temporaries within its initializer.
983  if (LK == LK_FullExpression)
984  return;
985 
986  // FIXME: consider moving the TemporaryVisitor and visitLocalsRetained*
987  // functions to a dedicated class.
988  auto TemporaryVisitor = [&](IndirectLocalPath &Path, Local L,
989  ReferenceKind RK) -> bool {
990  SourceRange DiagRange = nextPathEntryRange(Path, 0, L);
991  SourceLocation DiagLoc = DiagRange.getBegin();
992 
993  auto *MTE = dyn_cast<MaterializeTemporaryExpr>(L);
994 
995  bool IsGslPtrInitWithGslTempOwner = false;
996  bool IsLocalGslOwner = false;
997  if (pathOnlyInitializesGslPointer(Path)) {
998  if (isa<DeclRefExpr>(L)) {
999  // We do not want to follow the references when returning a pointer
1000  // originating from a local owner to avoid the following false positive:
1001  // int &p = *localUniquePtr;
1002  // someContainer.add(std::move(localUniquePtr));
1003  // return p;
1004  IsLocalGslOwner = isRecordWithAttr<OwnerAttr>(L->getType());
1005  if (pathContainsInit(Path) || !IsLocalGslOwner)
1006  return false;
1007  } else {
1008  IsGslPtrInitWithGslTempOwner =
1009  MTE && !MTE->getExtendingDecl() &&
1010  isRecordWithAttr<OwnerAttr>(MTE->getType());
1011  // Skipping a chain of initializing gsl::Pointer annotated objects.
1012  // We are looking only for the final source to find out if it was
1013  // a local or temporary owner or the address of a local variable/param.
1014  if (!IsGslPtrInitWithGslTempOwner)
1015  return true;
1016  }
1017  }
1018 
1019  switch (LK) {
1020  case LK_FullExpression:
1021  llvm_unreachable("already handled this");
1022 
1023  case LK_Extended: {
1024  if (!MTE) {
1025  // The initialized entity has lifetime beyond the full-expression,
1026  // and the local entity does too, so don't warn.
1027  //
1028  // FIXME: We should consider warning if a static / thread storage
1029  // duration variable retains an automatic storage duration local.
1030  return false;
1031  }
1032 
1033  if (IsGslPtrInitWithGslTempOwner && DiagLoc.isValid()) {
1034  SemaRef.Diag(DiagLoc, diag::warn_dangling_lifetime_pointer)
1035  << DiagRange;
1036  return false;
1037  }
1038 
1039  switch (shouldLifetimeExtendThroughPath(Path)) {
1040  case PathLifetimeKind::Extend:
1041  // Update the storage duration of the materialized temporary.
1042  // FIXME: Rebuild the expression instead of mutating it.
1043  MTE->setExtendingDecl(ExtendingEntity->getDecl(),
1044  ExtendingEntity->allocateManglingNumber());
1045  // Also visit the temporaries lifetime-extended by this initializer.
1046  return true;
1047 
1048  case PathLifetimeKind::ShouldExtend:
1049  // We're supposed to lifetime-extend the temporary along this path (per
1050  // the resolution of DR1815), but we don't support that yet.
1051  //
1052  // FIXME: Properly handle this situation. Perhaps the easiest approach
1053  // would be to clone the initializer expression on each use that would
1054  // lifetime extend its temporaries.
1055  SemaRef.Diag(DiagLoc, diag::warn_unsupported_lifetime_extension)
1056  << RK << DiagRange;
1057  break;
1058 
1059  case PathLifetimeKind::NoExtend:
1060  // If the path goes through the initialization of a variable or field,
1061  // it can't possibly reach a temporary created in this full-expression.
1062  // We will have already diagnosed any problems with the initializer.
1063  if (pathContainsInit(Path))
1064  return false;
1065 
1066  SemaRef.Diag(DiagLoc, diag::warn_dangling_variable)
1067  << RK << !InitEntity->getParent()
1068  << ExtendingEntity->getDecl()->isImplicit()
1069  << ExtendingEntity->getDecl() << Init->isGLValue() << DiagRange;
1070  break;
1071  }
1072  break;
1073  }
1074 
1075  case LK_Assignment: {
1076  if (!MTE)
1077  return false;
1078  assert(shouldLifetimeExtendThroughPath(Path) ==
1079  PathLifetimeKind::NoExtend &&
1080  "No lifetime extension for assignments");
1081  if (!pathContainsInit(Path))
1082  SemaRef.Diag(DiagLoc, diag::warn_dangling_pointer_assignment)
1083  << AEntity->LHS << DiagRange;
1084  return false;
1085  }
1086  case LK_MemInitializer: {
1087  if (MTE) {
1088  // Under C++ DR1696, if a mem-initializer (or a default member
1089  // initializer used by the absence of one) would lifetime-extend a
1090  // temporary, the program is ill-formed.
1091  if (auto *ExtendingDecl =
1092  ExtendingEntity ? ExtendingEntity->getDecl() : nullptr) {
1093  if (IsGslPtrInitWithGslTempOwner) {
1094  SemaRef.Diag(DiagLoc, diag::warn_dangling_lifetime_pointer_member)
1095  << ExtendingDecl << DiagRange;
1096  SemaRef.Diag(ExtendingDecl->getLocation(),
1097  diag::note_ref_or_ptr_member_declared_here)
1098  << true;
1099  return false;
1100  }
1101  bool IsSubobjectMember = ExtendingEntity != InitEntity;
1102  SemaRef.Diag(DiagLoc, shouldLifetimeExtendThroughPath(Path) !=
1103  PathLifetimeKind::NoExtend
1104  ? diag::err_dangling_member
1105  : diag::warn_dangling_member)
1106  << ExtendingDecl << IsSubobjectMember << RK << DiagRange;
1107  // Don't bother adding a note pointing to the field if we're inside
1108  // its default member initializer; our primary diagnostic points to
1109  // the same place in that case.
1110  if (Path.empty() ||
1111  Path.back().Kind != IndirectLocalPathEntry::DefaultInit) {
1112  SemaRef.Diag(ExtendingDecl->getLocation(),
1113  diag::note_lifetime_extending_member_declared_here)
1114  << RK << IsSubobjectMember;
1115  }
1116  } else {
1117  // We have a mem-initializer but no particular field within it; this
1118  // is either a base class or a delegating initializer directly
1119  // initializing the base-class from something that doesn't live long
1120  // enough.
1121  //
1122  // FIXME: Warn on this.
1123  return false;
1124  }
1125  } else {
1126  // Paths via a default initializer can only occur during error recovery
1127  // (there's no other way that a default initializer can refer to a
1128  // local). Don't produce a bogus warning on those cases.
1129  if (pathContainsInit(Path))
1130  return false;
1131 
1132  // Suppress false positives for code like the one below:
1133  // Ctor(unique_ptr<T> up) : member(*up), member2(move(up)) {}
1134  if (IsLocalGslOwner && pathOnlyInitializesGslPointer(Path))
1135  return false;
1136 
1137  auto *DRE = dyn_cast<DeclRefExpr>(L);
1138  auto *VD = DRE ? dyn_cast<VarDecl>(DRE->getDecl()) : nullptr;
1139  if (!VD) {
1140  // A member was initialized to a local block.
1141  // FIXME: Warn on this.
1142  return false;
1143  }
1144 
1145  if (auto *Member =
1146  ExtendingEntity ? ExtendingEntity->getDecl() : nullptr) {
1147  bool IsPointer = !Member->getType()->isReferenceType();
1148  SemaRef.Diag(DiagLoc,
1149  IsPointer ? diag::warn_init_ptr_member_to_parameter_addr
1150  : diag::warn_bind_ref_member_to_parameter)
1151  << Member << VD << isa<ParmVarDecl>(VD) << DiagRange;
1152  SemaRef.Diag(Member->getLocation(),
1153  diag::note_ref_or_ptr_member_declared_here)
1154  << (unsigned)IsPointer;
1155  }
1156  }
1157  break;
1158  }
1159 
1160  case LK_New:
1161  if (isa<MaterializeTemporaryExpr>(L)) {
1162  if (IsGslPtrInitWithGslTempOwner)
1163  SemaRef.Diag(DiagLoc, diag::warn_dangling_lifetime_pointer)
1164  << DiagRange;
1165  else
1166  SemaRef.Diag(DiagLoc, RK == RK_ReferenceBinding
1167  ? diag::warn_new_dangling_reference
1168  : diag::warn_new_dangling_initializer_list)
1169  << !InitEntity->getParent() << DiagRange;
1170  } else {
1171  // We can't determine if the allocation outlives the local declaration.
1172  return false;
1173  }
1174  break;
1175 
1176  case LK_Return:
1177  case LK_StmtExprResult:
1178  if (auto *DRE = dyn_cast<DeclRefExpr>(L)) {
1179  // We can't determine if the local variable outlives the statement
1180  // expression.
1181  if (LK == LK_StmtExprResult)
1182  return false;
1183  SemaRef.Diag(DiagLoc, diag::warn_ret_stack_addr_ref)
1184  << InitEntity->getType()->isReferenceType() << DRE->getDecl()
1185  << isa<ParmVarDecl>(DRE->getDecl()) << DiagRange;
1186  } else if (isa<BlockExpr>(L)) {
1187  SemaRef.Diag(DiagLoc, diag::err_ret_local_block) << DiagRange;
1188  } else if (isa<AddrLabelExpr>(L)) {
1189  // Don't warn when returning a label from a statement expression.
1190  // Leaving the scope doesn't end its lifetime.
1191  if (LK == LK_StmtExprResult)
1192  return false;
1193  SemaRef.Diag(DiagLoc, diag::warn_ret_addr_label) << DiagRange;
1194  } else if (auto *CLE = dyn_cast<CompoundLiteralExpr>(L)) {
1195  SemaRef.Diag(DiagLoc, diag::warn_ret_stack_addr_ref)
1196  << InitEntity->getType()->isReferenceType() << CLE->getInitializer()
1197  << 2 << DiagRange;
1198  } else {
1199  // P2748R5: Disallow Binding a Returned Glvalue to a Temporary.
1200  // [stmt.return]/p6: In a function whose return type is a reference,
1201  // other than an invented function for std::is_convertible ([meta.rel]),
1202  // a return statement that binds the returned reference to a temporary
1203  // expression ([class.temporary]) is ill-formed.
1204  if (SemaRef.getLangOpts().CPlusPlus26 &&
