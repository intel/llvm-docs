38 static bool shouldSkipSanitizeOption(const ToolChain &TC,
39  const llvm::opt::ArgList &DriverArgs,
40  StringRef TargetID,
41  const llvm::opt::Arg *A) {
42  // For actions without targetID, do nothing.
43  if (TargetID.empty())
44  return false;
45  Option O = A->getOption();
46  if (!O.matches(options::OPT_fsanitize_EQ))
47  return false;
48 
49  if (!DriverArgs.hasFlag(options::OPT_fgpu_sanitize,
50  options::OPT_fno_gpu_sanitize, true))
51  return true;
52 
53  auto &Diags = TC.getDriver().getDiags();
54 
55  // For simplicity, we only allow -fsanitize=address
56  SanitizerMask K = parseSanitizerValue(A->getValue(), /*AllowGroups=*/false);
57  if (K != SanitizerKind::Address)
58  return true;
59 
60  llvm::StringMap<bool> FeatureMap;
61  auto OptionalGpuArch = parseTargetID(TC.getTriple(), TargetID, &FeatureMap);
62 
63  assert(OptionalGpuArch && "Invalid Target ID");
64  (void)OptionalGpuArch;
65  auto Loc = FeatureMap.find("xnack");
66  if (Loc == FeatureMap.end() || !Loc->second) {
67  Diags.Report(
68  clang::diag::warn_drv_unsupported_option_for_offload_arch_req_feature)
69  << A->getAsString(DriverArgs) << TargetID << "xnack+";
70  return true;
71  }
72  return false;
73 }

75 void AMDGCN::Linker::constructLldCommand(Compilation &C, const JobAction &JA,
76  const InputInfoList &Inputs,
77  const InputInfo &Output,
78  const llvm::opt::ArgList &Args) const {
79  // Construct lld command.
80  // The output from ld.lld is an HSA code object file.
81  ArgStringList LldArgs{"-flavor",
82  "gnu",
83  "--no-undefined",
84  "-shared",
85  "-plugin-opt=-amdgpu-internalize-symbols",
86  "-plugin-opt=-sycl-enable-local-accessor"};
87 
88  auto &TC = getToolChain();
89  auto &D = TC.getDriver();
90  assert(!Inputs.empty() && "Must have at least one input.");
91  bool IsThinLTO = D.getLTOMode(/*IsOffload=*/true) == LTOK_Thin;
92  addLTOOptions(TC, Args, LldArgs, Output, Inputs[0], IsThinLTO);
93 
94  // Extract all the -m options
95  std::vector<llvm::StringRef> Features;
96  amdgpu::getAMDGPUTargetFeatures(D, TC.getTriple(), Args, Features);
97 
98  // Add features to mattr such as cumode
99  std::string MAttrString = "-plugin-opt=-mattr=";
100  for (auto OneFeature : unifyTargetFeatures(Features)) {
101  MAttrString.append(Args.MakeArgString(OneFeature));
102  if (OneFeature != Features.back())
103  MAttrString.append(",");
104  }
105  if (!Features.empty())
106  LldArgs.push_back(Args.MakeArgString(MAttrString));
107 
108  // ToDo: Remove this option after AMDGPU backend supports ISA-level linking.
109  // Since AMDGPU backend currently does not support ISA-level linking, all
110  // called functions need to be imported.
111  if (IsThinLTO)
112  LldArgs.push_back(Args.MakeArgString("-plugin-opt=-force-import-all"));
113 
114  for (const Arg *A : Args.filtered(options::OPT_mllvm)) {
115  LldArgs.push_back(
116  Args.MakeArgString(Twine("-plugin-opt=") + A->getValue(0)));
117  }
118 
119  if (C.getDriver().isSaveTempsEnabled())
120  LldArgs.push_back("-save-temps");
121 
122  addLinkerCompressDebugSectionsOption(TC, Args, LldArgs);
123 
124  LldArgs.append({"-o", Output.getFilename()});
125  for (auto Input : Inputs)
126  LldArgs.push_back(Input.getFilename());
127 
128  // Look for archive of bundled bitcode in arguments, and add temporary files
129  // for the extracted archive of bitcode to inputs.
130  auto TargetID = Args.getLastArgValue(options::OPT_mcpu_EQ);
131  AddStaticDeviceLibsLinking(C, *this, JA, Inputs, Args, LldArgs, "amdgcn",
132  TargetID,
133  /*IsBitCodeSDL=*/true,
134  /*PostClangLink=*/false);
135 
136  const char *Lld = Args.MakeArgString(getToolChain().GetProgramPath("lld"));
137  C.addCommand(std::make_unique<Command>(JA, *this, ResponseFileSupport::None(),
138  Lld, LldArgs, Inputs, Output));
139 }

143 void AMDGCN::Linker::ConstructJob(Compilation &C, const JobAction &JA,
144  const InputInfo &Output,
145  const InputInfoList &Inputs,
146  const ArgList &Args,
147  const char *LinkingOutput) const {
148  if (Inputs.size() > 0 &&
149  Inputs[0].getType() == types::TY_Image &&
150  JA.getType() == types::TY_Object)

157 
158  return constructLldCommand(C, JA, Inputs, Output, Args);
159 }

161 HIPAMDToolChain::HIPAMDToolChain(const Driver &D, const llvm::Triple &Triple,
162  const ToolChain &HostTC, const ArgList &Args,
163  const Action::OffloadKind OK)
164  : ROCMToolChain(D, Triple, Args), HostTC(HostTC), OK(OK) {
165  // Lookup binaries into the driver directory, this is used to
166  // discover the clang-offload-bundler executable.
167  getProgramPaths().push_back(getDriver().Dir);
168 
169  // Diagnose unsupported sanitizer options only once.
170  if (!Args.hasFlag(options::OPT_fgpu_sanitize, options::OPT_fno_gpu_sanitize,
171  true))
172  return;
173  for (auto A : Args.filtered(options::OPT_fsanitize_EQ)) {
174  SanitizerMask K = parseSanitizerValue(A->getValue(), /*AllowGroups=*/false);
175  if (K != SanitizerKind::Address)
176  D.getDiags().Report(clang::diag::warn_drv_unsupported_option_for_target)
177  << A->getAsString(Args) << getTriple().str();
178  }
179 }

185 void HIPAMDToolChain::addClangTargetOptions(
186  const llvm::opt::ArgList &DriverArgs, llvm::opt::ArgStringList &CC1Args,
187  Action::OffloadKind DeviceOffloadingKind) const {
188  HostTC.addClangTargetOptions(DriverArgs, CC1Args, DeviceOffloadingKind);
189 
190  assert((DeviceOffloadingKind == Action::OFK_HIP ||
191  DeviceOffloadingKind == Action::OFK_SYCL) &&
192  "Only HIP and SYCL offloading kinds are supported for GPUs.");
193 
194  CC1Args.push_back("-fcuda-is-device");
195 
196  if (DriverArgs.hasFlag(options::OPT_fcuda_approx_transcendentals,
197  options::OPT_fno_cuda_approx_transcendentals, false))
198  CC1Args.push_back("-fcuda-approx-transcendentals");
199 
200  if (!DriverArgs.hasFlag(options::OPT_fgpu_rdc, options::OPT_fno_gpu_rdc,
201  false))
202  CC1Args.append({"-mllvm", "-amdgpu-internalize-symbols"});
203 
204  StringRef MaxThreadsPerBlock =
205  DriverArgs.getLastArgValue(options::OPT_gpu_max_threads_per_block_EQ);
206  if (!MaxThreadsPerBlock.empty()) {
207  std::string ArgStr =
208  std::string("--gpu-max-threads-per-block=") + MaxThreadsPerBlock.str();
209  CC1Args.push_back(DriverArgs.MakeArgStringRef(ArgStr));
210  }
211 
212  CC1Args.push_back("-fcuda-allow-variadic-functions");
213 
214  // Default to "hidden" visibility, as object level linking will not be
215  // supported for the foreseeable future.
216  if (!DriverArgs.hasArg(options::OPT_fvisibility_EQ,
217  options::OPT_fvisibility_ms_compat)) {
218  CC1Args.append({"-fvisibility", "hidden"});
219  CC1Args.push_back("-fapply-global-visibility-to-externs");
220  }
221 
222  if (DeviceOffloadingKind == Action::OFK_SYCL) {
223  toolchains::SYCLToolChain::AddSYCLIncludeArgs(getDriver(), DriverArgs,
224  CC1Args);
225  }
226 
227  auto NoLibSpirv = DriverArgs.hasArg(options::OPT_fno_sycl_libspirv,
228  options::OPT_fsycl_device_only);
229  if (DeviceOffloadingKind == Action::OFK_SYCL && !NoLibSpirv) {
230  std::string LibSpirvFile;
231 
232  if (DriverArgs.hasArg(clang::driver::options::OPT_fsycl_libspirv_path_EQ)) {
233  auto ProvidedPath =
234  DriverArgs
235  .getLastArgValue(
236  clang::driver::options::OPT_fsycl_libspirv_path_EQ)
237  .str();
238  if (llvm::sys::fs::exists(ProvidedPath))
239  LibSpirvFile = ProvidedPath;
240  } else {
241  SmallVector<StringRef, 8> LibraryPaths;
242 
243  // Expected path w/out install.
244  SmallString<256> WithoutInstallPath(getDriver().ResourceDir);
245  llvm::sys::path::append(WithoutInstallPath, Twine("../../clc"));
246  LibraryPaths.emplace_back(WithoutInstallPath.c_str());
247 
248  // Expected path w/ install.
249  SmallString<256> WithInstallPath(getDriver().ResourceDir);
250  llvm::sys::path::append(WithInstallPath, Twine("../../../share/clc"));
251  LibraryPaths.emplace_back(WithInstallPath.c_str());
252 
253  std::string LibSpirvTargetName = getLibSpirvTargetName(HostTC);
254  for (StringRef LibraryPath : LibraryPaths) {
255  SmallString<128> LibSpirvTargetFile(LibraryPath);
256  llvm::sys::path::append(LibSpirvTargetFile, LibSpirvTargetName);
257  if (llvm::sys::fs::exists(LibSpirvTargetFile)) {
258  LibSpirvFile = std::string(LibSpirvTargetFile.str());
259  break;
260  }
261  }
262  }
263 
264  if (LibSpirvFile.empty()) {
265  getDriver().Diag(diag::err_drv_no_sycl_libspirv)
266  << getLibSpirvTargetName(HostTC);
267  return;
268  }
269 
270  CC1Args.push_back("-mlink-builtin-bitcode");
271  CC1Args.push_back(DriverArgs.MakeArgString(LibSpirvFile));
272  }
273 
274  llvm::for_each(
275  getHIPDeviceLibs(DriverArgs, DeviceOffloadingKind), [&](auto BCFile) {
276  CC1Args.push_back(BCFile.ShouldInternalize ? "-mlink-builtin-bitcode"
277  : "-mlink-bitcode-file");
278  CC1Args.push_back(DriverArgs.MakeArgString(BCFile.Path));
279  });
280 }

282 llvm::opt::DerivedArgList *
283 HIPAMDToolChain::TranslateArgs(const llvm::opt::DerivedArgList &Args,
284  StringRef BoundArch,
285  Action::OffloadKind DeviceOffloadKind) const {
286  DerivedArgList *DAL =
287  HostTC.TranslateArgs(Args, BoundArch, DeviceOffloadKind);
288  if (!DAL)
289  DAL = new DerivedArgList(Args.getBaseArgs());
290 
291  const OptTable &Opts = getDriver().getOpts();
292 
293  for (Arg *A : Args) {
294  if (!shouldSkipArgument(A) &&
295  !shouldSkipSanitizeOption(*this, Args, BoundArch, A))
296  DAL->append(A);
297  }
298 
299  if (!BoundArch.empty()) {
300  DAL->eraseArg(options::OPT_mcpu_EQ);
301  DAL->AddJoinedArg(nullptr, Opts.getOption(options::OPT_mcpu_EQ), BoundArch);
302  checkTargetID(*DAL);
303  }
304 
305  return DAL;
306 }

355 SanitizerMask HIPAMDToolChain::getSupportedSanitizers() const {
356  // The HIPAMDToolChain only supports sanitizers in the sense that it allows
357  // sanitizer arguments on the command line if they are supported by the host
358  // toolchain. The HIPAMDToolChain will actually ignore any command line
359  // arguments for any of these "supported" sanitizers. That means that no
360  // sanitization of device code is actually supported at this time.
361  //
362  // This behavior is necessary because the host and device toolchains
363  // invocations often share the command line, so the device toolchain must
364  // tolerate flags meant only for the host toolchain.
365  return HostTC.getSupportedSanitizers();
366 }

373 llvm::SmallVector<ToolChain::BitCodeLibraryInfo, 12>
374 HIPAMDToolChain::getHIPDeviceLibs(
375  const llvm::opt::ArgList &DriverArgs,
376  const Action::OffloadKind DeviceOffloadingKind) const {
377  llvm::SmallVector<BitCodeLibraryInfo, 12> BCLibs;
378  if (DriverArgs.hasArg(options::OPT_nogpulib))
379  return {};
380  ArgStringList LibraryPaths;
381 
382  // Find in --hip-device-lib-path and HIP_LIBRARY_PATH.
383  for (auto Path : RocmInstallation.getRocmDeviceLibPathArg())
384  LibraryPaths.push_back(DriverArgs.MakeArgString(Path));
385 
386  addDirectoryList(DriverArgs, LibraryPaths, "", "HIP_DEVICE_LIB_PATH");
387 
388  // Maintain compatability with --hip-device-lib.
389  auto BCLibArgs = DriverArgs.getAllArgValues(options::OPT_hip_device_lib_EQ);
390  if (!BCLibArgs.empty()) {
391  llvm::for_each(BCLibArgs, [&](StringRef BCName) {
392  StringRef FullName;
393  for (std::string LibraryPath : LibraryPaths) {
394  SmallString<128> Path(LibraryPath);
395  llvm::sys::path::append(Path, BCName);
396  FullName = Path;
397  if (llvm::sys::fs::exists(FullName)) {
398  BCLibs.push_back(FullName);
399  return;
400  }
401  }
402  getDriver().Diag(diag::err_drv_no_such_file) << BCName;
403  });
404  } else {
405  if (!RocmInstallation.hasDeviceLibrary()) {
406  getDriver().Diag(diag::err_drv_no_rocm_device_lib) << 0;
407  return {};
408  }
409  StringRef GpuArch = getGPUArch(DriverArgs);
410  assert(!GpuArch.empty() && "Must have an explicit GPU arch.");
411 
412  // If --hip-device-lib is not set, add the default bitcode libraries.
413  if (DriverArgs.hasFlag(options::OPT_fgpu_sanitize,
414  options::OPT_fno_gpu_sanitize, true) &&
415  getSanitizerArgs(DriverArgs).needsAsanRt()) {
416  auto AsanRTL = RocmInstallation.getAsanRTLPath();
417  if (AsanRTL.empty()) {
418  unsigned DiagID = getDriver().getDiags().getCustomDiagID(
419  DiagnosticsEngine::Error,
420  "AMDGPU address sanitizer runtime library (asanrtl) is not found. "
421  "Please install ROCm device library which supports address "
422  "sanitizer");
423  getDriver().Diag(DiagID);
424  return {};
425  } else
426  BCLibs.push_back({AsanRTL.str(), /*ShouldInternalize=*/false});
427  }
428 
429  // Add the HIP specific bitcode library.
430  BCLibs.push_back(RocmInstallation.getHIPPath());
431 
432  // Add common device libraries like ocml etc.
433  for (auto N : getCommonDeviceLibNames(DriverArgs, GpuArch.str(),
434  DeviceOffloadingKind))
435  BCLibs.push_back(StringRef(N));
436 
437  // Add instrument lib.
438  auto InstLib =
439  DriverArgs.getLastArgValue(options::OPT_gpu_instrument_lib_EQ);
440  if (InstLib.empty())
441  return BCLibs;
442  if (llvm::sys::fs::exists(InstLib))
443  BCLibs.push_back(InstLib);
444  else
445  getDriver().Diag(diag::err_drv_no_such_file) << InstLib;
446  }
447 
448  return BCLibs;
449 }
