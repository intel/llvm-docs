131 class SemaPPCallbacks : public PPCallbacks {
132  Sema *S = nullptr;
133  llvm::SmallVector<SourceLocation, 8> IncludeStack;
134 
135 public:
136  void set(Sema &S) { this->S = &S; }
137 
138  void reset() { S = nullptr; }

140  virtual void FileChanged(SourceLocation Loc, FileChangeReason Reason,
141  SrcMgr::CharacteristicKind FileType,
142  FileID PrevFID) override {
143  if (!S)
144  return;
145  switch (Reason) {
146  case EnterFile: {
147  SourceManager &SM = S->getSourceManager();
148  SourceLocation IncludeLoc = SM.getIncludeLoc(SM.getFileID(Loc));
149  if (IncludeLoc.isValid()) {
150  if (llvm::timeTraceProfilerEnabled()) {
151  const FileEntry *FE = SM.getFileEntryForID(SM.getFileID(Loc));
152  llvm::timeTraceProfilerBegin(
153  "Source", FE != nullptr ? FE->getName() : StringRef("<unknown>"));
154  }
155 
156  IncludeStack.push_back(IncludeLoc);
157  S->DiagnoseNonDefaultPragmaAlignPack(
158  Sema::PragmaAlignPackDiagnoseKind::NonDefaultStateAtInclude,
159  IncludeLoc);
160  }
161  break;
162  }
163  case ExitFile:
164  if (!IncludeStack.empty()) {
165  if (llvm::timeTraceProfilerEnabled())
166  llvm::timeTraceProfilerEnd();
167 
168  S->DiagnoseNonDefaultPragmaAlignPack(
169  Sema::PragmaAlignPackDiagnoseKind::ChangedStateAtExit,
170  IncludeStack.pop_back_val());
171  }
172  break;
173  default:
174  break;
175  }
176  }
177 };

279  if (SemaConsumer *SC = dyn_cast<SemaConsumer>(&Consumer))
280  SC->InitializeSema(*this);
281 
282  // Tell the external Sema source about this Sema object.
283  if (ExternalSemaSource *ExternalSema
284  = dyn_cast_or_null<ExternalSemaSource>(Context.getExternalSource()))
285  ExternalSema->InitializeSema(*this);
286 
287  // This needs to happen after ExternalSemaSource::InitializeSema(this) or we
288  // will not be able to merge any duplicate __va_list_tag decls correctly.
289  VAListTagName = PP.getIdentifierInfo("__va_list_tag");
290 
291  if (!TUScope)
292  return;
293 
294  // Initialize predefined 128-bit integer types, if needed.
295  if (Context.getTargetInfo().hasInt128Type() ||
296  (Context.getAuxTargetInfo() &&
297  Context.getAuxTargetInfo()->hasInt128Type())) {
298  // If either of the 128-bit integer types are unavailable to name lookup,
299  // define them now.
300  DeclarationName Int128 = &Context.Idents.get("__int128_t");
301  if (IdResolver.begin(Int128) == IdResolver.end())
302  PushOnScopeChains(Context.getInt128Decl(), TUScope);
303 
304  DeclarationName UInt128 = &Context.Idents.get("__uint128_t");
305  if (IdResolver.begin(UInt128) == IdResolver.end())
306  PushOnScopeChains(Context.getUInt128Decl(), TUScope);
307  }
308 
309 
310  // Initialize predefined Objective-C types:
311  if (getLangOpts().ObjC) {
312  // If 'SEL' does not yet refer to any declarations, make it refer to the
313  // predefined 'SEL'.
314  DeclarationName SEL = &Context.Idents.get("SEL");
315  if (IdResolver.begin(SEL) == IdResolver.end())
316  PushOnScopeChains(Context.getObjCSelDecl(), TUScope);
317 
318  // If 'id' does not yet refer to any declarations, make it refer to the
319  // predefined 'id'.
320  DeclarationName Id = &Context.Idents.get("id");
321  if (IdResolver.begin(Id) == IdResolver.end())
322  PushOnScopeChains(Context.getObjCIdDecl(), TUScope);
323 
324  // Create the built-in typedef for 'Class'.
325  DeclarationName Class = &Context.Idents.get("Class");
326  if (IdResolver.begin(Class) == IdResolver.end())
327  PushOnScopeChains(Context.getObjCClassDecl(), TUScope);
328 
329  // Create the built-in forward declaratino for 'Protocol'.
330  DeclarationName Protocol = &Context.Idents.get("Protocol");
331  if (IdResolver.begin(Protocol) == IdResolver.end())
332  PushOnScopeChains(Context.getObjCProtocolDecl(), TUScope);
333  }
334 
335  // Create the internal type for the *StringMakeConstantString builtins.
336  DeclarationName ConstantString = &Context.Idents.get("__NSConstantString");
337  if (IdResolver.begin(ConstantString) == IdResolver.end())
338  PushOnScopeChains(Context.getCFConstantStringDecl(), TUScope);
339 
340  // Initialize Microsoft "predefined C++ types".
341  if (getLangOpts().MSVCCompat) {
342  if (getLangOpts().CPlusPlus &&
343  IdResolver.begin(&Context.Idents.get("type_info")) == IdResolver.end())
344  PushOnScopeChains(Context.buildImplicitRecord("type_info", TTK_Class),
345  TUScope);
346 
347  addImplicitTypedef("size_t", Context.getSizeType());
348  }
349  if (getLangOpts().SYCLIsDevice) {
350  addImplicitTypedef("__ocl_event_t", Context.OCLEventTy);
351  addImplicitTypedef("__ocl_sampler_t", Context.OCLSamplerTy);
352 #ifdef SEMA_STRINGIZE
353 #error "Undefine SEMA_STRINGIZE macro."
354 #endif
355 #define SEMA_STRINGIZE(s) #s
356 #define IMAGE_TYPE(ImgType, Id, SingletonId, Access, Suffix) \
357  addImplicitTypedef(SEMA_STRINGIZE(__ocl_##ImgType##_##Suffix##_t), \
358  Context.SingletonId);
359 #include "clang/Basic/OpenCLImageTypes.def"
360 #undef SEMA_STRINGIZE
361  }
362 
363  if (getLangOpts().SYCLIsDevice || getLangOpts().OpenCL) {
364 #ifdef SEMA_STRINGIZE
365 #error "Undefine SEMA_STRINGIZE macro."
366 #endif
367 #define SEMA_STRINGIZE(s) #s
368 #define IMAGE_TYPE(ImgType, Id, SingletonId, Access, Suffix) \
369  addImplicitTypedef(SEMA_STRINGIZE(__ocl_sampled_##ImgType##_##Suffix##_t), \
370  Context.Sampled##SingletonId);
371 #define IMAGE_WRITE_TYPE(Type, Id, Ext)
372 #define IMAGE_READ_WRITE_TYPE(Type, Id, Ext)
373 #include "clang/Basic/OpenCLImageTypes.def"
374 #undef SEMA_STRINGIZE
375  }
376 
377  // Initialize predefined OpenCL types and supported extensions and (optional)
378  // core features.
379  if (getLangOpts().OpenCL) {
380  getOpenCLOptions().addSupport(
381  Context.getTargetInfo().getSupportedOpenCLOpts(), getLangOpts());
382  addImplicitTypedef("sampler_t", Context.OCLSamplerTy);
383  addImplicitTypedef("event_t", Context.OCLEventTy);
384  auto OCLCompatibleVersion = getLangOpts().getOpenCLCompatibleVersion();
385  if (OCLCompatibleVersion >= 200) {
386  if (getLangOpts().OpenCLCPlusPlus || getLangOpts().Blocks) {
387  addImplicitTypedef("clk_event_t", Context.OCLClkEventTy);
388  addImplicitTypedef("queue_t", Context.OCLQueueTy);
389  }
390  if (getLangOpts().OpenCLPipes)
391  addImplicitTypedef("reserve_id_t", Context.OCLReserveIDTy);
392  addImplicitTypedef("atomic_int", Context.getAtomicType(Context.IntTy));
393  addImplicitTypedef("atomic_uint",
394  Context.getAtomicType(Context.UnsignedIntTy));
395  addImplicitTypedef("atomic_float",
396  Context.getAtomicType(Context.FloatTy));
397  // OpenCLC v2.0, s6.13.11.6 requires that atomic_flag is implemented as
398  // 32-bit integer and OpenCLC v2.0, s6.1.1 int is always 32-bit wide.
399  addImplicitTypedef("atomic_flag", Context.getAtomicType(Context.IntTy));
400 
401 
402  // OpenCL v2.0 s6.13.11.6:
403  // - The atomic_long and atomic_ulong types are supported if the
404  // cl_khr_int64_base_atomics and cl_khr_int64_extended_atomics
405  // extensions are supported.
406  // - The atomic_double type is only supported if double precision
407  // is supported and the cl_khr_int64_base_atomics and
408  // cl_khr_int64_extended_atomics extensions are supported.
409  // - If the device address space is 64-bits, the data types

428  }

442  addImplicitTypedef("atomic_double", AtomicDoubleT);

446  auto AtomicULongT = Context.getAtomicType(Context.UnsignedLongTy);

448  addImplicitTypedef("atomic_ulong", AtomicULongT);
449 
450 
451  if (Context.getTypeSize(Context.getSizeType()) == 64) {

453  }
454  }
455  }

501 
502  // Kill all the active scopes.
503  for (sema::FunctionScopeInfo *FSI : FunctionScopes)
504  delete FSI;
505 
506  // Tell the SemaConsumer to forget about us; we're going out of scope.
507  if (SemaConsumer *SC = dyn_cast<SemaConsumer>(&Consumer))
508  SC->ForgetSema();
509 
510  // Detach from the external Sema source.

514 
515  // If Sema's ExternalSource is the multiplexer - we own it.
516  if (isMultiplexExternalSource)
517  delete ExternalSource;
518 
519  // Delete cached satisfactions.
520  std::vector<ConstraintSatisfaction *> Satisfactions;
521  Satisfactions.reserve(Satisfactions.size());
522  for (auto &Node : SatisfactionCache)
523  Satisfactions.push_back(&Node);
524  for (auto *Node : Satisfactions)
525  delete Node;
526 
527  threadSafety::threadSafetyCleanup(ThreadSafetyDeclCache);
528 
529  // Destroys data sharing attributes stack for OpenMP
530  DestroyDataSharingAttributesStack();
531 
532  // Detach from the PP callback handler which outlives Sema since it's owned
533  // by the preprocessor.
534  SemaPPCallbackHandler->reset();
535 }

756 /// Used to prune the decls of Sema's UnusedFileScopedDecls vector.
757 static bool ShouldRemoveFromUnused(Sema *SemaRef, const DeclaratorDecl *D) {
758  if (D->getMostRecentDecl()->isUsed())
759  return true;
760 
761  if (D->isExternallyVisible())
762  return true;
763 
764  if (const FunctionDecl *FD = dyn_cast<FunctionDecl>(D)) {
765  // If this is a function template and none of its specializations is used,
766  // we should warn.
767  if (FunctionTemplateDecl *Template = FD->getDescribedFunctionTemplate())
768  for (const auto *Spec : Template->specializations())
769  if (ShouldRemoveFromUnused(SemaRef, Spec))
770  return true;
771 
772  // UnusedFileScopedDecls stores the first declaration.
773  // The declaration may have become definition so check again.
774  const FunctionDecl *DeclToCheck;
775  if (FD->hasBody(DeclToCheck))
776  return !SemaRef->ShouldWarnIfUnusedFileScopedDecl(DeclToCheck);
777 
778  // Later redecls may add new information resulting in not having to warn,
779  // so check again.
780  DeclToCheck = FD->getMostRecentDecl();
781  if (DeclToCheck != FD)
782  return !SemaRef->ShouldWarnIfUnusedFileScopedDecl(DeclToCheck);
783  }
784 
785  if (const VarDecl *VD = dyn_cast<VarDecl>(D)) {
786  // If a variable usable in constant expressions is referenced,
787  // don't warn if it isn't used: if the value of a variable is required
788  // for the computation of a constant expression, it doesn't make sense to
789  // warn even if the variable isn't odr-used. (isReferenced doesn't
790  // precisely reflect that, but it's a decent approximation.)
791  if (VD->isReferenced() &&
792  VD->mightBeUsableInConstantExpressions(SemaRef->Context))
793  return true;
794 
795  if (VarTemplateDecl *Template = VD->getDescribedVarTemplate())
796  // If this is a variable template and none of its specializations is used,
797  // we should warn.
798  for (const auto *Spec : Template->specializations())
799  if (ShouldRemoveFromUnused(SemaRef, Spec))
800  return true;
801 
802  // UnusedFileScopedDecls stores the first declaration.
803  // The declaration may have become definition so check again.
804  const VarDecl *DeclToCheck = VD->getDefinition();
805  if (DeclToCheck)
806  return !SemaRef->ShouldWarnIfUnusedFileScopedDecl(DeclToCheck);
807 
808  // Later redecls may add new information resulting in not having to warn,
809  // so check again.
810  DeclToCheck = VD->getMostRecentDecl();
811  if (DeclToCheck != VD)
812  return !SemaRef->ShouldWarnIfUnusedFileScopedDecl(DeclToCheck);
813  }
814 
815  return false;
816 }

835 /// Obtains a sorted list of functions and variables that are undefined but
836 /// ODR-used.
837 void Sema::getUndefinedButUsed(
838  SmallVectorImpl<std::pair<NamedDecl *, SourceLocation> > &Undefined) {
839  for (const auto &UndefinedUse : UndefinedButUsed) {
840  NamedDecl *ND = UndefinedUse.first;
841 
842  // Ignore attributes that have become invalid.
843  if (ND->isInvalidDecl()) continue;
844 
845  // __attribute__((weakref)) is basically a definition.
846  if (ND->hasAttr<WeakRefAttr>()) continue;
847 
848  if (isa<CXXDeductionGuideDecl>(ND))
849  continue;
850 
851  if (ND->hasAttr<DLLImportAttr>() || ND->hasAttr<DLLExportAttr>()) {
852  // An exported function will always be emitted when defined, so even if
853  // the function is inline, it doesn't have to be emitted in this TU. An
854  // imported function implies that it has been exported somewhere else.
855  continue;
856  }
857 
858  if (FunctionDecl *FD = dyn_cast<FunctionDecl>(ND)) {
859  if (FD->isDefined())
860  continue;
861  if (FD->isExternallyVisible() &&
862  !isExternalWithNoLinkageType(FD) &&
863  !FD->getMostRecentDecl()->isInlined() &&
864  !FD->hasAttr<ExcludeFromExplicitInstantiationAttr>())
865  continue;
866  if (FD->getBuiltinID())
867  continue;
868  } else {
869  auto *VD = cast<VarDecl>(ND);
870  if (VD->hasDefinition() != VarDecl::DeclarationOnly)
871  continue;
872  if (VD->isExternallyVisible() &&
873  !isExternalWithNoLinkageType(VD) &&
874  !VD->getMostRecentDecl()->isInline() &&
875  !VD->hasAttr<ExcludeFromExplicitInstantiationAttr>())
876  continue;
877 
878  // Skip VarDecls that lack formal definitions but which we know are in
879  // fact defined somewhere.
880  if (VD->isKnownToBeDefined())
881  continue;
882  }
883 
884  Undefined.push_back(std::make_pair(ND, UndefinedUse.second));
885  }

888 /// checkUndefinedButUsed - Check for undefined objects with internal linkage
889 /// or that are inline.
890 static void checkUndefinedButUsed(Sema &S) {
891  if (S.UndefinedButUsed.empty()) return;
892 
893  // Collect all the still-undefined entities with internal linkage.
894  SmallVector<std::pair<NamedDecl *, SourceLocation>, 16> Undefined;
895  S.getUndefinedButUsed(Undefined);
896  if (Undefined.empty()) return;
897 
898  for (auto Undef : Undefined) {
899  ValueDecl *VD = cast<ValueDecl>(Undef.first);
900  SourceLocation UseLoc = Undef.second;
901 
902  if (S.isExternalWithNoLinkageType(VD)) {
903  // C++ [basic.link]p8:
904  // A type without linkage shall not be used as the type of a variable
905  // or function with external linkage unless
906  // -- the entity has C language linkage
907  // -- the entity is not odr-used or is defined in the same TU
908  //
909  // As an extension, accept this in cases where the type is externally
910  // visible, since the function or variable actually can be defined in
911  // another translation unit in that case.
912  S.Diag(VD->getLocation(), isExternallyVisible(VD->getType()->getLinkage())
913  ? diag::ext_undefined_internal_type
914  : diag::err_undefined_internal_type)
915  << isa<VarDecl>(VD) << VD;
916  } else if (!VD->isExternallyVisible()) {
917  // FIXME: We can promote this to an error. The function or variable can't
918  // be defined anywhere else, so the program must necessarily violate the
919  // one definition rule.
920  bool IsImplicitBase = false;
921  if (const auto *BaseD = dyn_cast<FunctionDecl>(VD)) {
922  auto *DVAttr = BaseD->getAttr<OMPDeclareVariantAttr>();
923  if (DVAttr && !DVAttr->getTraitInfo().isExtensionActive(
924  llvm::omp::TraitProperty::
925  implementation_extension_disable_implicit_base)) {
926  const auto *Func = cast<FunctionDecl>(
927  cast<DeclRefExpr>(DVAttr->getVariantFuncRef())->getDecl());
928  IsImplicitBase = BaseD->isImplicit() &&
929  Func->getIdentifier()->isMangledOpenMPVariantName();
930  }
931  }
932  if (!S.getLangOpts().OpenMP || !IsImplicitBase)
933  S.Diag(VD->getLocation(), diag::warn_undefined_internal)
934  << isa<VarDecl>(VD) << VD;
935  } else if (auto *FD = dyn_cast<FunctionDecl>(VD)) {
936  (void)FD;
937  assert(FD->getMostRecentDecl()->isInlined() &&
938  "used object requires definition but isn't inline or internal?");
939  // FIXME: This is ill-formed; we should reject.
940  S.Diag(VD->getLocation(), diag::warn_undefined_inline) << VD;
941  } else {
942  assert(cast<VarDecl>(VD)->getMostRecentDecl()->isInline() &&
943  "used var requires definition but isn't inline or internal?");
944  S.Diag(VD->getLocation(), diag::err_undefined_inline_var) << VD;
945  }
946  if (UseLoc.isValid())
947  S.Diag(UseLoc, diag::note_used_here);
948  }
949 
950  S.UndefinedButUsed.clear();
951 }

966 /// Returns true, if all methods and nested classes of the given
967 /// CXXRecordDecl are defined in this translation unit.
968 ///
969 /// Should only be called from ActOnEndOfTranslationUnit so that all
970 /// definitions are actually read.
971 static bool MethodsAndNestedClassesComplete(const CXXRecordDecl *RD,
972  RecordCompleteMap &MNCComplete) {
973  RecordCompleteMap::iterator Cache = MNCComplete.find(RD);
974  if (Cache != MNCComplete.end())
975  return Cache->second;
976  if (!RD->isCompleteDefinition())
977  return false;
978  bool Complete = true;
979  for (DeclContext::decl_iterator I = RD->decls_begin(),
980  E = RD->decls_end();
981  I != E && Complete; ++I) {
982  if (const CXXMethodDecl *M = dyn_cast<CXXMethodDecl>(*I))
983  Complete = M->isDefined() || M->isDefaulted() ||
984  (M->isPure() && !isa<CXXDestructorDecl>(M));
985  else if (const FunctionTemplateDecl *F = dyn_cast<FunctionTemplateDecl>(*I))
986  // If the template function is marked as late template parsed at this
987  // point, it has not been instantiated and therefore we have not
988  // performed semantic analysis on it yet, so we cannot know if the type
989  // can be considered complete.
990  Complete = !F->getTemplatedDecl()->isLateTemplateParsed() &&
991  F->getTemplatedDecl()->isDefined();
992  else if (const CXXRecordDecl *R = dyn_cast<CXXRecordDecl>(*I)) {
993  if (R->isInjectedClassName())
994  continue;
995  if (R->hasDefinition())
996  Complete = MethodsAndNestedClassesComplete(R->getDefinition(),
997  MNCComplete);
998  else
999  Complete = false;
1000  }
1001  }
1002  MNCComplete[RD] = Complete;
1003  return Complete;
1004 }

1006 /// Returns true, if the given CXXRecordDecl is fully defined in this
1007 /// translation unit, i.e. all methods are defined or pure virtual and all
1008 /// friends, friend functions and nested classes are fully defined in this
1009 /// translation unit.
1010 ///
1011 /// Should only be called from ActOnEndOfTranslationUnit so that all
1012 /// definitions are actually read.
1013 static bool IsRecordFullyDefined(const CXXRecordDecl *RD,
1014  RecordCompleteMap &RecordsComplete,
1015  RecordCompleteMap &MNCComplete) {
1016  RecordCompleteMap::iterator Cache = RecordsComplete.find(RD);
1017  if (Cache != RecordsComplete.end())
1018  return Cache->second;
1019  bool Complete = MethodsAndNestedClassesComplete(RD, MNCComplete);
1020  for (CXXRecordDecl::friend_iterator I = RD->friend_begin(),
1021  E = RD->friend_end();
1022  I != E && Complete; ++I) {
1023  // Check if friend classes and methods are complete.
1024  if (TypeSourceInfo *TSI = (*I)->getFriendType()) {
1025  // Friend classes are available as the TypeSourceInfo of the FriendDecl.
1026  if (CXXRecordDecl *FriendD = TSI->getType()->getAsCXXRecordDecl())
1027  Complete = MethodsAndNestedClassesComplete(FriendD, MNCComplete);
1028  else
1029  Complete = false;
1030  } else {
1031  // Friend functions are available through the NamedDecl of FriendDecl.
1032  if (const FunctionDecl *FD =
1033  dyn_cast<FunctionDecl>((*I)->getFriendDecl()))
1034  Complete = FD->isDefined();
1035  else
1036  // This is a template friend, give up.
1037  Complete = false;
1038  }
1039  }
1040  RecordsComplete[RD] = Complete;
1041  return Complete;
1042 }

1086  // transfer these over, even though they are not parsed. The end of the TU

1105  // instantiation will always happen at the end of the translation unit,
1106  // so it will find some names that are not required to be found. This is

1655 /// Helper class that emits deferred diagnostic messages if an entity directly
1656 /// or indirectly using the function that causes the deferred diagnostic
1657 /// messages is known to be emitted.
1658 ///
1659 /// During parsing of AST, certain diagnostic messages are recorded as deferred
1660 /// diagnostics since it is unknown whether the functions containing such
1661 /// diagnostics will be emitted. A list of potentially emitted functions and
1662 /// variables that may potentially trigger emission of functions are also
1663 /// recorded. DeferredDiagnosticsEmitter recursively visits used functions
1664 /// by each function to emit deferred diagnostics.
1665 ///
1666 /// During the visit, certain OpenMP directives or initializer of variables
1667 /// with certain OpenMP attributes will cause subsequent visiting of any
1668 /// functions enter a state which is called OpenMP device context in this
1669 /// implementation. The state is exited when the directive or initializer is
1670 /// exited. This state can change the emission states of subsequent uses
1671 /// of functions.
1672 ///
1673 /// Conceptually the functions or variables to be visited form a use graph
1674 /// where the parent node uses the child node. At any point of the visit,
1675 /// the tree nodes traversed from the tree root to the current node form a use
1676 /// stack. The emission state of the current node depends on two factors:
1677 /// 1. the emission state of the root node
1678 /// 2. whether the current node is in OpenMP device context
1679 /// If the function is decided to be emitted, its contained deferred diagnostics
1680 /// are emitted, together with the information about the use stack.
1681 ///
1682 class DeferredDiagnosticsEmitter
1683  : public UsedDeclVisitor<DeferredDiagnosticsEmitter> {
1684 public:
1685  typedef UsedDeclVisitor<DeferredDiagnosticsEmitter> Inherited;
1686 
1687  // Whether the function is already in the current use-path.
1688  llvm::SmallPtrSet<CanonicalDeclPtr<Decl>, 4> InUsePath;
1689 
1690  // The current use-path.
1691  llvm::SmallVector<CanonicalDeclPtr<FunctionDecl>, 4> UsePath;
1692 
1693  // Whether the visiting of the function has been done. Done[0] is for the
1694  // case not in OpenMP device context. Done[1] is for the case in OpenMP
1695  // device context. We need two sets because diagnostics emission may be
1696  // different depending on whether it is in OpenMP device context.
1697  llvm::SmallPtrSet<CanonicalDeclPtr<Decl>, 4> DoneMap[2];
1698 
1699  // Emission state of the root node of the current use graph.
1700  bool ShouldEmitRootNode;
1701 
1702  Sema::DeviceDiagnosticReason RootReason = Sema::DeviceDiagnosticReason::All;
1703 
1704  // Current OpenMP device context level. It is initialized to 0 and each
1705  // entering of device context increases it by 1 and each exit decreases
1706  // it by 1. Non-zero value indicates it is currently in device context.
1707  unsigned InOMPDeviceContext;
1708 
1709  DeferredDiagnosticsEmitter(Sema &S)
1710  : Inherited(S), ShouldEmitRootNode(false), InOMPDeviceContext(0) {}
1711 
1712  bool shouldVisitDiscardedStmt() const { return false; }

1871 // In CUDA, there are some constructs which may appear in semantically-valid
1872 // code, but trigger errors if we ever generate code for the function in which
1873 // they appear. Essentially every construct you're not allowed to use on the
1874 // device falls into this category, because you are allowed to use these
1875 // constructs in a __host__ __device__ function, but only if that function is
1876 // never codegen'ed on the device.
1877 //
1878 // To handle semantic checking for these constructs, we keep track of the set of
1879 // functions we know will be emitted, either because we could tell a priori that
1880 // they would be emitted, or because they were transitively called by a
1881 // known-emitted function.
1882 //
1883 // We also keep a partial call graph of which not-known-emitted functions call
1884 // which other not-known-emitted functions.
1885 //
1886 // When we see something which is illegal if the current function is emitted
1887 // (usually by way of CUDADiagIfDeviceCode, CUDADiagIfHostCode, or
1888 // CheckCUDACall), we first check if the current function is known-emitted. If
1889 // so, we immediately output the diagnostic.
1890 //
1891 // Otherwise, we "defer" the diagnostic. It sits in Sema::DeviceDeferredDiags
1892 // until we discover that the function is known-emitted, at which point we take
1893 // it out of this map and emit the diagnostic.
