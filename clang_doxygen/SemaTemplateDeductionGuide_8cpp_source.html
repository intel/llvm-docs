85  /// Returns true if it's safe to substitute \p Typedef with
86  /// \p OuterInstantiationArgs.
87  bool mightReferToOuterTemplateParameters(TypedefNameDecl *Typedef) {
88  if (!NestedPattern)
89  return false;
90 
91  static auto WalkUp = [](DeclContext *DC, DeclContext *TargetDC) {
92  if (DC->Equals(TargetDC))
93  return true;
94  while (DC->isRecord()) {
95  if (DC->Equals(TargetDC))
96  return true;
97  DC = DC->getParent();
98  }
99  return false;
100  };
101 
102  if (WalkUp(Typedef->getDeclContext(), NestedPattern->getTemplatedDecl()))
103  return true;
104  if (WalkUp(NestedPattern->getTemplatedDecl(), Typedef->getDeclContext()))
105  return true;
106  return false;
107  }

109  QualType
110  RebuildTemplateSpecializationType(TemplateName Template,
111  SourceLocation TemplateNameLoc,
112  TemplateArgumentListInfo &TemplateArgs) {
113  if (!OuterInstantiationArgs ||
114  !isa_and_present<TypeAliasTemplateDecl>(Template.getAsTemplateDecl()))
115  return Base::RebuildTemplateSpecializationType(Template, TemplateNameLoc,
116  TemplateArgs);
117 
118  auto *TATD = cast<TypeAliasTemplateDecl>(Template.getAsTemplateDecl());
119  auto *Pattern = TATD;
120  while (Pattern->getInstantiatedFromMemberTemplate())
121  Pattern = Pattern->getInstantiatedFromMemberTemplate();
122  if (!mightReferToOuterTemplateParameters(Pattern->getTemplatedDecl()))
123  return Base::RebuildTemplateSpecializationType(Template, TemplateNameLoc,
124  TemplateArgs);
125 
126  Decl *NewD =
127  TypedefNameInstantiator->InstantiateTypeAliasTemplateDecl(TATD);
128  if (!NewD)
129  return QualType();
130 
131  auto *NewTATD = cast<TypeAliasTemplateDecl>(NewD);
132  MaterializedTypedefs.push_back(NewTATD->getTemplatedDecl());
133 
134  return Base::RebuildTemplateSpecializationType(
135  TemplateName(NewTATD), TemplateNameLoc, TemplateArgs);
136  }

138  QualType TransformTypedefType(TypeLocBuilder &TLB, TypedefTypeLoc TL) {
139  ASTContext &Context = SemaRef.getASTContext();
140  TypedefNameDecl *OrigDecl = TL.getTypedefNameDecl();
141  TypedefNameDecl *Decl = OrigDecl;
142  // Transform the underlying type of the typedef and clone the Decl only if
143  // the typedef has a dependent context.
144  bool InDependentContext = OrigDecl->getDeclContext()->isDependentContext();
145 
146  // A typedef/alias Decl within the NestedPattern may reference the outer
147  // template parameters. They're substituted with corresponding instantiation
148  // arguments here and in RebuildTemplateSpecializationType() above.
149  // Otherwise, we would have a CTAD guide with "dangling" template
150  // parameters.
151  // For example,
152  // template <class T> struct Outer {
153  // using Alias = S<T>;
154  // template <class U> struct Inner {
155  // Inner(Alias);
156  // };
157  // };
158  if (OuterInstantiationArgs && InDependentContext &&
159  TL.getTypePtr()->isInstantiationDependentType()) {
160  Decl = cast_if_present<TypedefNameDecl>(
161  TypedefNameInstantiator->InstantiateTypedefNameDecl(
162  OrigDecl, /*IsTypeAlias=*/isa<TypeAliasDecl>(OrigDecl)));
163  if (!Decl)
164  return QualType();
165  MaterializedTypedefs.push_back(Decl);
166  } else if (InDependentContext) {
167  TypeLocBuilder InnerTLB;
168  QualType Transformed =
169  TransformType(InnerTLB, OrigDecl->getTypeSourceInfo()->getTypeLoc());
170  TypeSourceInfo *TSI = InnerTLB.getTypeSourceInfo(Context, Transformed);
171  if (isa<TypeAliasDecl>(OrigDecl))
172  Decl = TypeAliasDecl::Create(
173  Context, Context.getTranslationUnitDecl(), OrigDecl->getBeginLoc(),
174  OrigDecl->getLocation(), OrigDecl->getIdentifier(), TSI);
175  else {
176  assert(isa<TypedefDecl>(OrigDecl) && "Not a Type alias or typedef");
177  Decl = TypedefDecl::Create(
178  Context, Context.getTranslationUnitDecl(), OrigDecl->getBeginLoc(),
179  OrigDecl->getLocation(), OrigDecl->getIdentifier(), TSI);
180  }
181  MaterializedTypedefs.push_back(Decl);
182  }
183 
184  QualType TDTy = Context.getTypedefType(Decl);
185  TypedefTypeLoc TypedefTL = TLB.push<TypedefTypeLoc>(TDTy);
186  TypedefTL.setNameLoc(TL.getNameLoc());
187 
188  return TDTy;
189  }

192 // Build a deduction guide using the provided information.
193 //
194 // A deduction guide can be either a template or a non-template function
195 // declaration. If \p TemplateParams is null, a non-template function
196 // declaration will be created.
197 NamedDecl *buildDeductionGuide(
198  Sema &SemaRef, TemplateDecl *OriginalTemplate,
199  TemplateParameterList *TemplateParams, CXXConstructorDecl *Ctor,
200  ExplicitSpecifier ES, TypeSourceInfo *TInfo, SourceLocation LocStart,
201  SourceLocation Loc, SourceLocation LocEnd, bool IsImplicit,
202  llvm::ArrayRef<TypedefNameDecl *> MaterializedTypedefs = {}) {
203  DeclContext *DC = OriginalTemplate->getDeclContext();
204  auto DeductionGuideName =
205  SemaRef.Context.DeclarationNames.getCXXDeductionGuideName(
206  OriginalTemplate);
207 
208  DeclarationNameInfo Name(DeductionGuideName, Loc);
209  ArrayRef<ParmVarDecl *> Params =
210  TInfo->getTypeLoc().castAs<FunctionProtoTypeLoc>().getParams();
211 
212  // Build the implicit deduction guide template.
213  auto *Guide =
214  CXXDeductionGuideDecl::Create(SemaRef.Context, DC, LocStart, ES, Name,
215  TInfo->getType(), TInfo, LocEnd, Ctor);
216  Guide->setImplicit(IsImplicit);
217  Guide->setParams(Params);
218 
219  for (auto *Param : Params)
220  Param->setDeclContext(Guide);
221  for (auto *TD : MaterializedTypedefs)
222  TD->setDeclContext(Guide);
223  if (isa<CXXRecordDecl>(DC))
224  Guide->setAccess(AS_public);
225 
226  if (!TemplateParams) {
227  DC->addDecl(Guide);
228  return Guide;
229  }
230 
231  auto *GuideTemplate = FunctionTemplateDecl::Create(
232  SemaRef.Context, DC, Loc, DeductionGuideName, TemplateParams, Guide);
233  GuideTemplate->setImplicit(IsImplicit);
234  Guide->setDescribedFunctionTemplate(GuideTemplate);
235 
236  if (isa<CXXRecordDecl>(DC))
237  GuideTemplate->setAccess(AS_public);
238 
239  DC->addDecl(GuideTemplate);
240  return GuideTemplate;
241 }

252  SemaRef.Context, DC, TTP->getBeginLoc(), TTP->getLocation(), NewDepth,

269  return NewTTP;

287 /// Transform to convert portions of a constructor declaration into the
288 /// corresponding deduction guide, per C++1z [over.match.class.deduct]p1.
289 struct ConvertConstructorToDeductionGuideTransform {
290  ConvertConstructorToDeductionGuideTransform(Sema &S,
291  ClassTemplateDecl *Template)
292  : SemaRef(S), Template(Template) {
293  // If the template is nested, then we need to use the original
294  // pattern to iterate over the constructors.
295  ClassTemplateDecl *Pattern = Template;
296  while (Pattern->getInstantiatedFromMemberTemplate()) {
297  if (Pattern->isMemberSpecialization())
298  break;
299  Pattern = Pattern->getInstantiatedFromMemberTemplate();
300  NestedPattern = Pattern;
301  }
302 
303  if (NestedPattern)
304  OuterInstantiationArgs = SemaRef.getTemplateInstantiationArgs(Template);
305  }
306 
307  Sema &SemaRef;
308  ClassTemplateDecl *Template;
309  ClassTemplateDecl *NestedPattern = nullptr;
310 
311  DeclContext *DC = Template->getDeclContext();
312  CXXRecordDecl *Primary = Template->getTemplatedDecl();
313  DeclarationName DeductionGuideName =
314  SemaRef.Context.DeclarationNames.getCXXDeductionGuideName(Template);
315 
316  QualType DeducedType = SemaRef.Context.getTypeDeclType(Primary);
317 
318  // Index adjustment to apply to convert depth-1 template parameters into
319  // depth-0 template parameters.
320  unsigned Depth1IndexAdjustment = Template->getTemplateParameters()->size();
321 
322  // Instantiation arguments for the outermost depth-1 templates
323  // when the template is nested
324  MultiLevelTemplateArgumentList OuterInstantiationArgs;
325 
326  /// Transform a constructor declaration into a deduction guide.
327  NamedDecl *transformConstructor(FunctionTemplateDecl *FTD,
328  CXXConstructorDecl *CD) {
329  SmallVector<TemplateArgument, 16> SubstArgs;
330 
331  LocalInstantiationScope Scope(SemaRef);
332 
333  // C++ [over.match.class.deduct]p1:
334  // -- For each constructor of the class template designated by the
335  // template-name, a function template with the following properties:
336 
337  // -- The template parameters are the template parameters of the class
338  // template followed by the template parameters (including default
339  // template arguments) of the constructor, if any.
340  TemplateParameterList *TemplateParams =
341  SemaRef.GetTemplateParameterList(Template);
342  if (FTD) {
343  TemplateParameterList *InnerParams = FTD->getTemplateParameters();
344  SmallVector<NamedDecl *, 16> AllParams;
345  SmallVector<TemplateArgument, 16> Depth1Args;
346  AllParams.reserve(TemplateParams->size() + InnerParams->size());
347  AllParams.insert(AllParams.begin(), TemplateParams->begin(),
348  TemplateParams->end());
349  SubstArgs.reserve(InnerParams->size());
350  Depth1Args.reserve(InnerParams->size());
351 
352  // Later template parameters could refer to earlier ones, so build up
353  // a list of substituted template arguments as we go.
354  for (NamedDecl *Param : *InnerParams) {
355  MultiLevelTemplateArgumentList Args;
356  Args.setKind(TemplateSubstitutionKind::Rewrite);
357  Args.addOuterTemplateArguments(Depth1Args);
358  Args.addOuterRetainedLevel();
359  if (NestedPattern)
360  Args.addOuterRetainedLevels(NestedPattern->getTemplateDepth());
361  NamedDecl *NewParam = transformTemplateParameter(Param, Args);
362  if (!NewParam)
363  return nullptr;
364  // Constraints require that we substitute depth-1 arguments
365  // to match depths when substituted for evaluation later
366  Depth1Args.push_back(SemaRef.Context.getCanonicalTemplateArgument(
367  SemaRef.Context.getInjectedTemplateArg(NewParam)));
368 
369  if (NestedPattern) {
370  TemplateDeclInstantiator Instantiator(SemaRef, DC,
371  OuterInstantiationArgs);
372  Instantiator.setEvaluateConstraints(false);
373  SemaRef.runWithSufficientStackSpace(NewParam->getLocation(), [&] {
374  NewParam = cast<NamedDecl>(Instantiator.Visit(NewParam));
375  });
376  }
377 
378  assert(NewParam->getTemplateDepth() == 0 &&
379  "Unexpected template parameter depth");
380 
381  AllParams.push_back(NewParam);
382  SubstArgs.push_back(SemaRef.Context.getCanonicalTemplateArgument(
383  SemaRef.Context.getInjectedTemplateArg(NewParam)));
384  }
385 
386  // Substitute new template parameters into requires-clause if present.
387  Expr *RequiresClause = nullptr;
388  if (Expr *InnerRC = InnerParams->getRequiresClause()) {
389  MultiLevelTemplateArgumentList Args;
390  Args.setKind(TemplateSubstitutionKind::Rewrite);
391  Args.addOuterTemplateArguments(Depth1Args);
392  Args.addOuterRetainedLevel();
393  if (NestedPattern)
394  Args.addOuterRetainedLevels(NestedPattern->getTemplateDepth());
395  ExprResult E = SemaRef.SubstExpr(InnerRC, Args);
396  if (E.isInvalid())
397  return nullptr;
398  RequiresClause = E.getAs<Expr>();
399  }
400 
401  TemplateParams = TemplateParameterList::Create(
402  SemaRef.Context, InnerParams->getTemplateLoc(),
403  InnerParams->getLAngleLoc(), AllParams, InnerParams->getRAngleLoc(),
404  RequiresClause);
405  }
406 
407  // If we built a new template-parameter-list, track that we need to
408  // substitute references to the old parameters into references to the
409  // new ones.
410  MultiLevelTemplateArgumentList Args;
411  Args.setKind(TemplateSubstitutionKind::Rewrite);
412  if (FTD) {
413  Args.addOuterTemplateArguments(SubstArgs);
414  Args.addOuterRetainedLevel();
415  }
416 
417  FunctionProtoTypeLoc FPTL = CD->getTypeSourceInfo()
418  ->getTypeLoc()
419  .getAsAdjusted<FunctionProtoTypeLoc>();
420  assert(FPTL && "no prototype for constructor declaration");
421 
422  // Transform the type of the function, adjusting the return type and
423  // replacing references to the old parameters with references to the
424  // new ones.
425  TypeLocBuilder TLB;
426  SmallVector<ParmVarDecl *, 8> Params;
427  SmallVector<TypedefNameDecl *, 4> MaterializedTypedefs;
428  QualType NewType = transformFunctionProtoType(TLB, FPTL, Params, Args,
429  MaterializedTypedefs);
430  if (NewType.isNull())
431  return nullptr;
432  TypeSourceInfo *NewTInfo = TLB.getTypeSourceInfo(SemaRef.Context, NewType);
433 
434  return buildDeductionGuide(
435  SemaRef, Template, TemplateParams, CD, CD->getExplicitSpecifier(),
436  NewTInfo, CD->getBeginLoc(), CD->getLocation(), CD->getEndLoc(),
437  /*IsImplicit=*/true, MaterializedTypedefs);
438  }

441  NamedDecl *buildSimpleDeductionGuide(MutableArrayRef<QualType> ParamTypes) {
442  SourceLocation Loc = Template->getLocation();
443 
444  // Build the requested type.
445  FunctionProtoType::ExtProtoInfo EPI;

486  /// renumbering as we go.
487  NamedDecl *transformTemplateParameter(NamedDecl *TemplateParam,

503  QualType transformFunctionProtoType(
504  TypeLocBuilder &TLB, FunctionProtoTypeLoc TL,
505  SmallVectorImpl<ParmVarDecl *> &Params,
506  MultiLevelTemplateArgumentList &Args,
507  SmallVectorImpl<TypedefNameDecl *> &MaterializedTypedefs) {
508  SmallVector<QualType, 4> ParamTypes;
509  const FunctionProtoType *T = TL.getTypePtr();
510 
511  // -- The types of the function parameters are those of the constructor.
512  for (auto *OldParam : TL.getParams()) {
513  ParmVarDecl *NewParam = OldParam;
514  // Given
515  // template <class T> struct C {
516  // template <class U> struct D {
517  // template <class V> D(U, V);
518  // };
519  // };
520  // First, transform all the references to template parameters that are
521  // defined outside of the surrounding class template. That is T in the
522  // above example.
523  if (NestedPattern) {
524  NewParam = transformFunctionTypeParam(
525  NewParam, OuterInstantiationArgs, MaterializedTypedefs,
526  /*TransformingOuterPatterns=*/true);
527  if (!NewParam)
528  return QualType();
529  }
530  // Then, transform all the references to template parameters that are
531  // defined at the class template and the constructor. In this example,
532  // they're U and V, respectively.
533  NewParam =
534  transformFunctionTypeParam(NewParam, Args, MaterializedTypedefs,
535  /*TransformingOuterPatterns=*/false);
536  if (!NewParam)
537  return QualType();
538  ParamTypes.push_back(NewParam->getType());
539  Params.push_back(NewParam);
540  }
541 
542  // -- The return type is the class template specialization designated by
543  // the template-name and template arguments corresponding to the
544  // template parameters obtained from the class template.
545  //
546  // We use the injected-class-name type of the primary template instead.
547  // This has the convenient property that it is different from any type that
548  // the user can write in a deduction-guide (because they cannot enter the
549  // context of the template), so implicit deduction guides can never collide
550  // with explicit ones.
551  QualType ReturnType = DeducedType;
552  TLB.pushTypeSpec(ReturnType).setNameLoc(Primary->getLocation());
553 
554  // Resolving a wording defect, we also inherit the variadicness of the
555  // constructor.
556  FunctionProtoType::ExtProtoInfo EPI;
557  EPI.Variadic = T->isVariadic();
558  EPI.HasTrailingReturn = true;
559 
560  QualType Result = SemaRef.BuildFunctionType(
561  ReturnType, ParamTypes, TL.getBeginLoc(), DeductionGuideName, EPI);
562  if (Result.isNull())
563  return QualType();
564 
565  FunctionProtoTypeLoc NewTL = TLB.push<FunctionProtoTypeLoc>(Result);
566  NewTL.setLocalRangeBegin(TL.getLocalRangeBegin());
567  NewTL.setLParenLoc(TL.getLParenLoc());
568  NewTL.setRParenLoc(TL.getRParenLoc());
569  NewTL.setExceptionSpecRange(SourceRange());
570  NewTL.setLocalRangeEnd(TL.getLocalRangeEnd());
571  for (unsigned I = 0, E = NewTL.getNumParams(); I != E; ++I)
572  NewTL.setParam(I, Params[I]);
573 
574  return Result;
575  }

577  ParmVarDecl *transformFunctionTypeParam(
578  ParmVarDecl *OldParam, MultiLevelTemplateArgumentList &Args,
579  llvm::SmallVectorImpl<TypedefNameDecl *> &MaterializedTypedefs,
580  bool TransformingOuterPatterns) {
581  TypeSourceInfo *OldDI = OldParam->getTypeSourceInfo();
582  TypeSourceInfo *NewDI;
583  if (auto PackTL = OldDI->getTypeLoc().getAs<PackExpansionTypeLoc>()) {
584  // Expand out the one and only element in each inner pack.
585  Sema::ArgumentPackSubstitutionIndexRAII SubstIndex(SemaRef, 0);
586  NewDI =
587  SemaRef.SubstType(PackTL.getPatternLoc(), Args,
588  OldParam->getLocation(), OldParam->getDeclName());
589  if (!NewDI)
590  return nullptr;
591  NewDI =
592  SemaRef.CheckPackExpansion(NewDI, PackTL.getEllipsisLoc(),
593  PackTL.getTypePtr()->getNumExpansions());
594  } else

741 // Build the associated constraints for the alias deduction guides.
742 // C++ [over.match.class.deduct]p3.3:
743 // The associated constraints ([temp.constr.decl]) are the conjunction of the
744 // associated constraints of g and a constraint that is satisfied if and only
745 // if the arguments of A are deducible (see below) from the return type.
746 //
747 // The return result is expected to be the require-clause for the synthesized
748 // alias deduction guide.
749 Expr *
750 buildAssociatedConstraints(Sema &SemaRef, FunctionTemplateDecl *F,
751  TypeAliasTemplateDecl *AliasTemplate,
752  ArrayRef<DeducedTemplateArgument> DeduceResults,
753  unsigned FirstUndeducedParamIdx, Expr *IsDeducible) {
754  Expr *RC = F->getTemplateParameters()->getRequiresClause();
755  if (!RC)
756  return IsDeducible;
757 
758  ASTContext &Context = SemaRef.Context;
759  LocalInstantiationScope Scope(SemaRef);
760 
761  // In the clang AST, constraint nodes are deliberately not instantiated unless
762  // they are actively being evaluated. Consequently, occurrences of template
763  // parameters in the require-clause expression have a subtle "depth"
764  // difference compared to normal occurrences in places, such as function
765  // parameters. When transforming the require-clause, we must take this
766  // distinction into account:
767  //
768  // 1) In the transformed require-clause, occurrences of template parameters
769  // must use the "uninstantiated" depth;
770  // 2) When substituting on the require-clause expr of the underlying
771  // deduction guide, we must use the entire set of template argument lists;
772  //
773  // It's important to note that we're performing this transformation on an
774  // *instantiated* AliasTemplate.
775 
776  // For 1), if the alias template is nested within a class template, we
777  // calcualte the 'uninstantiated' depth by adding the substitution level back.
778  unsigned AdjustDepth = 0;
779  if (auto *PrimaryTemplate =
780  AliasTemplate->getInstantiatedFromMemberTemplate())
781  AdjustDepth = PrimaryTemplate->getTemplateDepth();
782 
783  // We rebuild all template parameters with the uninstantiated depth, and
784  // build template arguments refer to them.
785  SmallVector<TemplateArgument> AdjustedAliasTemplateArgs;
786 
787  for (auto *TP : *AliasTemplate->getTemplateParameters()) {
788  // Rebuild any internal references to earlier parameters and reindex
789  // as we go.
790  MultiLevelTemplateArgumentList Args;
791  Args.setKind(TemplateSubstitutionKind::Rewrite);
792  Args.addOuterTemplateArguments(AdjustedAliasTemplateArgs);
793  NamedDecl *NewParam = transformTemplateParameter(
794  SemaRef, AliasTemplate->getDeclContext(), TP, Args,
795  /*NewIndex=*/AdjustedAliasTemplateArgs.size(),
796  getTemplateParameterDepth(TP) + AdjustDepth);
797 
798  auto NewTemplateArgument = Context.getCanonicalTemplateArgument(
799  Context.getInjectedTemplateArg(NewParam));
800  AdjustedAliasTemplateArgs.push_back(NewTemplateArgument);
801  }
802  // Template arguments used to transform the template arguments in
803  // DeducedResults.
804  SmallVector<TemplateArgument> TemplateArgsForBuildingRC(
805  F->getTemplateParameters()->size());
806  // Transform the transformed template args
807  MultiLevelTemplateArgumentList Args;
808  Args.setKind(TemplateSubstitutionKind::Rewrite);
809  Args.addOuterTemplateArguments(AdjustedAliasTemplateArgs);
810 
811  for (unsigned Index = 0; Index < DeduceResults.size(); ++Index) {
812  const auto &D = DeduceResults[Index];
813  if (D.isNull()) { // non-deduced template parameters of f
814  NamedDecl *TP = F->getTemplateParameters()->getParam(Index);
815  MultiLevelTemplateArgumentList Args;
816  Args.setKind(TemplateSubstitutionKind::Rewrite);
817  Args.addOuterTemplateArguments(TemplateArgsForBuildingRC);
818  // Rebuild the template parameter with updated depth and index.
819  NamedDecl *NewParam = transformTemplateParameter(
820  SemaRef, F->getDeclContext(), TP, Args,
821  /*NewIndex=*/FirstUndeducedParamIdx,
822  getTemplateParameterDepth(TP) + AdjustDepth);
823  FirstUndeducedParamIdx += 1;
824  assert(TemplateArgsForBuildingRC[Index].isNull());
825  TemplateArgsForBuildingRC[Index] = Context.getCanonicalTemplateArgument(
826  Context.getInjectedTemplateArg(NewParam));
827  continue;
828  }
829  TemplateArgumentLoc Input =
830  SemaRef.getTrivialTemplateArgumentLoc(D, QualType(), SourceLocation{});
831  TemplateArgumentLoc Output;
832  if (!SemaRef.SubstTemplateArgument(Input, Args, Output)) {
833  assert(TemplateArgsForBuildingRC[Index].isNull() &&
834  "InstantiatedArgs must be null before setting");
835  TemplateArgsForBuildingRC[Index] = Output.getArgument();
836  }
837  }
838 
839  // A list of template arguments for transforming the require-clause of F.
840  // It must contain the entire set of template argument lists.
841  MultiLevelTemplateArgumentList ArgsForBuildingRC;
842  ArgsForBuildingRC.setKind(clang::TemplateSubstitutionKind::Rewrite);
843  ArgsForBuildingRC.addOuterTemplateArguments(TemplateArgsForBuildingRC);
844  // For 2), if the underlying deduction guide F is nested in a class template,
845  // we need the entire template argument list, as the constraint AST in the
846  // require-clause of F remains completely uninstantiated.
847  //
848  // For example:
849  // template <typename T> // depth 0
850  // struct Outer {
851  // template <typename U>
852  // struct Foo { Foo(U); };
853  //
854  // template <typename U> // depth 1
855  // requires C<U>
856  // Foo(U) -> Foo<int>;
857  // };
858  // template <typename U>
859  // using AFoo = Outer<int>::Foo<U>;
860  //
861  // In this scenario, the deduction guide for `Foo` inside `Outer<int>`:
862  // - The occurrence of U in the require-expression is [depth:1, index:0]
863  // - The occurrence of U in the function parameter is [depth:0, index:0]
864  // - The template parameter of U is [depth:0, index:0]
865  //
866  // We add the outer template arguments which is [int] to the multi-level arg
867  // list to ensure that the occurrence U in `C<U>` will be replaced with int
868  // during the substitution.
869  //
870  // NOTE: The underlying deduction guide F is instantiated -- either from an
871  // explicitly-written deduction guide member, or from a constructor.
872  // getInstantiatedFromMemberTemplate() can only handle the former case, so we
873  // check the DeclContext kind.
874  if (F->getLexicalDeclContext()->getDeclKind() ==
875  clang::Decl::ClassTemplateSpecialization) {
876  auto OuterLevelArgs = SemaRef.getTemplateInstantiationArgs(
877  F, F->getLexicalDeclContext(),
878  /*Final=*/false, /*Innermost=*/std::nullopt,
879  /*RelativeToPrimary=*/true,
880  /*Pattern=*/nullptr,
881  /*ForConstraintInstantiation=*/true);
882  for (auto It : OuterLevelArgs)
883  ArgsForBuildingRC.addOuterTemplateArguments(It.Args);
884  }
885 
886  ExprResult E = SemaRef.SubstExpr(RC, ArgsForBuildingRC);
887  if (E.isInvalid())
888  return nullptr;
889 
890  auto Conjunction =
891  SemaRef.BuildBinOp(SemaRef.getCurScope(), SourceLocation{},
892  BinaryOperatorKind::BO_LAnd, E.get(), IsDeducible);
893  if (Conjunction.isInvalid())
894  return nullptr;
895  return Conjunction.getAs<Expr>();
896 }

983 // Build deduction guides for a type alias template from the given underlying
984 // deduction guide F.
985 FunctionTemplateDecl *
986 BuildDeductionGuideForTypeAlias(Sema &SemaRef,
987  TypeAliasTemplateDecl *AliasTemplate,
988  FunctionTemplateDecl *F, SourceLocation Loc) {
989  LocalInstantiationScope Scope(SemaRef);
990  Sema::InstantiatingTemplate BuildingDeductionGuides(
991  SemaRef, AliasTemplate->getLocation(), F,
992  Sema::InstantiatingTemplate::BuildingDeductionGuidesTag{});
993  if (BuildingDeductionGuides.isInvalid())
994  return nullptr;
995 
996  auto &Context = SemaRef.Context;
997  auto [Template, AliasRhsTemplateArgs] =
998  getRHSTemplateDeclAndArgs(SemaRef, AliasTemplate);
999 
1000  auto RType = F->getTemplatedDecl()->getReturnType();
1001  // The (trailing) return type of the deduction guide.
1002  const TemplateSpecializationType *FReturnType =
1003  RType->getAs<TemplateSpecializationType>();
1004  if (const auto *InjectedCNT = RType->getAs<InjectedClassNameType>())
1005  // implicitly-generated deduction guide.
1006  FReturnType = InjectedCNT->getInjectedTST();
1007  else if (const auto *ET = RType->getAs<ElaboratedType>())
1008  // explicit deduction guide.
1009  FReturnType = ET->getNamedType()->getAs<TemplateSpecializationType>();
1010  assert(FReturnType && "expected to see a return type");
1011  // Deduce template arguments of the deduction guide f from the RHS of
1012  // the alias.
1013  //
1014  // C++ [over.match.class.deduct]p3: ...For each function or function
1015  // template f in the guides of the template named by the
1016  // simple-template-id of the defining-type-id, the template arguments
1017  // of the return type of f are deduced from the defining-type-id of A
1018  // according to the process in [temp.deduct.type] with the exception
1019  // that deduction does not fail if not all template arguments are
1020  // deduced.
1021  //
1022  //
1023  // template<typename X, typename Y>
1024  // f(X, Y) -> f<Y, X>;
1025  //
1026  // template<typename U>
1027  // using alias = f<int, U>;
1028  //
1029  // The RHS of alias is f<int, U>, we deduced the template arguments of
1030  // the return type of the deduction guide from it: Y->int, X->U
1031  sema::TemplateDeductionInfo TDeduceInfo(Loc);
1032  // Must initialize n elements, this is required by DeduceTemplateArguments.
1033  SmallVector<DeducedTemplateArgument> DeduceResults(
1034  F->getTemplateParameters()->size());
1035 
1036  // FIXME: DeduceTemplateArguments stops immediately at the first
1037  // non-deducible template argument. However, this doesn't seem to casue
1038  // issues for practice cases, we probably need to extend it to continue
1039  // performing deduction for rest of arguments to align with the C++
1040  // standard.
1041  SemaRef.DeduceTemplateArguments(
1042  F->getTemplateParameters(), FReturnType->template_arguments(),
1043  AliasRhsTemplateArgs, TDeduceInfo, DeduceResults,
1044  /*NumberOfArgumentsMustMatch=*/false);
1045 
1046  SmallVector<TemplateArgument> DeducedArgs;
1047  SmallVector<unsigned> NonDeducedTemplateParamsInFIndex;
1048  // !!NOTE: DeduceResults respects the sequence of template parameters of
1049  // the deduction guide f.
1050  for (unsigned Index = 0; Index < DeduceResults.size(); ++Index) {
1051  if (const auto &D = DeduceResults[Index]; !D.isNull()) // Deduced
1052  DeducedArgs.push_back(D);
1053  else
1054  NonDeducedTemplateParamsInFIndex.push_back(Index);
1055  }
1056  auto DeducedAliasTemplateParams =
1057  TemplateParamsReferencedInTemplateArgumentList(
1058  AliasTemplate->getTemplateParameters(), DeducedArgs);
1059  // All template arguments null by default.
1060  SmallVector<TemplateArgument> TemplateArgsForBuildingFPrime(
1061  F->getTemplateParameters()->size());
1062 
1063  // Create a template parameter list for the synthesized deduction guide f'.
1064  //
1065  // C++ [over.match.class.deduct]p3.2:
1066  // If f is a function template, f' is a function template whose template
1067  // parameter list consists of all the template parameters of A
1068  // (including their default template arguments) that appear in the above
1069  // deductions or (recursively) in their default template arguments
1070  SmallVector<NamedDecl *> FPrimeTemplateParams;
1071  // Store template arguments that refer to the newly-created template
1072  // parameters, used for building `TemplateArgsForBuildingFPrime`.
1073  SmallVector<TemplateArgument, 16> TransformedDeducedAliasArgs(
1074  AliasTemplate->getTemplateParameters()->size());
1075 
1076  for (unsigned AliasTemplateParamIdx : DeducedAliasTemplateParams) {
1077  auto *TP =
1078  AliasTemplate->getTemplateParameters()->getParam(AliasTemplateParamIdx);
1079  // Rebuild any internal references to earlier parameters and reindex as
1080  // we go.
1081  MultiLevelTemplateArgumentList Args;
1082  Args.setKind(TemplateSubstitutionKind::Rewrite);
1083  Args.addOuterTemplateArguments(TransformedDeducedAliasArgs);
1084  NamedDecl *NewParam = transformTemplateParameter(
1085  SemaRef, AliasTemplate->getDeclContext(), TP, Args,
1086  /*NewIndex=*/FPrimeTemplateParams.size(),
1087  getTemplateParameterDepth(TP));
1088  FPrimeTemplateParams.push_back(NewParam);
1089 
1090  auto NewTemplateArgument = Context.getCanonicalTemplateArgument(
1091  Context.getInjectedTemplateArg(NewParam));
1092  TransformedDeducedAliasArgs[AliasTemplateParamIdx] = NewTemplateArgument;
1093  }
1094  unsigned FirstUndeducedParamIdx = FPrimeTemplateParams.size();
1095  // ...followed by the template parameters of f that were not deduced
1096  // (including their default template arguments)
1097  for (unsigned FTemplateParamIdx : NonDeducedTemplateParamsInFIndex) {
1098  auto *TP = F->getTemplateParameters()->getParam(FTemplateParamIdx);
1099  MultiLevelTemplateArgumentList Args;
1100  Args.setKind(TemplateSubstitutionKind::Rewrite);
1101  // We take a shortcut here, it is ok to reuse the
1102  // TemplateArgsForBuildingFPrime.
1103  Args.addOuterTemplateArguments(TemplateArgsForBuildingFPrime);
1104  NamedDecl *NewParam = transformTemplateParameter(
1105  SemaRef, F->getDeclContext(), TP, Args, FPrimeTemplateParams.size(),
1106  getTemplateParameterDepth(TP));
1107  FPrimeTemplateParams.push_back(NewParam);
1108 
1109  assert(TemplateArgsForBuildingFPrime[FTemplateParamIdx].isNull() &&
1110  "The argument must be null before setting");
1111  TemplateArgsForBuildingFPrime[FTemplateParamIdx] =
1112  Context.getCanonicalTemplateArgument(
1113  Context.getInjectedTemplateArg(NewParam));
1114  }
1115 
1116  // To form a deduction guide f' from f, we leverage clang's instantiation
1117  // mechanism, we construct a template argument list where the template
1118  // arguments refer to the newly-created template parameters of f', and
1119  // then apply instantiation on this template argument list to instantiate
1120  // f, this ensures all template parameter occurrences are updated
1121  // correctly.
1122  //
1123  // The template argument list is formed from the `DeducedArgs`, two parts:
1124  // 1) appeared template parameters of alias: transfrom the deduced
1125  // template argument;
1126  // 2) non-deduced template parameters of f: rebuild a
1127  // template argument;
1128  //
1129  // 2) has been built already (when rebuilding the new template
1130  // parameters), we now perform 1).
1131  MultiLevelTemplateArgumentList Args;
1132  Args.setKind(TemplateSubstitutionKind::Rewrite);
1133  Args.addOuterTemplateArguments(TransformedDeducedAliasArgs);
1134  for (unsigned Index = 0; Index < DeduceResults.size(); ++Index) {
1135  const auto &D = DeduceResults[Index];
1136  if (D.isNull()) {
1137  // 2): Non-deduced template parameter has been built already.
1138  assert(!TemplateArgsForBuildingFPrime[Index].isNull() &&
1139  "template arguments for non-deduced template parameters should "
1140  "be been set!");
1141  continue;
1142  }
1143  TemplateArgumentLoc Input =
1144  SemaRef.getTrivialTemplateArgumentLoc(D, QualType(), SourceLocation{});
1145  TemplateArgumentLoc Output;
1146  if (!SemaRef.SubstTemplateArgument(Input, Args, Output)) {
1147  assert(TemplateArgsForBuildingFPrime[Index].isNull() &&
