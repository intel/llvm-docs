92 void tools::PS4cpu::Link::ConstructJob(Compilation &C, const JobAction &JA,
93  const InputInfo &Output,
94  const InputInfoList &Inputs,
95  const ArgList &Args,
96  const char *LinkingOutput) const {
97  const toolchains::FreeBSD &ToolChain =
98  static_cast<const toolchains::FreeBSD &>(getToolChain());
99  const Driver &D = ToolChain.getDriver();
100  ArgStringList CmdArgs;
101 
102  // Silence warning for "clang -g foo.o -o foo"
103  Args.ClaimAllArgs(options::OPT_g_Group);
104  // and "clang -emit-llvm foo.o -o foo"
105  Args.ClaimAllArgs(options::OPT_emit_llvm);
106  // and for "clang -w foo.o -o foo". Other warning options are already
107  // handled somewhere else.
108  Args.ClaimAllArgs(options::OPT_w);
109 
110  if (!D.SysRoot.empty())
111  CmdArgs.push_back(Args.MakeArgString("--sysroot=" + D.SysRoot));
112 
113  if (Args.hasArg(options::OPT_pie))
114  CmdArgs.push_back("-pie");
115 
116  if (Args.hasArg(options::OPT_rdynamic))
117  CmdArgs.push_back("-export-dynamic");
118  if (Args.hasArg(options::OPT_shared))
119  CmdArgs.push_back("--oformat=so");
120 
121  if (Output.isFilename()) {
122  CmdArgs.push_back("-o");
123  CmdArgs.push_back(Output.getFilename());
124  } else {
125  assert(Output.isNothing() && "Invalid output.");
126  }
127 
128  if(!Args.hasArg(options::OPT_nostdlib, options::OPT_nodefaultlibs))
129  AddPS4SanitizerArgs(ToolChain, Args, CmdArgs);
130 
131  Args.AddAllArgs(CmdArgs, options::OPT_L);
132  Args.AddAllArgs(CmdArgs, options::OPT_T_Group);
133  Args.AddAllArgs(CmdArgs, options::OPT_e);
134  Args.AddAllArgs(CmdArgs, options::OPT_s);
135  Args.AddAllArgs(CmdArgs, options::OPT_t);
136  Args.AddAllArgs(CmdArgs, options::OPT_r);
137 
138  if (Args.hasArg(options::OPT_Z_Xlinker__no_demangle))
139  CmdArgs.push_back("--no-demangle");
140 
141  AddLinkerInputs(ToolChain, Inputs, Args, CmdArgs, JA);
142 
143  if (Args.hasArg(options::OPT_pthread)) {
144  CmdArgs.push_back("-lpthread");
145  }
146 
147  if (Args.hasArg(options::OPT_fuse_ld_EQ)) {
148  D.Diag(diag::err_drv_unsupported_opt_for_target)
149  << "-fuse-ld" << getToolChain().getTriple().str();
150  }
151 
152  const char *Exec =
153  Args.MakeArgString(ToolChain.GetProgramPath("orbis-ld"));
154 
155  C.addCommand(std::make_unique<Command>(JA, *this,
156  ResponseFileSupport::AtFileUTF8(),
157  Exec, CmdArgs, Inputs, Output));
158 }

160 toolchains::PS4CPU::PS4CPU(const Driver &D, const llvm::Triple &Triple,
161  const ArgList &Args)
162  : Generic_ELF(D, Triple, Args) {
163  if (Args.hasArg(clang::driver::options::OPT_static))
164  D.Diag(clang::diag::err_drv_unsupported_opt_for_target) << "-static"
165  << "PS4";
166 
167  // Determine where to find the PS4 libraries. We use SCE_ORBIS_SDK_DIR
168  // if it exists; otherwise use the driver's installation path, which
169  // should be <SDK_DIR>/host_tools/bin.
170 
171  SmallString<512> PS4SDKDir;
172  if (const char *EnvValue = getenv("SCE_ORBIS_SDK_DIR")) {
173  if (!llvm::sys::fs::exists(EnvValue))
174  getDriver().Diag(clang::diag::warn_drv_ps4_sdk_dir) << EnvValue;
175  PS4SDKDir = EnvValue;
176  } else {
177  PS4SDKDir = getDriver().Dir;
178  llvm::sys::path::append(PS4SDKDir, "/../../");
179  }
180 
181  // By default, the driver won't report a warning if it can't find
182  // PS4's include or lib directories. This behavior could be changed if
183  // -Weverything or -Winvalid-or-nonexistent-directory options are passed.
184  // If -isysroot was passed, use that as the SDK base path.
185  std::string PrefixDir;
186  if (const Arg *A = Args.getLastArg(options::OPT_isysroot)) {
187  PrefixDir = A->getValue();
188  if (!llvm::sys::fs::exists(PrefixDir))
189  getDriver().Diag(clang::diag::warn_missing_sysroot) << PrefixDir;
190  } else
191  PrefixDir = std::string(PS4SDKDir.str());
192 
193  SmallString<512> PS4SDKIncludeDir(PrefixDir);
194  llvm::sys::path::append(PS4SDKIncludeDir, "target/include");
195  if (!Args.hasArg(options::OPT_nostdinc) &&
196  !Args.hasArg(options::OPT_nostdlibinc) &&
197  !Args.hasArg(options::OPT_isysroot) &&
198  !Args.hasArg(options::OPT__sysroot_EQ) &&
199  !llvm::sys::fs::exists(PS4SDKIncludeDir)) {
200  getDriver().Diag(clang::diag::warn_drv_unable_to_find_directory_expected)
201  << "PS4 system headers" << PS4SDKIncludeDir;
202  }
203 
204  SmallString<512> PS4SDKLibDir(PS4SDKDir);
205  llvm::sys::path::append(PS4SDKLibDir, "target/lib");
206  if (!Args.hasArg(options::OPT_nostdlib) &&
207  !Args.hasArg(options::OPT_nodefaultlibs) &&
208  !Args.hasArg(options::OPT__sysroot_EQ) && !Args.hasArg(options::OPT_E) &&
209  !Args.hasArg(options::OPT_c) && !Args.hasArg(options::OPT_S) &&
210  !Args.hasArg(options::OPT_emit_ast) &&
211  !llvm::sys::fs::exists(PS4SDKLibDir)) {
212  getDriver().Diag(clang::diag::warn_drv_unable_to_find_directory_expected)
213  << "PS4 system libraries" << PS4SDKLibDir;
214  return;
215  }
216  getFilePaths().push_back(std::string(PS4SDKLibDir.str()));
217 }

243  // PS4 does not use init arrays.

245  Arg *A = DriverArgs.getLastArg(options::OPT_fuse_init_array);
246  getDriver().Diag(clang::diag::err_drv_unsupported_opt_for_target)

249 
250  CC1Args.push_back("-fno-use-init-array");
251 
252  const Arg *A =

254  options::OPT_fno_visibility_from_dllstorageclass);
255  if (!A ||

257  CC1Args.push_back("-fvisibility-from-dllstorageclass");
258 
259  if (DriverArgs.hasArg(options::OPT_fvisibility_dllexport_EQ))
260  DriverArgs.AddLastArg(CC1Args, options::OPT_fvisibility_dllexport_EQ);
