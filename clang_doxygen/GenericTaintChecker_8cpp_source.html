83 /// Check if the region the expression evaluates to is the standard input,
84 /// and thus, is tainted.
85 /// FIXME: Move this to Taint.cpp.
86 bool isStdin(SVal Val, const ASTContext &ACtx) {
87  // FIXME: What if Val is NonParamVarRegion?
88 
89  // The region should be symbolic, we do not know it's value.
90  const auto *SymReg = dyn_cast_or_null<SymbolicRegion>(Val.getAsRegion());
91  if (!SymReg)
92  return false;
93 
94  // Get it's symbol and find the declaration region it's pointing to.
95  const auto *Sm = dyn_cast<SymbolRegionValue>(SymReg->getSymbol());
96  if (!Sm)
97  return false;
98  const auto *DeclReg = dyn_cast<DeclRegion>(Sm->getRegion());
99  if (!DeclReg)
100  return false;
101 
102  // This region corresponds to a declaration, find out if it's a global/extern
103  // variable named stdin with the proper type.
104  if (const auto *D = dyn_cast_or_null<VarDecl>(DeclReg->getDecl())) {
105  D = D->getCanonicalDecl();
106  // FIXME: This should look for an exact match.
107  if (D->getName().contains("stdin") && D->isExternC()) {
108  const QualType FILETy = ACtx.getFILEType().getCanonicalType();
109  const QualType Ty = D->getType().getCanonicalType();
110 
111  if (Ty->isPointerType())
112  return Ty->getPointeeType() == FILETy;
113  }
114  }
115  return false;
116 }

118 SVal getPointeeOf(const CheckerContext &C, Loc LValue) {
119  const QualType ArgTy = LValue.getType(C.getASTContext());
120  if (!ArgTy->isPointerType() || !ArgTy->getPointeeType()->isVoidType())
121  return C.getState()->getSVal(LValue);
122 
123  // Do not dereference void pointers. Treat them as byte pointers instead.
124  // FIXME: we might want to consider more than just the first byte.
125  return C.getState()->getSVal(LValue, C.getASTContext().CharTy);
126 }

129 Optional<SVal> getPointeeOf(const CheckerContext &C, SVal Arg) {
130  if (auto LValue = Arg.getAs<Loc>())
131  return getPointeeOf(C, *LValue);
132  return None;

135 /// Given a pointer, return the SVal of its pointee or if it is tainted,
136 /// otherwise return the pointer's SVal if tainted.
137 /// Also considers stdin as a taint source.
138 Optional<SVal> getTaintedPointeeOrPointer(const CheckerContext &C, SVal Arg) {
139  const ProgramStateRef State = C.getState();
140 
141  if (auto Pointee = getPointeeOf(C, Arg))
142  if (isTainted(State, *Pointee)) // FIXME: isTainted(...) ? Pointee : None;
143  return Pointee;
144 
145  if (isTainted(State, Arg))
146  return Arg;
147 
148  // FIXME: This should be done by the isTainted() API.
149  if (isStdin(Arg, C.getASTContext()))
150  return Arg;
151 
152  return None;
153 }

160 /// ArgSet is used to describe arguments relevant for taint detection or
161 /// taint application. A discrete set of argument indexes and a variadic
162 /// argument list signified by a starting index are supported.
163 class ArgSet {
164 public:
165  ArgSet() = default;
166  ArgSet(ArgVecTy &&DiscreteArgs, Optional<ArgIdxTy> VariadicIndex = None)
167  : DiscreteArgs(std::move(DiscreteArgs)),
168  VariadicIndex(std::move(VariadicIndex)) {}
169 
170  bool contains(ArgIdxTy ArgIdx) const {
171  if (llvm::is_contained(DiscreteArgs, ArgIdx))
172  return true;
173 
174  return VariadicIndex && ArgIdx >= *VariadicIndex;
175  }
176 
177  bool isEmpty() const { return DiscreteArgs.empty() && !VariadicIndex; }
178 
179  ArgVecTy ArgsUpTo(ArgIdxTy LastArgIdx) const {
180  ArgVecTy Args;
181  for (ArgIdxTy I = ReturnValueIndex; I <= LastArgIdx; ++I) {
182  if (contains(I))
183  Args.push_back(I);
184  }
185  return Args;
186  }

188 private:
189  ArgVecTy DiscreteArgs;
190  Optional<ArgIdxTy> VariadicIndex;

193 /// A struct used to specify taint propagation rules for a function.
194 ///
195 /// If any of the possible taint source arguments is tainted, all of the
196 /// destination arguments should also be tainted. If ReturnValueIndex is added
197 /// to the dst list, the return value will be tainted.
198 class GenericTaintRule {
199  /// Arguments which are taints sinks and should be checked, and a report
200  /// should be emitted if taint reaches these.
201  ArgSet SinkArgs;
202  /// Arguments which should be sanitized on function return.
203  ArgSet FilterArgs;
204  /// Arguments which can participate in taint propagationa. If any of the
205  /// arguments in PropSrcArgs is tainted, all arguments in PropDstArgs should
206  /// be tainted.
207  ArgSet PropSrcArgs;
208  ArgSet PropDstArgs;
209 
210  /// A message that explains why the call is sensitive to taint.
211  Optional<StringRef> SinkMsg;
212 
213  GenericTaintRule() = default;
214 
215  GenericTaintRule(ArgSet &&Sink, ArgSet &&Filter, ArgSet &&Src, ArgSet &&Dst,
216  Optional<StringRef> SinkMsg = None)
217  : SinkArgs(std::move(Sink)), FilterArgs(std::move(Filter)),
218  PropSrcArgs(std::move(Src)), PropDstArgs(std::move(Dst)),
219  SinkMsg(SinkMsg) {}
220 
221 public:
222  /// Make a rule that reports a warning if taint reaches any of \p FilterArgs
223  /// arguments.
224  static GenericTaintRule Sink(ArgSet &&SinkArgs,
225  Optional<StringRef> Msg = None) {
226  return {std::move(SinkArgs), {}, {}, {}, Msg};
227  }
228 
229  /// Make a rule that sanitizes all FilterArgs arguments.
230  static GenericTaintRule Filter(ArgSet &&FilterArgs) {
231  return {{}, std::move(FilterArgs), {}, {}};
232  }
233 
234  /// Make a rule that unconditionally taints all Args.
235  /// If Func is provided, it must also return true for taint to propagate.
236  static GenericTaintRule Source(ArgSet &&SourceArgs) {
237  return {{}, {}, {}, std::move(SourceArgs)};
238  }
239 
240  /// Make a rule that taints all PropDstArgs if any of PropSrcArgs is tainted.
241  static GenericTaintRule Prop(ArgSet &&SrcArgs, ArgSet &&DstArgs) {
242  return {{}, {}, std::move(SrcArgs), std::move(DstArgs)};
243  }
244 
245  /// Make a rule that taints all PropDstArgs if any of PropSrcArgs is tainted.
246  static GenericTaintRule SinkProp(ArgSet &&SinkArgs, ArgSet &&SrcArgs,
247  ArgSet &&DstArgs,
248  Optional<StringRef> Msg = None) {
249  return {
250  std::move(SinkArgs), {}, std::move(SrcArgs), std::move(DstArgs), Msg};
251  }
252 
253  /// Process a function which could either be a taint source, a taint sink, a
254  /// taint filter or a taint propagator.
255  void process(const GenericTaintChecker &Checker, const CallEvent &Call,
256  CheckerContext &C) const;
257 
258  /// Handles the resolution of indexes of type ArgIdxTy to Expr*-s.
259  static const Expr *GetArgExpr(ArgIdxTy ArgIdx, const CallEvent &Call) {
260  return ArgIdx == ReturnValueIndex ? Call.getOriginExpr()
261  : Call.getArgExpr(ArgIdx);
262  };
263 
264  /// Functions for custom taintedness propagation.
265  static bool UntrustedEnv(CheckerContext &C);
266 };

780 void GenericTaintChecker::checkPostCall(const CallEvent &Call,
781  CheckerContext &C) const {
782  // Set the marked values as tainted. The return value only accessible from
783  // checkPostStmt.
784  ProgramStateRef State = C.getState();
785  const StackFrameContext *CurrentFrame = C.getStackFrame();
786 
787  // Depending on what was tainted at pre-visit, we determined a set of
788  // arguments which should be tainted after the function returns. These are
789  // stored in the state as TaintArgsOnPostVisit set.
790  TaintArgsOnPostVisitTy TaintArgsMap = State->get<TaintArgsOnPostVisit>();
791 
792  const ImmutableSet<ArgIdxTy> *TaintArgs = TaintArgsMap.lookup(CurrentFrame);
793  if (!TaintArgs)
794  return;
795  assert(!TaintArgs->isEmpty());
796 
797  LLVM_DEBUG(for (ArgIdxTy I
798  : *TaintArgs) {
799  llvm::dbgs() << "PostCall<";
800  Call.dump(llvm::dbgs());
801  llvm::dbgs() << "> actually wants to taint arg index: " << I << '\n';
802  });
803 
804  for (ArgIdxTy ArgNum : *TaintArgs) {
805  // Special handling for the tainted return value.
806  if (ArgNum == ReturnValueIndex) {
807  State = addTaint(State, Call.getReturnValue());
808  continue;
809  }
810 
811  // The arguments are pointer arguments. The data they are pointing at is
812  // tainted after the call.
813  if (auto V = getPointeeOf(C, Call.getArgSVal(ArgNum)))
814  State = addTaint(State, *V);
815  }
816 
817  // Clear up the taint info from the state.
818  State = State->remove<TaintArgsOnPostVisit>(CurrentFrame);
819  C.addTransition(State);
820 }

827 void GenericTaintRule::process(const GenericTaintChecker &Checker,
828  const CallEvent &Call, CheckerContext &C) const {
829  ProgramStateRef State = C.getState();
830  const ArgIdxTy CallNumArgs = fromArgumentCount(Call.getNumArgs());
831 
832  /// Iterate every call argument, and get their corresponding Expr and SVal.
833  const auto ForEachCallArg = [&C, &Call, CallNumArgs](auto &&Fun) {
834  for (ArgIdxTy I = ReturnValueIndex; I < CallNumArgs; ++I) {
835  const Expr *E = GetArgExpr(I, Call);
836  Fun(I, E, C.getSVal(E));
837  }
838  };
839 
840  /// Check for taint sinks.
841  ForEachCallArg([this, &Checker, &C, &State](ArgIdxTy I, const Expr *E, SVal) {
842  if (SinkArgs.contains(I) && isTaintedOrPointsToTainted(E, State, C))
843  Checker.generateReportIfTainted(E, SinkMsg.getValueOr(MsgCustomSink), C);
844  });
845 
846  /// Check for taint filters.
847  ForEachCallArg([this, &C, &State](ArgIdxTy I, const Expr *E, SVal S) {
848  if (FilterArgs.contains(I)) {
849  State = removeTaint(State, S);
850  if (auto P = getPointeeOf(C, S))
851  State = removeTaint(State, *P);
852  }
853  });
854 
855  /// Check for taint propagation sources.
856  /// A rule is relevant if PropSrcArgs is empty, or if any of its signified
857  /// args are tainted in context of the current CallEvent.
858  bool IsMatching = PropSrcArgs.isEmpty();
859  ForEachCallArg(
860  [this, &C, &IsMatching, &State](ArgIdxTy I, const Expr *E, SVal) {
861  IsMatching = IsMatching || (PropSrcArgs.contains(I) &&
862  isTaintedOrPointsToTainted(E, State, C));
863  });
864 
865  if (!IsMatching)
866  return;
867 
868  const auto WouldEscape = [](SVal V, QualType Ty) -> bool {
869  if (!V.getAs<Loc>())
870  return false;
871 
872  const bool IsNonConstRef = Ty->isReferenceType() && !Ty.isConstQualified();
873  const bool IsNonConstPtr =
874  Ty->isPointerType() && !Ty->getPointeeType().isConstQualified();
875 
876  return IsNonConstRef || IsNonConstPtr;
877  };
878 
879  /// Propagate taint where it is necessary.
880  auto &F = State->getStateManager().get_context<ArgIdxFactory>();
881  ImmutableSet<ArgIdxTy> Result = F.getEmptySet();
882  ForEachCallArg(
883  [&](ArgIdxTy I, const Expr *E, SVal V) {
884  if (PropDstArgs.contains(I)) {
885  LLVM_DEBUG(llvm::dbgs() << "PreCall<"; Call.dump(llvm::dbgs());
886  llvm::dbgs()
887  << "> prepares tainting arg index: " << I << '\n';);
888  Result = F.add(Result, I);
889  }
890 
891  // TODO: We should traverse all reachable memory regions via the
892  // escaping parameter. Instead of doing that we simply mark only the
893  // referred memory region as tainted.
894  if (WouldEscape(V, E->getType())) {
895  LLVM_DEBUG(if (!Result.contains(I)) {
896  llvm::dbgs() << "PreCall<";
897  Call.dump(llvm::dbgs());
898  llvm::dbgs() << "> prepares tainting arg index: " << I << '\n';
899  });
900  Result = F.add(Result, I);
901  }
902  });
903 
904  if (!Result.isEmpty())
905  State = State->set<TaintArgsOnPostVisit>(C.getStackFrame(), Result);
906  C.addTransition(State);
907 }

916  CheckerContext &C) const {
917  assert(E);
918  Optional<SVal> TaintedSVal{getTaintedPointeeOrPointer(C, C.getSVal(E))};

921  return false;

977 void GenericTaintChecker::taintUnsafeSocketProtocol(const CallEvent &Call,
978  CheckerContext &C) const {
979  if (Call.getNumArgs() < 1)
980  return;
981  const IdentifierInfo *ID = Call.getCalleeIdentifier();
982  if (!ID)
983  return;
984  if (!ID->getName().equals("socket"))
985  return;
986 
987  SourceLocation DomLoc = Call.getArgExpr(0)->getExprLoc();
988  StringRef DomName = C.getMacroNameOrSpelling(DomLoc);
989  // Allow internal communication protocols.
990  bool SafeProtocol = DomName.equals("AF_SYSTEM") ||
991  DomName.equals("AF_LOCAL") || DomName.equals("AF_UNIX") ||
992  DomName.equals("AF_RESERVED_36");
993  if (SafeProtocol)
994  return;

997  auto &F = State->getStateManager().get_context<ArgIdxFactory>();
998  ImmutableSet<ArgIdxTy> Result = F.add(F.getEmptySet(), ReturnValueIndex);
999  State = State->set<TaintArgsOnPostVisit>(C.getStackFrame(), Result);
1000  C.addTransition(State);
