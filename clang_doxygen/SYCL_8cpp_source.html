61 void SYCL::constructLLVMForeachCommand(Compilation &C, const JobAction &JA,
62  std::unique_ptr<Command> InputCommand,
63  const InputInfoList &InputFiles,
64  const InputInfo &Output, const Tool *T,
65  StringRef Increment, StringRef Ext,
66  StringRef ParallelJobs) {
67  // Construct llvm-foreach command.
68  // The llvm-foreach command looks like this:
69  // llvm-foreach --in-file-list=a.list --in-replace='{}' -- echo '{}'
70  ArgStringList ForeachArgs;
71  std::string OutputFileName(Output.getFilename());
72  ForeachArgs.push_back(C.getArgs().MakeArgString("--out-ext=" + Ext));
73  for (auto &I : InputFiles) {
74  std::string Filename(I.getFilename());
75  ForeachArgs.push_back(
76  C.getArgs().MakeArgString("--in-file-list=" + Filename));
77  ForeachArgs.push_back(
78  C.getArgs().MakeArgString("--in-replace=" + Filename));
79  }
80 
81  ForeachArgs.push_back(
82  C.getArgs().MakeArgString("--out-file-list=" + OutputFileName));
83  ForeachArgs.push_back(
84  C.getArgs().MakeArgString("--out-replace=" + OutputFileName));
85  if (!Increment.empty())
86  ForeachArgs.push_back(
87  C.getArgs().MakeArgString("--out-increment=" + Increment));
88  if (!ParallelJobs.empty())
89  ForeachArgs.push_back(C.getArgs().MakeArgString("--jobs=" + ParallelJobs));
90 
91  if (C.getDriver().isSaveTempsEnabled()) {
92  SmallString<128> OutputDirName;
93  if (C.getDriver().isSaveTempsObj()) {
94  OutputDirName =
95  T->getToolChain().GetFilePath(OutputFileName.c_str()).c_str();
96  llvm::sys::path::remove_filename(OutputDirName);
97  }
98  // Use the current dir if the `GetFilePath` returned en empty string, which
99  // is the case when the `OutputFileName` does not contain any directory
100  // information, or if in CWD mode. This is necessary for `llvm-foreach`, as
101  // it would disregard the parameter without it. Otherwise append separator.
102  if (OutputDirName.empty())
103  llvm::sys::path::native(OutputDirName = "./");
104  else
105  OutputDirName.append(llvm::sys::path::get_separator());
106  ForeachArgs.push_back(
107  C.getArgs().MakeArgString("--out-dir=" + OutputDirName));
108  }
109 
110  ForeachArgs.push_back(C.getArgs().MakeArgString("--"));
111  ForeachArgs.push_back(
112  C.getArgs().MakeArgString(InputCommand->getExecutable()));
113 
114  for (auto &Arg : InputCommand->getArguments())
115  ForeachArgs.push_back(Arg);
116 
117  SmallString<128> ForeachPath(C.getDriver().Dir);
118  llvm::sys::path::append(ForeachPath, "llvm-foreach");
119  const char *Foreach = C.getArgs().MakeArgString(ForeachPath);
120 
121  auto Cmd = std::make_unique<Command>(JA, *T, ResponseFileSupport::None(),
122  Foreach, ForeachArgs, None);
123  // FIXME: Add the FPGA specific timing diagnostic to the foreach call.
124  // The foreach call obscures the return codes from the tool it is calling
125  // to the compiler itself.
126  addFPGATimingDiagnostic(Cmd, C);
127  C.addCommand(std::move(Cmd));
128 }

149 const char *SYCL::Linker::constructLLVMLinkCommand(
150  Compilation &C, const JobAction &JA, const InputInfo &Output,
151  const ArgList &Args, StringRef SubArchName, StringRef OutputFilePrefix,
152  const InputInfoList &InputFiles) const {
153  // Split inputs into libraries which have 'archive' type and other inputs
154  // which can be either objects or list files. Object files are linked together
155  // in a usual way, but the libraries/list files need to be linked differently.
156  // We need to fetch only required symbols from the libraries. With the current
157  // llvm-link command line interface that can be achieved with two step
158  // linking: at the first step we will link objects into an intermediate
159  // partially linked image which on the second step will be linked with the
160  // libraries with --only-needed option.
161  ArgStringList Opts;
162  ArgStringList Objs;
163  ArgStringList Libs;
164  // Add the input bc's created by compile step.
165  // When offloading, the input file(s) could be from unbundled partially
166  // linked archives. The unbundled information is a list of files and not
167  // an actual object/archive. Take that list and pass those to the linker
168  // instead of the original object.
169  if (JA.isDeviceOffloading(Action::OFK_SYCL)) {
170  auto isSYCLDeviceLib = [&C](const InputInfo &II) {
171  const ToolChain *HostTC = C.getSingleOffloadToolChain<Action::OFK_Host>();
172  StringRef LibPostfix = ".o";
173  if (HostTC->getTriple().isWindowsMSVCEnvironment() &&
174  C.getDriver().IsCLMode())
175  LibPostfix = ".obj";
176  StringRef InputFilename =
177  llvm::sys::path::filename(StringRef(II.getFilename()));
178  StringRef LibSyclPrefix("libsycl-");
179  if (!InputFilename.startswith(LibSyclPrefix) ||
180  !InputFilename.endswith(LibPostfix) || (InputFilename.count('-') < 2))
181  return false;
182  // Skip the prefix "libsycl-"
183  StringRef PureLibName = InputFilename.substr(LibSyclPrefix.size());
184  for (const auto &L : SYCLDeviceLibList) {
185  if (PureLibName.startswith(L))
186  return true;
187  }
188  return false;
189  };
190  size_t InputFileNum = InputFiles.size();
191  bool LinkSYCLDeviceLibs = (InputFileNum >= 2);
192  LinkSYCLDeviceLibs = LinkSYCLDeviceLibs && !isSYCLDeviceLib(InputFiles[0]);
193  for (size_t Idx = 1; Idx < InputFileNum; ++Idx)
194  LinkSYCLDeviceLibs =
195  LinkSYCLDeviceLibs && isSYCLDeviceLib(InputFiles[Idx]);
196  // Go through the Inputs to the link. When a listfile is encountered, we
197  // know it is an unbundled generated list.
198  if (LinkSYCLDeviceLibs)
199  Opts.push_back("-only-needed");
200  for (const auto &II : InputFiles) {
201  if (II.getType() == types::TY_Tempfilelist) {
202  // Pass the unbundled list with '@' to be processed.
203  std::string FileName(II.getFilename());
204  Libs.push_back(C.getArgs().MakeArgString("@" + FileName));
205  } else if (II.getType() == types::TY_Archive && !LinkSYCLDeviceLibs) {
206  Libs.push_back(II.getFilename());
207  } else
208  Objs.push_back(II.getFilename());
209  }
210  } else
211  for (const auto &II : InputFiles)
212  Objs.push_back(II.getFilename());
213 
214  // Get llvm-link path.
215  SmallString<128> ExecPath(C.getDriver().Dir);
216  llvm::sys::path::append(ExecPath, "llvm-link");
217  const char *Exec = C.getArgs().MakeArgString(ExecPath);
218 
219  auto AddLinkCommand = [this, &C, &JA, Exec](const char *Output,
220  const ArgStringList &Inputs,
221  const ArgStringList &Options) {
222  ArgStringList CmdArgs;
223  llvm::copy(Options, std::back_inserter(CmdArgs));
224  llvm::copy(Inputs, std::back_inserter(CmdArgs));
225  CmdArgs.push_back("-o");
226  CmdArgs.push_back(Output);
227  // TODO: temporary workaround for a problem with warnings reported by
228  // llvm-link when driver links LLVM modules with empty modules
229  CmdArgs.push_back("--suppress-warnings");
230  C.addCommand(std::make_unique<Command>(
231  JA, *this, ResponseFileSupport::AtFileUTF8(), Exec, CmdArgs, None));
232  };
233 
234  // Add an intermediate output file.
235  const char *OutputFileName = Output.getFilename();
236 
237  if (Libs.empty())
238  AddLinkCommand(OutputFileName, Objs, Opts);
239  else {
240  assert(Opts.empty() && "unexpected options");
241 
242  // Linker will be invoked twice if inputs contain libraries. First time we
243  // will link input objects into an intermediate temporary file, and on the
244  // second invocation intermediate temporary object will be linked with the
245  // libraries, but now only required symbols will be added to the final
246  // output.
247  std::string TempFile =
248  C.getDriver().GetTemporaryPath(OutputFilePrefix.str() + "-link", "bc");
249  const char *LinkOutput = C.addTempFile(C.getArgs().MakeArgString(TempFile));
250  AddLinkCommand(LinkOutput, Objs, {});
251 
252  // Now invoke linker for the second time to link required symbols from the
253  // input libraries.
254  ArgStringList LinkInputs{LinkOutput};
255  llvm::copy(Libs, std::back_inserter(LinkInputs));
256  AddLinkCommand(OutputFileName, LinkInputs, {"--only-needed"});
257  }
258  return OutputFileName;
259 }

279  const InputInfo &Output,
280  const InputInfoList &Inputs,
281  const ArgList &Args,
282  const char *LinkingOutput) const {
283 
284  assert((getToolChain().getTriple().isSPIR() ||
285  getToolChain().getTriple().isNVPTX() ||
286  getToolChain().getTriple().isAMDGCN()) &&
287  "Unsupported target");
288

291

294

297  if (getToolChain().getTriple().isNVPTX() ||
298  getToolChain().getTriple().isAMDGCN()) {
299  InputInfoList NvptxInputs;

301  if (!II.isFilename())
302  continue;
303  NvptxInputs.push_back(II);
304  }
305 
306  constructLLVMLinkCommand(C, JA, Output, Args, SubArchName, Prefix,
307  NvptxInputs);
308  return;
309  }
310 
311  InputInfoList SpirvInputs;
312  for (const auto &II : Inputs) {
313  if (!II.isFilename())
314  continue;
315  SpirvInputs.push_back(II);
316  }
317 
318  constructLLVMLinkCommand(C, JA, Output, Args, SubArchName, Prefix,
319  SpirvInputs);
320 }

330 void SYCL::fpga::BackendCompiler::constructOpenCLAOTCommand(
331  Compilation &C, const JobAction &JA, const InputInfo &Output,
332  const InputInfoList &Inputs, const ArgList &Args) const {
333  // Construct opencl-aot command. This is used for FPGA AOT compilations
334  // when performing emulation. Input file will be a SPIR-V binary which
335  // will be compiled to an aocx file.
336  InputInfoList ForeachInputs;
337  InputInfoList FPGADepFiles;
338  ArgStringList CmdArgs{"-device=fpga_fast_emu"};
339 
340  for (const auto &II : Inputs) {
341  if (II.getType() == types::TY_TempAOCOfilelist ||
342  II.getType() == types::TY_FPGA_Dependencies ||
343  II.getType() == types::TY_FPGA_Dependencies_List)
344  continue;
345  if (II.getType() == types::TY_Tempfilelist)
346  ForeachInputs.push_back(II);
347  CmdArgs.push_back(
348  C.getArgs().MakeArgString("-spv=" + Twine(II.getFilename())));
349  }
350  CmdArgs.push_back(
351  C.getArgs().MakeArgString("-ir=" + Twine(Output.getFilename())));
352 
353  StringRef ForeachExt = "aocx";
354  if (Arg *A = Args.getLastArg(options::OPT_fsycl_link_EQ))
355  if (A->getValue() == StringRef("early"))
356  ForeachExt = "aocr";
357 
358  // Add any implied arguments before user defined arguments.
359  const toolchains::SYCLToolChain &TC =
360  static_cast<const toolchains::SYCLToolChain &>(getToolChain());
361  llvm::Triple CPUTriple("spir64_x86_64");
362  TC.AddImpliedTargetArgs(CPUTriple, Args, CmdArgs);
363  // Add the target args passed in
364  TC.TranslateBackendTargetArgs(CPUTriple, Args, CmdArgs);
365  TC.TranslateLinkerTargetArgs(CPUTriple, Args, CmdArgs);
366 
367  SmallString<128> ExecPath(
368  getToolChain().GetProgramPath(makeExeName(C, "opencl-aot")));
369  const char *Exec = C.getArgs().MakeArgString(ExecPath);
370  auto Cmd = std::make_unique<Command>(JA, *this, ResponseFileSupport::None(),
371  Exec, CmdArgs, None);
372  if (!ForeachInputs.empty()) {
373  StringRef ParallelJobs =
374  Args.getLastArgValue(options::OPT_fsycl_max_parallel_jobs_EQ);
375  constructLLVMForeachCommand(C, JA, std::move(Cmd), ForeachInputs, Output,
376  this, "", ForeachExt, ParallelJobs);
377  } else
378  C.addCommand(std::move(Cmd));
379 }

381 void SYCL::fpga::BackendCompiler::ConstructJob(
382  Compilation &C, const JobAction &JA, const InputInfo &Output,
383  const InputInfoList &Inputs, const ArgList &Args,
384  const char *LinkingOutput) const {
385  assert((getToolChain().getTriple().getArch() == llvm::Triple::spir ||
386  getToolChain().getTriple().getArch() == llvm::Triple::spir64) &&
387  "Unsupported target");
388 
389  // Grab the -Xsycl-target* options.
390  const toolchains::SYCLToolChain &TC =
391  static_cast<const toolchains::SYCLToolChain &>(getToolChain());
392  ArgStringList TargetArgs;
393  TC.TranslateBackendTargetArgs(TC.getTriple(), Args, TargetArgs);
394 
395  // When performing emulation compilations for FPGA AOT, we want to use
396  // opencl-aot instead of aoc.
397  if (C.getDriver().isFPGAEmulationMode()) {
398  constructOpenCLAOTCommand(C, JA, Output, Inputs, Args);
399  return;
400  }
401 
402  InputInfoList ForeachInputs;
403  InputInfoList FPGADepFiles;
404  StringRef CreatedReportName;
405  ArgStringList CmdArgs{"-o", Output.getFilename()};
406  for (const auto &II : Inputs) {
407  std::string Filename(II.getFilename());
408  if (II.getType() == types::TY_Tempfilelist)
409  ForeachInputs.push_back(II);
410  if (II.getType() == types::TY_TempAOCOfilelist)
411  // Add any FPGA library lists. These come in as special tempfile lists.
412  CmdArgs.push_back(Args.MakeArgString(Twine("-library-list=") + Filename));
413  else if (II.getType() == types::TY_FPGA_Dependencies ||
414  II.getType() == types::TY_FPGA_Dependencies_List)
415  FPGADepFiles.push_back(II);
416  else
417  CmdArgs.push_back(C.getArgs().MakeArgString(Filename));
418  // Check for any AOCR input, if found use that as the project report name
419  StringRef Ext(llvm::sys::path::extension(Filename));
420  if (Ext.empty())
421  continue;
422  if (getToolChain().LookupTypeForExtension(Ext.drop_front()) ==
423  types::TY_FPGA_AOCR) {
424  // Keep the base of the .aocr file name. Input file is a temporary,
425  // so we are stripping off the additional naming information for a
426  // cleaner name. The suffix being stripped from the name is the
427  // added temporary string and the extension.
428  StringRef SuffixFormat("-XXXXXX.aocr");
429  SmallString<128> NameBase(
430  Filename.substr(0, Filename.length() - SuffixFormat.size()));
431  NameBase.append(".prj");
432  CreatedReportName =
433  Args.MakeArgString(llvm::sys::path::filename(NameBase));
434  }
435  }
436  CmdArgs.push_back("-sycl");
437 
438  StringRef ForeachExt = "aocx";
439  if (Arg *A = Args.getLastArg(options::OPT_fsycl_link_EQ))
440  if (A->getValue() == StringRef("early")) {
441  CmdArgs.push_back("-rtl");
442  ForeachExt = "aocr";
443  }
444 
445  for (auto *A : Args) {
446  // Any input file is assumed to have a dependency file associated and
447  // the report folder can also be named based on the first input.
448  if (A->getOption().getKind() != Option::InputClass)
449  continue;
450  SmallString<128> ArgName(A->getSpelling());
451  StringRef Ext(llvm::sys::path::extension(ArgName));
452  if (Ext.empty())
453  continue;
454  types::ID Ty = getToolChain().LookupTypeForExtension(Ext.drop_front());
455  if (Ty == types::TY_INVALID)
456  continue;
457  if (types::isSrcFile(Ty) || Ty == types::TY_Object) {
458  // The project report is created in CWD, so strip off any directory
459  // information if provided with the input file.
460  StringRef TrimmedArgName = llvm::sys::path::filename(ArgName);
461  if (types::isSrcFile(Ty)) {
462  SmallString<128> DepName(
463  C.getDriver().getFPGATempDepFile(std::string(TrimmedArgName)));
464  if (!DepName.empty())
465  FPGADepFiles.push_back(InputInfo(types::TY_Dependencies,
466  Args.MakeArgString(DepName),
467  Args.MakeArgString(DepName)));
468  }
469  if (CreatedReportName.empty()) {
470  // Project report should be saved into CWD, so strip off any
471  // directory information if provided with the input file.
472  llvm::sys::path::replace_extension(ArgName, "prj");
473  CreatedReportName = Args.MakeArgString(ArgName);
474  }
475  }
476  }
477 
478  // Add any dependency files.
479  if (!FPGADepFiles.empty()) {
480  SmallString<128> DepOpt("-dep-files=");
481  for (unsigned I = 0; I < FPGADepFiles.size(); ++I) {
482  if (I)
483  DepOpt += ',';
484  if (FPGADepFiles[I].getType() == types::TY_FPGA_Dependencies_List)
485  DepOpt += "@";
486  DepOpt += FPGADepFiles[I].getFilename();
487  }
488  CmdArgs.push_back(C.getArgs().MakeArgString(DepOpt));
489  }
490 
491  // Depending on output file designations, set the report folder
492  SmallString<128> ReportOptArg;
493  if (Arg *FinalOutput = Args.getLastArg(options::OPT_o, options::OPT__SLASH_o,
494  options::OPT__SLASH_Fe)) {
495  SmallString<128> FN(FinalOutput->getValue());
496  // For "-o file.xxx" where the option value has an extension, if the
497  // extension is one of .a .o .out .lib .obj .exe, the output project
498  // directory name will be file.proj which omits the extension. Otherwise
499  // the output project directory name will be file.xxx.prj which keeps
500  // the original extension.
501  StringRef Ext = llvm::sys::path::extension(FN);
502  SmallVector<StringRef, 6> Exts = {".o", ".a", ".out",
503  ".obj", ".lib", ".exe"};
504  if (std::find(Exts.begin(), Exts.end(), Ext) != Exts.end())
505  llvm::sys::path::replace_extension(FN, "prj");
506  else
507  FN.append(".prj");
508  const char *FolderName = Args.MakeArgString(FN);
509  ReportOptArg += FolderName;
510  } else {
511  // Output directory is based off of the first object name as captured
512  // above.
513  if (!CreatedReportName.empty())
514  ReportOptArg += CreatedReportName;
515  }
516  if (!ReportOptArg.empty())
517  CmdArgs.push_back(C.getArgs().MakeArgString(
518  Twine("-output-report-folder=") + ReportOptArg));
519 
520  // Add any implied arguments before user defined arguments.
521  TC.AddImpliedTargetArgs(getToolChain().getTriple(), Args, CmdArgs);
522 
523  // Add -Xsycl-target* options.
524  TC.TranslateBackendTargetArgs(getToolChain().getTriple(), Args, CmdArgs);
525  TC.TranslateLinkerTargetArgs(getToolChain().getTriple(), Args, CmdArgs);
526 
527  // Look for -reuse-exe=XX option
528  if (Arg *A = Args.getLastArg(options::OPT_reuse_exe_EQ)) {
529  Args.ClaimAllArgs(options::OPT_reuse_exe_EQ);
530  CmdArgs.push_back(Args.MakeArgString(A->getAsString(Args)));
531  }
532 
533  SmallString<128> ExecPath(
534  getToolChain().GetProgramPath(makeExeName(C, "aoc")));
535  const char *Exec = C.getArgs().MakeArgString(ExecPath);
536  auto Cmd = std::make_unique<Command>(JA, *this, ResponseFileSupport::None(),
537  Exec, CmdArgs, None);
538  addFPGATimingDiagnostic(Cmd, C);
539  if (!ForeachInputs.empty()) {
540  StringRef ParallelJobs =
541  Args.getLastArgValue(options::OPT_fsycl_max_parallel_jobs_EQ);
542  constructLLVMForeachCommand(C, JA, std::move(Cmd), ForeachInputs, Output,
543  this, ReportOptArg, ForeachExt, ParallelJobs);
544  } else
545  C.addCommand(std::move(Cmd));
546 }

638 llvm::opt::DerivedArgList *
639 SYCLToolChain::TranslateArgs(const llvm::opt::DerivedArgList &Args,
640  StringRef BoundArch,
641  Action::OffloadKind DeviceOffloadKind) const {
642  DerivedArgList *DAL =
643  HostTC.TranslateArgs(Args, BoundArch, DeviceOffloadKind);
644 
645  if (!DAL) {
646  DAL = new DerivedArgList(Args.getBaseArgs());
647  for (Arg *A : Args) {
648  // Filter out any options we do not want to pass along to the device
649  // compilation.
650  switch ((options::ID)A->getOption().getID()) {
651  default:
652  DAL->append(A);
653  break;
654  }
655  }
656  }

658  if (!getDriver().isFPGAEmulationMode() &&
659  getTriple().getSubArch() == llvm::Triple::SPIRSubArch_fpga)
660  DAL->eraseArg(options::OPT_O0);
661 
662  const OptTable &Opts = getDriver().getOpts();
663  if (!BoundArch.empty()) {
664  DAL->eraseArg(options::OPT_march_EQ);
665  DAL->AddJoinedArg(nullptr, Opts.getOption(options::OPT_march_EQ),
666  BoundArch);
667  }
668  return DAL;

672  llvm::opt::ArgStringList &CmdArgs) {

682 // Expects a specific type of option (e.g. -Xsycl-target-backend) and will
683 // extract the arguments.
684 void SYCLToolChain::TranslateTargetOpt(const llvm::opt::ArgList &Args,
685  llvm::opt::ArgStringList &CmdArgs,
686  OptSpecifier Opt,
687  OptSpecifier Opt_EQ) const {
688  for (auto *A : Args) {
689  bool OptNoTriple;
690  OptNoTriple = A->getOption().matches(Opt);
691  if (A->getOption().matches(Opt_EQ)) {
692  // Passing device args: -X<Opt>=<triple> -opt=val.
693  if (getDriver().MakeSYCLDeviceTriple(A->getValue()) != getTriple())
694  // Provided triple does not match current tool chain.
695  continue;
696  } else if (!OptNoTriple)
697  // Don't worry about any of the other args, we only want to pass what is
698  // passed in -X<Opt>
699  continue;
700 
701  // Add the argument from -X<Opt>
702  StringRef ArgString;
703  if (OptNoTriple) {
704  // With multiple -fsycl-targets, a triple is required so we know where
705  // the options should go.
706  const Arg *TargetArg = Args.getLastArg(options::OPT_fsycl_targets_EQ);
707  if (TargetArg && TargetArg->getValues().size() != 1) {
708  getDriver().Diag(diag::err_drv_Xsycl_target_missing_triple)
709  << A->getSpelling();
710  continue;
711  }
712  // No triple, so just add the argument.
713  ArgString = A->getValue();
714  } else
715  // Triple found, add the next argument in line.
716  ArgString = A->getValue(1);
717 
718  parseTargetOpts(ArgString, Args, CmdArgs);
719  A->claim();
720  }
721 }

723 void SYCLToolChain::AddImpliedTargetArgs(
724  const llvm::Triple &Triple, const llvm::opt::ArgList &Args,
725  llvm::opt::ArgStringList &CmdArgs) const {
726  // Current implied args are for debug information and disabling of
727  // optimizations. They are passed along to the respective areas as follows:
728  // FPGA and default device: -g -cl-opt-disable
729  // GEN: -options "-g -O0"
730  // CPU: "--bo=-g -cl-opt-disable"
731  llvm::opt::ArgStringList BeArgs;
732  bool IsGen = Triple.getSubArch() == llvm::Triple::SPIRSubArch_gen;
733  if (Arg *A = Args.getLastArg(options::OPT_g_Group, options::OPT__SLASH_Z7))
734  if (!A->getOption().matches(options::OPT_g0))
735  BeArgs.push_back("-g");
736  if (Args.getLastArg(options::OPT_O0))
737  BeArgs.push_back("-cl-opt-disable");
738  if (BeArgs.empty())
739  return;
740  if (Triple.getSubArch() == llvm::Triple::NoSubArch ||
741  Triple.getSubArch() == llvm::Triple::SPIRSubArch_fpga) {
742  for (StringRef A : BeArgs)
743  CmdArgs.push_back(Args.MakeArgString(A));
744  return;
745  }
746  SmallString<128> BeOpt;
747  if (IsGen)
748  CmdArgs.push_back("-options");
749  else
750  BeOpt = "--bo=";
751  for (unsigned I = 0; I < BeArgs.size(); ++I) {
752  if (I)
753  BeOpt += ' ';
754  BeOpt += BeArgs[I];
755  }
756  CmdArgs.push_back(Args.MakeArgString(BeOpt));
757 }

759 void SYCLToolChain::TranslateBackendTargetArgs(
760  const llvm::Triple &Triple, const llvm::opt::ArgList &Args,
761  llvm::opt::ArgStringList &CmdArgs) const {
762  // Handle -Xs flags.
763  for (auto *A : Args) {
764  // When parsing the target args, the -Xs<opt> type option applies to all
765  // target compilations is not associated with a specific triple. The
766  // option can be used in 3 different ways:
767  // -Xs -DFOO -Xs -DBAR
768  // -Xs "-DFOO -DBAR"
769  // -XsDFOO -XsDBAR
770  // All of the above examples will pass -DFOO -DBAR to the backend compiler.
771 
772  // Do not add the -Xs to the default SYCL triple (spir64) when we know we
773  // have implied the setting.
774  if ((A->getOption().matches(options::OPT_Xs) ||
775  A->getOption().matches(options::OPT_Xs_separate)) &&
776  Triple.getSubArch() == llvm::Triple::NoSubArch && Triple.isSPIR() &&
777  getDriver().isSYCLDefaultTripleImplied())
778  continue;
779 
780  if (A->getOption().matches(options::OPT_Xs)) {
781  // Take the arg and create an option out of it.
782  CmdArgs.push_back(Args.MakeArgString(Twine("-") + A->getValue()));
783  A->claim();
784  continue;
785  }
786  if (A->getOption().matches(options::OPT_Xs_separate)) {
787  StringRef ArgString(A->getValue());
788  parseTargetOpts(ArgString, Args, CmdArgs);
789  A->claim();
790  continue;
791  }
792  }
793  // Do not process -Xsycl-target-backend for implied spir64
794  if (Triple.getSubArch() == llvm::Triple::NoSubArch && Triple.isSPIR() &&
795  getDriver().isSYCLDefaultTripleImplied())
796  return;
797  // Handle -Xsycl-target-backend.
798  TranslateTargetOpt(Args, CmdArgs, options::OPT_Xsycl_backend,
799  options::OPT_Xsycl_backend_EQ);
800 }
