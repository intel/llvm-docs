65 void SYCL::constructLLVMForeachCommand(Compilation &C, const JobAction &JA,
66  std::unique_ptr<Command> InputCommand,
67  const InputInfoList &InputFiles,
68  const InputInfo &Output, const Tool *T,
69  StringRef Increment, StringRef Ext,
70  StringRef ParallelJobs) {
71  // Construct llvm-foreach command.
72  // The llvm-foreach command looks like this:
73  // llvm-foreach --in-file-list=a.list --in-replace='{}' -- echo '{}'
74  ArgStringList ForeachArgs;
75  std::string OutputFileName(T->getToolChain().getInputFilename(Output));
76  ForeachArgs.push_back(C.getArgs().MakeArgString("--out-ext=" + Ext));
77  for (auto &I : InputFiles) {
78  std::string Filename(T->getToolChain().getInputFilename(I));
79  ForeachArgs.push_back(
80  C.getArgs().MakeArgString("--in-file-list=" + Filename));
81  ForeachArgs.push_back(
82  C.getArgs().MakeArgString("--in-replace=" + Filename));
83  }
84 
85  ForeachArgs.push_back(
86  C.getArgs().MakeArgString("--out-file-list=" + OutputFileName));
87  ForeachArgs.push_back(
88  C.getArgs().MakeArgString("--out-replace=" + OutputFileName));
89  if (!Increment.empty())
90  ForeachArgs.push_back(
91  C.getArgs().MakeArgString("--out-increment=" + Increment));
92  if (!ParallelJobs.empty())
93  ForeachArgs.push_back(C.getArgs().MakeArgString("--jobs=" + ParallelJobs));
94 
95  if (C.getDriver().isSaveTempsEnabled()) {
96  SmallString<128> OutputDirName;
97  if (C.getDriver().isSaveTempsObj()) {
98  OutputDirName =
99  T->getToolChain().GetFilePath(OutputFileName.c_str()).c_str();
100  llvm::sys::path::remove_filename(OutputDirName);
101  }
102  // Use the current dir if the `GetFilePath` returned en empty string, which
103  // is the case when the `OutputFileName` does not contain any directory
104  // information, or if in CWD mode. This is necessary for `llvm-foreach`, as
105  // it would disregard the parameter without it. Otherwise append separator.
106  if (OutputDirName.empty())
107  llvm::sys::path::native(OutputDirName = "./");
108  else
109  OutputDirName.append(llvm::sys::path::get_separator());
110  ForeachArgs.push_back(
111  C.getArgs().MakeArgString("--out-dir=" + OutputDirName));
112  }
113 
114  // If fsycl-dump-device-code is passed, put the PTX files
115  // into the path provided in fsycl-dump-device-code.
116  if (T->getToolChain().getTriple().isNVPTX() &&
117  C.getDriver().isDumpDeviceCodeEnabled() && Ext.equals("s")) {
118  SmallString<128> OutputDir;
119 
120  Arg *DumpDeviceCodeArg =
121  C.getArgs().getLastArg(options::OPT_fsycl_dump_device_code_EQ);
122 
123  OutputDir = (DumpDeviceCodeArg ? DumpDeviceCodeArg->getValue() : "");
124 
125  // If the output directory path is empty, put the PTX files in the
126  // current directory.
127  if (OutputDir.empty())
128  llvm::sys::path::native(OutputDir = "./");
129  else
130  OutputDir.append(llvm::sys::path::get_separator());
131  ForeachArgs.push_back(C.getArgs().MakeArgString("--out-dir=" + OutputDir));
132  }
133 
134  ForeachArgs.push_back(C.getArgs().MakeArgString("--"));
135  ForeachArgs.push_back(
136  C.getArgs().MakeArgString(InputCommand->getExecutable()));
137 
138  for (auto &Arg : InputCommand->getArguments())
139  ForeachArgs.push_back(Arg);
140 
141  SmallString<128> ForeachPath(C.getDriver().Dir);
142  llvm::sys::path::append(ForeachPath, "llvm-foreach");
143  const char *Foreach = C.getArgs().MakeArgString(ForeachPath);
144 
145  auto Cmd = std::make_unique<Command>(JA, *T, ResponseFileSupport::None(),
146  Foreach, ForeachArgs, std::nullopt);
147  // FIXME: Add the FPGA specific timing diagnostic to the foreach call.
148  // The foreach call obscures the return codes from the tool it is calling
149  // to the compiler itself.
150  addFPGATimingDiagnostic(Cmd, C);
151  C.addCommand(std::move(Cmd));
152 }

159 // Return whether to use native bfloat16 library.
160 static bool selectBfloatLibs(const llvm::Triple &Triple, const Compilation &C,
161  bool &UseNative) {
162  const llvm::opt::ArgList &Args = C.getArgs();
163  bool NeedLibs = false;
164 
165  // spir64 target is actually JIT compilation, so we defer selection of
166  // bfloat16 libraries to runtime. For AOT we need libraries, but skip
167  // for Nvidia.
168  NeedLibs =
169  Triple.getSubArch() != llvm::Triple::NoSubArch && !Triple.isNVPTX();
170  UseNative = false;
171  if (NeedLibs && Triple.getSubArch() == llvm::Triple::SPIRSubArch_gen &&
172  C.hasOffloadToolChain<Action::OFK_SYCL>()) {
173  ArgStringList TargArgs;
174  auto ToolChains = C.getOffloadToolChains<Action::OFK_SYCL>();
175  // Match up the toolchain with the incoming Triple so we are grabbing the
176  // expected arguments to scrutinize.
177  for (auto TI = ToolChains.first, TE = ToolChains.second; TI != TE; ++TI) {
178  llvm::Triple SYCLTriple = TI->second->getTriple();
179  if (SYCLTriple == Triple) {
180  const toolchains::SYCLToolChain *SYCLTC =
181  static_cast<const toolchains::SYCLToolChain *>(TI->second);
182  SYCLTC->TranslateBackendTargetArgs(Triple, Args, TargArgs);
183  break;
184  }
185  }
186 
187  auto checkBF = [](StringRef Device) {
188  return Device.starts_with("pvc") || Device.starts_with("ats");
189  };
190 
191  std::string Params;
192  for (const auto &Arg : TargArgs) {
193  Params += " ";
194  Params += Arg;
195  }
196  size_t DevicesPos = Params.find("-device ");
197  UseNative = false;
198  if (DevicesPos != std::string::npos) {
199  UseNative = true;
200  std::istringstream Devices(Params.substr(DevicesPos + 8));
201  for (std::string S; std::getline(Devices, S, ',');)
202  UseNative &= checkBF(S);
203  }
204  }
205  return NeedLibs;
206 }

208 SmallVector<std::string, 8>
209 SYCL::getDeviceLibraries(const Compilation &C, const llvm::Triple &TargetTriple,
210  bool IsSpirvAOT) {
211  SmallVector<std::string, 8> LibraryList;
212  const llvm::opt::ArgList &Args = C.getArgs();
213 
214  struct DeviceLibOptInfo {
215  StringRef DeviceLibName;
216  StringRef DeviceLibOption;
217  };
218 
219  bool NoDeviceLibs = false;
220  // Currently, all SYCL device libraries will be linked by default. Linkage
221  // of "internal" libraries cannot be affected via -fno-sycl-device-lib.
222  llvm::StringMap<bool> DeviceLibLinkInfo = {
223  {"libc", true}, {"libm-fp32", true}, {"libm-fp64", true},
224  {"libimf-fp32", true}, {"libimf-fp64", true}, {"libimf-bf16", true},
225  {"libm-bfloat16", true}, {"internal", true}};
226  if (Arg *A = Args.getLastArg(options::OPT_fsycl_device_lib_EQ,
227  options::OPT_fno_sycl_device_lib_EQ)) {
228  if (A->getValues().size() == 0)
229  C.getDriver().Diag(diag::warn_drv_empty_joined_argument)
230  << A->getAsString(Args);
231  else {
232  if (A->getOption().matches(options::OPT_fno_sycl_device_lib_EQ))
233  NoDeviceLibs = true;
234 
235  for (StringRef Val : A->getValues()) {
236  if (Val == "all") {
237  for (const auto &K : DeviceLibLinkInfo.keys())
238  DeviceLibLinkInfo[K] =
239  true && (!NoDeviceLibs || K.equals("internal"));
240  break;
241  }
242  auto LinkInfoIter = DeviceLibLinkInfo.find(Val);
243  if (LinkInfoIter == DeviceLibLinkInfo.end() || Val.equals("internal")) {
244  // TODO: Move the diagnostic to the SYCL section of
245  // Driver::CreateOffloadingDeviceToolChains() to minimize code
246  // duplication.
247  C.getDriver().Diag(diag::err_drv_unsupported_option_argument)
248  << A->getSpelling() << Val;
249  }
250  DeviceLibLinkInfo[Val] = true && !NoDeviceLibs;
251  }
252  }
253  }
254  using SYCLDeviceLibsList = SmallVector<DeviceLibOptInfo, 5>;
255 
256  const SYCLDeviceLibsList SYCLDeviceWrapperLibs = {
257  {"libsycl-crt", "libc"},
258  {"libsycl-complex", "libm-fp32"},
259  {"libsycl-complex-fp64", "libm-fp64"},
260  {"libsycl-cmath", "libm-fp32"},
261  {"libsycl-cmath-fp64", "libm-fp64"},
262 #if defined(_WIN32)
263  {"libsycl-msvc-math", "libm-fp32"},
264 #endif
265  {"libsycl-imf", "libimf-fp32"},
266  {"libsycl-imf-fp64", "libimf-fp64"},
267  {"libsycl-imf-bf16", "libimf-bf16"}};
268  // For AOT compilation, we need to link sycl_device_fallback_libs as
269  // default too.
270  const SYCLDeviceLibsList SYCLDeviceFallbackLibs = {
271  {"libsycl-fallback-cassert", "libc"},
272  {"libsycl-fallback-cstring", "libc"},
273  {"libsycl-fallback-complex", "libm-fp32"},
274  {"libsycl-fallback-complex-fp64", "libm-fp64"},
275  {"libsycl-fallback-cmath", "libm-fp32"},
276  {"libsycl-fallback-cmath-fp64", "libm-fp64"},
277  {"libsycl-fallback-imf", "libimf-fp32"},
278  {"libsycl-fallback-imf-fp64", "libimf-fp64"},
279  {"libsycl-fallback-imf-bf16", "libimf-bf16"}};
280  const SYCLDeviceLibsList SYCLDeviceBfloat16FallbackLib = {
281  {"libsycl-fallback-bfloat16", "libm-bfloat16"}};
282  const SYCLDeviceLibsList SYCLDeviceBfloat16NativeLib = {
283  {"libsycl-native-bfloat16", "libm-bfloat16"}};
284  // ITT annotation libraries are linked in separately whenever the device
285  // code instrumentation is enabled.
286  const SYCLDeviceLibsList SYCLDeviceAnnotationLibs = {
287  {"libsycl-itt-user-wrappers", "internal"},
288  {"libsycl-itt-compiler-wrappers", "internal"},
289  {"libsycl-itt-stubs", "internal"}};
290 #if !defined(_WIN32)
291  const SYCLDeviceLibsList SYCLDeviceSanitizerLibs = {
292  {"libsycl-sanitizer", "internal"}};
293 #endif
294  StringRef LibSuffix = ".bc";
295  if (TargetTriple.isNVPTX())
296  // For NVidia, we are unbundling objects.
297  LibSuffix = C.getDefaultToolChain().getTriple().isWindowsMSVCEnvironment()
298  ? ".obj"
299  : ".o";
300  auto addLibraries = [&](const SYCLDeviceLibsList &LibsList) {
301  for (const DeviceLibOptInfo &Lib : LibsList) {
302  if (!DeviceLibLinkInfo[Lib.DeviceLibOption])
303  continue;
304  SmallString<128> LibName(Lib.DeviceLibName);
305  llvm::sys::path::replace_extension(LibName, LibSuffix);
306  LibraryList.push_back(Args.MakeArgString(LibName));
307  }
308  };
309 
310  addLibraries(SYCLDeviceWrapperLibs);
311  if (IsSpirvAOT || TargetTriple.isNVPTX())
312  addLibraries(SYCLDeviceFallbackLibs);
313 
314  bool NativeBfloatLibs;
315  bool NeedBfloatLibs = selectBfloatLibs(TargetTriple, C, NativeBfloatLibs);
316  if (NeedBfloatLibs) {
317  // Add native or fallback bfloat16 library.
318  if (NativeBfloatLibs)
319  addLibraries(SYCLDeviceBfloat16NativeLib);
320  else
321  addLibraries(SYCLDeviceBfloat16FallbackLib);
322  }
323 
324  if (Args.hasFlag(options::OPT_fsycl_instrument_device_code,
325  options::OPT_fno_sycl_instrument_device_code, true))
326  addLibraries(SYCLDeviceAnnotationLibs);
327 
328 #if !defined(_WIN32)
329  if (Arg *A = Args.getLastArg(options::OPT_fsanitize_EQ,
330  options::OPT_fno_sanitize_EQ)) {
331  if (A->getOption().matches(options::OPT_fsanitize_EQ) &&
332  A->getValues().size() == 1) {
333  std::string SanitizeVal = A->getValue();
334  if (SanitizeVal == "address")
335  addLibraries(SYCLDeviceSanitizerLibs);
336  }
337  } else {
338  // User can pass -fsanitize=address to device compiler via
339  // -Xsycl-target-frontend, sanitize device library must be
340  // linked with user's device image if so.
341  bool IsDeviceAsanEnabled = false;
342  auto SyclFEArg = Args.getAllArgValues(options::OPT_Xsycl_frontend);
343  IsDeviceAsanEnabled = (std::count(SyclFEArg.begin(), SyclFEArg.end(),
344  "-fsanitize=address") > 0);
345  if (!IsDeviceAsanEnabled) {
346  auto SyclFEArgEq = Args.getAllArgValues(options::OPT_Xsycl_frontend_EQ);
347  IsDeviceAsanEnabled = (std::count(SyclFEArgEq.begin(), SyclFEArgEq.end(),
348  "-fsanitize=address") > 0);
349  }
350 
351  // User can also enable asan for SYCL device via -Xarch_device option.
352  if (!IsDeviceAsanEnabled) {
353  auto DeviceArchVals = Args.getAllArgValues(options::OPT_Xarch_device);
354  for (auto DArchVal : DeviceArchVals) {
355  if (DArchVal.find("-fsanitize=address") != std::string::npos) {
356  IsDeviceAsanEnabled = true;
357  break;
358  }
359  }
360  }
361 
362  if (IsDeviceAsanEnabled)
363  addLibraries(SYCLDeviceSanitizerLibs);
364  }
365 #endif
366  return LibraryList;
367 }

402 const char *SYCL::Linker::constructLLVMLinkCommand(
403  Compilation &C, const JobAction &JA, const InputInfo &Output,
404  const ArgList &Args, StringRef SubArchName, StringRef OutputFilePrefix,
405  const InputInfoList &InputFiles) const {
406  // Split inputs into libraries which have 'archive' type and other inputs
407  // which can be either objects or list files. Object files are linked together
408  // in a usual way, but the libraries/list files need to be linked differently.
409  // We need to fetch only required symbols from the libraries. With the current
410  // llvm-link command line interface that can be achieved with two step
411  // linking: at the first step we will link objects into an intermediate
412  // partially linked image which on the second step will be linked with the
413  // libraries with --only-needed option.
414  ArgStringList Opts;
415  ArgStringList Objs;
416  ArgStringList Libs;
417  // Add the input bc's created by compile step.
418  // When offloading, the input file(s) could be from unbundled partially
419  // linked archives. The unbundled information is a list of files and not
420  // an actual object/archive. Take that list and pass those to the linker
421  // instead of the original object.
422  if (JA.isDeviceOffloading(Action::OFK_SYCL)) {
423  bool IsRDC = !shouldDoPerObjectFileLinking(C);
424  const bool IsSYCLNativeCPU = isSYCLNativeCPU(this->getToolChain());
425  auto isNoRDCDeviceCodeLink = [&](const InputInfo &II) {
426  if (IsRDC)
427  return false;
428  if (II.getType() != clang::driver::types::TY_LLVM_BC)
429  return false;
430  if (InputFiles.size() != 2)
431  return false;
432  return &II == &InputFiles[1];
433  };
434  auto isSYCLDeviceLib = [&](const InputInfo &II) {
435  const ToolChain *HostTC = C.getSingleOffloadToolChain<Action::OFK_Host>();
436  const bool IsNVPTX = this->getToolChain().getTriple().isNVPTX();
437  StringRef LibPostfix = ".bc";
438  if (IsNVPTX) {
439  LibPostfix = ".o";
440  if (HostTC->getTriple().isWindowsMSVCEnvironment() &&
441  C.getDriver().IsCLMode())
442  LibPostfix = ".obj";
443  }
444  std::string FileName = this->getToolChain().getInputFilename(II);
445  StringRef InputFilename = llvm::sys::path::filename(FileName);
446  if (IsNVPTX || IsSYCLNativeCPU) {
447  // Linking SYCL Device libs requires libclc as well as libdevice
448  if ((InputFilename.find("libspirv") != InputFilename.npos ||
449  InputFilename.find("libdevice") != InputFilename.npos))
450  return true;
451  if (IsNVPTX)
452  LibPostfix = ".cubin";
453  }
454  StringRef LibSyclPrefix("libsycl-");
455  if (!InputFilename.starts_with(LibSyclPrefix) ||
456  !InputFilename.ends_with(LibPostfix))
457  return false;
458  // Skip the prefix "libsycl-"
459  std::string PureLibName =
460  InputFilename.substr(LibSyclPrefix.size()).str();
461  if (isNoRDCDeviceCodeLink(II)) {
462  // Skip the final - until the . because we linked all device libs into a
463  // single BC in a previous action so we have a temp file name.
464  auto FinalDashPos = PureLibName.find_last_of('-');
465  auto DotPos = PureLibName.find_last_of('.');
466  assert((FinalDashPos != std::string::npos &&
467  DotPos != std::string::npos) &&
468  "Unexpected filename");
469  PureLibName =
470  PureLibName.substr(0, FinalDashPos) + PureLibName.substr(DotPos);
471  }
472  for (const auto &L : SYCLDeviceLibList) {
473  std::string DeviceLibName(L);
474  DeviceLibName.append(LibPostfix);
475  if (StringRef(PureLibName).equals(DeviceLibName) ||
476  (IsNVPTX && StringRef(PureLibName).starts_with(L)))
477  return true;
478  }
479  return false;
480  };
481  size_t InputFileNum = InputFiles.size();
482  bool LinkSYCLDeviceLibs = (InputFileNum >= 2);
483  LinkSYCLDeviceLibs = LinkSYCLDeviceLibs && !isSYCLDeviceLib(InputFiles[0]);
484  for (size_t Idx = 1; Idx < InputFileNum; ++Idx)
485  LinkSYCLDeviceLibs =
486  LinkSYCLDeviceLibs && isSYCLDeviceLib(InputFiles[Idx]);
487  if (LinkSYCLDeviceLibs) {
488  Opts.push_back("-only-needed");
489  }
490  // Go through the Inputs to the link. When a listfile is encountered, we
491  // know it is an unbundled generated list.
492  for (const auto &II : InputFiles) {
493  std::string FileName = getToolChain().getInputFilename(II);
494  if (II.getType() == types::TY_Tempfilelist) {
495  if (IsRDC) {
496  // Pass the unbundled list with '@' to be processed.
497  Libs.push_back(C.getArgs().MakeArgString("@" + FileName));
498  } else {
499  assert(InputFiles.size() == 2 &&
500  "Unexpected inputs for no-RDC with temp file list");
501  // If we're in no-RDC mode and the input is a temp file list,
502  // we want to link multiple object files each against device libs,
503  // so we should consider this input as an object and not pass '@'.
504  Objs.push_back(C.getArgs().MakeArgString(FileName));
505  }
506  } else if (II.getType() == types::TY_Archive && !LinkSYCLDeviceLibs) {
507  Libs.push_back(C.getArgs().MakeArgString(FileName));
508  } else
509  Objs.push_back(C.getArgs().MakeArgString(FileName));
510  }
511  } else
512  for (const auto &II : InputFiles)
513  Objs.push_back(
514  C.getArgs().MakeArgString(getToolChain().getInputFilename(II)));
515 
516  // Get llvm-link path.
517  SmallString<128> ExecPath(C.getDriver().Dir);
518  llvm::sys::path::append(ExecPath, "llvm-link");
519  const char *Exec = C.getArgs().MakeArgString(ExecPath);
520 
521  auto AddLinkCommand = [this, &C, &JA, Exec](const char *Output,
522  const ArgStringList &Inputs,
523  const ArgStringList &Options) {
524  ArgStringList CmdArgs;
525  llvm::copy(Options, std::back_inserter(CmdArgs));
526  llvm::copy(Inputs, std::back_inserter(CmdArgs));
527  CmdArgs.push_back("-o");
528  CmdArgs.push_back(Output);
529  // TODO: temporary workaround for a problem with warnings reported by
530  // llvm-link when driver links LLVM modules with empty modules
531  CmdArgs.push_back("--suppress-warnings");
532  C.addCommand(std::make_unique<Command>(JA, *this,
533  ResponseFileSupport::AtFileUTF8(),
534  Exec, CmdArgs, std::nullopt));
535  };
536 
537  // Add an intermediate output file.
538  const char *OutputFileName =
539  C.getArgs().MakeArgString(getToolChain().getInputFilename(Output));
540 
541  if (Libs.empty())
542  AddLinkCommand(OutputFileName, Objs, Opts);
543  else {
544  assert(Opts.empty() && "unexpected options");
545 
546  // Linker will be invoked twice if inputs contain libraries. First time we
547  // will link input objects into an intermediate temporary file, and on the
548  // second invocation intermediate temporary object will be linked with the
549  // libraries, but now only required symbols will be added to the final
550  // output.

557  // input libraries.

559  llvm::copy(Libs, std::back_inserter(LinkInputs));
560  AddLinkCommand(OutputFileName, LinkInputs, {"--only-needed"});
561  }

687 void SYCL::fpga::BackendCompiler::ConstructJob(
688  Compilation &C, const JobAction &JA, const InputInfo &Output,
689  const InputInfoList &Inputs, const ArgList &Args,
690  const char *LinkingOutput) const {
691  assert(getToolChain().getTriple().isSPIROrSPIRV() && "Unsupported target");
692 
693  // Grab the -Xsycl-target* options.
694  const toolchains::SYCLToolChain &TC =
695  static_cast<const toolchains::SYCLToolChain &>(getToolChain());
696  ArgStringList TargetArgs;
697  TC.TranslateBackendTargetArgs(TC.getTriple(), Args, TargetArgs);
698 
699  // When performing emulation compilations for FPGA AOT, we want to use
700  // opencl-aot instead of aoc.
701  if (C.getDriver().IsFPGAEmulationMode()) {
702  constructOpenCLAOTCommand(C, JA, Output, Inputs, Args);
703  return;
704  }
705 
706  InputInfoList ForeachInputs;
707  InputInfoList FPGADepFiles;
708  StringRef CreatedReportName;
709  ArgStringList CmdArgs{"-o", Output.getFilename()};
710  for (const auto &II : Inputs) {
711  std::string Filename(II.getFilename());
712  if (II.getType() == types::TY_Tempfilelist)
713  ForeachInputs.push_back(II);
714  if (II.getType() == types::TY_TempAOCOfilelist)
715  // Add any FPGA library lists. These come in as special tempfile lists.
716  CmdArgs.push_back(Args.MakeArgString(Twine("-library-list=") + Filename));
717  else if (II.getType() == types::TY_FPGA_Dependencies ||
718  II.getType() == types::TY_FPGA_Dependencies_List)
719  FPGADepFiles.push_back(II);
720  else
721  CmdArgs.push_back(C.getArgs().MakeArgString(Filename));
722  // Check for any AOCR input, if found use that as the project report name
723  StringRef Ext(llvm::sys::path::extension(Filename));
724  if (Ext.empty())
725  continue;
726  if (getToolChain().LookupTypeForExtension(Ext.drop_front()) ==
727  types::TY_FPGA_AOCR) {
728  // Keep the base of the .aocr file name. Input file is a temporary,
729  // so we are stripping off the additional naming information for a
730  // cleaner name. The suffix being stripped from the name is the
731  // added temporary string and the extension.
732  StringRef SuffixFormat("-XXXXXX.aocr");
733  SmallString<128> NameBase(
734  Filename.substr(0, Filename.length() - SuffixFormat.size()));
735  NameBase.append(".prj");
736  CreatedReportName =
737  Args.MakeArgString(llvm::sys::path::filename(NameBase));
738  }
739  }
740  CmdArgs.push_back("-sycl");
741 
742  StringRef ForeachExt = "aocx";
743  if (Arg *A = Args.getLastArg(options::OPT_fsycl_link_EQ))
744  if (A->getValue() == StringRef("early")) {
745  CmdArgs.push_back("-rtl");
746  ForeachExt = "aocr";
747  }
748 
749  for (auto *A : Args) {
750  // Any input file is assumed to have a dependency file associated and
751  // the report folder can also be named based on the first input.
752  if (A->getOption().getKind() != Option::InputClass)
753  continue;
754  SmallString<128> ArgName(A->getSpelling());
755  StringRef Ext(llvm::sys::path::extension(ArgName));
756  if (Ext.empty())
757  continue;
758  types::ID Ty = getToolChain().LookupTypeForExtension(Ext.drop_front());
759  if (Ty == types::TY_INVALID)
760  continue;
761  if (types::isSrcFile(Ty) || Ty == types::TY_Object) {
762  // The project report is created in CWD, so strip off any directory
763  // information if provided with the input file.
764  StringRef TrimmedArgName = llvm::sys::path::filename(ArgName);
765  if (types::isSrcFile(Ty)) {
766  SmallString<128> DepName(
767  C.getDriver().getFPGATempDepFile(std::string(TrimmedArgName)));
768  if (!DepName.empty())
769  FPGADepFiles.push_back(InputInfo(types::TY_Dependencies,
770  Args.MakeArgString(DepName),
771  Args.MakeArgString(DepName)));
772  }
773  if (CreatedReportName.empty()) {
774  // Project report should be saved into CWD, so strip off any
775  // directory information if provided with the input file.
776  llvm::sys::path::replace_extension(ArgName, "prj");
777  CreatedReportName = Args.MakeArgString(ArgName);
778  }
779  }
780  }
781 
782  // Add any dependency files.
783  if (!FPGADepFiles.empty()) {
784  SmallString<128> DepOpt("-dep-files=");
785  for (unsigned I = 0; I < FPGADepFiles.size(); ++I) {
786  if (I)
787  DepOpt += ',';
788  if (FPGADepFiles[I].getType() == types::TY_FPGA_Dependencies_List)
789  DepOpt += "@";
790  DepOpt += FPGADepFiles[I].getFilename();
791  }
792  CmdArgs.push_back(C.getArgs().MakeArgString(DepOpt));
793  }
794 
795  // Depending on output file designations, set the report folder
796  SmallString<128> ReportOptArg;
797  if (Arg *FinalOutput = Args.getLastArg(options::OPT_o, options::OPT__SLASH_o,
798  options::OPT__SLASH_Fe)) {
799  SmallString<128> FN(FinalOutput->getValue());
800  // For "-o file.xxx" where the option value has an extension, if the
801  // extension is one of .a .o .out .lib .obj .exe, the output project
802  // directory name will be file.proj which omits the extension. Otherwise
803  // the output project directory name will be file.xxx.prj which keeps
804  // the original extension.
805  StringRef Ext = llvm::sys::path::extension(FN);
806  SmallVector<StringRef, 6> Exts = {".o", ".a", ".out",
807  ".obj", ".lib", ".exe"};
808  if (std::find(Exts.begin(), Exts.end(), Ext) != Exts.end())
809  llvm::sys::path::replace_extension(FN, "prj");
810  else
811  FN.append(".prj");
812  const char *FolderName = Args.MakeArgString(FN);
813  ReportOptArg += FolderName;
814  } else {
815  // Default output directory should match default output executable name
816  ReportOptArg += "a.prj";
817  }
818  if (!ReportOptArg.empty())
819  CmdArgs.push_back(C.getArgs().MakeArgString(
820  Twine("-output-report-folder=") + ReportOptArg));
821 
822  // Add any implied arguments before user defined arguments.
823  const ToolChain *HostTC = C.getSingleOffloadToolChain<Action::OFK_Host>();
824  TC.AddImpliedTargetArgs(getToolChain().getTriple(), Args, CmdArgs, JA,
825  *HostTC);
826 
827  // Add -Xsycl-target* options.
828  TC.TranslateBackendTargetArgs(getToolChain().getTriple(), Args, CmdArgs);
829  TC.TranslateLinkerTargetArgs(getToolChain().getTriple(), Args, CmdArgs);
830 
831  // Look for -reuse-exe=XX option
832  if (Arg *A = Args.getLastArg(options::OPT_reuse_exe_EQ)) {
833  Args.ClaimAllArgs(options::OPT_reuse_exe_EQ);
834  CmdArgs.push_back(Args.MakeArgString(A->getAsString(Args)));
835  }
836 
837  SmallString<128> ExecPath(
838  getToolChain().GetProgramPath(makeExeName(C, "aoc")));
839  const char *Exec = C.getArgs().MakeArgString(ExecPath);
840  auto Cmd = std::make_unique<Command>(JA, *this, ResponseFileSupport::None(),
841  Exec, CmdArgs, std::nullopt);
842  addFPGATimingDiagnostic(Cmd, C);
843  if (!ForeachInputs.empty()) {
844  StringRef ParallelJobs =
845  Args.getLastArgValue(options::OPT_fsycl_max_parallel_jobs_EQ);
846  constructLLVMForeachCommand(C, JA, std::move(Cmd), ForeachInputs, Output,
847  this, ReportOptArg, ForeachExt, ParallelJobs);
848  } else
849  C.addCommand(std::move(Cmd));
850 }

1291 // Expects a specific type of option (e.g. -Xsycl-target-backend) and will
1292 // extract the arguments.
1293 void SYCLToolChain::TranslateTargetOpt(const llvm::opt::ArgList &Args,
1294  llvm::opt::ArgStringList &CmdArgs,
1295  OptSpecifier Opt, OptSpecifier Opt_EQ,
1296  StringRef Device) const {
1297  for (auto *A : Args) {
1298  bool OptNoTriple;
1299  OptNoTriple = A->getOption().matches(Opt);
1300  if (A->getOption().matches(Opt_EQ)) {
1301  // Passing device args: -X<Opt>=<triple> -opt=val.
1302  StringRef GenDevice = SYCL::gen::resolveGenDevice(A->getValue());
1303  bool IsGenTriple =
1304  getTriple().isSPIR() &&
1305  getTriple().getSubArch() == llvm::Triple::SPIRSubArch_gen;
1306  if (Device != GenDevice)
1307  continue;
1308  if (getDriver().MakeSYCLDeviceTriple(A->getValue()) != getTriple() &&
1309  (!IsGenTriple || (IsGenTriple && GenDevice.empty())))
1310  // Triples do not match, but only skip when we know we are not comparing
1311  // against intel_gpu_* and non-spir64_gen
1312  continue;
1313  } else if (!OptNoTriple)
1314  // Don't worry about any of the other args, we only want to pass what is
1315  // passed in -X<Opt>
1316  continue;
1317 
1318  // Add the argument from -X<Opt>
1319  StringRef ArgString;
1320  if (OptNoTriple) {
1321  // With multiple -fsycl-targets, a triple is required so we know where
1322  // the options should go.
1323  const Arg *TargetArg = Args.getLastArg(options::OPT_fsycl_targets_EQ);
1324  if (TargetArg && TargetArg->getValues().size() != 1) {
1325  getDriver().Diag(diag::err_drv_Xsycl_target_missing_triple)
1326  << A->getSpelling();
1327  continue;
1328  }
1329  // No triple, so just add the argument.
1330  ArgString = A->getValue();
1331  } else
1332  // Triple found, add the next argument in line.
1333  ArgString = A->getValue(1);
1334  WarnForDeprecatedBackendOpts(getDriver(), getTriple(), Device, ArgString,
1335  A);
1336  parseTargetOpts(ArgString, Args, CmdArgs);
1337  A->claim();
1338  }
1339 }

1341 void SYCLToolChain::AddImpliedTargetArgs(const llvm::Triple &Triple,
1342  const llvm::opt::ArgList &Args,
1343  llvm::opt::ArgStringList &CmdArgs,
1344  const JobAction &JA,
1345  const ToolChain &HostTC) const {
1346  // Current implied args are for debug information and disabling of
1347  // optimizations. They are passed along to the respective areas as follows:
1348  // FPGA: -g -cl-opt-disable
1349  // Default device AOT: -g -cl-opt-disable
1350  // Default device JIT: -g (-O0 is handled by the runtime)
1351  // GEN: -options "-g -O0"
1352  // CPU: "--bo=-g -cl-opt-disable"
1353  llvm::opt::ArgStringList BeArgs;
1354  // Per-device argument vector storing the device name and the backend argument
1355  // string
1356  llvm::SmallVector<std::pair<StringRef, StringRef>, 16> PerDeviceArgs;
1357  bool IsGen = Triple.getSubArch() == llvm::Triple::SPIRSubArch_gen;
1358  if (Arg *A = Args.getLastArg(options::OPT_g_Group, options::OPT__SLASH_Z7))
1359  if (!A->getOption().matches(options::OPT_g0))
1360  BeArgs.push_back("-g");
1361  // Only pass -cl-opt-disable for non-JIT, as the runtime
1362  // handles O0 for the JIT case.
1363  if (Triple.getSubArch() != llvm::Triple::NoSubArch)
1364  if (Arg *A = Args.getLastArg(options::OPT_O_Group))
1365  if (A->getOption().matches(options::OPT_O0))
1366  BeArgs.push_back("-cl-opt-disable");
1367  StringRef RegAllocModeOptName = "-ftarget-register-alloc-mode=";
1368  if (Arg *A = Args.getLastArg(options::OPT_ftarget_register_alloc_mode_EQ)) {
1369  StringRef RegAllocModeVal = A->getValue(0);
1370  auto ProcessElement = [&](StringRef Ele) {
1371  auto [DeviceName, RegAllocMode] = Ele.split(':');
1372  StringRef BackendOptName = SYCL::gen::getGenGRFFlag(RegAllocMode);
1373  bool IsDefault = RegAllocMode.equals("default");
1374  if (RegAllocMode.empty() || !DeviceName.equals("pvc") ||
1375  (BackendOptName.empty() && !IsDefault)) {
1376  getDriver().Diag(diag::err_drv_unsupported_option_argument)
1377  << A->getSpelling() << Ele;
1378  }
1379  // "default" means "provide no specification to the backend", so
1380  // we don't need to do anything here.
1381  if (IsDefault)
1382  return;
1383  if (IsGen) {
1384  // For AOT, Use ocloc's per-device options flag with the correct ocloc
1385  // option to honor the user's specification.
1386  PerDeviceArgs.push_back(
1387  {DeviceName, Args.MakeArgString(BackendOptName)});
1388  } else if (Triple.isSPIROrSPIRV() &&
1389  Triple.getSubArch() == llvm::Triple::NoSubArch) {
1390  // For JIT, pass -ftarget-register-alloc-mode=Device:BackendOpt to
1391  // clang-offload-wrapper to be processed by the runtime.
1392  BeArgs.push_back(Args.MakeArgString(RegAllocModeOptName + DeviceName +
1393  ":" + BackendOptName));
1394  }
1395  };
1396  llvm::SmallVector<StringRef, 16> RegAllocModeArgs;
1397  RegAllocModeVal.split(RegAllocModeArgs, ',');
1398  for (StringRef Elem : RegAllocModeArgs)
1399  ProcessElement(Elem);
1400  } else if (!HostTC.getTriple().isWindowsMSVCEnvironment()) {
1401  // If -ftarget-register-alloc-mode is not specified, the default is
1402  // pvc:default on Windows and and pvc:auto otherwise.
1403  StringRef DeviceName = "pvc";
1404  StringRef BackendOptName = SYCL::gen::getGenGRFFlag("auto");
1405  if (IsGen)
1406  PerDeviceArgs.push_back({DeviceName, Args.MakeArgString(BackendOptName)});
1407  else if (Triple.isSPIROrSPIRV() &&
1408  Triple.getSubArch() == llvm::Triple::NoSubArch) {
1409  BeArgs.push_back(Args.MakeArgString(RegAllocModeOptName + DeviceName +
1410  ":" + BackendOptName));
1411  }
1412  }
1413  if (IsGen) {
1414  // For GEN (spir64_gen) we have implied -device settings given usage
1415  // of intel_gpu_ as a target. Handle those here, and also check that no
1416  // other -device was passed, as that is a conflict.
1417  StringRef DepInfo = JA.getOffloadingArch();
1418  if (!DepInfo.empty()) {
1419  ArgStringList TargArgs;
1420  Args.AddAllArgValues(TargArgs, options::OPT_Xs, options::OPT_Xs_separate);
1421  Args.AddAllArgValues(TargArgs, options::OPT_Xsycl_backend);
1422  // For -Xsycl-target-backend=<triple> we need to scrutinize the triple
1423  for (auto *A : Args) {
1424  if (!A->getOption().matches(options::OPT_Xsycl_backend_EQ))
1425  continue;
1426  if (StringRef(A->getValue()).starts_with("intel_gpu"))
1427  TargArgs.push_back(A->getValue(1));
1428  }
1429  if (llvm::find_if(TargArgs, [&](auto Cur) {
1430  return !strncmp(Cur, "-device", sizeof("-device") - 1);
1431  }) != TargArgs.end()) {
1432  SmallString<64> Target("intel_gpu_");
1433  Target += DepInfo;
1434  getDriver().Diag(diag::err_drv_unsupported_opt_for_target)
1435  << "-device" << Target;
1436  }
1437  CmdArgs.push_back("-device");
1438  CmdArgs.push_back(Args.MakeArgString(DepInfo));
1439  }
1440  // -ftarget-compile-fast AOT
1441  if (Args.hasArg(options::OPT_ftarget_compile_fast))
1442  BeArgs.push_back("-igc_opts 'PartitionUnit=1,SubroutineThreshold=50000'");
1443  // -ftarget-export-symbols
1444  if (Args.hasFlag(options::OPT_ftarget_export_symbols,
1445  options::OPT_fno_target_export_symbols, false))
1446  BeArgs.push_back("-library-compilation");
1447  } else if (Triple.getSubArch() == llvm::Triple::NoSubArch &&
1448  Triple.isSPIROrSPIRV()) {
1449  // -ftarget-compile-fast JIT
1450  Args.AddLastArg(BeArgs, options::OPT_ftarget_compile_fast);
1451  }
1452  if (IsGen) {
1453  for (auto [DeviceName, BackendArgStr] : PerDeviceArgs) {
1454  CmdArgs.push_back("-device_options");
1455  CmdArgs.push_back(Args.MakeArgString(DeviceName));
1456  CmdArgs.push_back(Args.MakeArgString(BackendArgStr));
1457  }
1458  }
1459  if (BeArgs.empty())
1460  return;
1461  if (Triple.getSubArch() == llvm::Triple::NoSubArch ||
1462  Triple.getSubArch() == llvm::Triple::SPIRSubArch_fpga) {
1463  for (StringRef A : BeArgs)
1464  CmdArgs.push_back(Args.MakeArgString(A));
1465  return;
1466  }
1467  SmallString<128> BeOpt;
1468  if (IsGen)
1469  CmdArgs.push_back("-options");
1470  else
1471  BeOpt = "--bo=";
1472  for (unsigned I = 0; I < BeArgs.size(); ++I) {
1473  if (I)
1474  BeOpt += ' ';
1475  BeOpt += BeArgs[I];
1476  }
1477  CmdArgs.push_back(Args.MakeArgString(BeOpt));
1478 }

1480 void SYCLToolChain::TranslateBackendTargetArgs(
1481  const llvm::Triple &Triple, const llvm::opt::ArgList &Args,
1482  llvm::opt::ArgStringList &CmdArgs, StringRef Device) const {
1483  // Handle -Xs flags.
1484  for (auto *A : Args) {
1485  // When parsing the target args, the -Xs<opt> type option applies to all
1486  // target compilations is not associated with a specific triple. The
1487  // option can be used in 3 different ways:
1488  // -Xs -DFOO -Xs -DBAR
1489  // -Xs "-DFOO -DBAR"
1490  // -XsDFOO -XsDBAR
1491  // All of the above examples will pass -DFOO -DBAR to the backend compiler.
1492 
1493  // Do not add the -Xs to the default SYCL triple when we know we have
1494  // implied the setting.
1495  if ((A->getOption().matches(options::OPT_Xs) ||
1496  A->getOption().matches(options::OPT_Xs_separate)) &&
1497  Triple.getSubArch() == llvm::Triple::NoSubArch &&
1498  Triple.isSPIROrSPIRV() && getDriver().isSYCLDefaultTripleImplied())
1499  continue;
