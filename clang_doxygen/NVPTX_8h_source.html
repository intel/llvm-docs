162  /// \returns If a target requires an address within a target specific address
163  /// space \p AddressSpace to be converted in order to be used, then return the
164  /// corresponding target specific DWARF address space.
165  ///
166  /// \returns Otherwise return None and no conversion will be emitted in the
167  /// DWARF.
168  Optional<unsigned>
169  getDWARFAddressSpace(unsigned AddressSpace) const override {
170  if (AddressSpace >= llvm::array_lengthof(NVPTXDWARFAddrSpaceMap) ||
171  NVPTXDWARFAddrSpaceMap[AddressSpace] < 0)
172  return llvm::None;
173  return NVPTXDWARFAddrSpaceMap[AddressSpace];
174  }

176  CallingConvCheckResult checkCallingConvention(CallingConv CC) const override {
177  // CUDA compilations support all of the host's calling conventions.
178  //
179  // TODO: We should warn if you apply a non-default CC to anything other than
180  // a host function.
181  if (HostTarget)
182  return HostTarget->checkCallingConvention(CC);
183  return CCCR_Warning;
184  }
