167  /// \returns If a target requires an address within a target specific address
168  /// space \p AddressSpace to be converted in order to be used, then return the
169  /// corresponding target specific DWARF address space.
170  ///
171  /// \returns Otherwise return std::nullopt and no conversion will be emitted
172  /// in the DWARF.
173  std::optional<unsigned>
174  getDWARFAddressSpace(unsigned AddressSpace) const override {
175  if (AddressSpace >= std::size(NVPTXDWARFAddrSpaceMap) ||
176  NVPTXDWARFAddrSpaceMap[AddressSpace] < 0)
177  return std::nullopt;
178  return NVPTXDWARFAddrSpaceMap[AddressSpace];
179  }

185  // a host function.

187  return HostTarget->checkCallingConvention(CC);
188  return CCCR_Warning;
189  }
