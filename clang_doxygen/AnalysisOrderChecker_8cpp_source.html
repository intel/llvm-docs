158  void checkEndFunction(const ReturnStmt *S, CheckerContext &C) const {
159  if (isCallbackEnabled(C, "EndFunction")) {
160  llvm::errs() << "EndFunction\nReturnStmt: " << (S ? "yes" : "no") << "\n";
