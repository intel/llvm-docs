28 /// An aligned address.
29 class Address {

34 protected:

37 public:
38  Address(llvm::Value *Pointer, llvm::Type *ElementType, CharUnits Alignment,
39  KnownNonNull_t IsKnownNonNull = NotKnownNonNull)

42  assert(Pointer != nullptr && "Pointer cannot be null");
