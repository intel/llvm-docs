101 // CompressedOffloadBundle represents the format for the compressed offload
102 // bundles.
103 //
104 // The format is as follows:
105 // - Magic Number (4 bytes) - A constant "CCOB".
106 // - Version (2 bytes)
107 // - Compression Method (2 bytes) - Uses the values from
108 // llvm::compression::Format.
109 // - Total file size (4 bytes). Available in version 2 and above.
110 // - Uncompressed Size (4 bytes).
111 // - Truncated MD5 Hash (8 bytes).
112 // - Compressed Data (variable length).
113 
114 class CompressedOffloadBundle {
115 private:
116  static inline const size_t MagicSize = 4;
117  static inline const size_t VersionFieldSize = sizeof(uint16_t);
118  static inline const size_t MethodFieldSize = sizeof(uint16_t);
119  static inline const size_t FileSizeFieldSize = sizeof(uint32_t);
120  static inline const size_t UncompressedSizeFieldSize = sizeof(uint32_t);
121  static inline const size_t HashFieldSize = sizeof(uint64_t);
122  static inline const size_t V1HeaderSize =
123  MagicSize + VersionFieldSize + MethodFieldSize +
124  UncompressedSizeFieldSize + HashFieldSize;
125  static inline const size_t V2HeaderSize =
126  MagicSize + VersionFieldSize + FileSizeFieldSize + MethodFieldSize +
127  UncompressedSizeFieldSize + HashFieldSize;
128  static inline const llvm::StringRef MagicNumber = "CCOB";
129  static inline const uint16_t Version = 2;
