430  void addExtended(const Scope::Local &Local, const ValueDecl *ExtendingDecl) {
431  // Walk up the chain of scopes until we find the one for ExtendingDecl.
432  // If there is no such scope, attach it to the parent one.
433  VariableScope *P = this;
434  while (P) {
435  if (P->ValDecl == ExtendingDecl) {
436  P->addLocal(Local);
437  return;
438  }
439  P = P->Parent;
440  if (!P)
441  break;
442  }
443 
444  // Use the parent scope.
445  addExtended(Local);
446  }
447 
448  virtual void emitDestruction() {}
449  virtual bool emitDestructors() { return true; }
450  VariableScope *getParent() const { return Parent; }
451

453  /// Compiler instance.
454  Compiler<Emitter> *Ctx;
455  /// Link to the parent scope.
456  VariableScope *Parent;

500  if (!Idx)
501  return true;
502  // Emit destructor calls for local variables of record
503  // type with a destructor.
504  for (Scope::Local &Local : this->Ctx->Descriptors[*Idx]) {
505  if (!Local.Desc->isPrimitive() && !Local.Desc->isPrimitiveArray()) {
506  if (!this->Ctx->emitGetPtrLocal(Local.Offset, SourceInfo{}))
507  return false;
508 
509  if (!this->Ctx->emitDestruction(Local.Desc))
510  return false;
511 
512  if (!this->Ctx->emitPopPtr(SourceInfo{}))
513  return false;
514  removeIfStoredOpaqueValue(Local);
515  }
516  }

533  It != this->Ctx->OpaqueExprs.end())
534  this->Ctx->OpaqueExprs.erase(It);
