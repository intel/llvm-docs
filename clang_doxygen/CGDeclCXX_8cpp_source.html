28 static void EmitDeclInit(CodeGenFunction &CGF, const VarDecl &D,
29  ConstantAddress DeclPtr) {
30  assert((D.hasGlobalStorage() ||
31  (D.hasLocalStorage() &&
32  (CGF.getContext().getLangOpts().OpenCLCPlusPlus ||
33  CGF.getContext().getLangOpts().SYCLIsDevice))) &&
34  "VarDecl must have global or local (in the case of OpenCL and SYCL) "
35  "storage!");
36  assert(!D.getType()->isReferenceType() &&
37  "Should not call EmitDeclInit on a reference!");
38 
39  QualType type = D.getType();
40  LValue lv = CGF.MakeAddrLValue(DeclPtr, type);
41 
42  const Expr *Init = D.getInit();
43  switch (CGF.getEvaluationKind(type)) {
44  case TEK_Scalar: {
45  CodeGenModule &CGM = CGF.CGM;
46  if (lv.isObjCStrong())
47  CGM.getObjCRuntime().EmitObjCGlobalAssign(CGF, CGF.EmitScalarExpr(Init),
48  DeclPtr, D.getTLSKind());
49  else if (lv.isObjCWeak())
50  CGM.getObjCRuntime().EmitObjCWeakAssign(CGF, CGF.EmitScalarExpr(Init),
51  DeclPtr);
52  else
53  CGF.EmitScalarInit(Init, &D, lv, false);
54  return;
55  }
56  case TEK_Complex:
57  CGF.EmitComplexExprIntoLValue(Init, lv, /*isInit*/ true);
58  return;
59  case TEK_Aggregate:
60  CGF.EmitAggExpr(Init,
61  AggValueSlot::forLValue(lv, CGF, AggValueSlot::IsDestructed,
62  AggValueSlot::DoesNotNeedGCBarriers,
63  AggValueSlot::IsNotAliased,
64  AggValueSlot::DoesNotOverlap));
65  return;
66  }
67  llvm_unreachable("bad evaluation kind");
68 }

70 /// Emit code to cause the destruction of the given variable with
71 /// static storage duration.
72 static void EmitDeclDestroy(CodeGenFunction &CGF, const VarDecl &D,
73  ConstantAddress Addr) {
74  // Honor __attribute__((no_destroy)) and bail instead of attempting
75  // to emit a reference to a possibly nonexistent destructor, which
76  // in turn can cause a crash. This will result in a global constructor
77  // that isn't balanced out by a destructor call as intended by the
78  // attribute. This also checks for -fno-c++-static-destructors and
79  // bails even if the attribute is not present.
80  QualType::DestructionKind DtorKind = D.needsDestruction(CGF.getContext());
81 
82  // FIXME: __attribute__((cleanup)) ?
83 
84  switch (DtorKind) {
85  case QualType::DK_none:
86  return;
87 
88  case QualType::DK_cxx_destructor:
89  break;
90 
91  case QualType::DK_objc_strong_lifetime:
92  case QualType::DK_objc_weak_lifetime:
93  case QualType::DK_nontrivial_c_struct:
94  // We don't care about releasing objects during process teardown.
95  assert(!D.getTLSKind() && "should have rejected this");
96  return;
97  }
98 
99  llvm::FunctionCallee Func;
100  llvm::Constant *Argument;
101 
102  CodeGenModule &CGM = CGF.CGM;
103  QualType Type = D.getType();
104 
105  // Special-case non-array C++ destructors, if they have the right signature.
106  // Under some ABIs, destructors return this instead of void, and cannot be
107  // passed directly to __cxa_atexit if the target does not allow this
108  // mismatch.
109  const CXXRecordDecl *Record = Type->getAsCXXRecordDecl();
110  bool CanRegisterDestructor =
111  Record && (!CGM.getCXXABI().HasThisReturn(
112  GlobalDecl(Record->getDestructor(), Dtor_Complete)) ||
113  CGM.getCXXABI().canCallMismatchedFunctionType());
114  // If __cxa_atexit is disabled via a flag, a different helper function is
115  // generated elsewhere which uses atexit instead, and it takes the destructor
116  // directly.
117  bool UsingExternalHelper = !CGM.getCodeGenOpts().CXAAtExit;
118  if (Record && (CanRegisterDestructor || UsingExternalHelper)) {
119  assert(!Record->hasTrivialDestructor());
120  CXXDestructorDecl *Dtor = Record->getDestructor();
121 
122  Func = CGM.getAddrAndTypeOfCXXStructor(GlobalDecl(Dtor, Dtor_Complete));
123  if (CGF.getContext().getLangOpts().OpenCL) {
124  auto DestAS =
125  CGM.getTargetCodeGenInfo().getAddrSpaceOfCxaAtexitPtrParam();
126  auto DestTy = CGF.getTypes().ConvertType(Type)->getPointerTo(
127  CGM.getContext().getTargetAddressSpace(DestAS));
128  auto SrcAS = D.getType().getQualifiers().getAddressSpace();
129  if (DestAS == SrcAS)
130  Argument = llvm::ConstantExpr::getBitCast(Addr.getPointer(), DestTy);
131  else
132  // FIXME: On addr space mismatch we are passing NULL. The generation
133  // of the global destructor function should be adjusted accordingly.
134  Argument = llvm::ConstantPointerNull::get(DestTy);
135  } else {
136  Argument = llvm::ConstantExpr::getBitCast(
137  Addr.getPointer(), CGF.getTypes().ConvertType(Type)->getPointerTo());
138  }
139  // Otherwise, the standard logic requires a helper function.
140  } else {
141  Addr = Addr.getElementBitCast(CGF.ConvertTypeForMem(Type));
142  Func = CodeGenFunction(CGM)
143  .generateDestroyHelper(Addr, Type, CGF.getDestroyer(DtorKind),
144  CGF.needsEHCleanup(DtorKind), &D);
145  Argument = llvm::Constant::getNullValue(CGF.Int8PtrTy);
146  }
147 
148  CGM.getCXXABI().registerGlobalDtor(CGF, D, Func, Argument);
149 }

159 void CodeGenFunction::EmitInvariantStart(llvm::Constant *Addr, CharUnits Size) {
160  // Do not emit the intrinsic if we're not optimizing.
161  if (!CGM.getCodeGenOpts().OptimizationLevel)
162  return;
163 
164  // Grab the llvm.invariant.start intrinsic.
165  llvm::Intrinsic::ID InvStartID = llvm::Intrinsic::invariant_start;
166  // Overloaded address space type.
167  llvm::Type *ResTy = llvm::PointerType::getInt8PtrTy(
168  CGM.getLLVMContext(), Addr->getType()->getPointerAddressSpace());
169  llvm::Type *ObjectPtr[1] = {ResTy};
170  llvm::Function *InvariantStart = CGM.getIntrinsic(InvStartID, ObjectPtr);
171 
172  // Emit a call with the size in bytes of the object.
173  uint64_t Width = Size.getQuantity();
174  llvm::Value *Args[2] = {llvm::ConstantInt::getSigned(Int64Ty, Width),
175  llvm::ConstantExpr::getBitCast(Addr, ResTy)};
176  Builder.CreateCall(InvariantStart, Args);
177 }

179 void CodeGenFunction::EmitCXXGlobalVarDeclInit(const VarDecl &D,
180  llvm::GlobalVariable *GV,
181  bool PerformInit) {
182 
183  const Expr *Init = D.getInit();
184  QualType T = D.getType();
185 
186  // The address space of a static local variable (DeclPtr) may be different
187  // from the address space of the "this" argument of the constructor. In that
188  // case, we need an addrspacecast before calling the constructor.
189  //
190  // struct StructWithCtor {
191  // __device__ StructWithCtor() {...}
192  // };
193  // __device__ void foo() {
194  // __shared__ StructWithCtor s;
195  // ...
196  // }
197  //
198  // For example, in the above CUDA code, the static local variable s has a
199  // "shared" address space qualifier, but the constructor of StructWithCtor
200  // expects "this" in the "generic" address space.
201  unsigned ExpectedAddrSpace = getContext().getTargetAddressSpace(T);
202  unsigned ActualAddrSpace = GV->getAddressSpace();
203  llvm::Constant *DeclPtr = GV;
204  if (ActualAddrSpace != ExpectedAddrSpace) {
205  llvm::PointerType *PTy = llvm::PointerType::getWithSamePointeeType(
206  GV->getType(), ExpectedAddrSpace);
207  DeclPtr = llvm::ConstantExpr::getAddrSpaceCast(DeclPtr, PTy);
208  }
209 
210  ConstantAddress DeclAddr(
211  DeclPtr, GV->getValueType(), getContext().getDeclAlign(&D));
212 
213  if (!T->isReferenceType()) {
214  if (getLangOpts().OpenMP && !getLangOpts().OpenMPSimd &&
215  D.hasAttr<OMPThreadPrivateDeclAttr>()) {
216  (void)CGM.getOpenMPRuntime().emitThreadPrivateVarDefinition(
217  &D, DeclAddr, D.getAttr<OMPThreadPrivateDeclAttr>()->getLocation(),
218  PerformInit, this);
219  }
220  if (PerformInit)
221  EmitDeclInit(*this, D, DeclAddr);
222  if (CGM.isTypeConstant(D.getType(), true))
223  EmitDeclInvariant(*this, D, DeclPtr);
224  else
225  EmitDeclDestroy(*this, D, DeclAddr);
226  return;
227  }
228 
229  assert(PerformInit && "cannot have constant initializer which needs "
230  "destruction for reference");
231  RValue RV = EmitReferenceBindingToExpr(Init);
232  EmitStoreOfScalar(RV.getScalarVal(), DeclAddr, false, T);
233 }

238  llvm::FunctionCallee dtor,

241  llvm::FunctionType *ty = llvm::FunctionType::get(CGM.VoidTy, false);
242  SmallString<256> FnName;
243  {
244  llvm::raw_svector_ostream Out(FnName);
245  CGM.getCXXABI().getMangleContext().mangleDynamicAtExitDestructor(&VD, Out);
246  }
247 
248  const CGFunctionInfo &FI = CGM.getTypes().arrangeNullaryFunction();

250  ty, FnName.str(), FI, VD.getLocation());

252  CodeGenFunction CGF(CGM);
253

255  CGM.getContext().VoidTy, fn, FI, FunctionArgList(),

257  // Emit an artificial location for this function.
258  auto AL = ApplyDebugLocation::CreateArtificial(CGF);
259 
260  llvm::CallInst *call = CGF.Builder.CreateCall(dtor, addr);
261 
262  // Make sure the call and the callee agree on calling convention.

264  dtor.getCallee()->stripPointerCastsAndAliases()))
265  call->setCallingConv(dtorFn->getCallingConv());
266 
267  CGF.FinishFunction();
268 
269  return fn;
270 }

393 void CodeGenFunction::EmitCXXGuardedInitBranch(llvm::Value *NeedsInit,
394  llvm::BasicBlock *InitBlock,
395  llvm::BasicBlock *NoInitBlock,
396  GuardKind Kind,
397  const VarDecl *D) {
398  assert((Kind == GuardKind::TlsGuard || D) && "no guarded variable");
399 
400  // A guess at how many times we will enter the initialization of a
401  // variable, depending on the kind of variable.
402  static const uint64_t InitsPerTLSVar = 1024;
403  static const uint64_t InitsPerLocalVar = 1024 * 1024;
404 
405  llvm::MDNode *Weights;
406  if (Kind == GuardKind::VariableGuard && !D->isLocalVarDecl()) {
407  // For non-local variables, don't apply any weighting for now. Due to our
408  // use of COMDATs, we expect there to be at most one initialization of the
409  // variable per DSO, but we have no way to know how many DSOs will try to
410  // initialize the variable.
411  Weights = nullptr;
412  } else {
413  uint64_t NumInits;
414  // FIXME: For the TLS case, collect and use profiling information to
415  // determine a more accurate brach weight.
416  if (Kind == GuardKind::TlsGuard || D->getTLSKind())
417  NumInits = InitsPerTLSVar;
418  else
419  NumInits = InitsPerLocalVar;
420 
421  // The probability of us entering the initializer is
422  // 1 / (total number of times we attempt to initialize the variable).
423  llvm::MDBuilder MDHelper(CGM.getLLVMContext());
424  Weights = MDHelper.createBranchWeights(1, NumInits - 1);
425  }
426 
427  Builder.CreateCondBr(NeedsInit, InitBlock, NoInitBlock, Weights);
428 }

510 void
511 CodeGenModule::EmitCXXGlobalVarDeclInitFunc(const VarDecl *D,
512  llvm::GlobalVariable *Addr,
513  bool PerformInit) {
514 
515  // According to E.2.3.1 in CUDA-7.5 Programming guide: __device__,
516  // __constant__ and __shared__ variables defined in namespace scope,
517  // that are of class type, cannot have a non-empty constructor. All
518  // the checks have been done in Sema by now. Whatever initializers
519  // are allowed are empty and we just need to ignore them here.
520  if (getLangOpts().CUDAIsDevice && !getLangOpts().GPUAllowDeviceInit &&
521  (D->hasAttr<CUDADeviceAttr>() || D->hasAttr<CUDAConstantAttr>() ||
522  D->hasAttr<CUDASharedAttr>()))
523  return;
524 
525  if (getLangOpts().OpenMP &&
526  getOpenMPRuntime().emitDeclareTargetVarDefinition(D, Addr, PerformInit))
527  return;
528 
529  // Check if we've already initialized this decl.
530  auto I = DelayedCXXInitPosition.find(D);
531  if (I != DelayedCXXInitPosition.end() && I->second == ~0U)
532  return;
533 
534  llvm::FunctionType *FTy = llvm::FunctionType::get(VoidTy, false);
535  SmallString<256> FnName;
536  {
537  llvm::raw_svector_ostream Out(FnName);
538  getCXXABI().getMangleContext().mangleDynamicInitializer(D, Out);
539  }
540 
541  // Create a variable initialization function.
542  llvm::Function *Fn = CreateGlobalInitOrCleanUpFunction(
543  FTy, FnName.str(), getTypes().arrangeNullaryFunction(), D->getLocation());
544 
545  auto *ISA = D->getAttr<InitSegAttr>();
546  CodeGenFunction(*this).GenerateCXXGlobalVarDeclInitFunc(Fn, D, Addr,
547  PerformInit);
548 
549  llvm::GlobalVariable *COMDATKey =
550  supportsCOMDAT() && D->isExternallyVisible() ? Addr : nullptr;
551 
552  if (D->getTLSKind()) {
553  // FIXME: Should we support init_priority for thread_local?
554  // FIXME: We only need to register one __cxa_thread_atexit function for the
555  // entire TU.
556  CXXThreadLocalInits.push_back(Fn);
557  CXXThreadLocalInitVars.push_back(D);
558  } else if (PerformInit && ISA) {
559  EmitPointerToInitFunc(D, Addr, Fn, ISA);
560  } else if (auto *IPA = D->getAttr<InitPriorityAttr>()) {
561  OrderGlobalInitsOrStermFinalizers Key(IPA->getPriority(),
562  PrioritizedCXXGlobalInits.size());
563  PrioritizedCXXGlobalInits.push_back(std::make_pair(Key, Fn));
564  } else if (isTemplateInstantiation(D->getTemplateSpecializationKind()) ||
565  getContext().GetGVALinkageForVariable(D) == GVA_DiscardableODR ||
566  D->hasAttr<SelectAnyAttr>()) {
567  // C++ [basic.start.init]p2:
568  // Definitions of explicitly specialized class template static data
569  // members have ordered initialization. Other class template static data
570  // members (i.e., implicitly or explicitly instantiated specializations)
571  // have unordered initialization.
572  //
573  // As a consequence, we can put them into their own llvm.global_ctors entry.
574  //
575  // If the global is externally visible, put the initializer into a COMDAT
576  // group with the global being initialized. On most platforms, this is a
577  // minor startup time optimization. In the MS C++ ABI, there are no guard
578  // variables, so this COMDAT key is required for correctness.
579  //
580  // SelectAny globals will be comdat-folded. Put the initializer into a
581  // COMDAT group associated with the global, so the initializers get folded
582  // too.
583 
584  AddGlobalCtor(Fn, 65535, COMDATKey);
585  if (COMDATKey && (getTriple().isOSBinFormatELF() ||
586  getTarget().getCXXABI().isMicrosoft())) {
587  // When COMDAT is used on ELF or in the MS C++ ABI, the key must be in
588  // llvm.used to prevent linker GC.
589  addUsedGlobal(COMDATKey);
590  }
591 
592  // If we used a COMDAT key for the global ctor, the init function can be
593  // discarded if the global ctor entry is discarded.
594  // FIXME: Do we need to restrict this to ELF and Wasm?
595  llvm::Comdat *C = Addr->getComdat();
596  if (COMDATKey && C &&
597  (getTarget().getTriple().isOSBinFormatELF() ||
598  getTarget().getTriple().isOSBinFormatWasm())) {
599  Fn->setComdat(C);
600  }
601  } else {
602  I = DelayedCXXInitPosition.find(D); // Re-do lookup in case of re-hash.
603  if (I == DelayedCXXInitPosition.end()) {
604  CXXGlobalInits.push_back(Fn);
605  } else if (I->second != ~0U) {
606  assert(I->second < CXXGlobalInits.size() &&
607  CXXGlobalInits[I->second] == nullptr);
608  CXXGlobalInits[I->second] = Fn;
609  }
610  }
611 
612  // Remember that we already emitted the initializer for this global.
613  DelayedCXXInitPosition[D] = ~0U;
614 }

652 void
653 CodeGenModule::EmitCXXGlobalInitFunc() {
654  while (!CXXGlobalInits.empty() && !CXXGlobalInits.back())
655  CXXGlobalInits.pop_back();
656 
657  if (CXXGlobalInits.empty() && PrioritizedCXXGlobalInits.empty())
658  return;
659 
660  llvm::FunctionType *FTy = llvm::FunctionType::get(VoidTy, false);
661  const CGFunctionInfo &FI = getTypes().arrangeNullaryFunction();
662 
663  // Create our global prioritized initialization function.
664  if (!PrioritizedCXXGlobalInits.empty()) {
665  SmallVector<llvm::Function *, 8> LocalCXXGlobalInits;
666  llvm::array_pod_sort(PrioritizedCXXGlobalInits.begin(),
667  PrioritizedCXXGlobalInits.end());
668  // Iterate over "chunks" of ctors with same priority and emit each chunk
669  // into separate function. Note - everything is sorted first by priority,
670  // second - by lex order, so we emit ctor functions in proper order.
671  for (SmallVectorImpl<GlobalInitData >::iterator
672  I = PrioritizedCXXGlobalInits.begin(),
673  E = PrioritizedCXXGlobalInits.end(); I != E; ) {
674  SmallVectorImpl<GlobalInitData >::iterator
675  PrioE = std::upper_bound(I + 1, E, *I, GlobalInitPriorityCmp());
676 
677  LocalCXXGlobalInits.clear();
678 
679  unsigned int Priority = I->first.priority;
680  llvm::Function *Fn = CreateGlobalInitOrCleanUpFunction(
681  FTy, "_GLOBAL__I_" + getPrioritySuffix(Priority), FI);
682 
683  for (; I < PrioE; ++I)
684  LocalCXXGlobalInits.push_back(I->second);
685 
686  CodeGenFunction(*this).GenerateCXXGlobalInitFunc(Fn, LocalCXXGlobalInits);
687  AddGlobalCtor(Fn, Priority);
688  }
689  PrioritizedCXXGlobalInits.clear();
690  }
691 
692  if (getCXXABI().useSinitAndSterm() && CXXGlobalInits.empty())
693  return;
694 
695  // Include the filename in the symbol name. Including "sub_" matches gcc
696  // and makes sure these symbols appear lexicographically behind the symbols
697  // with priority emitted above.
698  llvm::Function *Fn = CreateGlobalInitOrCleanUpFunction(
699  FTy, llvm::Twine("_GLOBAL__sub_I_", getTransformedFileName(getModule())),
700  FI);
701 
702  CodeGenFunction(*this).GenerateCXXGlobalInitFunc(Fn, CXXGlobalInits);
703  AddGlobalCtor(Fn);
704 
705  // In OpenCL global init functions must be converted to kernels in order to
706  // be able to launch them from the host.
707  // FIXME: Some more work might be needed to handle destructors correctly.
708  // Current initialization function makes use of function pointers callbacks.
709  // We can't support function pointers especially between host and device.
710  // However it seems global destruction has little meaning without any
711  // dynamic resource allocation on the device and program scope variables are
712  // destroyed by the runtime when program is released.
713  if (getLangOpts().OpenCL) {
714  GenOpenCLArgMetadata(Fn);
715  Fn->setCallingConv(llvm::CallingConv::SPIR_KERNEL);
716  }
717 
718  assert(!getLangOpts().CUDA || !getLangOpts().CUDAIsDevice ||
719  getLangOpts().GPUAllowDeviceInit);
720  if (getLangOpts().HIP && getLangOpts().CUDAIsDevice) {
721  Fn->setCallingConv(llvm::CallingConv::AMDGPU_KERNEL);
722  Fn->addFnAttr("device-init");
723  }
724 
725  CXXGlobalInits.clear();
726 }

728 void CodeGenModule::EmitCXXGlobalCleanUpFunc() {

730  PrioritizedCXXStermFinalizers.empty())
731  return;

733  llvm::FunctionType *FTy = llvm::FunctionType::get(VoidTy, false);

737  if (!PrioritizedCXXStermFinalizers.empty()) {

739  llvm::array_pod_sort(PrioritizedCXXStermFinalizers.begin(),
740  PrioritizedCXXStermFinalizers.end());
741  // Iterate over "chunks" of dtors with same priority and emit each chunk
742  // into separate function. Note - everything is sorted first by priority,
743  // second - by lex order, so we emit dtor functions in proper order.

751  LocalCXXStermFinalizers.clear();

753  unsigned int Priority = I->first.priority;
754  llvm::Function *Fn = CreateGlobalInitOrCleanUpFunction(
755  FTy, "_GLOBAL__a_" + getPrioritySuffix(Priority), FI);

759  LocalCXXStermFinalizers.emplace_back(DtorFn.getFunctionType(),

771  return;

775  CreateGlobalInitOrCleanUpFunction(FTy, "_GLOBAL__D_a", FI);

780  CXXGlobalDtorsOrStermFinalizers.clear();

818 void
819 CodeGenFunction::GenerateCXXGlobalInitFunc(llvm::Function *Fn,
820  ArrayRef<llvm::Function *> Decls,
821  ConstantAddress Guard) {
822  {
823  auto NL = ApplyDebugLocation::CreateEmpty(*this);
824  StartFunction(GlobalDecl(), getContext().VoidTy, Fn,
825  getTypes().arrangeNullaryFunction(), FunctionArgList());
826  // Emit an artificial location for this function.
827  auto AL = ApplyDebugLocation::CreateArtificial(*this);
828 
829  llvm::BasicBlock *ExitBlock = nullptr;
830  if (Guard.isValid()) {
831  // If we have a guard variable, check whether we've already performed
832  // these initializations. This happens for TLS initialization functions.
833  llvm::Value *GuardVal = Builder.CreateLoad(Guard);
834  llvm::Value *Uninit = Builder.CreateIsNull(GuardVal,
835  "guard.uninitialized");
836  llvm::BasicBlock *InitBlock = createBasicBlock("init");
837  ExitBlock = createBasicBlock("exit");
838  EmitCXXGuardedInitBranch(Uninit, InitBlock, ExitBlock,
839  GuardKind::TlsGuard, nullptr);
840  EmitBlock(InitBlock);
841  // Mark as initialized before initializing anything else. If the
842  // initializers use previously-initialized thread_local vars, that's
843  // probably supposed to be OK, but the standard doesn't say.
844  Builder.CreateStore(llvm::ConstantInt::get(GuardVal->getType(),1), Guard);
845 
846  // The guard variable can't ever change again.
847  EmitInvariantStart(
848  Guard.getPointer(),
849  CharUnits::fromQuantity(
850  CGM.getDataLayout().getTypeAllocSize(GuardVal->getType())));
851  }
852 
853  RunCleanupsScope Scope(*this);
854 
855  // When building in Objective-C++ ARC mode, create an autorelease pool
856  // around the global initializers.
857  if (getLangOpts().ObjCAutoRefCount && getLangOpts().CPlusPlus) {
858  llvm::Value *token = EmitObjCAutoreleasePoolPush();
859  EmitObjCAutoreleasePoolCleanup(token);
860  }
861 
862  for (unsigned i = 0, e = Decls.size(); i != e; ++i)
863  if (Decls[i])
864  EmitRuntimeCall(Decls[i]);
865 
866  Scope.ForceCleanup();
867 
868  if (ExitBlock) {
869  Builder.CreateBr(ExitBlock);
870  EmitBlock(ExitBlock);
871  }
872  }
873 
874  FinishFunction();
875 }
