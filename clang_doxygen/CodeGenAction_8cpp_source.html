280  for (const CodeGenOptions::BitcodeFileToLink &F :

282  auto BCBuf = FileMgr.getBufferForFile(F.Filename);
283  if (!BCBuf) {

294  if (!ModuleOrErr) {

296  Diags.Report(diag::err_cannot_open_file) << F.Filename

309  // Links each entry in LinkModules into our module. Returns true on error.
310  bool LinkInModules(llvm::Module *M) {
311  for (auto &LM : LinkModules) {
312  assert(LM.Module && "LinkModule does not actually have a module");
313  if (LM.PropagateAttrs)
314  for (Function &F : *LM.Module) {
315  // Skip intrinsics. Keep consistent with how intrinsics are created
316  // in LLVM IR.
317  if (F.isIntrinsic())
318  continue;
319  CodeGen::mergeDefaultFunctionDefinitionAttributes(
320  F, CodeGenOpts, LangOpts, TargetOpts, LM.Internalize);
321  }
322 
323  CurLinkModule = LM.Module.get();
324 
325  bool Err;
326  if (LM.Internalize) {
327  Err = Linker::linkModules(
328  *M, std::move(LM.Module), LM.LinkFlags,
329  [](llvm::Module &M, const llvm::StringSet<> &GVS) {
330  internalizeModule(M, [&GVS](const llvm::GlobalValue &GV) {
331  return !GV.hasName() || (GVS.count(GV.getName()) == 0);
332  });
333  });
334  } else
335  Err = Linker::linkModules(*M, std::move(LM.Module), LM.LinkFlags);
336 
337  if (Err)
338  return true;
339  }
340 
341  LinkModules.clear();
342  return false; // success
343  }

345  void HandleTranslationUnit(ASTContext &C) override {
346  {
347  llvm::TimeTraceScope TimeScope("Frontend");
348  PrettyStackTraceString CrashInfo("Per-file LLVM IR generation");
349  if (TimerIsEnabled) {
350  LLVMIRGenerationRefCount += 1;
351  if (LLVMIRGenerationRefCount == 1)
352  LLVMIRGeneration.startTimer();
353  }
354 
355  Gen->HandleTranslationUnit(C);
356 
357  if (TimerIsEnabled) {
358  LLVMIRGenerationRefCount -= 1;
359  if (LLVMIRGenerationRefCount == 0)
360  LLVMIRGeneration.stopTimer();
361  }
362 
363  IRGenFinished = true;
364  }
365 
366  // Silently ignore if we weren't initialized for some reason.
367  if (!getModule())
368  return;
369 
370  LLVMContext &Ctx = getModule()->getContext();
371 
372  std::unique_ptr<DiagnosticHandler> OldDiagnosticHandler =
373  Ctx.getDiagnosticHandler();
374  Ctx.setDiagnosticHandler(std::make_unique<ClangDiagnosticHandler>(
375  CodeGenOpts, this));
376  // The diagnostic handler is now processed in OptRecordFileRAII.
377 
378  // The parallel_for_work_group legalization pass can emit calls to
379  // builtins function. Definitions of those builtins can be provided in
380  // LinkModule. We force the pass to legalize the code before the link
381  // happens.
382  if (LangOpts.SYCLIsDevice) {
383  PrettyStackTraceString CrashInfo("Pre-linking SYCL passes");
384 
385  FunctionAnalysisManager FAM;
386  ModuleAnalysisManager MAM;
387  MAM.registerPass([&] { return PassInstrumentationAnalysis(); });
388  MAM.registerPass(
389  [&] { return FunctionAnalysisManagerModuleProxy(FAM); });
390  FAM.registerPass(
391  [&] { return ModuleAnalysisManagerFunctionProxy(MAM); });
392 
393  ModulePassManager PreLinkingSyclPasses;
394  PreLinkingSyclPasses.addPass(
395  createModuleToFunctionPassAdaptor(SYCLLowerWGScopePass()));
396  PreLinkingSyclPasses.run(*getModule(), MAM);
397  }
398 
399  if (CodeGenOpts.MisExpect) {
400  Ctx.setMisExpectWarningRequested(true);
401  }
402 
403  if (CodeGenOpts.DiagnosticsMisExpectTolerance) {
404  Ctx.setDiagnosticsMisExpectTolerance(
405  CodeGenOpts.DiagnosticsMisExpectTolerance);
406  }
407 
408  // Link each LinkModule into our module.
409  if (LinkInModules(getModule()))
410  return;
411 
412  for (auto &F : getModule()->functions()) {
413  if (const Decl *FD = Gen->GetDeclForMangledName(F.getName())) {
414  auto Loc = FD->getASTContext().getFullLoc(FD->getLocation());
415  // TODO: use a fast content hash when available.
416  auto NameHash = llvm::hash_value(F.getName());
417  ManglingFullSourceLocs.push_back(std::make_pair(NameHash, Loc));
418  }
419  }
420 
421  if (CodeGenOpts.ClearASTBeforeBackend) {
422  LLVM_DEBUG(llvm::dbgs() << "Clearing AST...\n");
423  // Access to the AST is no longer available after this.
424  // Other things that the ASTContext manages are still available, e.g.
425  // the SourceManager. It'd be nice if we could separate out all the
426  // things in ASTContext used after this point and null out the
427  // ASTContext, but too many various parts of the ASTContext are still
428  // used in various parts.
429  C.cleanup();
430  C.getAllocator().Reset();
431  }
432 
433  EmbedBitcode(getModule(), CodeGenOpts, llvm::MemoryBufferRef());
434 
435  EmitBackendOutput(Diags, HeaderSearchOpts, CodeGenOpts, TargetOpts,
436  LangOpts, C.getTargetInfo().getDataLayoutString(),
437  getModule(), Action, FS, std::move(AsmOutStream));
438 
439  Ctx.setDiagnosticHandler(std::move(OldDiagnosticHandler));
440  }

526 /// ConvertBackendLocation - Convert a location in a temporary llvm::SourceMgr
527 /// buffer to be a valid FullSourceLoc.
528 static FullSourceLoc ConvertBackendLocation(const llvm::SMDiagnostic &D,
529  SourceManager &CSM) {
530  // Get both the clang and llvm source managers. The location is relative to
531  // a memory buffer that the LLVM Source Manager is handling, we need to add
532  // a copy to the Clang source manager.
533  const llvm::SourceMgr &LSM = *D.getSourceMgr();
534 
535  // We need to copy the underlying LLVM memory buffer because llvm::SourceMgr
536  // already owns its one and clang::SourceManager wants to own its one.
537  const MemoryBuffer *LBuf =
538  LSM.getMemoryBuffer(LSM.FindBufferContainingLoc(D.getLoc()));
539 
540  // Create the copy and transfer ownership to clang::SourceManager.
541  // TODO: Avoid copying files into memory.
542  std::unique_ptr<llvm::MemoryBuffer> CBuf =
543  llvm::MemoryBuffer::getMemBufferCopy(LBuf->getBuffer(),
544  LBuf->getBufferIdentifier());
545  // FIXME: Keep a file ID map instead of creating new IDs for each location.
546  FileID FID = CSM.createFileID(std::move(CBuf));
547 
548  // Translate the offset into the file.
549  unsigned Offset = D.getLoc().getPointer() - LBuf->getBufferStart();
550  SourceLocation NewLoc =
551  CSM.getLocForStartOfFile(FID).getLocWithOffset(Offset);
552  return FullSourceLoc(NewLoc, CSM);
553 }

620  // If this problem has clang-level source location information, report the
621  // issue in the source with a note showing the instantiated
622  // code.
623  if (DI.isInlineAsmDiag()) {
624  SourceLocation LocCookie =
625  SourceLocation::getFromRawEncoding(DI.getLocCookie());
626  if (LocCookie.isValid()) {
627  Diags.Report(LocCookie, DiagID).AddString(Message);
628 
629  if (D.getLoc().isValid()) {
630  DiagnosticBuilder B = Diags.Report(Loc, diag::note_fe_inline_asm_here);
631  // Convert the SMDiagnostic ranges into SourceRange and attach them
632  // to the diagnostic.
633  for (const std::pair<unsigned, unsigned> &Range : D.getRanges()) {
634  unsigned Column = D.getColumnNo();
635  B << SourceRange(Loc.getLocWithOffset(Range.first - Column),
636  Loc.getLocWithOffset(Range.second - Column));
637  }
638  }
639  return;
640  }
641  }
642 
643  // Otherwise, report the backend issue as occurring in the generated .s file.
644  // If Loc is invalid, we still need to report the issue, it just gets no
645  // location info.
646  Diags.Report(Loc, DiagID).AddString(Message);
647 }

705 const FullSourceLoc BackendConsumer::getBestLocationFromDebugLoc(
706  const llvm::DiagnosticInfoWithLocationBase &D, bool &BadDebugInfo,
707  StringRef &Filename, unsigned &Line, unsigned &Column) const {
708  SourceManager &SourceMgr = Context->getSourceManager();
709  FileManager &FileMgr = SourceMgr.getFileManager();
710  SourceLocation DILoc;
711 
712  if (D.isLocationAvailable()) {
713  D.getLocation(Filename, Line, Column);
714  if (Line > 0) {
715  auto FE = FileMgr.getFile(Filename);
716  if (!FE)
717  FE = FileMgr.getFile(D.getAbsolutePath());
718  if (FE) {
719  // If -gcolumn-info was not used, Column will be 0. This upsets the
720  // source manager, so pass 1 if Column is not set.
721  DILoc = SourceMgr.translateFileLineCol(*FE, Line, Column ? Column : 1);
722  }
723  }
724  BadDebugInfo = DILoc.isInvalid();
725  }
726 
727  // If a location isn't available, try to approximate it using the associated
728  // function definition. We use the definition's right brace to differentiate
729  // from diagnostics that genuinely relate to the function itself.
730  FullSourceLoc Loc(DILoc, SourceMgr);
731  if (Loc.isInvalid()) {
732  if (auto MaybeLoc = getFunctionSourceLocation(D.getFunction()))
733  Loc = *MaybeLoc;
734  }
735 
736  if (DILoc.isInvalid() && D.isLocationAvailable())
737  // If we were not able to translate the file:line:col information
738  // back to a SourceLocation, at least emit a note stating that
739  // we could not translate this location. This can happen in the
740  // case of #line directives.
741  Diags.Report(Loc, diag::note_fe_backend_invalid_loc)
742  << Filename << Line << Column;
743 
744  return Loc;
745 }

946 /// This function is invoked when the backend needs
947 /// to report something to the user.
948 void BackendConsumer::DiagnosticHandlerImpl(const DiagnosticInfo &DI) {
949  unsigned DiagID = diag::err_fe_inline_asm;
950  llvm::DiagnosticSeverity Severity = DI.getSeverity();
951  // Get the diagnostic ID based.
952  switch (DI.getKind()) {
953  case llvm::DK_InlineAsm:
954  if (InlineAsmDiagHandler(cast<DiagnosticInfoInlineAsm>(DI)))
955  return;
956  ComputeDiagID(Severity, inline_asm, DiagID);
957  break;
958  case llvm::DK_SrcMgr:
959  SrcMgrDiagHandler(cast<DiagnosticInfoSrcMgr>(DI));
960  return;
961  case llvm::DK_StackSize:
962  if (StackSizeDiagHandler(cast<DiagnosticInfoStackSize>(DI)))
963  return;
964  ComputeDiagID(Severity, backend_frame_larger_than, DiagID);
965  break;
966  case llvm::DK_ResourceLimit:
967  if (ResourceLimitDiagHandler(cast<DiagnosticInfoResourceLimit>(DI)))
968  return;
969  ComputeDiagID(Severity, backend_resource_limit, DiagID);
970  break;
971  case DK_Linker:
972  ComputeDiagID(Severity, linking_module, DiagID);
973  break;
974  case llvm::DK_OptimizationRemark:
975  // Optimization remarks are always handled completely by this
976  // handler. There is no generic way of emitting them.
977  OptimizationRemarkHandler(cast<OptimizationRemark>(DI));
978  return;
979  case llvm::DK_OptimizationRemarkMissed:
980  // Optimization remarks are always handled completely by this
981  // handler. There is no generic way of emitting them.
982  OptimizationRemarkHandler(cast<OptimizationRemarkMissed>(DI));
983  return;
984  case llvm::DK_OptimizationRemarkAnalysis:
985  // Optimization remarks are always handled completely by this
986  // handler. There is no generic way of emitting them.
987  OptimizationRemarkHandler(cast<OptimizationRemarkAnalysis>(DI));
988  return;
989  case llvm::DK_OptimizationRemarkAnalysisFPCommute:
990  // Optimization remarks are always handled completely by this
991  // handler. There is no generic way of emitting them.
992  OptimizationRemarkHandler(cast<OptimizationRemarkAnalysisFPCommute>(DI));
993  return;
994  case llvm::DK_OptimizationRemarkAnalysisAliasing:
995  // Optimization remarks are always handled completely by this
996  // handler. There is no generic way of emitting them.
997  OptimizationRemarkHandler(cast<OptimizationRemarkAnalysisAliasing>(DI));
998  return;
999  case llvm::DK_MachineOptimizationRemark:
1000  // Optimization remarks are always handled completely by this
1001  // handler. There is no generic way of emitting them.
1002  OptimizationRemarkHandler(cast<MachineOptimizationRemark>(DI));
1003  return;
1004  case llvm::DK_MachineOptimizationRemarkMissed:
1005  // Optimization remarks are always handled completely by this
1006  // handler. There is no generic way of emitting them.
1007  OptimizationRemarkHandler(cast<MachineOptimizationRemarkMissed>(DI));
1008  return;
1009  case llvm::DK_MachineOptimizationRemarkAnalysis:
1010  // Optimization remarks are always handled completely by this
1011  // handler. There is no generic way of emitting them.
1012  OptimizationRemarkHandler(cast<MachineOptimizationRemarkAnalysis>(DI));
1013  return;
1014  case llvm::DK_OptimizationFailure:
1015  // Optimization failures are always handled completely by this
1016  // handler.
1017  OptimizationFailureHandler(cast<DiagnosticInfoOptimizationFailure>(DI));
1018  return;
1019  case llvm::DK_Unsupported:
1020  UnsupportedDiagHandler(cast<DiagnosticInfoUnsupported>(DI));
1021  return;
1022  case llvm::DK_DontCall:
1023  DontCallDiagHandler(cast<DiagnosticInfoDontCall>(DI));
1024  return;
1025  case llvm::DK_MisExpect:
1026  MisExpectDiagHandler(cast<DiagnosticInfoMisExpect>(DI));
1027  return;
1028  case llvm::DK_AspectMismatch:
1029  AspectMismatchDiagHandler(cast<DiagnosticInfoAspectsMismatch>(DI));
1030  return;
1031  default:
1032  // Plugin IDs are not bound to any value as they are set dynamically.
1033  ComputeDiagRemarkID(Severity, backend_plugin, DiagID);
1034  break;
1035  }
1036  std::string MsgStorage;
1037  {
1038  raw_string_ostream Stream(MsgStorage);
1039  DiagnosticPrinterRawOStream DP(Stream);
1040  DI.print(DP);
1041  }
1042 
1043  if (DI.getKind() == DK_Linker) {
1044  assert(CurLinkModule && "CurLinkModule must be set for linker diagnostics");
1045  Diags.Report(DiagID) << CurLinkModule->getModuleIdentifier() << MsgStorage;
1046  return;
1047  }
1048 
1049  // Report the backend message using the usual diagnostic mechanism.
1050  FullSourceLoc Loc;
1051  Diags.Report(Loc, DiagID).AddString(MsgStorage);
1052 }
