32 StmtResult SemaObjC::ActOnObjCForCollectionStmt(SourceLocation ForLoc,
33  Stmt *First, Expr *collection,
34  SourceLocation RParenLoc) {
35  ASTContext &Context = getASTContext();
36  SemaRef.setFunctionHasBranchProtectedScope();
37 
38  ExprResult CollectionExprResult =
39  CheckObjCForCollectionOperand(ForLoc, collection);
40 
41  if (First) {
42  QualType FirstType;
43  if (DeclStmt *DS = dyn_cast<DeclStmt>(First)) {
44  if (!DS->isSingleDecl())
45  return StmtError(Diag((*DS->decl_begin())->getLocation(),
46  diag::err_toomany_element_decls));
47 
48  VarDecl *D = dyn_cast<VarDecl>(DS->getSingleDecl());
49  if (!D || D->isInvalidDecl())
50  return StmtError();
51 
52  FirstType = D->getType();
53  // C99 6.8.5p3: The declaration part of a 'for' statement shall only
54  // declare identifiers for objects having storage class 'auto' or
55  // 'register'.
56  if (!D->hasLocalStorage())
57  return StmtError(
58  Diag(D->getLocation(), diag::err_non_local_variable_decl_in_for));
59 
60  // If the type contained 'auto', deduce the 'auto' to 'id'.
61  if (FirstType->getContainedAutoType()) {
62  SourceLocation Loc = D->getLocation();
63  OpaqueValueExpr OpaqueId(Loc, Context.getObjCIdType(), VK_PRValue);
64  Expr *DeducedInit = &OpaqueId;
65  sema::TemplateDeductionInfo Info(Loc);
66  FirstType = QualType();
67  TemplateDeductionResult Result = SemaRef.DeduceAutoType(
68  D->getTypeSourceInfo()->getTypeLoc(), DeducedInit, FirstType, Info);
69  if (Result != TemplateDeductionResult::Success &&
70  Result != TemplateDeductionResult::AlreadyDiagnosed)
71  SemaRef.DiagnoseAutoDeductionFailure(D, DeducedInit);
72  if (FirstType.isNull()) {
73  D->setInvalidDecl();
74  return StmtError();
75  }
76 
77  D->setType(FirstType);
78 
79  if (!SemaRef.inTemplateInstantiation()) {
80  SourceLocation Loc =
81  D->getTypeSourceInfo()->getTypeLoc().getBeginLoc();
82  Diag(Loc, diag::warn_auto_var_is_id) << D->getDeclName();
83  }
84  }
85 
86  } else {
87  Expr *FirstE = cast<Expr>(First);
88  if (!FirstE->isTypeDependent() && !FirstE->isLValue())
89  return StmtError(
90  Diag(First->getBeginLoc(), diag::err_selector_element_not_lvalue)
91  << First->getSourceRange());
92 
93  FirstType = static_cast<Expr *>(First)->getType();
94  if (FirstType.isConstQualified())
95  Diag(ForLoc, diag::err_selector_element_const_type)
96  << FirstType << First->getSourceRange();
97  }
98  if (!FirstType->isDependentType() &&
99  !FirstType->isObjCObjectPointerType() &&
100  !FirstType->isBlockPointerType())
101  return StmtError(Diag(ForLoc, diag::err_selector_element_type)
102  << FirstType << First->getSourceRange());
103  }
104 
105  if (CollectionExprResult.isInvalid())
106  return StmtError();
107 
108  CollectionExprResult = SemaRef.ActOnFinishFullExpr(CollectionExprResult.get(),
109  /*DiscardedValue*/ false);
110  if (CollectionExprResult.isInvalid())
111  return StmtError();
112 
113  return new (Context) ObjCForCollectionStmt(First, CollectionExprResult.get(),
114  nullptr, ForLoc, RParenLoc);
115 }

117 ExprResult SemaObjC::CheckObjCForCollectionOperand(SourceLocation forLoc,
118  Expr *collection) {
119  ASTContext &Context = getASTContext();
120  if (!collection)
121  return ExprError();
122 
123  ExprResult result = SemaRef.CorrectDelayedTyposInExpr(collection);
124  if (!result.isUsable())
125  return ExprError();
126  collection = result.get();
127 
128  // Bail out early if we've got a type-dependent expression.
129  if (collection->isTypeDependent())
130  return collection;
131 
132  // Perform normal l-value conversion.
133  result = SemaRef.DefaultFunctionArrayLvalueConversion(collection);
134  if (result.isInvalid())
135  return ExprError();
136  collection = result.get();
137 
138  // The operand needs to have object-pointer type.
139  // TODO: should we do a contextual conversion?
140  const ObjCObjectPointerType *pointerType =
141  collection->getType()->getAs<ObjCObjectPointerType>();
142  if (!pointerType)
143  return Diag(forLoc, diag::err_collection_expr_type)
144  << collection->getType() << collection->getSourceRange();
145 
146  // Check that the operand provides
147  // - countByEnumeratingWithState:objects:count:
148  const ObjCObjectType *objectType = pointerType->getObjectType();
149  ObjCInterfaceDecl *iface = objectType->getInterface();
150 
151  // If we have a forward-declared type, we can't do this check.
152  // Under ARC, it is an error not to have a forward-declared class.
153  if (iface &&
154  (getLangOpts().ObjCAutoRefCount
155  ? SemaRef.RequireCompleteType(forLoc, QualType(objectType, 0),
156  diag::err_arc_collection_forward,
157  collection)
158  : !SemaRef.isCompleteType(forLoc, QualType(objectType, 0)))) {
159  // Otherwise, if we have any useful type information, check that
160  // the type declares the appropriate method.
161  } else if (iface || !objectType->qual_empty()) {
162  const IdentifierInfo *selectorIdents[] = {
163  &Context.Idents.get("countByEnumeratingWithState"),
164  &Context.Idents.get("objects"), &Context.Idents.get("count")};
165  Selector selector = Context.Selectors.getSelector(3, &selectorIdents[0]);
166 
167  ObjCMethodDecl *method = nullptr;
168 
169  // If there's an interface, look in both the public and private APIs.
170  if (iface) {
171  method = iface->lookupInstanceMethod(selector);
172  if (!method)
173  method = iface->lookupPrivateMethod(selector);
174  }
175 
176  // Also check protocol qualifiers.
177  if (!method)
178  method = LookupMethodInQualifiedType(selector, pointerType,
179  /*instance*/ true);
180 
181  // If we didn't find it anywhere, give up.
182  if (!method) {
183  Diag(forLoc, diag::warn_collection_expr_type)
184  << collection->getType() << selector << collection->getSourceRange();
185  }
186 
187  // TODO: check for an incompatible signature?
188  }
189 
190  // Wrap up any cleanups in the expression.
191  return collection;
192 }

266 StmtResult SemaObjC::ActOnObjCAtThrowStmt(SourceLocation AtLoc, Expr *Throw,
267  Scope *CurScope) {
268  if (!getLangOpts().ObjCExceptions)
269  Diag(AtLoc, diag::err_objc_exceptions_disabled) << "@throw";
270 
271  if (!Throw) {
272  // @throw without an expression designates a rethrow (which must occur
273  // in the context of an @catch clause).
274  Scope *AtCatchParent = CurScope;
275  while (AtCatchParent && !AtCatchParent->isAtCatchScope())
276  AtCatchParent = AtCatchParent->getParent();
277  if (!AtCatchParent)
278  return StmtError(Diag(AtLoc, diag::err_rethrow_used_outside_catch));
279  }
280  return BuildObjCAtThrowStmt(AtLoc, Throw);

284  Expr *operand) {
285  ExprResult result = SemaRef.DefaultLvalueConversion(operand);
286  if (result.isInvalid())
287  return ExprError();
288  operand = result.get();
289 
290  // Make sure the expression type is an ObjC pointer or "void *".
291  QualType type = operand->getType();
292  if (!type->isDependentType() && !type->isObjCObjectPointerType()) {
293  const PointerType *pointerType = type->getAs<PointerType>();
294  if (!pointerType || !pointerType->getPointeeType()->isVoidType()) {
295  if (getLangOpts().CPlusPlus) {
296  if (SemaRef.RequireCompleteType(atLoc, type,
297  diag::err_incomplete_receiver_type))
298  return Diag(atLoc, diag::err_objc_synchronized_expects_object)
299  << type << operand->getSourceRange();
300 
301  ExprResult result =
302  SemaRef.PerformContextuallyConvertToObjCPointer(operand);
303  if (result.isInvalid())
304  return ExprError();
305  if (!result.isUsable())
306  return Diag(atLoc, diag::err_objc_synchronized_expects_object)
307  << type << operand->getSourceRange();
308 
309  operand = result.get();
310  } else {
311  return Diag(atLoc, diag::err_objc_synchronized_expects_object)
312  << type << operand->getSourceRange();
313  }
314  }
315  }
316 
317  // The operand to @synchronized is a full-expression.
318  return SemaRef.ActOnFinishFullExpr(operand, /*DiscardedValue*/ false);

337 TypeResult SemaObjC::actOnObjCProtocolQualifierType(
338  SourceLocation lAngleLoc, ArrayRef<Decl *> protocols,
339  ArrayRef<SourceLocation> protocolLocs, SourceLocation rAngleLoc) {
340  ASTContext &Context = getASTContext();
341  // Form id<protocol-list>.
342  QualType Result = Context.getObjCObjectType(
343  Context.ObjCBuiltinIdTy, {},
344  llvm::ArrayRef((ObjCProtocolDecl *const *)protocols.data(),
345  protocols.size()),
346  false);
347  Result = Context.getObjCObjectPointerType(Result);
348 
349  TypeSourceInfo *ResultTInfo = Context.CreateTypeSourceInfo(Result);
350  TypeLoc ResultTL = ResultTInfo->getTypeLoc();
351 
352  auto ObjCObjectPointerTL = ResultTL.castAs<ObjCObjectPointerTypeLoc>();
353  ObjCObjectPointerTL.setStarLoc(SourceLocation()); // implicit
354 
355  auto ObjCObjectTL =
356  ObjCObjectPointerTL.getPointeeLoc().castAs<ObjCObjectTypeLoc>();
357  ObjCObjectTL.setHasBaseTypeAsWritten(false);
358  ObjCObjectTL.getBaseLoc().initialize(Context, SourceLocation());
359 
360  // No type arguments.
361  ObjCObjectTL.setTypeArgsLAngleLoc(SourceLocation());
362  ObjCObjectTL.setTypeArgsRAngleLoc(SourceLocation());
363 
364  // Fill in protocol qualifiers.
365  ObjCObjectTL.setProtocolLAngleLoc(lAngleLoc);
366  ObjCObjectTL.setProtocolRAngleLoc(rAngleLoc);
367  for (unsigned i = 0, n = protocols.size(); i != n; ++i)
368  ObjCObjectTL.setProtocolLoc(i, protocolLocs[i]);
369 
370  // We're done. Return the completed type to the parser.
371  return SemaRef.CreateParsedType(Result, ResultTInfo);
372 }

374 TypeResult SemaObjC::actOnObjCTypeArgsAndProtocolQualifiers(
375  Scope *S, SourceLocation Loc, ParsedType BaseType,
376  SourceLocation TypeArgsLAngleLoc, ArrayRef<ParsedType> TypeArgs,
377  SourceLocation TypeArgsRAngleLoc, SourceLocation ProtocolLAngleLoc,
378  ArrayRef<Decl *> Protocols, ArrayRef<SourceLocation> ProtocolLocs,
379  SourceLocation ProtocolRAngleLoc) {
380  ASTContext &Context = getASTContext();
381  TypeSourceInfo *BaseTypeInfo = nullptr;
382  QualType T = SemaRef.GetTypeFromParser(BaseType, &BaseTypeInfo);
383  if (T.isNull())
384  return true;
385 
386  // Handle missing type-source info.
387  if (!BaseTypeInfo)
388  BaseTypeInfo = Context.getTrivialTypeSourceInfo(T, Loc);
389 
390  // Extract type arguments.
391  SmallVector<TypeSourceInfo *, 4> ActualTypeArgInfos;
392  for (unsigned i = 0, n = TypeArgs.size(); i != n; ++i) {
393  TypeSourceInfo *TypeArgInfo = nullptr;
394  QualType TypeArg = SemaRef.GetTypeFromParser(TypeArgs[i], &TypeArgInfo);
395  if (TypeArg.isNull()) {
396  ActualTypeArgInfos.clear();
397  break;
398  }
399 
400  assert(TypeArgInfo && "No type source info?");
401  ActualTypeArgInfos.push_back(TypeArgInfo);
402  }
403 
404  // Build the object type.
405  QualType Result = BuildObjCObjectType(
406  T, BaseTypeInfo->getTypeLoc().getSourceRange().getBegin(),
407  TypeArgsLAngleLoc, ActualTypeArgInfos, TypeArgsRAngleLoc,
408  ProtocolLAngleLoc,
409  llvm::ArrayRef((ObjCProtocolDecl *const *)Protocols.data(),
410  Protocols.size()),
411  ProtocolLocs, ProtocolRAngleLoc,
412  /*FailOnError=*/false,
413  /*Rebuilding=*/false);
414 
415  if (Result == T)
416  return BaseType;
417 
418  // Create source information for this type.
419  TypeSourceInfo *ResultTInfo = Context.CreateTypeSourceInfo(Result);
420  TypeLoc ResultTL = ResultTInfo->getTypeLoc();
421 
422  // For id<Proto1, Proto2> or Class<Proto1, Proto2>, we'll have an
423  // object pointer type. Fill in source information for it.
424  if (auto ObjCObjectPointerTL = ResultTL.getAs<ObjCObjectPointerTypeLoc>()) {
425  // The '*' is implicit.
426  ObjCObjectPointerTL.setStarLoc(SourceLocation());
427  ResultTL = ObjCObjectPointerTL.getPointeeLoc();
428  }
429 
430  if (auto OTPTL = ResultTL.getAs<ObjCTypeParamTypeLoc>()) {
431  // Protocol qualifier information.
432  if (OTPTL.getNumProtocols() > 0) {
433  assert(OTPTL.getNumProtocols() == Protocols.size());
434  OTPTL.setProtocolLAngleLoc(ProtocolLAngleLoc);
435  OTPTL.setProtocolRAngleLoc(ProtocolRAngleLoc);
436  for (unsigned i = 0, n = Protocols.size(); i != n; ++i)
437  OTPTL.setProtocolLoc(i, ProtocolLocs[i]);
438  }
439 
440  // We're done. Return the completed type to the parser.
441  return SemaRef.CreateParsedType(Result, ResultTInfo);
442  }
443 
444  auto ObjCObjectTL = ResultTL.castAs<ObjCObjectTypeLoc>();
445 
446  // Type argument information.
447  if (ObjCObjectTL.getNumTypeArgs() > 0) {
448  assert(ObjCObjectTL.getNumTypeArgs() == ActualTypeArgInfos.size());
449  ObjCObjectTL.setTypeArgsLAngleLoc(TypeArgsLAngleLoc);
450  ObjCObjectTL.setTypeArgsRAngleLoc(TypeArgsRAngleLoc);
451  for (unsigned i = 0, n = ActualTypeArgInfos.size(); i != n; ++i)
452  ObjCObjectTL.setTypeArgTInfo(i, ActualTypeArgInfos[i]);
453  } else {
454  ObjCObjectTL.setTypeArgsLAngleLoc(SourceLocation());
455  ObjCObjectTL.setTypeArgsRAngleLoc(SourceLocation());
456  }
457 
458  // Protocol qualifier information.
459  if (ObjCObjectTL.getNumProtocols() > 0) {
460  assert(ObjCObjectTL.getNumProtocols() == Protocols.size());
461  ObjCObjectTL.setProtocolLAngleLoc(ProtocolLAngleLoc);
462  ObjCObjectTL.setProtocolRAngleLoc(ProtocolRAngleLoc);
463  for (unsigned i = 0, n = Protocols.size(); i != n; ++i)
464  ObjCObjectTL.setProtocolLoc(i, ProtocolLocs[i]);
465  } else {
466  ObjCObjectTL.setProtocolLAngleLoc(SourceLocation());
467  ObjCObjectTL.setProtocolRAngleLoc(SourceLocation());
468  }
469 
470  // Base type.
471  ObjCObjectTL.setHasBaseTypeAsWritten(true);
472  if (ObjCObjectTL.getType() == T)
473  ObjCObjectTL.getBaseLoc().initializeFullCopy(BaseTypeInfo->getTypeLoc());
474  else
475  ObjCObjectTL.getBaseLoc().initialize(Context, Loc);
476 
477  // We're done. Return the completed type to the parser.
478  return SemaRef.CreateParsedType(Result, ResultTInfo);
479 }

504 /// Apply Objective-C type arguments to the given type.
505 static QualType applyObjCTypeArgs(Sema &S, SourceLocation loc, QualType type,
506  ArrayRef<TypeSourceInfo *> typeArgs,
507  SourceRange typeArgsRange, bool failOnError,
508  bool rebuilding) {
509  // We can only apply type arguments to an Objective-C class type.
510  const auto *objcObjectType = type->getAs<ObjCObjectType>();
511  if (!objcObjectType || !objcObjectType->getInterface()) {
512  S.Diag(loc, diag::err_objc_type_args_non_class) << type << typeArgsRange;
513 
514  if (failOnError)
515  return QualType();
516  return type;
517  }
518 
519  // The class type must be parameterized.
520  ObjCInterfaceDecl *objcClass = objcObjectType->getInterface();
521  ObjCTypeParamList *typeParams = objcClass->getTypeParamList();
522  if (!typeParams) {
523  S.Diag(loc, diag::err_objc_type_args_non_parameterized_class)
524  << objcClass->getDeclName() << FixItHint::CreateRemoval(typeArgsRange);
525 
526  if (failOnError)
527  return QualType();
528 
529  return type;
530  }
531 
532  // The type must not already be specialized.
533  if (objcObjectType->isSpecialized()) {
534  S.Diag(loc, diag::err_objc_type_args_specialized_class)
535  << type << FixItHint::CreateRemoval(typeArgsRange);
536 
537  if (failOnError)
538  return QualType();
539 
540  return type;
541  }
542 
543  // Check the type arguments.
544  SmallVector<QualType, 4> finalTypeArgs;
545  unsigned numTypeParams = typeParams->size();
546  bool anyPackExpansions = false;
547  for (unsigned i = 0, n = typeArgs.size(); i != n; ++i) {
548  TypeSourceInfo *typeArgInfo = typeArgs[i];
549  QualType typeArg = typeArgInfo->getType();
550 
551  // Type arguments cannot have explicit qualifiers or nullability.
552  // We ignore indirect sources of these, e.g. behind typedefs or
553  // template arguments.
554  if (TypeLoc qual = typeArgInfo->getTypeLoc().findExplicitQualifierLoc()) {
555  bool diagnosed = false;
556  SourceRange rangeToRemove;
557  if (auto attr = qual.getAs<AttributedTypeLoc>()) {
558  rangeToRemove = attr.getLocalSourceRange();
559  if (attr.getTypePtr()->getImmediateNullability()) {
560  typeArg = attr.getTypePtr()->getModifiedType();
561  S.Diag(attr.getBeginLoc(),
562  diag::err_objc_type_arg_explicit_nullability)
563  << typeArg << FixItHint::CreateRemoval(rangeToRemove);
564  diagnosed = true;
565  }
566  }
567 
568  // When rebuilding, qualifiers might have gotten here through a
569  // final substitution.
570  if (!rebuilding && !diagnosed) {
571  S.Diag(qual.getBeginLoc(), diag::err_objc_type_arg_qualified)
572  << typeArg << typeArg.getQualifiers().getAsString()
573  << FixItHint::CreateRemoval(rangeToRemove);
574  }
575  }
576 
577  // Remove qualifiers even if they're non-local.
578  typeArg = typeArg.getUnqualifiedType();
579 
580  finalTypeArgs.push_back(typeArg);
581 
582  if (typeArg->getAs<PackExpansionType>())
583  anyPackExpansions = true;
584 
585  // Find the corresponding type parameter, if there is one.
586  ObjCTypeParamDecl *typeParam = nullptr;
587  if (!anyPackExpansions) {
588  if (i < numTypeParams) {
589  typeParam = typeParams->begin()[i];
590  } else {
591  // Too many arguments.
592  S.Diag(loc, diag::err_objc_type_args_wrong_arity)
593  << false << objcClass->getDeclName() << (unsigned)typeArgs.size()
594  << numTypeParams;
595  S.Diag(objcClass->getLocation(), diag::note_previous_decl) << objcClass;
596 
597  if (failOnError)
598  return QualType();
599 
600  return type;
601  }
602  }
603 
604  // Objective-C object pointer types must be substitutable for the bounds.
605  if (const auto *typeArgObjC = typeArg->getAs<ObjCObjectPointerType>()) {
606  // If we don't have a type parameter to match against, assume
607  // everything is fine. There was a prior pack expansion that
608  // means we won't be able to match anything.
609  if (!typeParam) {
610  assert(anyPackExpansions && "Too many arguments?");
611  continue;
612  }
613 
614  // Retrieve the bound.
615  QualType bound = typeParam->getUnderlyingType();
616  const auto *boundObjC = bound->castAs<ObjCObjectPointerType>();
617 
618  // Determine whether the type argument is substitutable for the bound.
619  if (typeArgObjC->isObjCIdType()) {
620  // When the type argument is 'id', the only acceptable type
621  // parameter bound is 'id'.
622  if (boundObjC->isObjCIdType())
623  continue;
624  } else if (S.Context.canAssignObjCInterfaces(boundObjC, typeArgObjC)) {
625  // Otherwise, we follow the assignability rules.
626  continue;
627  }
628 
629  // Diagnose the mismatch.
630  S.Diag(typeArgInfo->getTypeLoc().getBeginLoc(),
631  diag::err_objc_type_arg_does_not_match_bound)
632  << typeArg << bound << typeParam->getDeclName();
633  S.Diag(typeParam->getLocation(), diag::note_objc_type_param_here)
634  << typeParam->getDeclName();
635 
636  if (failOnError)
637  return QualType();
638 
639  return type;
640  }
641 
642  // Block pointer types are permitted for unqualified 'id' bounds.
643  if (typeArg->isBlockPointerType()) {
644  // If we don't have a type parameter to match against, assume
645  // everything is fine. There was a prior pack expansion that
646  // means we won't be able to match anything.
647  if (!typeParam) {
648  assert(anyPackExpansions && "Too many arguments?");
649  continue;
650  }
651 
652  // Retrieve the bound.
653  QualType bound = typeParam->getUnderlyingType();
654  if (bound->isBlockCompatibleObjCPointerType(S.Context))
655  continue;
656 
657  // Diagnose the mismatch.
658  S.Diag(typeArgInfo->getTypeLoc().getBeginLoc(),
659  diag::err_objc_type_arg_does_not_match_bound)
660  << typeArg << bound << typeParam->getDeclName();
661  S.Diag(typeParam->getLocation(), diag::note_objc_type_param_here)
662  << typeParam->getDeclName();
663 
664  if (failOnError)
665  return QualType();
666 
667  return type;
668  }
669 
670  // Types that have __attribute__((NSObject)) are permitted.
671  if (typeArg->isObjCNSObjectType()) {
672  continue;
673  }
674 
675  // Dependent types will be checked at instantiation time.
676  if (typeArg->isDependentType()) {
677  continue;
678  }
679 
680  // Diagnose non-id-compatible type arguments.
681  S.Diag(typeArgInfo->getTypeLoc().getBeginLoc(),
682  diag::err_objc_type_arg_not_id_compatible)
683  << typeArg << typeArgInfo->getTypeLoc().getSourceRange();
684 
685  if (failOnError)
686  return QualType();
687 
688  return type;
689  }
690 
691  // Make sure we didn't have the wrong number of arguments.
692  if (!anyPackExpansions && finalTypeArgs.size() != numTypeParams) {
693  S.Diag(loc, diag::err_objc_type_args_wrong_arity)
694  << (typeArgs.size() < typeParams->size()) << objcClass->getDeclName()
695  << (unsigned)finalTypeArgs.size() << (unsigned)numTypeParams;
696  S.Diag(objcClass->getLocation(), diag::note_previous_decl) << objcClass;
697 
698  if (failOnError)
699  return QualType();
700 
701  return type;
702  }
703 
704  // Success. Form the specialized type.
705  return S.Context.getObjCObjectType(type, finalTypeArgs, {}, false);
706 }

770 static bool considerVariable(VarDecl *var, Expr *ref, RetainCycleOwner &owner) {
771  // In ARC, it's captured strongly iff the variable has __strong
772  // lifetime. In MRR, it's captured strongly if the variable is
773  // __block and has an appropriate type.
774  if (var->getType().getObjCLifetime() != Qualifiers::OCL_Strong)

783 static bool findRetainCycleOwner(Sema &S, Expr *e, RetainCycleOwner &owner) {
784  while (true) {
785  e = e->IgnoreParens();
786  if (CastExpr *cast = dyn_cast<CastExpr>(e)) {
787  switch (cast->getCastKind()) {
788  case CK_BitCast:
789  case CK_LValueBitCast:
790  case CK_LValueToRValue:
791  case CK_ARCReclaimReturnedObject:
792  e = cast->getSubExpr();
793  continue;
794 
795  default:
796  return false;
797  }
798  }
799 
800  if (ObjCIvarRefExpr *ref = dyn_cast<ObjCIvarRefExpr>(e)) {
801  ObjCIvarDecl *ivar = ref->getDecl();
802  if (ivar->getType().getObjCLifetime() != Qualifiers::OCL_Strong)
803  return false;
804 
805  // Try to find a retain cycle in the base.
806  if (!findRetainCycleOwner(S, ref->getBase(), owner))
807  return false;
808 
809  if (ref->isFreeIvar())
810  owner.setLocsFrom(ref);
811  owner.Indirect = true;
812  return true;
813  }
814 
815  if (DeclRefExpr *ref = dyn_cast<DeclRefExpr>(e)) {
816  VarDecl *var = dyn_cast<VarDecl>(ref->getDecl());
817  if (!var)
818  return false;
819  return considerVariable(var, ref, owner);
820  }
821 
822  if (MemberExpr *member = dyn_cast<MemberExpr>(e)) {
823  if (member->isArrow())
824  return false;
825 
826  // Don't count this as an indirect ownership.
827  e = member->getBase();
828  continue;
829  }
830 
831  if (PseudoObjectExpr *pseudo = dyn_cast<PseudoObjectExpr>(e)) {
832  // Only pay attention to pseudo-objects on property references.
833  ObjCPropertyRefExpr *pre = dyn_cast<ObjCPropertyRefExpr>(
834  pseudo->getSyntacticForm()->IgnoreParens());
835  if (!pre)
836  return false;
837  if (pre->isImplicitProperty())
838  return false;
839  ObjCPropertyDecl *property = pre->getExplicitProperty();
840  if (!property->isRetaining() &&
841  !(property->getPropertyIvarDecl() &&
842  property->getPropertyIvarDecl()->getType().getObjCLifetime() ==
843  Qualifiers::OCL_Strong))
844  return false;
845 
846  owner.Indirect = true;
847  if (pre->isSuperReceiver()) {
848  owner.Variable = S.getCurMethodDecl()->getSelfDecl();
849  if (!owner.Variable)
850  return false;
851  owner.Loc = pre->getLocation();
852  owner.Range = pre->getSourceRange();
853  return true;
854  }
855  e = const_cast<Expr *>(
856  cast<OpaqueValueExpr>(pre->getBase())->getSourceExpr());
857  continue;
858  }
859 
860  // Array ivars?
861 
862  return false;
863  }

922 /// Check whether the given argument is a block which captures a
923 /// variable.
924 static Expr *findCapturingExpr(Sema &S, Expr *e, RetainCycleOwner &owner) {
925  assert(owner.Variable && owner.Loc.isValid());
926 
927  e = e->IgnoreParenCasts();
928 
929  // Look through [^{...} copy] and Block_copy(^{...}).
930  if (ObjCMessageExpr *ME = dyn_cast<ObjCMessageExpr>(e)) {
931  Selector Cmd = ME->getSelector();
932  if (Cmd.isUnarySelector() && Cmd.getNameForSlot(0) == "copy") {
933  e = ME->getInstanceReceiver();
934  if (!e)
935  return nullptr;
936  e = e->IgnoreParenCasts();
937  }
938  } else if (CallExpr *CE = dyn_cast<CallExpr>(e)) {
939  if (CE->getNumArgs() == 1) {
940  FunctionDecl *Fn = dyn_cast_or_null<FunctionDecl>(CE->getCalleeDecl());
941  if (Fn) {
942  const IdentifierInfo *FnI = Fn->getIdentifier();
943  if (FnI && FnI->isStr("_Block_copy")) {
944  e = CE->getArg(0)->IgnoreParenCasts();
945  }
946  }
947  }
948  }
949 
950  BlockExpr *block = dyn_cast<BlockExpr>(e);
951  if (!block || !block->getBlockDecl()->capturesVariable(owner.Variable))
952  return nullptr;
953 
954  FindCaptureVisitor visitor(S.Context, owner.Variable);
955  visitor.Visit(block->getBlockDecl()->getBody());
956  return visitor.VarWillBeReased ? nullptr : visitor.Capturer;
957 }

1156 /// Check a message send to see if it's likely to cause a retain cycle.
1157 void SemaObjC::checkRetainCycles(ObjCMessageExpr *msg) {
1158  // Only check instance methods whose selector looks like a setter.
1159  if (!msg->isInstanceMessage() || !isSetterLikeSelector(msg->getSelector()))
1160  return;
1161 
1162  // Try to find a variable that the receiver is strongly owned by.
1163  RetainCycleOwner owner;
1164  if (msg->getReceiverKind() == ObjCMessageExpr::Instance) {
1165  if (!findRetainCycleOwner(SemaRef, msg->getInstanceReceiver(), owner))
1166  return;
1167  } else {
1168  assert(msg->getReceiverKind() == ObjCMessageExpr::SuperInstance);
1169  owner.Variable = SemaRef.getCurMethodDecl()->getSelfDecl();
1170  owner.Loc = msg->getSuperLoc();
1171  owner.Range = msg->getSuperLoc();
1172  }
1173 
1174  // Check whether the receiver is captured by any of the arguments.
1175  const ObjCMethodDecl *MD = msg->getMethodDecl();
1176  for (unsigned i = 0, e = msg->getNumArgs(); i != e; ++i) {
1177  if (Expr *capturer = findCapturingExpr(SemaRef, msg->getArg(i), owner)) {
1178  // noescape blocks should not be retained by the method.
1179  if (MD && MD->parameters()[i]->hasAttr<NoEscapeAttr>())
1180  continue;
1181  return diagnoseRetainCycle(SemaRef, capturer, owner);
1182  }
1183  }

1305 /// Determine whether this is an Objective-C writeback conversion,
1306 /// used for parameter passing when performing automatic reference counting.
1307 ///
1308 /// \param FromType The type we're converting form.
1309 ///
1310 /// \param ToType The type we're converting to.
1311 ///
1312 /// \param ConvertedType The type that will be produced after applying
1313 /// this conversion.
1314 bool SemaObjC::isObjCWritebackConversion(QualType FromType, QualType ToType,
1315  QualType &ConvertedType) {
1316  ASTContext &Context = getASTContext();
1317  if (!getLangOpts().ObjCAutoRefCount ||
1318  Context.hasSameUnqualifiedType(FromType, ToType))
1319  return false;
1320 
1321  // Parameter must be a pointer to __autoreleasing (with no other qualifiers).
1322  QualType ToPointee;
1323  if (const PointerType *ToPointer = ToType->getAs<PointerType>())
1324  ToPointee = ToPointer->getPointeeType();
1325  else
1326  return false;
1327 
1328  Qualifiers ToQuals = ToPointee.getQualifiers();
1329  if (!ToPointee->isObjCLifetimeType() ||
1330  ToQuals.getObjCLifetime() != Qualifiers::OCL_Autoreleasing ||
1331  !ToQuals.withoutObjCLifetime().empty())
1332  return false;
1333 
1334  // Argument must be a pointer to __strong to __weak.
1335  QualType FromPointee;
1336  if (const PointerType *FromPointer = FromType->getAs<PointerType>())
1337  FromPointee = FromPointer->getPointeeType();
1338  else
1339  return false;
1340 
1341  Qualifiers FromQuals = FromPointee.getQualifiers();
1342  if (!FromPointee->isObjCLifetimeType() ||
1343  (FromQuals.getObjCLifetime() != Qualifiers::OCL_Strong &&
1344  FromQuals.getObjCLifetime() != Qualifiers::OCL_Weak))
1345  return false;
1346 
1347  // Make sure that we have compatible qualifiers.
1348  FromQuals.setObjCLifetime(Qualifiers::OCL_Autoreleasing);
1349  if (!ToQuals.compatiblyIncludes(FromQuals))
1350  return false;
1351 
1352  // Remove qualifiers from the pointee type we're converting from; they
1353  // aren't used in the compatibility check belong, and we'll be adding back
1354  // qualifiers (with __autoreleasing) if the compatibility check succeeds.
1355  FromPointee = FromPointee.getUnqualifiedType();
1356 
1357  // The unqualified form of the pointee types must be compatible.
1358  ToPointee = ToPointee.getUnqualifiedType();
1359  bool IncompatibleObjC;
1360  if (Context.typesAreCompatible(FromPointee, ToPointee))
1361  FromPointee = ToPointee;
1362  else if (!SemaRef.isObjCPointerConversion(FromPointee, ToPointee, FromPointee,
1363  IncompatibleObjC))
1364  return false;
1365 
1366  /// Construct the type we're converting to, which is a pointer to
1367  /// __autoreleasing pointee.
1368  FromPointee = Context.getQualifiedType(FromPointee, FromQuals);
1369  ConvertedType = Context.getPointerType(FromPointee);
1370  return true;
1371 }

1373 /// CheckSubscriptingKind - This routine decide what type
1374 /// of indexing represented by "FromE" is being done.
1375 SemaObjC::ObjCSubscriptKind SemaObjC::CheckSubscriptingKind(Expr *FromE) {
1376  // If the expression already has integral or enumeration type, we're golden.
1377  QualType T = FromE->getType();
1378  if (T->isIntegralOrEnumerationType())
1379  return SemaObjC::OS_Array;
1380 
1381  // If we don't have a class type in C++, there's no way we can get an
1382  // expression of integral or enumeration type.
1383  const RecordType *RecordTy = T->getAs<RecordType>();
1384  if (!RecordTy && (T->isObjCObjectPointerType() || T->isVoidPointerType()))
1385  // All other scalar cases are assumed to be dictionary indexing which
1386  // caller handles, with diagnostics if needed.
1387  return SemaObjC::OS_Dictionary;
1388  if (!getLangOpts().CPlusPlus || !RecordTy || RecordTy->isIncompleteType()) {
1389  // No indexing can be done. Issue diagnostics and quit.
1390  const Expr *IndexExpr = FromE->IgnoreParenImpCasts();
1391  if (isa<StringLiteral>(IndexExpr))
1392  Diag(FromE->getExprLoc(), diag::err_objc_subscript_pointer)
1393  << T << FixItHint::CreateInsertion(FromE->getExprLoc(), "@");
1394  else
1395  Diag(FromE->getExprLoc(), diag::err_objc_subscript_type_conversion) << T;
1396  return SemaObjC::OS_Error;
1397  }
1398 
1399  // We must have a complete class type.
1400  if (SemaRef.RequireCompleteType(FromE->getExprLoc(), T,
1401  diag::err_objc_index_incomplete_class_type,
1402  FromE))
1403  return SemaObjC::OS_Error;
1404 
1405  // Look for a conversion to an integral, enumeration type, or
1406  // objective-C pointer type.
1407  int NoIntegrals = 0, NoObjCIdPointers = 0;
1408  SmallVector<CXXConversionDecl *, 4> ConversionDecls;
1409 
1410  for (NamedDecl *D : cast<CXXRecordDecl>(RecordTy->getDecl())
1411  ->getVisibleConversionFunctions()) {
1412  if (CXXConversionDecl *Conversion =
1413  dyn_cast<CXXConversionDecl>(D->getUnderlyingDecl())) {
1414  QualType CT = Conversion->getConversionType().getNonReferenceType();
1415  if (CT->isIntegralOrEnumerationType()) {
1416  ++NoIntegrals;
1417  ConversionDecls.push_back(Conversion);
1418  } else if (CT->isObjCIdType() || CT->isBlockPointerType()) {
1419  ++NoObjCIdPointers;
1420  ConversionDecls.push_back(Conversion);
1421  }
1422  }
1423  }
1424  if (NoIntegrals == 1 && NoObjCIdPointers == 0)
1425  return SemaObjC::OS_Array;
1426  if (NoIntegrals == 0 && NoObjCIdPointers == 1)
1427  return SemaObjC::OS_Dictionary;
1428  if (NoIntegrals == 0 && NoObjCIdPointers == 0) {
1429  // No conversion function was found. Issue diagnostic and return.
1430  Diag(FromE->getExprLoc(), diag::err_objc_subscript_type_conversion)
1431  << FromE->getType();
1432  return SemaObjC::OS_Error;
1433  }
1434  Diag(FromE->getExprLoc(), diag::err_objc_multiple_subscript_type_conversion)
1435  << FromE->getType();
1436  for (unsigned int i = 0; i < ConversionDecls.size(); i++)
1437  Diag(ConversionDecls[i]->getLocation(),
1438  diag::note_conv_function_declared_at);
1439 
1440  return SemaObjC::OS_Error;
1441 }
