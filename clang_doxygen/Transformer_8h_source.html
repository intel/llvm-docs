90 /// Handles the matcher and callback registration for a single `RewriteRule`, as
91 /// defined by the arguments of the constructor.
92 class Transformer : public ast_matchers::MatchFinder::MatchCallback {
93 public:
94  /// Provides the set of changes to the consumer. The callback is free to move
95  /// or destructively consume the changes as needed.
96  ///
97  /// We use \c MutableArrayRef as an abstraction to provide decoupling, and we
98  /// expect the majority of consumers to copy or move the individual values
99  /// into a separate data structure.
100  using ChangeSetConsumer = std::function<void(
101  Expected<llvm::MutableArrayRef<AtomicChange>> Changes)>;
102 
103  /// \param Consumer receives all rewrites for a single match, or an error.
104  /// Will not necessarily be called for each match; for example, if the rule
105  /// generates no edits but does not fail. Note that clients are responsible
106  /// for handling the case that independent \c AtomicChanges conflict with each
107  /// other.
108  explicit Transformer(transformer::RewriteRuleWith<void> Rule,
109  ChangeSetConsumer Consumer)
110  : Impl(std::make_unique<detail::NoMetadataImpl>(std::move(Rule),
111  std::move(Consumer))) {}
112 
113  /// \param Consumer receives all rewrites and the associated metadata for a
114  /// single match, or an error. Will always be called for each match, even if
115  /// the rule generates no edits. Note that clients are responsible for
116  /// handling the case that independent \c AtomicChanges conflict with each
117  /// other.
118  template <typename MetadataT>
119  explicit Transformer(
120  transformer::RewriteRuleWith<MetadataT> Rule,
121  std::function<void(llvm::Expected<TransformerResult<
122  typename detail::type_identity<MetadataT>::type>>)>
123  Consumer);
124 
125  /// N.B. Passes `this` pointer to `MatchFinder`. So, this object should not
126  /// be moved after this call.
127  void registerMatchers(ast_matchers::MatchFinder *MatchFinder);
128 
129  /// Not called directly by users -- called by the framework, via base class
130  /// pointer.
131  void run(const ast_matchers::MatchFinder::MatchResult &Result) override;
132 
133 private:
134  std::unique_ptr<detail::TransformerImpl> Impl;
135 };

138 /// Implementation when metadata is generated as a part of the rewrite. This
139 /// happens when we have a \c RewriteRuleWith<T>.
140 template <typename T> class WithMetadataImpl final : public TransformerImpl {
141  transformer::RewriteRuleWith<T> Rule;
142  std::function<void(llvm::Expected<TransformerResult<T>>)> Consumer;
143 
144 public:
145  explicit WithMetadataImpl(
146  transformer::RewriteRuleWith<T> R,
147  std::function<void(llvm::Expected<TransformerResult<T>>)> Consumer)
148  : Rule(std::move(R)), Consumer(std::move(Consumer)) {
149  assert(llvm::all_of(Rule.Cases,
150  [](const transformer::RewriteRuleBase::Case &Case)
151  -> bool { return !!Case.Edits; }) &&
152  "edit generator must be provided for each rule");
153  assert(llvm::all_of(Rule.Metadata,
154  [](const typename transformer::Generator<T> &Metadata)
155  -> bool { return !!Metadata; }) &&
156  "metadata generator must be provided for each rule");
157  }
158 
159 private:
160  void onMatchImpl(const ast_matchers::MatchFinder::MatchResult &Result) final {
161  size_t I = transformer::detail::findSelectedCase(Result, Rule);
162  auto Transformations = Rule.Cases[I].Edits(Result);
163  if (!Transformations) {
164  Consumer(Transformations.takeError());
165  return;
166  }
167 
168  llvm::SmallVector<AtomicChange, 1> Changes;
169  if (!Transformations->empty()) {
170  auto C = convertToAtomicChanges(*Transformations, Result);
171  if (C) {
172  Changes = std::move(*C);
173  } else {
174  Consumer(C.takeError());
175  return;
176  }
177  }
178 
179  auto Metadata = Rule.Metadata[I]->eval(Result);
180  if (!Metadata) {
181  Consumer(Metadata.takeError());
182  return;
183  }
184 
185  Consumer(TransformerResult<T>{llvm::MutableArrayRef<AtomicChange>(Changes),
186  std::move(*Metadata)});
187  }
188 
189  std::vector<ast_matchers::internal::DynTypedMatcher>
190  buildMatchers() const final {
191  return transformer::detail::buildMatchers(Rule);
192  }
193 };
194 } // namespace detail
