56 static bool getFullyQualifiedTemplateName(const ASTContext &Ctx,
57  TemplateName &TName,
58  bool WithGlobalNsPrefix) {
59  bool Changed = false;
60  NestedNameSpecifier *NNS = nullptr;
61 
62  TemplateDecl *ArgTDecl = TName.getAsTemplateDecl();
63  // ArgTDecl won't be NULL because we asserted that this isn't a
64  // dependent context very early in the call chain.
65  assert(ArgTDecl != nullptr);
66  QualifiedTemplateName *QTName = TName.getAsQualifiedTemplateName();
67 
68  if (QTName && !QTName->hasTemplateKeyword()) {
69  NNS = QTName->getQualifier();
70  NestedNameSpecifier *QNNS = getFullyQualifiedNestedNameSpecifier(
71  Ctx, NNS, WithGlobalNsPrefix);
72  if (QNNS != NNS) {
73  Changed = true;
74  NNS = QNNS;
75  } else {
76  NNS = nullptr;
77  }
78  } else {
79  NNS = createNestedNameSpecifierForScopeOf(
80  Ctx, ArgTDecl, true, WithGlobalNsPrefix);
81  }
82  if (NNS) {
83  TemplateName UnderlyingTN(ArgTDecl);
84  if (UsingShadowDecl *USD = TName.getAsUsingShadowDecl())
85  UnderlyingTN = TemplateName(USD);
86  TName =
87  Ctx.getQualifiedTemplateName(NNS,
88  /*TemplateKeyword=*/false, UnderlyingTN);
89  Changed = true;
90  }
91  return Changed;
92 }

94 static bool getFullyQualifiedTemplateArgument(const ASTContext &Ctx,
95  TemplateArgument &Arg,
96  bool WithGlobalNsPrefix) {
97  bool Changed = false;
98 
99  // Note: we do not handle TemplateArgument::Expression, to replace it
100  // we need the information for the template instance decl.
101 
102  if (Arg.getKind() == TemplateArgument::Template) {
103  TemplateName TName = Arg.getAsTemplate();
104  Changed = getFullyQualifiedTemplateName(Ctx, TName, WithGlobalNsPrefix);
105  if (Changed) {
106  Arg = TemplateArgument(TName);
107  }
108  } else if (Arg.getKind() == TemplateArgument::Type) {
109  QualType SubTy = Arg.getAsType();
110  // Check if the type needs more desugaring and recurse.
111  QualType QTFQ = getFullyQualifiedType(SubTy, Ctx, WithGlobalNsPrefix);
112  if (QTFQ != SubTy) {
113  Arg = TemplateArgument(QTFQ);
114  Changed = true;
115  }
116  } else if (Arg.getKind() == TemplateArgument::Pack) {
117  for (const auto &It : Arg.pack_elements()) {
118  Changed |= getFullyQualifiedTemplateArgument(
119  Ctx, const_cast<TemplateArgument &>(It), WithGlobalNsPrefix);
120  }
121  }
122  return Changed;
123 }

125 static const Type *getFullyQualifiedTemplateType(const ASTContext &Ctx,
126  const Type *TypePtr,
127  bool WithGlobalNsPrefix) {
128  // DependentTemplateTypes exist within template declarations and
129  // definitions. Therefore we shouldn't encounter them at the end of
130  // a translation unit. If we do, the caller has made an error.
131  assert(!isa<DependentTemplateSpecializationType>(TypePtr));
132  // In case of template specializations, iterate over the arguments
133  // and fully qualify them as well.
134  if (const auto *TST = dyn_cast<const TemplateSpecializationType>(TypePtr)) {
135  bool MightHaveChanged = false;
136  SmallVector<TemplateArgument, 4> FQArgs;
137  // Cheap to copy and potentially modified by
138  // getFullyQualifedTemplateArgument.
139  for (TemplateArgument Arg : TST->template_arguments()) {
140  MightHaveChanged |= getFullyQualifiedTemplateArgument(
141  Ctx, Arg, WithGlobalNsPrefix);
142  FQArgs.push_back(Arg);
143  }
144 
145  // If a fully qualified arg is different from the unqualified arg,
146  // allocate new type in the AST.
147  if (MightHaveChanged) {
148  QualType QT = Ctx.getTemplateSpecializationType(
149  TST->getTemplateName(), FQArgs,
150  TST->getCanonicalTypeInternal());
151  // getTemplateSpecializationType returns a fully qualified
152  // version of the specialization itself, so no need to qualify
153  // it.
154  return QT.getTypePtr();
155  }
156  } else if (const auto *TSTRecord = dyn_cast<const RecordType>(TypePtr)) {
157  // We are asked to fully qualify and we have a Record Type,
158  // which can point to a template instantiation with no sugar in any of
159  // its template argument, however we still need to fully qualify them.
160 
161  if (const auto *TSTDecl =
162  dyn_cast<ClassTemplateSpecializationDecl>(TSTRecord->getDecl())) {
163  const TemplateArgumentList &TemplateArgs = TSTDecl->getTemplateArgs();
164 
165  bool MightHaveChanged = false;
166  SmallVector<TemplateArgument, 4> FQArgs;
167  for (unsigned int I = 0, E = TemplateArgs.size(); I != E; ++I) {
168  // cheap to copy and potentially modified by
169  // getFullyQualifedTemplateArgument
170  TemplateArgument Arg(TemplateArgs[I]);
171  MightHaveChanged |= getFullyQualifiedTemplateArgument(
172  Ctx, Arg, WithGlobalNsPrefix);
173  FQArgs.push_back(Arg);
174  }
175 
176  // If a fully qualified arg is different from the unqualified arg,
177  // allocate new type in the AST.
178  if (MightHaveChanged) {
179  TemplateName TN(TSTDecl->getSpecializedTemplate());
180  QualType QT = Ctx.getTemplateSpecializationType(
181  TN, FQArgs,
182  TSTRecord->getCanonicalTypeInternal());
183  // getTemplateSpecializationType returns a fully qualified
184  // version of the specialization itself, so no need to qualify
185  // it.
186  return QT.getTypePtr();
187  }
188  }
189  }
190  return TypePtr;
191 }

193 static NestedNameSpecifier *createOuterNNS(const ASTContext &Ctx, const Decl *D,
194  bool FullyQualify,
195  bool WithGlobalNsPrefix) {
196  const DeclContext *DC = D->getDeclContext();
197  if (const auto *NS = dyn_cast<NamespaceDecl>(DC)) {
198  while (NS && NS->isInline()) {
199  // Ignore inline namespace;
200  NS = dyn_cast<NamespaceDecl>(NS->getDeclContext());
201  }
202  if (NS && NS->getDeclName()) {
203  return createNestedNameSpecifier(Ctx, NS, WithGlobalNsPrefix);
204  }
205  return nullptr; // no starting '::', no anonymous
206  } else if (const auto *TD = dyn_cast<TagDecl>(DC)) {
207  return createNestedNameSpecifier(Ctx, TD, FullyQualify, WithGlobalNsPrefix);
208  } else if (const auto *TDD = dyn_cast<TypedefNameDecl>(DC)) {
209  return createNestedNameSpecifier(
210  Ctx, TDD, FullyQualify, WithGlobalNsPrefix);
211  } else if (WithGlobalNsPrefix && DC->isTranslationUnit()) {
212  return NestedNameSpecifier::GlobalSpecifier(Ctx);
213  }
214  return nullptr; // no starting '::' if |WithGlobalNsPrefix| is false
215 }

217 /// Return a fully qualified version of this name specifier.
218 static NestedNameSpecifier *getFullyQualifiedNestedNameSpecifier(
219  const ASTContext &Ctx, NestedNameSpecifier *Scope,
220  bool WithGlobalNsPrefix) {
221  switch (Scope->getKind()) {
222  case NestedNameSpecifier::Global:
223  // Already fully qualified
224  return Scope;
225  case NestedNameSpecifier::Namespace:
226  return TypeName::createNestedNameSpecifier(
227  Ctx, Scope->getAsNamespace(), WithGlobalNsPrefix);
228  case NestedNameSpecifier::NamespaceAlias:
229  // Namespace aliases are only valid for the duration of the
230  // scope where they were introduced, and therefore are often
231  // invalid at the end of the TU. So use the namespace name more
232  // likely to be valid at the end of the TU.
233  return TypeName::createNestedNameSpecifier(
234  Ctx,
235  Scope->getAsNamespaceAlias()->getNamespace()->getCanonicalDecl(),
236  WithGlobalNsPrefix);
237  case NestedNameSpecifier::Identifier:
238  // A function or some other construct that makes it un-namable
239  // at the end of the TU. Skip the current component of the name,
240  // but use the name of it's prefix.
241  return getFullyQualifiedNestedNameSpecifier(
242  Ctx, Scope->getPrefix(), WithGlobalNsPrefix);
243  case NestedNameSpecifier::Super:
244  case NestedNameSpecifier::TypeSpec:
245  case NestedNameSpecifier::TypeSpecWithTemplate: {
246  const Type *Type = Scope->getAsType();
247  // Find decl context.
248  const TagDecl *TD = nullptr;
249  if (const TagType *TagDeclType = Type->getAs<TagType>()) {
250  TD = TagDeclType->getDecl();
251  } else {
252  TD = Type->getAsCXXRecordDecl();
253  }
254  if (TD) {
255  return TypeName::createNestedNameSpecifier(Ctx, TD,
256  true /*FullyQualified*/,
257  WithGlobalNsPrefix);
258  } else if (const auto *TDD = dyn_cast<TypedefType>(Type)) {
259  return TypeName::createNestedNameSpecifier(Ctx, TDD->getDecl(),
260  true /*FullyQualified*/,
261  WithGlobalNsPrefix);
262  }
263  return Scope;
264  }
265  }
266  llvm_unreachable("bad NNS kind");
267 }

269 /// Create a nested name specifier for the declaring context of
270 /// the type.
271 static NestedNameSpecifier *createNestedNameSpecifierForScopeOf(
272  const ASTContext &Ctx, const Decl *Decl,
273  bool FullyQualified, bool WithGlobalNsPrefix) {
274  assert(Decl);
275 
276  const DeclContext *DC = Decl->getDeclContext()->getRedeclContext();
277  const auto *Outer = dyn_cast_or_null<NamedDecl>(DC);
278  const auto *OuterNS = dyn_cast_or_null<NamespaceDecl>(DC);
279  if (Outer && !(OuterNS && OuterNS->isAnonymousNamespace())) {
280  if (const auto *CxxDecl = dyn_cast<CXXRecordDecl>(DC)) {
281  if (ClassTemplateDecl *ClassTempl =
282  CxxDecl->getDescribedClassTemplate()) {
283  // We are in the case of a type(def) that was declared in a
284  // class template but is *not* type dependent. In clang, it
285  // gets attached to the class template declaration rather than
286  // any specific class template instantiation. This result in
287  // 'odd' fully qualified typename:
288  //
289  // vector<_Tp,_Alloc>::size_type
290  //
291  // Make the situation is 'useable' but looking a bit odd by
292  // picking a random instance as the declaring context.
293  if (ClassTempl->spec_begin() != ClassTempl->spec_end()) {
294  Decl = *(ClassTempl->spec_begin());
295  Outer = dyn_cast<NamedDecl>(Decl);
296  OuterNS = dyn_cast<NamespaceDecl>(Decl);
297  }
298  }
299  }
300 
301  if (OuterNS) {
302  return createNestedNameSpecifier(Ctx, OuterNS, WithGlobalNsPrefix);
303  } else if (const auto *TD = dyn_cast<TagDecl>(Outer)) {
304  return createNestedNameSpecifier(
305  Ctx, TD, FullyQualified, WithGlobalNsPrefix);
306  } else if (isa<TranslationUnitDecl>(Outer)) {
307  // Context is the TU. Nothing needs to be done.
308  return nullptr;
309  } else {
310  // Decl's context was neither the TU, a namespace, nor a
311  // TagDecl, which means it is a type local to a scope, and not
312  // accessible at the end of the TU.
313  return nullptr;
314  }
315  } else if (WithGlobalNsPrefix && DC->isTranslationUnit()) {
316  return NestedNameSpecifier::GlobalSpecifier(Ctx);
317  }
318  return nullptr;
319 }

321 /// Create a nested name specifier for the declaring context of
322 /// the type.
323 static NestedNameSpecifier *createNestedNameSpecifierForScopeOf(
324  const ASTContext &Ctx, const Type *TypePtr,
325  bool FullyQualified, bool WithGlobalNsPrefix) {
326  if (!TypePtr) return nullptr;
327 
328  Decl *Decl = nullptr;
329  // There are probably other cases ...
330  if (const auto *TDT = dyn_cast<TypedefType>(TypePtr)) {
331  Decl = TDT->getDecl();
332  } else if (const auto *TagDeclType = dyn_cast<TagType>(TypePtr)) {
333  Decl = TagDeclType->getDecl();
334  } else if (const auto *TST = dyn_cast<TemplateSpecializationType>(TypePtr)) {
335  Decl = TST->getTemplateName().getAsTemplateDecl();
336  } else {
337  Decl = TypePtr->getAsCXXRecordDecl();
338  }
339 
340  if (!Decl) return nullptr;
341 
342  return createNestedNameSpecifierForScopeOf(
343  Ctx, Decl, FullyQualified, WithGlobalNsPrefix);
344 }

346 NestedNameSpecifier *createNestedNameSpecifier(const ASTContext &Ctx,
347  const NamespaceDecl *Namespace,
348  bool WithGlobalNsPrefix) {
349  while (Namespace && Namespace->isInline()) {
350  // Ignore inline namespace;
351  Namespace = dyn_cast<NamespaceDecl>(Namespace->getDeclContext());
352  }
353  if (!Namespace) return nullptr;
354 
355  bool FullyQualified = true; // doesn't matter, DeclContexts are namespaces
356  return NestedNameSpecifier::Create(
357  Ctx,
358  createOuterNNS(Ctx, Namespace, FullyQualified, WithGlobalNsPrefix),
359  Namespace);

362 NestedNameSpecifier *createNestedNameSpecifier(const ASTContext &Ctx,
363  const TypeDecl *TD,
364  bool FullyQualify,
365  bool WithGlobalNsPrefix) {
366  const Type *TypePtr = TD->getTypeForDecl();
367  if (isa<const TemplateSpecializationType>(TypePtr) ||
368  isa<const RecordType>(TypePtr)) {
369  // We are asked to fully qualify and we have a Record Type (which
370  // may point to a template specialization) or Template
371  // Specialization Type. We need to fully qualify their arguments.
372 
373  TypePtr = getFullyQualifiedTemplateType(Ctx, TypePtr, WithGlobalNsPrefix);
374  }
375 
376  return NestedNameSpecifier::Create(
377  Ctx, createOuterNNS(Ctx, TD, FullyQualify, WithGlobalNsPrefix),
378  false /*No TemplateKeyword*/, TypePtr);
379 }

381 /// Return the fully qualified type, including fully-qualified
382 /// versions of any template parameters.
383 QualType getFullyQualifiedType(QualType QT, const ASTContext &Ctx,
384  bool WithGlobalNsPrefix) {
385  // In case of myType* we need to strip the pointer first, fully
386  // qualify and attach the pointer once again.
387  if (isa<PointerType>(QT.getTypePtr())) {
388  // Get the qualifiers.
389  Qualifiers Quals = QT.getQualifiers();
390  QT = getFullyQualifiedType(QT->getPointeeType(), Ctx, WithGlobalNsPrefix);
391  QT = Ctx.getPointerType(QT);
392  // Add back the qualifiers.
393  QT = Ctx.getQualifiedType(QT, Quals);
394  return QT;
395  }
396 
397  if (auto *MPT = dyn_cast<MemberPointerType>(QT.getTypePtr())) {
398  // Get the qualifiers.
399  Qualifiers Quals = QT.getQualifiers();
400  // Fully qualify the pointee and class types.
401  QT = getFullyQualifiedType(QT->getPointeeType(), Ctx, WithGlobalNsPrefix);
402  QualType Class = getFullyQualifiedType(QualType(MPT->getClass(), 0), Ctx,
403  WithGlobalNsPrefix);
404  QT = Ctx.getMemberPointerType(QT, Class.getTypePtr());
405  // Add back the qualifiers.
406  QT = Ctx.getQualifiedType(QT, Quals);
407  return QT;
408  }
409 
410  // In case of myType& we need to strip the reference first, fully
411  // qualify and attach the reference once again.
412  if (isa<ReferenceType>(QT.getTypePtr())) {
413  // Get the qualifiers.
414  bool IsLValueRefTy = isa<LValueReferenceType>(QT.getTypePtr());
415  Qualifiers Quals = QT.getQualifiers();
416  QT = getFullyQualifiedType(QT->getPointeeType(), Ctx, WithGlobalNsPrefix);
417  // Add the r- or l-value reference type back to the fully
418  // qualified one.
419  if (IsLValueRefTy)
420  QT = Ctx.getLValueReferenceType(QT);
421  else
422  QT = Ctx.getRValueReferenceType(QT);
423  // Add back the qualifiers.
424  QT = Ctx.getQualifiedType(QT, Quals);
425  return QT;
426  }
427 
428  // Remove the part of the type related to the type being a template
429  // parameter (we won't report it as part of the 'type name' and it
430  // is actually make the code below to be more complex (to handle
431  // those)
432  while (isa<SubstTemplateTypeParmType>(QT.getTypePtr())) {
433  // Get the qualifiers.
434  Qualifiers Quals = QT.getQualifiers();
435 
436  QT = cast<SubstTemplateTypeParmType>(QT.getTypePtr())->desugar();
437 
438  // Add back the qualifiers.
439  QT = Ctx.getQualifiedType(QT, Quals);
440  }
441 
442  NestedNameSpecifier *Prefix = nullptr;
443  // Local qualifiers are attached to the QualType outside of the
444  // elaborated type. Retrieve them before descending into the
445  // elaborated type.
446  Qualifiers PrefixQualifiers = QT.getLocalQualifiers();
447  QT = QualType(QT.getTypePtr(), 0);
448  ElaboratedTypeKeyword Keyword = ElaboratedTypeKeyword::None;
449  if (const auto *ETypeInput = dyn_cast<ElaboratedType>(QT.getTypePtr())) {
450  QT = ETypeInput->getNamedType();
451  assert(!QT.hasLocalQualifiers());
452  Keyword = ETypeInput->getKeyword();
453  }
454 
455  // We don't consider the alias introduced by `using a::X` as a new type.
456  // The qualified name is still a::X.
457  if (const auto *UT = QT->getAs<UsingType>()) {
458  QT = Ctx.getQualifiedType(UT->getUnderlyingType(), PrefixQualifiers);
459  return getFullyQualifiedType(QT, Ctx, WithGlobalNsPrefix);
460  }
461 
462  // Create a nested name specifier if needed.
463  Prefix = createNestedNameSpecifierForScopeOf(Ctx, QT.getTypePtr(),
464  true /*FullyQualified*/,
465  WithGlobalNsPrefix);
466 
467  // In case of template specializations iterate over the arguments and
468  // fully qualify them as well.
469  if (isa<const TemplateSpecializationType>(QT.getTypePtr()) ||
470  isa<const RecordType>(QT.getTypePtr())) {
471  // We are asked to fully qualify and we have a Record Type (which
472  // may point to a template specialization) or Template
473  // Specialization Type. We need to fully qualify their arguments.
474 
475  const Type *TypePtr = getFullyQualifiedTemplateType(
476  Ctx, QT.getTypePtr(), WithGlobalNsPrefix);
477  QT = QualType(TypePtr, 0);
478  }
479  if (Prefix || Keyword != ElaboratedTypeKeyword::None) {
480  QT = Ctx.getElaboratedType(Keyword, Prefix, QT);
481  }
482  QT = Ctx.getQualifiedType(QT, PrefixQualifiers);
483  return QT;
484 }
