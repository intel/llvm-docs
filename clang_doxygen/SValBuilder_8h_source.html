363  /// Create NULL pointer, with proper pointer bit-width for given address
364  /// space.
365  /// \param type pointer type.
366  Loc makeNullWithType(QualType type) {
367  // We cannot use the `isAnyPointerType()`.
368  assert((type->isPointerType() || type->isObjCObjectPointerType() ||
369  type->isBlockPointerType() || type->isNullPtrType() ||
370  type->isReferenceType()) &&
371  "makeNullWithType must use pointer type");
372 
373  // The `sizeof(T&)` is `sizeof(T)`, thus we replace the reference with a
374  // pointer. Here we assume that references are actually implemented by
375  // pointers under-the-hood.
376  type = type->isReferenceType()
377  ? Context.getPointerType(type->getPointeeType())
378  : type;
379  return loc::ConcreteInt(BasicVals.getZeroWithTypeSize(type));
380  }
