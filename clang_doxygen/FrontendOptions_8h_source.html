150 /// The kind of a file that we've been handed as an input.
151 class InputKind {
152 private:
153  Language Lang;
154  unsigned Fmt : 3;
155  unsigned Preprocessed : 1;
156 
157 public:
158  /// The input file format.
159  enum Format {
160  Source,
161  ModuleMap,
162  Precompiled
163  };
164 
165  constexpr InputKind(Language L = Language::Unknown, Format F = Source,
166  bool PP = false)
167  : Lang(L), Fmt(F), Preprocessed(PP) {}
168 
169  Language getLanguage() const { return static_cast<Language>(Lang); }
170  Format getFormat() const { return static_cast<Format>(Fmt); }
171  bool isPreprocessed() const { return Preprocessed; }
172 
173  /// Is the input kind fully-unknown?
174  bool isUnknown() const { return Lang == Language::Unknown && Fmt == Source; }
175 
176  /// Is the language of the input some dialect of Objective-C?
177  bool isObjectiveC() const {
178  return Lang == Language::ObjC || Lang == Language::ObjCXX;
179  }
180 
181  InputKind getPreprocessed() const {
182  return InputKind(getLanguage(), getFormat(), true);
183  }
184 
185  InputKind withFormat(Format F) const {
186  return InputKind(getLanguage(), F, isPreprocessed());
187  }
188 };

190 /// An input file for the front end.
191 class FrontendInputFile {
192  /// The file name, or "-" to read from standard input.
193  std::string File;
194 
195  /// The input, if it comes from a buffer rather than a file. This object
196  /// does not own the buffer, and the caller is responsible for ensuring
197  /// that it outlives any users.
198  llvm::Optional<llvm::MemoryBufferRef> Buffer;
199 
200  /// The kind of input, e.g., C source, AST file, LLVM IR.
201  InputKind Kind;
202 
203  /// Whether we're dealing with a 'system' input (vs. a 'user' input).
204  bool IsSystem = false;
205 
206 public:
207  FrontendInputFile() = default;
208  FrontendInputFile(StringRef File, InputKind Kind, bool IsSystem = false)
209  : File(File.str()), Kind(Kind), IsSystem(IsSystem) {}
210  FrontendInputFile(llvm::MemoryBufferRef Buffer, InputKind Kind,
211  bool IsSystem = false)
212  : Buffer(Buffer), Kind(Kind), IsSystem(IsSystem) {}
213 
214  InputKind getKind() const { return Kind; }
215  bool isSystem() const { return IsSystem; }
216 
217  bool isEmpty() const { return File.empty() && Buffer == None; }
218  bool isFile() const { return !isBuffer(); }
219  bool isBuffer() const { return Buffer != None; }
220  bool isPreprocessed() const { return Kind.isPreprocessed(); }
221 
222  StringRef getFile() const {
223  assert(isFile());
224  return File;
225  }
226 
227  llvm::MemoryBufferRef getBuffer() const {
228  assert(isBuffer());
229  return *Buffer;
230  }
231 };
