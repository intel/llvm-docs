63 const DerivedArgList &
64 Compilation::getArgsForToolChain(const ToolChain *TC, StringRef BoundArch,
65  Action::OffloadKind DeviceOffloadKind) {
66  if (!TC)
67  TC = &DefaultToolChain;
68 
69  DerivedArgList *&Entry = TCArgs[{TC, BoundArch, DeviceOffloadKind}];
70  if (!Entry) {
71  SmallVector<Arg *, 4> AllocatedArgs;
72  DerivedArgList *OffloadArgs = nullptr;
73  // Translate offload toolchain arguments provided via the -Xopenmp-target
74  // or -Xsycl-target-frontend flags.
75  if (DeviceOffloadKind == Action::OFK_OpenMP ||
76  DeviceOffloadKind == Action::OFK_SYCL) {
77  const ToolChain *HostTC = getSingleOffloadToolChain<Action::OFK_Host>();
78  bool SameTripleAsHost = (TC->getTriple() == HostTC->getTriple());
79  OffloadArgs = TC->TranslateOffloadTargetArgs(
80  *TranslatedArgs, SameTripleAsHost, AllocatedArgs, DeviceOffloadKind);
81  }
82 
83  DerivedArgList *NewDAL = nullptr;
84  if (!OffloadArgs) {
85  NewDAL = TC->TranslateXarchArgs(*TranslatedArgs, BoundArch,
86  DeviceOffloadKind, &AllocatedArgs);
87  } else {
88  NewDAL = TC->TranslateXarchArgs(*OffloadArgs, BoundArch, DeviceOffloadKind,
89  &AllocatedArgs);
90  if (!NewDAL)
91  NewDAL = OffloadArgs;
92  else
93  delete OffloadArgs;
94  }
95 
96  if (!NewDAL) {
97  Entry = TC->TranslateArgs(*TranslatedArgs, BoundArch, DeviceOffloadKind);
98  if (!Entry)
99  Entry = TranslatedArgs;
100  } else {
101  Entry = TC->TranslateArgs(*NewDAL, BoundArch, DeviceOffloadKind);
102  if (!Entry)
103  Entry = NewDAL;
104  else
105  delete NewDAL;
106  }
107 
108  // Add allocated arguments to the final DAL.
109  for (auto ArgPtr : AllocatedArgs)
110  Entry->AddSynthesizedArg(ArgPtr);
111  }
112 
113  return *Entry;
114 }

116 bool Compilation::CleanupFile(const char *File, bool IssueErrors) const {
117  // FIXME: Why are we trying to remove files that we have not created? For
118  // example we should only try to remove a temporary assembly file if
119  // "clang -cc1" succeed in writing it. Was this a workaround for when
120  // clang was writing directly to a .s file and sometimes leaving it behind
121  // during a failure?
122 
123  // FIXME: If this is necessary, we can still try to split
124  // llvm::sys::fs::remove into a removeFile and a removeDir and avoid the
125  // duplicated stat from is_regular_file.
126 
127  // Don't try to remove files which we don't have write access to (but may be
128  // able to remove), or non-regular files. Underlying tools may have
129  // intentionally not overwritten them.
130  if (!llvm::sys::fs::can_write(File) || !llvm::sys::fs::is_regular_file(File))
131  return true;
132 
133  if (std::error_code EC = llvm::sys::fs::remove(File)) {
134  // Failure is only failure if the file exists and is "regular". We checked
135  // for it being regular before, and llvm::sys::fs::remove ignores ENOENT,
136  // so we don't need to check again.
137 
138  if (IssueErrors)
139  getDriver().Diag(diag::err_drv_unable_to_remove_file)
140  << EC.message();
141  return false;
142  }
143  return true;
144 }

146 bool Compilation::CleanupFileList(const TempFileList &Files,
147  bool IssueErrors) const {
148  bool Success = true;
149  for (const auto &File: Files) {
150  // Temporary file lists contain files that need to be cleaned. The
151  // file containing the information is also removed
152  if (File.second == types::TY_Tempfilelist ||
153  File.second == types::TY_Tempfiletable ||
154  File.second == types::TY_FPGA_Dependencies_List) {
155  // These are temporary files and need to be removed.
156  bool IsTable = File.second == types::TY_Tempfiletable;
157 
158  if (IsTable) {
159  if (llvm::sys::fs::exists(File.first)) {
160  auto T = llvm::util::SimpleTable::read(File.first);
161  if (!T) {
162  Success = false;
163  continue;
164  }
165  std::vector<std::string> TmpFileNames;
166  T->get()->linearize(TmpFileNames);
167 
168  for (const auto &TmpFileName : TmpFileNames) {
169  if (!TmpFileName.empty())
170  Success &= CleanupFile(TmpFileName.c_str(), IssueErrors);
171  }
172  }
173  } else {
174  std::ifstream ListFile(File.first);
175  std::string TmpFileName;
176  while (std::getline(ListFile, TmpFileName) && !TmpFileName.empty())
177  Success &= CleanupFile(TmpFileName.c_str(), IssueErrors);
178  }
179  }
180  Success &= CleanupFile(File.first, IssueErrors);
181  }
182  return Success;
183 }

199 int Compilation::ExecuteCommand(const Command &C,
200  const Command *&FailingCommand) const {
201  if ((getDriver().CCPrintOptions ||
202  getArgs().hasArg(options::OPT_v)) && !getDriver().CCGenDiagnostics) {
203  raw_ostream *OS = &llvm::errs();
204  std::unique_ptr<llvm::raw_fd_ostream> OwnedStream;
205 
206  // Follow gcc implementation of CC_PRINT_OPTIONS; we could also cache the
207  // output stream.
208  if (getDriver().CCPrintOptions &&
209  !getDriver().CCPrintOptionsFilename.empty()) {
210  std::error_code EC;
211  OwnedStream.reset(new llvm::raw_fd_ostream(
212  getDriver().CCPrintOptionsFilename, EC,
213  llvm::sys::fs::OF_Append | llvm::sys::fs::OF_TextWithCRLF));
214  if (EC) {
215  getDriver().Diag(diag::err_drv_cc_print_options_failure)
216  << EC.message();
217  FailingCommand = &C;
218  return 1;
219  }
220  OS = OwnedStream.get();
221  }
222 
223  if (getDriver().CCPrintOptions)
224  *OS << "[Logging clang options]\n";
225 
226  C.Print(*OS, "\n", /*Quote=*/getDriver().CCPrintOptions);
227  }
228 
229  std::string Error;
230  bool ExecutionFailed;
231  int Res = C.Execute(Redirects, &Error, &ExecutionFailed);
232  if (PostCallback)
233  PostCallback(C, Res);
234  if (!Error.empty()) {
235  assert(Res && "Error string set with 0 result code!");
236  getDriver().Diag(diag::err_drv_command_failure) << Error;
237  }
238 
239  // When performing preprocessing, we need to be able to produce the
240  // preprocessed output even if the compilation is not valid. If
241  // the device compilation fails for SYCL allow the failure to pass
242  // through so we can still generate the expected preprocessed files.
243  if (Res && C.getSource().isDeviceOffloading(Action::OFK_SYCL) &&
244  getArgs().hasArg(options::OPT_E))
245  return 0;
246 
247  if (Res)
248  FailingCommand = &C;
249 
250  return ExecutionFailed ? 1 : Res;
251 }

255 static bool ActionFailed(const Action *A,
256  const FailingCommandList &FailingCommands) {
257  if (FailingCommands.empty())
258  return false;
259 
260  for (const auto &CI : FailingCommands)
261  if (!CI.second->getWillExitForErrorCode(CI.first))
262  return false;
263 
264  // CUDA/HIP/SYCL can have the same input source code compiled multiple times
265  // so do not compile again if there are already failures. It is OK to abort
266  // the CUDA pipeline on errors.
267  if (A->isOffloading(Action::OFK_Cuda) || A->isOffloading(Action::OFK_HIP) ||
268  A->isOffloading(Action::OFK_SYCL))
269  return true;
270 
271  for (const auto &CI : FailingCommands)
272  if (A == &(CI.second->getSource()))
273  return true;
274 
275  for (const auto *AI : A->inputs())
276  if (ActionFailed(AI, FailingCommands))
277  return true;
278 
279  return false;
280 }

287 void Compilation::ExecuteJobs(const JobList &Jobs,
288  FailingCommandList &FailingCommands) const {
289  // According to UNIX standard, driver need to continue compiling all the
290  // inputs on the command line even one of them failed.
291  // In all but CLMode, execute all the jobs unless the necessary inputs for the
292  // job is missing due to previous failures.
293  for (const auto &Job : Jobs) {
294  if (!InputsOk(Job, FailingCommands))
295  continue;
296  const Command *FailingCommand = nullptr;
297  if (int Res = ExecuteCommand(Job, FailingCommand)) {
298  FailingCommands.push_back(std::make_pair(Res, FailingCommand));
299  // Bail as soon as one command fails in cl driver mode.
300  // Do not bail when the tool is setup to allow for continuation upon
301  // failure.
302  if (TheDriver.IsCLMode() && FailingCommand->getWillExitForErrorCode(Res))
303  return;
304  }
305  }
306 }

308 void Compilation::initCompilationForDiagnostics() {
309  ForDiagnostics = true;
310 
311  // Free actions and jobs.
312  Actions.clear();
313  AllActions.clear();
314  Jobs.clear();
315 
316  // Remove temporary files.
317  if (!TheDriver.isSaveTempsEnabled() && !ForceKeepTempFiles)
318  CleanupFileList(TempFiles);
319 
320  // Clear temporary/results file lists.
321  TempFiles.clear();
322  ResultFiles.clear();
323  FailureResultFiles.clear();
324 
325  // Remove any user specified output. Claim any unclaimed arguments, so as
326  // to avoid emitting warnings about unused args.
327  OptSpecifier OutputOpts[] = {
328  options::OPT_o, options::OPT_MD, options::OPT_MMD, options::OPT_M,
329  options::OPT_MM, options::OPT_MF, options::OPT_MG, options::OPT_MJ,
330  options::OPT_MQ, options::OPT_MT, options::OPT_MV};
331  for (unsigned i = 0, e = llvm::array_lengthof(OutputOpts); i != e; ++i) {
332  if (TranslatedArgs->hasArg(OutputOpts[i]))
333  TranslatedArgs->eraseArg(OutputOpts[i]);
334  }
335  TranslatedArgs->ClaimAllArgs();
336 
337  // Force re-creation of the toolchain Args, otherwise our modifications just
338  // above will have no effect.
339  for (auto Arg : TCArgs)
340  if (Arg.second != TranslatedArgs)
341  delete Arg.second;
342  TCArgs.clear();
343 
344  // Redirect stdout/stderr to /dev/null.
345  Redirects = {None, {""}, {""}};
346 
347  // Temporary files added by diagnostics should be kept.
348  ForceKeepTempFiles = true;
349 }
