63 static void HandleXarchArgs(DerivedArgList *OffloadArgList, const Driver &D,
64  bool IsDevice) {
65  if (!OffloadArgList)
66  return;
67 
68  if (IsDevice && !OffloadArgList->hasArg(options::OPT_Xarch_device))
69  return;
70 
71  if (!IsDevice && !OffloadArgList->hasArg(options::OPT_Xarch_host))
72  return;
73 
74  bool NeedHandle = false;
75  std::vector<std::string> XarchValues;
76  XarchValues = IsDevice
77  ? OffloadArgList->getAllArgValues(options::OPT_Xarch_device)
78  : OffloadArgList->getAllArgValues(options::OPT_Xarch_host);
79  SmallVector<StringRef, 8> XarchValueRefs;
80  for (auto XarchV : XarchValues) {
81  if (XarchV.find(' ') != std::string::npos) {
82  NeedHandle = true;
83  StringRef XarchVRef(XarchV);
84  SmallVector<StringRef, 8> XarchVecs;
85  XarchVRef.trim().split(XarchVecs, ' ', -1, false);
86  size_t Index;
87  const size_t XSize = XarchVecs.size();
88  for (Index = 0; Index < XSize; ++Index) {
89  if (XarchVecs[Index].compare("-mllvm") == 0) {
90  if (Index < (XSize - 1)) {
91  XarchValueRefs.push_back(OffloadArgList->MakeArgStringRef(
92  (StringRef("-mllvm=") + XarchVecs[Index + 1]).str()));
93  Index++;
94  continue;
95  } else
96  D.Diag(clang::diag::err_drv_missing_argument) << "-mllvm" << 1;
97  } else
98  XarchValueRefs.push_back(
99  OffloadArgList->MakeArgStringRef(XarchVecs[Index]));
100  }
101  } else
102  XarchValueRefs.push_back(OffloadArgList->MakeArgString(XarchV));
103  }
104 
105  if (NeedHandle) {
106  auto Xarch_OPT =
107  IsDevice ? options::OPT_Xarch_device : options::OPT_Xarch_host;
108  OffloadArgList->eraseArg(Xarch_OPT);
109  for (auto XarchV : XarchValueRefs) {
110  Arg *A = OffloadArgList->MakeSeparateArg(
111  nullptr, D.getOpts().getOption(Xarch_OPT), XarchV);
112  A->claim();
113  OffloadArgList->append(A);
114  }
115  }
116 }

118 const DerivedArgList &
119 Compilation::getArgsForToolChain(const ToolChain *TC, StringRef BoundArch,
120  Action::OffloadKind DeviceOffloadKind) {
121  if (!TC)
122  TC = &DefaultToolChain;
123 
124  DerivedArgList *&Entry = TCArgs[{TC, BoundArch, DeviceOffloadKind}];
125  if (!Entry) {
126  SmallVector<Arg *, 4> AllocatedArgs;
127  DerivedArgList *OffloadArgs = nullptr;
128  // Translate offload toolchain arguments provided via the -Xopenmp-target
129  // or -Xsycl-target-frontend flags.
130  if (DeviceOffloadKind == Action::OFK_OpenMP ||
131  DeviceOffloadKind == Action::OFK_SYCL) {
132  const ToolChain *HostTC = getSingleOffloadToolChain<Action::OFK_Host>();
133  bool SameTripleAsHost = (TC->getTriple() == HostTC->getTriple());
134  OffloadArgs = TC->TranslateOffloadTargetArgs(
135  *TranslatedArgs, SameTripleAsHost, AllocatedArgs, DeviceOffloadKind);
136  }
137 
138  DerivedArgList *NewDAL = nullptr;
139  if (!OffloadArgs) {
140  HandleXarchArgs(TranslatedArgs, getDriver(), false);
141  NewDAL = TC->TranslateXarchArgs(*TranslatedArgs, BoundArch,
142  DeviceOffloadKind, &AllocatedArgs);
143  } else {
144  HandleXarchArgs(OffloadArgs, getDriver(), true);
145  NewDAL = TC->TranslateXarchArgs(*OffloadArgs, BoundArch, DeviceOffloadKind,
146  &AllocatedArgs);
147  if (!NewDAL)
148  NewDAL = OffloadArgs;
149  else
150  delete OffloadArgs;
151  }
152 
153  if (!NewDAL) {
154  Entry = TC->TranslateArgs(*TranslatedArgs, BoundArch, DeviceOffloadKind);
155  if (!Entry)
156  Entry = TranslatedArgs;
157  } else {
158  Entry = TC->TranslateArgs(*NewDAL, BoundArch, DeviceOffloadKind);
159  if (!Entry)
160  Entry = NewDAL;
161  else
162  delete NewDAL;
163  }
164 
165  // Add allocated arguments to the final DAL.
166  for (auto *ArgPtr : AllocatedArgs)
167  Entry->AddSynthesizedArg(ArgPtr);
168  }
169 
170  return *Entry;
171 }

173 bool Compilation::CleanupFile(const char *File, bool IssueErrors) const {
174  // FIXME: Why are we trying to remove files that we have not created? For
175  // example we should only try to remove a temporary assembly file if
176  // "clang -cc1" succeed in writing it. Was this a workaround for when
177  // clang was writing directly to a .s file and sometimes leaving it behind
178  // during a failure?
179 
180  // FIXME: If this is necessary, we can still try to split
181  // llvm::sys::fs::remove into a removeFile and a removeDir and avoid the
182  // duplicated stat from is_regular_file.
183 
184  // Don't try to remove files which we don't have write access to (but may be
185  // able to remove), or non-regular files. Underlying tools may have
186  // intentionally not overwritten them.
187 
188  // Save the device code files if -fsycl-dump-device-code option is enabled.
189  if (TheDriver.isDumpDeviceCodeEnabled()) {
190  Arg *DumpDeviceCodeArg =
191  getArgs().getLastArg(options::OPT_fsycl_dump_device_code_EQ);
192  std::string ExpectedDir =
193  DumpDeviceCodeArg ? DumpDeviceCodeArg->getValue() : "";
194  std::string ActualFile(File);
195 
196  if (ActualFile.find(ExpectedDir) != std::string::npos) {
197  // Save PTX files generated by LLVM NVPTX Back-End,
198  // when the nvptx*-nvidia-cuda is passed to -fsycl-targets.
199  if (DefaultToolChain.getTriple().isNVPTX())
200  return false;
201  if (llvm::sys::path::extension(ActualFile).equals(".spv"))
202  return false;
203  }
204  }
205 
206  if (!llvm::sys::fs::can_write(File) || !llvm::sys::fs::is_regular_file(File))
207  return true;
208 
209  if (std::error_code EC = llvm::sys::fs::remove(File)) {
210  // Failure is only failure if the file exists and is "regular". We checked
211  // for it being regular before, and llvm::sys::fs::remove ignores ENOENT,
212  // so we don't need to check again.
213 
214  if (IssueErrors)
215  getDriver().Diag(diag::err_drv_unable_to_remove_file)
216  << EC.message();
217  return false;
218  }
219  return true;
220 }

222 bool Compilation::CleanupFileList(const TempFileList &Files,
223  bool IssueErrors) const {
224  bool Success = true;
225  for (const auto &File : Files) {
226  // Temporary file lists contain files that need to be cleaned. The
227  // file containing the information is also removed
228  if (File.second == types::TY_Tempfilelist ||
229  File.second == types::TY_Tempfiletable ||
230  File.second == types::TY_FPGA_Dependencies_List) {
231  // These are temporary files and need to be removed.
232  bool IsTable = File.second == types::TY_Tempfiletable;
233 
234  if (IsTable) {
235  if (llvm::sys::fs::exists(File.first)) {
236  auto T = llvm::util::SimpleTable::read(File.first);
237  if (!T) {
238  Success = false;
239  continue;
240  }
241 
242  std::vector<std::string> TmpFileNames;
243  T->get()->linearize(TmpFileNames);
244 
245  for (const auto &TmpFileName : TmpFileNames) {
246  if (!TmpFileName.empty())
247  Success &= CleanupFile(TmpFileName.c_str(), IssueErrors);
248  }
249  }
250  } else {
251  std::ifstream ListFile(File.first);
252  std::string TmpFileName;
253  while (std::getline(ListFile, TmpFileName))
254  Success &= CleanupFile(TmpFileName.c_str(), IssueErrors);
255  }
256  }
257  Success &= CleanupFile(File.first, IssueErrors);
258  }
259  return Success;
260 }

276 int Compilation::ExecuteCommand(const Command &C,
277  const Command *&FailingCommand,
278  bool LogOnly) const {
279  if ((getDriver().CCPrintOptions ||
280  getArgs().hasArg(options::OPT_v)) && !getDriver().CCGenDiagnostics) {
281  raw_ostream *OS = &llvm::errs();
282  std::unique_ptr<llvm::raw_fd_ostream> OwnedStream;
283 
284  // Follow gcc implementation of CC_PRINT_OPTIONS; we could also cache the
285  // output stream.
286  if (getDriver().CCPrintOptions &&
287  !getDriver().CCPrintOptionsFilename.empty()) {
288  std::error_code EC;
289  OwnedStream.reset(new llvm::raw_fd_ostream(
290  getDriver().CCPrintOptionsFilename, EC,
291  llvm::sys::fs::OF_Append | llvm::sys::fs::OF_TextWithCRLF));
292  if (EC) {
293  getDriver().Diag(diag::err_drv_cc_print_options_failure)
294  << EC.message();
295  FailingCommand = &C;
296  return 1;
297  }
298  OS = OwnedStream.get();
299  }
300 
301  if (getDriver().CCPrintOptions)
302  *OS << "[Logging clang options]\n";
303 
304  C.Print(*OS, "\n", /*Quote=*/getDriver().CCPrintOptions);
305  }
306 
307  if (LogOnly)
308  return 0;
309 
310  std::string Error;
311  bool ExecutionFailed;
312  int Res = C.Execute(Redirects, &Error, &ExecutionFailed);
313  if (PostCallback)
314  PostCallback(C, Res);
315  if (!Error.empty()) {
316  assert(Res && "Error string set with 0 result code!");
317  getDriver().Diag(diag::err_drv_command_failure) << Error;
318  }
319 
320  // When performing preprocessing, we need to be able to produce the
321  // preprocessed output even if the compilation is not valid. If
322  // the device compilation fails for SYCL allow the failure to pass
323  // through so we can still generate the expected preprocessed files.
324  if (Res && C.getSource().isDeviceOffloading(Action::OFK_SYCL) &&
325  getArgs().hasArg(options::OPT_E))
326  return 0;
327 
328  if (Res)
329  FailingCommand = &C;
330 
331  return ExecutionFailed ? 1 : Res;
332 }

336 static bool ActionFailed(const Action *A,
337  const FailingCommandList &FailingCommands) {
338  if (FailingCommands.empty())
339  return false;
340 
341  for (const auto &CI : FailingCommands)
342  if (!CI.second->getWillExitForErrorCode(CI.first))
343  return false;
344 
345  // CUDA/HIP/SYCL can have the same input source code compiled multiple times
346  // so do not compile again if there are already failures. It is OK to abort
347  // the CUDA pipeline on errors.
348  if (A->isOffloading(Action::OFK_Cuda) || A->isOffloading(Action::OFK_HIP) ||
349  A->isOffloading(Action::OFK_SYCL))
350  return true;
351 
352  for (const auto &CI : FailingCommands)
353  if (A == &(CI.second->getSource()))
354  return true;
355 
356  for (const auto *AI : A->inputs())
357  if (ActionFailed(AI, FailingCommands))
358  return true;
359 
360  return false;
361 }

368 void Compilation::ExecuteJobs(const JobList &Jobs,
369  FailingCommandList &FailingCommands,
370  bool LogOnly) const {
371  // According to UNIX standard, driver need to continue compiling all the
372  // inputs on the command line even one of them failed.
373  // In all but CLMode, execute all the jobs unless the necessary inputs for the
374  // job is missing due to previous failures.
375  for (const auto &Job : Jobs) {
376  if (!InputsOk(Job, FailingCommands))
377  continue;
378  const Command *FailingCommand = nullptr;
379  if (int Res = ExecuteCommand(Job, FailingCommand, LogOnly)) {
380  FailingCommands.push_back(std::make_pair(Res, FailingCommand));
381  // Bail as soon as one command fails in cl driver mode.
382  // Do not bail when the tool is setup to allow for continuation upon
383  // failure.
384  if (TheDriver.IsCLMode() && FailingCommand->getWillExitForErrorCode(Res))
385  return;
386  }
387  }
388 }

390 void Compilation::initCompilationForDiagnostics() {
391  ForDiagnostics = true;
392 
393  // Free actions and jobs.
394  Actions.clear();
395  AllActions.clear();
396  Jobs.clear();
397 
398  // Remove temporary files.
399  if (!TheDriver.isSaveTempsEnabled() && !ForceKeepTempFiles)
400  CleanupFileList(TempFiles);
401 
402  // Clear temporary/results file lists.
403  TempFiles.clear();
404  ResultFiles.clear();
405  FailureResultFiles.clear();
406 
407  // Remove any user specified output. Claim any unclaimed arguments, so as
408  // to avoid emitting warnings about unused args.
409  OptSpecifier OutputOpts[] = {
410  options::OPT_o, options::OPT_MD, options::OPT_MMD, options::OPT_M,
411  options::OPT_MM, options::OPT_MF, options::OPT_MG, options::OPT_MJ,
412  options::OPT_MQ, options::OPT_MT, options::OPT_MV};
413  for (const auto &Opt : OutputOpts) {
414  if (TranslatedArgs->hasArg(Opt))
415  TranslatedArgs->eraseArg(Opt);
416  }
417  TranslatedArgs->ClaimAllArgs();
418 
419  // Force re-creation of the toolchain Args, otherwise our modifications just
420  // above will have no effect.
421  for (auto Arg : TCArgs)
422  if (Arg.second != TranslatedArgs)
423  delete Arg.second;
424  TCArgs.clear();
425 
426  // Redirect stdout/stderr to /dev/null.
427  Redirects = {std::nullopt, {""}, {""}};
428 
429  // Temporary files added by diagnostics should be kept.
430  ForceKeepTempFiles = true;
431 }
