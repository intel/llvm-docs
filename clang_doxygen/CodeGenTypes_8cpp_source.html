77  else

88 /// ConvertTypeForMem - Convert type T into a llvm::Type. This differs from
89 /// ConvertType in that it is used to convert to the memory representation for
90 /// a type. For example, the scalar representation for _Bool is i1, but the
91 /// memory representation is usually i8 or i32, depending on the target.
92 llvm::Type *CodeGenTypes::ConvertTypeForMem(QualType T, bool ForBitField) {
93  if (T->isConstantMatrixType()) {
94  const Type *Ty = Context.getCanonicalType(T).getTypePtr();
95  const ConstantMatrixType *MT = cast<ConstantMatrixType>(Ty);
96  return llvm::ArrayType::get(ConvertType(MT->getElementType()),
97  MT->getNumRows() * MT->getNumColumns());
98  }
99 
100  llvm::Type *R = ConvertType(T);
101 
102  // Check for the boolean vector case.
103  if (T->isExtVectorBoolType()) {
104  auto *FixedVT = cast<llvm::FixedVectorType>(R);
105  // Pad to at least one byte.
106  uint64_t BytePadded = std::max<uint64_t>(FixedVT->getNumElements(), 8);
107  return llvm::IntegerType::get(FixedVT->getContext(), BytePadded);
108  }
109 
110  // If this is a bool type, or a bit-precise integer type in a bitfield
111  // representation, map this integer to the target-specified size.
112  if ((ForBitField && T->isBitIntType()) ||
113  (!T->isBitIntType() && R->isIntegerTy(1)))
114  return llvm::IntegerType::get(getLLVMContext(),
115  (unsigned)Context.getTypeSize(T));
116 
117  // Else, don't map it.
118  return R;
119 }

149 /// Code to verify a given function type is complete, i.e. the return type
150 /// and all of the parameter types are complete. Also check to see if we are in
151 /// a RS_StructPointer context, and if so whether any struct types have been
152 /// pended. If so, we don't want to ask the ABI lowering code to handle a type
153 /// that cannot be converted to an IR type.
154 bool CodeGenTypes::isFuncTypeConvertible(const FunctionType *FT) {
155  if (!isFuncParamTypeConvertible(FT->getReturnType()))
156  return false;
157 
158  if (const FunctionProtoType *FPT = dyn_cast<FunctionProtoType>(FT))
159  for (unsigned i = 0, e = FPT->getNumParams(); i != e; i++)
160  if (!isFuncParamTypeConvertible(FPT->getParamType(i)))
161  return false;
162 
163  return true;

166 /// UpdateCompletedType - When we find the full definition for a TagDecl,
167 /// replace the 'opaque' type we previously made for it if applicable.
168 void CodeGenTypes::UpdateCompletedType(const TagDecl *TD) {
169  // If this is an enum being completed, then we flush all non-struct types from
170  // the cache. This allows function types and other things that may be derived
171  // from the enum to be recomputed.
172  if (const EnumDecl *ED = dyn_cast<EnumDecl>(TD)) {
173  // Only flush the cache if we've actually already converted this type.
174  if (TypeCache.count(ED->getTypeForDecl())) {
175  // Okay, we formed some types based on this. We speculated that the enum
176  // would be lowered to i32, so we only need to flush the cache if this
177  // didn't happen.
178  if (!ConvertType(ED->getIntegerType())->isIntegerTy(32))
179  TypeCache.clear();
180  }
181  // If this is the SYCL aspect enum it is saved for later processing.
182  if (const auto *Attr = ED->getAttr<SYCLTypeAttr>())
183  if (Attr->getType() == SYCLTypeAttr::SYCLType::aspect)
184  CGM.setAspectsEnumDecl(ED);
185  // If necessary, provide the full definition of a type only used with a
186  // declaration so far.
187  if (CGDebugInfo *DI = CGM.getModuleDebugInfo())
188  DI->completeType(ED);
189  return;
190  }
191 
192  // If we completed a RecordDecl that we previously used and converted to an
193  // anonymous type, then go ahead and complete it now.
194  const RecordDecl *RD = cast<RecordDecl>(TD);
195  if (RD->isDependentType()) return;
196 
197  // Only complete it if we converted it already. If we haven't converted it
198  // yet, we'll just do it lazily.
199  if (RecordDeclTypes.count(Context.getTagDeclType(RD).getTypePtr()))
200  ConvertRecordDeclType(RD);
201 
202  // If necessary, provide the full definition of a type only used with a
203  // declaration so far.
204  if (CGDebugInfo *DI = CGM.getModuleDebugInfo())
205  DI->completeType(RD);
206 }

219 static llvm::Type *getTypeForFormat(llvm::LLVMContext &VMContext,
220  const llvm::fltSemantics &format,
221  bool UseNativeHalf = false) {
222  if (&format == &llvm::APFloat::IEEEhalf()) {
223  if (UseNativeHalf)
224  return llvm::Type::getHalfTy(VMContext);
225  else
226  return llvm::Type::getInt16Ty(VMContext);
227  }
228  if (&format == &llvm::APFloat::BFloat())
229  return llvm::Type::getBFloatTy(VMContext);
230  if (&format == &llvm::APFloat::IEEEsingle())
231  return llvm::Type::getFloatTy(VMContext);
232  if (&format == &llvm::APFloat::IEEEdouble())
233  return llvm::Type::getDoubleTy(VMContext);
234  if (&format == &llvm::APFloat::IEEEquad())
235  return llvm::Type::getFP128Ty(VMContext);
236  if (&format == &llvm::APFloat::PPCDoubleDouble())
237  return llvm::Type::getPPC_FP128Ty(VMContext);
238  if (&format == &llvm::APFloat::x87DoubleExtended())
239  return llvm::Type::getX86_FP80Ty(VMContext);
240  llvm_unreachable("Unknown float format!");
241 }

243 llvm::Type *CodeGenTypes::ConvertFunctionTypeInternal(QualType QFT) {
244  assert(QFT.isCanonical());
245  const FunctionType *FT = cast<FunctionType>(QFT.getTypePtr());
246  // First, check whether we can build the full function type. If the
247  // function type depends on an incomplete type (e.g. a struct or enum), we
248  // cannot lower the function type.
249  if (!isFuncTypeConvertible(FT)) {
250  // This function's type depends on an incomplete tag type.
251 
252  // Force conversion of all the relevant record types, to make sure
253  // we re-convert the FunctionType when appropriate.
254  if (const RecordType *RT = FT->getReturnType()->getAs<RecordType>())
255  ConvertRecordDeclType(RT->getDecl());
256  if (const FunctionProtoType *FPT = dyn_cast<FunctionProtoType>(FT))
257  for (unsigned i = 0, e = FPT->getNumParams(); i != e; i++)
258  if (const RecordType *RT = FPT->getParamType(i)->getAs<RecordType>())
259  ConvertRecordDeclType(RT->getDecl());
260 
261  SkippedLayout = true;
262 
263  // Return a placeholder type.
264  return llvm::StructType::get(getLLVMContext());
265  }
266 
267  // The function type can be built; call the appropriate routines to
268  // build it.
269  const CGFunctionInfo *FI;
270  if (const FunctionProtoType *FPT = dyn_cast<FunctionProtoType>(FT)) {
271  FI = &arrangeFreeFunctionType(
272  CanQual<FunctionProtoType>::CreateUnsafe(QualType(FPT, 0)));
273  } else {
274  const FunctionNoProtoType *FNPT = cast<FunctionNoProtoType>(FT);
275  FI = &arrangeFreeFunctionType(
276  CanQual<FunctionNoProtoType>::CreateUnsafe(QualType(FNPT, 0)));
277  }
278 
279  llvm::Type *ResultType = nullptr;
280  // If there is something higher level prodding our CGFunctionInfo, then
281  // don't recurse into it again.
282  if (FunctionsBeingProcessed.count(FI)) {
283 
284  ResultType = llvm::StructType::get(getLLVMContext());
285  SkippedLayout = true;
286  } else {
287 
288  // Otherwise, we're good to go, go ahead and convert it.
289  ResultType = GetFunctionType(*FI);
290  }
291 
292  return ResultType;
293 }

295 template <bool NeedTypeInterpret = false>
296 llvm::Type *getJointMatrixINTELExtType(llvm::Type *CompTy,
297  ArrayRef<TemplateArgument> TemplateArgs,
298  const unsigned Val = 0) {
299  // TODO: we should actually have exactly 5 template parameters: 1 for
300  // type and 4 for type parameters. But in previous version of the SPIR-V
301  // spec we have Layout matrix type parameter, that was later removed.
302  // Once we update to the newest version of the spec - this should be updated.
303  assert((TemplateArgs.size() == 5 || TemplateArgs.size() == 6) &&
304  "Wrong JointMatrixINTEL template parameters number");
305  // This is required to represent optional 'Component Type Interpretation'
306  // parameter
307  std::vector<unsigned> Params;
308  for (size_t I = 1; I != TemplateArgs.size(); ++I) {
309  assert(TemplateArgs[I].getKind() == TemplateArgument::Integral &&
310  "Wrong JointMatrixINTEL template parameter");
311  Params.push_back(TemplateArgs[I].getAsIntegral().getExtValue());
312  }
313  // Don't add type interpretation for legacy matrices.
314  // Legacy matrices has 5 template parameters, while new representation
315  // has 6.
316  if (NeedTypeInterpret && TemplateArgs.size() != 5)
317  Params.push_back(Val);
318 
319  return llvm::TargetExtType::get(CompTy->getContext(),
320  "spirv.JointMatrixINTEL", {CompTy}, Params);
321 }

323 /// ConvertSYCLJointMatrixINTELType - Convert SYCL joint_matrix type
324 /// which is represented as a pointer to a structure to LLVM extension type
325 /// with the parameters that follow SPIR-V JointMatrixINTEL type.
326 /// The expected representation is:
327 /// target("spirv.JointMatrixINTEL", %element_type, %rows%, %cols%, %scope%,
328 /// %use%, (optional) %element_type_interpretation%)
329 llvm::Type *CodeGenTypes::ConvertSYCLJointMatrixINTELType(RecordDecl *RD) {
330  auto *TemplateDecl = cast<ClassTemplateSpecializationDecl>(RD);
331  ArrayRef<TemplateArgument> TemplateArgs =
332  TemplateDecl->getTemplateArgs().asArray();
333  assert(TemplateArgs[0].getKind() == TemplateArgument::Type &&
334  "1st JointMatrixINTEL template parameter must be type");
335  llvm::Type *CompTy = ConvertType(TemplateArgs[0].getAsType());
336 
337  // Per JointMatrixINTEL spec the type can have an optional
338  // 'Component Type Interpretation' parameter. We should emit it in case
339  // if on SYCL level joint matrix accepts 'bfloat16' or 'tf32' objects as
340  // matrix's components. Yet 'bfloat16' should be represented as 'int16' and
341  // 'tf32' as 'float' types.
342  if (CompTy->isStructTy()) {
343  StringRef LlvmTyName = CompTy->getStructName();
344  // Emit half/int16/float for sycl[::*]::{half,bfloat16,tf32}
345  if (LlvmTyName.starts_with("class.sycl::") ||
346  LlvmTyName.starts_with("class.__sycl_internal::"))
347  LlvmTyName = LlvmTyName.rsplit("::").second;
348  if (LlvmTyName == "half") {
349  CompTy = llvm::Type::getHalfTy(getLLVMContext());
350  return getJointMatrixINTELExtType(CompTy, TemplateArgs);
351  } else if (LlvmTyName == "tf32") {
352  CompTy = llvm::Type::getFloatTy(getLLVMContext());
353  // 'tf32' interpretation is mapped to '0'
354  return getJointMatrixINTELExtType<true>(CompTy, TemplateArgs, 0);
355  } else if (LlvmTyName == "bfloat16") {
356  CompTy = llvm::Type::getInt16Ty(getLLVMContext());
357  // 'bfloat16' interpretation is mapped to '1'
358  return getJointMatrixINTELExtType<true>(CompTy, TemplateArgs, 1);
359  } else {
360  llvm_unreachable("Wrong matrix base type!");
361  }
362  }
363  return getJointMatrixINTELExtType(CompTy, TemplateArgs);
364 }

375  if (T->isCUDADeviceBuiltinSurfaceType()) {
376  if (auto *Ty = CGM.getTargetCodeGenInfo()

834 llvm::StructType *CodeGenTypes::ConvertRecordDeclType(const RecordDecl *RD) {
835  // TagDecl's are not necessarily unique, instead use the (clang)
836  // type connected to the decl.
837  const Type *Key = Context.getTagDeclType(RD).getTypePtr();
838 
839  llvm::StructType *&Entry = RecordDeclTypes[Key];
840 
841  // If we don't have a StructType at all yet, create the forward declaration.
842  if (!Entry) {
843  Entry = llvm::StructType::create(getLLVMContext());
844  addRecordTypeName(RD, Entry, "");
845  if (RD->hasAttr<SYCLUsesAspectsAttr>())
846  CGM.addTypeWithAspects(Entry->getName(), RD);
847  }
848  llvm::StructType *Ty = Entry;
849 
850  // If this is still a forward declaration, or the LLVM type is already
851  // complete, there's nothing more to do.
852  RD = RD->getDefinition();
853  if (!RD || !RD->isCompleteDefinition() || !Ty->isOpaque())
854  return Ty;
855 
856  // Force conversion of non-virtual base classes recursively.
857  if (const CXXRecordDecl *CRD = dyn_cast<CXXRecordDecl>(RD)) {
858  for (const auto &I : CRD->bases()) {
859  if (I.isVirtual()) continue;
860  ConvertRecordDeclType(I.getType()->castAs<RecordType>()->getDecl());
861  }
862  }
863 
864  // Layout fields.
865  std::unique_ptr<CGRecordLayout> Layout = ComputeRecordLayout(RD, Ty);
866  CGRecordLayouts[Key] = std::move(Layout);
867 
868  // If this struct blocked a FunctionType conversion, then recompute whatever
869  // was derived from that.
870  // FIXME: This is hugely overconservative.
871  if (SkippedLayout)
872  TypeCache.clear();
873 
874  return Ty;
875 }
