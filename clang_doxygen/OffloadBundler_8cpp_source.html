317  /// Information about the bundles extracted from the header.
318  struct BinaryBundleInfo final : public BundleInfo {
319  /// Size of the bundle.
320  uint64_t Size = 0u;
321  /// Offset at which the bundle starts in the bundled file.
322  uint64_t Offset = 0u;

329  /// Map between a triple and the corresponding bundle information.
330  StringMap<BinaryBundleInfo> BundlesInfo;
331 
332  /// Iterator for the bundle information that is being read.
333  StringMap<BinaryBundleInfo>::iterator CurBundleInfo;
334  StringMap<BinaryBundleInfo>::iterator NextBundleInfo;
335 
336  /// Current bundle target to be written.
337  std::string CurWriteBundleTarget;
338 
339  /// Configuration options and arrays for this bundler job
340  const OffloadBundlerConfig &BundlerConfig;
341 
342 public:
343  // TODO: Add error checking from ClangOffloadBundler.cpp
344  BinaryFileHandler(const OffloadBundlerConfig &BC) : BundlerConfig(BC) {}
345 
346  ~BinaryFileHandler() final {}
347 
348  Error ReadHeader(MemoryBuffer &Input) final {
349  StringRef FC = Input.getBuffer();
350 
351  // Initialize the current bundle with the end of the container.
352  CurBundleInfo = BundlesInfo.end();
353 
354  // Check if buffer is smaller than magic string.
355  size_t ReadChars = sizeof(OFFLOAD_BUNDLER_MAGIC_STR) - 1;
356  if (ReadChars > FC.size())
357  return Error::success();
358 
359  // Check if no magic was found.
360  if (llvm::identify_magic(FC) != llvm::file_magic::offload_bundle)
361  return Error::success();
362 
363  // Read number of bundles.
364  if (ReadChars + 8 > FC.size())
365  return Error::success();
366 
367  uint64_t NumberOfBundles = Read8byteIntegerFromBuffer(FC, ReadChars);
368  ReadChars += 8;
369 
370  // Read bundle offsets, sizes and triples.
371  for (uint64_t i = 0; i < NumberOfBundles; ++i) {
372 
373  // Read offset.
374  if (ReadChars + 8 > FC.size())
375  return Error::success();
376 
377  uint64_t Offset = Read8byteIntegerFromBuffer(FC, ReadChars);
378  ReadChars += 8;
379 
380  // Read size.
381  if (ReadChars + 8 > FC.size())
382  return Error::success();
383 
384  uint64_t Size = Read8byteIntegerFromBuffer(FC, ReadChars);
385  ReadChars += 8;
386 
387  // Read triple size.
388  if (ReadChars + 8 > FC.size())
389  return Error::success();
390 
391  uint64_t TripleSize = Read8byteIntegerFromBuffer(FC, ReadChars);
392  ReadChars += 8;
393 
394  // Read triple.
395  if (ReadChars + TripleSize > FC.size())
396  return Error::success();
397 
398  StringRef Triple(&FC.data()[ReadChars], TripleSize);
399  ReadChars += TripleSize;
400 
401  // Check if the offset and size make sense.
402  if (!Offset || Offset + Size > FC.size())
403  return Error::success();
404 
405  assert(!BundlesInfo.contains(Triple) && "Triple is duplicated??");
406  BundlesInfo[Triple] = BinaryBundleInfo(Size, Offset);
407  }
408  // Set the iterator to where we will start to read.
409  CurBundleInfo = BundlesInfo.end();
410  NextBundleInfo = BundlesInfo.begin();
411  return Error::success();
412  }
413 
414  Expected<std::optional<StringRef>>
415  ReadBundleStart(MemoryBuffer &Input) final {
416  if (NextBundleInfo == BundlesInfo.end())
417  return std::nullopt;
418  CurBundleInfo = NextBundleInfo++;
419  return CurBundleInfo->first();
420  }
421 
422  Error ReadBundleEnd(MemoryBuffer &Input) final {
423  assert(CurBundleInfo != BundlesInfo.end() && "Invalid reader info!");
424  return Error::success();
425  }
426 
427  Error ReadBundle(raw_ostream &OS, MemoryBuffer &Input) final {
428  assert(CurBundleInfo != BundlesInfo.end() && "Invalid reader info!");
429  StringRef FC = Input.getBuffer();
430  OS.write(FC.data() + CurBundleInfo->second.Offset,
431  CurBundleInfo->second.Size);
432  return Error::success();
433  }

435  Error WriteHeader(raw_ostream &OS,
436  ArrayRef<std::unique_ptr<MemoryBuffer>> Inputs) final {
437 
438  // Compute size of the header.
439  uint64_t HeaderSize = 0;
440 
441  HeaderSize += sizeof(OFFLOAD_BUNDLER_MAGIC_STR) - 1;
442  HeaderSize += 8; // Number of Bundles
443 
444  for (auto &T : BundlerConfig.TargetNames) {
445  HeaderSize += 3 * 8; // Bundle offset, Size of bundle and size of triple.
446  HeaderSize += T.size(); // The triple.
447  }
448 
449  // Write to the buffer the header.
450  OS << OFFLOAD_BUNDLER_MAGIC_STR;
451 
452  Write8byteIntegerToBuffer(OS, BundlerConfig.TargetNames.size());
453 
454  unsigned Idx = 0;
455  for (auto &T : BundlerConfig.TargetNames) {
456  MemoryBuffer &MB = *Inputs[Idx++];
457  HeaderSize = alignTo(HeaderSize, BundlerConfig.BundleAlignment);
458  // Bundle offset.
459  Write8byteIntegerToBuffer(OS, HeaderSize);
460  // Size of the bundle (adds to the next bundle's offset)
461  Write8byteIntegerToBuffer(OS, MB.getBufferSize());
462  BundlesInfo[T] = BinaryBundleInfo(MB.getBufferSize(), HeaderSize);
463  HeaderSize += MB.getBufferSize();
464  // Size of the triple
465  Write8byteIntegerToBuffer(OS, T.size());
466  // Triple
467  OS << T;
468  }
469  return Error::success();
470  }

472  Error WriteBundleStart(raw_ostream &OS, StringRef TargetTriple) final {
473  CurWriteBundleTarget = TargetTriple.str();
474  return Error::success();
475  }
476 
477  Error WriteBundleEnd(raw_ostream &OS, StringRef TargetTriple) final {

481  Error WriteBundle(raw_ostream &OS, MemoryBuffer &Input) final {
482  auto BI = BundlesInfo[CurWriteBundleTarget];
483 
484  // Pad with 0 to reach specified offset.
485  size_t CurrentPos = OS.tell();
486  size_t PaddingSize = BI.Offset > CurrentPos ? BI.Offset - CurrentPos : 0;
487  for (size_t I = 0; I < PaddingSize; ++I)
488  OS.write('\0');
489  assert(OS.tell() == BI.Offset);
490 
491  OS.write(Input.getBufferStart(), Input.getBufferSize());
492 
493  return Error::success();
494  }

529 /// designated name.
530 ///
531 /// To unbundle, we just copy the contents of the designated section.
532 ///
533 /// The bundler produces object file in host target native format (e.g. ELF for
534 /// Linux). The sections it creates are:
535 ///
536 /// <OFFLOAD_BUNDLER_MAGIC_STR><target triple 1>
537 /// |
538 /// | binary data for the <target 1>'s bundle
539 /// |
540 /// ...
541 /// <OFFLOAD_BUNDLER_MAGIC_STR><target triple N>
542 /// |
543 /// | binary data for the <target N>'s bundle
544 /// |
545 /// ...
546 /// <OFFLOAD_BUNDLER_MAGIC_STR><host target>
547 /// | 0 (1 byte long)
548 /// ...
549 ///
550 /// The alignment of all the added sections is set to one to avoid padding
551 /// between concatenated parts.
552 ///

561  /// Return bundle name (<kind>-<triple>) if the provided section is an offload
562  /// section.
563  static Expected<std::optional<StringRef>>
564  IsOffloadSection(SectionRef CurSection) {
565  Expected<StringRef> NameOrErr = CurSection.getName();
566  if (!NameOrErr)
567  return NameOrErr.takeError();
568

570  if (llvm::identify_magic(*NameOrErr) != llvm::file_magic::offload_bundle)
571  return std::nullopt;

574  return NameOrErr->substr(sizeof(OFFLOAD_BUNDLER_MAGIC_STR) - 1);
575  }

578  unsigned NumberOfInputs = 0;

581  /// read from the buffers.

591  // Return a buffer with symbol names that are defined in target objects.
592  // Each symbol name is prefixed by a target name <kind>-<triple> to uniquely
593  // identify the target it belongs to, and symbol names are separated from each
594  // other by '\0' characters.
595  Expected<SmallVector<char, 0>> makeTargetSymbolTable() {
596  SmallVector<char, 0> SymbolsBuf;
597  raw_svector_ostream SymbolsOS(SymbolsBuf);
598  LLVMContext Context;
599 
600  for (unsigned I = 0; I < NumberOfInputs; ++I) {
601  if (I == BundlerConfig.HostInputIndex)
602  continue;
603 
604  // Get the list of symbols defined in the target object. Open file and
605  // check if it is a symbolic file.
606  ErrorOr<std::unique_ptr<MemoryBuffer>> BufOrErr =
607  MemoryBuffer::getFileOrSTDIN(BundlerConfig.InputFileNames[I]);
608  if (!BufOrErr)
609  return createFileError(BundlerConfig.InputFileNames[I], BufOrErr.getError());
610 
611  std::unique_ptr<MemoryBuffer> Buf = std::move(*BufOrErr);
612 
613  // Workaround for the absence of assembly parser for spir target. If this
614  // input is a bitcode for spir target we need to remove module-level
615  // inline asm from it, if there is one, and recreate the buffer with new
616  // contents.
617  // TODO: remove this workaround once spir/spirv target gets asm parser.
618  if (isBitcode((const unsigned char *)Buf->getBufferStart(),
619  (const unsigned char *)Buf->getBufferEnd()))
620  if (getTargetTriple(BundlerConfig.TargetNames[I], BundlerConfig)
621  .isSPIROrSPIRV()) {
622  SMDiagnostic Err;
623  std::unique_ptr<Module> Mod = parseIR(*Buf, Err, Context);
624  if (!Mod)
625  return createStringError(inconvertibleErrorCode(),
626  Err.getMessage());
627 
628  bool UpdateBuf = false;
629  if (!Mod->getModuleInlineAsm().empty()) {
630  Mod->setModuleInlineAsm("");
631  UpdateBuf = true;
632  }
633  for (auto I = Mod->global_begin(), E = Mod->global_end(); I != E;) {
634  GlobalVariable &GV = *I++;
635  // Do not add globals with constant address space to the tgtsym.
636  if (!GV.isDeclaration() && !GV.hasLocalLinkage() &&
637  GV.getAddressSpace() == 2) {
638  GV.replaceAllUsesWith(UndefValue::get(GV.getType()));
639  GV.dropAllReferences();
640  GV.eraseFromParent();
641  UpdateBuf = true;
642  }
643  }
644  if (UpdateBuf) {
645  SmallVector<char, 0> ModuleBuf;
646  raw_svector_ostream ModuleOS(ModuleBuf);
647  WriteBitcodeToFile(*Mod, ModuleOS);
648 
649  Buf = MemoryBuffer::getMemBufferCopy(ModuleOS.str(),
650  Buf->getBufferIdentifier());
651  }
652  }
653 
654  Expected<std::unique_ptr<Binary>> BinOrErr =
655  createBinary(Buf->getMemBufferRef(), &Context);
656 
657  // If it is not a symbolic file just ignore it since we cannot do anything
658  // with it.
659  if (!BinOrErr) {
660  if (auto Err = isNotObjectErrorInvalidFileType(BinOrErr.takeError()))
661  return std::move(Err);
662  continue;
663  }
664  auto *SF = dyn_cast<SymbolicFile>(&**BinOrErr);
665  if (!SF)
666  continue;

954  /// String that begins a line comment.
955  StringRef Comment;
956 
957  /// String that initiates a bundle.
958  std::string BundleStartString;
959 
960  /// String that closes a bundle.
961  std::string BundleEndString;
962 
963  /// Number of chars read from input.
964  size_t ReadChars = 0u;
965 
966 protected:
967  Error ReadHeader(MemoryBuffer &Input) final { return Error::success(); }
968 
969  Expected<std::optional<StringRef>>
970  ReadBundleStart(MemoryBuffer &Input) final {
971  StringRef FC = Input.getBuffer();
972 
973  // Find start of the bundle.
974  ReadChars = FC.find(BundleStartString, ReadChars);
975  if (ReadChars == FC.npos)
976  return std::nullopt;
977 
978  // Get position of the triple.
979  size_t TripleStart = ReadChars = ReadChars + BundleStartString.size();
980 
981  // Get position that closes the triple.
982  size_t TripleEnd = ReadChars = FC.find("\n", ReadChars);
983  if (TripleEnd == FC.npos)
984  return std::nullopt;
985 
986  // Next time we read after the new line.
987  ++ReadChars;
988 
989  return StringRef(&FC.data()[TripleStart], TripleEnd - TripleStart);
990  }
991 
992  Error ReadBundleEnd(MemoryBuffer &Input) final {
993  StringRef FC = Input.getBuffer();
994 
995  // Read up to the next new line.
996  assert(FC[ReadChars] == '\n' && "The bundle should end with a new line.");
997 
998  size_t TripleEnd = ReadChars = FC.find("\n", ReadChars + 1);
999  if (TripleEnd != FC.npos)
1000  // Next time we read after the new line.
1001  ++ReadChars;
1002 
1003  return Error::success();
1004  }
1005 
1006  Error ReadBundle(raw_ostream &OS, MemoryBuffer &Input) final {
1007  StringRef FC = Input.getBuffer();
1008  size_t BundleStart = ReadChars;
1009 
1010  // Find end of the bundle.
1011  size_t BundleEnd = ReadChars = FC.find(BundleEndString, ReadChars);
1012 
1013  StringRef Bundle(&FC.data()[BundleStart], BundleEnd - BundleStart);
1014  OS << Bundle;
1015 
1016  return Error::success();
1017  }

1061 /// Archive file handler. Only unbundling is supported so far.
1062 class ArchiveFileHandler final : public FileHandler {

1065

1067  const OffloadBundlerConfig &BundlerConfig;

1080  FileList, // Output is a list file with extracted object file names
1081  Object, // Output is a single object file
1082  Archive // Output is an archive with extracted objects

1085  StringSwitch<OutputType>(BundlerConfig.FilesType)
1086  .Cases("aoo", "aocx", "aocr", OutputType::FileList)
1087  .Case("ao", OutputType::Object)
1088  .Case("a", OutputType::Archive)
1089  .Default(OutputType::Unknown);
1090

1092  // unbundling.

1099  Error ReadHeader(MemoryBuffer &Input) override {
1100  assert(Mode != OutputType::Unknown && "unknown output mode");
1101 
1102  // Create archive instance for the given input.
1103  auto ArOrErr = Archive::create(Input);
1104  if (!ArOrErr)
1105  return ArOrErr.takeError();
1106  Ar = std::move(*ArOrErr);
1107 
1108  // Read all children.
1109  ssize_t ChildIndex = -1;
1110  Error Err = Error::success();
1111  for (auto &C : Ar->children(Err)) {
1112  ++ChildIndex;
1113  auto BinOrErr = C.getAsBinary();
1114 
1115  std::unique_ptr<FileHandler> FH{nullptr};
1116  std::unique_ptr<MemoryBuffer> Buf{nullptr};
1117 
1118  if (!BinOrErr) {
1119  if (auto Err = isNotObjectErrorInvalidFileType(BinOrErr.takeError()))
1120  return Err;
1121 
1122  // Handle bundled BC Files
1123  FH = std::make_unique<BinaryFileHandler>(BundlerConfig);
1124  auto MR = C.getMemoryBufferRef();
1125  assert(MR);
1126  Buf = MemoryBuffer::getMemBuffer(*MR, false);
1127  } else {
1128  auto &Bin = BinOrErr.get();
1129  if (!Bin->isObject())
1130  continue;
1131

1133  if (!CheckOrErr)
1134  return CheckOrErr.takeError();

1137  LLVM_DEBUG(outs()
1138  << "Add child to ban list. Index: " << ChildIndex << "\n");
1139  ExcludedChildIndexes.emplace(ChildIndex);
1140  }

1142  auto Obj = std::unique_ptr<ObjectFile>(cast<ObjectFile>(Bin.release()));
1143  Buf = MemoryBuffer::getMemBuffer(Obj->getMemoryBufferRef(), false);
1144 
1145  FH = std::make_unique<ObjectFileHandler>(std::move(Obj), BundlerConfig);
1146  }

1198

1203  ++ChildIndex;

1205  LLVM_DEBUG(outs() << "Skip Child. Index: " << ChildIndex << "\n");

1213  BundlerConfig.FilesType == "aocx")

1215

1219  if (auto Err = isNotObjectErrorInvalidFileType(BinOrErr.takeError()))

1229  } else

1233  if (!Bin->isObject())
1234  continue;

1236  Buf = MemoryBuffer::getMemBuffer(Obj->getMemoryBufferRef(), false);
1237  FH = std::make_unique<ObjectFileHandler>(std::move(Obj), BundlerConfig);
1238  }

1243  if (!NameOrErr)

1269  // list.
1270  OS << ChildFileName << "\n";
1271  } else if (Mode == OutputType::Object) {
1272  // Extract the bundle to the output file in single file mode.
1273  if (Error Err = FH->ReadBundle(OS, *Buf))

1290  Member.MemberName = Member.Buf->getBufferIdentifier();

1293  return Err;
1294  }

1296  if (!NameOrErr)
1297  return NameOrErr.takeError();
1298  }
1299  }
1300  if (Err)
1301  return Err;

1304  // Determine archive kind for the offload target.

1511 llvm::Expected<std::unique_ptr<llvm::MemoryBuffer>>
1512 CompressedOffloadBundle::compress(llvm::compression::Params P,
1513  const llvm::MemoryBuffer &Input,
1514  bool Verbose) {
1515  if (!llvm::compression::zstd::isAvailable() &&
1516  !llvm::compression::zlib::isAvailable())
1517  return createStringError(llvm::inconvertibleErrorCode(),
1518  "Compression not supported");
1519 
1520  llvm::Timer HashTimer("Hash Calculation Timer", "Hash calculation time",
1521  ClangOffloadBundlerTimerGroup);
1522  if (Verbose)
1523  HashTimer.startTimer();
1524  llvm::MD5 Hash;
1525  llvm::MD5::MD5Result Result;
1526  Hash.update(Input.getBuffer());
1527  Hash.final(Result);
1528  uint64_t TruncatedHash = Result.low();
1529  if (Verbose)
1530  HashTimer.stopTimer();
1531 
1532  SmallVector<uint8_t, 0> CompressedBuffer;
1533  auto BufferUint8 = llvm::ArrayRef<uint8_t>(
1534  reinterpret_cast<const uint8_t *>(Input.getBuffer().data()),
1535  Input.getBuffer().size());
1536

1538  ClangOffloadBundlerTimerGroup);
1539  if (Verbose)
1540  CompressTimer.startTimer();
1541  llvm::compression::compress(P, BufferUint8, CompressedBuffer);
1542  if (Verbose)
1543  CompressTimer.stopTimer();
1544 
1545  uint16_t CompressionMethod = static_cast<uint16_t>(P.format);
1546  uint32_t UncompressedSize = Input.getBuffer().size();
1547  uint32_t TotalFileSize = MagicNumber.size() + sizeof(TotalFileSize) +
1548  sizeof(Version) + sizeof(CompressionMethod) +
1549  sizeof(UncompressedSize) + sizeof(TruncatedHash) +
1550  CompressedBuffer.size();
1551 
1552  SmallVector<char, 0> FinalBuffer;
1553  llvm::raw_svector_ostream OS(FinalBuffer);
1554  OS << MagicNumber;
1555  OS.write(reinterpret_cast<const char *>(&Version), sizeof(Version));
1556  OS.write(reinterpret_cast<const char *>(&CompressionMethod),
1557  sizeof(CompressionMethod));
1558  OS.write(reinterpret_cast<const char *>(&TotalFileSize),
1559  sizeof(TotalFileSize));
1560  OS.write(reinterpret_cast<const char *>(&UncompressedSize),
1561  sizeof(UncompressedSize));
1562  OS.write(reinterpret_cast<const char *>(&TruncatedHash),
1563  sizeof(TruncatedHash));
1564  OS.write(reinterpret_cast<const char *>(CompressedBuffer.data()),
1565  CompressedBuffer.size());
1566 
1567  if (Verbose) {
1568  auto MethodUsed =
1569  P.format == llvm::compression::Format::Zstd ? "zstd" : "zlib";
1570  double CompressionRate =
1571  static_cast<double>(UncompressedSize) / CompressedBuffer.size();
1572  double CompressionTimeSeconds = CompressTimer.getTotalTime().getWallTime();
1573  double CompressionSpeedMBs =
1574  (UncompressedSize / (1024.0 * 1024.0)) / CompressionTimeSeconds;
1575 
1576  llvm::errs() << "Compressed bundle format version: " << Version << "\n"
1577  << "Total file size (including headers): "
1578  << formatWithCommas(TotalFileSize) << " bytes\n"
1579  << "Compression method used: " << MethodUsed << "\n"
1580  << "Compression level: " << P.level << "\n"
1581  << "Binary size before compression: "
1582  << formatWithCommas(UncompressedSize) << " bytes\n"
1583  << "Binary size after compression: "
1584  << formatWithCommas(CompressedBuffer.size()) << " bytes\n"
1585  << "Compression rate: "
1586  << llvm::format("%.2lf", CompressionRate) << "\n"
1587  << "Compression ratio: "
1588  << llvm::format("%.2lf%%", 100.0 / CompressionRate) << "\n"
1589  << "Compression speed: "
1590  << llvm::format("%.2lf MB/s", CompressionSpeedMBs) << "\n"
1591  << "Truncated MD5 hash: "
1592  << llvm::format_hex(TruncatedHash, 16) << "\n";
1593  }
1594  return llvm::MemoryBuffer::getMemBufferCopy(
1595  llvm::StringRef(FinalBuffer.data(), FinalBuffer.size()));
1596 }

1598 llvm::Expected<std::unique_ptr<llvm::MemoryBuffer>>
1599 CompressedOffloadBundle::decompress(const llvm::MemoryBuffer &Input,
1600  bool Verbose) {
1601 
1602  StringRef Blob = Input.getBuffer();
1603 
1604  if (Blob.size() < V1HeaderSize)
1605  return llvm::MemoryBuffer::getMemBufferCopy(Blob);
1606 
1607  if (llvm::identify_magic(Blob) !=
1608  llvm::file_magic::offload_bundle_compressed) {
1609  if (Verbose)
1610  llvm::errs() << "Uncompressed bundle.\n";
1611  return llvm::MemoryBuffer::getMemBufferCopy(Blob);
1612  }
1613 
1614  size_t CurrentOffset = MagicSize;
1615 
1616  uint16_t ThisVersion;
1617  memcpy(&ThisVersion, Blob.data() + CurrentOffset, sizeof(uint16_t));
1618  CurrentOffset += VersionFieldSize;
1619 
1620  uint16_t CompressionMethod;
1621  memcpy(&CompressionMethod, Blob.data() + CurrentOffset, sizeof(uint16_t));
1622  CurrentOffset += MethodFieldSize;
1623 
1624  uint32_t TotalFileSize;
1625  if (ThisVersion >= 2) {
1626  if (Blob.size() < V2HeaderSize)
1627  return createStringError(inconvertibleErrorCode(),
1628  "Compressed bundle header size too small");
1629  memcpy(&TotalFileSize, Blob.data() + CurrentOffset, sizeof(uint32_t));
1630  CurrentOffset += FileSizeFieldSize;
1631  }
1632 
1633  uint32_t UncompressedSize;
1634  memcpy(&UncompressedSize, Blob.data() + CurrentOffset, sizeof(uint32_t));
1635  CurrentOffset += UncompressedSizeFieldSize;
1636 
1637  uint64_t StoredHash;
1638  memcpy(&StoredHash, Blob.data() + CurrentOffset, sizeof(uint64_t));
1639  CurrentOffset += HashFieldSize;
1640 
1641  llvm::compression::Format CompressionFormat;
1642  if (CompressionMethod ==
1643  static_cast<uint16_t>(llvm::compression::Format::Zlib))
1644  CompressionFormat = llvm::compression::Format::Zlib;
1645  else if (CompressionMethod ==
1646  static_cast<uint16_t>(llvm::compression::Format::Zstd))
1647  CompressionFormat = llvm::compression::Format::Zstd;
1648  else
1649  return createStringError(inconvertibleErrorCode(),
1650  "Unknown compressing method");
1651 
1652  llvm::Timer DecompressTimer("Decompression Timer", "Decompression time",
1653  ClangOffloadBundlerTimerGroup);
1654  if (Verbose)
1655  DecompressTimer.startTimer();
1656 
1657  SmallVector<uint8_t, 0> DecompressedData;
1658  StringRef CompressedData = Blob.substr(CurrentOffset);
1659  if (llvm::Error DecompressionError = llvm::compression::decompress(
1660  CompressionFormat, llvm::arrayRefFromStringRef(CompressedData),
1661  DecompressedData, UncompressedSize))
1662  return createStringError(inconvertibleErrorCode(),
1663  "Could not decompress embedded file contents: " +
1664  llvm::toString(std::move(DecompressionError)));
1665 
1666  if (Verbose) {
1667  DecompressTimer.stopTimer();
1668 
1669  double DecompressionTimeSeconds =
1670  DecompressTimer.getTotalTime().getWallTime();
1671 
1672  // Recalculate MD5 hash for integrity check
1673  llvm::Timer HashRecalcTimer("Hash Recalculation Timer",
1674  "Hash recalculation time",
1675  ClangOffloadBundlerTimerGroup);
1676  HashRecalcTimer.startTimer();
1677  llvm::MD5 Hash;
1678  llvm::MD5::MD5Result Result;
1679  Hash.update(llvm::ArrayRef<uint8_t>(DecompressedData.data(),
1680  DecompressedData.size()));
1681  Hash.final(Result);
1682  uint64_t RecalculatedHash = Result.low();
1683  HashRecalcTimer.stopTimer();
1684  bool HashMatch = (StoredHash == RecalculatedHash);
1685 
1686  double CompressionRate =
1687  static_cast<double>(UncompressedSize) / CompressedData.size();
1688  double DecompressionSpeedMBs =
1689  (UncompressedSize / (1024.0 * 1024.0)) / DecompressionTimeSeconds;
1690 
1691  llvm::errs() << "Compressed bundle format version: " << ThisVersion << "\n";
1692  if (ThisVersion >= 2)
1693  llvm::errs() << "Total file size (from header): "
1694  << formatWithCommas(TotalFileSize) << " bytes\n";
1695  llvm::errs() << "Decompression method: "
1696  << (CompressionFormat == llvm::compression::Format::Zlib
1697  ? "zlib"
1698  : "zstd")
1699  << "\n"
1700  << "Size before decompression: "
1701  << formatWithCommas(CompressedData.size()) << " bytes\n"
1702  << "Size after decompression: "
1703  << formatWithCommas(UncompressedSize) << " bytes\n"
1704  << "Compression rate: "
1705  << llvm::format("%.2lf", CompressionRate) << "\n"
1706  << "Compression ratio: "
1707  << llvm::format("%.2lf%%", 100.0 / CompressionRate) << "\n"
1708  << "Decompression speed: "
1709  << llvm::format("%.2lf MB/s", DecompressionSpeedMBs) << "\n"
1710  << "Stored hash: " << llvm::format_hex(StoredHash, 16) << "\n"
1711  << "Recalculated hash: "
1712  << llvm::format_hex(RecalculatedHash, 16) << "\n"
1713  << "Hashes match: " << (HashMatch ? "Yes" : "No") << "\n";
1714  }
1715 
1716  return llvm::MemoryBuffer::getMemBufferCopy(
1717  llvm::toStringRef(DecompressedData));

1751 /// @brief Checks if a code object \p CodeObjectInfo is compatible with a given
1752 /// target \p TargetInfo.
1753 /// @link https://clang.llvm.org/docs/ClangOffloadBundler.html#bundle-entry-id
1754 bool isCodeObjectCompatible(const OffloadTargetInfo &CodeObjectInfo,
1755  const OffloadTargetInfo &TargetInfo) {
1756 
1757  // Compatible in case of exact match.
1758  if (CodeObjectInfo == TargetInfo) {
1759  DEBUG_WITH_TYPE("CodeObjectCompatibility",
1760  dbgs() << "Compatible: Exact match: \t[CodeObject: "
1761  << CodeObjectInfo.str()
1762  << "]\t:\t[Target: " << TargetInfo.str() << "]\n");
1763  return true;
1764  }
1765 
1766  // Incompatible if Kinds or Triples mismatch.
1767  if (!CodeObjectInfo.isOffloadKindCompatible(TargetInfo.OffloadKind) ||
1768  !CodeObjectInfo.Triple.isCompatibleWith(TargetInfo.Triple)) {
1769  DEBUG_WITH_TYPE(
1770  "CodeObjectCompatibility",
1771  dbgs() << "Incompatible: Kind/Triple mismatch \t[CodeObject: "
1772  << CodeObjectInfo.str() << "]\t:\t[Target: " << TargetInfo.str()
1773  << "]\n");
1774  return false;
1775  }
1776 
1777  // Incompatible if Processors mismatch.
1778  llvm::StringMap<bool> CodeObjectFeatureMap, TargetFeatureMap;
1779  std::optional<StringRef> CodeObjectProc = clang::parseTargetID(
1780  CodeObjectInfo.Triple, CodeObjectInfo.TargetID, &CodeObjectFeatureMap);
1781  std::optional<StringRef> TargetProc = clang::parseTargetID(
1782  TargetInfo.Triple, TargetInfo.TargetID, &TargetFeatureMap);
1783 
1784  // Both TargetProc and CodeObjectProc can't be empty here.
1785  if (!TargetProc || !CodeObjectProc ||
1786  CodeObjectProc.value() != TargetProc.value()) {
1787  DEBUG_WITH_TYPE("CodeObjectCompatibility",
1788  dbgs() << "Incompatible: Processor mismatch \t[CodeObject: "
1789  << CodeObjectInfo.str()
1790  << "]\t:\t[Target: " << TargetInfo.str() << "]\n");
1791  return false;
1792  }
1793 
1794  // Incompatible if CodeObject has more features than Target, irrespective of
1795  // type or sign of features.
1796  if (CodeObjectFeatureMap.getNumItems() > TargetFeatureMap.getNumItems()) {
1797  DEBUG_WITH_TYPE("CodeObjectCompatibility",
1798  dbgs() << "Incompatible: CodeObject has more features "
1799  "than target \t[CodeObject: "
1800  << CodeObjectInfo.str()
1801  << "]\t:\t[Target: " << TargetInfo.str() << "]\n");
1802  return false;
1803  }
1804 
1805  // Compatible if each target feature specified by target is compatible with
1806  // target feature of code object. The target feature is compatible if the
1807  // code object does not specify it (meaning Any), or if it specifies it
1808  // with the same value (meaning On or Off).
1809  for (const auto &CodeObjectFeature : CodeObjectFeatureMap) {
1810  auto TargetFeature = TargetFeatureMap.find(CodeObjectFeature.getKey());
1811  if (TargetFeature == TargetFeatureMap.end()) {
1812  DEBUG_WITH_TYPE(
1813  "CodeObjectCompatibility",
1814  dbgs()
1815  << "Incompatible: Value of CodeObject's non-ANY feature is "
1816  "not matching with Target feature's ANY value \t[CodeObject: "
1817  << CodeObjectInfo.str() << "]\t:\t[Target: " << TargetInfo.str()
1818  << "]\n");
1819  return false;
1820  } else if (TargetFeature->getValue() != CodeObjectFeature.getValue()) {
1821  DEBUG_WITH_TYPE(
1822  "CodeObjectCompatibility",
1823  dbgs() << "Incompatible: Value of CodeObject's non-ANY feature is "
1824  "not matching with Target feature's non-ANY value "
1825  "\t[CodeObject: "
1826  << CodeObjectInfo.str()
1827  << "]\t:\t[Target: " << TargetInfo.str() << "]\n");
1828  return false;
1829  }
1830  }
1831 
1832  // CodeObject is compatible if all features of Target are:
1833  // - either, present in the Code Object's features map with the same sign,
1834  // - or, the feature is missing from CodeObjects's features map i.e. it is
1835  // set to ANY
1836  DEBUG_WITH_TYPE(
1837  "CodeObjectCompatibility",
1838  dbgs() << "Compatible: Target IDs are compatible \t[CodeObject: "
1839  << CodeObjectInfo.str() << "]\t:\t[Target: " << TargetInfo.str()
1840  << "]\n");
1841  return true;
1842 }
