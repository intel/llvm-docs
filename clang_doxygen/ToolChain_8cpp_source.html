139 static const DriverSuffix *FindDriverSuffix(StringRef ProgName, size_t &Pos) {
140  // A list of known driver suffixes. Suffixes are compared against the
141  // program name in order. If there is a match, the frontend type is updated as
142  // necessary by applying the ModeFlag.
143  static const DriverSuffix DriverSuffixes[] = {
144  {"clang", nullptr},
145  {"clang++", "--driver-mode=g++"},
146  {"clang-c++", "--driver-mode=g++"},
147  {"clang-cc", nullptr},
148  {"clang-cpp", "--driver-mode=cpp"},
149  {"clang-g++", "--driver-mode=g++"},
150  {"clang-gcc", nullptr},
151  {"clang-cl", "--driver-mode=cl"},
152  {"cc", nullptr},
153  {"cpp", "--driver-mode=cpp"},
154  {"cl", "--driver-mode=cl"},
155  {"++", "--driver-mode=g++"},
156  {"flang", "--driver-mode=flang"},
157  };
158 
159  for (size_t i = 0; i < llvm::array_lengthof(DriverSuffixes); ++i) {
160  StringRef Suffix(DriverSuffixes[i].Suffix);
161  if (ProgName.endswith(Suffix)) {
162  Pos = ProgName.size() - Suffix.size();
163  return &DriverSuffixes[i];
164  }
165  }
166  return nullptr;
167 }

169 /// Normalize the program name from argv[0] by stripping the file extension if
170 /// present and lower-casing the string on Windows.
171 static std::string normalizeProgramName(llvm::StringRef Argv0) {
172  std::string ProgName = std::string(llvm::sys::path::stem(Argv0));
173  if (is_style_windows(llvm::sys::path::Style::native)) {
174  // Transform to lowercase for case insensitive file systems.
175  std::transform(ProgName.begin(), ProgName.end(), ProgName.begin(),
176  ::tolower);
177  }
178  return ProgName;
179 }

181 static const DriverSuffix *parseDriverSuffix(StringRef ProgName, size_t &Pos) {
182  // Try to infer frontend type and default target from the program name by
183  // comparing it against DriverSuffixes in order.
184 
185  // If there is a match, the function tries to identify a target as prefix.
186  // E.g. "x86_64-linux-clang" as interpreted as suffix "clang" with target
187  // prefix "x86_64-linux". If such a target prefix is found, it may be
188  // added via -target as implicit first argument.
189  const DriverSuffix *DS = FindDriverSuffix(ProgName, Pos);
190 
191  if (!DS) {
192  // Try again after stripping any trailing version number:
193  // clang++3.5 -> clang++
194  ProgName = ProgName.rtrim("0123456789.");
195  DS = FindDriverSuffix(ProgName, Pos);
196  }
197 
198  if (!DS) {
199  // Try again after stripping trailing -component.
200  // clang++-tot -> clang++
201  ProgName = ProgName.slice(0, ProgName.rfind('-'));
202  DS = FindDriverSuffix(ProgName, Pos);
203  }
204  return DS;
205 }

207 ParsedClangName
208 ToolChain::getTargetAndModeFromProgramName(StringRef PN) {
209  std::string ProgName = normalizeProgramName(PN);
210  size_t SuffixPos;
211  const DriverSuffix *DS = parseDriverSuffix(ProgName, SuffixPos);
212  if (!DS)
213  return {};
214  size_t SuffixEnd = SuffixPos + strlen(DS->Suffix);
215 
216  size_t LastComponent = ProgName.rfind('-', SuffixPos);
217  if (LastComponent == std::string::npos)
218  return ParsedClangName(ProgName.substr(0, SuffixEnd), DS->ModeFlag);
219  std::string ModeSuffix = ProgName.substr(LastComponent + 1,
220  SuffixEnd - LastComponent - 1);
221 
222  // Infer target from the prefix.
223  StringRef Prefix(ProgName);
224  Prefix = Prefix.slice(0, LastComponent);
225  std::string IgnoredError;
226  bool IsRegistered =
227  llvm::TargetRegistry::lookupTarget(std::string(Prefix), IgnoredError);
228  return ParsedClangName{std::string(Prefix), ModeSuffix, DS->ModeFlag,
229  IsRegistered};
230 }

461 static StringRef getArchNameForCompilerRTLib(const ToolChain &TC,
462  const ArgList &Args) {
463  const llvm::Triple &Triple = TC.getTriple();
464  bool IsWindows = Triple.isOSWindows();
465 
466  if (TC.getArch() == llvm::Triple::arm || TC.getArch() == llvm::Triple::armeb)
467  return (arm::getARMFloatABI(TC, Args) == arm::FloatABI::Hard && !IsWindows)
468  ? "armhf"
469  : "arm";
470 
471  // For historic reasons, Android library is using i686 instead of i386.
472  if (TC.getArch() == llvm::Triple::x86 && Triple.isAndroid())
473  return "i686";
474 
475  if (TC.getArch() == llvm::Triple::x86_64 && Triple.isX32())
476  return "x32";
477 
478  return llvm::Triple::getArchTypeName(TC.getArch());
479 }

518 std::string ToolChain::buildCompilerRTBasename(const llvm::opt::ArgList &Args,
519  StringRef Component,
520  FileType Type,
521  bool AddArch) const {
522  const llvm::Triple &TT = getTriple();
523  bool IsITANMSVCWindows =
524  TT.isWindowsMSVCEnvironment() || TT.isWindowsItaniumEnvironment();
525 
526  const char *Prefix =
527  IsITANMSVCWindows || Type == ToolChain::FT_Object ? "" : "lib";
528  const char *Suffix;
529  switch (Type) {
530  case ToolChain::FT_Object:
531  Suffix = IsITANMSVCWindows ? ".obj" : ".o";
532  break;
533  case ToolChain::FT_Static:
534  Suffix = IsITANMSVCWindows ? ".lib" : ".a";
535  break;
536  case ToolChain::FT_Shared:
537  Suffix = TT.isOSWindows()
538  ? (TT.isWindowsGNUEnvironment() ? ".dll.a" : ".lib")
539  : ".so";
540  break;
541  }
542 
543  std::string ArchAndEnv;
544  if (AddArch) {
545  StringRef Arch = getArchNameForCompilerRTLib(*this, Args);
546  const char *Env = TT.isAndroid() ? "-android" : "";
547  ArchAndEnv = ("-" + Arch + Env).str();
548  }
549  return (Prefix + Twine("clang_rt.") + Component + ArchAndEnv + Suffix).str();
550 }

552 std::string ToolChain::getCompilerRT(const ArgList &Args, StringRef Component,
553  FileType Type) const {
554  // Check for runtime files in the new layout without the architecture first.
555  std::string CRTBasename =
556  buildCompilerRTBasename(Args, Component, Type, /*AddArch=*/false);
557  for (const auto &LibPath : getLibraryPaths()) {
558  SmallString<128> P(LibPath);
559  llvm::sys::path::append(P, CRTBasename);
560  if (getVFS().exists(P))
561  return std::string(P.str());
562  }
563 
564  // Fall back to the old expected compiler-rt name if the new one does not
565  // exist.
566  CRTBasename =
567  buildCompilerRTBasename(Args, Component, Type, /*AddArch=*/true);
568  SmallString<128> Path(getCompilerRTPath());
569  llvm::sys::path::append(Path, CRTBasename);
570  return std::string(Path.str());
571 }

658  // Get -fuse-ld= first to prevent -Wunused-command-line-argument. -fuse-ld= is
659  // considered as the linker flavor, e.g. "bfd", "gold", or "lld".
660  const Arg* A = Args.getLastArg(options::OPT_fuse_ld_EQ);
661  StringRef UseLinker = A ? A->getValue() : CLANG_DEFAULT_LINKER;
662 
663  // --ld-path= takes precedence over -fuse-ld= and specifies the executable
664  // name. -B, COMPILER_PATH and PATH and consulted if the value does not
665  // contain a path component separator.
666  if (const Arg *A = Args.getLastArg(options::OPT_ld_path_EQ)) {
667  std::string Path(A->getValue());
668  if (!Path.empty()) {
669  if (llvm::sys::path::parent_path(Path).empty())
670  Path = GetProgramPath(A->getValue());
671  if (llvm::sys::fs::can_execute(Path))
672  return std::string(Path);
673  }
674  getDriver().Diag(diag::err_drv_invalid_linker_name) << A->getAsString(Args);
675  return GetProgramPath(getDefaultLinker());
676  }
677  // If we're passed -fuse-ld= with no argument, or with the argument ld,
678  // then use whatever the default system linker is.
679  if (UseLinker.empty() || UseLinker == "ld") {
680  const char *DefaultLinker = getDefaultLinker();
681  if (llvm::sys::path::is_absolute(DefaultLinker))
682  return std::string(DefaultLinker);
683  else
684  return GetProgramPath(DefaultLinker);
685  }
686 
687  // Extending -fuse-ld= to an absolute or relative path is unexpected. Checking
688  // for the linker flavor is brittle. In addition, prepending "ld." or "ld64."
689  // to a relative path is surprising. This is more complex due to priorities
690  // among -B, COMPILER_PATH and PATH. --ld-path= should be used instead.
691  if (UseLinker.contains('/'))
692  getDriver().Diag(diag::warn_drv_fuse_ld_path);
693 
694  if (llvm::sys::path::is_absolute(UseLinker)) {
695  // If we're passed what looks like an absolute path, don't attempt to
696  // second-guess that.
697  if (llvm::sys::fs::can_execute(UseLinker))
698  return std::string(UseLinker);
699  } else {
700  llvm::SmallString<8> LinkerName;
701  if (Triple.isOSDarwin())
702  LinkerName.append("ld64.");
703  else
704  LinkerName.append("ld.");
705  LinkerName.append(UseLinker);
706 
707  std::string LinkerPath(GetProgramPath(LinkerName.c_str()));
708  if (llvm::sys::fs::can_execute(LinkerPath)) {
709  if (LinkerIsLLD)
710  *LinkerIsLLD = UseLinker == "lld";
711  return LinkerPath;
712  }
713  }
714 
715  if (A)
716  getDriver().Diag(diag::err_drv_invalid_linker_name) << A->getAsString(Args);
717 
718  return GetProgramPath(getDefaultLinker());
719 }

1196  const llvm::opt::DerivedArgList &Args, bool SameTripleAsHost,
1197  SmallVectorImpl<llvm::opt::Arg *> &AllocatedArgs,
1198  Action::OffloadKind DeviceOffloadKind) const {

1200  DeviceOffloadKind == Action::OFK_SYCL) &&
1201  "requires OpenMP or SYCL offload kind");
1202  DerivedArgList *DAL = new DerivedArgList(Args.getBaseArgs());
1203  const OptTable &Opts = getDriver().getOpts();
1204  bool Modified = false;
1205 
1206  // Handle -Xopenmp-target and -Xsycl-target-frontend flags
1207  for (auto *A : Args) {
1208  // Exclude flags which may only apply to the host toolchain.
1209  // Do not exclude flags when the host triple (AuxTriple)
1210  // matches the current toolchain triple. If it is not present
1211  // at all, target and host share a toolchain.
1212  if (A->getOption().matches(options::OPT_m_Group)) {

1214  // compilation, except for SYCL which uses --offload-arch.
1215  if (SameTripleAsHost || (getTriple().getArch() == llvm::Triple::amdgcn &&
1216  DeviceOffloadKind != Action::OFK_SYCL))
1217  DAL->append(A);
1218  else
1219  Modified = true;
1220  continue;
1221  }
1222 
1223  // Exclude -fsycl
1224  if (A->getOption().matches(options::OPT_fsycl)) {
1225  Modified = true;
1226  continue;
1227  }
1228 
1229  unsigned Index = 0;
1230  unsigned Prev;
1231  bool XOffloadTargetNoTriple;
1232 
1233  // TODO: functionality between OpenMP offloading and SYCL offloading
1234  // is similar, can be improved
1235  if (DeviceOffloadKind == Action::OFK_OpenMP) {
1236  XOffloadTargetNoTriple =
1237  A->getOption().matches(options::OPT_Xopenmp_target);
