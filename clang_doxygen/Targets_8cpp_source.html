53 /// DefineStd - Define a macro name and standard variants. For example if
54 /// MacroName is "unix", then this will define "__unix", "__unix__", and "unix"
55 /// when in GNU mode.
56 void DefineStd(MacroBuilder &Builder, StringRef MacroName,
57  const LangOptions &Opts) {
58  assert(MacroName[0] != '_' && "Identifier should be in the user's namespace");
59 
60  // If in GNU mode (e.g. -std=gnu99 but not -std=c99) define the raw identifier
61  // in the user's namespace.
62  if (Opts.GNUMode)
63  Builder.defineMacro(MacroName);
64 
65  // Define __unix.
66  Builder.defineMacro("__" + MacroName);
67 
68  // Define __unix__.
69  Builder.defineMacro("__" + MacroName + "__");
70 }

72 void defineCPUMacros(MacroBuilder &Builder, StringRef CPUName, bool Tuning) {
73  Builder.defineMacro("__" + CPUName);
74  Builder.defineMacro("__" + CPUName + "__");
75  if (Tuning)
76  Builder.defineMacro("__tune_" + CPUName + "__");
77 }

79 void addCygMingDefines(const LangOptions &Opts, MacroBuilder &Builder) {
80  // Mingw and cygwin define __declspec(a) to __attribute__((a)). Clang
81  // supports __declspec natively under -fms-extensions, but we define a no-op
82  // __declspec macro anyway for pre-processor compatibility.
83  if (Opts.MicrosoftExt)
84  Builder.defineMacro("__declspec", "__declspec");
85  else
86  Builder.defineMacro("__declspec(a)", "__attribute__((a))");
87 
88  if (!Opts.MicrosoftExt) {
89  // Provide macros for all the calling convention keywords. Provide both
90  // single and double underscore prefixed variants. These are available on
91  // x64 as well as x86, even though they have no effect.
92  const char *CCs[] = {"cdecl", "stdcall", "fastcall", "thiscall", "pascal"};
93  for (const char *CC : CCs) {
94  std::string GCCSpelling = "__attribute__((__";
95  GCCSpelling += CC;
96  GCCSpelling += "__))";
97  Builder.defineMacro(Twine("_") + CC, GCCSpelling);
98  Builder.defineMacro(Twine("__") + CC, GCCSpelling);
99  }
100  }
101 }

690 /// CreateTargetInfo - Return the target info object for the specified target
691 /// options.
692 TargetInfo *
693 TargetInfo::CreateTargetInfo(DiagnosticsEngine &Diags,
694  const std::shared_ptr<TargetOptions> &Opts) {
695  llvm::Triple Triple(Opts->Triple);
696 
697  // Construct the target
698  std::unique_ptr<TargetInfo> Target(AllocateTarget(Triple, *Opts));
699  if (!Target) {
700  Diags.Report(diag::err_target_unknown_triple) << Triple.str();
701  return nullptr;
702  }
703  Target->TargetOpts = Opts;
704 
705  // Set the target CPU if specified.
706  if (!Opts->CPU.empty() && !Target->setCPU(Opts->CPU)) {
707  Diags.Report(diag::err_target_unknown_cpu) << Opts->CPU;
708  SmallVector<StringRef, 32> ValidList;
709  Target->fillValidCPUList(ValidList);
710  if (!ValidList.empty())
711  Diags.Report(diag::note_valid_options) << llvm::join(ValidList, ", ");
712  return nullptr;
713  }
714 
715  // Check the TuneCPU name if specified.
716  if (!Opts->TuneCPU.empty() &&
717  !Target->isValidTuneCPUName(Opts->TuneCPU)) {
718  Diags.Report(diag::err_target_unknown_cpu) << Opts->TuneCPU;
719  SmallVector<StringRef, 32> ValidList;
720  Target->fillValidTuneCPUList(ValidList);
721  if (!ValidList.empty())
722  Diags.Report(diag::note_valid_options) << llvm::join(ValidList, ", ");
723  return nullptr;
724  }
725 
726  // Set the target ABI if specified.
727  if (!Opts->ABI.empty() && !Target->setABI(Opts->ABI)) {
728  Diags.Report(diag::err_target_unknown_abi) << Opts->ABI;
729  return nullptr;
730  }
731 
732  // Set the fp math unit.
733  if (!Opts->FPMath.empty() && !Target->setFPMath(Opts->FPMath)) {
734  Diags.Report(diag::err_target_unknown_fpmath) << Opts->FPMath;
735  return nullptr;
736  }
737 
738  // Compute the default target features, we need the target to handle this
739  // because features may have dependencies on one another.
740  if (!Target->initFeatureMap(Opts->FeatureMap, Diags, Opts->CPU,
741  Opts->FeaturesAsWritten))
742  return nullptr;
743 
744  // Add the features to the compile options.
745  Opts->Features.clear();
746  for (const auto &F : Opts->FeatureMap)
747  Opts->Features.push_back((F.getValue() ? "+" : "-") + F.getKey().str());
748  // Sort here, so we handle the features in a predictable order. (This matters
749  // when we're dealing with features that overlap.)
750  llvm::sort(Opts->Features);
751 
752  if (!Target->handleTargetFeatures(Opts->Features, Diags))
753  return nullptr;
754 
755  Target->setSupportedOpenCLOpts();
756  Target->setCommandLineOpenCLOpts();
757  Target->setMaxAtomicWidth();
758 
759  if (!Opts->DarwinTargetVariantTriple.empty())
760  Target->DarwinTargetVariantTriple =
761  llvm::Triple(Opts->DarwinTargetVariantTriple);
762 
763  if (!Target->validateTarget(Diags))
764  return nullptr;
765 
766  Target->CheckFixedPointBits();
767 
768  return Target.release();
769 }

781  << Opts.getOpenCLVersionTuple().getAsString();
782  };

793  OpenCLOptions::diagnoseFeatureExtensionDifferences(*this, Diags);
