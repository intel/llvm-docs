147 void CodeGenFunction::CGFPOptionsRAII::ConstructorHelper(FPOptions FPFeatures) {
148  OldFPFeatures = CGF.CurFPFeatures;
149  CGF.CurFPFeatures = FPFeatures;
150 
151  OldExcept = CGF.Builder.getDefaultConstrainedExcept();
152  OldRounding = CGF.Builder.getDefaultConstrainedRounding();
153 
154  if (OldFPFeatures == FPFeatures)
155  return;
156 
157  FMFGuard.emplace(CGF.Builder);
158 
159  llvm::RoundingMode NewRoundingBehavior = FPFeatures.getRoundingMode();
160  CGF.Builder.setDefaultConstrainedRounding(NewRoundingBehavior);
161  auto NewExceptionBehavior =
162  ToConstrainedExceptMD(static_cast<LangOptions::FPExceptionModeKind>(
163  FPFeatures.getExceptionMode()));
164  CGF.Builder.setDefaultConstrainedExcept(NewExceptionBehavior);
165 
166  CGF.SetFastMathFlags(FPFeatures);
167 
168  assert((CGF.CurFuncDecl == nullptr || CGF.Builder.getIsFPConstrained() ||
169  isa<CXXConstructorDecl>(CGF.CurFuncDecl) ||
170  isa<CXXDestructorDecl>(CGF.CurFuncDecl) ||
171  (NewExceptionBehavior == llvm::fp::ebIgnore &&
172  NewRoundingBehavior == llvm::RoundingMode::NearestTiesToEven)) &&
173  "FPConstrained should be enabled on entire function");
174 
175  auto mergeFnAttrValue = [&](StringRef Name, bool Value) {
176  auto OldValue =
177  CGF.CurFn->getFnAttribute(Name).getValueAsBool();
178  auto NewValue = OldValue & Value;
179  if (OldValue != NewValue)
180  CGF.CurFn->addFnAttr(Name, llvm::toStringRef(NewValue));
181  };
182  mergeFnAttrValue("no-infs-fp-math", FPFeatures.getNoHonorInfs());
183  mergeFnAttrValue("no-nans-fp-math", FPFeatures.getNoHonorNaNs());
184  mergeFnAttrValue("no-signed-zeros-fp-math", FPFeatures.getNoSignedZero());
185  mergeFnAttrValue(
186  "unsafe-fp-math",
187  FPFeatures.getAllowFPReassociate() && FPFeatures.getAllowReciprocal() &&
188  FPFeatures.getAllowApproxFunc() && FPFeatures.getNoSignedZero() &&
189  FPFeatures.allowFPContractAcrossStatement());
190 }

236 TypeEvaluationKind CodeGenFunction::getEvaluationKind(QualType type) {
237  type = type.getCanonicalType();
238  while (true) {
239  switch (type->getTypeClass()) {
240 #define TYPE(name, parent)
241 #define ABSTRACT_TYPE(name, parent)
242 #define NON_CANONICAL_TYPE(name, parent) case Type::name:
243 #define DEPENDENT_TYPE(name, parent) case Type::name:
244 #define NON_CANONICAL_UNLESS_DEPENDENT_TYPE(name, parent) case Type::name:
245 #include "clang/AST/TypeNodes.inc"
246  llvm_unreachable("non-canonical or dependent type in IR-generation");
247 
248  case Type::Auto:
249  case Type::DeducedTemplateSpecialization:
250  llvm_unreachable("undeduced type in IR-generation");
251 
252  // Various scalar types.
253  case Type::Builtin:
254  case Type::Pointer:
255  case Type::BlockPointer:
256  case Type::LValueReference:
257  case Type::RValueReference:
258  case Type::MemberPointer:
259  case Type::Vector:
260  case Type::ExtVector:
261  case Type::ConstantMatrix:
262  case Type::FunctionProto:
263  case Type::FunctionNoProto:
264  case Type::Enum:
265  case Type::ObjCObjectPointer:
266  case Type::Pipe:
267  case Type::BitInt:
268  return TEK_Scalar;
269 
270  // Complexes.
271  case Type::Complex:
272  return TEK_Complex;
273 
274  // Arrays, records, and Objective-C objects.
275  case Type::ConstantArray:
276  case Type::IncompleteArray:
277  case Type::VariableArray:
278  case Type::Record:
279  case Type::ObjCObject:
280  case Type::ObjCInterface:
281  case Type::ArrayParameter:
282  return TEK_Aggregate;
283 
284  // We operate on atomic values according to their underlying type.
285  case Type::Atomic:
286  type = cast<AtomicType>(type)->getValueType();
287  continue;
288  }
289  llvm_unreachable("unknown type kind!");
290  }
291 }

293 llvm::DebugLoc CodeGenFunction::EmitReturnBlock() {
294  // For cleanliness, we try to avoid emitting the return block for
295  // simple cases.
296  llvm::BasicBlock *CurBB = Builder.GetInsertBlock();
297 
298  if (CurBB) {
299  assert(!CurBB->getTerminator() && "Unexpected terminated block.");
300 
301  // We have a valid insert point, reuse it if it is empty or there are no
302  // explicit jumps to the return block.
303  if (CurBB->empty() || ReturnBlock.getBlock()->use_empty()) {
304  ReturnBlock.getBlock()->replaceAllUsesWith(CurBB);
305  delete ReturnBlock.getBlock();
306  ReturnBlock = JumpDest();
307  } else
308  EmitBlock(ReturnBlock.getBlock());
309  return llvm::DebugLoc();
310  }
311 
312  // Otherwise, if the return block is the target of a single direct
313  // branch then we can just put the code in that block instead. This
314  // cleans up functions which started with a unified return block.
315  if (ReturnBlock.getBlock()->hasOneUse()) {
316  llvm::BranchInst *BI =
317  dyn_cast<llvm::BranchInst>(*ReturnBlock.getBlock()->user_begin());
318  if (BI && BI->isUnconditional() &&
319  BI->getSuccessor(0) == ReturnBlock.getBlock()) {
320  // Record/return the DebugLoc of the simple 'return' expression to be used
321  // later by the actual 'ret' instruction.
322  llvm::DebugLoc Loc = BI->getDebugLoc();
323  Builder.SetInsertPoint(BI->getParent());
324  BI->eraseFromParent();
325  delete ReturnBlock.getBlock();
326  ReturnBlock = JumpDest();
327  return Loc;
328  }
329  }
330 
331  // FIXME: We are at an unreachable point, there is no reason to emit the block
332  // unless it has uses. However, we still need a place to put the debug
333  // region.end for now.
334 
335  EmitBlock(ReturnBlock.getBlock());
336  return llvm::DebugLoc();
337 }

348 void CodeGenFunction::FinishFunction(SourceLocation EndLoc) {
349  assert(BreakContinueStack.empty() &&
350  "mismatched push/pop in break/continue stack!");
351 
352  bool OnlySimpleReturnStmts = NumSimpleReturnExprs > 0
353  && NumSimpleReturnExprs == NumReturnExprs
354  && ReturnBlock.getBlock()->use_empty();
355  // Usually the return expression is evaluated before the cleanup
356  // code. If the function contains only a simple return statement,
357  // such as a constant, the location before the cleanup code becomes
358  // the last useful breakpoint in the function, because the simple
359  // return expression will be evaluated after the cleanup code. To be
360  // safe, set the debug location for cleanup code to the location of
361  // the return statement. Otherwise the cleanup code should be at the
362  // end of the function's lexical scope.
363  //
364  // If there are multiple branches to the return block, the branch
365  // instructions will get the location of the return statements and
366  // all will be fine.
367  if (CGDebugInfo *DI = getDebugInfo()) {
368  if (OnlySimpleReturnStmts)
369  DI->EmitLocation(Builder, LastStopPoint);
370  else
371  DI->EmitLocation(Builder, EndLoc);
372  }
373 
374  // Pop any cleanups that might have been associated with the
375  // parameters. Do this in whatever block we're currently in; it's
376  // important to do this before we enter the return block or return
377  // edges will be *really* confused.
378  bool HasCleanups = EHStack.stable_begin() != PrologueCleanupDepth;
379  bool HasOnlyLifetimeMarkers =
380  HasCleanups && EHStack.containsOnlyLifetimeMarkers(PrologueCleanupDepth);
381  bool EmitRetDbgLoc = !HasCleanups || HasOnlyLifetimeMarkers;
382 
383  std::optional<ApplyDebugLocation> OAL;
384  if (HasCleanups) {
385  // Make sure the line table doesn't jump back into the body for
386  // the ret after it's been at EndLoc.
387  if (CGDebugInfo *DI = getDebugInfo()) {
388  if (OnlySimpleReturnStmts)
389  DI->EmitLocation(Builder, EndLoc);
390  else
391  // We may not have a valid end location. Try to apply it anyway, and
392  // fall back to an artificial location if needed.
393  OAL = ApplyDebugLocation::CreateDefaultArtificial(*this, EndLoc);
394  }
395 
396  PopCleanupBlocks(PrologueCleanupDepth);
397  }
398 
399  // Emit function epilog (to return).
400  llvm::DebugLoc Loc = EmitReturnBlock();
401 
402  if (ShouldInstrumentFunction()) {
403  if (CGM.getCodeGenOpts().InstrumentFunctions)
404  CurFn->addFnAttr("instrument-function-exit", "__cyg_profile_func_exit");
405  if (CGM.getCodeGenOpts().InstrumentFunctionsAfterInlining)
406  CurFn->addFnAttr("instrument-function-exit-inlined",
407  "__cyg_profile_func_exit");
408  }
409 
410  // Emit debug descriptor for function end.
411  if (CGDebugInfo *DI = getDebugInfo())
412  DI->EmitFunctionEnd(Builder, CurFn);
413 
414  // Reset the debug location to that of the simple 'return' expression, if any
415  // rather than that of the end of the function's scope '}'.
416  ApplyDebugLocation AL(*this, Loc);
417  EmitFunctionEpilog(*CurFnInfo, EmitRetDbgLoc, EndLoc);
418  EmitEndEHSpec(CurCodeDecl);
419 
420  assert(EHStack.empty() &&
421  "did not remove all scopes from cleanup stack!");
422 
423  // If someone did an indirect goto, emit the indirect goto block at the end of
424  // the function.
425  if (IndirectBranch) {
426  EmitBlock(IndirectBranch->getParent());
427  Builder.ClearInsertionPoint();
428  }
429 
430  // If some of our locals escaped, insert a call to llvm.localescape in the
431  // entry block.
432  if (!EscapedLocals.empty()) {
433  // Invert the map from local to index into a simple vector. There should be
434  // no holes.
435  SmallVector<llvm::Value *, 4> EscapeArgs;
436  EscapeArgs.resize(EscapedLocals.size());
437  for (auto &Pair : EscapedLocals)
438  EscapeArgs[Pair.second] = Pair.first;
439  llvm::Function *FrameEscapeFn = llvm::Intrinsic::getDeclaration(
440  &CGM.getModule(), llvm::Intrinsic::localescape);
441  CGBuilderTy(*this, AllocaInsertPt).CreateCall(FrameEscapeFn, EscapeArgs);
442  }
443 
444  // Remove the AllocaInsertPt instruction, which is just a convenience for us.
445  llvm::Instruction *Ptr = AllocaInsertPt;
446  AllocaInsertPt = nullptr;
447  Ptr->eraseFromParent();
448 
449  // PostAllocaInsertPt, if created, was lazily created when it was required,
450  // remove it now since it was just created for our own convenience.
451  if (PostAllocaInsertPt) {
452  llvm::Instruction *PostPtr = PostAllocaInsertPt;
453  PostAllocaInsertPt = nullptr;
454  PostPtr->eraseFromParent();
455  }
456 
457  // If someone took the address of a label but never did an indirect goto, we
458  // made a zero entry PHI node, which is illegal, zap it now.
459  if (IndirectBranch) {
460  llvm::PHINode *PN = cast<llvm::PHINode>(IndirectBranch->getAddress());
461  if (PN->getNumIncomingValues() == 0) {
462  PN->replaceAllUsesWith(llvm::UndefValue::get(PN->getType()));
463  PN->eraseFromParent();
464  }
465  }
466 
467  EmitIfUsed(*this, EHResumeBlock);
468  EmitIfUsed(*this, TerminateLandingPad);
469  EmitIfUsed(*this, TerminateHandler);
470  EmitIfUsed(*this, UnreachableBlock);
471 
472  for (const auto &FuncletAndParent : TerminateFunclets)
473  EmitIfUsed(*this, FuncletAndParent.second);
474 
475  if (CGM.getCodeGenOpts().EmitDeclMetadata)
476  EmitDeclMetadata();
477 
478  for (const auto &R : DeferredReplacements) {
479  if (llvm::Value *Old = R.first) {
480  Old->replaceAllUsesWith(R.second);
481  cast<llvm::Instruction>(Old)->eraseFromParent();
482  }
483  }
484  DeferredReplacements.clear();
485 
486  // Eliminate CleanupDestSlot alloca by replacing it with SSA values and
487  // PHIs if the current function is a coroutine. We don't do it for all
488  // functions as it may result in slight increase in numbers of instructions
489  // if compiled with no optimizations. We do it for coroutine as the lifetime
490  // of CleanupDestSlot alloca make correct coroutine frame building very
491  // difficult.
492  if (NormalCleanupDest.isValid() && isCoroutine()) {
493  llvm::DominatorTree DT(*CurFn);
494  llvm::PromoteMemToReg(
495  cast<llvm::AllocaInst>(NormalCleanupDest.getPointer()), DT);
496  NormalCleanupDest = Address::invalid();
497  }
498 
499  // Scan function arguments for vector width.
500  for (llvm::Argument &A : CurFn->args())
501  if (auto *VT = dyn_cast<llvm::VectorType>(A.getType()))
502  LargestVectorWidth =
503  std::max((uint64_t)LargestVectorWidth,
504  VT->getPrimitiveSizeInBits().getKnownMinValue());
505 
506  // Update vector width based on return type.
507  if (auto *VT = dyn_cast<llvm::VectorType>(CurFn->getReturnType()))
508  LargestVectorWidth =
509  std::max((uint64_t)LargestVectorWidth,
510  VT->getPrimitiveSizeInBits().getKnownMinValue());
511 
512  if (CurFnInfo->getMaxVectorWidth() > LargestVectorWidth)
513  LargestVectorWidth = CurFnInfo->getMaxVectorWidth();
514 
515  // Add the min-legal-vector-width attribute. This contains the max width from:
516  // 1. min-vector-width attribute used in the source program.
517  // 2. Any builtins used that have a vector width specified.
518  // 3. Values passed in and out of inline assembly.
519  // 4. Width of vector arguments and return types for this function.
520  // 5. Width of vector arguments and return types for functions called by this
521  // function.
522  if (getContext().getTargetInfo().getTriple().isX86())
523  CurFn->addFnAttr("min-legal-vector-width",
524  llvm::utostr(LargestVectorWidth));
525 
526  // Add vscale_range attribute if appropriate.
527  std::optional<std::pair<unsigned, unsigned>> VScaleRange =
528  getContext().getTargetInfo().getVScaleRange(getLangOpts());
529  if (VScaleRange) {
530  CurFn->addFnAttr(llvm::Attribute::getWithVScaleRangeArgs(
531  getLLVMContext(), VScaleRange->first, VScaleRange->second));
532  }
533 
534  // If we generated an unreachable return block, delete it now.
535  if (ReturnBlock.isValid() && ReturnBlock.getBlock()->use_empty()) {
536  Builder.ClearInsertionPoint();
537  ReturnBlock.getBlock()->eraseFromParent();
538  }
539  if (ReturnValue.isValid()) {
540  auto *RetAlloca =
541  dyn_cast<llvm::AllocaInst>(ReturnValue.emitRawPointer(*this));
542  if (RetAlloca && RetAlloca->use_empty()) {
543  RetAlloca->eraseFromParent();
544  ReturnValue = Address::invalid();
545  }
546  }
547 }

912 void CodeGenFunction::StartFunction(GlobalDecl GD, QualType RetTy,
913  llvm::Function *Fn,
914  const CGFunctionInfo &FnInfo,
915  const FunctionArgList &Args,
916  SourceLocation Loc,
917  SourceLocation StartLoc) {
918  assert(!CurFn &&
919  "Do not use a CodeGenFunction object for more than one function");
920 
921  const Decl *D = GD.getDecl();
922 
923  DidCallStackSave = false;
924  CurCodeDecl = D;
925  const FunctionDecl *FD = dyn_cast_or_null<FunctionDecl>(D);
926  if (FD && FD->usesSEHTry())
927  CurSEHParent = GD;
928  CurFuncDecl = (D ? D->getNonClosureContext() : nullptr);
929  FnRetTy = RetTy;
930  CurFn = Fn;
931  CurFnInfo = &FnInfo;
932  assert(CurFn->isDeclaration() && "Function already has body?");
933 
934  // If this function is ignored for any of the enabled sanitizers,
935  // disable the sanitizer for the function.
936  do {
937 #define SANITIZER(NAME, ID) \
938  if (SanOpts.empty()) \
939  break; \
940  if (SanOpts.has(SanitizerKind::ID)) \
941  if (CGM.isInNoSanitizeList(SanitizerKind::ID, Fn, Loc)) \
942  SanOpts.set(SanitizerKind::ID, false);
943 
944 #include "clang/Basic/Sanitizers.def"
945 #undef SANITIZER
946  } while (false);
947 
948  if (D) {
949  const bool SanitizeBounds = SanOpts.hasOneOf(SanitizerKind::Bounds);
950  SanitizerMask no_sanitize_mask;
951  bool NoSanitizeCoverage = false;
952 
953  for (auto *Attr : D->specific_attrs<NoSanitizeAttr>()) {
954  no_sanitize_mask |= Attr->getMask();
955  // SanitizeCoverage is not handled by SanOpts.
956  if (Attr->hasCoverage())
957  NoSanitizeCoverage = true;
958  }
959 
960  // Apply the no_sanitize* attributes to SanOpts.
961  SanOpts.Mask &= ~no_sanitize_mask;
962  if (no_sanitize_mask & SanitizerKind::Address)
963  SanOpts.set(SanitizerKind::KernelAddress, false);
964  if (no_sanitize_mask & SanitizerKind::KernelAddress)
965  SanOpts.set(SanitizerKind::Address, false);
966  if (no_sanitize_mask & SanitizerKind::HWAddress)
967  SanOpts.set(SanitizerKind::KernelHWAddress, false);
968  if (no_sanitize_mask & SanitizerKind::KernelHWAddress)
969  SanOpts.set(SanitizerKind::HWAddress, false);
970 
971  if (SanitizeBounds && !SanOpts.hasOneOf(SanitizerKind::Bounds))
972  Fn->addFnAttr(llvm::Attribute::NoSanitizeBounds);
973 
974  if (NoSanitizeCoverage && CGM.getCodeGenOpts().hasSanitizeCoverage())
975  Fn->addFnAttr(llvm::Attribute::NoSanitizeCoverage);
976 
977  // Some passes need the non-negated no_sanitize attribute. Pass them on.
978  if (CGM.getCodeGenOpts().hasSanitizeBinaryMetadata()) {
979  if (no_sanitize_mask & SanitizerKind::Thread)
980  Fn->addFnAttr("no_sanitize_thread");
981  }
982  }
983 
984  if (ShouldSkipSanitizerInstrumentation()) {
985  CurFn->addFnAttr(llvm::Attribute::DisableSanitizerInstrumentation);
986  } else {
987  // Apply sanitizer attributes to the function.
988  if (SanOpts.hasOneOf(SanitizerKind::Address | SanitizerKind::KernelAddress))
989  Fn->addFnAttr(llvm::Attribute::SanitizeAddress);
990  if (SanOpts.hasOneOf(SanitizerKind::HWAddress |
991  SanitizerKind::KernelHWAddress))
992  Fn->addFnAttr(llvm::Attribute::SanitizeHWAddress);
993  if (SanOpts.has(SanitizerKind::MemtagStack))
994  Fn->addFnAttr(llvm::Attribute::SanitizeMemTag);
995  if (SanOpts.has(SanitizerKind::Thread))
996  Fn->addFnAttr(llvm::Attribute::SanitizeThread);
997  if (SanOpts.hasOneOf(SanitizerKind::Memory | SanitizerKind::KernelMemory))
998  Fn->addFnAttr(llvm::Attribute::SanitizeMemory);
999  }
1000  if (SanOpts.has(SanitizerKind::SafeStack))
1001  Fn->addFnAttr(llvm::Attribute::SafeStack);
1002  if (SanOpts.has(SanitizerKind::ShadowCallStack))
1003  Fn->addFnAttr(llvm::Attribute::ShadowCallStack);
1004 
1005  // Apply fuzzing attribute to the function.
1006  if (SanOpts.hasOneOf(SanitizerKind::Fuzzer | SanitizerKind::FuzzerNoLink))
1007  Fn->addFnAttr(llvm::Attribute::OptForFuzzing);
1008 
1009  // Ignore TSan memory acesses from within ObjC/ObjC++ dealloc, initialize,
1010  // .cxx_destruct, __destroy_helper_block_ and all of their calees at run time.
1011  if (SanOpts.has(SanitizerKind::Thread)) {
1012  if (const auto *OMD = dyn_cast_or_null<ObjCMethodDecl>(D)) {
1013  const IdentifierInfo *II = OMD->getSelector().getIdentifierInfoForSlot(0);
1014  if (OMD->getMethodFamily() == OMF_dealloc ||
1015  OMD->getMethodFamily() == OMF_initialize ||
1016  (OMD->getSelector().isUnarySelector() && II->isStr(".cxx_destruct"))) {
1017  markAsIgnoreThreadCheckingAtRuntime(Fn);
1018  }
1019  }
1020  }
1021 
1022  // Ignore unrelated casts in STL allocate() since the allocator must cast
1023  // from void* to T* before object initialization completes. Don't match on the
1024  // namespace because not all allocators are in std::
1025  if (D && SanOpts.has(SanitizerKind::CFIUnrelatedCast)) {
1026  if (matchesStlAllocatorFn(D, getContext()))
1027  SanOpts.Mask &= ~SanitizerKind::CFIUnrelatedCast;
1028  }
1029 
1030  // Ignore null checks in coroutine functions since the coroutines passes
1031  // are not aware of how to move the extra UBSan instructions across the split
1032  // coroutine boundaries.
1033  if (D && SanOpts.has(SanitizerKind::Null))
1034  if (FD && FD->getBody() &&
1035  FD->getBody()->getStmtClass() == Stmt::CoroutineBodyStmtClass)
1036  SanOpts.Mask &= ~SanitizerKind::Null;
1037 
1038  // Apply xray attributes to the function (as a string, for now)
1039  bool AlwaysXRayAttr = false;
1040  if (const auto *XRayAttr = D ? D->getAttr<XRayInstrumentAttr>() : nullptr) {
1041  if (CGM.getCodeGenOpts().XRayInstrumentationBundle.has(
1042  XRayInstrKind::FunctionEntry) ||
1043  CGM.getCodeGenOpts().XRayInstrumentationBundle.has(
1044  XRayInstrKind::FunctionExit)) {
1045  if (XRayAttr->alwaysXRayInstrument() && ShouldXRayInstrumentFunction()) {
1046  Fn->addFnAttr("function-instrument", "xray-always");
1047  AlwaysXRayAttr = true;
1048  }
1049  if (XRayAttr->neverXRayInstrument())
1050  Fn->addFnAttr("function-instrument", "xray-never");
1051  if (const auto *LogArgs = D->getAttr<XRayLogArgsAttr>())
1052  if (ShouldXRayInstrumentFunction())
1053  Fn->addFnAttr("xray-log-args",
1054  llvm::utostr(LogArgs->getArgumentCount()));
1055  }
1056  } else {
1057  if (ShouldXRayInstrumentFunction() && !CGM.imbueXRayAttrs(Fn, Loc))
1058  Fn->addFnAttr(
1059  "xray-instruction-threshold",
1060  llvm::itostr(CGM.getCodeGenOpts().XRayInstructionThreshold));
1061  }
1062 
1063  if (ShouldXRayInstrumentFunction()) {
1064  if (CGM.getCodeGenOpts().XRayIgnoreLoops)
1065  Fn->addFnAttr("xray-ignore-loops");
1066 
1067  if (!CGM.getCodeGenOpts().XRayInstrumentationBundle.has(
1068  XRayInstrKind::FunctionExit))
1069  Fn->addFnAttr("xray-skip-exit");
1070 
1071  if (!CGM.getCodeGenOpts().XRayInstrumentationBundle.has(
1072  XRayInstrKind::FunctionEntry))
1073  Fn->addFnAttr("xray-skip-entry");
1074 
1075  auto FuncGroups = CGM.getCodeGenOpts().XRayTotalFunctionGroups;
1076  if (FuncGroups > 1) {
1077  auto FuncName = llvm::ArrayRef<uint8_t>(CurFn->getName().bytes_begin(),
1078  CurFn->getName().bytes_end());
1079  auto Group = crc32(FuncName) % FuncGroups;
1080  if (Group != CGM.getCodeGenOpts().XRaySelectedFunctionGroup &&
1081  !AlwaysXRayAttr)
1082  Fn->addFnAttr("function-instrument", "xray-never");
1083  }
1084  }
1085

1609  const CGFunctionInfo &FnInfo) {
1610  assert(Fn && "generating code for null Function");
1611  const FunctionDecl *FD = cast<FunctionDecl>(GD.getDecl());
1612  CurGD = GD;
1613 
1614  FunctionArgList Args;
1615  QualType ResTy = BuildFunctionArgList(GD, Args);
1616 
1617  CGM.getTargetCodeGenInfo().checkFunctionABI(CGM, FD);
1618 
1619  if (FD->isInlineBuiltinDeclaration()) {
1620  // When generating code for a builtin with an inline declaration, use a
1621  // mangled name to hold the actual body, while keeping an external
1622  // definition in case the function pointer is referenced somewhere.
1623  std::string FDInlineName = (Fn->getName() + ".inline").str();
1624  llvm::Module *M = Fn->getParent();
1625  llvm::Function *Clone = M->getFunction(FDInlineName);
1626  if (!Clone) {
1627  Clone = llvm::Function::Create(Fn->getFunctionType(),
1628  llvm::GlobalValue::InternalLinkage,
1629  Fn->getAddressSpace(), FDInlineName, M);
1630  Clone->addFnAttr(llvm::Attribute::AlwaysInline);
1631  }
1632  Fn->setLinkage(llvm::GlobalValue::ExternalLinkage);
1633  Fn = Clone;
1634  } else {
1635  // Detect the unusual situation where an inline version is shadowed by a
1636  // non-inline version. In that case we should pick the external one
1637  // everywhere. That's GCC behavior too. Unfortunately, I cannot find a way
1638  // to detect that situation before we reach codegen, so do some late
1639  // replacement.
1640  for (const FunctionDecl *PD = FD->getPreviousDecl(); PD;
1641  PD = PD->getPreviousDecl()) {
1642  if (LLVM_UNLIKELY(PD->isInlineBuiltinDeclaration())) {
1643  std::string FDInlineName = (Fn->getName() + ".inline").str();
1644  llvm::Module *M = Fn->getParent();
1645  if (llvm::Function *Clone = M->getFunction(FDInlineName)) {
1646  Clone->replaceAllUsesWith(Fn);
1647  Clone->eraseFromParent();
1648  }
1649  break;
1650  }
1651  }
1652  }
1653 
1654  // Check if we should generate debug info for this function.
1655  if (FD->hasAttr<NoDebugAttr>() || noSystemDebugInfo(FD, CGM)) {
1656  // Clear non-distinct debug info that was possibly attached to the function
1657  // due to an earlier declaration without the nodebug attribute
1658  Fn->setSubprogram(nullptr);
1659  // Disable debug info indefinitely for this function
1660  DebugInfo = nullptr;
1661  }
1662 
1663  // The function might not have a body if we're generating thunks for a
1664  // function declaration.
1665  SourceRange BodyRange;
1666  if (Stmt *Body = FD->getBody())
1667  BodyRange = Body->getSourceRange();
1668  else
1669  BodyRange = FD->getLocation();
1670  CurEHLocation = BodyRange.getEnd();
1671 
1672  // Use the location of the start of the function to determine where
1673  // the function definition is located. By default use the location
1674  // of the declaration as the location for the subprogram. A function
1675  // may lack a declaration in the source code if it is created by code
1676  // gen. (examples: _GLOBAL__I_a, __cxx_global_array_dtor, thunk).
1677  SourceLocation Loc = FD->getLocation();
1678 
1679  // If this is a function specialization then use the pattern body
1680  // as the location for the function.
1681  if (const FunctionDecl *SpecDecl = FD->getTemplateInstantiationPattern())
1682  if (SpecDecl->hasBody(SpecDecl))
1683  Loc = SpecDecl->getLocation();
1684 
1685  Stmt *Body = FD->getBody();
1686 
1687  if (Body) {
1688  // Coroutines always emit lifetime markers.
1689  if (isa<CoroutineBodyStmt>(Body))
1690  ShouldEmitLifetimeMarkers = true;
1691 
1692  // Initialize helper which will detect jumps which can cause invalid
1693  // lifetime markers.
1694  if (ShouldEmitLifetimeMarkers)
1695  Bypasses.Init(Body);
1696  }
1697 
1698  // Emit the standard function prologue.
1699  StartFunction(GD, ResTy, Fn, FnInfo, Args, Loc, BodyRange.getBegin());
1700  if (!getLangOpts().OptRecordFile.empty()) {
1701  SyclOptReportHandler &SyclOptReport = CGM.getDiags().getSYCLOptReport();
1702  if (SyclOptReport.HasOptReportInfo(FD)) {
1703  llvm::OptimizationRemarkEmitter ORE(Fn);
1704  for (auto ORI : llvm::enumerate(SyclOptReport.GetInfo(FD))) {
1705  llvm::DiagnosticLocation DL =
1706  SourceLocToDebugLoc(ORI.value().KernelArgLoc);
1707  StringRef NameInDesc = ORI.value().KernelArgDescName;
1708  StringRef ArgType = ORI.value().KernelArgType;
1709  StringRef ArgDesc = ORI.value().KernelArgDesc;
1710  unsigned ArgSize = ORI.value().KernelArgSize;
1711  StringRef ArgDecomposedField = ORI.value().KernelArgDecomposedField;
1712 
1713  llvm::OptimizationRemark Remark("sycl", "Region", DL,
1714  &Fn->getEntryBlock());
1715  Remark << "Arg " << llvm::ore::NV("Argument", ORI.index()) << ":"
1716  << ArgDesc << NameInDesc << " (" << ArgDecomposedField
1717  << "Type:" << ArgType << ", "
1718  << "Size: " << llvm::ore::NV("Argument", ArgSize) << ")";
1719  ORE.emit(Remark);
1720  }
1721  }
1722  }
1723 
1724  // Save parameters for coroutine function.
1725  if (Body && isa_and_nonnull<CoroutineBodyStmt>(Body))
1726  llvm::append_range(FnArgs, FD->parameters());
1727 
1728  // Generate a dummy __host__ function for compiling CUDA sources in SYCL.
1729  if (getLangOpts().CUDA && !getLangOpts().CUDAIsDevice &&
1730  getLangOpts().SYCLIsHost && !FD->hasAttr<CUDAHostAttr>() &&
1731  FD->hasAttr<CUDADeviceAttr>()) {
1732  if (FD->getReturnType()->isVoidType())
1733  Builder.CreateRetVoid();
1734  else
1735  Builder.CreateRet(llvm::UndefValue::get(Fn->getReturnType()));
1736  return;
1737  }
1738  // When compiling a CUDA file in SYCL device mode,
1739  // set weak ODR linkage for possibly duplicated functions.
1740  if (getLangOpts().CUDA && !getLangOpts().CUDAIsDevice &&
1741  getLangOpts().SYCLIsDevice &&
1742  (FD->hasAttr<CUDADeviceAttr>() || FD->hasAttr<CUDAHostAttr>()))
1743  Fn->setLinkage(llvm::Function::WeakODRLinkage);
1744 
1745  // Ensure that the function adheres to the forward progress guarantee, which
1746  // is required by certain optimizations.
1747  if (checkIfFunctionMustProgress())
1748  CurFn->addFnAttr(llvm::Attribute::MustProgress);
1749 
1750  // Generate the body of the function.
1751  PGO.assignRegionCounters(GD, CurFn);
1752  if (isa<CXXDestructorDecl>(FD))
1753  EmitDestructorBody(Args);
1754  else if (isa<CXXConstructorDecl>(FD))
1755  EmitConstructorBody(Args);
1756  else if (getLangOpts().CUDA &&
1757  !getLangOpts().CUDAIsDevice &&
1758  FD->hasAttr<CUDAGlobalAttr>())
1759  CGM.getCUDARuntime().emitDeviceStub(*this, Args);
1760  else if (isa<CXXMethodDecl>(FD) &&
1761  cast<CXXMethodDecl>(FD)->isLambdaStaticInvoker()) {
1762  // The lambda static invoker function is special, because it forwards or
1763  // clones the body of the function call operator (but is actually static).
1764  EmitLambdaStaticInvokeBody(cast<CXXMethodDecl>(FD));
1765  } else if (isa<CXXMethodDecl>(FD) &&
1766  isLambdaCallOperator(cast<CXXMethodDecl>(FD)) &&
1767  !FnInfo.isDelegateCall() &&
1768  cast<CXXMethodDecl>(FD)->getParent()->getLambdaStaticInvoker() &&
1769  hasInAllocaArg(cast<CXXMethodDecl>(FD))) {
1770  // If emitting a lambda with static invoker on X86 Windows, change
1771  // the call operator body.
1772  // Make sure that this is a call operator with an inalloca arg and check
1773  // for delegate call to make sure this is the original call op and not the
1774  // new forwarding function for the static invoker.
1775  EmitLambdaInAllocaCallOpBody(cast<CXXMethodDecl>(FD));
1776  } else if (FD->isDefaulted() && isa<CXXMethodDecl>(FD) &&
1777  (cast<CXXMethodDecl>(FD)->isCopyAssignmentOperator() ||
1778  cast<CXXMethodDecl>(FD)->isMoveAssignmentOperator())) {
1779  // Implicit copy-assignment gets the same special treatment as implicit
1780  // copy-constructors.
1781  emitImplicitAssignmentOperatorBody(Args);
1782  } else if (Body) {
1783  EmitFunctionBody(Body);
1784  } else
1785  llvm_unreachable("no definition for emitted function");
1786 
1787  // C++11 [stmt.return]p2:
1788  // Flowing off the end of a function [...] results in undefined behavior in
1789  // a value-returning function.
1790  // C11 6.9.1p12:
1791  // If the '}' that terminates a function is reached, and the value of the
1792  // function call is used by the caller, the behavior is undefined.
1793  if (getLangOpts().CPlusPlus && !FD->hasImplicitReturnZero() && !SawAsmBlock &&
1794  !FD->getReturnType()->isVoidType() && Builder.GetInsertBlock()) {
1795  bool ShouldEmitUnreachable =
1796  CGM.getCodeGenOpts().StrictReturn ||
1797  !CGM.MayDropFunctionReturn(FD->getASTContext(), FD->getReturnType());
1798  if (SanOpts.has(SanitizerKind::Return)) {
1799  SanitizerScope SanScope(this);
1800  llvm::Value *IsFalse = Builder.getFalse();
1801  EmitCheck(std::make_pair(IsFalse, SanitizerKind::Return),
1802  SanitizerHandler::MissingReturn,
1803  EmitCheckSourceLocation(FD->getLocation()), std::nullopt);
1804  } else if (ShouldEmitUnreachable) {
1805  if (CGM.getCodeGenOpts().OptimizationLevel == 0)
1806  EmitTrapCall(llvm::Intrinsic::trap);
1807  }
1808  if (SanOpts.has(SanitizerKind::Return) || ShouldEmitUnreachable) {
1809  Builder.CreateUnreachable();
1810  Builder.ClearInsertionPoint();
1811  }
1812  }
1813 
1814  // Emit the standard function epilogue.
1815  FinishFunction(BodyRange.getEnd());
1816 
1817  // If we haven't marked the function nothrow through other means, do
1818  // a quick pass now to see if we can.
1819  if (!CurFn->doesNotThrow())
1820  TryMarkNoThrow(CurFn);
1821 }

1823 /// ContainsLabel - Return true if the statement contains a label in it. If
1824 /// this statement is not executed normally, it not containing a label means
1825 /// that we can just remove the code.
1826 bool CodeGenFunction::ContainsLabel(const Stmt *S, bool IgnoreCaseStmts) {
1827  // Null statement, not a label!
1828  if (!S) return false;
1829 
1830  // If this is a label, we have to emit the code, consider something like:
1831  // if (0) { ... foo: bar(); } goto foo;
1832  //
1833  // TODO: If anyone cared, we could track __label__'s, since we know that you
1834  // can't jump to one from outside their declared region.
1835  if (isa<LabelStmt>(S))
1836  return true;
1837 
1838  // If this is a case/default statement, and we haven't seen a switch, we have
1839  // to emit the code.
1840  if (isa<SwitchCase>(S) && !IgnoreCaseStmts)
1841  return true;
1842 
1843  // If this is a switch statement, we want to ignore cases below it.
1844  if (isa<SwitchStmt>(S))
1845  IgnoreCaseStmts = true;
1846 
1847  // Scan subexpressions for verboten labels.
1848  for (const Stmt *SubStmt : S->children())
1849  if (ContainsLabel(SubStmt, IgnoreCaseStmts))
1850  return true;
1851 
1852  return false;
1853 }

1855 /// containsBreak - Return true if the statement contains a break out of it.
1856 /// If the statement (recursively) contains a switch or loop with a break
1857 /// inside of it, this is fine.
1858 bool CodeGenFunction::containsBreak(const Stmt *S) {
1859  // Null statement, not a label!
1860  if (!S) return false;
1861 
1862  // If this is a switch or loop that defines its own break scope, then we can
1863  // include it and anything inside of it.
1864  if (isa<SwitchStmt>(S) || isa<WhileStmt>(S) || isa<DoStmt>(S) ||

1866  return false;
1867

1869  return true;
1870 
1871  // Scan subexpressions for verboten breaks.
1872  for (const Stmt *SubStmt : S->children())
1873  if (containsBreak(SubStmt))
1874  return true;
1875 
1876  return false;
1877 }

1960 /// EmitBranchToCounterBlock - Emit a conditional branch to a new block that
1961 /// increments a profile counter based on the semantics of the given logical
1962 /// operator opcode. This is used to instrument branch condition coverage for
1963 /// logical operators.
1964 void CodeGenFunction::EmitBranchToCounterBlock(
1965  const Expr *Cond, BinaryOperator::Opcode LOp, llvm::BasicBlock *TrueBlock,
1966  llvm::BasicBlock *FalseBlock, uint64_t TrueCount /* = 0 */,
1967  Stmt::Likelihood LH /* =None */, const Expr *CntrIdx /* = nullptr */) {
1968  // If not instrumenting, just emit a branch.
1969  bool InstrumentRegions = CGM.getCodeGenOpts().hasProfileClangInstr();
1970  if (!InstrumentRegions || !isInstrumentedCondition(Cond))
1971  return EmitBranchOnBoolExpr(Cond, TrueBlock, FalseBlock, TrueCount, LH);
1972 
1973  llvm::BasicBlock *ThenBlock = nullptr;
1974  llvm::BasicBlock *ElseBlock = nullptr;
1975  llvm::BasicBlock *NextBlock = nullptr;
1976 
1977  // Create the block we'll use to increment the appropriate counter.
1978  llvm::BasicBlock *CounterIncrBlock = createBasicBlock("lop.rhscnt");
1979 
1980  // Set block pointers according to Logical-AND (BO_LAnd) semantics. This
1981  // means we need to evaluate the condition and increment the counter on TRUE:
1982  //
1983  // if (Cond)
1984  // goto CounterIncrBlock;
1985  // else
1986  // goto FalseBlock;
1987  //
1988  // CounterIncrBlock:
1989  // Counter++;
1990  // goto TrueBlock;
1991 
1992  if (LOp == BO_LAnd) {
1993  ThenBlock = CounterIncrBlock;
1994  ElseBlock = FalseBlock;
1995  NextBlock = TrueBlock;
1996  }
1997 
1998  // Set block pointers according to Logical-OR (BO_LOr) semantics. This means
1999  // we need to evaluate the condition and increment the counter on FALSE:
2000  //
2001  // if (Cond)
2002  // goto TrueBlock;
2003  // else
2004  // goto CounterIncrBlock;
2005  //
2006  // CounterIncrBlock:
2007  // Counter++;
2008  // goto FalseBlock;
2009 
2010  else if (LOp == BO_LOr) {
2011  ThenBlock = TrueBlock;
2012  ElseBlock = CounterIncrBlock;
2013  NextBlock = FalseBlock;
2014  } else {
2015  llvm_unreachable("Expected Opcode must be that of a Logical Operator");
2016  }
2017 
2018  // Emit Branch based on condition.
2019  EmitBranchOnBoolExpr(Cond, ThenBlock, ElseBlock, TrueCount, LH);
2020 
2021  // Emit the block containing the counter increment(s).
2022  EmitBlock(CounterIncrBlock);
2023 
2024  // Increment corresponding counter; if index not provided, use Cond as index.
2025  incrementProfileCounter(CntrIdx ? CntrIdx : Cond);
2026 
2027  // Go to the next block.
2028  EmitBranch(NextBlock);
2029 }

2031 /// EmitBranchOnBoolExpr - Emit a branch on a boolean condition (e.g. for an if
2032 /// statement) to the specified blocks. Based on the condition, this might try
2033 /// to simplify the codegen of the conditional based on the branch.
2034 /// \param LH The value of the likelihood attribute on the True branch.
2035 /// \param ConditionalOp Used by MC/DC code coverage to track the result of the
2036 /// ConditionalOperator (ternary) through a recursive call for the operator's
2037 /// LHS and RHS nodes.
2038 void CodeGenFunction::EmitBranchOnBoolExpr(
2039  const Expr *Cond, llvm::BasicBlock *TrueBlock, llvm::BasicBlock *FalseBlock,
2040  uint64_t TrueCount, Stmt::Likelihood LH, const Expr *ConditionalOp) {
2041  Cond = Cond->IgnoreParens();
2042 
2043  if (const BinaryOperator *CondBOp = dyn_cast<BinaryOperator>(Cond)) {
2044  // Handle X && Y in a condition.
2045  if (CondBOp->getOpcode() == BO_LAnd) {
2046  MCDCLogOpStack.push_back(CondBOp);
2047 
2048  // If we have "1 && X", simplify the code. "0 && X" would have constant
2049  // folded if the case was simple enough.
2050  bool ConstantBool = false;
2051  if (ConstantFoldsToSimpleInteger(CondBOp->getLHS(), ConstantBool) &&
2052  ConstantBool) {
2053  // br(1 && X) -> br(X).
2054  incrementProfileCounter(CondBOp);
2055  EmitBranchToCounterBlock(CondBOp->getRHS(), BO_LAnd, TrueBlock,
2056  FalseBlock, TrueCount, LH);
2057  MCDCLogOpStack.pop_back();
2058  return;
2059  }
2060 
2061  // If we have "X && 1", simplify the code to use an uncond branch.
2062  // "X && 0" would have been constant folded to 0.
2063  if (ConstantFoldsToSimpleInteger(CondBOp->getRHS(), ConstantBool) &&
2064  ConstantBool) {
2065  // br(X && 1) -> br(X).
2066  EmitBranchToCounterBlock(CondBOp->getLHS(), BO_LAnd, TrueBlock,
2067  FalseBlock, TrueCount, LH, CondBOp);
2068  MCDCLogOpStack.pop_back();
2069  return;
2070  }
2071 
2072  // Emit the LHS as a conditional. If the LHS conditional is false, we
2073  // want to jump to the FalseBlock.
2074  llvm::BasicBlock *LHSTrue = createBasicBlock("land.lhs.true");
2075  // The counter tells us how often we evaluate RHS, and all of TrueCount
2076  // can be propagated to that branch.
2077  uint64_t RHSCount = getProfileCount(CondBOp->getRHS());
2078 
2079  ConditionalEvaluation eval(*this);
2080  {
2081  ApplyDebugLocation DL(*this, Cond);
2082  // Propagate the likelihood attribute like __builtin_expect
2083  // __builtin_expect(X && Y, 1) -> X and Y are likely
2084  // __builtin_expect(X && Y, 0) -> only Y is unlikely
2085  EmitBranchOnBoolExpr(CondBOp->getLHS(), LHSTrue, FalseBlock, RHSCount,
2086  LH == Stmt::LH_Unlikely ? Stmt::LH_None : LH);
2087  EmitBlock(LHSTrue);
2088  }
2089 
2090  incrementProfileCounter(CondBOp);
2091  setCurrentProfileCount(getProfileCount(CondBOp->getRHS()));
2092 
2093  // Any temporaries created here are conditional.
2094  eval.begin(*this);
2095  EmitBranchToCounterBlock(CondBOp->getRHS(), BO_LAnd, TrueBlock,
2096  FalseBlock, TrueCount, LH);
2097  eval.end(*this);
2098  MCDCLogOpStack.pop_back();
2099  return;
2100  }
2101 
2102  if (CondBOp->getOpcode() == BO_LOr) {
2103  MCDCLogOpStack.push_back(CondBOp);
2104 
2105  // If we have "0 || X", simplify the code. "1 || X" would have constant
2106  // folded if the case was simple enough.
2107  bool ConstantBool = false;
2108  if (ConstantFoldsToSimpleInteger(CondBOp->getLHS(), ConstantBool) &&
2109  !ConstantBool) {
2110  // br(0 || X) -> br(X).
2111  incrementProfileCounter(CondBOp);
2112  EmitBranchToCounterBlock(CondBOp->getRHS(), BO_LOr, TrueBlock,
2113  FalseBlock, TrueCount, LH);
2114  MCDCLogOpStack.pop_back();
2115  return;
2116  }
2117 
2118  // If we have "X || 0", simplify the code to use an uncond branch.
2119  // "X || 1" would have been constant folded to 1.
2120  if (ConstantFoldsToSimpleInteger(CondBOp->getRHS(), ConstantBool) &&
2121  !ConstantBool) {
2122  // br(X || 0) -> br(X).
2123  EmitBranchToCounterBlock(CondBOp->getLHS(), BO_LOr, TrueBlock,
2124  FalseBlock, TrueCount, LH, CondBOp);
2125  MCDCLogOpStack.pop_back();
2126  return;
2127  }
2128  // Emit the LHS as a conditional. If the LHS conditional is true, we
2129  // want to jump to the TrueBlock.
2130  llvm::BasicBlock *LHSFalse = createBasicBlock("lor.lhs.false");
2131  // We have the count for entry to the RHS and for the whole expression
2132  // being true, so we can divy up True count between the short circuit and
2133  // the RHS.
2134  uint64_t LHSCount =
2135  getCurrentProfileCount() - getProfileCount(CondBOp->getRHS());
2136  uint64_t RHSCount = TrueCount - LHSCount;
2137 
2138  ConditionalEvaluation eval(*this);
2139  {
2140  // Propagate the likelihood attribute like __builtin_expect
2141  // __builtin_expect(X || Y, 1) -> only Y is likely
2142  // __builtin_expect(X || Y, 0) -> both X and Y are unlikely
2143  ApplyDebugLocation DL(*this, Cond);
2144  EmitBranchOnBoolExpr(CondBOp->getLHS(), TrueBlock, LHSFalse, LHSCount,
2145  LH == Stmt::LH_Likely ? Stmt::LH_None : LH);
2146  EmitBlock(LHSFalse);
2147  }
2148 
2149  incrementProfileCounter(CondBOp);
2150  setCurrentProfileCount(getProfileCount(CondBOp->getRHS()));
2151 
2152  // Any temporaries created here are conditional.
2153  eval.begin(*this);
2154  EmitBranchToCounterBlock(CondBOp->getRHS(), BO_LOr, TrueBlock, FalseBlock,
2155  RHSCount, LH);
2156 
2157  eval.end(*this);
