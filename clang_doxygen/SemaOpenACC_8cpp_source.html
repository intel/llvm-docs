25 bool diagnoseConstructAppertainment(SemaOpenACC &S, OpenACCDirectiveKind K,
26  SourceLocation StartLoc, bool IsStmt) {
27  switch (K) {
28  default:
29  case OpenACCDirectiveKind::Invalid:
30  // Nothing to do here, both invalid and unimplemented don't really need to
31  // do anything.
32  break;
33  case OpenACCDirectiveKind::Parallel:
34  case OpenACCDirectiveKind::Serial:
35  case OpenACCDirectiveKind::Kernels:
36  if (!IsStmt)
37  return S.Diag(StartLoc, diag::err_acc_construct_appertainment) << K;
38  break;
39  }
40  return false;
41 }

256 bool checkAlreadyHasClauseOfKind(
257  SemaOpenACC &S, ArrayRef<const OpenACCClause *> ExistingClauses,
258  SemaOpenACC::OpenACCParsedClause &Clause) {
259  const auto *Itr = llvm::find_if(ExistingClauses, [&](const OpenACCClause *C) {
260  return C->getClauseKind() == Clause.getClauseKind();
261  });
262  if (Itr != ExistingClauses.end()) {
263  S.Diag(Clause.getBeginLoc(), diag::err_acc_duplicate_clause_disallowed)
264  << Clause.getDirectiveKind() << Clause.getClauseKind();
265  S.Diag((*Itr)->getBeginLoc(), diag::note_acc_previous_clause_here);
266  return true;
267  }
268  return false;
269 }

271 /// Implement check from OpenACC3.3: section 2.5.4:
272 /// Only the async, wait, num_gangs, num_workers, and vector_length clauses may
273 /// follow a device_type clause.
274 bool checkValidAfterDeviceType(
275  SemaOpenACC &S, const OpenACCDeviceTypeClause &DeviceTypeClause,
276  const SemaOpenACC::OpenACCParsedClause &NewClause) {
277  // This is only a requirement on compute constructs so far, so this is fine
278  // otherwise.
279  if (!isOpenACCComputeDirectiveKind(NewClause.getDirectiveKind()))
280  return false;
281  switch (NewClause.getClauseKind()) {
282  case OpenACCClauseKind::Async:
283  case OpenACCClauseKind::Wait:
284  case OpenACCClauseKind::NumGangs:
285  case OpenACCClauseKind::NumWorkers:
286  case OpenACCClauseKind::VectorLength:
287  case OpenACCClauseKind::DType:
288  case OpenACCClauseKind::DeviceType:
289  return false;
290  default:
291  S.Diag(NewClause.getBeginLoc(), diag::err_acc_clause_after_device_type)
292  << NewClause.getClauseKind() << DeviceTypeClause.getClauseKind();
293  S.Diag(DeviceTypeClause.getBeginLoc(), diag::note_acc_previous_clause_here);
294  return true;
295  }

301 OpenACCClause *
302 SemaOpenACC::ActOnClause(ArrayRef<const OpenACCClause *> ExistingClauses,
303  OpenACCParsedClause &Clause) {
304  if (Clause.getClauseKind() == OpenACCClauseKind::Invalid)
305  return nullptr;
306 
307  // Diagnose that we don't support this clause on this directive.
308  if (!doesClauseApplyToDirective(Clause.getDirectiveKind(),
309  Clause.getClauseKind())) {
310  Diag(Clause.getBeginLoc(), diag::err_acc_clause_appertainment)
311  << Clause.getDirectiveKind() << Clause.getClauseKind();
312  return nullptr;
313  }
314 
315  if (const auto *DevTypeClause =
316  llvm::find_if(ExistingClauses,
317  [&](const OpenACCClause *C) {
318  return isa<OpenACCDeviceTypeClause>(C);
319  });
320  DevTypeClause != ExistingClauses.end()) {
321  if (checkValidAfterDeviceType(
322  *this, *cast<OpenACCDeviceTypeClause>(*DevTypeClause), Clause))
323  return nullptr;
324  }
325 
326  switch (Clause.getClauseKind()) {
327  case OpenACCClauseKind::Default: {
328  // Restrictions only properly implemented on 'compute' constructs, and
329  // 'compute' constructs are the only construct that can do anything with
330  // this yet, so skip/treat as unimplemented in this case.
331  if (!isOpenACCComputeDirectiveKind(Clause.getDirectiveKind()))
332  break;
333 
334  // Don't add an invalid clause to the AST.
335  if (Clause.getDefaultClauseKind() == OpenACCDefaultClauseKind::Invalid)
336  return nullptr;
337 
338  // OpenACC 3.3, Section 2.5.4:
339  // At most one 'default' clause may appear, and it must have a value of
340  // either 'none' or 'present'.
341  // Second half of the sentence is diagnosed during parsing.
342  if (checkAlreadyHasClauseOfKind(*this, ExistingClauses, Clause))
343  return nullptr;
344 
345  return OpenACCDefaultClause::Create(
346  getASTContext(), Clause.getDefaultClauseKind(), Clause.getBeginLoc(),
347  Clause.getLParenLoc(), Clause.getEndLoc());
348  }
349 
350  case OpenACCClauseKind::If: {
351  // Restrictions only properly implemented on 'compute' constructs, and
352  // 'compute' constructs are the only construct that can do anything with
353  // this yet, so skip/treat as unimplemented in this case.
354  if (!isOpenACCComputeDirectiveKind(Clause.getDirectiveKind()))
355  break;
356 
357  // There is no prose in the standard that says duplicates aren't allowed,
358  // but this diagnostic is present in other compilers, as well as makes
359  // sense.
360  if (checkAlreadyHasClauseOfKind(*this, ExistingClauses, Clause))
361  return nullptr;
362 
363  // The parser has ensured that we have a proper condition expr, so there
364  // isn't really much to do here.
365 
366  // If the 'if' clause is true, it makes the 'self' clause have no effect,
367  // diagnose that here.
368  // TODO OpenACC: When we add these two to other constructs, we might not
369  // want to warn on this (for example, 'update').
370  const auto *Itr =
371  llvm::find_if(ExistingClauses, llvm::IsaPred<OpenACCSelfClause>);
372  if (Itr != ExistingClauses.end()) {
373  Diag(Clause.getBeginLoc(), diag::warn_acc_if_self_conflict);
374  Diag((*Itr)->getBeginLoc(), diag::note_acc_previous_clause_here);
375  }
376 
377  return OpenACCIfClause::Create(
378  getASTContext(), Clause.getBeginLoc(), Clause.getLParenLoc(),
379  Clause.getConditionExpr(), Clause.getEndLoc());
380  }
381 
382  case OpenACCClauseKind::Self: {
383  // Restrictions only properly implemented on 'compute' constructs, and
384  // 'compute' constructs are the only construct that can do anything with
385  // this yet, so skip/treat as unimplemented in this case.
386  if (!isOpenACCComputeDirectiveKind(Clause.getDirectiveKind()))
387  break;
388 
389  // TODO OpenACC: When we implement this for 'update', this takes a
390  // 'var-list' instead of a condition expression, so semantics/handling has
391  // to happen differently here.
392 
393  // There is no prose in the standard that says duplicates aren't allowed,
394  // but this diagnostic is present in other compilers, as well as makes
395  // sense.
396  if (checkAlreadyHasClauseOfKind(*this, ExistingClauses, Clause))
397  return nullptr;
398 
399  // If the 'if' clause is true, it makes the 'self' clause have no effect,
400  // diagnose that here.
401  // TODO OpenACC: When we add these two to other constructs, we might not
402  // want to warn on this (for example, 'update').
403  const auto *Itr =
404  llvm::find_if(ExistingClauses, llvm::IsaPred<OpenACCIfClause>);
405  if (Itr != ExistingClauses.end()) {
406  Diag(Clause.getBeginLoc(), diag::warn_acc_if_self_conflict);
407  Diag((*Itr)->getBeginLoc(), diag::note_acc_previous_clause_here);
408  }
409 
410  return OpenACCSelfClause::Create(
411  getASTContext(), Clause.getBeginLoc(), Clause.getLParenLoc(),
412  Clause.getConditionExpr(), Clause.getEndLoc());
413  }
414  case OpenACCClauseKind::NumGangs: {
415  // Restrictions only properly implemented on 'compute' constructs, and
416  // 'compute' constructs are the only construct that can do anything with
417  // this yet, so skip/treat as unimplemented in this case.
418  if (!isOpenACCComputeDirectiveKind(Clause.getDirectiveKind()))
419  break;
420 
421  // There is no prose in the standard that says duplicates aren't allowed,
422  // but this diagnostic is present in other compilers, as well as makes
423  // sense.
424  if (checkAlreadyHasClauseOfKind(*this, ExistingClauses, Clause))
425  return nullptr;
426 
427  if (Clause.getIntExprs().empty())
428  Diag(Clause.getBeginLoc(), diag::err_acc_num_gangs_num_args)
429  << /*NoArgs=*/0;
430 
431  unsigned MaxArgs =
432  (Clause.getDirectiveKind() == OpenACCDirectiveKind::Parallel ||
433  Clause.getDirectiveKind() == OpenACCDirectiveKind::ParallelLoop)
434  ? 3
435  : 1;
436  if (Clause.getIntExprs().size() > MaxArgs)
437  Diag(Clause.getBeginLoc(), diag::err_acc_num_gangs_num_args)
438  << /*NoArgs=*/1 << Clause.getDirectiveKind() << MaxArgs
439  << Clause.getIntExprs().size();
440 
441  // OpenACC 3.3 Section 2.5.4:
442  // A reduction clause may not appear on a parallel construct with a
443  // num_gangs clause that has more than one argument.
444  if (Clause.getDirectiveKind() == OpenACCDirectiveKind::Parallel &&
445  Clause.getIntExprs().size() > 1) {
446  auto *Parallel =
447  llvm::find_if(ExistingClauses, llvm::IsaPred<OpenACCReductionClause>);
448 
449  if (Parallel != ExistingClauses.end()) {
450  Diag(Clause.getBeginLoc(), diag::err_acc_reduction_num_gangs_conflict)
451  << Clause.getIntExprs().size();
452  Diag((*Parallel)->getBeginLoc(), diag::note_acc_previous_clause_here);
453  return nullptr;
454  }
455  }
456 
457  // Create the AST node for the clause even if the number of expressions is
458  // incorrect.
459  return OpenACCNumGangsClause::Create(
460  getASTContext(), Clause.getBeginLoc(), Clause.getLParenLoc(),
461  Clause.getIntExprs(), Clause.getEndLoc());
462  break;
463  }
464  case OpenACCClauseKind::NumWorkers: {
465  // Restrictions only properly implemented on 'compute' constructs, and
466  // 'compute' constructs are the only construct that can do anything with
467  // this yet, so skip/treat as unimplemented in this case.
468  if (!isOpenACCComputeDirectiveKind(Clause.getDirectiveKind()))
469  break;
470 
471  // There is no prose in the standard that says duplicates aren't allowed,
472  // but this diagnostic is present in other compilers, as well as makes
473  // sense.
474  if (checkAlreadyHasClauseOfKind(*this, ExistingClauses, Clause))
475  return nullptr;
476 
477  assert(Clause.getIntExprs().size() == 1 &&
478  "Invalid number of expressions for NumWorkers");
479  return OpenACCNumWorkersClause::Create(
480  getASTContext(), Clause.getBeginLoc(), Clause.getLParenLoc(),
481  Clause.getIntExprs()[0], Clause.getEndLoc());
482  }
483  case OpenACCClauseKind::VectorLength: {
484  // Restrictions only properly implemented on 'compute' constructs, and
485  // 'compute' constructs are the only construct that can do anything with
486  // this yet, so skip/treat as unimplemented in this case.
487  if (!isOpenACCComputeDirectiveKind(Clause.getDirectiveKind()))
488  break;
489 
490  // There is no prose in the standard that says duplicates aren't allowed,
491  // but this diagnostic is present in other compilers, as well as makes
492  // sense.
493  if (checkAlreadyHasClauseOfKind(*this, ExistingClauses, Clause))
494  return nullptr;
495 
496  assert(Clause.getIntExprs().size() == 1 &&
497  "Invalid number of expressions for VectorLength");
498  return OpenACCVectorLengthClause::Create(
499  getASTContext(), Clause.getBeginLoc(), Clause.getLParenLoc(),
500  Clause.getIntExprs()[0], Clause.getEndLoc());
501  }
502  case OpenACCClauseKind::Async: {
503  // Restrictions only properly implemented on 'compute' constructs, and
504  // 'compute' constructs are the only construct that can do anything with
505  // this yet, so skip/treat as unimplemented in this case.
506  if (!isOpenACCComputeDirectiveKind(Clause.getDirectiveKind()))
507  break;
508 
509  // There is no prose in the standard that says duplicates aren't allowed,
510  // but this diagnostic is present in other compilers, as well as makes
511  // sense.
512  if (checkAlreadyHasClauseOfKind(*this, ExistingClauses, Clause))
513  return nullptr;
514 
515  assert(Clause.getNumIntExprs() < 2 &&
516  "Invalid number of expressions for Async");
517 
518  return OpenACCAsyncClause::Create(
519  getASTContext(), Clause.getBeginLoc(), Clause.getLParenLoc(),
520  Clause.getNumIntExprs() != 0 ? Clause.getIntExprs()[0] : nullptr,
521  Clause.getEndLoc());
522  }
523  case OpenACCClauseKind::Private: {
524  // Restrictions only properly implemented on 'compute' constructs, and
525  // 'compute' constructs are the only construct that can do anything with
526  // this yet, so skip/treat as unimplemented in this case.
527  if (!isOpenACCComputeDirectiveKind(Clause.getDirectiveKind()))
528  break;
529 
530  // ActOnVar ensured that everything is a valid variable reference, so there
531  // really isn't anything to do here. GCC does some duplicate-finding, though
532  // it isn't apparent in the standard where this is justified.
533 
534  return OpenACCPrivateClause::Create(
535  getASTContext(), Clause.getBeginLoc(), Clause.getLParenLoc(),
536  Clause.getVarList(), Clause.getEndLoc());
537  }
538  case OpenACCClauseKind::FirstPrivate: {
539  // Restrictions only properly implemented on 'compute' constructs, and
540  // 'compute' constructs are the only construct that can do anything with
541  // this yet, so skip/treat as unimplemented in this case.
542  if (!isOpenACCComputeDirectiveKind(Clause.getDirectiveKind()))
543  break;

546  // really isn't anything to do here. GCC does some duplicate-finding, though
547  // it isn't apparent in the standard where this is justified.
548 
549  return OpenACCFirstPrivateClause::Create(
550  getASTContext(), Clause.getBeginLoc(), Clause.getLParenLoc(),
551  Clause.getVarList(), Clause.getEndLoc());
552  }
553  case OpenACCClauseKind::NoCreate: {
554  // Restrictions only properly implemented on 'compute' constructs, and
555  // 'compute' constructs are the only construct that can do anything with
556  // this yet, so skip/treat as unimplemented in this case.

567  }

569  // Restrictions only properly implemented on 'compute' constructs, and

786 /// OpenACC 3.3 section 2.5.15:
787 /// At a mininmum, the supported data types include ... the numerical data types
788 /// in C, C++, and Fortran.
789 ///
790 /// If the reduction var is a composite variable, each
791 /// member of the composite variable must be a supported datatype for the
792 /// reduction operation.
793 ExprResult SemaOpenACC::CheckReductionVar(Expr *VarExpr) {
794  VarExpr = VarExpr->IgnoreParenCasts();
795 
796  auto TypeIsValid = [](QualType Ty) {
797  return Ty->isDependentType() || Ty->isScalarType();
798  };
799 
800  if (isa<ArraySectionExpr>(VarExpr)) {
801  Expr *ASExpr = VarExpr;
802  QualType BaseTy = ArraySectionExpr::getBaseOriginalType(ASExpr);
803  QualType EltTy = getASTContext().getBaseElementType(BaseTy);
804 
805  if (!TypeIsValid(EltTy)) {
806  Diag(VarExpr->getExprLoc(), diag::err_acc_reduction_type)
807  << EltTy << /*Sub array base type*/ 1;
808  return ExprError();
809  }
810  } else if (auto *RD = VarExpr->getType()->getAsRecordDecl()) {
811  if (!RD->isStruct() && !RD->isClass()) {
812  Diag(VarExpr->getExprLoc(), diag::err_acc_reduction_composite_type)
813  << /*not class or struct*/ 0 << VarExpr->getType();
814  return ExprError();
815  }
816 
817  if (!RD->isCompleteDefinition()) {
818  Diag(VarExpr->getExprLoc(), diag::err_acc_reduction_composite_type)
819  << /*incomplete*/ 1 << VarExpr->getType();
820  return ExprError();
821  }
822  if (const auto *CXXRD = dyn_cast<CXXRecordDecl>(RD);
823  CXXRD && !CXXRD->isAggregate()) {
824  Diag(VarExpr->getExprLoc(), diag::err_acc_reduction_composite_type)
825  << /*aggregate*/ 2 << VarExpr->getType();
826  return ExprError();
827  }
828 
829  for (FieldDecl *FD : RD->fields()) {
830  if (!TypeIsValid(FD->getType())) {
831  Diag(VarExpr->getExprLoc(),
832  diag::err_acc_reduction_composite_member_type);
833  Diag(FD->getLocation(), diag::note_acc_reduction_composite_member_loc);
834  return ExprError();
835  }
836  }
837  } else if (!TypeIsValid(VarExpr->getType())) {
838  Diag(VarExpr->getExprLoc(), diag::err_acc_reduction_type)
839  << VarExpr->getType() << /*Sub array base type*/ 0;
840  return ExprError();
841  }
842 
843  return VarExpr;
844 }

995 ExprResult SemaOpenACC::ActOnVar(OpenACCClauseKind CK, Expr *VarExpr) {
996  Expr *CurVarExpr = VarExpr->IgnoreParenImpCasts();
997 
998  // Sub-arrays/subscript-exprs are fine as long as the base is a
999  // VarExpr/MemberExpr. So strip all of those off.
1000  while (isa<ArraySectionExpr, ArraySubscriptExpr>(CurVarExpr)) {
1001  if (auto *SubScrpt = dyn_cast<ArraySubscriptExpr>(CurVarExpr))
1002  CurVarExpr = SubScrpt->getBase()->IgnoreParenImpCasts();
1003  else
1004  CurVarExpr =
1005  cast<ArraySectionExpr>(CurVarExpr)->getBase()->IgnoreParenImpCasts();
1006  }
1007 
1008  // References to a VarDecl are fine.
1009  if (const auto *DRE = dyn_cast<DeclRefExpr>(CurVarExpr)) {
1010  if (isa<VarDecl, NonTypeTemplateParmDecl>(
1011  DRE->getFoundDecl()->getCanonicalDecl()))
1012  return VarExpr;
1013  }
1014 
1015  // If CK is a Reduction, this special cases for OpenACC3.3 2.5.15: "A var in a
1016  // reduction clause must be a scalar variable name, an aggregate variable
1017  // name, an array element, or a subarray.
1018  // A MemberExpr that references a Field is valid.
1019  if (CK != OpenACCClauseKind::Reduction) {
1020  if (const auto *ME = dyn_cast<MemberExpr>(CurVarExpr)) {
1021  if (isa<FieldDecl>(ME->getMemberDecl()->getCanonicalDecl()))
1022  return VarExpr;
1023  }
1024  }
1025 
1026  // Referring to 'this' is always OK.
1027  if (isa<CXXThisExpr>(CurVarExpr))
1028  return VarExpr;
1029 
1030  // Nothing really we can do here, as these are dependent. So just return they
1031  // are valid.
1032  if (isa<DependentScopeDeclRefExpr>(CurVarExpr) ||
1033  (CK != OpenACCClauseKind::Reduction &&
1034  isa<CXXDependentScopeMemberExpr>(CurVarExpr)))
1035  return VarExpr;
1036 
1037  // There isn't really anything we can do in the case of a recovery expr, so
1038  // skip the diagnostic rather than produce a confusing diagnostic.
1039  if (isa<RecoveryExpr>(CurVarExpr))
1040  return ExprError();
1041 
1042  Diag(VarExpr->getExprLoc(), diag::err_acc_not_a_var_ref)
1043  << (CK != OpenACCClauseKind::Reduction);
1044  return ExprError();
1045 }

1048  Expr *LowerBound,
1049  SourceLocation ColonLoc,
1050  Expr *Length,
1051  SourceLocation RBLoc) {
1052  ASTContext &Context = getASTContext();
1053 
1054  // Handle placeholders.
1055  if (Base->hasPlaceholderType() &&
1056  !Base->hasPlaceholderType(BuiltinType::ArraySection)) {
1057  ExprResult Result = SemaRef.CheckPlaceholderExpr(Base);
1058  if (Result.isInvalid())
1059  return ExprError();
1060  Base = Result.get();
1061  }
1062  if (LowerBound && LowerBound->getType()->isNonOverloadPlaceholderType()) {
1063  ExprResult Result = SemaRef.CheckPlaceholderExpr(LowerBound);
1064  if (Result.isInvalid())
1065  return ExprError();

1067  if (Result.isInvalid())
1068  return ExprError();

1070  }
1071  if (Length && Length->getType()->isNonOverloadPlaceholderType()) {
1072  ExprResult Result = SemaRef.CheckPlaceholderExpr(Length);
1073  if (Result.isInvalid())
1074  return ExprError();

1076  if (Result.isInvalid())
1077  return ExprError();

1082  // a function type.
1083  QualType OriginalBaseTy = ArraySectionExpr::getBaseOriginalType(Base);
1084  QualType ResultTy;
1085  if (!Base->isTypeDependent()) {
1086  if (OriginalBaseTy->isAnyPointerType()) {
1087  ResultTy = OriginalBaseTy->getPointeeType();
1088  } else if (OriginalBaseTy->isArrayType()) {
1089  ResultTy = OriginalBaseTy->getAsArrayTypeUnsafe()->getElementType();
1090  } else {
1091  return ExprError(
1092  Diag(Base->getExprLoc(), diag::err_acc_typecheck_subarray_value)
1093  << Base->getSourceRange());
1094  }
1095

1097  Diag(Base->getExprLoc(), diag::err_acc_subarray_function_type)
1098  << ResultTy << Base->getSourceRange();
1099  return ExprError();
1100  }
1101 
1102  if (SemaRef.RequireCompleteType(Base->getExprLoc(), ResultTy,
1103  diag::err_acc_subarray_incomplete_type,
1104  Base))
1105  return ExprError();
1106 
1107  if (!Base->hasPlaceholderType(BuiltinType::ArraySection)) {
1108  ExprResult Result = SemaRef.DefaultFunctionArrayLvalueConversion(Base);
1109  if (Result.isInvalid())
1110  return ExprError();
1111  Base = Result.get();
1112  }
1113  }

1115  auto GetRecovery = [&](Expr *E, QualType Ty) {
1116  ExprResult Recovery =
1117  SemaRef.CreateRecoveryExpr(E->getBeginLoc(), E->getEndLoc(), E, Ty);
1118  return Recovery.isUsable() ? Recovery.get() : nullptr;
1119  };
1120 
1121  // Ensure both of the expressions are int-exprs.
1122  if (LowerBound && !LowerBound->isTypeDependent()) {
1123  ExprResult LBRes =
1124  ActOnIntExpr(OpenACCDirectiveKind::Invalid, OpenACCClauseKind::Invalid,
1125  LowerBound->getExprLoc(), LowerBound);
1126 
1127  if (LBRes.isUsable())
1128  LBRes = SemaRef.DefaultLvalueConversion(LBRes.get());
1129  LowerBound =
1130  LBRes.isUsable() ? LBRes.get() : GetRecovery(LowerBound, Context.IntTy);
1131  }
1132 
1133  if (Length && !Length->isTypeDependent()) {
1134  ExprResult LenRes =
1135  ActOnIntExpr(OpenACCDirectiveKind::Invalid, OpenACCClauseKind::Invalid,
1136  Length->getExprLoc(), Length);
1137 
1138  if (LenRes.isUsable())
1139  LenRes = SemaRef.DefaultLvalueConversion(LenRes.get());
1140  Length =
1141  LenRes.isUsable() ? LenRes.get() : GetRecovery(Length, Context.IntTy);
1142  }
1143 
1144  // Length is required if the base type is not an array of known bounds.
1145  if (!Length && (OriginalBaseTy.isNull() ||
1146  (!OriginalBaseTy->isDependentType() &&
1147  !OriginalBaseTy->isConstantArrayType() &&
1148  !OriginalBaseTy->isDependentSizedArrayType()))) {
1149  bool IsArray = !OriginalBaseTy.isNull() && OriginalBaseTy->isArrayType();
1150  Diag(ColonLoc, diag::err_acc_subarray_no_length) << IsArray;
1151  // Fill in a dummy 'length' so that when we instantiate this we don't
1152  // double-diagnose here.
1153  ExprResult Recovery = SemaRef.CreateRecoveryExpr(
1154  ColonLoc, SourceLocation(), ArrayRef<Expr *>{std::nullopt},
1155  Context.IntTy);
1156  Length = Recovery.isUsable() ? Recovery.get() : nullptr;
1157  }

1159  // Check the values of each of the arguments, they cannot be negative(we
1160  // assume), and if the array bound is known, must be within range. As we do
1161  // so, do our best to continue with evaluation, we can set the
1162  // value/expression to nullptr/nullopt if they are invalid, and treat them as
1163  // not present for the rest of evaluation.
1164 
1165  // We don't have to check for dependence, because the dependent size is
1166  // represented as a different AST node.
1167  std::optional<llvm::APSInt> BaseSize;
