21 bool diagnoseConstructAppertainment(Sema &S, OpenACCDirectiveKind K,
22  SourceLocation StartLoc, bool IsStmt) {
23  switch (K) {
24  default:
25  case OpenACCDirectiveKind::Invalid:
26  // Nothing to do here, both invalid and unimplemented don't really need to
27  // do anything.
28  break;
29  case OpenACCDirectiveKind::Parallel:
30  case OpenACCDirectiveKind::Serial:
31  case OpenACCDirectiveKind::Kernels:
32  if (!IsStmt)
33  return S.Diag(StartLoc, diag::err_acc_construct_appertainment) << K;
34  break;
35  }
36  return false;
37 }
38 } // namespace

40 bool Sema::ActOnOpenACCClause(OpenACCClauseKind ClauseKind,
41  SourceLocation StartLoc) {
42  if (ClauseKind == OpenACCClauseKind::Invalid)
43  return false;
44  // For now just diagnose that it is unsupported and leave the parsing to do
45  // whatever it can do. This function will eventually need to start returning
46  // some sort of Clause AST type, but for now just return true/false based on
47  // success.
48  return Diag(StartLoc, diag::warn_acc_clause_unimplemented) << ClauseKind;
49 }
50 void Sema::ActOnOpenACCConstruct(OpenACCDirectiveKind K,
51  SourceLocation StartLoc) {
52  switch (K) {
53  case OpenACCDirectiveKind::Invalid:
54  // Nothing to do here, an invalid kind has nothing we can check here. We
55  // want to continue parsing clauses as far as we can, so we will just
56  // ensure that we can still work and don't check any construct-specific
57  // rules anywhere.
58  break;
59  case OpenACCDirectiveKind::Parallel:
60  case OpenACCDirectiveKind::Serial:
61  case OpenACCDirectiveKind::Kernels:
62  // Nothing to do here, there is no real legalization that needs to happen
63  // here as these constructs do not take any arguments.
64  break;
65  default:
66  Diag(StartLoc, diag::warn_acc_construct_unimplemented) << K;
67  break;
68  }
69 }

76 StmtResult Sema::ActOnEndOpenACCStmtDirective(OpenACCDirectiveKind K,
77  SourceLocation StartLoc,
78  SourceLocation EndLoc,
79  StmtResult AssocStmt) {
80  switch (K) {
81  default:
82  return StmtEmpty();
83  case OpenACCDirectiveKind::Invalid:
84  return StmtError();
85  case OpenACCDirectiveKind::Parallel:
86  case OpenACCDirectiveKind::Serial:
87  case OpenACCDirectiveKind::Kernels:
88  return OpenACCComputeConstruct::Create(
89  getASTContext(), K, StartLoc, EndLoc,
90  AssocStmt.isUsable() ? AssocStmt.get() : nullptr);
91  }
92  llvm_unreachable("Unhandled case in directive handling?");
93 }

95 StmtResult Sema::ActOnOpenACCAssociatedStmt(OpenACCDirectiveKind K,
96  StmtResult AssocStmt) {
97  switch (K) {
98  default:
99  llvm_unreachable("Unimplemented associated statement application");
100  case OpenACCDirectiveKind::Parallel:
101  case OpenACCDirectiveKind::Serial:
102  case OpenACCDirectiveKind::Kernels:
103  // There really isn't any checking here that could happen. As long as we
104  // have a statement to associate, this should be fine.
105  // OpenACC 3.3 Section 6:
106  // Structured Block: in C or C++, an executable statement, possibly
107  // compound, with a single entry at the top and a single exit at the
108  // bottom.
109  // FIXME: Should we reject DeclStmt's here? The standard isn't clear, and
110  // an interpretation of it is to allow this and treat the initializer as
111  // the 'structured block'.
112  return AssocStmt;
113  }
114  llvm_unreachable("Invalid associated statement application");
115 }
