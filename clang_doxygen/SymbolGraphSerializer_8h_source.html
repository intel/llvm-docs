31 /// The serializer that organizes API information in the Symbol Graph format.
32 ///
33 /// The Symbol Graph format (https://github.com/apple/swift-docc-symbolkit)
34 /// models an API set as a directed graph, where nodes are symbol declarations,
35 /// and edges are relationships between the connected symbols.
36 class SymbolGraphSerializer : public APISerializer {
37  virtual void anchor();
38 
39  /// A JSON array of formatted symbols in \c APISet.
40  Array Symbols;
41 
42  /// A JSON array of formatted symbol relationships in \c APISet.
43  Array Relationships;
44 
45  /// The Symbol Graph format version used by this serializer.
46  static const VersionTuple FormatVersion;
47 
48 public:
49  /// Serialize the APIs in \c APISet in the Symbol Graph format.
50  ///
51  /// \returns a JSON object that contains the root of the formatted
52  /// Symbol Graph.
53  Object serialize();
54 
55  /// Implement the APISerializer::serialize interface. Wrap serialize(void) and
56  /// write out the serialized JSON object to \p os.
57  void serialize(raw_ostream &os) override;
58 
59  /// The kind of a relationship between two symbols.
60  enum RelationshipKind {
61  /// The source symbol is a member of the target symbol.
62  /// For example enum constants are members of the enum, class/instance
63  /// methods are members of the class, etc.
64  MemberOf,
65  };
66 
67  /// Get the string representation of the relationship kind.
68  static StringRef getRelationshipString(RelationshipKind Kind);
69 
70 private:
71  /// Synthesize the metadata section of the Symbol Graph format.
72  ///
73  /// The metadata section describes information about the Symbol Graph itself,
74  /// including the format version and the generator information.
75  Object serializeMetadata() const;
76 
77  /// Synthesize the module section of the Symbol Graph format.
78  ///
79  /// The module section contains information about the product that is defined
80  /// by the given API set.
81  /// Note that "module" here is not to be confused with the Clang/C++ module
82  /// concept.
83  Object serializeModule() const;
84 
85  /// Determine if the given \p Record should be skipped during serialization.
86  bool shouldSkip(const APIRecord &Record) const;
87 
88  /// Format the common API information for \p Record.
89  ///
90  /// This handles the shared information of all kinds of API records,
91  /// for example identifier and source location. The resulting object is then
92  /// augmented with kind-specific symbol information by the caller.
93  /// This method also checks if the given \p Record should be skipped during
94  /// serialization.
95  ///
96  /// \returns \c None if this \p Record should be skipped, or a JSON object
97  /// containing common symbol information of \p Record.
98  Optional<Object> serializeAPIRecord(const APIRecord &Record) const;
99 
100  /// Serialize the \p Kind relationship between \p Source and \p Target.
101  ///
102  /// Record the relationship between the two symbols in
103  /// SymbolGraphSerializer::Relationships.
104  void serializeRelationship(RelationshipKind Kind, const APIRecord &Source,
105  const APIRecord &Target);
106 
107  /// Serialize a global record.
108  void serializeGlobalRecord(const GlobalRecord &Record);
109 
110  /// Serialize an enum record.
111  void serializeEnumRecord(const EnumRecord &Record);
112 
113  /// Serialize a struct record.
114  void serializeStructRecord(const StructRecord &Record);
115 
116 public:
117  SymbolGraphSerializer(const APISet &API, StringRef ProductName,
118  APISerializerOption Options = {})
119  : APISerializer(API, ProductName, Options) {}
120 };
