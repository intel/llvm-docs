17 /// Loads a memory sequence containing the specified memory address into
18 /// all data cache levels.
19 ///
20 /// The cache-coherency state is set to exclusive. Data can be read from
21 /// and written to the cache line without additional delay.
22 ///
23 /// \headerfile <x86intrin.h>
24 ///
25 /// This intrinsic corresponds to the \c PREFETCHT0 instruction.
26 ///
27 /// \param __P
28 /// A pointer specifying the memory address to be prefetched.
29 static __inline__ void __attribute__((__always_inline__, __nodebug__))
30 _m_prefetch(void *__P)
31 {
32  __builtin_prefetch (__P, 0, 3 /* _MM_HINT_T0 */);
33 }
