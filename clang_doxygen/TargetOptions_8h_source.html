92  };

95

99  // code model.
100  std::string CodeModel;
101 
102  /// The version of the SDK which was used during the compilation.
103  /// The option is used for two different purposes:
104  /// * on darwin the version is propagated to LLVM where it's used

108  llvm::VersionTuple SDKVersion;
109 
110  /// The name of the darwin target- ariant triple to compile for.
111  std::string DarwinTargetVariantTriple;
112 
113  /// The version of the darwin target variant SDK which was used during the
114  /// compilation.
115  llvm::VersionTuple DarwinTargetVariantSDKVersion;
