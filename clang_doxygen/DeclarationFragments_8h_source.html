30 /// DeclarationFragments is a vector of tagged important parts of a symbol's
31 /// declaration.
32 ///
33 /// The fragments sequence can be joined to form spans of declaration text, with
34 /// attached information useful for purposes like syntax-highlighting etc.
35 /// For example:
36 /// \code
37 /// const -> keyword "const"
38 /// int -> type "int"
39 /// pi; -> identifier "pi"
40 /// \endcode
41 class DeclarationFragments {
42 public:
43  DeclarationFragments() = default;
44 
45  /// The kind of a fragment.
46  enum class FragmentKind {
47  /// Unknown fragment kind.
48  None,
49 
50  Keyword,
51  Attribute,
52  NumberLiteral,
53  StringLiteral,
54  Identifier,
55 
56  /// Identifier that refers to a type in the context.
57  TypeIdentifier,
58 
59  /// Parameter that's used as generics in the context. For example template
60  /// parameters.
61  GenericParameter,
62 
63  /// External parameters in Objective-C methods.
64  /// For example, \c forKey in
65  /// \code{.m}
66  /// - (void) setValue:(Value)value forKey(Key)key
67  /// \endcode
68  ExternalParam,
69 
70  /// Internal/local parameters in Objective-C methods.
71  /// For example, \c key in
72  /// \code{.m}
73  /// - (void) setValue:(Value)value forKey(Key)key
74  /// \endcode
75  InternalParam,
76 
77  Text,
78  };
79 
80  /// Fragment holds information of a single fragment.
81  struct Fragment {
82  std::string Spelling;
83  FragmentKind Kind;
84 
85  /// The USR of the fragment symbol, if applicable.
86  std::string PreciseIdentifier;
87 
88  Fragment(StringRef Spelling, FragmentKind Kind, StringRef PreciseIdentifier)
89  : Spelling(Spelling), Kind(Kind), PreciseIdentifier(PreciseIdentifier) {
90  }
91  };
92 
93  const std::vector<Fragment> &getFragments() const { return Fragments; }
94 
95  /// Append a new Fragment to the end of the Fragments.
96  ///
97  /// \returns a reference to the DeclarationFragments object itself after
98  /// appending to chain up consecutive appends.
99  DeclarationFragments &append(StringRef Spelling, FragmentKind Kind,
100  StringRef PreciseIdentifier = "") {
101  if (Kind == FragmentKind::Text && !Fragments.empty() &&
102  Fragments.back().Kind == FragmentKind::Text) {
103  // If appending a text fragment, and the last fragment is also text,
104  // merge into the last fragment.
105  Fragments.back().Spelling.append(Spelling.data(), Spelling.size());
106  } else {
107  Fragments.emplace_back(Spelling, Kind, PreciseIdentifier);
108  }
109  return *this;
110  }
111 
112  /// Append another DeclarationFragments to the end.
113  ///
114  /// Note: \p Other is moved from and cannot be used after a call to this
115  /// method.
116  ///
117  /// \returns a reference to the DeclarationFragments object itself after
118  /// appending to chain up consecutive appends.
119  DeclarationFragments &append(DeclarationFragments &&Other) {
120  Fragments.insert(Fragments.end(),
121  std::make_move_iterator(Other.Fragments.begin()),
122  std::make_move_iterator(Other.Fragments.end()));
123  Other.Fragments.clear();
124  return *this;
125  }
126 
127  /// Append a text Fragment of a space character.
128  ///
129  /// \returns a reference to the DeclarationFragments object itself after
130  /// appending to chain up consecutive appends.
131  DeclarationFragments &appendSpace();
132 
133  /// Get the string description of a FragmentKind \p Kind.
134  static StringRef getFragmentKindString(FragmentKind Kind);
135 
136  /// Get the corresponding FragmentKind from string \p S.
137  static FragmentKind parseFragmentKindFromString(StringRef S);
138 
139 private:
140  std::vector<Fragment> Fragments;
141 };

143 /// Store function signature information with DeclarationFragments of the
144 /// return type and parameters.
145 class FunctionSignature {
146 public:
147  FunctionSignature() = default;
148 
149  /// Parameter holds the name and DeclarationFragments of a single parameter.
150  struct Parameter {
151  std::string Name;
152  DeclarationFragments Fragments;
153 
154  Parameter(StringRef Name, DeclarationFragments Fragments)
155  : Name(Name), Fragments(Fragments) {}
156  };
157 
158  const std::vector<Parameter> &getParameters() const { return Parameters; }
159  const DeclarationFragments &getReturnType() const { return ReturnType; }
160 
161  FunctionSignature &addParameter(StringRef Name,
162  DeclarationFragments Fragments) {
163  Parameters.emplace_back(Name, Fragments);
164  return *this;
165  }
166 
167  void setReturnType(DeclarationFragments RT) { ReturnType = RT; }
168 
169  /// Determine if the FunctionSignature is empty.
170  ///
171  /// \returns true if the return type DeclarationFragments is empty and there
172  /// is no parameter, otherwise false.
173  bool empty() const {
174  return Parameters.empty() && ReturnType.getFragments().empty();
175  }
176 
177 private:
178  std::vector<Parameter> Parameters;
179  DeclarationFragments ReturnType;
180 };
