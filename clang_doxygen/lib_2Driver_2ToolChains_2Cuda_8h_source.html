52  /// Emit an error if Version does not support the given Arch.
53  ///
54  /// If either Version or Arch is unknown, does not emit an error. Emits at
55  /// most one error per Arch.
56  void CheckCudaVersionSupportsArch(CudaArch Arch) const;
57 
58  /// Check whether we detected a valid Cuda install.
59  bool isValid() const { return IsValid; }
60  /// Print information about the detected CUDA installation.
61  void print(raw_ostream &OS) const;
62 
63  /// Get the detected Cuda install's version.
64  CudaVersion version() const {
65  return Version == CudaVersion::NEW ? CudaVersion::PARTIALLY_SUPPORTED
66  : Version;
67  }
68  /// Get the detected Cuda installation path.
69  StringRef getInstallPath() const { return InstallPath; }
70  /// Get the detected path to Cuda's bin directory.
71  StringRef getBinPath() const { return BinPath; }
72  /// Get the detected Cuda Include path.
73  StringRef getIncludePath() const { return IncludePath; }
74  /// Get the detected Cuda library path.
75  StringRef getLibPath() const { return LibPath; }
76  /// Get the detected Cuda device library path.
77  StringRef getLibDevicePath() const { return LibDevicePath; }
78  /// Get libdevice file for given architecture
79  std::string getLibDeviceFile(StringRef Gpu) const {
80  return LibDeviceMap.lookup(Gpu);
81  }
82  void WarnIfUnsupportedVersion();
83 };
