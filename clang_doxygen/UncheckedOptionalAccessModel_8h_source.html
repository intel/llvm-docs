27 // FIXME: Explore using an allowlist-approach, where constructs supported by the
28 // analysis are always enabled and additional constructs are enabled through the

30 struct UncheckedOptionalAccessModelOptions {
31  /// Ignore optionals reachable through overloaded `operator*` or `operator->`
32  /// (other than those of the optional type itself). The analysis does not
33  /// equate the results of such calls, so it can't identify when their results
34  /// are used safely (across calls), resulting in false positives in all such
35  /// cases. Note: this option does not cover access through `operator[]`.
36  bool IgnoreSmartPointerDereference = false;
37 };
