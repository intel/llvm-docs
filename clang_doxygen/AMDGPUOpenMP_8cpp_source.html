34 static const char *getOutputFileName(Compilation &C, StringRef Base,
35  const char *Postfix,
36  const char *Extension) {
37  const char *OutputFileName;
38  if (C.getDriver().isSaveTempsEnabled()) {
39  OutputFileName =
40  C.getArgs().MakeArgString(Base.str() + Postfix + "." + Extension);
41  } else {
42  std::string TmpName =
43  C.getDriver().GetTemporaryPath(Base.str() + Postfix, Extension);
44  OutputFileName = C.addTempFile(C.getArgs().MakeArgString(TmpName));
45  }
46  return OutputFileName;
47 }

49 static void addLLCOptArg(const llvm::opt::ArgList &Args,
50  llvm::opt::ArgStringList &CmdArgs) {
51  if (Arg *A = Args.getLastArg(options::OPT_O_Group)) {
52  StringRef OOpt = "0";
53  if (A->getOption().matches(options::OPT_O4) ||
54  A->getOption().matches(options::OPT_Ofast))
55  OOpt = "3";
56  else if (A->getOption().matches(options::OPT_O0))
57  OOpt = "0";
58  else if (A->getOption().matches(options::OPT_O)) {
59  // Clang and opt support -Os/-Oz; llc only supports -O0, -O1, -O2 and -O3
60  // so we map -Os/-Oz to -O2.
61  // Only clang supports -Og, and maps it to -O1.
62  // We map anything else to -O2.
63  OOpt = llvm::StringSwitch<const char *>(A->getValue())
64  .Case("1", "1")
65  .Case("2", "2")
66  .Case("3", "3")
67  .Case("s", "2")
68  .Case("z", "2")
69  .Case("g", "1")
70  .Default("0");
71  }
72  CmdArgs.push_back(Args.MakeArgString("-O" + OOpt));
73  }
74 }

77 const char *AMDGCN::OpenMPLinker::constructLLVMLinkCommand(
78  const toolchains::AMDGPUOpenMPToolChain &AMDGPUOpenMPTC, Compilation &C,
79  const JobAction &JA, const InputInfoList &Inputs, const ArgList &Args,
80  StringRef SubArchName, StringRef OutputFilePrefix) const {
81  ArgStringList CmdArgs;
82 
83  for (const auto &II : Inputs)
84  if (II.isFilename())
85  CmdArgs.push_back(II.getFilename());
86 
87  bool HasLibm = false;
88  if (Args.hasArg(options::OPT_l)) {
89  auto Lm = Args.getAllArgValues(options::OPT_l);
90  for (auto &Lib : Lm) {
91  if (Lib == "m") {
92  HasLibm = true;
93  break;
94  }
95  }
96 
97  if (HasLibm) {
98  // This is not certain to work. The device libs added here, and passed to
99  // llvm-link, are missing attributes that they expect to be inserted when
100  // passed to mlink-builtin-bitcode. The amdgpu backend does not generate
101  // conservatively correct code when attributes are missing, so this may
102  // be the root cause of miscompilations. Passing via mlink-builtin-bitcode
103  // ultimately hits CodeGenModule::addDefaultFunctionDefinitionAttributes
104  // on each function, see D28538 for context.
105  // Potential workarounds:
106  // - unconditionally link all of the device libs to every translation
107  // unit in clang via mlink-builtin-bitcode
108  // - build a libm bitcode file as part of the DeviceRTL and explictly
109  // mlink-builtin-bitcode the rocm device libs components at build time
110  // - drop this llvm-link fork in favour or some calls into LLVM, chosen
111  // to do basically the same work as llvm-link but with that call first
112  // - write an opt pass that sets that on every function it sees and pipe
113  // the device-libs bitcode through that on the way to this llvm-link
114  SmallVector<std::string, 12> BCLibs =
115  AMDGPUOpenMPTC.getCommonDeviceLibNames(Args, SubArchName.str(),
116  Action::OFK_OpenMP);
117  for (StringRef BCFile : BCLibs)
118  CmdArgs.push_back(Args.MakeArgString(BCFile));
119  }
120  }
121 
122  AddStaticDeviceLibsLinking(C, *this, JA, Inputs, Args, CmdArgs, "amdgcn",
123  SubArchName, /*isBitCodeSDL=*/true);
124  // Add an intermediate output file.
125  CmdArgs.push_back("-o");
126  const char *OutputFileName =
127  getOutputFileName(C, OutputFilePrefix, "-linked", "bc");
128  CmdArgs.push_back(OutputFileName);
129  const char *Exec =
130  Args.MakeArgString(getToolChain().GetProgramPath("llvm-link"));
131  C.addCommand(std::make_unique<Command>(
132  JA, *this, ResponseFileSupport::AtFileCurCP(), Exec, CmdArgs, Inputs,
133  InputInfo(&JA, Args.MakeArgString(OutputFileName))));
134 
135  // If we linked in libm definitions late we run another round of optimizations
136  // to inline the definitions and fold what is foldable.
137  if (HasLibm) {
138  ArgStringList OptCmdArgs;
139  const char *OptOutputFileName =
140  getOutputFileName(C, OutputFilePrefix, "-linked-opt", "bc");
141  addLLCOptArg(Args, OptCmdArgs);
142  OptCmdArgs.push_back(OutputFileName);
143  OptCmdArgs.push_back("-o");
144  OptCmdArgs.push_back(OptOutputFileName);
145  const char *OptExec =
146  Args.MakeArgString(getToolChain().GetProgramPath("opt"));
147  C.addCommand(std::make_unique<Command>(
148  JA, *this, ResponseFileSupport::AtFileCurCP(), OptExec, OptCmdArgs,
149  InputInfo(&JA, Args.MakeArgString(OutputFileName)),
150  InputInfo(&JA, Args.MakeArgString(OptOutputFileName))));
151  OutputFileName = OptOutputFileName;
152  }
153 
154  return OutputFileName;
155 }

157 const char *AMDGCN::OpenMPLinker::constructLlcCommand(
158  Compilation &C, const JobAction &JA, const InputInfoList &Inputs,
159  const llvm::opt::ArgList &Args, llvm::StringRef SubArchName,
160  llvm::StringRef OutputFilePrefix, const char *InputFileName,
161  bool OutputIsAsm) const {
162  // Construct llc command.
163  ArgStringList LlcArgs;
164  // The input to llc is the output from opt.
165  LlcArgs.push_back(InputFileName);
166  // Pass optimization arg to llc.
167  addLLCOptArg(Args, LlcArgs);
168  LlcArgs.push_back("-mtriple=amdgcn-amd-amdhsa");
169  LlcArgs.push_back(Args.MakeArgString("-mcpu=" + SubArchName));
170  LlcArgs.push_back(
171  Args.MakeArgString(Twine("-filetype=") + (OutputIsAsm ? "asm" : "obj")));
172 
173  for (const Arg *A : Args.filtered(options::OPT_mllvm)) {
174  LlcArgs.push_back(A->getValue(0));
175  }
176 
177  // Add output filename
178  LlcArgs.push_back("-o");
179  const char *LlcOutputFile =
180  getOutputFileName(C, OutputFilePrefix, "", OutputIsAsm ? "s" : "o");
181  LlcArgs.push_back(LlcOutputFile);
182  const char *Llc = Args.MakeArgString(getToolChain().GetProgramPath("llc"));
183  C.addCommand(std::make_unique<Command>(
184  JA, *this, ResponseFileSupport::AtFileCurCP(), Llc, LlcArgs, Inputs,
185  InputInfo(&JA, Args.MakeArgString(LlcOutputFile))));
186  return LlcOutputFile;
187 }

205 // For amdgcn the inputs of the linker job are device bitcode and output is
206 // object file. It calls llvm-link, opt, llc, then lld steps.
207 void AMDGCN::OpenMPLinker::ConstructJob(Compilation &C, const JobAction &JA,
208  const InputInfo &Output,
209  const InputInfoList &Inputs,
210  const ArgList &Args,
211  const char *LinkingOutput) const {
212  const ToolChain &TC = getToolChain();
213  assert(getToolChain().getTriple().isAMDGCN() && "Unsupported target");
214 
215  const toolchains::AMDGPUOpenMPToolChain &AMDGPUOpenMPTC =
216  static_cast<const toolchains::AMDGPUOpenMPToolChain &>(TC);
217 
218  StringRef GPUArch = Args.getLastArgValue(options::OPT_march_EQ);
219  assert(!GPUArch.empty() && "Must have an explicit GPU arch.");
220 
221  // Prefix for temporary file name.
222  std::string Prefix;
223  for (const auto &II : Inputs)
224  if (II.isFilename())
225  Prefix =
226  llvm::sys::path::stem(II.getFilename()).str() + "-" + GPUArch.str();
227  assert(Prefix.length() && "no linker inputs are files ");
228 
229  // Each command outputs different files.
230  const char *LLVMLinkCommand = constructLLVMLinkCommand(
231  AMDGPUOpenMPTC, C, JA, Inputs, Args, GPUArch, Prefix);
232 
233  // Produce readable assembly if save-temps is enabled.
234  if (C.getDriver().isSaveTempsEnabled())
235  constructLlcCommand(C, JA, Inputs, Args, GPUArch, Prefix, LLVMLinkCommand,
236  /*OutputIsAsm=*/true);
237  const char *LlcCommand = constructLlcCommand(C, JA, Inputs, Args, GPUArch,
238  Prefix, LLVMLinkCommand);
239  constructLldCommand(C, JA, Inputs, Output, Args, LlcCommand);
240 }

252 void AMDGPUOpenMPToolChain::addClangTargetOptions(
253  const llvm::opt::ArgList &DriverArgs, llvm::opt::ArgStringList &CC1Args,
254  Action::OffloadKind DeviceOffloadingKind) const {
255  HostTC.addClangTargetOptions(DriverArgs, CC1Args, DeviceOffloadingKind);
256 
257  StringRef GPUArch = DriverArgs.getLastArgValue(options::OPT_march_EQ);
258  assert(!GPUArch.empty() && "Must have an explicit GPU arch.");
259 
260  assert(DeviceOffloadingKind == Action::OFK_OpenMP &&
261  "Only OpenMP offloading kinds are supported.");
262 
263  CC1Args.push_back("-target-cpu");
264  CC1Args.push_back(DriverArgs.MakeArgStringRef(GPUArch));
265  CC1Args.push_back("-fcuda-is-device");
266 
267  if (DriverArgs.hasArg(options::OPT_nogpulib))
268  return;
269 
270  for (auto BCFile : getDeviceLibs(DriverArgs, DeviceOffloadingKind)) {
271  CC1Args.push_back(BCFile.ShouldInternalize ? "-mlink-builtin-bitcode"
272  : "-mlink-bitcode-file");
273  CC1Args.push_back(DriverArgs.MakeArgString(BCFile.Path));
274  }
275 
276  // Link the bitcode library late if we're using device LTO.
277  if (getDriver().isUsingLTO(/* IsOffload */ true))
278  return;
279 }

281 llvm::opt::DerivedArgList *AMDGPUOpenMPToolChain::TranslateArgs(
282  const llvm::opt::DerivedArgList &Args, StringRef BoundArch,
283  Action::OffloadKind DeviceOffloadKind) const {
284  DerivedArgList *DAL =
285  HostTC.TranslateArgs(Args, BoundArch, DeviceOffloadKind);
286  if (!DAL)
287  DAL = new DerivedArgList(Args.getBaseArgs());
288 
289  const OptTable &Opts = getDriver().getOpts();
290 
291  if (DeviceOffloadKind == Action::OFK_OpenMP) {
292  for (Arg *A : Args)
293  if (!llvm::is_contained(*DAL, A))
294  DAL->append(A);
295 
296  if (!DAL->hasArg(options::OPT_march_EQ)) {
297  StringRef Arch = BoundArch;
298  if (Arch.empty()) {
299  auto ArchsOrErr = getSystemGPUArchs(Args);
300  if (!ArchsOrErr) {
301  std::string ErrMsg =
302  llvm::formatv("{0}", llvm::fmt_consume(ArchsOrErr.takeError()));
303  getDriver().Diag(diag::err_drv_undetermined_gpu_arch)
304  << llvm::Triple::getArchTypeName(getArch()) << ErrMsg << "-march";
305  Arch = CudaArchToString(CudaArch::HIPDefault);
306  } else {
307  Arch = Args.MakeArgString(ArchsOrErr->front());
308  }
309  }
310  DAL->AddJoinedArg(nullptr, Opts.getOption(options::OPT_march_EQ), Arch);
311  }
312 
313  return DAL;
314  }
315 
316  for (Arg *A : Args) {
317  DAL->append(A);
318  }
319 
320  if (!BoundArch.empty()) {
321  DAL->eraseArg(options::OPT_march_EQ);
322  DAL->AddJoinedArg(nullptr, Opts.getOption(options::OPT_march_EQ),
323  BoundArch);
324  }
325 
326  return DAL;
327 }

355 SanitizerMask AMDGPUOpenMPToolChain::getSupportedSanitizers() const {
356  // The AMDGPUOpenMPToolChain only supports sanitizers in the sense that it
357  // allows sanitizer arguments on the command line if they are supported by the
358  // host toolchain. The AMDGPUOpenMPToolChain will actually ignore any command
359  // line arguments for any of these "supported" sanitizers. That means that no
360  // sanitization of device code is actually supported at this time.
361  //
362  // This behavior is necessary because the host and device toolchains
363  // invocations often share the command line, so the device toolchain must
364  // tolerate flags meant only for the host toolchain.
365  return HostTC.getSupportedSanitizers();
366 }

374 llvm::SmallVector<ToolChain::BitCodeLibraryInfo, 12>
375 AMDGPUOpenMPToolChain::getDeviceLibs(
376  const llvm::opt::ArgList &Args,
377  const Action::OffloadKind DeviceOffloadingKind) const {
378  if (Args.hasArg(options::OPT_nogpulib))
379  return {};
380 
381  if (!RocmInstallation->hasDeviceLibrary()) {
382  getDriver().Diag(diag::err_drv_no_rocm_device_lib) << 0;
383  return {};
384  }
385 
386  StringRef GpuArch = getProcessorFromTargetID(
387  getTriple(), Args.getLastArgValue(options::OPT_march_EQ));
388 
389  SmallVector<BitCodeLibraryInfo, 12> BCLibs;
390  for (auto BCLib :
391  getCommonDeviceLibNames(Args, GpuArch.str(), DeviceOffloadingKind,
392  /*IsOpenMP=*/true))
393  BCLibs.emplace_back(BCLib);
394 
395  return BCLibs;
396 }
