235 inline bool isSYCLNativeCPU(const llvm::opt::ArgList &Args) {
236  if (auto SYCLTargets = Args.getLastArg(options::OPT_fsycl_targets_EQ)) {
237  if (SYCLTargets->containsValue("native_cpu"))
238  return true;
239  }
240  return false;
241 }

247 inline bool isSYCLNativeCPU(const ToolChain &TC) {
