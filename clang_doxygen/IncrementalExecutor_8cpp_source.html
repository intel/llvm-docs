40 llvm::Expected<std::unique_ptr<llvm::orc::LLJITBuilder>>
41 IncrementalExecutor::createDefaultJITBuilder(

79 llvm::Error IncrementalExecutor::removeModule(PartialTranslationUnit &PTU) {
80 
81  llvm::orc::ResourceTrackerSP RT = std::move(ResourceTrackers[&PTU]);
82  if (!RT)
83  return llvm::Error::success();
84 
85  ResourceTrackers.erase(&PTU);
86  if (llvm::Error Err = RT->remove())
87  return Err;
88  return llvm::Error::success();
89 }

101 llvm::Expected<llvm::orc::ExecutorAddr>
102 IncrementalExecutor::getSymbolAddress(llvm::StringRef Name,
103  SymbolNameKind NameKind) const {
104  using namespace llvm::orc;
105  auto SO = makeJITDylibSearchOrder({&Jit->getMainJITDylib(),
106  Jit->getPlatformJITDylib().get(),
107  Jit->getProcessSymbolsJITDylib().get()});
108 
109  ExecutionSession &ES = Jit->getExecutionSession();
110 
111  auto SymOrErr =
112  ES.lookup(SO, (NameKind == LinkerName) ? ES.intern(Name)
113  : Jit->mangleAndIntern(Name));
114  if (auto Err = SymOrErr.takeError())
115  return std::move(Err);
116  return SymOrErr->getAddress();
117 }
