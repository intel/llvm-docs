122 LLVM_NODISCARD static bool isRawStringLiteral(const char *First,
123  const char *Current) {
124  assert(First <= Current);
125 
126  // Check if we can even back up.
127  if (*Current != '"' || First == Current)
128  return false;
129 
130  // Check for an "R".
131  --Current;
132  if (*Current != 'R')
133  return false;
134  if (First == Current || !isAsciiIdentifierContinue(*--Current))
135  return true;
136 
137  // Check for a prefix of "u", "U", or "L".
138  if (*Current == 'u' || *Current == 'U' || *Current == 'L')
139  return First == Current || !isAsciiIdentifierContinue(*--Current);
140 
141  // Check for a prefix of "u8".
142  if (*Current != '8' || First == Current || *Current-- != 'u')
143  return false;
144  return First == Current || !isAsciiIdentifierContinue(*--Current);
145 }

147 static void skipRawString(const char *&First, const char *const End) {
148  assert(First[0] == '"');
149  assert(First[-1] == 'R');
150 
151  const char *Last = ++First;
152  while (Last != End && *Last != '(')
153  ++Last;
154  if (Last == End) {
155  First = Last; // Hit the end... just give up.
156  return;
157  }
158 
159  StringRef Terminator(First, Last - First);
160  for (;;) {
161  // Move First to just past the next ")".
162  First = Last;
163  while (First != End && *First != ')')
164  ++First;
165  if (First == End)
166  return;
167  ++First;
168 
169  // Look ahead for the terminator sequence.
170  Last = First;
171  while (Last != End && size_t(Last - First) < Terminator.size() &&
172  Terminator[Last - First] == *Last)
173  ++Last;
174 
175  // Check if we hit it (or the end of the file).
176  if (Last == End) {
177  First = Last;
178  return;
179  }
180  if (size_t(Last - First) < Terminator.size())
181  continue;
182  if (*Last != '"')
183  continue;
184  First = Last + 1;
185  return;
186  }
187 }

189 // Returns the length of EOL, either 0 (no end-of-line), 1 (\n) or 2 (\r\n)
190 static unsigned isEOL(const char *First, const char *const End) {
191  if (First == End)
192  return 0;
193  if (End - First > 1 && isVerticalWhitespace(First[0]) &&
194  isVerticalWhitespace(First[1]) && First[0] != First[1])
195  return 2;
196  return !!isVerticalWhitespace(First[0]);
197 }

199 static void skipString(const char *&First, const char *const End) {
200  assert(*First == '\'' || *First == '"' || *First == '<');
201  const char Terminator = *First == '<' ? '>' : *First;
202  for (++First; First != End && *First != Terminator; ++First) {
203  // String and character literals don't extend past the end of the line.
204  if (isVerticalWhitespace(*First))
205  return;
206  if (*First != '\\')
207  continue;
208  // Skip past backslash to the next character. This ensures that the
209  // character right after it is skipped as well, which matters if it's
210  // the terminator.
211  if (++First == End)
212  return;
213  if (!isWhitespace(*First))
214  continue;
215  // Whitespace after the backslash might indicate a line continuation.
216  const char *FirstAfterBackslashPastSpace = First;
217  skipOverSpaces(FirstAfterBackslashPastSpace, End);
218  if (unsigned NLSize = isEOL(FirstAfterBackslashPastSpace, End)) {
219  // Advance the character pointer to the next line for the next
220  // iteration.
221  First = FirstAfterBackslashPastSpace + NLSize - 1;
222  }
223  }
224  if (First != End)
225  ++First; // Finish off the string.
226 }

229 static unsigned skipNewline(const char *&First, const char *End) {
230  if (First == End)
231  return 0;
232  assert(isVerticalWhitespace(*First));
233  unsigned Len = isEOL(First, End);
234  assert(Len && "expected newline");
235  First += Len;
236  return Len;
237 }

243 static void skipToNewlineRaw(const char *&First, const char *const End) {
244  for (;;) {
245  if (First == End)
246  return;
247 
248  unsigned Len = isEOL(First, End);
249  if (Len)
250  return;
251 
252  do {
253  if (++First == End)
254  return;
255  Len = isEOL(First, End);
256  } while (!Len);
257 
258  if (First[-1] != '\\')
259  return;
260 
261  First += Len;
262  // Keep skipping lines...
263  }
264 }

297 static void skipBlockComment(const char *&First, const char *const End) {
298  assert(First[0] == '/' && First[1] == '*');
299  if (End - First < 4) {
300  First = End;
301  return;
302  }

310 /// \returns True if the current single quotation mark character is a C++ 14
311 /// digit separator.
312 static bool isQuoteCppDigitSeparator(const char *const Start,
313  const char *const Cur,
314  const char *const End) {
315  assert(*Cur == '\'' && "expected quotation character");
316  // skipLine called in places where we don't expect a valid number
317  // body before `start` on the same line, so always return false at the start.
318  if (Start == Cur)
319  return false;
320  // The previous character must be a valid PP number character.
321  // Make sure that the L, u, U, u8 prefixes don't get marked as a
322  // separator though.
323  char Prev = *(Cur - 1);
324  if (Prev == 'L' || Prev == 'U' || Prev == 'u')
325  return false;
326  if (Prev == '8' && (Cur - 1 != Start) && *(Cur - 2) == 'u')
327  return false;
328  if (!isPreprocessingNumberBody(Prev))
329  return false;
330  // The next character should be a valid identifier body character.
331  return (Cur + 1) < End && isAsciiIdentifierContinue(*(Cur + 1));
332 }

334 static void skipLine(const char *&First, const char *const End) {
335  for (;;) {
336  assert(First <= End);
337  if (First == End)
338  return;
339 
340  if (isVerticalWhitespace(*First)) {
341  skipNewline(First, End);
342  return;
343  }
344  const char *Start = First;
345  while (First != End && !isVerticalWhitespace(*First)) {
346  // Iterate over strings correctly to avoid comments and newlines.
347  if (*First == '"' ||
348  (*First == '\'' && !isQuoteCppDigitSeparator(Start, First, End))) {
349  if (isRawStringLiteral(Start, First))
350  skipRawString(First, End);
351  else
352  skipString(First, End);
353  continue;
354  }
355 
356  // Iterate over comments correctly.
357  if (*First != '/' || End - First < 2) {
358  ++First;
359  continue;
360  }
361 
362  if (First[1] == '/') {
363  // "//...".
364  skipLineComment(First, End);
365  continue;
366  }
367 
368  if (First[1] != '*') {
369  ++First;
370  continue;
371  }
372 
373  // "/*...*/".
374  skipBlockComment(First, End);
375  }
376  if (First == End)
377  return;
378 
379  // Skip over the newline.
380  unsigned Len = skipNewline(First, End);
381  if (!wasLineContinuation(First, Len)) // Continue past line-continuations.
382  break;
383  }
384 }

398 void Minimizer::printToNewline(const char *&First, const char *const End) {
399  while (First != End && !isVerticalWhitespace(*First)) {
400  const char *Last = First;
401  do {
402  // Iterate over strings correctly to avoid comments and newlines.
403  if (*Last == '"' || *Last == '\'' ||
404  (*Last == '<' && (top() == pp_include || top() == pp_import))) {
405  if (LLVM_UNLIKELY(isRawStringLiteral(First, Last)))
406  skipRawString(Last, End);
407  else
408  skipString(Last, End);
409  continue;
410  }
411  if (*Last != '/' || End - Last < 2) {
412  ++Last;
413  continue; // Gather the rest up to print verbatim.
414  }
415 
416  if (Last[1] != '/' && Last[1] != '*') {
417  ++Last;
418  continue;
419  }
420 
421  // Deal with "//..." and "/*...*/".
422  append(First, findFirstTrailingSpace(First, Last));
423  First = Last;
424 
425  if (Last[1] == '/') {
426  skipLineComment(First, End);
427  return;
428  }
429 
430  put(' ');
431  skipBlockComment(First, End);
432  skipOverSpaces(First, End);
433  Last = First;
434  } while (Last != End && !isVerticalWhitespace(*Last));
435 
436  // Print out the string.
437  const char *LastBeforeTrailingSpace = findLastNonSpace(First, Last);
438  if (Last == End || LastBeforeTrailingSpace == First ||
439  LastBeforeTrailingSpace[-1] != '\\') {
440  append(First, LastBeforeTrailingSpace);
441  First = Last;
442  skipNewline(First, End);
443  return;
444  }
445 
446  // Print up to the last character that's not a whitespace or backslash.
447  // Then print exactly one space, which matters when tokens are separated by
448  // a line continuation.
449  append(First, findLastNonSpaceNonBackslash(First, Last));
450  put(' ');
451 
452  First = Last;
453  skipNewline(First, End);
454  skipOverSpaces(First, End);
455  }
456 }

458 static void skipWhitespace(const char *&First, const char *const End) {
459  for (;;) {
460  assert(First <= End);
461  skipOverSpaces(First, End);
462 
463  if (End - First < 2)
464  return;
465 
466  if (First[0] == '\\' && isVerticalWhitespace(First[1])) {
467  skipNewline(++First, End);
468  continue;
469  }
470 
471  // Check for a non-comment character.
472  if (First[0] != '/')
473  return;
474 
475  // "// ...".
476  if (First[1] == '/') {
477  skipLineComment(First, End);
478  return;
479  }
480 
481  // Cannot be a comment.
482  if (First[1] != '*')
483  return;
484 
485  // "/*...*/".
486  skipBlockComment(First, End);
487  }
488 }

555 Minimizer::IdInfo Minimizer::lexIdentifier(const char *First,
556  const char *const End) {
557  const char *Last = lexRawIdentifier(First, End);
558  const char *Next = getIdentifierContinuation(Last, End);
559  if (LLVM_LIKELY(!Next))
560  return IdInfo{Last, StringRef(First, Last - First)};
561 
562  // Slow path, where identifiers are split over lines.
563  SmallVector<char, 64> Id(First, Last);
564  while (Next) {
565  Last = lexRawIdentifier(Next, End);
566  Id.append(Next, Last);
567  Next = getIdentifierContinuation(Last, End);
568  }
569  return IdInfo{
570  Last,
571  SplitIds.try_emplace(StringRef(Id.begin(), Id.size()), 0).first->first()};
572 }

644 bool Minimizer::lexModule(const char *&First, const char *const End) {
645  IdInfo Id = lexIdentifier(First, End);
646  First = Id.Last;
647  bool Export = false;
648  if (Id.Name == "export") {
649  Export = true;
650  skipWhitespace(First, End);
651  if (!isAsciiIdentifierContinue(*First)) {
652  skipLine(First, End);
653  return false;
654  }
655  Id = lexIdentifier(First, End);
656  First = Id.Last;
657  }
658 
659  if (Id.Name != "module" && Id.Name != "import") {
660  skipLine(First, End);
661  return false;
662  }
663 
664  skipWhitespace(First, End);
665 
666  // Ignore this as a module directive if the next character can't be part of
667  // an import.
668 
669  switch (*First) {
670  case ':':
671  case '<':
672  case '"':
673  break;
674  default:
675  if (!isAsciiIdentifierContinue(*First)) {
676  skipLine(First, End);
677  return false;
678  }
679  }
680 
681  if (Export) {
682  makeToken(cxx_export_decl);
683  append("export ");
684  }
685 
686  if (Id.Name == "module")
687  makeToken(cxx_module_decl);
688  else
689  makeToken(cxx_import_decl);
690  append(Id.Name);
691  append(" ");
692  printToNewline(First, End);
693  append("\n");
694  return false;
695 }

697 bool Minimizer::lexDefine(const char *&First, const char *const End) {
698  makeToken(pp_define);
699  append("#define ");
700  skipWhitespace(First, End);
701 
702  if (!isAsciiIdentifierStart(*First))
703  return reportError(First, diag::err_pp_macro_not_identifier);
704 
705  IdInfo Id = lexIdentifier(First, End);
706  const char *Last = Id.Last;
707  append(Id.Name);
708  if (Last == End)
709  return false;
710  if (*Last == '(') {
711  size_t Size = Out.size();
712  if (printMacroArgs(Last, End)) {
713  // Be robust to bad macro arguments, since they can show up in disabled
714  // code.
715  Out.resize(Size);
716  append("(/* invalid */\n");
717  skipLine(Last, End);
718  return false;
719  }
720  }
721  skipWhitespace(Last, End);
722  if (Last == End)
723  return false;

731 bool Minimizer::lexPragma(const char *&First, const char *const End) {
732  // #pragma.
733  skipWhitespace(First, End);
734  if (First == End || !isAsciiIdentifierStart(*First))
735  return false;
736 
737  IdInfo FoundId = lexIdentifier(First, End);
738  First = FoundId.Last;
739  if (FoundId.Name == "once") {
740  // #pragma once
741  skipLine(First, End);
742  makeToken(pp_pragma_once);
743  append("#pragma once\n");
744  return false;
745  }
746  if (FoundId.Name == "push_macro") {
747  // #pragma push_macro
748  makeToken(pp_pragma_push_macro);
749  append("#pragma push_macro");
750  printDirectiveBody(First, End);
751  return false;
752  }
753  if (FoundId.Name == "pop_macro") {
754  // #pragma pop_macro
755  makeToken(pp_pragma_pop_macro);
756  append("#pragma pop_macro");
757  printDirectiveBody(First, End);
758  return false;
759  }
760  if (FoundId.Name == "include_alias") {
761  // #pragma include_alias
762  makeToken(pp_pragma_include_alias);
763  append("#pragma include_alias");
764  printDirectiveBody(First, End);
765  return false;
766  }
767 
768  if (FoundId.Name != "clang") {
769  skipLine(First, End);
770  return false;
771  }
772 
773  // #pragma clang.
774  if (!isNextIdentifier("module", First, End)) {
775  skipLine(First, End);
776  return false;
777  }
778 
779  // #pragma clang module.
780  if (!isNextIdentifier("import", First, End)) {
781  skipLine(First, End);
782  return false;
783  }
784 
785  // #pragma clang module import.
786  makeToken(pp_pragma_import);
787  append("#pragma clang module import ");
788  printDirectiveBody(First, End);
789  return false;
790 }

792 bool Minimizer::lexEndif(const char *&First, const char *const End) {
793  // Strip out "#else" if it's empty.
794  if (top() == pp_else)
795  popToken();
796 
797  // If "#ifdef" is empty, strip it and skip the "#endif".
798  //
799  // FIXME: Once/if Clang starts disallowing __has_include in macro expansions,
800  // we can skip empty `#if` and `#elif` blocks as well after scanning for a
801  // literal __has_include in the condition. Even without that rule we could
802  // drop the tokens if we scan for identifiers in the condition and find none.
803  if (top() == pp_ifdef || top() == pp_ifndef) {
804  popToken();
805  skipLine(First, End);
806  return false;
807  }
808 
809  return lexDefault(pp_endif, "endif", First, End);
810 }

832 bool Minimizer::lexPPLine(const char *&First, const char *const End) {
833  assert(First != End);
834 
835  skipWhitespace(First, End);
836  assert(First <= End);
837  if (First == End)
838  return false;
839 
840  if (!isStartOfRelevantLine(*First)) {
841  skipLine(First, End);
842  assert(First <= End);
843  return false;
844  }
845 
846  // Handle "@import".
847  if (*First == '@')
848  return lexAt(First, End);
849 
850  if (*First == 'i' || *First == 'e' || *First == 'm')
851  return lexModule(First, End);
852 
853  // Handle preprocessing directives.
854  ++First; // Skip over '#'.
855  skipWhitespace(First, End);
856 
857  if (First == End)
858  return reportError(First, diag::err_pp_expected_eol);
859 
860  if (!isAsciiIdentifierStart(*First)) {
861  skipLine(First, End);
862  return false;
863  }
864 
865  // Figure out the token.
866  IdInfo Id = lexIdentifier(First, End);
867  First = Id.Last;
868 
869  if (Id.Name == "pragma")
870  return lexPragma(First, End);
871 
872  auto Kind = llvm::StringSwitch<TokenKind>(Id.Name)
873  .Case("include", pp_include)
874  .Case("__include_macros", pp___include_macros)
875  .Case("define", pp_define)
876  .Case("undef", pp_undef)
877  .Case("import", pp_import)
878  .Case("include_next", pp_include_next)
879  .Case("if", pp_if)
880  .Case("ifdef", pp_ifdef)
881  .Case("ifndef", pp_ifndef)
882  .Case("elif", pp_elif)
883  .Case("elifdef", pp_elifdef)
884  .Case("elifndef", pp_elifndef)
885  .Case("else", pp_else)
886  .Case("endif", pp_endif)
887  .Default(pp_none);
888  if (Kind == pp_none) {
889  skipDirective(Id.Name, First, End);
890  return false;
891  }
892 
893  if (Kind == pp_endif)
894  return lexEndif(First, End);
895 
896  if (Kind == pp_define)
897  return lexDefine(First, End);
898 
899  // Everything else.
900  return lexDefault(Kind, Id.Name, First, End);
901 }

934 bool clang::minimize_source_to_dependency_directives::computeSkippedRanges(
935  ArrayRef<Token> Input, llvm::SmallVectorImpl<SkippedRange> &Range) {
936  struct Directive {
937  enum DirectiveKind {
938  If, // if/ifdef/ifndef
939  Else // elif/elifdef/elifndef, else
940  };
941  int Offset;
942  DirectiveKind Kind;
943  };
944  llvm::SmallVector<Directive, 32> Offsets;
945  for (const Token &T : Input) {
946  switch (T.K) {
947  case pp_if:
948  case pp_ifdef:
949  case pp_ifndef:
950  Offsets.push_back({T.Offset, Directive::If});
951  break;
952 
953  case pp_elif:
954  case pp_elifdef:
955  case pp_elifndef:
956  case pp_else: {
957  if (Offsets.empty())
958  return true;
959  int PreviousOffset = Offsets.back().Offset;
960  Range.push_back({PreviousOffset, T.Offset - PreviousOffset});
961  Offsets.push_back({T.Offset, Directive::Else});
962  break;
963  }
964 
965  case pp_endif: {
966  if (Offsets.empty())
967  return true;
968  int PreviousOffset = Offsets.back().Offset;
969  Range.push_back({PreviousOffset, T.Offset - PreviousOffset});
970  do {
971  Directive::DirectiveKind Kind = Offsets.pop_back_val().Kind;
972  if (Kind == Directive::If)
973  break;
974  } while (!Offsets.empty());
975  break;
976  }
977  default:
978  break;
979  }
980  }
981  return false;
982 }
