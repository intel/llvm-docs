417 void LoopInfoStack::addSYCLIVDepInfo(llvm::LLVMContext &Ctx, unsigned SafeLen,
418  const ValueDecl *Array) {
419  // If there is a global that beats this one out, don't add/change anything.
420  if (StagedAttrs.GlobalSYCLIVDepInfo &&
421  (StagedAttrs.GlobalSYCLIVDepInfo->SafeLen == 0 ||
422  (SafeLen != 0 && StagedAttrs.GlobalSYCLIVDepInfo->SafeLen >= SafeLen)))
423  return;
424 
425  if (!Array) {
426  // Updating the global setting.
427  if (!StagedAttrs.GlobalSYCLIVDepInfo)
428  StagedAttrs.GlobalSYCLIVDepInfo = LoopAttributes::SYCLIVDepInfo{SafeLen};
429  else
430  StagedAttrs.GlobalSYCLIVDepInfo->SafeLen = SafeLen;
431 
432  // Remove any array collections that don't have a greater safelen than the
433  // global.
434  StagedAttrs.ArraySYCLIVDepInfo.erase(
435  llvm::remove_if(StagedAttrs.ArraySYCLIVDepInfo,
436  [SafeLen](const auto &A) {
437  return !A.isSafeLenGreaterOrEqual(SafeLen);
438  }),
439  StagedAttrs.ArraySYCLIVDepInfo.end());
440  return;
441  }
442 
443  auto SafeLenItr = llvm::find_if(
444  StagedAttrs.ArraySYCLIVDepInfo,
445  [SafeLen](const auto &Info) { return Info.SafeLen == SafeLen; });
446  auto ArrayItr =
447  llvm::find_if(StagedAttrs.ArraySYCLIVDepInfo,
448  [Array](const auto &Info) { return Info.hasArray(Array); });
449 
450  if (ArrayItr != StagedAttrs.ArraySYCLIVDepInfo.end()) {
451  // Ensure that the current array's safelen is greater than the existing one.
452  // Otherwise, there is nothing to do. We've already been checked against
453  // the global safelen.
454  if (ArrayItr->isSafeLenGreaterOrEqual(SafeLen))
455  return;
456 
457  // We know this exists, so no need to check the result of find_if, but
458  // remove the last array.
459  ArrayItr->eraseArray(Array);
460  }
461 
462  // Add this to the new safelen version.
463  if (SafeLenItr != StagedAttrs.ArraySYCLIVDepInfo.end()) {
464  SafeLenItr->Arrays.emplace_back(Array, MDNode::getDistinct(Ctx, {}));
465  return;
466  }
467 
468  StagedAttrs.ArraySYCLIVDepInfo.emplace_back(SafeLen, Array,
469  MDNode::getDistinct(Ctx, {}));
470 }

507 MDNode *LoopInfo::createMetadata(
508  const LoopAttributes &Attrs,
509  llvm::ArrayRef<llvm::Metadata *> AdditionalLoopProperties,
510  bool &HasUserTransforms) {
511  SmallVector<Metadata *, 3> LoopProperties;
512 
513  // If we have a valid start debug location for the loop, add it.
514  if (StartLoc) {
515  LoopProperties.push_back(StartLoc.getAsMDNode());
516 
517  // If we also have a valid end debug location for the loop, add it.
518  if (EndLoc)
519  LoopProperties.push_back(EndLoc.getAsMDNode());
520  }
521 
522  LLVMContext &Ctx = Header->getContext();
523  if (Attrs.MustProgress)
524  LoopProperties.push_back(
525  MDNode::get(Ctx, MDString::get(Ctx, "llvm.loop.mustprogress")));
526 
527  assert(!!AccGroup == Attrs.IsParallel &&
528  "There must be an access group iff the loop is parallel");
529  if (Attrs.IsParallel) {
530  LoopProperties.push_back(MDNode::get(
531  Ctx, {MDString::get(Ctx, "llvm.loop.parallel_accesses"), AccGroup}));
532  }
533 
534  if (Attrs.GlobalSYCLIVDepInfo.hasValue()) {
535  EmitIVDepLoopMetadata(Ctx, LoopProperties, *Attrs.GlobalSYCLIVDepInfo);
536  // The legacy metadata also needs to be emitted to provide backwards
537  // compatibility with any conformant backend. This is done exclusively
538  // for the "global" ivdep specification so as not to impose unnecessarily
539  // tight safe length constraints on the array-specific cases.
540  EmitLegacyIVDepLoopMetadata(Ctx, LoopProperties,
541  *Attrs.GlobalSYCLIVDepInfo);
542  }
543  for (const auto &I : Attrs.ArraySYCLIVDepInfo)
544  EmitIVDepLoopMetadata(Ctx, LoopProperties, I);
545 
546  // Setting ii attribute with an initiation interval
547  if (Attrs.SYCLIInterval > 0) {
548  Metadata *Vals[] = {MDString::get(Ctx, "llvm.loop.ii.count"),
549  ConstantAsMetadata::get(ConstantInt::get(
550  llvm::Type::getInt32Ty(Ctx), Attrs.SYCLIInterval))};
551  LoopProperties.push_back(MDNode::get(Ctx, Vals));
552  }
553 
554  // Setting max_concurrency attribute with number of threads
555  if (Attrs.SYCLMaxConcurrencyNThreads) {
556  Metadata *Vals[] = {
557  MDString::get(Ctx, "llvm.loop.max_concurrency.count"),
558  ConstantAsMetadata::get(ConstantInt::get(
559  llvm::Type::getInt32Ty(Ctx), *Attrs.SYCLMaxConcurrencyNThreads))};
560  LoopProperties.push_back(MDNode::get(Ctx, Vals));
561  }
562 
563  if (Attrs.SYCLLoopCoalesceEnable) {
564  Metadata *Vals[] = {MDString::get(Ctx, "llvm.loop.coalesce.enable")};
565  LoopProperties.push_back(MDNode::get(Ctx, Vals));
566  }
567 
568  if (Attrs.SYCLLoopCoalesceNLevels > 0) {
569  Metadata *Vals[] = {
570  MDString::get(Ctx, "llvm.loop.coalesce.count"),
571  ConstantAsMetadata::get(ConstantInt::get(
572  llvm::Type::getInt32Ty(Ctx), Attrs.SYCLLoopCoalesceNLevels))};
573  LoopProperties.push_back(MDNode::get(Ctx, Vals));
574  }
575 
576  // disable_loop_pipelining attribute corresponds to

696 void LoopInfo::finish() {
697  // We did not annotate the loop body instructions because there are no
698  // attributes for this loop.
699  if (!TempLoopID)
700  return;
701 
702  MDNode *LoopID;
703  LoopAttributes CurLoopAttr = Attrs;
704  LLVMContext &Ctx = Header->getContext();
705 
706  if (Parent && (Parent->Attrs.UnrollAndJamEnable ||
707  Parent->Attrs.UnrollAndJamCount != 0)) {
708  // Parent unroll-and-jams this loop.
709  // Split the transformations in those that happens before the unroll-and-jam
710  // and those after.
711 
712  LoopAttributes BeforeJam, AfterJam;
713 
714  BeforeJam.IsParallel = AfterJam.IsParallel = Attrs.IsParallel;
715 
716  BeforeJam.VectorizeWidth = Attrs.VectorizeWidth;
717  BeforeJam.VectorizeScalable = Attrs.VectorizeScalable;
718  BeforeJam.InterleaveCount = Attrs.InterleaveCount;
719  BeforeJam.VectorizeEnable = Attrs.VectorizeEnable;
720  BeforeJam.DistributeEnable = Attrs.DistributeEnable;
721  BeforeJam.VectorizePredicateEnable = Attrs.VectorizePredicateEnable;
722 
723  switch (Attrs.UnrollEnable) {
724  case LoopAttributes::Unspecified:
725  case LoopAttributes::Disable:
726  BeforeJam.UnrollEnable = Attrs.UnrollEnable;
727  AfterJam.UnrollEnable = Attrs.UnrollEnable;
728  break;
729  case LoopAttributes::Full:
730  BeforeJam.UnrollEnable = LoopAttributes::Full;
731  break;
732  case LoopAttributes::Enable:
733  AfterJam.UnrollEnable = LoopAttributes::Enable;
734  break;
735  }
736 
737  AfterJam.VectorizePredicateEnable = Attrs.VectorizePredicateEnable;
738  AfterJam.UnrollCount = Attrs.UnrollCount;
739  AfterJam.PipelineDisabled = Attrs.PipelineDisabled;
740  AfterJam.PipelineInitiationInterval = Attrs.PipelineInitiationInterval;
741 
742  // If this loop is subject of an unroll-and-jam by the parent loop, and has
743  // an unroll-and-jam annotation itself, we have to decide whether to first
744  // apply the parent's unroll-and-jam or this loop's unroll-and-jam. The
745  // UnrollAndJam pass processes loops from inner to outer, so we apply the
746  // inner first.
747  BeforeJam.UnrollAndJamCount = Attrs.UnrollAndJamCount;
748  BeforeJam.UnrollAndJamEnable = Attrs.UnrollAndJamEnable;
749 
750  // Set the inner followup metadata to process by the outer loop. Only
751  // consider the first inner loop.
752  if (!Parent->UnrollAndJamInnerFollowup) {
753  // Splitting the attributes into a BeforeJam and an AfterJam part will
754  // stop 'llvm.loop.isvectorized' (generated by vectorization in BeforeJam)
755  // to be forwarded to the AfterJam part. We detect the situation here and
756  // add it manually.
757  SmallVector<Metadata *, 1> BeforeLoopProperties;
758  if (BeforeJam.VectorizeEnable != LoopAttributes::Unspecified ||
759  BeforeJam.VectorizePredicateEnable != LoopAttributes::Unspecified ||
760  BeforeJam.InterleaveCount != 0 || BeforeJam.VectorizeWidth != 0 ||
761  BeforeJam.VectorizeScalable == LoopAttributes::Enable)
762  BeforeLoopProperties.push_back(
763  MDNode::get(Ctx, MDString::get(Ctx, "llvm.loop.isvectorized")));
764 
765  bool InnerFollowupHasTransform = false;
766  MDNode *InnerFollowup = createMetadata(AfterJam, BeforeLoopProperties,
767  InnerFollowupHasTransform);
768  if (InnerFollowupHasTransform)
769  Parent->UnrollAndJamInnerFollowup = InnerFollowup;
770  }
771 
772  CurLoopAttr = BeforeJam;
773  }
774 
775  bool HasUserTransforms = false;
776  LoopID = createMetadata(CurLoopAttr, {}, HasUserTransforms);
777  TempLoopID->replaceAllUsesWith(LoopID);
778 }

789 void LoopInfoStack::push(BasicBlock *Header, clang::ASTContext &Ctx,
790  const clang::CodeGenOptions &CGOpts,
791  ArrayRef<const clang::Attr *> Attrs,
792  const llvm::DebugLoc &StartLoc,
793  const llvm::DebugLoc &EndLoc, bool MustProgress) {
794  // Identify loop hint attributes from Attrs.
795  for (const auto *Attr : Attrs) {
796  const LoopHintAttr *LH = dyn_cast<LoopHintAttr>(Attr);
797  const OpenCLUnrollHintAttr *OpenCLHint =
798  dyn_cast<OpenCLUnrollHintAttr>(Attr);
799  const LoopUnrollHintAttr *UnrollHint = dyn_cast<LoopUnrollHintAttr>(Attr);
800 
801  // Skip non loop hint attributes
802  if (!LH && !OpenCLHint && !UnrollHint) {
803  continue;
804  }
805 
806  LoopHintAttr::OptionType Option = LoopHintAttr::Unroll;
807  LoopHintAttr::LoopHintState State = LoopHintAttr::Disable;
808  unsigned ValueInt = 1;
809  // Translate opencl_unroll_hint and clang::unroll attribute
810  // argument to equivalent LoopHintAttr enums.
811  // OpenCL v2.0 s6.11.5:
812  // 0 - enable unroll (no argument).
813  // 1 - disable unroll.
814  // other positive integer n - unroll by n.
815  if (OpenCLHint || UnrollHint) {
816  ValueInt = 0;
817  if (OpenCLHint)
818  ValueInt = OpenCLHint->getUnrollHint();
819  else if (Expr *E = UnrollHint->getUnrollHintExpr())
820  ValueInt = E->EvaluateKnownConstInt(Ctx).getSExtValue();
821 
822  if (ValueInt == 0) {
823  State = LoopHintAttr::Enable;
824  } else if (ValueInt != 1) {
825  Option = LoopHintAttr::UnrollCount;
826  State = LoopHintAttr::Numeric;
827  }
828  } else if (LH) {
829  auto *ValueExpr = LH->getValue();
830  if (ValueExpr) {
831  llvm::APSInt ValueAPS = ValueExpr->EvaluateKnownConstInt(Ctx);
832  ValueInt = ValueAPS.getSExtValue();
833  }
834 
835  Option = LH->getOption();
836  State = LH->getState();
837  }
838  switch (State) {
839  case LoopHintAttr::Disable:
840  switch (Option) {
841  case LoopHintAttr::Vectorize:
842  // Disable vectorization by specifying a width of 1.
843  setVectorizeWidth(1);
844  setVectorizeScalable(LoopAttributes::Unspecified);
845  break;
846  case LoopHintAttr::Interleave:
847  // Disable interleaving by speciyfing a count of 1.
848  setInterleaveCount(1);
849  break;
850  case LoopHintAttr::Unroll:
851  setUnrollState(LoopAttributes::Disable);
852  break;
853  case LoopHintAttr::UnrollAndJam:
854  setUnrollAndJamState(LoopAttributes::Disable);
855  break;
856  case LoopHintAttr::VectorizePredicate:
857  setVectorizePredicateState(LoopAttributes::Disable);
858  break;
859  case LoopHintAttr::Distribute:
860  setDistributeState(false);
861  break;
862  case LoopHintAttr::PipelineDisabled:
863  setPipelineDisabled(true);
864  break;
865  case LoopHintAttr::UnrollCount:
866  case LoopHintAttr::UnrollAndJamCount:
867  case LoopHintAttr::VectorizeWidth:
868  case LoopHintAttr::InterleaveCount:
869  case LoopHintAttr::PipelineInitiationInterval:
870  llvm_unreachable("Options cannot be disabled.");
871  break;
872  }
873  break;
874  case LoopHintAttr::Enable:
875  switch (Option) {
876  case LoopHintAttr::Vectorize:
877  case LoopHintAttr::Interleave:
878  setVectorizeEnable(true);
879  break;
880  case LoopHintAttr::Unroll:
881  setUnrollState(LoopAttributes::Enable);
882  break;
883  case LoopHintAttr::UnrollAndJam:
884  setUnrollAndJamState(LoopAttributes::Enable);
885  break;
886  case LoopHintAttr::VectorizePredicate:
887  setVectorizePredicateState(LoopAttributes::Enable);
888  break;
889  case LoopHintAttr::Distribute:
890  setDistributeState(true);
891  break;
892  case LoopHintAttr::UnrollCount:
893  case LoopHintAttr::UnrollAndJamCount:
894  case LoopHintAttr::VectorizeWidth:
895  case LoopHintAttr::InterleaveCount:
896  case LoopHintAttr::PipelineDisabled:
897  case LoopHintAttr::PipelineInitiationInterval:
898  llvm_unreachable("Options cannot enabled.");
899  break;
900  }
901  break;
902  case LoopHintAttr::AssumeSafety:
903  switch (Option) {
904  case LoopHintAttr::Vectorize:
905  case LoopHintAttr::Interleave:
906  // Apply "llvm.mem.parallel_loop_access" metadata to load/stores.
907  setParallel(true);
908  setVectorizeEnable(true);
909  break;
910  case LoopHintAttr::Unroll:
911  case LoopHintAttr::UnrollAndJam:
912  case LoopHintAttr::VectorizePredicate:
913  case LoopHintAttr::UnrollCount:
914  case LoopHintAttr::UnrollAndJamCount:
915  case LoopHintAttr::VectorizeWidth:
916  case LoopHintAttr::InterleaveCount:
917  case LoopHintAttr::Distribute:
918  case LoopHintAttr::PipelineDisabled:
919  case LoopHintAttr::PipelineInitiationInterval:
920  llvm_unreachable("Options cannot be used to assume mem safety.");
921  break;
922  }
923  break;
924  case LoopHintAttr::Full:
925  switch (Option) {
926  case LoopHintAttr::Unroll:
927  setUnrollState(LoopAttributes::Full);
928  break;
929  case LoopHintAttr::UnrollAndJam:
930  setUnrollAndJamState(LoopAttributes::Full);
931  break;
932  case LoopHintAttr::Vectorize:
933  case LoopHintAttr::Interleave:
934  case LoopHintAttr::UnrollCount:
935  case LoopHintAttr::UnrollAndJamCount:
936  case LoopHintAttr::VectorizeWidth:
937  case LoopHintAttr::InterleaveCount:
938  case LoopHintAttr::Distribute:
939  case LoopHintAttr::PipelineDisabled:
940  case LoopHintAttr::PipelineInitiationInterval:
941  case LoopHintAttr::VectorizePredicate:
942  llvm_unreachable("Options cannot be used with 'full' hint.");
943  break;
944  }
945  break;
946  case LoopHintAttr::FixedWidth:
947  case LoopHintAttr::ScalableWidth:
948  switch (Option) {
949  case LoopHintAttr::VectorizeWidth:
950  setVectorizeScalable(State == LoopHintAttr::ScalableWidth
951  ? LoopAttributes::Enable
952  : LoopAttributes::Disable);
953  if (LH->getValue())
954  setVectorizeWidth(ValueInt);
955  break;
956  default:
957  llvm_unreachable("Options cannot be used with 'scalable' hint.");
958  break;
959  }
960  break;
961  case LoopHintAttr::Numeric:
962  switch (Option) {
963  case LoopHintAttr::InterleaveCount:
964  setInterleaveCount(ValueInt);
965  break;
966  case LoopHintAttr::UnrollCount:
967  setUnrollCount(ValueInt);
968  break;
969  case LoopHintAttr::UnrollAndJamCount:
970  setUnrollAndJamCount(ValueInt);
971  break;
972  case LoopHintAttr::PipelineInitiationInterval:
973  setPipelineInitiationInterval(ValueInt);
974  break;
975  case LoopHintAttr::Unroll:
976  case LoopHintAttr::UnrollAndJam:
977  case LoopHintAttr::VectorizePredicate:
978  case LoopHintAttr::Vectorize:
979  case LoopHintAttr::VectorizeWidth:
980  case LoopHintAttr::Interleave:
981  case LoopHintAttr::Distribute:
982  case LoopHintAttr::PipelineDisabled:
983  llvm_unreachable("Options cannot be assigned a value.");
984  break;
985  }
986  break;
987  }
988  }
989 
990  // Translate intelfpga loop attributes' arguments to equivalent Attr enums.
991  // It's being handled separately from LoopHintAttrs not to support
992  // legacy GNU attributes and pragma styles.
993  //
994  // For attribute ivdep:
995  // Metadata 'llvm.loop.parallel_access_indices' & index group metadata
996  // will be emitted, depending on the conditions described at the
997  // helpers' site
998  // For attribute ii:
999  // n - 'llvm.loop.ii.count, i32 n' metadata will be emitted
1000  // For attribute max_concurrency:
1001  // n - 'llvm.loop.max_concurrency.count, i32 n' metadata will be emitted
1002  // For attribute loop_coalesce:
1003  // without parameter - 'lvm.loop.coalesce.enable' metadata will be emitted
1004  // n - 'llvm.loop.coalesce.count, i32 n' metadata will be emitted
1005  // For attribute disable_loop_pipelining:
1006  // 'llvm.loop.intel.pipelining.enable, i32 0' metadata will be emitted
1007  // For attribute max_interleaving:
1008  // n - 'llvm.loop.max_interleaving.count, i32 n' metadata will be emitted
1009  // For attribute speculated_iterations:
1010  // n - 'llvm.loop.intel.speculated.iterations.count, i32 n' metadata will be
1011  // emitted
1012  // For attribute nofusion:
1013  // 'llvm.loop.fusion.disable' metadata will be emitted
1014  for (const auto *A : Attrs) {
1015  if (const auto *IntelFPGAIVDep = dyn_cast<SYCLIntelFPGAIVDepAttr>(A))
1016  addSYCLIVDepInfo(Header->getContext(), IntelFPGAIVDep->getSafelenValue(),
1017  IntelFPGAIVDep->getArrayDecl());
1018 
1019  if (const auto *IntelFPGAII =
1020  dyn_cast<SYCLIntelFPGAInitiationIntervalAttr>(A)) {
1021  const auto *CE = cast<ConstantExpr>(IntelFPGAII->getIntervalExpr());
1022  llvm::APSInt ArgVal = CE->getResultAsAPSInt();
1023  setSYCLIInterval(ArgVal.getSExtValue());
1024  }
1025 
1026  if (const auto *IntelFPGAMaxConcurrency =
1027  dyn_cast<SYCLIntelFPGAMaxConcurrencyAttr>(A)) {
1028  const auto *CE =
1029  cast<ConstantExpr>(IntelFPGAMaxConcurrency->getNThreadsExpr());
1030  llvm::APSInt ArgVal = CE->getResultAsAPSInt();
1031  setSYCLMaxConcurrencyNThreads(ArgVal.getSExtValue());
1032  }
1033 
1034  if (const auto *IntelFPGALoopCountAvg =
1035  dyn_cast<SYCLIntelFPGALoopCountAttr>(A)) {
1036  const auto *CE =
1037  cast<ConstantExpr>(IntelFPGALoopCountAvg->getNTripCount());
1038  llvm::APSInt ArgVal = CE->getResultAsAPSInt();
1039  const char *Var =
1040  IntelFPGALoopCountAvg->isMax() ? "llvm.loop.intel.loopcount_max"
1041  : IntelFPGALoopCountAvg->isMin() ? "llvm.loop.intel.loopcount_min"
1042  : IntelFPGALoopCountAvg->isAvg() ? "llvm.loop.intel.loopcount_avg"
1043  : "llvm.loop.intel.loopcount";
1044  setSYCLIntelFPGAVariantCount(Var, ArgVal.getSExtValue());
1045  }
1046 
1047  if (const auto *IntelFPGALoopCoalesce =
1048  dyn_cast<SYCLIntelFPGALoopCoalesceAttr>(A)) {
1049  if (const auto *LCE = IntelFPGALoopCoalesce->getNExpr()) {
1050  const auto *CE = cast<ConstantExpr>(LCE);
1051  llvm::APSInt ArgVal = CE->getResultAsAPSInt();
1052  setSYCLLoopCoalesceNLevels(ArgVal.getSExtValue());
1053  } else {
1054  setSYCLLoopCoalesceEnable();
1055  }
1056  }
1057 
1058  if (isa<SYCLIntelFPGADisableLoopPipeliningAttr>(A))
1059  setSYCLLoopPipeliningDisable();
1060 
1061  if (const auto *IntelFPGAMaxInterleaving =
1062  dyn_cast<SYCLIntelFPGAMaxInterleavingAttr>(A)) {
1063  const auto *CE = cast<ConstantExpr>(IntelFPGAMaxInterleaving->getNExpr());
1064  llvm::APSInt ArgVal = CE->getResultAsAPSInt();
1065  setSYCLMaxInterleavingNInvocations(ArgVal.getSExtValue());
1066  }
1067 
1068  if (const auto *IntelFPGASpeculatedIterations =
1069  dyn_cast<SYCLIntelFPGASpeculatedIterationsAttr>(A)) {
1070  const auto *CE =
1071  cast<ConstantExpr>(IntelFPGASpeculatedIterations->getNExpr());
1072  llvm::APSInt ArgVal = CE->getResultAsAPSInt();
1073  setSYCLSpeculatedIterationsNIterations(ArgVal.getSExtValue());
1074  }
1075 
1076  if (isa<SYCLIntelFPGANofusionAttr>(A))
1077  setSYCLNofusionEnable();
1078  }
1079 
1080  setMustProgress(MustProgress);
1081 
1082  if (CGOpts.OptimizationLevel > 0)
1083  // Disable unrolling for the loop, if unrolling is disabled (via
1084  // -fno-unroll-loops) and no pragmas override the decision.
1085  if (!CGOpts.UnrollLoops &&
1086  (StagedAttrs.UnrollEnable == LoopAttributes::Unspecified &&
1087  StagedAttrs.UnrollCount == 0))
1088  setUnrollState(LoopAttributes::Disable);
1089 
1090  /// Stage the attributes.
1091  push(Header, StartLoc, EndLoc);
1092 }

1100 void LoopInfoStack::InsertHelper(Instruction *I) const {
1101  if (I->mayReadOrWriteMemory()) {
1102  SmallVector<Metadata *, 4> AccessGroups;
1103  for (const auto &AL : Active) {
1104  // Here we assume that every loop that has an access group is parallel.
1105  if (MDNode *Group = AL->getAccessGroup())
1106  AccessGroups.push_back(Group);
1107  }
1108  MDNode *UnionMD = nullptr;
1109  if (AccessGroups.size() == 1)
1110  UnionMD = cast<MDNode>(AccessGroups[0]);
1111  else if (AccessGroups.size() >= 2)
1112  UnionMD = MDNode::get(I->getContext(), AccessGroups);
1113  I->setMetadata("llvm.access.group", UnionMD);
1114  }
1115 
1116  if (!hasInfo())
1117  return;
1118 
1119  const LoopInfo &L = getInfo();
1120  if (!L.getLoopID())
1121  return;
1122 
1123  if (I->isTerminator()) {
1124  for (BasicBlock *Succ : successors(I))
1125  if (Succ == L.getHeader()) {
1126  I->setMetadata(llvm::LLVMContext::MD_loop, L.getLoopID());
1127  break;
1128  }
1129  return;
1130  }

1133 void LoopInfo::collectIVDepMetadata(
1134  const ValueDecl *Array, llvm::SmallVectorImpl<llvm::Metadata *> &MD) const {
1135  if (Parent)
1136  Parent->collectIVDepMetadata(Array, MD);
1137 
1138  auto ArrayIVDep =
1139  llvm::find_if(Attrs.ArraySYCLIVDepInfo,
1140  [Array](const auto &Info) { return Info.hasArray(Array); });
1141 
1142  // If this array is associated with an array, use this one.
1143  if (ArrayIVDep != Attrs.ArraySYCLIVDepInfo.end()) {
1144  MD.push_back(ArrayIVDep->getArrayPairItr(Array)->second);
1145  return;
1146  }
1147 
1148  if (!Attrs.GlobalSYCLIVDepInfo)
1149  return;
1150 
1151  auto GlobalArrayPairItr = Attrs.GlobalSYCLIVDepInfo->getArrayPairItr(Array);
1152  if (GlobalArrayPairItr == Attrs.GlobalSYCLIVDepInfo->Arrays.end()) {
