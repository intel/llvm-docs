263 static __inline int __get_cpuid_max (unsigned int __leaf, unsigned int *__sig)
264 {
265  unsigned int __eax, __ebx, __ecx, __edx;
266 #if __i386__
267  int __cpuid_supported;
268 
269  __asm(" pushfl\n"
270  " popl %%eax\n"
271  " movl %%eax,%%ecx\n"
272  " xorl $0x00200000,%%eax\n"
273  " pushl %%eax\n"
274  " popfl\n"
275  " pushfl\n"
276  " popl %%eax\n"
277  " movl $0,%0\n"
278  " cmpl %%eax,%%ecx\n"
279  " je 1f\n"
280  " movl $1,%0\n"
281  "1:"
282  : "=r" (__cpuid_supported) : : "eax", "ecx");
283  if (!__cpuid_supported)
284  return 0;
285 #endif
286 
287  __cpuid(__leaf, __eax, __ebx, __ecx, __edx);
288  if (__sig)
289  *__sig = __ebx;
290  return __eax;
291 }

294  unsigned int *__ebx, unsigned int *__ecx,
295  unsigned int *__edx)
296 {
