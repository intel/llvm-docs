239 static IVDepExprResult HandleFPGAIVDepAttrExpr(Sema &S, Expr *E,
240  unsigned &SafelenValue) {
241  if (!E)
242  return IVDepExprResult::Null;
243 
244  if (E->isInstantiationDependent())
245  return IVDepExprResult::Dependent;
246 
247  Optional<llvm::APSInt> ArgVal = E->getIntegerConstantExpr(S.getASTContext());
248  if (ArgVal) {
249  if (checkSYCLIntelFPGAIVDepSafeLen(S, *ArgVal, E))
250  return IVDepExprResult::Invalid;
251  SafelenValue = ArgVal->getZExtValue();
252  return IVDepExprResult::SafeLen;
253  }
254 
255  if (isa<DeclRefExpr>(E) || isa<MemberExpr>(E)) {
256  if (!E->getType()->isArrayType() && !E->getType()->isPointerType()) {
257  S.Diag(E->getExprLoc(), diag::err_ivdep_declrefexpr_arg);
258  return IVDepExprResult::Invalid;
259  }
260  return IVDepExprResult::Array;
261  }
262 
263  S.Diag(E->getExprLoc(), diag::err_ivdep_unknown_arg);
264  return IVDepExprResult::Invalid;
265 }

269 SYCLIntelFPGAIVDepAttr *
270 Sema::BuildSYCLIntelFPGAIVDepAttr(const AttributeCommonInfo &CI, Expr *Expr1,
271  Expr *Expr2) {
272  unsigned SafelenValue = 0;
273  IVDepExprResult E1 = HandleFPGAIVDepAttrExpr(*this, Expr1, SafelenValue);
274  IVDepExprResult E2 = HandleFPGAIVDepAttrExpr(*this, Expr2, SafelenValue);
275 
276  if (E1 == IVDepExprResult::Invalid || E2 == IVDepExprResult::Invalid)
277  return nullptr;
278 
279  if (E1 == E2 && E1 != IVDepExprResult::Dependent &&
280  E1 != IVDepExprResult::Null) {
281  Diag(Expr2->getExprLoc(), diag::err_ivdep_duplicate_arg);
282  return nullptr;
283  }
284 
285  // Try to put Safelen in the 1st one so codegen can count on the ordering.
286  Expr *SafeLenExpr;
287  Expr *ArrayExpr;
288  if (E1 == IVDepExprResult::SafeLen) {
289  SafeLenExpr = Expr1;
290  ArrayExpr = Expr2;
291  } else {
292  SafeLenExpr = Expr2;
293  ArrayExpr = Expr1;
294  }
295 
296  return new (Context)
297  SYCLIntelFPGAIVDepAttr(Context, CI, SafeLenExpr, ArrayExpr, SafelenValue);
298 }

300 // Filters out any attributes from the list that are either not the specified
301 // type, or whose function isDependent returns true.
302 template <typename T>
303 static void FilterAttributeList(ArrayRef<const Attr *> Attrs,
304  SmallVectorImpl<const T *> &FilteredAttrs) {
305 
306  llvm::transform(Attrs, std::back_inserter(FilteredAttrs),
307  [](const Attr *A) -> const T * {
308  if (const auto *Cast = dyn_cast<T>(A))
309  return Cast->isDependent() ? nullptr : Cast;
310  return nullptr;
311  });
312  FilteredAttrs.erase(
313  std::remove(FilteredAttrs.begin(), FilteredAttrs.end(), nullptr),
314  FilteredAttrs.end());
315 }

317 static void
318 CheckRedundantSYCLIntelFPGAIVDepAttrs(Sema &S, ArrayRef<const Attr *> Attrs) {
319  // Skip SEMA if we're in a template, this will be diagnosed later.
320  if (S.getCurLexicalContext()->isDependentContext())
321  return;
322 
323  SmallVector<const SYCLIntelFPGAIVDepAttr *, 8> FilteredAttrs;
324  // Filter down to just non-dependent ivdeps.
325  FilterAttributeList(Attrs, FilteredAttrs);
326  if (FilteredAttrs.empty())
327  return;
328 
329  SmallVector<const SYCLIntelFPGAIVDepAttr *, 8> SortedAttrs(FilteredAttrs);
330  llvm::stable_sort(SortedAttrs, SYCLIntelFPGAIVDepAttr::SafelenCompare);
331 
332  // Find the maximum without an array expression, which ends up in the 2nd
333  // expr.
334  const auto *GlobalMaxItr =
335  llvm::find_if(SortedAttrs, [](const SYCLIntelFPGAIVDepAttr *A) {
336  return !A->getArrayExpr();
337  });
338  const SYCLIntelFPGAIVDepAttr *GlobalMax =
339  GlobalMaxItr == SortedAttrs.end() ? nullptr : *GlobalMaxItr;
340 
341  for (const auto *A : FilteredAttrs) {
342  if (A == GlobalMax)
343  continue;
344 
345  if (GlobalMax && !SYCLIntelFPGAIVDepAttr::SafelenCompare(A, GlobalMax)) {
346  S.Diag(A->getLocation(), diag::warn_ivdep_redundant)
347  << !GlobalMax->isInf() << GlobalMax->getSafelenValue() << !A->isInf()
348  << A->getSafelenValue();
349  S.Diag(GlobalMax->getLocation(), diag::note_previous_attribute);
350  continue;
351  }
352 
353  if (!A->getArrayExpr())
354  continue;
355 
356  const ValueDecl *ArrayDecl = A->getArrayDecl();
357  auto Other = llvm::find_if(SortedAttrs,
358  [ArrayDecl](const SYCLIntelFPGAIVDepAttr *A) {
359  return ArrayDecl == A->getArrayDecl();
360  });
361  assert(Other != SortedAttrs.end() && "Should find at least itself");
362 
363  // Diagnose if lower/equal to the lowest with this array.
364  if (*Other != A && !SYCLIntelFPGAIVDepAttr::SafelenCompare(A, *Other)) {
365  S.Diag(A->getLocation(), diag::warn_ivdep_redundant)
366  << !(*Other)->isInf() << (*Other)->getSafelenValue() << !A->isInf()
367  << A->getSafelenValue();
368  S.Diag((*Other)->getLocation(), diag::note_previous_attribute);
369  }
370  }
371 }

469  LoopHintAttr::OptionType Option;
470  LoopHintAttr::LoopHintState State;
471 
472  auto SetHints = [&Option, &State](LoopHintAttr::OptionType O,
473  LoopHintAttr::LoopHintState S) {
474  Option = O;
475  State = S;
476  };
477 
478  if (PragmaName == "nounroll") {
479  SetHints(LoopHintAttr::Unroll, LoopHintAttr::Disable);
480  } else if (PragmaName == "unroll") {
481  // #pragma unroll N
482  if (ValueExpr)
483  SetHints(LoopHintAttr::UnrollCount, LoopHintAttr::Numeric);
484  else
485  SetHints(LoopHintAttr::Unroll, LoopHintAttr::Enable);
486  } else if (PragmaName == "nounroll_and_jam") {
487  SetHints(LoopHintAttr::UnrollAndJam, LoopHintAttr::Disable);
488  } else if (PragmaName == "unroll_and_jam") {
489  // #pragma unroll_and_jam N
490  if (ValueExpr)
491  SetHints(LoopHintAttr::UnrollAndJamCount, LoopHintAttr::Numeric);
492  else
493  SetHints(LoopHintAttr::UnrollAndJam, LoopHintAttr::Enable);
494  } else {
495  // #pragma clang loop ...
496  assert(OptionLoc && OptionLoc->Ident &&
497  "Attribute must have valid option info.");
498  Option = llvm::StringSwitch<LoopHintAttr::OptionType>(
499  OptionLoc->Ident->getName())
500  .Case("vectorize", LoopHintAttr::Vectorize)
501  .Case("vectorize_width", LoopHintAttr::VectorizeWidth)
502  .Case("interleave", LoopHintAttr::Interleave)
503  .Case("vectorize_predicate", LoopHintAttr::VectorizePredicate)
504  .Case("interleave_count", LoopHintAttr::InterleaveCount)
505  .Case("unroll", LoopHintAttr::Unroll)
506  .Case("unroll_count", LoopHintAttr::UnrollCount)
507  .Case("pipeline", LoopHintAttr::PipelineDisabled)
508  .Case("pipeline_initiation_interval",
509  LoopHintAttr::PipelineInitiationInterval)
510  .Case("distribute", LoopHintAttr::Distribute)
511  .Default(LoopHintAttr::Vectorize);
512  if (Option == LoopHintAttr::VectorizeWidth) {
513  assert((ValueExpr || (StateLoc && StateLoc->Ident)) &&
514  "Attribute must have a valid value expression or argument.");
515  if (ValueExpr && S.CheckLoopHintExpr(ValueExpr, St->getBeginLoc()))
516  return nullptr;
517  if (StateLoc && StateLoc->Ident && StateLoc->Ident->isStr("scalable"))
518  State = LoopHintAttr::ScalableWidth;
519  else
520  State = LoopHintAttr::FixedWidth;
521  } else if (Option == LoopHintAttr::InterleaveCount ||
522  Option == LoopHintAttr::UnrollCount ||
523  Option == LoopHintAttr::PipelineInitiationInterval) {
524  assert(ValueExpr && "Attribute must have a valid value expression.");
525  if (S.CheckLoopHintExpr(ValueExpr, St->getBeginLoc()))
526  return nullptr;
527  State = LoopHintAttr::Numeric;
528  } else if (Option == LoopHintAttr::Vectorize ||
529  Option == LoopHintAttr::Interleave ||
530  Option == LoopHintAttr::VectorizePredicate ||
531  Option == LoopHintAttr::Unroll ||
532  Option == LoopHintAttr::Distribute ||
533  Option == LoopHintAttr::PipelineDisabled) {
534  assert(StateLoc && StateLoc->Ident && "Loop hint must have an argument");
535  if (StateLoc->Ident->isStr("disable"))
536  State = LoopHintAttr::Disable;
537  else if (StateLoc->Ident->isStr("assume_safety"))
538  State = LoopHintAttr::AssumeSafety;
539  else if (StateLoc->Ident->isStr("full"))
540  State = LoopHintAttr::Full;
541  else if (StateLoc->Ident->isStr("enable"))
542  State = LoopHintAttr::Enable;
543  else
544  llvm_unreachable("bad loop hint argument");
545  } else
546  llvm_unreachable("bad loop hint");
547  }
548 
549  return LoopHintAttr::CreateImplicit(S.Context, Option, State, ValueExpr, A);
550 }

673 static void
674 CheckForIncompatibleAttributes(Sema &S,
675  const SmallVectorImpl<const Attr *> &Attrs) {
676  // The vast majority of attributed statements will only have one attribute
677  // on them, so skip all of the checking in the common case.
678  if (Attrs.size() < 2)
679  return;
680 
681  // First, check for the easy cases that are table-generated for us.
682  if (!DiagnoseMutualExclusions(S, Attrs))
683  return;
684 
685  // There are 6 categories of loop hints attributes: vectorize, interleave,
686  // unroll, unroll_and_jam, pipeline and distribute. Except for distribute they
687  // come in two variants: a state form and a numeric form. The state form
688  // selectively defaults/enables/disables the transformation for the loop
689  // (for unroll, default indicates full unrolling rather than enabling the
690  // transformation). The numeric form form provides an integer hint (for
691  // example, unroll count) to the transformer. The following array accumulates
692  // the hints encountered while iterating through the attributes to check for
693  // compatibility.
694  struct {
695  const LoopHintAttr *StateAttr;
696  const LoopHintAttr *NumericAttr;
697  } HintAttrs[] = {{nullptr, nullptr}, {nullptr, nullptr}, {nullptr, nullptr},
698  {nullptr, nullptr}, {nullptr, nullptr}, {nullptr, nullptr},
699  {nullptr, nullptr}};
700 
701  for (const auto *I : Attrs) {
702  const LoopHintAttr *LH = dyn_cast<LoopHintAttr>(I);
703 
704  // Skip non loop hint attributes
705  if (!LH)
706  continue;
707 
708  LoopHintAttr::OptionType Option = LH->getOption();
709  enum {
710  Vectorize,
711  Interleave,
712  Unroll,
713  UnrollAndJam,
714  Distribute,
715  Pipeline,
716  VectorizePredicate
717  } Category;
718  switch (Option) {
719  case LoopHintAttr::Vectorize:
720  case LoopHintAttr::VectorizeWidth:
721  Category = Vectorize;
722  break;
723  case LoopHintAttr::Interleave:
724  case LoopHintAttr::InterleaveCount:
725  Category = Interleave;
726  break;
727  case LoopHintAttr::Unroll:
728  case LoopHintAttr::UnrollCount:
729  Category = Unroll;
730  break;
731  case LoopHintAttr::UnrollAndJam:
732  case LoopHintAttr::UnrollAndJamCount:
733  Category = UnrollAndJam;
734  break;
735  case LoopHintAttr::Distribute:
736  // Perform the check for duplicated 'distribute' hints.
737  Category = Distribute;
738  break;
739  case LoopHintAttr::PipelineDisabled:
740  case LoopHintAttr::PipelineInitiationInterval:
741  Category = Pipeline;
742  break;
743  case LoopHintAttr::VectorizePredicate:
744  Category = VectorizePredicate;
745  break;
746  };
747 
748  assert(Category < sizeof(HintAttrs) / sizeof(HintAttrs[0]));
749  auto &CategoryState = HintAttrs[Category];
750  const LoopHintAttr *PrevAttr;
751  if (Option == LoopHintAttr::Vectorize ||
752  Option == LoopHintAttr::Interleave || Option == LoopHintAttr::Unroll ||
753  Option == LoopHintAttr::UnrollAndJam ||
754  Option == LoopHintAttr::VectorizePredicate ||
755  Option == LoopHintAttr::PipelineDisabled ||
756  Option == LoopHintAttr::Distribute) {
757  // Enable|Disable|AssumeSafety hint. For example, vectorize(enable).
758  PrevAttr = CategoryState.StateAttr;
759  CategoryState.StateAttr = LH;
760  } else {
761  // Numeric hint. For example, vectorize_width(8).
762  PrevAttr = CategoryState.NumericAttr;
763  CategoryState.NumericAttr = LH;
764  }
765 
766  PrintingPolicy Policy(S.Context.getLangOpts());
767  SourceLocation OptionLoc = LH->getRange().getBegin();
768  if (PrevAttr)
769  // Cannot specify same type of attribute twice.
770  S.Diag(OptionLoc, diag::err_pragma_loop_compatibility)
771  << /*Duplicate=*/true << PrevAttr->getDiagnosticName(Policy)
772  << LH->getDiagnosticName(Policy);
773 
774  if (CategoryState.StateAttr && CategoryState.NumericAttr &&
775  (Category == Unroll || Category == UnrollAndJam ||
776  CategoryState.StateAttr->getState() == LoopHintAttr::Disable)) {
777  // Disable hints are not compatible with numeric hints of the same
778  // category. As a special case, numeric unroll hints are also not
779  // compatible with enable or full form of the unroll pragma because these
780  // directives indicate full unrolling.
781  S.Diag(OptionLoc, diag::err_pragma_loop_compatibility)
782  << /*Duplicate=*/false
783  << CategoryState.StateAttr->getDiagnosticName(Policy)
784  << CategoryState.NumericAttr->getDiagnosticName(Policy);
785  }
786  }
787 }
