245 static IVDepExprResult HandleIVDepAttrExpr(Sema &S, Expr *E,
246  unsigned &SafelenValue) {
247  if (!E)
248  return IVDepExprResult::Null;
249 
250  if (E->isInstantiationDependent())
251  return IVDepExprResult::Dependent;
252 
253  std::optional<llvm::APSInt> ArgVal = E->getIntegerConstantExpr(S.getASTContext());
254  if (ArgVal) {
255  if (checkSYCLIntelIVDepSafeLen(S, *ArgVal, E))
256  return IVDepExprResult::Invalid;
257  SafelenValue = ArgVal->getZExtValue();
258  // ivdep attribute allows both safelen = 0 and safelen = 1 with a warning.
259  if (SafelenValue == 0 || SafelenValue == 1) {
260  S.Diag(E->getExprLoc(), diag::warn_ivdep_attribute_argument)
261  << SafelenValue;
262  return IVDepExprResult::Invalid;
263  }
264  return IVDepExprResult::SafeLen;
265  }
266 
267  if (isa<DeclRefExpr>(E) || isa<MemberExpr>(E)) {
268  if (!E->getType()->isArrayType() && !E->getType()->isPointerType()) {
269  S.Diag(E->getExprLoc(), diag::err_ivdep_declrefexpr_arg);
270  return IVDepExprResult::Invalid;
271  }
272  return IVDepExprResult::Array;
273  }
274 
275  S.Diag(E->getExprLoc(), diag::err_ivdep_unknown_arg);
276  return IVDepExprResult::Invalid;
277 }

281 SYCLIntelIVDepAttr *
282 Sema::BuildSYCLIntelIVDepAttr(const AttributeCommonInfo &CI, Expr *Expr1,
283  Expr *Expr2) {
284  unsigned SafelenValue = 0;
285  IVDepExprResult E1 = HandleIVDepAttrExpr(*this, Expr1, SafelenValue);
286  IVDepExprResult E2 = HandleIVDepAttrExpr(*this, Expr2, SafelenValue);
287 
288  if (E1 == IVDepExprResult::Invalid || E2 == IVDepExprResult::Invalid)
289  return nullptr;
290 
291  if (E1 == E2 && E1 != IVDepExprResult::Dependent &&
292  E1 != IVDepExprResult::Null) {
293  Diag(Expr2->getExprLoc(), diag::err_ivdep_duplicate_arg);
294  return nullptr;
295  }
296 
297  // Try to put Safelen in the 1st one so codegen can count on the ordering.
298  Expr *SafeLenExpr = Expr1;
299  Expr *ArrayExpr = Expr2;
300 
301  // Both can be null or dependent, so swap if we're really sure.
302  if (E2 == IVDepExprResult::SafeLen || E1 == IVDepExprResult::Array) {
303  SafeLenExpr = Expr2;
304  ArrayExpr = Expr1;
305  }
306 
307  return new (Context)
308  SYCLIntelIVDepAttr(Context, CI, SafeLenExpr, ArrayExpr, SafelenValue);
309 }

311 // Filters out any attributes from the list that are either not the specified
312 // type, or whose function isDependent returns true.
313 template <typename T>
314 static void FilterAttributeList(ArrayRef<const Attr *> Attrs,
315  SmallVectorImpl<const T *> &FilteredAttrs) {
316 
317  llvm::transform(Attrs, std::back_inserter(FilteredAttrs),
318  [](const Attr *A) -> const T * {
319  if (const auto *Cast = dyn_cast<T>(A))
320  return Cast->isDependent() ? nullptr : Cast;
321  return nullptr;
322  });
323  FilteredAttrs.erase(
324  std::remove(FilteredAttrs.begin(), FilteredAttrs.end(), nullptr),
325  FilteredAttrs.end());
326 }

328 static void
329 CheckRedundantSYCLIntelIVDepAttrs(Sema &S, ArrayRef<const Attr *> Attrs) {
330  // Skip SEMA if we're in a template, this will be diagnosed later.
331  if (S.getCurLexicalContext()->isDependentContext())
332  return;
333 
334  SmallVector<const SYCLIntelIVDepAttr *, 8> FilteredAttrs;
335  // Filter down to just non-dependent ivdeps.
336  FilterAttributeList(Attrs, FilteredAttrs);
337  if (FilteredAttrs.empty())
338  return;
339 
340  SmallVector<const SYCLIntelIVDepAttr *, 8> SortedAttrs(FilteredAttrs);
341  llvm::stable_sort(SortedAttrs, SYCLIntelIVDepAttr::SafelenCompare);
342 
343  // Find the maximum without an array expression, which ends up in the 2nd
344  // expr.
345  const auto *GlobalMaxItr =
346  llvm::find_if(SortedAttrs, [](const SYCLIntelIVDepAttr *A) {
347  return !A->getArrayExpr();
348  });
349  const SYCLIntelIVDepAttr *GlobalMax =
350  GlobalMaxItr == SortedAttrs.end() ? nullptr : *GlobalMaxItr;
351 
352  for (const auto *A : FilteredAttrs) {
353  if (A == GlobalMax)
354  continue;
355 
356  if (GlobalMax && !SYCLIntelIVDepAttr::SafelenCompare(A, GlobalMax)) {
357  S.Diag(A->getLocation(), diag::warn_ivdep_redundant)
358  << !GlobalMax->isInf() << GlobalMax->getSafelenValue() << !A->isInf()
359  << A->getSafelenValue();
360  S.Diag(GlobalMax->getLocation(), diag::note_previous_attribute);
361  continue;
362  }
363 
364  if (!A->getArrayExpr())
365  continue;
366 
367  const ValueDecl *ArrayDecl = A->getArrayDecl();
368  auto Other = llvm::find_if(SortedAttrs,
369  [ArrayDecl](const SYCLIntelIVDepAttr *A) {
370  return ArrayDecl == A->getArrayDecl();
371  });
372  assert(Other != SortedAttrs.end() && "Should find at least itself");
373 
374  // Diagnose if lower/equal to the lowest with this array.
375  if (*Other != A && !SYCLIntelIVDepAttr::SafelenCompare(A, *Other)) {
376  S.Diag(A->getLocation(), diag::warn_ivdep_redundant)
377  << !(*Other)->isInf() << (*Other)->getSafelenValue() << !A->isInf()
378  << A->getSafelenValue();
379  S.Diag((*Other)->getLocation(), diag::note_previous_attribute);
380  }
381  }
382 }

492 static Attr *handleLoopHintAttr(Sema &S, Stmt *St, const ParsedAttr &A,
493  SourceRange) {
494  IdentifierLoc *PragmaNameLoc = A.getArgAsIdent(0);
495  IdentifierLoc *OptionLoc = A.getArgAsIdent(1);
496  IdentifierLoc *StateLoc = A.getArgAsIdent(2);
497  Expr *ValueExpr = A.getArgAsExpr(3);
498 
499  StringRef PragmaName =
500  llvm::StringSwitch<StringRef>(PragmaNameLoc->Ident->getName())
501  .Cases("unroll", "nounroll", "unroll_and_jam", "nounroll_and_jam",
502  PragmaNameLoc->Ident->getName())
503  .Default("clang loop");
504 
505  // This could be handled automatically by adding a Subjects definition in
506  // Attr.td, but that would make the diagnostic behavior worse in this case
507  // because the user spells this attribute as a pragma.
508  if (!isa<DoStmt, ForStmt, CXXForRangeStmt, WhileStmt>(St)) {
509  std::string Pragma = "#pragma " + std::string(PragmaName);
510  S.Diag(St->getBeginLoc(), diag::err_pragma_loop_precedes_nonloop) << Pragma;
511  return nullptr;
512  }
513 
514  LoopHintAttr::OptionType Option;
515  LoopHintAttr::LoopHintState State;
516 
517  auto SetHints = [&Option, &State](LoopHintAttr::OptionType O,
518  LoopHintAttr::LoopHintState S) {
519  Option = O;
520  State = S;
521  };
522 
523  if (PragmaName == "nounroll") {
524  SetHints(LoopHintAttr::Unroll, LoopHintAttr::Disable);
525  } else if (PragmaName == "unroll") {
526  // #pragma unroll N
527  if (ValueExpr && !ValueExpr->isValueDependent()) {
528  llvm::APSInt ValueAPS;
529  ExprResult R = S.VerifyIntegerConstantExpression(ValueExpr, &ValueAPS);
530  assert(!R.isInvalid() && "unroll count value must be a valid value, it's "
531  "should be checked in Sema::CheckLoopHintExpr");
532  (void)R;
533  // The values of 0 and 1 block any unrolling of the loop.
534  if (ValueAPS.isZero() || ValueAPS.isOne())
535  SetHints(LoopHintAttr::UnrollCount, LoopHintAttr::Disable);
536  else
537  SetHints(LoopHintAttr::UnrollCount, LoopHintAttr::Numeric);
538  } else
539  SetHints(LoopHintAttr::Unroll, LoopHintAttr::Enable);
540  } else if (PragmaName == "nounroll_and_jam") {
541  SetHints(LoopHintAttr::UnrollAndJam, LoopHintAttr::Disable);
542  } else if (PragmaName == "unroll_and_jam") {
543  // #pragma unroll_and_jam N
544  if (ValueExpr)
545  SetHints(LoopHintAttr::UnrollAndJamCount, LoopHintAttr::Numeric);
546  else
547  SetHints(LoopHintAttr::UnrollAndJam, LoopHintAttr::Enable);
548  } else {
549  // #pragma clang loop ...
550  assert(OptionLoc && OptionLoc->Ident &&
551  "Attribute must have valid option info.");
552  Option = llvm::StringSwitch<LoopHintAttr::OptionType>(
553  OptionLoc->Ident->getName())
554  .Case("vectorize", LoopHintAttr::Vectorize)
555  .Case("vectorize_width", LoopHintAttr::VectorizeWidth)
556  .Case("interleave", LoopHintAttr::Interleave)
557  .Case("vectorize_predicate", LoopHintAttr::VectorizePredicate)
558  .Case("interleave_count", LoopHintAttr::InterleaveCount)
559  .Case("unroll", LoopHintAttr::Unroll)
560  .Case("unroll_count", LoopHintAttr::UnrollCount)
561  .Case("pipeline", LoopHintAttr::PipelineDisabled)
562  .Case("pipeline_initiation_interval",
563  LoopHintAttr::PipelineInitiationInterval)
564  .Case("distribute", LoopHintAttr::Distribute)
565  .Default(LoopHintAttr::Vectorize);
566  if (Option == LoopHintAttr::VectorizeWidth) {
567  assert((ValueExpr || (StateLoc && StateLoc->Ident)) &&
568  "Attribute must have a valid value expression or argument.");
569  if (ValueExpr && S.CheckLoopHintExpr(ValueExpr, St->getBeginLoc(),
570  /*AllowZero=*/false))
571  return nullptr;
572  if (StateLoc && StateLoc->Ident && StateLoc->Ident->isStr("scalable"))
573  State = LoopHintAttr::ScalableWidth;
574  else
575  State = LoopHintAttr::FixedWidth;
576  } else if (Option == LoopHintAttr::InterleaveCount ||
577  Option == LoopHintAttr::UnrollCount ||
578  Option == LoopHintAttr::PipelineInitiationInterval) {
579  assert(ValueExpr && "Attribute must have a valid value expression.");
580  if (S.CheckLoopHintExpr(ValueExpr, St->getBeginLoc(),
581  /*AllowZero=*/false))
582  return nullptr;
583  State = LoopHintAttr::Numeric;
584  } else if (Option == LoopHintAttr::Vectorize ||
585  Option == LoopHintAttr::Interleave ||
586  Option == LoopHintAttr::VectorizePredicate ||
587  Option == LoopHintAttr::Unroll ||
588  Option == LoopHintAttr::Distribute ||
589  Option == LoopHintAttr::PipelineDisabled) {
590  assert(StateLoc && StateLoc->Ident && "Loop hint must have an argument");
591  if (StateLoc->Ident->isStr("disable"))
592  State = LoopHintAttr::Disable;
593  else if (StateLoc->Ident->isStr("assume_safety"))
594  State = LoopHintAttr::AssumeSafety;
595  else if (StateLoc->Ident->isStr("full"))
596  State = LoopHintAttr::Full;
597  else if (StateLoc->Ident->isStr("enable"))

649 template <typename OtherAttr, int DiagIdx>
650 static bool CheckStmtInlineAttr(Sema &SemaRef, const Stmt *OrigSt,
651  const Stmt *CurSt,
652  const AttributeCommonInfo &A) {
653  CallExprFinder OrigCEF(SemaRef, OrigSt);
654  CallExprFinder CEF(SemaRef, CurSt);
655 
656  // If the call expressions lists are equal in size, we can skip
657  // previously emitted diagnostics. However, if the statement has a pack
658  // expansion, we have no way of telling which CallExpr is the instantiated
659  // version of the other. In this case, we will end up re-diagnosing in the
660  // instantiation.
661  // ie: [[clang::always_inline]] non_dependent(), (other_call<Pack>()...)
662  // will diagnose nondependent again.
663  bool CanSuppressDiag =
664  OrigSt && CEF.getCallExprs().size() == OrigCEF.getCallExprs().size();

666  if (!CEF.foundCallExpr()) {
667  return SemaRef.Diag(CurSt->getBeginLoc(),
668  diag::warn_attribute_ignored_no_calls_in_stmt)
669  << A;
670  }
671 
672  for (const auto &Tup :
673  llvm::zip_longest(OrigCEF.getCallExprs(), CEF.getCallExprs())) {
674  // If the original call expression already had a callee, we already
675  // diagnosed this, so skip it here. We can't skip if there isn't a 1:1
676  // relationship between the two lists of call expressions.
677  if (!CanSuppressDiag || !(*std::get<0>(Tup))->getCalleeDecl()) {
678  const Decl *Callee = (*std::get<1>(Tup))->getCalleeDecl();
679  if (Callee &&
680  (Callee->hasAttr<OtherAttr>() || Callee->hasAttr<FlattenAttr>())) {
681  SemaRef.Diag(CurSt->getBeginLoc(),
682  diag::warn_function_stmt_attribute_precedence)
683  << A << (Callee->hasAttr<OtherAttr>() ? DiagIdx : 1);
684  SemaRef.Diag(Callee->getBeginLoc(), diag::note_conflicting_attribute);
685  }
686  }
687  }
688 
689  return false;

865  enum CategoryType {
866  // For the following categories, they come in two variants: a state form and
867  // a numeric form. The state form may be one of default, enable, and
868  // disable. The numeric form provides an integer hint (for example, unroll
869  // count) to the transformer.
870  Vectorize,
871  Interleave,
872  UnrollAndJam,
873  Pipeline,
874  // For unroll, default indicates full unrolling rather than enabling the
875  // transformation.
876  Unroll,
877  // The loop distribution transformation only has a state form that is
878  // exposed by #pragma clang loop distribute (enable | disable).
879  Distribute,
880  // The vector predication only has a state form that is exposed by
881  // #pragma clang loop vectorize_predicate (enable | disable).
882  VectorizePredicate,
883  // This serves as a indicator to how many category are listed in this enum.
884  NumberOfCategories
885  };
886  // The following array accumulates the hints encountered while iterating
887  // through the attributes to check for compatibility.
888  struct {
889  const LoopHintAttr *StateAttr;
890  const LoopHintAttr *NumericAttr;
891  } HintAttrs[CategoryType::NumberOfCategories] = {};
892 
893  for (const auto *I : Attrs) {
894  const LoopHintAttr *LH = dyn_cast<LoopHintAttr>(I);
895 
896  // Skip non loop hint attributes
897  if (!LH)
898  continue;
899 
900  CategoryType Category = CategoryType::NumberOfCategories;
901  LoopHintAttr::OptionType Option = LH->getOption();
902  switch (Option) {
903  case LoopHintAttr::Vectorize:
904  case LoopHintAttr::VectorizeWidth:
905  Category = Vectorize;
906  break;
907  case LoopHintAttr::Interleave:
908  case LoopHintAttr::InterleaveCount:
909  Category = Interleave;
910  break;
911  case LoopHintAttr::Unroll:
912  case LoopHintAttr::UnrollCount:
913  Category = Unroll;
914  break;
915  case LoopHintAttr::UnrollAndJam:
916  case LoopHintAttr::UnrollAndJamCount:
917  Category = UnrollAndJam;
918  break;
919  case LoopHintAttr::Distribute:
920  // Perform the check for duplicated 'distribute' hints.
921  Category = Distribute;
922  break;
923  case LoopHintAttr::PipelineDisabled:
924  case LoopHintAttr::PipelineInitiationInterval:
925  Category = Pipeline;
926  break;
927  case LoopHintAttr::VectorizePredicate:
928  Category = VectorizePredicate;
929  break;
930  };
931 
932  assert(Category != NumberOfCategories && "Unhandled loop hint option");
933  auto &CategoryState = HintAttrs[Category];
934  const LoopHintAttr *PrevAttr;
935  if (Option == LoopHintAttr::Vectorize ||
936  Option == LoopHintAttr::Interleave || Option == LoopHintAttr::Unroll ||
937  Option == LoopHintAttr::UnrollAndJam ||
938  Option == LoopHintAttr::VectorizePredicate ||
939  Option == LoopHintAttr::PipelineDisabled ||
940  Option == LoopHintAttr::Distribute) {
941  // Enable|Disable|AssumeSafety hint. For example, vectorize(enable).
942  PrevAttr = CategoryState.StateAttr;
943  CategoryState.StateAttr = LH;
944  } else {
945  // Numeric hint. For example, vectorize_width(8).
946  PrevAttr = CategoryState.NumericAttr;
947  CategoryState.NumericAttr = LH;
948  }
949 
950  PrintingPolicy Policy(S.Context.getLangOpts());
951  SourceLocation OptionLoc = LH->getRange().getBegin();
952  if (PrevAttr)
953  // Cannot specify same type of attribute twice.
954  S.Diag(OptionLoc, diag::err_pragma_loop_compatibility)
955  << /*Duplicate=*/true << PrevAttr->getDiagnosticName(Policy)
956  << LH->getDiagnosticName(Policy);
957 
958  if (CategoryState.StateAttr && CategoryState.NumericAttr &&
959  (Category == Unroll || Category == UnrollAndJam ||
960  CategoryState.StateAttr->getState() == LoopHintAttr::Disable)) {
961  // Disable hints are not compatible with numeric hints of the same
962  // category. As a special case, numeric unroll hints are also not
963  // compatible with enable or full form of the unroll pragma because these
964  // directives indicate full unrolling.
965  S.Diag(OptionLoc, diag::err_pragma_loop_compatibility)
966  << /*Duplicate=*/false
967  << CategoryState.StateAttr->getDiagnosticName(Policy)
968  << CategoryState.NumericAttr->getDiagnosticName(Policy);
969  }
970  }
971 }
