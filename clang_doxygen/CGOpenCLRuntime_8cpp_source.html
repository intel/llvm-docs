164 // Get the block literal from an expression derived from the block expression.
165 // OpenCL v2.0 s6.12.5:
166 // Block variable declarations are implicitly qualified with const. Therefore
167 // all block variables must be initialized at declaration time and may not be
168 // reassigned.
169 static const BlockExpr *getBlockExpr(const Expr *E) {
170  const Expr *Prev = nullptr; // to make sure we do not stuck in infinite loop.
171  while(!isa<BlockExpr>(E) && E != Prev) {
172  Prev = E;
173  E = E->IgnoreCasts();
174  if (auto DR = dyn_cast<DeclRefExpr>(E)) {
175  E = cast<VarDecl>(DR->getDecl())->getInit();
176  }
177  }
178  return cast<BlockExpr>(E);
179 }

183 void CGOpenCLRuntime::recordBlockInfo(const BlockExpr *E,
184  llvm::Function *InvokeF,
185  llvm::Value *Block, llvm::Type *BlockTy) {
186  assert(EnqueuedBlockMap.find(E) == EnqueuedBlockMap.end() &&
187  "Block expression emitted twice");
188  assert(isa<llvm::Function>(InvokeF) && "Invalid invoke function");
189  assert(Block->getType()->isPointerTy() && "Invalid block literal type");
190  EnqueuedBlockMap[E].InvokeFunc = InvokeF;
191  EnqueuedBlockMap[E].BlockArg = Block;
192  EnqueuedBlockMap[E].BlockTy = BlockTy;
193  EnqueuedBlockMap[E].Kernel = nullptr;
194 }

200 CGOpenCLRuntime::EnqueuedBlockInfo
201 CGOpenCLRuntime::emitOpenCLEnqueuedBlock(CodeGenFunction &CGF, const Expr *E) {
202  CGF.EmitScalarExpr(E);
203 
204  // The block literal may be assigned to a const variable. Chasing down
205  // to get the block literal.
206  const BlockExpr *Block = getBlockExpr(E);
207 
208  assert(EnqueuedBlockMap.find(Block) != EnqueuedBlockMap.end() &&
209  "Block expression not emitted");
210 
211  // Do not emit the block wrapper again if it has been emitted.
212  if (EnqueuedBlockMap[Block].Kernel) {
213  return EnqueuedBlockMap[Block];
214  }
215 
216  auto *F = CGF.getTargetHooks().createEnqueuedBlockKernel(
217  CGF, EnqueuedBlockMap[Block].InvokeFunc, EnqueuedBlockMap[Block].BlockTy);
218 
219  // The common part of the post-processing of the kernel goes here.
220  F->addFnAttr(llvm::Attribute::NoUnwind);
221  F->setCallingConv(
222  CGF.getTypes().ClangCallConvToLLVMCallConv(CallingConv::CC_OpenCLKernel));
223  EnqueuedBlockMap[Block].Kernel = F;
224  return EnqueuedBlockMap[Block];
225 }
