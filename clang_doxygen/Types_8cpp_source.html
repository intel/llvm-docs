23 struct TypeInfo {

27  class PhasesBitSet {
28  unsigned Bits = 0;
29 
30  public:
31  constexpr PhasesBitSet(std::initializer_list<phases::ID> Phases) {
32  for (auto Id : Phases)
33  Bits |= 1 << Id;
34  }
35  bool contains(phases::ID Id) const { return Bits & (1 << Id); }

364 types::ID types::lookupTypeForTypeSpecifier(const char *Name) {
365  for (unsigned i=0; i<numTypes; ++i) {
366  types::ID Id = (types::ID) (i + 1);
367  if (canTypeBeUserSpecified(Id) &&
368  strcmp(Name, getInfo(Id).Name) == 0)
369  return Id;
370  }
371  // Accept "cu" as an alias for "cuda" for NVCC compatibility
372  if (strcmp(Name, "cu") == 0) {
373  return types::TY_CUDA;
374  }
375  return TY_INVALID;
376 }

378 llvm::SmallVector<phases::ID, phases::MaxNumberOfPhases>
379 types::getCompilationPhases(ID Id, phases::ID LastPhase) {
380  llvm::SmallVector<phases::ID, phases::MaxNumberOfPhases> P;
381  const auto &Info = getInfo(Id);
382  for (int I = 0; I <= LastPhase; ++I)
383  if (Info.Phases.contains(static_cast<phases::ID>(I)))
384  P.push_back(static_cast<phases::ID>(I));
385  assert(P.size() <= phases::MaxNumberOfPhases && "Too many phases in list");
386  return P;
387 }
