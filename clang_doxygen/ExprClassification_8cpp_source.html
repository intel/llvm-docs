37 Cl Expr::ClassifyImpl(ASTContext &Ctx, SourceLocation *Loc) const {
38  assert(!TR->isReferenceType() && "Expressions can't have reference type.");
39 
40  Cl::Kinds kind = ClassifyInternal(Ctx, this);
41  // C99 6.3.2.1: An lvalue is an expression with an object type or an
42  // incomplete type other than void.
43  if (!Ctx.getLangOpts().CPlusPlus) {
44  // Thus, no functions.
45  if (TR->isFunctionType() || TR == Ctx.OverloadTy)
46  kind = Cl::CL_Function;
47  // No void either, but qualified void is OK because it is "other than void".
48  // Void "lvalues" are classified as addressable void values, which are void
49  // expressions whose address can be taken.
50  else if (TR->isVoidType() && !TR.hasQualifiers())
51  kind = (kind == Cl::CL_LValue ? Cl::CL_AddressableVoid : Cl::CL_Void);
52  }
53 
54  // Enable this assertion for testing.
55  switch (kind) {
56  case Cl::CL_LValue:
57  assert(isLValue());
58  break;
59  case Cl::CL_XValue:
60  assert(isXValue());
61  break;
62  case Cl::CL_Function:
63  case Cl::CL_Void:
64  case Cl::CL_AddressableVoid:
65  case Cl::CL_DuplicateVectorComponents:
66  case Cl::CL_MemberFunction:
67  case Cl::CL_SubObjCPropertySetting:
68  case Cl::CL_ClassTemporary:
69  case Cl::CL_ArrayTemporary:
70  case Cl::CL_ObjCMessageRValue:
71  case Cl::CL_PRValue:
72  assert(isPRValue());
73  break;
74  }
75 
76  Cl::ModifiableType modifiable = Cl::CM_Untested;
77  if (Loc)
78  modifiable = IsModifiable(Ctx, this, kind, *Loc);
79  return Classification(kind, modifiable);
80 }

82 /// Classify an expression which creates a temporary, based on its type.
83 static Cl::Kinds ClassifyTemporary(QualType T) {
84  if (T->isRecordType())
85  return Cl::CL_ClassTemporary;
86  if (T->isArrayType())
87  return Cl::CL_ArrayTemporary;

90  // prvalues.
91  return Cl::CL_PRValue;
92 }

94 static Cl::Kinds ClassifyExprValueKind(const LangOptions &Lang,
95  const Expr *E,
96  ExprValueKind Kind) {
97  switch (Kind) {
98  case VK_PRValue:
99  return Lang.CPlusPlus ? ClassifyTemporary(E->getType()) : Cl::CL_PRValue;
100  case VK_LValue:

108 static Cl::Kinds ClassifyInternal(ASTContext &Ctx, const Expr *E) {
109  // This function takes the first stab at classifying expressions.
110  const LangOptions &Lang = Ctx.getLangOpts();
111 
112  switch (E->getStmtClass()) {
113  case Stmt::NoStmtClass:
114 #define ABSTRACT_STMT(Kind)
115 #define STMT(Kind, Base) case Expr::Kind##Class:
116 #define EXPR(Kind, Base)
117 #include "clang/AST/StmtNodes.inc"
118  llvm_unreachable("cannot classify a statement");
119 
120  // First come the expressions that are always lvalues, unconditionally.
121  case Expr::ObjCIsaExprClass:
122  // C++ [expr.prim.general]p1: A string literal is an lvalue.
123  case Expr::StringLiteralClass:
124  // @encode is equivalent to its string
125  case Expr::ObjCEncodeExprClass:
126  // __func__ and friends are too.
127  case Expr::PredefinedExprClass:
128  // Property references are lvalues
129  case Expr::ObjCSubscriptRefExprClass:
130  case Expr::ObjCPropertyRefExprClass:
131  // C++ [expr.typeid]p1: The result of a typeid expression is an lvalue of...
132  case Expr::CXXTypeidExprClass:
133  case Expr::CXXUuidofExprClass:
134  // Unresolved lookups and uncorrected typos get classified as lvalues.
135  // FIXME: Is this wise? Should they get their own kind?
136  case Expr::UnresolvedLookupExprClass:
137  case Expr::UnresolvedMemberExprClass:
138  case Expr::TypoExprClass:
139  case Expr::DependentCoawaitExprClass:
140  case Expr::CXXDependentScopeMemberExprClass:
141  case Expr::DependentScopeDeclRefExprClass:
142  // ObjC instance variables are lvalues
143  // FIXME: ObjC++0x might have different rules
144  case Expr::ObjCIvarRefExprClass:
145  case Expr::FunctionParmPackExprClass:
146  case Expr::MSPropertyRefExprClass:
147  case Expr::MSPropertySubscriptExprClass:
148  case Expr::OMPArraySectionExprClass:
149  case Expr::OMPArrayShapingExprClass:
150  case Expr::OMPIteratorExprClass:
151  return Cl::CL_LValue;
152 
153  // C99 6.5.2.5p5 says that compound literals are lvalues.
154  // In C++, they're prvalue temporaries, except for file-scope arrays.
155  case Expr::CompoundLiteralExprClass:
156  return !E->isLValue() ? ClassifyTemporary(E->getType()) : Cl::CL_LValue;
157 
158  // Expressions that are prvalues.
159  case Expr::CXXBoolLiteralExprClass:
160  case Expr::CXXPseudoDestructorExprClass:
161  case Expr::UnaryExprOrTypeTraitExprClass:
162  case Expr::CXXNewExprClass:
163  case Expr::CXXNullPtrLiteralExprClass:
164  case Expr::ImaginaryLiteralClass:
165  case Expr::GNUNullExprClass:
166  case Expr::OffsetOfExprClass:
167  case Expr::CXXThrowExprClass:
168  case Expr::ShuffleVectorExprClass:
169  case Expr::ConvertVectorExprClass:
170  case Expr::IntegerLiteralClass:
171  case Expr::FixedPointLiteralClass:
172  case Expr::CharacterLiteralClass:
173  case Expr::AddrLabelExprClass:
174  case Expr::CXXDeleteExprClass:
175  case Expr::ImplicitValueInitExprClass:
176  case Expr::BlockExprClass:
177  case Expr::FloatingLiteralClass:
178  case Expr::CXXNoexceptExprClass:
179  case Expr::CXXScalarValueInitExprClass:
180  case Expr::TypeTraitExprClass:
181  case Expr::ArrayTypeTraitExprClass:
182  case Expr::ExpressionTraitExprClass:
183  case Expr::ObjCSelectorExprClass:
184  case Expr::ObjCProtocolExprClass:
185  case Expr::ObjCStringLiteralClass:
186  case Expr::ObjCBoxedExprClass:
187  case Expr::ObjCArrayLiteralClass:
188  case Expr::ObjCDictionaryLiteralClass:
189  case Expr::ObjCBoolLiteralExprClass:
190  case Expr::ObjCAvailabilityCheckExprClass:
191  case Expr::ParenListExprClass:
192  case Expr::SizeOfPackExprClass:
193  case Expr::SubstNonTypeTemplateParmPackExprClass:
194  case Expr::AsTypeExprClass:
195  case Expr::ObjCIndirectCopyRestoreExprClass:
196  case Expr::AtomicExprClass:
197  case Expr::CXXFoldExprClass:
198  case Expr::ArrayInitLoopExprClass:
199  case Expr::ArrayInitIndexExprClass:
200  case Expr::NoInitExprClass:
201  case Expr::DesignatedInitUpdateExprClass:
202  case Expr::SourceLocExprClass:
203  case Expr::ConceptSpecializationExprClass:
204  case Expr::RequiresExprClass:
205  case Expr::SYCLBuiltinNumFieldsExprClass:
206  case Expr::SYCLBuiltinFieldTypeExprClass:
207  case Expr::SYCLBuiltinNumBasesExprClass:
208  case Expr::SYCLBuiltinBaseTypeExprClass:
209  return Cl::CL_PRValue;
210 
211  // Make HLSL this reference-like
212  case Expr::CXXThisExprClass:
213  return Lang.HLSL ? Cl::CL_LValue : Cl::CL_PRValue;
214 
215  case Expr::ConstantExprClass:
216  return ClassifyInternal(Ctx, cast<ConstantExpr>(E)->getSubExpr());
217 
218  // Next come the complicated cases.
219  case Expr::SubstNonTypeTemplateParmExprClass:
220  return ClassifyInternal(Ctx,
221  cast<SubstNonTypeTemplateParmExpr>(E)->getReplacement());
222 
223  case Expr::PackIndexingExprClass:
224  return ClassifyInternal(Ctx, cast<PackIndexingExpr>(E)->getSelectedExpr());
225 
226  // C, C++98 [expr.sub]p1: The result is an lvalue of type "T".
227  // C++11 (DR1213): in the case of an array operand, the result is an lvalue
228  // if that operand is an lvalue and an xvalue otherwise.
229  // Subscripting vector types is more like member access.
230  case Expr::ArraySubscriptExprClass:
231  if (cast<ArraySubscriptExpr>(E)->getBase()->getType()->isVectorType())
232  return ClassifyInternal(Ctx, cast<ArraySubscriptExpr>(E)->getBase());
233  if (Lang.CPlusPlus11) {
234  // Step over the array-to-pointer decay if present, but not over the
235  // temporary materialization.
236  auto *Base = cast<ArraySubscriptExpr>(E)->getBase()->IgnoreImpCasts();
237  if (Base->getType()->isArrayType())
238  return ClassifyInternal(Ctx, Base);
239  }
240  return Cl::CL_LValue;
241 
242  // Subscripting matrix types behaves like member accesses.
243  case Expr::MatrixSubscriptExprClass:
244  return ClassifyInternal(Ctx, cast<MatrixSubscriptExpr>(E)->getBase());
245 
246  // C++ [expr.prim.general]p3: The result is an lvalue if the entity is a
247  // function or variable and a prvalue otherwise.
248  case Expr::DeclRefExprClass:
249  if (E->getType() == Ctx.UnknownAnyTy)
250  return isa<FunctionDecl>(cast<DeclRefExpr>(E)->getDecl())
251  ? Cl::CL_PRValue : Cl::CL_LValue;
252  return ClassifyDecl(Ctx, cast<DeclRefExpr>(E)->getDecl());
253 
254  // Member access is complex.
255  case Expr::MemberExprClass:
256  return ClassifyMemberExpr(Ctx, cast<MemberExpr>(E));
257 
258  case Expr::UnaryOperatorClass:
259  switch (cast<UnaryOperator>(E)->getOpcode()) {
260  // C++ [expr.unary.op]p1: The unary * operator performs indirection:
261  // [...] the result is an lvalue referring to the object or function
262  // to which the expression points.
263  case UO_Deref:
264  return Cl::CL_LValue;
265 
266  // GNU extensions, simply look through them.
267  case UO_Extension:
268  return ClassifyInternal(Ctx, cast<UnaryOperator>(E)->getSubExpr());
269 
270  // Treat _Real and _Imag basically as if they were member
271  // expressions: l-value only if the operand is a true l-value.
272  case UO_Real:
273  case UO_Imag: {
274  const Expr *Op = cast<UnaryOperator>(E)->getSubExpr()->IgnoreParens();
275  Cl::Kinds K = ClassifyInternal(Ctx, Op);
276  if (K != Cl::CL_LValue) return K;
277 
278  if (isa<ObjCPropertyRefExpr>(Op))
279  return Cl::CL_SubObjCPropertySetting;
280  return Cl::CL_LValue;
281  }
282 
283  // C++ [expr.pre.incr]p1: The result is the updated operand; it is an
284  // lvalue, [...]
285  // Not so in C.
286  case UO_PreInc:
287  case UO_PreDec:
288  return Lang.CPlusPlus ? Cl::CL_LValue : Cl::CL_PRValue;
289 
290  default:
291  return Cl::CL_PRValue;
292  }
293 
294  case Expr::RecoveryExprClass:
295  case Expr::OpaqueValueExprClass:
296  return ClassifyExprValueKind(Lang, E, E->getValueKind());
297 
298  // Pseudo-object expressions can produce l-values with reference magic.
299  case Expr::PseudoObjectExprClass:
300  return ClassifyExprValueKind(Lang, E,
301  cast<PseudoObjectExpr>(E)->getValueKind());
302 
303  // Implicit casts are lvalues if they're lvalue casts. Other than that, we
304  // only specifically record class temporaries.
305  case Expr::ImplicitCastExprClass:
306  return ClassifyExprValueKind(Lang, E, E->getValueKind());
307 
308  // C++ [expr.prim.general]p4: The presence of parentheses does not affect
309  // whether the expression is an lvalue.
310  case Expr::ParenExprClass:
311  return ClassifyInternal(Ctx, cast<ParenExpr>(E)->getSubExpr());
312 
313  // C11 6.5.1.1p4: [A generic selection] is an lvalue, a function designator,
314  // or a void expression if its result expression is, respectively, an
315  // lvalue, a function designator, or a void expression.
316  case Expr::GenericSelectionExprClass:
317  if (cast<GenericSelectionExpr>(E)->isResultDependent())
318  return Cl::CL_PRValue;
319  return ClassifyInternal(Ctx,cast<GenericSelectionExpr>(E)->getResultExpr());
320 
321  case Expr::BinaryOperatorClass:
322  case Expr::CompoundAssignOperatorClass:
323  // C doesn't have any binary expressions that are lvalues.
324  if (Lang.CPlusPlus)
325  return ClassifyBinaryOp(Ctx, cast<BinaryOperator>(E));
326  return Cl::CL_PRValue;
327 
328  case Expr::CallExprClass:
329  case Expr::CXXOperatorCallExprClass:
330  case Expr::CXXMemberCallExprClass:
331  case Expr::UserDefinedLiteralClass:
332  case Expr::CUDAKernelCallExprClass:
333  return ClassifyUnnamed(Ctx, cast<CallExpr>(E)->getCallReturnType(Ctx));
334 
335  case Expr::CXXRewrittenBinaryOperatorClass:
336  return ClassifyInternal(
337  Ctx, cast<CXXRewrittenBinaryOperator>(E)->getSemanticForm());
338 
339  // __builtin_choose_expr is equivalent to the chosen expression.
340  case Expr::ChooseExprClass:
341  return ClassifyInternal(Ctx, cast<ChooseExpr>(E)->getChosenSubExpr());
342 
343  // Extended vector element access is an lvalue unless there are duplicates
344  // in the shuffle expression.
345  case Expr::ExtVectorElementExprClass:
346  if (cast<ExtVectorElementExpr>(E)->containsDuplicateElements())
347  return Cl::CL_DuplicateVectorComponents;
348  if (cast<ExtVectorElementExpr>(E)->isArrow())
349  return Cl::CL_LValue;
350  return ClassifyInternal(Ctx, cast<ExtVectorElementExpr>(E)->getBase());
351 
352  // Simply look at the actual default argument.
353  case Expr::CXXDefaultArgExprClass:
354  return ClassifyInternal(Ctx, cast<CXXDefaultArgExpr>(E)->getExpr());
355 
356  // Same idea for default initializers.
357  case Expr::CXXDefaultInitExprClass:
358  return ClassifyInternal(Ctx, cast<CXXDefaultInitExpr>(E)->getExpr());
359 
360  // Same idea for temporary binding.
361  case Expr::CXXBindTemporaryExprClass:
362  return ClassifyInternal(Ctx, cast<CXXBindTemporaryExpr>(E)->getSubExpr());
363 
364  // And the cleanups guard.
365  case Expr::ExprWithCleanupsClass:
366  return ClassifyInternal(Ctx, cast<ExprWithCleanups>(E)->getSubExpr());
367 
368  // Casts depend completely on the target type. All casts work the same.
369  case Expr::CStyleCastExprClass:
370  case Expr::CXXFunctionalCastExprClass:
371  case Expr::CXXStaticCastExprClass:
372  case Expr::CXXDynamicCastExprClass:
373  case Expr::CXXReinterpretCastExprClass:
374  case Expr::CXXConstCastExprClass:
375  case Expr::CXXAddrspaceCastExprClass:
376  case Expr::ObjCBridgedCastExprClass:
377  case Expr::BuiltinBitCastExprClass:
378  // Only in C++ can casts be interesting at all.
379  if (!Lang.CPlusPlus) return Cl::CL_PRValue;
380  return ClassifyUnnamed(Ctx, cast<ExplicitCastExpr>(E)->getTypeAsWritten());
381 
382  case Expr::CXXUnresolvedConstructExprClass:
383  return ClassifyUnnamed(Ctx,
384  cast<CXXUnresolvedConstructExpr>(E)->getTypeAsWritten());
385 
386  case Expr::BinaryConditionalOperatorClass: {
387  if (!Lang.CPlusPlus) return Cl::CL_PRValue;
388  const auto *co = cast<BinaryConditionalOperator>(E);
389  return ClassifyConditional(Ctx, co->getTrueExpr(), co->getFalseExpr());
390  }
391 
392  case Expr::ConditionalOperatorClass: {
393  // Once again, only C++ is interesting.
394  if (!Lang.CPlusPlus) return Cl::CL_PRValue;
395  const auto *co = cast<ConditionalOperator>(E);
396  return ClassifyConditional(Ctx, co->getTrueExpr(), co->getFalseExpr());
397  }
398 
399  // ObjC message sends are effectively function calls, if the target function
400  // is known.
401  case Expr::ObjCMessageExprClass:
402  if (const ObjCMethodDecl *Method =
403  cast<ObjCMessageExpr>(E)->getMethodDecl()) {
404  Cl::Kinds kind = ClassifyUnnamed(Ctx, Method->getReturnType());
405  return (kind == Cl::CL_PRValue) ? Cl::CL_ObjCMessageRValue : kind;
406  }
407  return Cl::CL_PRValue;
408 
409  // Some C++ expressions are always class temporaries.
410  case Expr::CXXConstructExprClass:
411  case Expr::CXXInheritedCtorInitExprClass:
412  case Expr::CXXTemporaryObjectExprClass:
413  case Expr::LambdaExprClass:
414  case Expr::CXXStdInitializerListExprClass:
415  return Cl::CL_ClassTemporary;
416 
417  case Expr::VAArgExprClass:
418  return ClassifyUnnamed(Ctx, E->getType());
419 
420  case Expr::DesignatedInitExprClass:
421  return ClassifyInternal(Ctx, cast<DesignatedInitExpr>(E)->getInit());
422 
423  case Expr::StmtExprClass: {
424  const CompoundStmt *S = cast<StmtExpr>(E)->getSubStmt();
425  if (const auto *LastExpr = dyn_cast_or_null<Expr>(S->body_back()))
426  return ClassifyUnnamed(Ctx, LastExpr->getType());
427  return Cl::CL_PRValue;
428  }
429 
430  case Expr::PackExpansionExprClass:
431  return ClassifyInternal(Ctx, cast<PackExpansionExpr>(E)->getPattern());
432 
433  case Expr::MaterializeTemporaryExprClass:
434  return cast<MaterializeTemporaryExpr>(E)->isBoundToLvalueReference()
435  ? Cl::CL_LValue
436  : Cl::CL_XValue;
437 
438  case Expr::InitListExprClass:
439  // An init list can be an lvalue if it is bound to a reference and
440  // contains only one element. In that case, we look at that element
441  // for an exact classification. Init list creation takes care of the
442  // value kind for us, so we only need to fine-tune.
443  if (E->isPRValue())
444  return ClassifyExprValueKind(Lang, E, E->getValueKind());
445  assert(cast<InitListExpr>(E)->getNumInits() == 1 &&
446  "Only 1-element init lists can be glvalues.");
447  return ClassifyInternal(Ctx, cast<InitListExpr>(E)->getInit(0));
448 
449  case Expr::CoawaitExprClass:
450  case Expr::CoyieldExprClass:
451  return ClassifyInternal(Ctx, cast<CoroutineSuspendExpr>(E)->getResumeExpr());
452  case Expr::SYCLUniqueStableNameExprClass:
453  case Expr::SYCLUniqueStableIdExprClass:
454  return Cl::CL_PRValue;
455  break;
456 
457  case Expr::CXXParenListInitExprClass:
458  if (isa<ArrayType>(E->getType()))
459  return Cl::CL_ArrayTemporary;
460  return Cl::CL_ClassTemporary;
461  }
462 
463  llvm_unreachable("unhandled expression kind in classification");
464 }

466 /// ClassifyDecl - Return the classification of an expression referencing the
467 /// given declaration.
468 static Cl::Kinds ClassifyDecl(ASTContext &Ctx, const Decl *D) {
469  // C++ [expr.prim.general]p6: The result is an lvalue if the entity is a
470  // function, variable, or data member and a prvalue otherwise.
471  // In C, functions are not lvalues.
472  // In addition, NonTypeTemplateParmDecl derives from VarDecl but isn't an
473  // lvalue unless it's a reference type (C++ [temp.param]p6), so we need to
474  // special-case this.
475 
476  if (const auto *M = dyn_cast<CXXMethodDecl>(D)) {
477  if (M->isImplicitObjectMemberFunction())
478  return Cl::CL_MemberFunction;
479  if (M->isStatic())
480  return Cl::CL_LValue;
481  return Cl::CL_PRValue;
482  }
483 
484  bool islvalue;
485  if (const auto *NTTParm = dyn_cast<NonTypeTemplateParmDecl>(D))
486  islvalue = NTTParm->getType()->isReferenceType() ||
487  NTTParm->getType()->isRecordType();
488  else
489  islvalue =
490  isa<VarDecl, FieldDecl, IndirectFieldDecl, BindingDecl, MSGuidDecl,
491  UnnamedGlobalConstantDecl, TemplateParamObjectDecl>(D) ||
492  (Ctx.getLangOpts().CPlusPlus &&
493  (isa<FunctionDecl, MSPropertyDecl, FunctionTemplateDecl>(D)));
494 
495  return islvalue ? Cl::CL_LValue : Cl::CL_PRValue;
496 }

498 /// ClassifyUnnamed - Return the classification of an expression yielding an
499 /// unnamed value of the given type. This applies in particular to function
500 /// calls and casts.
501 static Cl::Kinds ClassifyUnnamed(ASTContext &Ctx, QualType T) {
502  // In C, function calls are always rvalues.
503  if (!Ctx.getLangOpts().CPlusPlus) return Cl::CL_PRValue;
504 
505  // C++ [expr.call]p10: A function call is an lvalue if the result type is an
506  // lvalue reference type or an rvalue reference to function type, an xvalue
507  // if the result type is an rvalue reference to object type, and a prvalue
508  // otherwise.
509  if (T->isLValueReferenceType())
510  return Cl::CL_LValue;
511  const auto *RV = T->getAs<RValueReferenceType>();
512  if (!RV) // Could still be a class temporary, though.
513  return ClassifyTemporary(T);
514 
515  return RV->getPointeeType()->isFunctionType() ? Cl::CL_LValue : Cl::CL_XValue;
516 }

518 static Cl::Kinds ClassifyMemberExpr(ASTContext &Ctx, const MemberExpr *E) {
519  if (E->getType() == Ctx.UnknownAnyTy)
520  return (isa<FunctionDecl>(E->getMemberDecl())
521  ? Cl::CL_PRValue : Cl::CL_LValue);
522 
523  // Handle C first, it's easier.
524  if (!Ctx.getLangOpts().CPlusPlus) {
525  // C99 6.5.2.3p3
526  // For dot access, the expression is an lvalue if the first part is. For
527  // arrow access, it always is an lvalue.
528  if (E->isArrow())
529  return Cl::CL_LValue;
530  // ObjC property accesses are not lvalues, but get special treatment.
531  Expr *Base = E->getBase()->IgnoreParens();
532  if (isa<ObjCPropertyRefExpr>(Base))
533  return Cl::CL_SubObjCPropertySetting;
534  return ClassifyInternal(Ctx, Base);
535  }
536 
537  NamedDecl *Member = E->getMemberDecl();
538  // C++ [expr.ref]p3: E1->E2 is converted to the equivalent form (*(E1)).E2.
539  // C++ [expr.ref]p4: If E2 is declared to have type "reference to T", then
540  // E1.E2 is an lvalue.
541  if (const auto *Value = dyn_cast<ValueDecl>(Member))
542  if (Value->getType()->isReferenceType())
543  return Cl::CL_LValue;
544 
545  // Otherwise, one of the following rules applies.
546  // -- If E2 is a static member [...] then E1.E2 is an lvalue.
547  if (isa<VarDecl>(Member) && Member->getDeclContext()->isRecord())
548  return Cl::CL_LValue;
549 
550  // -- If E2 is a non-static data member [...]. If E1 is an lvalue, then
551  // E1.E2 is an lvalue; if E1 is an xvalue, then E1.E2 is an xvalue;
552  // otherwise, it is a prvalue.
553  if (isa<FieldDecl>(Member)) {
554  // *E1 is an lvalue
555  if (E->isArrow())
556  return Cl::CL_LValue;
557  Expr *Base = E->getBase()->IgnoreParenImpCasts();
558  if (isa<ObjCPropertyRefExpr>(Base))
559  return Cl::CL_SubObjCPropertySetting;
560  return ClassifyInternal(Ctx, E->getBase());
561  }
562 
563  // -- If E2 is a [...] member function, [...]
564  // -- If it refers to a static member function [...], then E1.E2 is an
565  // lvalue; [...]
566  // -- Otherwise [...] E1.E2 is a prvalue.
567  if (const auto *Method = dyn_cast<CXXMethodDecl>(Member)) {
568  if (Method->isStatic())
569  return Cl::CL_LValue;
570  if (Method->isImplicitObjectMemberFunction())
571  return Cl::CL_MemberFunction;
572  return Cl::CL_PRValue;
573  }
574 
575  // -- If E2 is a member enumerator [...], the expression E1.E2 is a prvalue.
576  // So is everything else we haven't handled yet.
577  return Cl::CL_PRValue;
578 }

580 static Cl::Kinds ClassifyBinaryOp(ASTContext &Ctx, const BinaryOperator *E) {
581  assert(Ctx.getLangOpts().CPlusPlus &&
582  "This is only relevant for C++.");
583  // C++ [expr.ass]p1: All [...] return an lvalue referring to the left operand.
584  // Except we override this for writes to ObjC properties.
585  if (E->isAssignmentOp())
586  return (E->getLHS()->getObjectKind() == OK_ObjCProperty
587  ? Cl::CL_PRValue : Cl::CL_LValue);
588 
589  // C++ [expr.comma]p1: the result is of the same value category as its right
590  // operand, [...].
591  if (E->getOpcode() == BO_Comma)
592  return ClassifyInternal(Ctx, E->getRHS());
593 
594  // C++ [expr.mptr.oper]p6: The result of a .* expression whose second operand
595  // is a pointer to a data member is of the same value category as its first
596  // operand.
597  if (E->getOpcode() == BO_PtrMemD)
598  return (E->getType()->isFunctionType() ||
599  E->hasPlaceholderType(BuiltinType::BoundMember))
600  ? Cl::CL_MemberFunction
601  : ClassifyInternal(Ctx, E->getLHS());
602 
603  // C++ [expr.mptr.oper]p6: The result of an ->* expression is an lvalue if its
604  // second operand is a pointer to data member and a prvalue otherwise.
605  if (E->getOpcode() == BO_PtrMemI)
606  return (E->getType()->isFunctionType() ||
607  E->hasPlaceholderType(BuiltinType::BoundMember))
608  ? Cl::CL_MemberFunction
609  : Cl::CL_LValue;
610 
611  // All other binary operations are prvalues.
612  return Cl::CL_PRValue;
613 }

615 static Cl::Kinds ClassifyConditional(ASTContext &Ctx, const Expr *True,
616  const Expr *False) {
617  assert(Ctx.getLangOpts().CPlusPlus &&
618  "This is only relevant for C++.");
619 
620  // C++ [expr.cond]p2
621  // If either the second or the third operand has type (cv) void,
622  // one of the following shall hold:
623  if (True->getType()->isVoidType() || False->getType()->isVoidType()) {
624  // The second or the third operand (but not both) is a (possibly
625  // parenthesized) throw-expression; the result is of the [...] value
626  // category of the other.
627  bool TrueIsThrow = isa<CXXThrowExpr>(True->IgnoreParenImpCasts());
628  bool FalseIsThrow = isa<CXXThrowExpr>(False->IgnoreParenImpCasts());
629  if (const Expr *NonThrow = TrueIsThrow ? (FalseIsThrow ? nullptr : False)
630  : (FalseIsThrow ? True : nullptr))
631  return ClassifyInternal(Ctx, NonThrow);
632 
633  // [Otherwise] the result [...] is a prvalue.
634  return Cl::CL_PRValue;
635  }
636 
637  // Note that at this point, we have already performed all conversions
638  // according to [expr.cond]p3.
639  // C++ [expr.cond]p4: If the second and third operands are glvalues of the
640  // same value category [...], the result is of that [...] value category.
641  // C++ [expr.cond]p5: Otherwise, the result is a prvalue.
642  Cl::Kinds LCl = ClassifyInternal(Ctx, True),
643  RCl = ClassifyInternal(Ctx, False);
644  return LCl == RCl ? LCl : Cl::CL_PRValue;
645 }

647 static Cl::ModifiableType IsModifiable(ASTContext &Ctx, const Expr *E,
648  Cl::Kinds Kind, SourceLocation &Loc) {
649  // As a general rule, we only care about lvalues. But there are some rvalues
650  // for which we want to generate special results.
651  if (Kind == Cl::CL_PRValue) {
652  // For the sake of better diagnostics, we want to specifically recognize
653  // use of the GCC cast-as-lvalue extension.
654  if (const auto *CE = dyn_cast<ExplicitCastExpr>(E->IgnoreParens())) {
655  if (CE->getSubExpr()->IgnoreParenImpCasts()->isLValue()) {
656  Loc = CE->getExprLoc();
657  return Cl::CM_LValueCast;
658  }
659  }
660  }
661  if (Kind != Cl::CL_LValue)
662  return Cl::CM_RValue;
663 
664  // This is the lvalue case.
665  // Functions are lvalues in C++, but not modifiable. (C++ [basic.lval]p6)
666  if (Ctx.getLangOpts().CPlusPlus && E->getType()->isFunctionType())
667  return Cl::CM_Function;
668 
669  // Assignment to a property in ObjC is an implicit setter access. But a
670  // setter might not exist.
671  if (const auto *Expr = dyn_cast<ObjCPropertyRefExpr>(E)) {
672  if (Expr->isImplicitProperty() &&
673  Expr->getImplicitPropertySetter() == nullptr)
674  return Cl::CM_NoSetterProperty;
675  }
676 
677  CanQualType CT = Ctx.getCanonicalType(E->getType());
678  // Const stuff is obviously not modifiable.
679  if (CT.isConstQualified())
680  return Cl::CM_ConstQualified;
681  if (Ctx.getLangOpts().OpenCL &&
682  CT.getQualifiers().getAddressSpace() == LangAS::opencl_constant)
683  return Cl::CM_ConstAddrSpace;
684 
685  // Arrays are not modifiable, only their elements are.
686  if (CT->isArrayType())
687  return Cl::CM_ArrayType;
688  // Incomplete types are not modifiable.
689  if (CT->isIncompleteType())
690  return Cl::CM_IncompleteType;
691 
692  // Records with any const fields (recursively) are not modifiable.
693  if (const RecordType *R = CT->getAs<RecordType>())
694  if (R->hasConstFields())
695  return Cl::CM_ConstQualifiedField;
696 
697  return Cl::CM_Modifiable;
698 }
