23 /// Given a pointer-authentication schema, return a concrete "other"
24 /// discriminator for it.
25 llvm::ConstantInt *CodeGenModule::getPointerAuthOtherDiscriminator(
26  const PointerAuthSchema &Schema, GlobalDecl Decl, QualType Type) {
27  switch (Schema.getOtherDiscrimination()) {
28  case PointerAuthSchema::Discrimination::None:
29  return nullptr;
30 
31  case PointerAuthSchema::Discrimination::Type:
32  assert(!Type.isNull() && "type not provided for type-discriminated schema");
33  return llvm::ConstantInt::get(
34  IntPtrTy, getContext().getPointerAuthTypeDiscriminator(Type));
35 
36  case PointerAuthSchema::Discrimination::Decl:
37  assert(Decl.getDecl() &&
38  "declaration not provided for decl-discriminated schema");
39  return llvm::ConstantInt::get(IntPtrTy,
40  getPointerAuthDeclDiscriminator(Decl));
41 
42  case PointerAuthSchema::Discrimination::Constant:
43  return llvm::ConstantInt::get(IntPtrTy, Schema.getConstantDiscrimination());
44  }
45  llvm_unreachable("bad discrimination kind");
46 }

71 /// Return the abstract pointer authentication schema for a pointer to the given
72 /// function type.
73 CGPointerAuthInfo CodeGenModule::getFunctionPointerAuthInfo(QualType T) {
74  const auto &Schema = getCodeGenOpts().PointerAuth.FunctionPointers;
75  if (!Schema)
76  return CGPointerAuthInfo();
77 
78  assert(!Schema.isAddressDiscriminated() &&
79  "function pointers cannot use address-specific discrimination");
80 
81  llvm::Constant *Discriminator = nullptr;
82  if (T->isFunctionPointerType() || T->isFunctionReferenceType())
83  T = T->getPointeeType();
84  if (T->isFunctionType())
85  Discriminator = getPointerAuthOtherDiscriminator(Schema, GlobalDecl(), T);
86 
87  return CGPointerAuthInfo(Schema.getKey(), Schema.getAuthenticationMode(),
88  /*IsaPointer=*/false, /*AuthenticatesNull=*/false,
89  Discriminator);
90 }

100 /// Emit the concrete pointer authentication informaton for the
101 /// given authentication schema.
102 CGPointerAuthInfo CodeGenFunction::EmitPointerAuthInfo(
103  const PointerAuthSchema &Schema, llvm::Value *StorageAddress,
104  GlobalDecl SchemaDecl, QualType SchemaType) {
105  if (!Schema)
106  return CGPointerAuthInfo();
107 
108  llvm::Value *Discriminator =
109  CGM.getPointerAuthOtherDiscriminator(Schema, SchemaDecl, SchemaType);
110 
111  if (Schema.isAddressDiscriminated()) {
112  assert(StorageAddress &&
113  "address not provided for address-discriminated schema");
114 
115  if (Discriminator)
116  Discriminator =
117  EmitPointerAuthBlendDiscriminator(StorageAddress, Discriminator);
118  else
119  Discriminator = Builder.CreatePtrToInt(StorageAddress, IntPtrTy);
120  }
121 
122  return CGPointerAuthInfo(Schema.getKey(), Schema.getAuthenticationMode(),
123  Schema.isIsaPointer(),
124  Schema.authenticatesNullValues(), Discriminator);
125 }

131  if (PointeeType.isNull())

168 llvm::Constant *
169 CodeGenModule::getConstantSignedPointer(llvm::Constant *Pointer, unsigned Key,
170  llvm::Constant *StorageAddress,
171  llvm::ConstantInt *OtherDiscriminator) {
172  llvm::Constant *AddressDiscriminator;
173  if (StorageAddress) {
174  assert(StorageAddress->getType() == UnqualPtrTy);
175  AddressDiscriminator = StorageAddress;
176  } else {
177  AddressDiscriminator = llvm::Constant::getNullValue(UnqualPtrTy);
178  }
179 
180  llvm::ConstantInt *IntegerDiscriminator;
181  if (OtherDiscriminator) {
182  assert(OtherDiscriminator->getType() == Int64Ty);
183  IntegerDiscriminator = OtherDiscriminator;
184  } else {
185  IntegerDiscriminator = llvm::ConstantInt::get(Int64Ty, 0);
186  }
187 
188  return llvm::ConstantPtrAuth::get(Pointer,
189  llvm::ConstantInt::get(Int32Ty, Key),
190  IntegerDiscriminator, AddressDiscriminator);
191 }

245 std::optional<PointerAuthQualifier>
246 CodeGenModule::computeVTPointerAuthentication(const CXXRecordDecl *ThisClass) {
247  auto DefaultAuthentication = getCodeGenOpts().PointerAuth.CXXVTablePointers;
248  if (!DefaultAuthentication)
249  return std::nullopt;
250  const CXXRecordDecl *PrimaryBase =
251  Context.baseForVTableAuthentication(ThisClass);
252 
253  unsigned Key = DefaultAuthentication.getKey();
254  bool AddressDiscriminated = DefaultAuthentication.isAddressDiscriminated();
255  auto DefaultDiscrimination = DefaultAuthentication.getOtherDiscrimination();
256  unsigned TypeBasedDiscriminator =
257  Context.getPointerAuthVTablePointerDiscriminator(PrimaryBase);
258  unsigned Discriminator;
259  if (DefaultDiscrimination == PointerAuthSchema::Discrimination::Type) {
260  Discriminator = TypeBasedDiscriminator;
261  } else if (DefaultDiscrimination ==
262  PointerAuthSchema::Discrimination::Constant) {
263  Discriminator = DefaultAuthentication.getConstantDiscrimination();
264  } else {
265  assert(DefaultDiscrimination == PointerAuthSchema::Discrimination::None);
266  Discriminator = 0;
267  }
268  if (auto ExplicitAuthentication =
269  PrimaryBase->getAttr<VTablePointerAuthenticationAttr>()) {
270  auto ExplicitAddressDiscrimination =
271  ExplicitAuthentication->getAddressDiscrimination();
272  auto ExplicitDiscriminator =
273  ExplicitAuthentication->getExtraDiscrimination();
274 
275  unsigned ExplicitKey = ExplicitAuthentication->getKey();
276  if (ExplicitKey == VTablePointerAuthenticationAttr::NoKey)
277  return std::nullopt;
278 
279  if (ExplicitKey != VTablePointerAuthenticationAttr::DefaultKey) {
280  if (ExplicitKey == VTablePointerAuthenticationAttr::ProcessIndependent)
281  Key = (unsigned)PointerAuthSchema::ARM8_3Key::ASDA;
282  else {
283  assert(ExplicitKey ==
284  VTablePointerAuthenticationAttr::ProcessDependent);
285  Key = (unsigned)PointerAuthSchema::ARM8_3Key::ASDB;
286  }
287  }
288 
289  if (ExplicitAddressDiscrimination !=
290  VTablePointerAuthenticationAttr::DefaultAddressDiscrimination)
291  AddressDiscriminated =
292  ExplicitAddressDiscrimination ==
293  VTablePointerAuthenticationAttr::AddressDiscrimination;
294 
295  if (ExplicitDiscriminator ==
296  VTablePointerAuthenticationAttr::TypeDiscrimination)
297  Discriminator = TypeBasedDiscriminator;
298  else if (ExplicitDiscriminator ==
299  VTablePointerAuthenticationAttr::CustomDiscrimination)
300  Discriminator = ExplicitAuthentication->getCustomDiscriminationValue();
301  else if (ExplicitDiscriminator ==
302  VTablePointerAuthenticationAttr::NoExtraDiscrimination)
303  Discriminator = 0;
304  }
305  return PointerAuthQualifier::Create(Key, AddressDiscriminated, Discriminator,
306  PointerAuthenticationMode::SignAndAuth,
307  /* IsIsaPointer */ false,
308  /* AuthenticatesNullValues */ false);
309 }

311 std::optional<PointerAuthQualifier>
312 CodeGenModule::getVTablePointerAuthentication(const CXXRecordDecl *Record) {
313  if (!Record->getDefinition() || !Record->isPolymorphic())
314  return std::nullopt;
315 
316  auto Existing = VTablePtrAuthInfos.find(Record);
317  std::optional<PointerAuthQualifier> Authentication;
318  if (Existing != VTablePtrAuthInfos.end()) {
319  Authentication = Existing->getSecond();
320  } else {
321  Authentication = computeVTPointerAuthentication(Record);
322  VTablePtrAuthInfos.insert(std::make_pair(Record, Authentication));
323  }
324  return Authentication;
325 }

328 CodeGenModule::getVTablePointerAuthInfo(CodeGenFunction *CGF,
329  const CXXRecordDecl *Record,
330  llvm::Value *StorageAddress) {

332  if (!Authentication)
333  return std::nullopt;
334 
335  llvm::Value *Discriminator = nullptr;
336  if (auto ExtraDiscriminator = Authentication->getExtraDiscriminator())
337  Discriminator = llvm::ConstantInt::get(IntPtrTy, ExtraDiscriminator);

344  CGF->EmitPointerAuthBlendDiscriminator(StorageAddress, Discriminator);
345  else

347  }
348 
349  return CGPointerAuthInfo(Authentication->getKey(),
350  PointerAuthenticationMode::SignAndAuth,
351  /* IsIsaPointer */ false,
352  /* AuthenticatesNullValues */ false, Discriminator);
