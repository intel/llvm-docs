349 /// Helper class that saves the current stream position and
350 /// then restores it when destroyed.
351 struct SavedStreamPosition {
352  explicit SavedStreamPosition(llvm::BitstreamCursor &Cursor)
353  : Cursor(Cursor), Offset(Cursor.GetCurrentBitNo()) {}
354 
355  ~SavedStreamPosition() {
356  if (llvm::Error Err = Cursor.JumpToBit(Offset))
357  llvm::report_fatal_error(
358  llvm::Twine("Cursor should always be able to go back, failed: ") +
359  toString(std::move(Err)));
360  }
361 
362 private:
