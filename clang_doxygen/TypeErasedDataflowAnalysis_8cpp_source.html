104  void extendFlowCondition(const Expr &Cond) {
105  // The terminator sub-expression might not be evaluated.
106  if (Env.getValue(Cond, SkipPast::None) == nullptr)
107  transfer(StmtToEnv, Cond, Env);
108 
109  auto *Val =
110  cast_or_null<BoolValue>(Env.getValue(Cond, SkipPast::Reference));
111  if (Val == nullptr)
112  return;
113 
114  // The condition must be inverted for the successor that encompasses the
115  // "else" branch, if such exists.
116  if (BlockSuccIdx == 1)
117  Val = &Env.makeNot(*Val);
118 
119  Env.addToFlowCondition(*Val);
120  }
121 
122  const StmtToEnvMap &StmtToEnv;
123  Environment &Env;
124  int BlockSuccIdx;

127 /// Computes the input state for a given basic block by joining the output
128 /// states of its predecessors.
129 ///
130 /// Requirements:
131 ///
132 /// All predecessors of `Block` except those with loop back edges must have
133 /// already been transferred. States in `BlockStates` that are set to
134 /// `llvm::None` represent basic blocks that are not evaluated yet.
135 static TypeErasedDataflowAnalysisState computeBlockInputState(
136  const ControlFlowContext &CFCtx,
137  std::vector<llvm::Optional<TypeErasedDataflowAnalysisState>> &BlockStates,
138  const CFGBlock &Block, const Environment &InitEnv,
139  TypeErasedDataflowAnalysis &Analysis) {
140  llvm::DenseSet<const CFGBlock *> Preds;
141  Preds.insert(Block.pred_begin(), Block.pred_end());
142  if (Block.getTerminator().isTemporaryDtorsBranch()) {
143  // This handles a special case where the code that produced the CFG includes
144  // a conditional operator with a branch that constructs a temporary and
145  // calls a destructor annotated as noreturn. The CFG models this as follows:
146  //
147  // B1 (contains the condition of the conditional operator) - succs: B2, B3
148  // B2 (contains code that does not call a noreturn destructor) - succs: B4
149  // B3 (contains code that calls a noreturn destructor) - succs: B4
150  // B4 (has temporary destructor terminator) - succs: B5, B6
151  // B5 (noreturn block that is associated with the noreturn destructor call)
152  // B6 (contains code that follows the conditional operator statement)
153  //
154  // The first successor (B5 above) of a basic block with a temporary
155  // destructor terminator (B4 above) is the block that evaluates the
156  // destructor. If that block has a noreturn element then the predecessor
157  // block that constructed the temporary object (B3 above) is effectively a
158  // noreturn block and its state should not be used as input for the state
159  // of the block that has a temporary destructor terminator (B4 above). This
160  // holds regardless of which branch of the ternary operator calls the
161  // noreturn destructor. However, it doesn't cases where a nested ternary
162  // operator includes a branch that contains a noreturn destructor call.
163  //
164  // See `NoreturnDestructorTest` for concrete examples.
165  if (Block.succ_begin()->getReachableBlock()->hasNoReturnElement()) {
166  auto StmtBlock = CFCtx.getStmtToBlock().find(Block.getTerminatorStmt());
167  assert(StmtBlock != CFCtx.getStmtToBlock().end());
168  Preds.erase(StmtBlock->getSecond());
169  }
170  }
171 
172  llvm::Optional<TypeErasedDataflowAnalysisState> MaybeState;
173  bool ApplyBuiltinTransfer = Analysis.applyBuiltinTransfer();
174 
175  for (const CFGBlock *Pred : Preds) {
176  // Skip if the `Block` is unreachable or control flow cannot get past it.
177  if (!Pred || Pred->hasNoReturnElement())
178  continue;
179 
180  // Skip if `Pred` was not evaluated yet. This could happen if `Pred` has a
181  // loop back edge to `Block`.
182  const llvm::Optional<TypeErasedDataflowAnalysisState> &MaybePredState =
183  BlockStates[Pred->getBlockID()];
184  if (!MaybePredState.hasValue())
185  continue;
186 
187  TypeErasedDataflowAnalysisState PredState = MaybePredState.getValue();
188  if (ApplyBuiltinTransfer) {
189  if (const Stmt *PredTerminatorStmt = Pred->getTerminatorStmt()) {
190  const StmtToEnvMapImpl StmtToEnv(CFCtx, BlockStates);
191  TerminatorVisitor(StmtToEnv, PredState.Env,
192  blockIndexInPredecessor(*Pred, Block))
193  .Visit(PredTerminatorStmt);
194  }
195  }
196 
197  if (MaybeState.hasValue()) {
198  Analysis.joinTypeErased(MaybeState->Lattice, PredState.Lattice);
199  MaybeState->Env.join(PredState.Env, Analysis);
200  } else {
201  MaybeState = std::move(PredState);
202  }
203  }
204  if (!MaybeState.hasValue()) {
205  // FIXME: Consider passing `Block` to `Analysis.typeErasedInitialElement()`
206  // to enable building analyses like computation of dominators that
207  // initialize the state of each basic block differently.
208  MaybeState.emplace(Analysis.typeErasedInitialElement(), InitEnv);
209  }
210  return *MaybeState;
211 }

213 /// Transfers `State` by evaluating `CfgStmt` in the context of `Analysis`.
214 /// `HandleTransferredStmt` (if provided) will be applied to `CfgStmt`, after it
215 /// is evaluated.
216 static void transferCFGStmt(
217  const ControlFlowContext &CFCtx,
218  llvm::ArrayRef<llvm::Optional<TypeErasedDataflowAnalysisState>> BlockStates,
219  const CFGStmt &CfgStmt, TypeErasedDataflowAnalysis &Analysis,
220  TypeErasedDataflowAnalysisState &State,
221  std::function<void(const CFGStmt &,
222  const TypeErasedDataflowAnalysisState &)>
223  HandleTransferredStmt) {
224  const Stmt *S = CfgStmt.getStmt();
225  assert(S != nullptr);
226 
227  if (Analysis.applyBuiltinTransfer())
228  transfer(StmtToEnvMapImpl(CFCtx, BlockStates), *S, State.Env);
229  Analysis.transferTypeErased(S, State.Lattice, State.Env);
230 
231  if (HandleTransferredStmt != nullptr)
232  HandleTransferredStmt(CfgStmt, State);
233 }

235 /// Transfers `State` by evaluating `CfgInit`.
236 static void transferCFGInitializer(const CFGInitializer &CfgInit,
237  TypeErasedDataflowAnalysisState &State) {
238  const auto &ThisLoc = *cast<AggregateStorageLocation>(
239  State.Env.getThisPointeeStorageLocation());
240 
241  const CXXCtorInitializer *Initializer = CfgInit.getInitializer();
242  assert(Initializer != nullptr);
243 
244  auto *InitStmt = Initializer->getInit();
245  assert(InitStmt != nullptr);
246 
247  auto *InitStmtLoc =
248  State.Env.getStorageLocation(*InitStmt, SkipPast::Reference);
249  if (InitStmtLoc == nullptr)
250  return;
251 
252  auto *InitStmtVal = State.Env.getValue(*InitStmtLoc);
253  if (InitStmtVal == nullptr)
254  return;
255 
256  const FieldDecl *Member = Initializer->getMember();
257  assert(Member != nullptr);
258 
259  if (Member->getType()->isReferenceType()) {
260  auto &MemberLoc = ThisLoc.getChild(*Member);
261  State.Env.setValue(MemberLoc,
262  State.Env.takeOwnership(
263  std::make_unique<ReferenceValue>(*InitStmtLoc)));
264  } else {
265  auto &MemberLoc = ThisLoc.getChild(*Member);
266  State.Env.setValue(MemberLoc, *InitStmtVal);
267  }
268 }

270 TypeErasedDataflowAnalysisState transferBlock(
271  const ControlFlowContext &CFCtx,
272  std::vector<llvm::Optional<TypeErasedDataflowAnalysisState>> &BlockStates,
273  const CFGBlock &Block, const Environment &InitEnv,
274  TypeErasedDataflowAnalysis &Analysis,
275  std::function<void(const CFGStmt &,
276  const TypeErasedDataflowAnalysisState &)>
277  HandleTransferredStmt) {
278  TypeErasedDataflowAnalysisState State =
279  computeBlockInputState(CFCtx, BlockStates, Block, InitEnv, Analysis);
280  for (const CFGElement &Element : Block) {
281  switch (Element.getKind()) {
282  case CFGElement::Statement:
283  transferCFGStmt(CFCtx, BlockStates, *Element.getAs<CFGStmt>(), Analysis,
284  State, HandleTransferredStmt);
285  break;
286  case CFGElement::Initializer:
287  if (Analysis.applyBuiltinTransfer())
288  transferCFGInitializer(*Element.getAs<CFGInitializer>(), State);
289  break;
290  default:
291  // FIXME: Evaluate other kinds of `CFGElement`.
292  break;
293  }
294  }
295  return State;
296 }

298 llvm::Expected<std::vector<llvm::Optional<TypeErasedDataflowAnalysisState>>>
299 runTypeErasedDataflowAnalysis(const ControlFlowContext &CFCtx,
300  TypeErasedDataflowAnalysis &Analysis,
301  const Environment &InitEnv) {
302  PostOrderCFGView POV(&CFCtx.getCFG());
303  ForwardDataflowWorklist Worklist(CFCtx.getCFG(), &POV);
304 
305  std::vector<llvm::Optional<TypeErasedDataflowAnalysisState>> BlockStates;
306  BlockStates.resize(CFCtx.getCFG().size(), llvm::None);
307 
308  // The entry basic block doesn't contain statements so it can be skipped.
309  const CFGBlock &Entry = CFCtx.getCFG().getEntry();
310  BlockStates[Entry.getBlockID()] = {Analysis.typeErasedInitialElement(),
311  InitEnv};
312  Worklist.enqueueSuccessors(&Entry);
313 
314  // Bugs in lattices and transfer functions can prevent the analysis from
315  // converging. To limit the damage (infinite loops) that these bugs can cause,
316  // limit the number of iterations.
317  // FIXME: Consider making the maximum number of iterations configurable.
318  // FIXME: Set up statistics (see llvm/ADT/Statistic.h) to count average number
319  // of iterations, number of functions that time out, etc.
320  uint32_t Iterations = 0;
321  static constexpr uint32_t MaxIterations = 1 << 16;
322  while (const CFGBlock *Block = Worklist.dequeue()) {
323  if (++Iterations > MaxIterations) {
324  return llvm::createStringError(std::errc::timed_out,
325  "maximum number of iterations reached");
326  }
327 
328  const llvm::Optional<TypeErasedDataflowAnalysisState> &OldBlockState =
329  BlockStates[Block->getBlockID()];
330  TypeErasedDataflowAnalysisState NewBlockState =
331  transferBlock(CFCtx, BlockStates, *Block, InitEnv, Analysis);
332 
333  if (OldBlockState.hasValue() &&
334  Analysis.isEqualTypeErased(OldBlockState.getValue().Lattice,
335  NewBlockState.Lattice) &&
336  OldBlockState->Env.equivalentTo(NewBlockState.Env, Analysis)) {
337  // The state of `Block` didn't change after transfer so there's no need to
338  // revisit its successors.
339  continue;
340  }
341 
342  BlockStates[Block->getBlockID()] = std::move(NewBlockState);
343 
344  // Do not add unreachable successor blocks to `Worklist`.
345  if (Block->hasNoReturnElement())
346  continue;
347 
348  Worklist.enqueueSuccessors(Block);
349  }
350  // FIXME: Consider evaluating unreachable basic blocks (those that have a
351  // state set to `llvm::None` at this point) to also analyze dead code.
352 
353  return BlockStates;
354 }
