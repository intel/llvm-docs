36 /// Indicates what kind of indirections should be skipped past when retrieving

278  /// Returns a boolean value represents `LHS` => `RHS`. Subsequent calls with
279  /// the same arguments, regardless of their order, will return the same
280  /// result. If the given boolean values represent the same value, the result
281  /// will be a value that represents the true boolean literal.
282  BoolValue &makeImplication(BoolValue &LHS, BoolValue &RHS) const {
283  return &LHS == &RHS ? getBoolLiteralValue(true) : makeOr(makeNot(LHS), RHS);
284  }
285 
286  /// Returns a boolean value represents `LHS` <=> `RHS`. Subsequent calls with
287  /// the same arguments, regardless of their order, will return the same
288  /// result. If the given boolean values represent the same value, the result
289  /// will be a value that represents the true boolean literal.
290  BoolValue &makeIff(BoolValue &LHS, BoolValue &RHS) const {
291  return &LHS == &RHS
292  ? getBoolLiteralValue(true)
293  : makeAnd(makeImplication(LHS, RHS), makeImplication(RHS, LHS));
294  }
