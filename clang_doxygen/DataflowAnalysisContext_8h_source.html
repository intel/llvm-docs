33 /// Owns objects that encompass the state of a program and stores context that
34 /// is used during dataflow analysis.
35 class DataflowAnalysisContext {
36 public:
37  /// Constructs a dataflow analysis context.
38  ///
39  /// Requirements:
40  ///
41  /// `S` must not be null.
42  DataflowAnalysisContext(std::unique_ptr<Solver> S)
43  : S(std::move(S)), TrueVal(createAtomicBoolValue()),
44  FalseVal(createAtomicBoolValue()) {
45  assert(this->S != nullptr);
46  }
47 
48  /// Returns the SAT solver instance that is available in this context.
49  Solver &getSolver() const { return *S; }
50 
51  /// Takes ownership of `Loc` and returns a reference to it.
52  ///
53  /// Requirements:
54  ///
55  /// `Loc` must not be null.
56  template <typename T>
57  typename std::enable_if<std::is_base_of<StorageLocation, T>::value, T &>::type
58  takeOwnership(std::unique_ptr<T> Loc) {
59  assert(Loc != nullptr);
60  Locs.push_back(std::move(Loc));
61  return *cast<T>(Locs.back().get());
62  }
63 
64  /// Takes ownership of `Val` and returns a reference to it.
65  ///
66  /// Requirements:
67  ///
68  /// `Val` must not be null.
69  template <typename T>
70  typename std::enable_if<std::is_base_of<Value, T>::value, T &>::type
71  takeOwnership(std::unique_ptr<T> Val) {
72  assert(Val != nullptr);
73  Vals.push_back(std::move(Val));
74  return *cast<T>(Vals.back().get());
75  }
76 
77  /// Assigns `Loc` as the storage location of `D`.
78  ///
79  /// Requirements:
80  ///
81  /// `D` must not be assigned a storage location.
82  void setStorageLocation(const ValueDecl &D, StorageLocation &Loc) {
83  assert(DeclToLoc.find(&D) == DeclToLoc.end());
84  DeclToLoc[&D] = &Loc;
85  }
86 
87  /// Returns the storage location assigned to `D` or null if `D` has no
88  /// assigned storage location.
89  StorageLocation *getStorageLocation(const ValueDecl &D) const {
90  auto It = DeclToLoc.find(&D);
91  return It == DeclToLoc.end() ? nullptr : It->second;
92  }
93 
94  /// Assigns `Loc` as the storage location of `E`.
95  ///
96  /// Requirements:
97  ///
98  /// `E` must not be assigned a storage location.
99  void setStorageLocation(const Expr &E, StorageLocation &Loc) {
100  assert(ExprToLoc.find(&E) == ExprToLoc.end());
101  ExprToLoc[&E] = &Loc;
102  }
103 
104  /// Returns the storage location assigned to `E` or null if `E` has no
105  /// assigned storage location.
106  StorageLocation *getStorageLocation(const Expr &E) const {
107  auto It = ExprToLoc.find(&E);
108  return It == ExprToLoc.end() ? nullptr : It->second;
109  }
110 
111  /// Assigns `Loc` as the storage location of the `this` pointee.
112  ///
113  /// Requirements:
114  ///
115  /// The `this` pointee must not be assigned a storage location.
116  void setThisPointeeStorageLocation(StorageLocation &Loc) {
117  assert(ThisPointeeLoc == nullptr);
118  ThisPointeeLoc = &Loc;
119  }
120 
121  /// Returns the storage location assigned to the `this` pointee or null if the
122  /// `this` pointee has no assigned storage location.
123  StorageLocation *getThisPointeeStorageLocation() const {
124  return ThisPointeeLoc;
125  }
126 
127  /// Returns a symbolic boolean value that models a boolean literal equal to
128  /// `Value`.
129  AtomicBoolValue &getBoolLiteralValue(bool Value) const {
130  return Value ? TrueVal : FalseVal;
131  }
132 
133  /// Creates an atomic boolean value.
134  AtomicBoolValue &createAtomicBoolValue() {
135  return takeOwnership(std::make_unique<AtomicBoolValue>());
136  }
137 
138  /// Returns a boolean value that represents the conjunction of `LHS` and
139  /// `RHS`. Subsequent calls with the same arguments, regardless of their
140  /// order, will return the same result. If the given boolean values represent
141  /// the same value, the result will be the value itself.
142  BoolValue &getOrCreateConjunctionValue(BoolValue &LHS, BoolValue &RHS);
143 
144  /// Returns a boolean value that represents the disjunction of `LHS` and
145  /// `RHS`. Subsequent calls with the same arguments, regardless of their
146  /// order, will return the same result. If the given boolean values represent
147  /// the same value, the result will be the value itself.
148  BoolValue &getOrCreateDisjunctionValue(BoolValue &LHS, BoolValue &RHS);
149 
150  /// Returns a boolean value that represents the negation of `Val`. Subsequent
151  /// calls with the same argument will return the same result.
152  BoolValue &getOrCreateNegationValue(BoolValue &Val);
153 
154 private:
155  std::unique_ptr<Solver> S;
156 
157  // Storage for the state of a program.
158  std::vector<std::unique_ptr<StorageLocation>> Locs;
159  std::vector<std::unique_ptr<Value>> Vals;
160 
161  // Maps from program declarations and statements to storage locations that are
162  // assigned to them. These assignments are global (aggregated across all basic
163  // blocks) and are used to produce stable storage locations when the same
164  // basic blocks are evaluated multiple times. The storage locations that are
165  // in scope for a particular basic block are stored in `Environment`.
166  llvm::DenseMap<const ValueDecl *, StorageLocation *> DeclToLoc;
167  llvm::DenseMap<const Expr *, StorageLocation *> ExprToLoc;
168 
169  StorageLocation *ThisPointeeLoc = nullptr;
170 
171  AtomicBoolValue &TrueVal;
172  AtomicBoolValue &FalseVal;
173 
174  // Indices that are used to avoid recreating the same composite boolean
175  // values.
176  llvm::DenseMap<std::pair<BoolValue *, BoolValue *>, ConjunctionValue *>
177  ConjunctionVals;
178  llvm::DenseMap<std::pair<BoolValue *, BoolValue *>, DisjunctionValue *>
179  DisjunctionVals;
180  llvm::DenseMap<BoolValue *, NegationValue *> NegationVals;
181 };
