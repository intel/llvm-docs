283 unsigned serialization::ComputeHash(Selector Sel) {
284  unsigned N = Sel.getNumArgs();
285  if (N == 0)
286  ++N;
287  unsigned R = 5381;
288  for (unsigned I = 0; I != N; ++I)
289  if (IdentifierInfo *II = Sel.getIdentifierInfoForSlot(I))
290  R = llvm::djbHash(II->getName(), R);
291  return R;
292 }

294 const DeclContext *
295 serialization::getDefinitiveDeclContext(const DeclContext *DC) {
296  switch (DC->getDeclKind()) {
297  // These entities may have multiple definitions.
298  case Decl::TranslationUnit:
299  case Decl::ExternCContext:
300  case Decl::Namespace:
301  case Decl::LinkageSpec:
302  case Decl::Export:
303  return nullptr;
304 
305  // C/C++ tag types can only be defined in one place.
306  case Decl::Enum:
307  case Decl::Record:
308  if (const TagDecl *Def = cast<TagDecl>(DC)->getDefinition())
309  return Def;
310  return nullptr;
311 
312  // FIXME: These can be defined in one place... except special member
313  // functions and out-of-line definitions.
314  case Decl::CXXRecord:
315  case Decl::ClassTemplateSpecialization:
316  case Decl::ClassTemplatePartialSpecialization:
317  return nullptr;
318 
319  // Each function, method, and block declaration is its own DeclContext.
320  case Decl::Function:
321  case Decl::CXXMethod:
322  case Decl::CXXConstructor:
323  case Decl::CXXDestructor:
324  case Decl::CXXConversion:
325  case Decl::ObjCMethod:
326  case Decl::Block:
327  case Decl::Captured:
328  // Objective C categories, category implementations, and class
329  // implementations can only be defined in one place.
330  case Decl::ObjCCategory:
331  case Decl::ObjCCategoryImpl:
332  case Decl::ObjCImplementation:
333  return DC;
334 
335  case Decl::ObjCProtocol:
336  if (const ObjCProtocolDecl *Def
337  = cast<ObjCProtocolDecl>(DC)->getDefinition())
338  return Def;
339  return nullptr;
340 
341  // FIXME: These are defined in one place, but properties in class extensions
342  // end up being back-patched into the main interface. See
343  // Sema::HandlePropertyInClassExtension for the offending code.
344  case Decl::ObjCInterface:
345  return nullptr;
346 
347  default:
348  llvm_unreachable("Unhandled DeclContext in AST reader");
349  }
350 
351  llvm_unreachable("Unhandled decl kind");
352 }

455 bool serialization::needsAnonymousDeclarationNumber(const NamedDecl *D) {
456  // Friend declarations in dependent contexts aren't anonymous in the usual
457  // sense, but they cannot be found by name lookup in their semantic context
458  // (or indeed in any context), so we treat them as anonymous.
459  //
460  // This doesn't apply to friend tag decls; Sema makes those available to name
461  // lookup in the surrounding context.
462  if (D->getFriendObjectKind() &&
463  D->getLexicalDeclContext()->isDependentContext() && !isa<TagDecl>(D)) {
464  // For function templates and class templates, the template is numbered and
465  // not its pattern.
466  if (auto *FD = dyn_cast<FunctionDecl>(D))
467  return !FD->getDescribedFunctionTemplate();
468  if (auto *RD = dyn_cast<CXXRecordDecl>(D))
469  return !RD->getDescribedClassTemplate();
470  return true;
471  }
472 
473  // At block scope, we number everything that we need to deduplicate, since we
474  // can't just use name matching to keep things lined up.
475  // FIXME: This is only necessary for an inline function or a template or
476  // similar.
477  if (D->getLexicalDeclContext()->isFunctionOrMethod()) {
478  if (auto *VD = dyn_cast<VarDecl>(D))
479  return VD->isStaticLocal();
480  // FIXME: What about CapturedDecls (and declarations nested within them)?
481  return isa<TagDecl>(D) || isa<BlockDecl>(D);
482  }
483 
484  // Otherwise, we only care about anonymous class members / block-scope decls.
485  // FIXME: We need to handle lambdas and blocks within inline / templated
486  // variables too.
487  if (D->getDeclName() || !isa<RecordDecl>(D->getLexicalDeclContext()))
488  return false;
489  return isa<TagDecl>(D) || isa<FieldDecl>(D);
490 }
