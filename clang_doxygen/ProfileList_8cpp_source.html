74 static StringRef getSectionName(CodeGenOptions::ProfileInstrKind Kind) {
75  switch (Kind) {
76  case CodeGenOptions::ProfileNone:
77  return "";
78  case CodeGenOptions::ProfileClangInstr:

80  case CodeGenOptions::ProfileIRInstr:
