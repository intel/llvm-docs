52 /// Check if the compiler flag in question should be skipped when
53 /// emitting a reproducer. Also track how many arguments it has and if the
54 /// option is some kind of include path.
55 static bool skipArgs(const char *Flag, bool HaveCrashVFS, int &SkipNum,
56  bool &IsInclude) {
57  SkipNum = 2;
58  // These flags are all of the form -Flag <Arg> and are treated as two
59  // arguments. Therefore, we need to skip the flag and the next argument.
60  bool ShouldSkip = llvm::StringSwitch<bool>(Flag)
61  .Cases("-MF", "-MT", "-MQ", "-serialize-diagnostic-file", true)
62  .Cases("-o", "-dependency-file", true)
63  .Cases("-fdebug-compilation-dir", "-diagnostic-log-file", true)
64  .Cases("-dwarf-debug-flags", "-ivfsoverlay", true)
65  .Default(false);
66  if (ShouldSkip)
67  return true;
68 
69  // Some include flags shouldn't be skipped if we have a crash VFS
70  IsInclude = llvm::StringSwitch<bool>(Flag)
71  .Cases("-include", "-header-include-file", true)
72  .Cases("-idirafter", "-internal-isystem", "-iwithprefix", true)
73  .Cases("-internal-externc-isystem", "-iprefix", true)
74  .Cases("-iwithprefixbefore", "-isystem", "-iquote", true)
75  .Cases("-isysroot", "-I", "-F", "-resource-dir", true)
76  .Cases("-iframework", "-include-pch", true)
77  .Default(false);
78  if (IsInclude)
79  return !HaveCrashVFS;
80 
81  // The remaining flags are treated as a single argument.
82 
83  // These flags are all of the form -Flag and have no second argument.
84  ShouldSkip = llvm::StringSwitch<bool>(Flag)
85  .Cases("-M", "-MM", "-MG", "-MP", "-MD", true)
86  .Case("-MMD", true)
87  .Default(false);
88 
89  // Match found.
90  SkipNum = 1;
91  if (ShouldSkip)
92  return true;
93 
94  // These flags are treated as a single argument (e.g., -F<Dir>).
95  StringRef FlagRef(Flag);
96  IsInclude = FlagRef.startswith("-F") || FlagRef.startswith("-I");
97  if (IsInclude)
98  return !HaveCrashVFS;
99  if (FlagRef.startswith("-fmodules-cache-path="))
100  return true;
101 
102  SkipNum = 0;
103  return false;
104 }

106 void Command::writeResponseFile(raw_ostream &OS) const {
107  // In a file list, we only write the set of inputs to the response file
108  if (ResponseSupport.ResponseKind == ResponseFileSupport::RF_FileList) {
109  for (const auto *Arg : InputFileList) {
110  OS << Arg << '\n';
111  }
112  return;
113  }
114 
115  // In regular response files, we send all arguments to the response file.
116  // Wrapping all arguments in double quotes ensures that both Unix tools and
117  // Windows tools understand the response file.
118  for (const auto *Arg : Arguments) {
119  OS << '"';
120 
121  for (; *Arg != '\0'; Arg++) {
122  if (*Arg == '\"' || *Arg == '\\') {
123  OS << '\\';
124  }
125  OS << *Arg;
126  }
127 
128  OS << "\" ";
129  }
130 }

132 void Command::buildArgvForResponseFile(
133  llvm::SmallVectorImpl<const char *> &Out) const {
134  // When not a file list, all arguments are sent to the response file.
135  // This leaves us to set the argv to a single parameter, requesting the tool
136  // to read the response file.
137  if (ResponseSupport.ResponseKind != ResponseFileSupport::RF_FileList) {
138  Out.push_back(Executable);
139  Out.push_back(ResponseFileFlag.c_str());
140  return;
141  }
142 
143  llvm::StringSet<> Inputs;
144  for (const auto *InputName : InputFileList)
145  Inputs.insert(InputName);
146  Out.push_back(Executable);
147  // In a file list, build args vector ignoring parameters that will go in the
148  // response file (elements of the InputFileList vector)
149  bool FirstInput = true;
150  for (const auto *Arg : Arguments) {
151  if (Inputs.count(Arg) == 0) {
152  Out.push_back(Arg);
153  } else if (FirstInput) {
154  FirstInput = false;
155  Out.push_back(ResponseSupport.ResponseFlag);
156  Out.push_back(ResponseFile);
157  }
158  }
159 }

183 /// Rewrite relative include-like flag paths to absolute ones.
184 static void
185 rewriteIncludes(const llvm::ArrayRef<const char *> &Args, size_t Idx,
186  size_t NumArgs,
187  llvm::SmallVectorImpl<llvm::SmallString<128>> &IncFlags) {
188  using namespace llvm;
189  using namespace sys;
190 
191  auto getAbsPath = [](StringRef InInc, SmallVectorImpl<char> &OutInc) -> bool {
192  if (path::is_absolute(InInc)) // Nothing to do here...
193  return false;
194  std::error_code EC = fs::current_path(OutInc);
195  if (EC)
196  return false;
197  path::append(OutInc, InInc);
198  return true;
199  };
200 
201  SmallString<128> NewInc;
202  if (NumArgs == 1) {
203  StringRef FlagRef(Args[Idx + NumArgs - 1]);
204  assert((FlagRef.startswith("-F") || FlagRef.startswith("-I")) &&
205  "Expecting -I or -F");
206  StringRef Inc = FlagRef.slice(2, StringRef::npos);
207  if (getAbsPath(Inc, NewInc)) {
208  SmallString<128> NewArg(FlagRef.slice(0, 2));
209  NewArg += NewInc;
210  IncFlags.push_back(std::move(NewArg));
211  }
212  return;
213  }
214 
215  assert(NumArgs == 2 && "Not expecting more than two arguments");
216  StringRef Inc(Args[Idx + NumArgs - 1]);
217  if (!getAbsPath(Inc, NewInc))
218  return;
219  IncFlags.push_back(SmallString<128>(Args[Idx]));
220  IncFlags.push_back(std::move(NewInc));
221 }

223 void Command::Print(raw_ostream &OS, const char *Terminator, bool Quote,
224  CrashReportInfo *CrashInfo) const {
225  // Always quote the exe.
226  OS << ' ';
227  llvm::sys::printArg(OS, Executable, /*Quote=*/true);
228 
229  ArrayRef<const char *> Args = Arguments;
230  SmallVector<const char *, 128> ArgsRespFile;
231  if (ResponseFile != nullptr) {
232  buildArgvForResponseFile(ArgsRespFile);
233  Args = ArrayRef<const char *>(ArgsRespFile).slice(1); // no executable name
234  }
235 
236  bool HaveCrashVFS = CrashInfo && !CrashInfo->VFSPath.empty();
237  for (size_t i = 0, e = Args.size(); i < e; ++i) {
238  const char *const Arg = Args[i];
239 
240  if (CrashInfo) {
241  int NumArgs = 0;
242  bool IsInclude = false;
243  if (skipArgs(Arg, HaveCrashVFS, NumArgs, IsInclude)) {
244  i += NumArgs - 1;
245  continue;
246  }
247 
248  // Relative includes need to be expanded to absolute paths.
249  if (HaveCrashVFS && IsInclude) {
250  SmallVector<SmallString<128>, 2> NewIncFlags;
251  rewriteIncludes(Args, i, NumArgs, NewIncFlags);
252  if (!NewIncFlags.empty()) {
253  for (auto &F : NewIncFlags) {
254  OS << ' ';
255  llvm::sys::printArg(OS, F.c_str(), Quote);
256  }
257  i += NumArgs - 1;
258  continue;
259  }
260  }
261 
262  auto Found = llvm::find_if(InputInfoList, [&Arg](const InputInfo &II) {
263  return II.getFilename() == Arg;
264  });
265  if (Found != InputInfoList.end() &&
266  (i == 0 || StringRef(Args[i - 1]) != "-main-file-name")) {
267  // Replace the input file name with the crashinfo's file name.
268  OS << ' ';
269  StringRef ShortName = llvm::sys::path::filename(CrashInfo->Filename);
270  llvm::sys::printArg(OS, ShortName.str(), Quote);
271  continue;
272  }
273  }
274 
275  OS << ' ';
276  llvm::sys::printArg(OS, Arg, Quote);
277  }
278 
279  if (CrashInfo && HaveCrashVFS) {
280  OS << ' ';
281  llvm::sys::printArg(OS, "-ivfsoverlay", Quote);
282  OS << ' ';
283  llvm::sys::printArg(OS, CrashInfo->VFSPath.str(), Quote);
284 
285  // The leftover modules from the crash are stored in
286  // <name>.cache/vfs/modules
287  // Leave it untouched for pcm inspection and provide a clean/empty dir
288  // path to contain the future generated module cache:
289  // <name>.cache/vfs/repro-modules
290  SmallString<128> RelModCacheDir = llvm::sys::path::parent_path(
291  llvm::sys::path::parent_path(CrashInfo->VFSPath));
292  llvm::sys::path::append(RelModCacheDir, "repro-modules");
293 
294  std::string ModCachePath = "-fmodules-cache-path=";
295  ModCachePath.append(RelModCacheDir.c_str());
296 
297  OS << ' ';
298  llvm::sys::printArg(OS, ModCachePath, Quote);
299  }
300 
301  if (ResponseFile != nullptr) {
302  OS << "\n Arguments passed via response file:\n";
303  writeResponseFile(OS);
304  // Avoiding duplicated newline terminator, since FileLists are
305  // newline-separated.
306  if (ResponseSupport.ResponseKind != ResponseFileSupport::RF_FileList)
307  OS << "\n";
308  OS << " (end of response file)";
309  }
310 
311  OS << Terminator;
312 }

334 int Command::Execute(ArrayRef<llvm::Optional<StringRef>> Redirects,
335  std::string *ErrMsg, bool *ExecutionFailed) const {
336  PrintFileNames();
337 
338  SmallVector<const char *, 128> Argv;
339  if (ResponseFile == nullptr) {
340  Argv.push_back(Executable);
341  Argv.append(Arguments.begin(), Arguments.end());
342  Argv.push_back(nullptr);
343  } else {
344  // If the command is too large, we need to put arguments in a response file.
345  std::string RespContents;
346  llvm::raw_string_ostream SS(RespContents);
347 
348  // Write file contents and build the Argv vector
349  writeResponseFile(SS);
350  buildArgvForResponseFile(Argv);
351  Argv.push_back(nullptr);
352  SS.flush();
353 
354  // Save the response file in the appropriate encoding
355  if (std::error_code EC = writeFileWithEncoding(
356  ResponseFile, RespContents, ResponseSupport.ResponseEncoding)) {
357  if (ErrMsg)
358  *ErrMsg = EC.message();
359  if (ExecutionFailed)
360  *ExecutionFailed = true;
361  // Return -1 by convention (see llvm/include/llvm/Support/Program.h) to
362  // indicate the requested executable cannot be started.
363  return -1;
364  }
365  }
366 
367  Optional<ArrayRef<StringRef>> Env;
368  std::vector<StringRef> ArgvVectorStorage;
369  if (!Environment.empty()) {
370  assert(Environment.back() == nullptr &&
371  "Environment vector should be null-terminated by now");
372  ArgvVectorStorage = llvm::toStringRefArray(Environment.data());
373  Env = makeArrayRef(ArgvVectorStorage);
374  }
375 
376  auto Args = llvm::toStringRefArray(Argv.data());
377  return llvm::sys::ExecuteAndWait(Executable, Args, Env, Redirects,
378  /*secondsToWait*/ 0, /*memoryLimit*/ 0,
379  ErrMsg, ExecutionFailed, &ProcStat);
380 }

399 int CC1Command::Execute(ArrayRef<llvm::Optional<StringRef>> Redirects,
400  std::string *ErrMsg, bool *ExecutionFailed) const {
401  // FIXME: Currently, if there're more than one job, we disable
402  // -fintegrate-cc1. If we're no longer a integrated-cc1 job, fallback to
403  // out-of-process execution. See discussion in https://reviews.llvm.org/D74447
404  if (!InProcess)
405  return Command::Execute(Redirects, ErrMsg, ExecutionFailed);
406 
407  PrintFileNames();
408 
409  SmallVector<const char *, 128> Argv;
410  Argv.push_back(getExecutable());
411  Argv.append(getArguments().begin(), getArguments().end());
412  Argv.push_back(nullptr);
413  Argv.pop_back(); // The terminating null element shall not be part of the
414  // slice (main() behavior).
415 
416  // This flag simply indicates that the program couldn't start, which isn't
417  // applicable here.
418  if (ExecutionFailed)
419  *ExecutionFailed = false;
420 
421  llvm::CrashRecoveryContext CRC;
422  CRC.DumpStackAndCleanupOnFailure = true;
423 
424  const void *PrettyState = llvm::SavePrettyStackState();
425  const Driver &D = getCreator().getToolChain().getDriver();
426 
427  int R = 0;
428  // Enter ExecuteCC1Tool() instead of starting up a new process
429  if (!CRC.RunSafely([&]() { R = D.CC1Main(Argv); })) {
430  llvm::RestorePrettyStackState(PrettyState);
431  return CRC.RetCode;
432  }
433  return R;
434 }
