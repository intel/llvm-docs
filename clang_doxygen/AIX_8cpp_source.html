26 void aix::Assembler::ConstructJob(Compilation &C, const JobAction &JA,
27  const InputInfo &Output,
28  const InputInfoList &Inputs,
29  const ArgList &Args,
30  const char *LinkingOutput) const {
31  ArgStringList CmdArgs;
32 
33  const bool IsArch32Bit = getToolChain().getTriple().isArch32Bit();
34  const bool IsArch64Bit = getToolChain().getTriple().isArch64Bit();
35  // Only support 32 and 64 bit.
36  if (!IsArch32Bit && !IsArch64Bit)
37  llvm_unreachable("Unsupported bit width value.");
38 
39  // Specify the mode in which the as(1) command operates.
40  if (IsArch32Bit) {
41  CmdArgs.push_back("-a32");
42  } else {
43  // Must be 64-bit, otherwise asserted already.
44  CmdArgs.push_back("-a64");
45  }
46 
47  // Accept any mixture of instructions.
48  // On Power for AIX and Linux, this behaviour matches that of GCC for both the
49  // user-provided assembler source case and the compiler-produced assembler
50  // source case. Yet XL with user-provided assembler source would not add this.
51  CmdArgs.push_back("-many");
52 
53  Args.AddAllArgValues(CmdArgs, options::OPT_Wa_COMMA, options::OPT_Xassembler);
54 
55  // Specify assembler output file.
56  assert((Output.isFilename() || Output.isNothing()) && "Invalid output.");
57  if (Output.isFilename()) {
58  CmdArgs.push_back("-o");
59  CmdArgs.push_back(Output.getFilename());
60  }
61 
62  // Specify assembler input file.
63  // The system assembler on AIX takes exactly one input file. The driver is
64  // expected to invoke as(1) separately for each assembler source input file.
65  if (Inputs.size() != 1)
66  llvm_unreachable("Invalid number of input files.");
67  const InputInfo &II = Inputs[0];
68  assert((II.isFilename() || II.isNothing()) && "Invalid input.");
69  if (II.isFilename())
70  CmdArgs.push_back(II.getFilename());
71 
72  const char *Exec = Args.MakeArgString(getToolChain().GetProgramPath("as"));
73  C.addCommand(std::make_unique<Command>(JA, *this, ResponseFileSupport::None(),
74  Exec, CmdArgs, Inputs, Output));
75 }

77 void aix::Linker::ConstructJob(Compilation &C, const JobAction &JA,
78  const InputInfo &Output,
79  const InputInfoList &Inputs, const ArgList &Args,
80  const char *LinkingOutput) const {
81  const AIX &ToolChain = static_cast<const AIX &>(getToolChain());
82  const Driver &D = ToolChain.getDriver();
83  ArgStringList CmdArgs;
84 
85  const bool IsArch32Bit = ToolChain.getTriple().isArch32Bit();
86  const bool IsArch64Bit = ToolChain.getTriple().isArch64Bit();
87  // Only support 32 and 64 bit.
88  if (!(IsArch32Bit || IsArch64Bit))
89  llvm_unreachable("Unsupported bit width value.");
90 
91  // Force static linking when "-static" is present.
92  if (Args.hasArg(options::OPT_static))
93  CmdArgs.push_back("-bnso");
94 
95  // Add options for shared libraries.
96  if (Args.hasArg(options::OPT_shared)) {
97  CmdArgs.push_back("-bM:SRE");
98  CmdArgs.push_back("-bnoentry");
99  }
100 
101  // Specify PGO linker option without LTO
102  if (!D.isUsingLTO() &&
103  (Args.hasFlag(options::OPT_fprofile_arcs, options::OPT_fno_profile_arcs,
104  false) ||
105  Args.hasFlag(options::OPT_fprofile_generate,
106  options::OPT_fno_profile_generate, false) ||
107  Args.hasFlag(options::OPT_fprofile_generate_EQ,
108  options::OPT_fno_profile_generate, false) ||
109  Args.hasFlag(options::OPT_fprofile_instr_generate,
110  options::OPT_fno_profile_instr_generate, false) ||
111  Args.hasFlag(options::OPT_fprofile_instr_generate_EQ,
112  options::OPT_fno_profile_instr_generate, false) ||
113  Args.hasFlag(options::OPT_fcs_profile_generate,
114  options::OPT_fno_profile_generate, false) ||
115  Args.hasFlag(options::OPT_fcs_profile_generate_EQ,
116  options::OPT_fno_profile_generate, false) ||
117  Args.hasArg(options::OPT_fcreate_profile) ||
118  Args.hasArg(options::OPT_coverage)))
119  CmdArgs.push_back("-bdbg:namedsects");
120 
121  // Specify linker output file.
122  assert((Output.isFilename() || Output.isNothing()) && "Invalid output.");
123  if (Output.isFilename()) {
124  CmdArgs.push_back("-o");
125  CmdArgs.push_back(Output.getFilename());
126  }
127 
128  // Set linking mode (i.e., 32/64-bit) and the address of
129  // text and data sections based on arch bit width.
130  if (IsArch32Bit) {
131  CmdArgs.push_back("-b32");
132  CmdArgs.push_back("-bpT:0x10000000");
133  CmdArgs.push_back("-bpD:0x20000000");
134  } else {
135  // Must be 64-bit, otherwise asserted already.
136  CmdArgs.push_back("-b64");
137  CmdArgs.push_back("-bpT:0x100000000");
138  CmdArgs.push_back("-bpD:0x110000000");
139  }
140 
141  auto getCrt0Basename = [&Args, IsArch32Bit] {
142  // Enable gprofiling when "-pg" is specified.
143  if (Args.hasArg(options::OPT_pg))
144  return IsArch32Bit ? "gcrt0.o" : "gcrt0_64.o";
145  // Enable profiling when "-p" is specified.
146  else if (Args.hasArg(options::OPT_p))
147  return IsArch32Bit ? "mcrt0.o" : "mcrt0_64.o";
148  else
149  return IsArch32Bit ? "crt0.o" : "crt0_64.o";
150  };
151 
152  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nostartfiles,
153  options::OPT_shared)) {
154  CmdArgs.push_back(
155  Args.MakeArgString(ToolChain.GetFilePath(getCrt0Basename())));
156 
157  CmdArgs.push_back(Args.MakeArgString(
158  ToolChain.GetFilePath(IsArch32Bit ? "crti.o" : "crti_64.o")));
159  }
160 
161  // Collect all static constructor and destructor functions in both C and CXX
162  // language link invocations. This has to come before AddLinkerInputs as the
163  // implied option needs to precede any other '-bcdtors' settings or
164  // '-bnocdtors' that '-Wl' might forward.
165  CmdArgs.push_back("-bcdtors:all:0:s");
166 
167  // Specify linker input file(s).
168  AddLinkerInputs(ToolChain, Inputs, Args, CmdArgs, JA);
169 
170  // Add directory to library search path.
171  Args.AddAllArgs(CmdArgs, options::OPT_L);
172  ToolChain.AddFilePathLibArgs(Args, CmdArgs);
173  ToolChain.addProfileRTLibs(Args, CmdArgs);
174 
175  if (getToolChain().ShouldLinkCXXStdlib(Args))
176  getToolChain().AddCXXStdlibLibArgs(Args, CmdArgs);
177 
178  if (!Args.hasArg(options::OPT_nostdlib, options::OPT_nodefaultlibs)) {
179  AddRunTimeLibs(ToolChain, D, CmdArgs, Args);
180 
181  // Support POSIX threads if "-pthreads" or "-pthread" is present.
182  if (Args.hasArg(options::OPT_pthreads, options::OPT_pthread))
183  CmdArgs.push_back("-lpthreads");
184 
185  if (D.CCCIsCXX())
186  CmdArgs.push_back("-lm");
187 
188  CmdArgs.push_back("-lc");
189  }
190 
191  const char *Exec = Args.MakeArgString(ToolChain.GetLinkerPath());
192  C.addCommand(std::make_unique<Command>(JA, *this, ResponseFileSupport::None(),
193  Exec, CmdArgs, Inputs, Output));
194 }

215 void AIX::AddClangSystemIncludeArgs(const ArgList &DriverArgs,
216  ArgStringList &CC1Args) const {
217  // Return if -nostdinc is specified as a driver option.
218  if (DriverArgs.hasArg(options::OPT_nostdinc))
219  return;
220 
221  llvm::StringRef Sysroot = GetHeaderSysroot(DriverArgs);
222  const Driver &D = getDriver();
223 
224  // Add the Clang builtin headers (<resource>/include).
225  if (!DriverArgs.hasArg(options::OPT_nobuiltininc)) {
226  SmallString<128> P(D.ResourceDir);
227  path::append(P, "/include");
228  addSystemInclude(DriverArgs, CC1Args, P.str());
229  }
230 
231  // Return if -nostdlibinc is specified as a driver option.
232  if (DriverArgs.hasArg(options::OPT_nostdlibinc))
233  return;
234 
235  // Add <sysroot>/usr/include.
236  SmallString<128> UP(Sysroot);
237  path::append(UP, "/usr/include");
238  addSystemInclude(DriverArgs, CC1Args, UP.str());
239 }

241 void AIX::AddClangCXXStdlibIncludeArgs(
242  const llvm::opt::ArgList &DriverArgs,
243  llvm::opt::ArgStringList &CC1Args) const {
244 
245  if (DriverArgs.hasArg(options::OPT_nostdinc) ||
246  DriverArgs.hasArg(options::OPT_nostdincxx) ||
247  DriverArgs.hasArg(options::OPT_nostdlibinc))
248  return;
249 
250  switch (GetCXXStdlibType(DriverArgs)) {
251  case ToolChain::CST_Libstdcxx:
252  llvm::report_fatal_error(
253  "picking up libstdc++ headers is unimplemented on AIX");
254  case ToolChain::CST_Libcxx: {
255  llvm::StringRef Sysroot = GetHeaderSysroot(DriverArgs);
256  SmallString<128> PathCPP(Sysroot);
257  llvm::sys::path::append(PathCPP, "opt/IBM/openxlCSDK", "include", "c++",
258  "v1");
259  addSystemInclude(DriverArgs, CC1Args, PathCPP.str());
260  // Required in order to suppress conflicting C++ overloads in the system
261  // libc headers that were used by XL C++.
262  CC1Args.push_back("-D__LIBC_NO_CPP_MATH_OVERLOADS__");
263  return;
264  }
265  }
266 
267  llvm_unreachable("Unexpected C++ library type; only libc++ is supported.");
268 }
