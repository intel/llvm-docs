118 StringRef APISet::copyString(StringRef String) {
119  if (String.empty())
120  return {};
121 
122  // No need to allocate memory and copy if the string has already been stored.
123  if (StringAllocator.identifyObject(String.data()))
124  return String;
125 
126  void *Ptr = StringAllocator.Allocate(String.size(), 1);
127  memcpy(Ptr, String.data(), String.size());
128  return StringRef(reinterpret_cast<const char *>(Ptr), String.size());
129 }
