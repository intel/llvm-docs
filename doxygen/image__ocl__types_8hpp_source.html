187 template <typename RetType, typename ImageT, typename CoordT>
188 static RetType __invoke__ImageReadSampler(ImageT Img, CoordT Coords,
189  const __ocl_sampler_t &Smpl) {
190 
191  // Convert from sycl types to builtin types to get correct function mangling.
192  using TempRetT = sycl::detail::ConvertToOpenCLType_t<RetType>;
193  using SampledT =
194  typename sycl::detail::sampled_opencl_image_type<ImageT>::type;
195 
196  auto TmpCoords = sycl::detail::convertToOpenCLType(Coords);
197  // According to validation rules(SPIR-V specification, section 2.16.1) result
198  // of __spirv_SampledImage is allowed to be an operand of image lookup
199  // and query instructions explicitly specified to take an operand whose
200  // type is OpTypeSampledImage.
201  //
202  // According to SPIR-V specification section 3.32.10 at least one operand
203  // setting the level of detail must be present. The last two arguments of
204  // __spirv_ImageSampleExplicitLod represent image operand type and value.
205  // From the SPIR-V specification section 3.14:
206  enum ImageOperands { Lod = 0x2 };
207 
208  // Lod value is zero as mipmap is not supported.
209  return sycl::detail::convertFromOpenCLTypeFor<RetType>(
210  __spirv_ImageSampleExplicitLod<SampledT, TempRetT, decltype(TmpCoords)>(
211  __spirv_SampledImage<ImageT, SampledT>(Img, Smpl), TmpCoords,
212  ImageOperands::Lod, 0.0f));
213 }

219 // Function to return the number of channels for Image Channel Order returned by
220 // SPIR-V call to OpImageQueryOrder.
221 // The returned int value represents an enum from Image Channel Order. The enums
222 // for Image Channel Order are mapped differently in sycl and SPIR-V spec.
223 inline int getSPIRVNumChannels(int ImageChannelOrder) {
224  switch (ImageChannelOrder) {
225  case 0: // R
226  case 1: // A
227  case 10: // Rx
228  case 8: // Intensity
229  case 9: // Luminance
230  return 1;
231  case 2: // RG
232  case 3: // RA
233  case 11: // RGx
234  return 2;
235  case 4: // RGB
236  return 3;
237  case 5: // RGBA
238  case 6: // BGRA
239  case 7: // ARGB
240  case 12: // RGBx
241  case 19: // ABGR
242  case 17: // sRGBA
243  return 4;
244  case 13: // Depth
245  case 14: // DepthStencil
246  case 18: // sBGRA
247  // TODO: Enable the below assert after assert is supported for device
248  // compiler. assert(!"Unhandled image channel order in sycl.");
249  default:
250  return 0;

254 // Function to compute the Element Size for a given Image Channel Type and Image
255 // Channel Order, returned by SPIR-V calls to OpImageQueryFormat and
256 // OpImageQueryOrder respectively.
257 // The returned int value from OpImageQueryFormat represents an enum from Image
258 // Channel Data Type. The enums for Image Channel Data Type are mapped
259 // differently in sycl and SPIR-V spec.
260 inline int getSPIRVElementSize(int ImageChannelType, int ImageChannelOrder) {
261  int NumChannels = getSPIRVNumChannels(ImageChannelOrder);
262  switch (ImageChannelType) {
263  case 0: // SnormInt8
264  case 2: // UnormInt8
265  case 7: // SignedInt8
266  case 10: // UnsignedInt8
267  return NumChannels;
268  case 1: // SnormInt16
269  case 3: // UnormInt16
270  case 8: // SignedInt16
271  case 11: // UnsignedInt16
272  case 13: // HalfFloat
273  return 2 * NumChannels;
274  case 4: // UnormShort565
275  case 5: // UnormShort555
276  return 2;
277  case 6: // UnormInt101010
278  return 4;
279  case 9: // SignedInt32
280  case 12: // UnsignedInt32
281  case 14: // Float
282  return 4 * NumChannels;
283  case 15: // UnormInt24
284  case 16: // UnormInt101010_2
285  default:
286  // TODO: Enable the below assert after assert is supported for device
287  // compiler. assert(!"Unhandled image channel type in sycl.");
288  return 0;
