172 template <typename T>
173 EnableIfFP<T, unsigned> floatingPointToDecStr(T AbsVal, char *Digits,
174  int Precision, bool IsSci) {
175  int Exp = 0;
176 
177  // For the case that the value is larger than 10.0
178  while (AbsVal >= 10.0) {
179  ++Exp;
180  AbsVal /= 10.0;
181  }
182  // For the case that the value is less than 1.0
183  while (AbsVal > 0.0 && AbsVal < 1.0) {
184  --Exp;
185  AbsVal *= 10.0;
186  }
187 
188  auto IntegralPart = static_cast<int>(AbsVal);
189  auto FractionPart = AbsVal - IntegralPart;
190 
191  int FractionDigits[MAX_FLOATING_POINT_DIGITS] = {0};
192 
193  // Exponent
194  int P = Precision > 0 ? Precision : 4;
195  size_t FractionLength = Exp + P;
196 
197  // After normalization integral part contains 1 symbol, also there could be
198  // '.', 'e', sign of the exponent and sign of the number, overall 5 symbols.
199  // So, clamp fraction length if required according to maximum size of the
200  // buffer for floating point number.
201  if (FractionLength > MAX_FLOATING_POINT_DIGITS - 5)
202  FractionLength = MAX_FLOATING_POINT_DIGITS - 5;
203 
204  for (unsigned I = 0; I < FractionLength; ++I) {
205  FractionPart *= 10.0;
206  FractionDigits[I] = static_cast<int>(FractionPart);
207  FractionPart -= static_cast<int>(FractionPart);
208  }
209 
210  int Carry = FractionPart > static_cast<T>(0.5) ? 1 : 0;
211 
212  // Propagate the Carry
213  for (int I = FractionLength - 1; I >= 0 && Carry; --I) {
214  auto Digit = FractionDigits[I] + Carry;
215  FractionDigits[I] = Digit % 10;
216  Carry = Digit / 10;
217  }
218 
219  // Carry from the fraction part is propagated to integral part
220  IntegralPart += Carry;
221  if (IntegralPart == 10) {
222  IntegralPart = 1;
223  ++Exp;
224  }
225 
226  unsigned Offset = 0;
227 
228  // Assemble the final string correspondingly
229  if (IsSci) { // scientific mode
230  // Append the integral part
231  Digits[Offset++] = digitToChar(IntegralPart);
232  Digits[Offset++] = '.';
233 
234  // Append all fraction
235  for (unsigned I = 0; I < FractionLength; ++I)
236  Digits[Offset++] = digitToChar(FractionDigits[I]);
237 
238  // Exponent part
239  Digits[Offset++] = 'e';
240  Digits[Offset++] = Exp >= 0 ? '+' : '-';
241  Digits[Offset++] = digitToChar(abs(Exp) / 10);
242  Digits[Offset++] = digitToChar(abs(Exp) % 10);
243  } else { // normal mode
244  if (Exp < 0) {
245  Digits[Offset++] = '0';
246  Digits[Offset++] = '.';
247  while (++Exp)
248  Digits[Offset++] = '0';
249 
250  // Append the integral part
251  Digits[Offset++] = digitToChar(IntegralPart);
252 
253  // Append all fraction
254  for (unsigned I = 0; I < FractionLength; ++I)
255  Digits[Offset++] = digitToChar(FractionDigits[I]);
256  } else {
257  // Append the integral part
258  Digits[Offset++] = digitToChar(IntegralPart);
259  unsigned I = 0;
260  // Append the integral part first
261  for (; I < FractionLength && Exp--; ++I)
262  Digits[Offset++] = digitToChar(FractionDigits[I]);
263 
264  // Put the dot
265  Digits[Offset++] = '.';
266 
267  // Append the rest of fraction part, or the real fraction part
268  for (; I < FractionLength; ++I)
269  Digits[Offset++] = digitToChar(FractionDigits[I]);
270  }
271  // The normal mode requires no tailing zero digit, then we need to first
272  // find the first non-zero digit
273  while (Digits[Offset - 1] == '0')
274  Offset--;
275 
276  // If dot is the last digit, it should be stripped off as well
277  if (Digits[Offset - 1] == '.')
278  Offset--;
279  }
280  return Offset;
281 }

283 // Helper method to update offset in the global buffer atomically according to
284 // the provided size of the data in the flush buffer. Return true if offset is
285 // updated and false in case of overflow.
286 inline bool updateOffset(stream_impl::GlobalOffsetAccessorT &GlobalOffset,
287  stream_impl::GlobalBufAccessorT &GlobalBuf,
288  unsigned Size, unsigned &Cur) {
289  unsigned New;
290  Cur = GlobalOffset[0].load();
291  do {
292  if (GlobalBuf.get_range().size() - Cur < Size)
293  // Overflow
294  return false;
295  New = Cur + Size;
296  } while (!GlobalOffset[0].compare_exchange_strong(Cur, New));
297  return true;
298 }

300 inline void flushBuffer(stream_impl::GlobalOffsetAccessorT &GlobalOffset,
301  stream_impl::GlobalBufAccessorT &GlobalBuf,
302  stream_impl::GlobalBufAccessorT &GlobalFlushBuf,
303  unsigned &WIOffset, unsigned &Offset) {
304 
305  unsigned Cur = 0;
306  if (!updateOffset(GlobalOffset, GlobalBuf, Offset, Cur))
307  return;
308 
309  for (unsigned I = WIOffset; I < WIOffset + Offset; I++) {
310  GlobalBuf[Cur++] = GlobalFlushBuf[I];
311  }
312  // Reset the offset in the flush buffer
313  Offset = 0;

333 inline void reverseBuf(char *Buf, unsigned Len) {
334  int I = Len - 1;
335  int J = 0;
336  while (I > J) {
337  int Temp = Buf[I];
338  Buf[I] = Buf[J];
339  Buf[J] = Temp;
340  I--;
341  J++;
342  }
343 }

345 inline unsigned append(char *Dst, const char *Src) {
346  unsigned Len = 0;
347  for (; Src[Len] != '\0'; ++Len)
348  ;
349 
350  for (unsigned I = 0; I < Len; ++I)
351  Dst[I] = Src[I];
352  return Len;
353 }

355 template <typename T>
356 inline typename std::enable_if<std::is_same<T, half>::value, unsigned>::type
357 checkForInfNan(char *Buf, T Val) {
358  if (Val != Val)
359  return append(Buf, "nan");
360 
361  // Extract the sign from the bits
362  const uint16_t Sign = reinterpret_cast<uint16_t &>(Val) & 0x8000;
363  // Extract the exponent from the bits
364  const uint16_t Exp16 = (reinterpret_cast<uint16_t &>(Val) & 0x7c00) >> 10;
365 
366  if (Exp16 == 0x1f) {
367  if (Sign)
368  return append(Buf, "-inf");
369  return append(Buf, "inf");
370  }
371  return 0;
372 }

389 // Returns number of symbols written to the buffer
390 template <typename T>
391 inline EnableIfFP<T, unsigned>
392 ScalarToStr(const T &Val, char *Buf, unsigned Flags, int, int Precision = -1) {
393  unsigned Offset = checkForInfNan(Buf, Val);
394  if (Offset)
395  return Offset;
396 
397  T Neg = -Val;
398  auto AbsVal = Val < 0 ? Neg : Val;
399 
400  if (Val < 0) {
401  Buf[Offset++] = '-';
402  } else if (Flags & ShowPos) {
403  Buf[Offset++] = '+';
404  }
405 
406  bool IsSci = false;
407  if (Flags & detail::Scientific)
408  IsSci = true;
409 
410  // TODO: manipulators for floating-point output - hexfloat, fixed
411  Offset += floatingPointToDecStr(AbsVal, Buf + Offset, Precision, IsSci);
412 
413  return Offset;
414 }

416 // Returns number of symbols written to the buffer
417 template <typename T>
418 inline typename std::enable_if<std::is_integral<T>::value, unsigned>::type
419 ScalarToStr(const T &Val, char *Buf, unsigned Flags, int, int Precision = -1) {
420  (void)Precision;
421  int Base = 10;
422 
423  // append base manipulator
424  switch (Flags & BaseField) {
425  case Dec:
426  Base = 10;
427  break;
428  case Hex:
429  Base = 16;
430  break;
431  case Oct:
432  Base = 8;
433  break;
434  default:
435  // default value is 10
436  break;
437  }
438 
439  unsigned Offset = 0;
440 
441  // write '+' to the stream if the base is 10 and the value is non-negative
442  // or write '-' to stream if base is 10 and the value is negative
443  if (Base == 10) {
444  if ((Flags & ShowPos) && Val >= 0)
445  Buf[Offset++] = '+';
446  else if (Val < 0)
447  Buf[Offset++] = '-';
448  }
449 
450  // write 0 or 0x to the stream if base is not 10 and the manipulator is set
451  if (Base != 10 && (Flags & ShowBase)) {
452  Buf[Offset++] = '0';
453  if (Base == 16)
454  Buf[Offset++] = 'x';
455  }
456 
457  auto AbsVal = getAbsVal(Val, Base);
458 
459  const unsigned NumBuf = integralToBase(AbsVal, Base, Buf + Offset);
460 
461  reverseBuf(Buf + Offset, NumBuf);
462  return Offset + NumBuf;
463 }
