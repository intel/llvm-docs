20 std::ostream &operator<<(std::ostream &Out, const DeviceBinaryProperty &P) {
21  switch (P.Prop->Type) {
22  case SYCL_PROPERTY_TYPE_UINT32:
23  Out << "[UINT32] ";
24  break;
25  case SYCL_PROPERTY_TYPE_BYTE_ARRAY:
26  Out << "[Byte array] ";
27  break;
28  case SYCL_PROPERTY_TYPE_STRING:
29  Out << "[String] ";
30  break;
31  default:
32  assert(false && "unsupported property");
33  return Out;
34  }
35  Out << P.Prop->Name << "=";
36 
37  switch (P.Prop->Type) {
38  case SYCL_PROPERTY_TYPE_UINT32:
39  Out << P.asUint32();
40  break;
41  case SYCL_PROPERTY_TYPE_BYTE_ARRAY: {
42  ByteArray BA = P.asByteArray();
43  std::ios_base::fmtflags FlagsBackup = Out.flags();
44  Out << std::hex;
45  for (const auto &Byte : BA) {
46  Out << "0x" << static_cast<unsigned>(Byte) << " ";
47  }
48  Out.flags(FlagsBackup);
49  break;
50  }
51  case SYCL_PROPERTY_TYPE_STRING:
52  Out << P.asCString();
53  break;
54  default:
55  assert(false && "Unsupported property");
56  return Out;
57  }
58  return Out;
59 }

61 uint32_t DeviceBinaryProperty::asUint32() const {
62  assert(Prop->Type == SYCL_PROPERTY_TYPE_UINT32 && "property type mismatch");
63  // if type fits into the ValSize - it is used to store the property value
64  assert(Prop->ValAddr == nullptr && "primitive types must be stored inline");
65  const auto *P = reinterpret_cast<const unsigned char *>(&Prop->ValSize);
66  return (*P) | (*(P + 1) << 8) | (*(P + 2) << 16) | (*(P + 3) << 24);
67 }

77 const char *DeviceBinaryProperty::asCString() const {
78  assert((Prop->Type == SYCL_PROPERTY_TYPE_STRING ||
79  Prop->Type == SYCL_PROPERTY_TYPE_BYTE_ARRAY) &&
80  "property type mismatch");

84  return ur::cast<const char *>(Prop->ValAddr) + Shift;

87 void RTDeviceBinaryImage::PropertyRange::init(sycl_device_binary Bin,
88  const char *PropSetName) {
89  assert(!this->Begin && !this->End && "already initialized");
90  sycl_device_binary_property_set PS = nullptr;
91 
92  for (PS = Bin->PropertySetsBegin; PS != Bin->PropertySetsEnd; ++PS) {
93  assert(PS->Name && "nameless property set - bug in the offload wrapper?");
94  if (!strcmp(PropSetName, PS->Name))
95  break;
96  }
97  if (PS == Bin->PropertySetsEnd) {
98  Begin = End = nullptr;
99  return;
100  }
101  Begin = PS->PropertiesBegin;
102  End = Begin ? PS->PropertiesEnd : nullptr;
103 }

160 void RTDeviceBinaryImage::init(sycl_device_binary Bin) {
161  // Bin != nullptr is guaranteed here.
162  this->Bin = Bin;
163  // If device binary image format wasn't set by its producer, then can't change
164  // now, because 'Bin' data is part of the executable image loaded into memory
165  // which can't be modified (easily).
166  // TODO clang driver + ClangOffloadWrapper can figure out the format and set
167  // it when invoking the offload wrapper job
168  Format = static_cast<ur::DeviceBinaryType>(Bin->Format);
169 
170  if (Format == SYCL_DEVICE_BINARY_TYPE_NONE)
171  // try to determine the format; may remain "NONE"
172  Format = ur::getBinaryImageFormat(Bin->BinaryStart, getSize());
173 
174  SpecConstIDMap.init(Bin, __SYCL_PROPERTY_SET_SPEC_CONST_MAP);
175  SpecConstDefaultValuesMap.init(
176  Bin, __SYCL_PROPERTY_SET_SPEC_CONST_DEFAULT_VALUES_MAP);
177  DeviceLibReqMask.init(Bin, __SYCL_PROPERTY_SET_DEVICELIB_REQ_MASK);
178  KernelParamOptInfo.init(Bin, __SYCL_PROPERTY_SET_KERNEL_PARAM_OPT_INFO);
179  AssertUsed.init(Bin, __SYCL_PROPERTY_SET_SYCL_ASSERT_USED);
180  ProgramMetadata.init(Bin, __SYCL_PROPERTY_SET_PROGRAM_METADATA);
181  // Convert ProgramMetadata into the UR format
182  for (const auto &Prop : ProgramMetadata) {
183  ProgramMetadataUR.push_back(
184  ur::mapDeviceBinaryPropertyToProgramMetadata(Prop));
185  }
186  ExportedSymbols.init(Bin, __SYCL_PROPERTY_SET_SYCL_EXPORTED_SYMBOLS);
187  ImportedSymbols.init(Bin, __SYCL_PROPERTY_SET_SYCL_IMPORTED_SYMBOLS);
188  DeviceGlobals.init(Bin, __SYCL_PROPERTY_SET_SYCL_DEVICE_GLOBALS);
189  DeviceRequirements.init(Bin, __SYCL_PROPERTY_SET_SYCL_DEVICE_REQUIREMENTS);
190  HostPipes.init(Bin, __SYCL_PROPERTY_SET_SYCL_HOST_PIPES);
191  VirtualFunctions.init(Bin, __SYCL_PROPERTY_SET_SYCL_VIRTUAL_FUNCTIONS);
192 
193  ImageId = ImageCounter++;
194 }

198 DynRTDeviceBinaryImage::DynRTDeviceBinaryImage(
199  std::unique_ptr<char[]> &&DataPtr, size_t DataSize)
200  : RTDeviceBinaryImage() {
201  Data = std::move(DataPtr);
202  Bin = new sycl_device_binary_struct();
203  Bin->Version = SYCL_DEVICE_BINARY_VERSION;
204  Bin->Kind = SYCL_DEVICE_BINARY_OFFLOAD_KIND_SYCL;
205  Bin->CompileOptions = "";
206  Bin->LinkOptions = "";
207  Bin->ManifestStart = nullptr;
208  Bin->ManifestEnd = nullptr;
209  Bin->BinaryStart = reinterpret_cast<unsigned char *>(Data.get());
210  Bin->BinaryEnd = Bin->BinaryStart + DataSize;
211  Bin->EntriesBegin = nullptr;
212  Bin->EntriesEnd = nullptr;
213  Bin->Format = ur::getBinaryImageFormat(Bin->BinaryStart, DataSize);
214  switch (Bin->Format) {
215  case SYCL_DEVICE_BINARY_TYPE_SPIRV:
216  Bin->DeviceTargetSpec = __SYCL_DEVICE_BINARY_TARGET_SPIRV64;
217  break;
218  default:
219  Bin->DeviceTargetSpec = __SYCL_DEVICE_BINARY_TARGET_UNKNOWN;
220  }
221  init(Bin);
222 }
