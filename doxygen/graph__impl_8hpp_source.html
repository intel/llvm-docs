183  bool hasRequirementDependency(sycl::detail::AccessorImplHost *IncomingReq) {
184  access_mode InMode = IncomingReq->MAccessMode;
185  switch (InMode) {
186  case access_mode::read:
187  case access_mode::read_write:
188  case access_mode::atomic:
189  break;
190  // These access modes don't care about existing buffer data, so we don't
191  // need a dependency.
192  case access_mode::write:
193  case access_mode::discard_read_write:
194  case access_mode::discard_write:
195  return false;
196  }
197 
198  for (sycl::detail::AccessorImplHost *CurrentReq :
199  MCommandGroup->getRequirements()) {
200  if (IncomingReq->MSYCLMemObj == CurrentReq->MSYCLMemObj) {
201  access_mode CurrentMode = CurrentReq->MAccessMode;
202  // Since we have an incoming read requirement, we only care
203  // about requirements on this node if they are write
204  if (CurrentMode != access_mode::read) {
205  return true;
206  }
207  }
208  }
209  // No dependency necessary
210  return false;
211  }

314  bool isSimilar(const std::shared_ptr<node_impl> &Node,
315  bool CompareContentOnly = false) const {
316  if (!CompareContentOnly) {
317  if (MSuccessors.size() != Node->MSuccessors.size())
318  return false;
319 
320  if (MPredecessors.size() != Node->MPredecessors.size())
321  return false;
322  }
323  if (MCGType != Node->MCGType)
324  return false;
325 
326  switch (MCGType) {
327  case sycl::detail::CG::CGTYPE::Kernel: {
328  sycl::detail::CGExecKernel *ExecKernelA =
329  static_cast<sycl::detail::CGExecKernel *>(MCommandGroup.get());
330  sycl::detail::CGExecKernel *ExecKernelB =
331  static_cast<sycl::detail::CGExecKernel *>(Node->MCommandGroup.get());
332  return ExecKernelA->MKernelName.compare(ExecKernelB->MKernelName) == 0;
333  }
334  case sycl::detail::CG::CGTYPE::CopyUSM: {
335  sycl::detail::CGCopyUSM *CopyA =
336  static_cast<sycl::detail::CGCopyUSM *>(MCommandGroup.get());
337  sycl::detail::CGCopyUSM *CopyB =
338  static_cast<sycl::detail::CGCopyUSM *>(Node->MCommandGroup.get());
339  return (CopyA->getSrc() == CopyB->getSrc()) &&
340  (CopyA->getDst() == CopyB->getDst()) &&
341  (CopyA->getLength() == CopyB->getLength());
342  }
343  case sycl::detail::CG::CGTYPE::CopyAccToAcc:
344  case sycl::detail::CG::CGTYPE::CopyAccToPtr:
345  case sycl::detail::CG::CGTYPE::CopyPtrToAcc: {
346  sycl::detail::CGCopy *CopyA =
347  static_cast<sycl::detail::CGCopy *>(MCommandGroup.get());
348  sycl::detail::CGCopy *CopyB =
349  static_cast<sycl::detail::CGCopy *>(Node->MCommandGroup.get());
350  return (CopyA->getSrc() == CopyB->getSrc()) &&
351  (CopyA->getDst() == CopyB->getDst());
352  }
353  default:
354  assert(false && "Unexpected command group type!");
355  return false;
356  }

366  std::vector<node_impl *> &Visited, bool Verbose) {
367  // if Node has been already visited, we skip it
368  if (std::find(Visited.begin(), Visited.end(), this) != Visited.end())
369  return;
370 
371  Visited.push_back(this);
372 
373  printDotCG(Stream, Verbose);
374  for (const auto &Dep : MPredecessors) {
375  auto NodeDep = Dep.lock();
376  Stream << " \"" << NodeDep.get() << "\" -> \"" << this << "\""
377  << std::endl;
378  }

380  for (std::weak_ptr<node_impl> Succ : MSuccessors) {
381  if (MPartitionNum == Succ.lock()->MPartitionNum)
382  Succ.lock()->printDotRecursive(Stream, Visited, Verbose);
383  }

405  void updateAccessor(int ArgIndex, const sycl::detail::AccessorBaseHost *Acc) {
406  auto &Args =
407  static_cast<sycl::detail::CGExecKernel *>(MCommandGroup.get())->MArgs;
408  auto NewAccImpl = sycl::detail::getSyclObjImpl(*Acc);
409  for (auto &Arg : Args) {
410  if (Arg.MIndex != ArgIndex) {
411  continue;
412  }
413  assert(Arg.MType == sycl::detail::kernel_param_kind_t::kind_accessor);
414 
415  // Find old accessor in accessor storage and replace with new one
416  if (static_cast<sycl::detail::SYCLMemObjT *>(NewAccImpl->MSYCLMemObj)
417  ->needsWriteBack()) {
418  throw sycl::exception(
419  make_error_code(errc::invalid),
420  "Accessors to buffers which have write_back enabled "
421  "are not allowed to be used in command graphs.");
422  }
423 
424  // All accessors passed to this function will be placeholders, so we must
425  // perform steps similar to what happens when handler::require() is
426  // called here.
427  sycl::detail::Requirement *NewReq = NewAccImpl.get();
428  if (NewReq->MAccessMode != sycl::access_mode::read) {
429  auto SYCLMemObj =
430  static_cast<sycl::detail::SYCLMemObjT *>(NewReq->MSYCLMemObj);
431  SYCLMemObj->handleWriteAccessorCreation();
432  }
433 
434  for (auto &Acc : MCommandGroup->getAccStorage()) {
435  if (auto OldAcc =
436  static_cast<sycl::detail::AccessorImplHost *>(Arg.MPtr);
437  Acc.get() == OldAcc) {
438  Acc = NewAccImpl;
439  }
440  }
441 
442  for (auto &Req : MCommandGroup->getRequirements()) {
443  if (auto OldReq =
444  static_cast<sycl::detail::AccessorImplHost *>(Arg.MPtr);
445  Req == OldReq) {
446  Req = NewReq;
447  }
448  }
449  Arg.MPtr = NewAccImpl.get();
450  break;
451  }
452  }

543  void rebuildArgStorage(std::vector<sycl::detail::ArgDesc> &Args,
544  const std::vector<std::vector<char>> &OldArgStorage,
545  std::vector<std::vector<char>> &NewArgStorage) const {
546  // Clear the arg storage so we can rebuild it
547  NewArgStorage.clear();
548 
549  // Loop over all the args, any std_layout ones need their pointers updated
550  // to point to the new arg storage.
551  for (auto &Arg : Args) {
552  if (Arg.MType != sycl::detail::kernel_param_kind_t::kind_std_layout) {
553  continue;
554  }
555  // Find which ArgStorage Arg.MPtr is pointing to
556  for (auto &ArgStorage : OldArgStorage) {
557  if (ArgStorage.data() != Arg.MPtr) {
558  continue;
559  }
560  NewArgStorage.emplace_back(Arg.MSize);
561  // Memcpy contents from old storage to new storage
562  std::memcpy(NewArgStorage.back().data(), ArgStorage.data(), Arg.MSize);
563  // Update MPtr to point to the new storage instead of the old
564  Arg.MPtr = NewArgStorage.back().data();
565 
566  break;
567  }
568  }
