74 std::vector<std::string> split_string(const std::string &str, char delimeter) {
75  std::vector<std::string> Result;
76  size_t Start = 0;
77  size_t End = 0;
78  while ((End = str.find(delimeter, Start)) != std::string::npos) {
79  Result.push_back(str.substr(Start, End - Start));
80  Start = End + 1;
81  }
82  // Get the last substring and ignore the null character so we wouldn't get
83  // double null characters \0\0 at the end of the substring
84  End = str.find('\0');
85  if (Start < End) {
86  std::string LastSubStr(str.substr(Start, End - Start));
87  // In case str has a delimeter at the end, the substring will be empty, so
88  // we shouldn't add it to the final vector
89  if (!LastSubStr.empty())
90  Result.push_back(LastSubStr);
91  }
92  return Result;
93 }
