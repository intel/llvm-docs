76 static pi_result getExtFuncFromContext(pi_context context, T *fptr) {
77  // TODO
78  // Potentially redo caching as PI interface changes.
79  thread_local static std::map<pi_context, T> FuncPtrs;
80 
81  // if cached, return cached FuncPtr
82  if (auto F = FuncPtrs[context]) {
83  // if cached that extension is not available return nullptr and
84  // PI_INVALID_VALUE
85  *fptr = F;
86  return F ? PI_SUCCESS : PI_INVALID_VALUE;
87  }
88 
89  cl_uint deviceCount;
90  cl_int ret_err =
91  clGetContextInfo(cast<cl_context>(context), CL_CONTEXT_NUM_DEVICES,
92  sizeof(cl_uint), &deviceCount, nullptr);
93 
94  if (ret_err != CL_SUCCESS || deviceCount < 1) {
95  return PI_INVALID_CONTEXT;
96  }
97 
98  std::vector<cl_device_id> devicesInCtx(deviceCount);
99  ret_err = clGetContextInfo(cast<cl_context>(context), CL_CONTEXT_DEVICES,
100  deviceCount * sizeof(cl_device_id),
101  devicesInCtx.data(), nullptr);
102 
103  if (ret_err != CL_SUCCESS) {
104  return PI_INVALID_CONTEXT;
105  }
106 
107  cl_platform_id curPlatform;
108  ret_err = clGetDeviceInfo(devicesInCtx[0], CL_DEVICE_PLATFORM,
109  sizeof(cl_platform_id), &curPlatform, nullptr);
110 
111  if (ret_err != CL_SUCCESS) {
112  return PI_INVALID_CONTEXT;
113  }
114 
115  T FuncPtr =
116  (T)clGetExtensionFunctionAddressForPlatform(curPlatform, FuncName);
117 
118  if (!FuncPtr) {
119  // Cache that the extension is not available
120  FuncPtrs[context] = nullptr;
121  return PI_INVALID_VALUE;
122  }
123 
124  *fptr = FuncPtr;
125  FuncPtrs[context] = FuncPtr;
126 
127  return cast<pi_result>(ret_err);
128 }

134 static pi_result USMSetIndirectAccess(pi_kernel kernel) {
135  // We test that each alloc type is supported before we actually try to
136  // set KernelExecInfo.
137  cl_bool TrueVal = CL_TRUE;
138  clHostMemAllocINTEL_fn HFunc = nullptr;
139  clSharedMemAllocINTEL_fn SFunc = nullptr;
140  clDeviceMemAllocINTEL_fn DFunc = nullptr;
141  cl_context CLContext;
142  cl_int CLErr = clGetKernelInfo(cast<cl_kernel>(kernel), CL_KERNEL_CONTEXT,
143  sizeof(cl_context), &CLContext, nullptr);
144  if (CLErr != CL_SUCCESS) {
145  return cast<pi_result>(CLErr);
146  }
147 
148  getExtFuncFromContext<clHostMemAllocName, clHostMemAllocINTEL_fn>(
149  cast<pi_context>(CLContext), &HFunc);
150  if (HFunc) {
151  clSetKernelExecInfo(cast<cl_kernel>(kernel),
152  CL_KERNEL_EXEC_INFO_INDIRECT_HOST_ACCESS_INTEL,
153  sizeof(cl_bool), &TrueVal);
154  }
155 
156  getExtFuncFromContext<clDeviceMemAllocName, clDeviceMemAllocINTEL_fn>(
157  cast<pi_context>(CLContext), &DFunc);
158  if (DFunc) {
159  clSetKernelExecInfo(cast<cl_kernel>(kernel),
160  CL_KERNEL_EXEC_INFO_INDIRECT_DEVICE_ACCESS_INTEL,
161  sizeof(cl_bool), &TrueVal);
162  }
163 
164  getExtFuncFromContext<clSharedMemAllocName, clSharedMemAllocINTEL_fn>(
165  cast<pi_context>(CLContext), &SFunc);
166  if (SFunc) {
167  clSetKernelExecInfo(cast<cl_kernel>(kernel),
168  CL_KERNEL_EXEC_INFO_INDIRECT_SHARED_ACCESS_INTEL,
169  sizeof(cl_bool), &TrueVal);
170  }
171  return PI_SUCCESS;
172 }

227  std::memcpy(paramValue, &result, sizeof(cl_bool));

243  if (paramValueSize >= 3 * sizeof(size_t))
244  out[2] = Max;
245  return PI_SUCCESS;
246  }
247 
248  default:
249  cl_int result = clGetDeviceInfo(
250  cast<cl_device_id>(device), cast<cl_device_info>(paramName),
251  paramValueSize, paramValue, paramValueSizeRet);
252  return static_cast<pi_result>(result);
253  }

296 pi_result piextDeviceSelectBinary(pi_device device, pi_device_binary *images,
297  pi_uint32 num_images,
298  pi_uint32 *selected_image_ind) {
299 
300  // TODO: this is a bare-bones implementation for choosing a device image
301  // that would be compatible with the targeted device. An AOT-compiled
302  // image is preferred over SPIR-V for known devices (i.e. Intel devices)
303  // The implementation makes no effort to differentiate between multiple images
304  // for the given device, and simply picks the first one compatible
305  // Real implementation will use the same mechanism OpenCL ICD dispatcher
306  // uses. Something like:
307  // PI_VALIDATE_HANDLE_RETURN_HANDLE(ctx, PI_INVALID_CONTEXT);
308  // return context->dispatch->piextDeviceSelectIR(
309  // ctx, images, num_images, selected_image);
310  // where context->dispatch is set to the dispatch table provided by PI
311  // plugin for platform/device the ctx was created for.
312 
313  // Choose the binary target for the provided device
314  const char *image_target = nullptr;
315  // Get the type of the device
316  cl_device_type device_type;
317  constexpr pi_uint32 invalid_ind = std::numeric_limits<pi_uint32>::max();
318  cl_int ret_err =
319  clGetDeviceInfo(cast<cl_device_id>(device), CL_DEVICE_TYPE,
320  sizeof(cl_device_type), &device_type, nullptr);
321  if (ret_err != CL_SUCCESS) {
322  *selected_image_ind = invalid_ind;
323  return cast<pi_result>(ret_err);
324  }
325 
326  switch (device_type) {
327  // TODO: Factor out vendor specifics into a separate source
328  // E.g. sycl/source/detail/vendor/intel/detail/pi_opencl.cpp?
329 
330  // We'll attempt to find an image that was AOT-compiled
331  // from a SPIR-V image into an image specific for:
332 
333  case CL_DEVICE_TYPE_CPU: // OpenCL 64-bit CPU
334  image_target = __SYCL_PI_DEVICE_BINARY_TARGET_SPIRV64_X86_64;
335  break;
336  case CL_DEVICE_TYPE_GPU: // OpenCL 64-bit GEN GPU
337  image_target = __SYCL_PI_DEVICE_BINARY_TARGET_SPIRV64_GEN;
338  break;
339  case CL_DEVICE_TYPE_ACCELERATOR: // OpenCL 64-bit FPGA
340  image_target = __SYCL_PI_DEVICE_BINARY_TARGET_SPIRV64_FPGA;
341  break;
342  default:
343  // Otherwise, we'll attempt to find and JIT-compile
344  // a device-independent SPIR-V image
345  image_target = __SYCL_PI_DEVICE_BINARY_TARGET_SPIRV64;
346  break;
347  }
348 
349  // Find the appropriate device image, fallback to spirv if not found
350  pi_uint32 fallback = invalid_ind;
351  for (pi_uint32 i = 0; i < num_images; ++i) {
352  if (strcmp(images[i]->DeviceTargetSpec, image_target) == 0) {
353  *selected_image_ind = i;
354  return PI_SUCCESS;
355  }
356  if (strcmp(images[i]->DeviceTargetSpec,
357  __SYCL_PI_DEVICE_BINARY_TARGET_SPIRV64) == 0)
358  fallback = i;
359  }
360  // Points to a spirv image, if such indeed was found
361  if ((*selected_image_ind = fallback) != invalid_ind)
362  return PI_SUCCESS;
363  // No image can be loaded for the given device
364  return PI_INVALID_BINARY;
365 }

562 // Function gets characters between delimeter's in str
563 // then checks if they are equal to the sub_str.
564 // returns true if there is at least one instance
565 // returns false if there are no instances of the name
566 static bool is_in_separated_string(const std::string &str, char delimiter,
567  const std::string &sub_str) {
568  size_t beg = 0;
569  size_t length = 0;
570  for (const auto &x : str) {
571  if (x == delimiter) {
572  if (str.substr(beg, length) == sub_str)
573  return true;
574 
575  beg += length + 1;
576  length = 0;
577  continue;
578  }
579  length++;
580  }
581  if (length != 0)
582  if (str.substr(beg, length) == sub_str)
583  return true;
584 
585  return false;
586 }
