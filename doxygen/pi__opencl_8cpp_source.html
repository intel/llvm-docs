70  // TODO
71  // Potentially redo caching as PI interface changes.
72  thread_local static std::map<pi_context, T> FuncPtrs;
73 
74  // if cached, return cached FuncPtr
75  if (auto F = FuncPtrs[context]) {
76  *fptr = F;
77  return PI_SUCCESS;
78  }
79 
80  size_t deviceCount;
81  cl_int ret_err = clGetContextInfo(
82  cast<cl_context>(context), CL_CONTEXT_DEVICES, 0, nullptr, &deviceCount);
83 
84  if (ret_err != CL_SUCCESS || deviceCount < 1) {
85  return PI_INVALID_CONTEXT;
86  }
87 
88  std::vector<cl_device_id> devicesInCtx(deviceCount);
89  ret_err = clGetContextInfo(cast<cl_context>(context), CL_CONTEXT_DEVICES,
90  deviceCount * sizeof(cl_device_id),
91  devicesInCtx.data(), nullptr);
92 
93  if (ret_err != CL_SUCCESS) {
94  return PI_INVALID_CONTEXT;
95  }
96 
97  cl_platform_id curPlatform;
98  ret_err = clGetDeviceInfo(devicesInCtx[0], CL_DEVICE_PLATFORM,
99  sizeof(cl_platform_id), &curPlatform, nullptr);
100 
101  if (ret_err != CL_SUCCESS) {
102  return PI_INVALID_CONTEXT;
103  }
104 
105  T FuncPtr =
106  (T)clGetExtensionFunctionAddressForPlatform(curPlatform, FuncName);
107 
108  if (!FuncPtr)
109  return PI_INVALID_VALUE;
110 
111  *fptr = FuncPtr;
112  FuncPtrs[context] = FuncPtr;
113 
114  return cast<pi_result>(ret_err);
115 }

121 static pi_result USMSetIndirectAccess(pi_kernel kernel) {
122  // We test that each alloc type is supported before we actually try to
123  // set KernelExecInfo.
124  cl_bool TrueVal = CL_TRUE;
125  clHostMemAllocINTEL_fn HFunc = nullptr;
126  clSharedMemAllocINTEL_fn SFunc = nullptr;
127  clDeviceMemAllocINTEL_fn DFunc = nullptr;
128  cl_context CLContext;
129  cl_int CLErr = clGetKernelInfo(cast<cl_kernel>(kernel), CL_KERNEL_CONTEXT,
130  sizeof(cl_context), &CLContext, nullptr);
131  if (CLErr != CL_SUCCESS) {
132  return cast<pi_result>(CLErr);
133  }
134 
135  getExtFuncFromContext<clHostMemAllocName, clHostMemAllocINTEL_fn>(
136  cast<pi_context>(CLContext), &HFunc);
137  if (HFunc) {
138  clSetKernelExecInfo(cast<cl_kernel>(kernel),
139  CL_KERNEL_EXEC_INFO_INDIRECT_HOST_ACCESS_INTEL,
140  sizeof(cl_bool), &TrueVal);
141  }
142 
143  getExtFuncFromContext<clDeviceMemAllocName, clDeviceMemAllocINTEL_fn>(
144  cast<pi_context>(CLContext), &DFunc);
145  if (DFunc) {
146  clSetKernelExecInfo(cast<cl_kernel>(kernel),
147  CL_KERNEL_EXEC_INFO_INDIRECT_DEVICE_ACCESS_INTEL,
148  sizeof(cl_bool), &TrueVal);
149  }
150 
151  getExtFuncFromContext<clSharedMemAllocName, clSharedMemAllocINTEL_fn>(
152  cast<pi_context>(CLContext), &SFunc);
153  if (SFunc) {
154  clSetKernelExecInfo(cast<cl_kernel>(kernel),
155  CL_KERNEL_EXEC_INFO_INDIRECT_SHARED_ACCESS_INTEL,
156  sizeof(cl_bool), &TrueVal);
157  }
158  return PI_SUCCESS;
159 }

205 pi_result piextDeviceSelectBinary(pi_device device, pi_device_binary *images,
206  pi_uint32 num_images,
207  pi_uint32 *selected_image_ind) {
208 
209  // TODO: this is a bare-bones implementation for choosing a device image
210  // that would be compatible with the targeted device. An AOT-compiled
211  // image is preferred over SPIR-V for known devices (i.e. Intel devices)
212  // The implementation makes no effort to differentiate between multiple images
213  // for the given device, and simply picks the first one compatible
214  // Real implementation will use the same mechanism OpenCL ICD dispatcher
215  // uses. Something like:
216  // PI_VALIDATE_HANDLE_RETURN_HANDLE(ctx, PI_INVALID_CONTEXT);
217  // return context->dispatch->piextDeviceSelectIR(
218  // ctx, images, num_images, selected_image);
219  // where context->dispatch is set to the dispatch table provided by PI
220  // plugin for platform/device the ctx was created for.
221 
222  // Choose the binary target for the provided device
223  const char *image_target = nullptr;
224  // Get the type of the device
225  cl_device_type device_type;
226  constexpr pi_uint32 invalid_ind = std::numeric_limits<pi_uint32>::max();
227  cl_int ret_err =
228  clGetDeviceInfo(cast<cl_device_id>(device), CL_DEVICE_TYPE,
229  sizeof(cl_device_type), &device_type, nullptr);
230  if (ret_err != CL_SUCCESS) {
231  *selected_image_ind = invalid_ind;
232  return cast<pi_result>(ret_err);
233  }
234 
235  switch (device_type) {
236  // TODO: Factor out vendor specifics into a separate source
237  // E.g. sycl/source/detail/vendor/intel/detail/pi_opencl.cpp?
238 
239  // We'll attempt to find an image that was AOT-compiled
240  // from a SPIR-V image into an image specific for:
241 
242  case CL_DEVICE_TYPE_CPU: // OpenCL 64-bit CPU
243  image_target = PI_DEVICE_BINARY_TARGET_SPIRV64_X86_64;
244  break;
245  case CL_DEVICE_TYPE_GPU: // OpenCL 64-bit GEN GPU
246  image_target = PI_DEVICE_BINARY_TARGET_SPIRV64_GEN;
247  break;
248  case CL_DEVICE_TYPE_ACCELERATOR: // OpenCL 64-bit FPGA
249  image_target = PI_DEVICE_BINARY_TARGET_SPIRV64_FPGA;
250  break;
251  default:
252  // Otherwise, we'll attempt to find and JIT-compile
253  // a device-independent SPIR-V image
254  image_target = PI_DEVICE_BINARY_TARGET_SPIRV64;
255  break;
256  }
257 
258  // Find the appropriate device image, fallback to spirv if not found
259  pi_uint32 fallback = invalid_ind;
260  for (pi_uint32 i = 0; i < num_images; ++i) {
261  if (strcmp(images[i]->DeviceTargetSpec, image_target) == 0) {
262  *selected_image_ind = i;
263  return PI_SUCCESS;
264  }
265  if (strcmp(images[i]->DeviceTargetSpec, PI_DEVICE_BINARY_TARGET_SPIRV64) ==
266  0)
267  fallback = i;
268  }
269  // Points to a spirv image, if such indeed was found
270  if ((*selected_image_ind = fallback) != invalid_ind)
271  return PI_SUCCESS;
272  // No image can be loaded for the given device
273  return PI_INVALID_BINARY;
274 }
