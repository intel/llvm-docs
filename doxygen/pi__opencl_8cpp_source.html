69 static pi_result getExtFuncFromContext(pi_context context, T *fptr) {
70  // TODO
71  // Potentially redo caching as PI interface changes.
72  thread_local static std::map<pi_context, T> FuncPtrs;
73 
74  // if cached, return cached FuncPtr
75  if (auto F = FuncPtrs[context]) {
76  *fptr = F;
77  return PI_SUCCESS;
78  }
79 
80  size_t deviceCount;
81  cl_int ret_err = clGetContextInfo(
82  cast<cl_context>(context), CL_CONTEXT_DEVICES, 0, nullptr, &deviceCount);
83 
84  if (ret_err != CL_SUCCESS || deviceCount < 1) {
85  return PI_INVALID_CONTEXT;
86  }
87 
88  std::vector<cl_device_id> devicesInCtx(deviceCount);
89  ret_err = clGetContextInfo(cast<cl_context>(context), CL_CONTEXT_DEVICES,
90  deviceCount * sizeof(cl_device_id),
91  devicesInCtx.data(), nullptr);
92 
93  if (ret_err != CL_SUCCESS) {
94  return PI_INVALID_CONTEXT;
95  }
96 
97  cl_platform_id curPlatform;
98  ret_err = clGetDeviceInfo(devicesInCtx[0], CL_DEVICE_PLATFORM,
99  sizeof(cl_platform_id), &curPlatform, nullptr);
100 
101  if (ret_err != CL_SUCCESS) {
102  return PI_INVALID_CONTEXT;
103  }
104 
105  T FuncPtr =
106  (T)clGetExtensionFunctionAddressForPlatform(curPlatform, FuncName);
107 
108  if (!FuncPtr)
109  return PI_INVALID_VALUE;
110 
111  *fptr = FuncPtr;
112  FuncPtrs[context] = FuncPtr;
113 
114  return cast<pi_result>(ret_err);
115 }

121 static pi_result USMSetIndirectAccess(pi_kernel kernel) {
122  // We test that each alloc type is supported before we actually try to
123  // set KernelExecInfo.
124  cl_bool TrueVal = CL_TRUE;
125  clHostMemAllocINTEL_fn HFunc = nullptr;
126  clSharedMemAllocINTEL_fn SFunc = nullptr;
127  clDeviceMemAllocINTEL_fn DFunc = nullptr;
128  cl_context CLContext;
129  cl_int CLErr = clGetKernelInfo(cast<cl_kernel>(kernel), CL_KERNEL_CONTEXT,
130  sizeof(cl_context), &CLContext, nullptr);
131  if (CLErr != CL_SUCCESS) {
132  return cast<pi_result>(CLErr);
133  }
134 
135  getExtFuncFromContext<clHostMemAllocName, clHostMemAllocINTEL_fn>(
136  cast<pi_context>(CLContext), &HFunc);
137  if (HFunc) {
138  clSetKernelExecInfo(cast<cl_kernel>(kernel),
139  CL_KERNEL_EXEC_INFO_INDIRECT_HOST_ACCESS_INTEL,
140  sizeof(cl_bool), &TrueVal);
141  }
142 
143  getExtFuncFromContext<clDeviceMemAllocName, clDeviceMemAllocINTEL_fn>(
144  cast<pi_context>(CLContext), &DFunc);
145  if (DFunc) {
146  clSetKernelExecInfo(cast<cl_kernel>(kernel),
147  CL_KERNEL_EXEC_INFO_INDIRECT_DEVICE_ACCESS_INTEL,
148  sizeof(cl_bool), &TrueVal);
149  }
150 
151  getExtFuncFromContext<clSharedMemAllocName, clSharedMemAllocINTEL_fn>(
152  cast<pi_context>(CLContext), &SFunc);
153  if (SFunc) {
154  clSetKernelExecInfo(cast<cl_kernel>(kernel),
155  CL_KERNEL_EXEC_INFO_INDIRECT_SHARED_ACCESS_INTEL,
156  sizeof(cl_bool), &TrueVal);
157  }
158  return PI_SUCCESS;
159 }

218 pi_result OCL(piextDeviceSelectBinary)(pi_device device,
219  pi_device_binary *images,
220  pi_uint32 num_images,
221  pi_uint32 *selected_image_ind) {
222 
223  // TODO: this is a bare-bones implementation for choosing a device image
224  // that would be compatible with the targeted device. An AOT-compiled
225  // image is preferred over SPIR-V for known devices (i.e. Intel devices)
226  // The implementation makes no effort to differentiate between multiple images
227  // for the given device, and simply picks the first one compatible
228  // Real implementation will use the same mechanism OpenCL ICD dispatcher
229  // uses. Something like:
230  // PI_VALIDATE_HANDLE_RETURN_HANDLE(ctx, PI_INVALID_CONTEXT);
231  // return context->dispatch->piextDeviceSelectIR(
232  // ctx, images, num_images, selected_image);
233  // where context->dispatch is set to the dispatch table provided by PI
234  // plugin for platform/device the ctx was created for.
235 
236  // Choose the binary target for the provided device
237  const char *image_target = nullptr;
238  // Get the type of the device
239  cl_device_type device_type;
240  constexpr pi_uint32 invalid_ind = std::numeric_limits<pi_uint32>::max();
241  cl_int ret_err =
242  clGetDeviceInfo(cast<cl_device_id>(device), CL_DEVICE_TYPE,
243  sizeof(cl_device_type), &device_type, nullptr);
244  if (ret_err != CL_SUCCESS) {
245  *selected_image_ind = invalid_ind;
246  return cast<pi_result>(ret_err);
247  }
248 
249  switch (device_type) {
250  // TODO: Factor out vendor specifics into a separate source
251  // E.g. sycl/source/detail/vendor/intel/detail/pi_opencl.cpp?
252 
253  // We'll attempt to find an image that was AOT-compiled
254  // from a SPIR-V image into an image specific for:
255 
256  case CL_DEVICE_TYPE_CPU: // OpenCL 64-bit CPU
257  image_target = PI_DEVICE_BINARY_TARGET_SPIRV64_X86_64;
258  break;
259  case CL_DEVICE_TYPE_GPU: // OpenCL 64-bit GEN GPU
260  image_target = PI_DEVICE_BINARY_TARGET_SPIRV64_GEN;
261  break;
262  case CL_DEVICE_TYPE_ACCELERATOR: // OpenCL 64-bit FPGA
263  image_target = PI_DEVICE_BINARY_TARGET_SPIRV64_FPGA;
264  break;
265  default:
266  // Otherwise, we'll attempt to find and JIT-compile
267  // a device-independent SPIR-V image
268  image_target = PI_DEVICE_BINARY_TARGET_SPIRV64;
269  break;
270  }
271 
272  // Find the appropriate device image, fallback to spirv if not found
273  pi_uint32 fallback = invalid_ind;
274  for (pi_uint32 i = 0; i < num_images; ++i) {
275  if (strcmp(images[i]->DeviceTargetSpec, image_target) == 0) {
276  *selected_image_ind = i;
277  return PI_SUCCESS;
278  }
279  if (strcmp(images[i]->DeviceTargetSpec, PI_DEVICE_BINARY_TARGET_SPIRV64) ==
280  0)
281  fallback = i;
282  }
283  // Points to a spirv image, if such indeed was found
284  if ((*selected_image_ind = fallback) != invalid_ind)
285  return PI_SUCCESS;
286  // No image can be loaded for the given device
287  return PI_INVALID_BINARY;
288 }
