406  template <typename RTy>
407  ESIMD_INLINE void
408  writeRegion(RTy Region,
409  const detail::vector_type_t<typename RTy::element_type,
410  RTy::length> &Val) {
411  using ElemTy = typename RTy::element_type;
412  if constexpr (N * sizeof(Ty) == RTy::length * sizeof(ElemTy))
413  // update the entire vector
414  set(detail::bitcast<Ty, ElemTy, RTy::length>(Val));
415  else {
416  static_assert(!RTy::Is_2D);
417  // If element type differs, do bitcast conversion first.
418  auto Base = detail::bitcast<ElemTy, Ty, N>(data());
419  constexpr int BN = (N * sizeof(Ty)) / sizeof(ElemTy);
420  // Access the region information.
421  constexpr int M = RTy::Size_x;
422  constexpr int Stride = RTy::Stride_x;
423  uint16_t Offset = Region.M_offset_x * sizeof(ElemTy);
424 
425  // Merge and update.
426  auto Merged = __esimd_wrregion<ElemTy, BN, M,
427  /*VS*/ 0, M, Stride>(Base, Val, Offset);
428  // Convert back to the original element type, if needed.
429  set(detail::bitcast<Ty, ElemTy, BN>(Merged));
430  }
431  }

434  template <typename TR, typename UR>
435  ESIMD_INLINE void writeRegion(
436  std::pair<TR, UR> Region,
437  const detail::vector_type_t<typename TR::element_type, TR::length> &Val) {
438  // parent-region type
439  using PaTy = typename shape_type<UR>::type;
440  using ElemTy = typename TR::element_type;
441  using BT = typename PaTy::element_type;
442  constexpr int BN = PaTy::length;
443 
444  if constexpr (PaTy::Size_in_bytes == TR::Size_in_bytes) {
445  writeRegion(Region.second, detail::bitcast<BT, ElemTy, TR::length>(Val));
446  } else {
447  // Recursively read the base
448  auto Base = detail::readRegion<Ty, N>(data(), Region.second);
449  // If element type differs, do bitcast conversion first.
450  auto Base1 = detail::bitcast<ElemTy, BT, BN>(Base);
451  constexpr int BN1 = PaTy::Size_in_bytes / sizeof(ElemTy);
452 
453  if constexpr (!TR::Is_2D) {
454  // Access the region information.
455  constexpr int M = TR::Size_x;
456  constexpr int Stride = TR::Stride_x;
457  uint16_t Offset = Region.first.M_offset_x * sizeof(ElemTy);
458 
459  // Merge and update.
460  Base1 = __esimd_wrregion<ElemTy, BN1, M,
461  /*VS*/ 0, M, Stride>(Base1, Val, Offset);
462  } else {
463  static_assert(std::is_same<ElemTy, BT>::value);
464  // Read columns with non-trivial horizontal stride.
465  constexpr int M = TR::length;
466  constexpr int VS = PaTy::Size_x * TR::Stride_y;
467  constexpr int W = TR::Size_x;
468  constexpr int HS = TR::Stride_x;
469  constexpr int ParentWidth = PaTy::Size_x;
470 
471  // Compute the byte offset for the starting element.
472  uint16_t Offset = static_cast<uint16_t>(
473  (Region.first.M_offset_y * PaTy::Size_x + Region.first.M_offset_x) *
474  sizeof(ElemTy));
475 
476  // Merge and update.
477  Base1 = __esimd_wrregion<ElemTy, BN1, M, VS, W, HS, ParentWidth>(
478  Base1, Val, Offset);
479  }
480  // Convert back to the original element type, if needed.
481  auto Merged1 = detail::bitcast<BT, ElemTy, BN1>(Base1);
482  // recursively write it back to the base
483  writeRegion(Region.second, Merged1);
484  }
485  }
