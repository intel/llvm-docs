89  constexpr simd(Ty Val, Ty Step = Ty()) noexcept {
90  if (Step == Ty())
91  M_data = Val;
92  else {
93 #pragma unroll
94  for (int i = 0; i < N; ++i) {
95  M_data[i] = Val;
96  Val += Step;
97  }

401  template <typename RTy>
402  ESIMD_INLINE void writeRegion(
403  RTy Region,
404  const vector_type_t<typename RTy::element_type, RTy::length> &Val) {
405  using ElemTy = typename RTy::element_type;
406  if constexpr (N * sizeof(Ty) == RTy::length * sizeof(ElemTy))
407  // update the entire vector
408  set(bitcast<Ty, ElemTy, RTy::length>(Val));
409  else {
410  static_assert(!RTy::Is_2D);
411  // If element type differs, do bitcast conversion first.
412  auto Base = bitcast<ElemTy, Ty, N>(data());
413  constexpr int BN = (N * sizeof(Ty)) / sizeof(ElemTy);
414  // Access the region information.
415  constexpr int M = RTy::Size_x;
416  constexpr int Stride = RTy::Stride_x;
417  uint16_t Offset = Region.M_offset_x * sizeof(ElemTy);
418 
419  // Merge and update.
420  auto Merged = __esimd_wrregion<ElemTy, BN, M,
421  /*VS*/ 0, M, Stride>(Base, Val, Offset);
422  // Convert back to the original element type, if needed.
423  set(bitcast<Ty, ElemTy, BN>(Merged));
424  }
425  }

428  template <typename TR, typename UR>
429  ESIMD_INLINE void
430  writeRegion(std::pair<TR, UR> Region,
431  const vector_type_t<typename TR::element_type, TR::length> &Val) {
432  // parent-region type
433  using PaTy = typename shape_type<UR>::type;
434  using ElemTy = typename TR::element_type;
435  using BT = typename PaTy::element_type;
436  constexpr int BN = PaTy::length;
437 
438  if constexpr (PaTy::Size_in_bytes == TR::Size_in_bytes) {
439  writeRegion(Region.second, bitcast<BT, ElemTy, TR::length>(Val));
440  } else {
441  // Recursively read the base
442  auto Base = readRegion<Ty, N>(data(), Region.second);
443  // If element type differs, do bitcast conversion first.
444  auto Base1 = bitcast<ElemTy, BT, BN>(Base);
445  constexpr int BN1 = PaTy::Size_in_bytes / sizeof(ElemTy);
446 
447  if constexpr (!TR::Is_2D) {
448  // Access the region information.
449  constexpr int M = TR::Size_x;
450  constexpr int Stride = TR::Stride_x;
451  uint16_t Offset = Region.first.M_offset_x * sizeof(ElemTy);
452 
453  // Merge and update.
454  Base1 = __esimd_wrregion<ElemTy, BN1, M,
455  /*VS*/ 0, M, Stride>(Base1, Val, Offset);
456  } else {
457  static_assert(std::is_same<ElemTy, BT>::value);
458  // Read columns with non-trivial horizontal stride.
459  constexpr int M = TR::length;
460  constexpr int VS = PaTy::Size_x * TR::Stride_y;
461  constexpr int W = TR::Size_x;
462  constexpr int HS = TR::Stride_x;
463  constexpr int ParentWidth = PaTy::Size_x;
464 
465  // Compute the byte offset for the starting element.
466  uint16_t Offset = static_cast<uint16_t>(
467  (Region.first.M_offset_y * PaTy::Size_x + Region.first.M_offset_x) *
468  sizeof(ElemTy));
469 
470  // Merge and update.
471  Base1 = __esimd_wrregion<ElemTy, BN1, M, VS, W, HS, ParentWidth>(
472  Base1, Val, Offset);
473  }
474  // Convert back to the original element type, if needed.
475  auto Merged1 = bitcast<BT, ElemTy, BN1>(Base1);
476  // recursively write it back to the base
477  writeRegion(Region.second, Merged1);
478  }
