116 constexpr vector_type_t<T, N> make_vector_impl(const T (&&Arr)[N],
117  std::index_sequence<Is...>) {

121 template <class T, int N>
122 constexpr vector_type_t<T, N> make_vector(const T (&&Arr)[N]) {
123  return make_vector_impl<T, N>(std::move(Arr), std::make_index_sequence<N>{});
124 }
125 
126 template <class T, int N, size_t... Is>
127 constexpr auto make_vector_impl(T Base, T Stride, std::index_sequence<Is...>) {
128  if constexpr (std::is_integral_v<T> && N <= 3) {
129  // This sequence is a bit more efficient for integral types and N <= 3.
130  return vector_type_t<T, N>{(T)(Base + ((T)Is) * Stride)...};
131  } else {
132  using CppT = typename element_type_traits<T>::EnclosingCppT;
133  CppT BaseCpp = Base;
134  CppT StrideCpp = Stride;
135  vector_type_t<CppT, N> VBase = BaseCpp;
136  vector_type_t<CppT, N> VStride = StrideCpp;
137  vector_type_t<CppT, N> VStrideCoef{(CppT)(Is)...};
138  vector_type_t<CppT, N> Result{VBase + VStride * VStrideCoef};
139  return wrapper_type_converter<T>::template to_vector<N>(Result);
140  }
141 }
142 
143 template <class T, int N> constexpr auto make_vector(T Base, T Stride) {
144  return make_vector_impl<T, N>(Base, Stride, std::make_index_sequence<N>{});
145 }

645  template <typename RTy, class ElemTy = __raw_t<typename RTy::element_type>>
646  ESIMD_INLINE void writeRegion(RTy Region,
647  const vector_type_t<ElemTy, RTy::length> &Val) {
648 
649  if constexpr (N * sizeof(RawTy) == RTy::length * sizeof(ElemTy))
650  // update the entire vector
651  set(bitcast<RawTy, ElemTy, RTy::length>(Val));
652  else {
653  static_assert(!RTy::Is_2D);
654  // If element type differs, do bitcast conversion first.
655  auto Base = bitcast<ElemTy, RawTy, N>(data());
656  constexpr int BN = (N * sizeof(RawTy)) / sizeof(ElemTy);
657  // Access the region information.
658  constexpr int M = RTy::Size_x;
659  constexpr int Stride = RTy::Stride_x;
660  uint16_t Offset = Region.M_offset_x * sizeof(ElemTy);
661  check_wrregion_params<BN, M, /*VS*/ 0, M, Stride>();
662  // Merge and update.
663  auto Merged = __esimd_wrregion<ElemTy, BN, M,
664  /*VS*/ 0, M, Stride>(Base, Val, Offset);
665  // Convert back to the original element type, if needed.
666  set(bitcast<RawTy, ElemTy, BN>(Merged));
667  }
668  }

672  template <typename TR, typename UR,
673  class ElemTy = __raw_t<typename TR::element_type>>
674  ESIMD_INLINE void writeRegion(std::pair<TR, UR> Region,
675  const vector_type_t<ElemTy, TR::length> &Val) {
676  // parent-region type
677  using PaTy = typename shape_type<UR>::type;
678  using BT = __raw_t<typename PaTy::element_type>;
679  constexpr int BN = PaTy::length;
680 
681  if constexpr (PaTy::Size_in_bytes == TR::Size_in_bytes) {
682  writeRegion(Region.second, bitcast<BT, ElemTy, TR::length>(Val));
683  } else {
684  // Recursively read the base
685  auto Base = readRegion<RawTy, N>(data(), Region.second);
686  // If element type differs, do bitcast conversion first.
687  auto Base1 = bitcast<ElemTy, BT, BN>(Base);
688  constexpr int BN1 = PaTy::Size_in_bytes / sizeof(ElemTy);
689 
690  if constexpr (!TR::Is_2D) {
691  // Access the region information.
692  constexpr int M = TR::Size_x;
693  constexpr int Stride = TR::Stride_x;
694  uint16_t Offset = Region.first.M_offset_x * sizeof(ElemTy);
695 
696  check_wrregion_params<BN1, M, /*VS*/ 0, M, Stride>();
697  // Merge and update.
698  Base1 = __esimd_wrregion<ElemTy, BN1, M,
699  /*VS*/ 0, M, Stride>(Base1, Val, Offset);
700  } else {
701  static_assert(std::is_same<ElemTy, BT>::value);
702  // Read columns with non-trivial horizontal stride.
703  constexpr int M = TR::length;
704  constexpr int VS = PaTy::Size_x * TR::Stride_y;
705  constexpr int W = TR::Size_x;
706  constexpr int HS = TR::Stride_x;
707  constexpr int ParentWidth = PaTy::Size_x;
708 
709  // Compute the byte offset for the starting element.
710  uint16_t Offset = static_cast<uint16_t>(
711  (Region.first.M_offset_y * PaTy::Size_x + Region.first.M_offset_x) *
712  sizeof(ElemTy));
713 
714  check_wrregion_params<BN1, M, VS, W, HS>();
715  // Merge and update.
716  Base1 = __esimd_wrregion<ElemTy, BN1, M, VS, W, HS, ParentWidth>(
717  Base1, Val, Offset);
718  }
719  // Convert back to the original element type, if needed.
720  auto Merged1 = bitcast<BT, ElemTy, BN1>(Base1);
721  // recursively write it back to the base
722  writeRegion(Region.second, Merged1);
723  }
724  }
