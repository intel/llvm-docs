71 template <typename T, typename T0, typename T1, int N>
72 __ESIMD_API __ESIMD_NS::simd<T, N> imul_impl(__ESIMD_NS::simd<T, N> &rmd,
73  __ESIMD_NS::simd<T0, N> src0,
74  __ESIMD_NS::simd<T1, N> src1) {
75  static_assert(__ESIMD_DNS::is_dword_type<T>::value &&
76  __ESIMD_DNS::is_dword_type<T0>::value &&
77  __ESIMD_DNS::is_dword_type<T1>::value,
78  "expected 32-bit integer vector operands.");
79  using Comp32T = __ESIMD_DNS::computation_type_t<T0, T1>;
80  auto Src0 = src0.template bit_cast_view<Comp32T>();
81  auto Src1 = src1.template bit_cast_view<Comp32T>();
82 
83  // Compute the result using 64-bit multiplication operation.
84  using Comp64T =
85  std::conditional_t<std::is_signed_v<Comp32T>, int64_t, uint64_t>;
86  __ESIMD_NS::simd<Comp64T, N> Product64 = Src0;
87  Product64 *= Src1;
88 
89  // Split the 32-bit high and low parts to return them from this function.
90  auto Product32 = Product64.template bit_cast_view<T>();
91  if constexpr (N == 1) {
92  rmd = Product32[0];
93  return Product32[1];
94  } else {
95  rmd = Product32.template select<N, 2>(0);
96  return Product32.template select<N, 2>(1);
97  }
98 }

296 template <typename T, int SZ>
297 __ESIMD_API __ESIMD_NS::simd<T, SZ> atan(__ESIMD_NS::simd<T, SZ> src0) {
298  static_assert(std::is_floating_point<T>::value,
299  "Floating point argument type is expected.");
300  __ESIMD_NS::simd<T, SZ> Src0 = __ESIMD_NS::abs(src0);
301 
302  __ESIMD_NS::simd<T, SZ> OneP((T)1.0);
303  __ESIMD_NS::simd<T, SZ> OneN((T)-1.0);
304  __ESIMD_NS::simd<T, SZ> sign;
305  __ESIMD_NS::simd_mask<SZ> Gt1 = Src0 > T(1.0);
306 
307  sign.merge(OneN, OneP, src0 < 0);
308 
309  Src0.merge(__ESIMD_NS::inv(Src0), Gt1);
310 
311  __ESIMD_NS::simd<T, SZ> Src0P2 = Src0 * Src0;
312  __ESIMD_NS::simd<T, SZ> Src0P4 = Src0P2 * Src0P2;
313 
314  __ESIMD_NS::simd<T, SZ> Result =
315  (Src0P4 * T(0.185696) + ((Src0 * T(0.787997) + T(0.63693)) * Src0P2) +
316  Src0) /
317  (((((Src0 * -T(0.000121387) + T(0.00202308)) * Src0P2) +
318  (Src0 * -T(0.0149145)) + T(0.182569)) *
319  Src0P4) +
320  ((Src0 * T(0.395889) + T(1.12158)) * Src0P2) + (Src0 * T(0.636918)) +
321  T(1.0));
322 
323  Result.merge(Result - T(detail::__ESIMD_CONST_PI) / T(2.0), Gt1);
324 
325  return __ESIMD_NS::abs(Result) * sign;
326 }

338 template <typename T, int SZ>
339 __ESIMD_API
340  std::enable_if_t<std::is_floating_point<T>::value, __ESIMD_NS::simd<T, SZ>>
341  acos(__ESIMD_NS::simd<T, SZ> src0) {
342  __ESIMD_NS::simd<T, SZ> Src0 = __ESIMD_NS::abs(src0);
343 
344  __ESIMD_NS::simd_mask<SZ> Neg = src0 < T(0.0);
345  __ESIMD_NS::simd_mask<SZ> TooBig = Src0 >= T(0.999998);
346 
347  // Replace oversized values to ensure no possibility of sqrt of
348  // a negative value later
349  Src0.merge(T(0.0), TooBig);
350 
351  __ESIMD_NS::simd<T, SZ> Src01m = T(1.0) - Src0;
352 
353  __ESIMD_NS::simd<T, SZ> Src0P2 = Src01m * Src01m;
354  __ESIMD_NS::simd<T, SZ> Src0P4 = Src0P2 * Src0P2;
355 
356  __ESIMD_NS::simd<T, SZ> Result =
357  (((Src01m * T(0.015098965761299077) - T(0.005516443930088506)) * Src0P4) +
358  ((Src01m * T(0.047654245891495528) + T(0.163910606547823220)) * Src0P2) +
359  Src01m * T(2.000291665285952400) - T(0.000007239283986332)) *
360  __ESIMD_NS::rsqrt(Src01m * T(2.0));
361 
362  Result.merge(T(0.0), TooBig);
363  Result.merge(T(detail::__ESIMD_CONST_PI) - Result, Neg);
364  return Result;
365 }

454 // For vector input
455 template <int N>
456 ESIMD_INLINE __ESIMD_NS::simd<float, N>
457 atan2_fast(__ESIMD_NS::simd<float, N> y, __ESIMD_NS::simd<float, N> x) {
458  /* smallest such that 1.0+CONST_DBL_EPSILON != 1.0 */
459  constexpr float CONST_DBL_EPSILON = 0.00001f;
460  __ESIMD_NS::simd<float, N> OneP(1.0f);
461  __ESIMD_NS::simd<float, N> OneN(-1.0f);
462  __ESIMD_NS::simd<float, N> sign;
463  __ESIMD_NS::simd<float, N> atan2;
464  __ESIMD_NS::simd<float, N> r;
465  __ESIMD_NS::simd_mask<N> mask = x < 0;
466  __ESIMD_NS::simd<float, N> abs_y = __ESIMD_NS::abs(y) + CONST_DBL_EPSILON;
467 
468  r.merge((x + abs_y) / (abs_y - x), (x - abs_y) / (x + abs_y), mask);
469  atan2.merge(float(detail::__ESIMD_CONST_PI) * 0.75f,
470  float(detail::__ESIMD_CONST_PI) * 0.25f, mask);
471  atan2 += (0.1963f * r * r - 0.9817f) * r;
472 
473  sign.merge(OneN, OneP, y < 0);
474 
475  return atan2 * sign;
476 }

486 // atan2
487 // For Vector input
488 template <int N>
489 ESIMD_INLINE __ESIMD_NS::simd<float, N> atan2(__ESIMD_NS::simd<float, N> y,
490  __ESIMD_NS::simd<float, N> x) {
491  __ESIMD_NS::simd<float, N> atan2;
492  __ESIMD_NS::simd_mask<N> mask;
493  __ESIMD_NS::simd<float, N> atan = esimd::atan(y / x);
494 
495  constexpr float CONST_DBL_EPSILON = 0.00001f;
496 
497  mask = (__ESIMD_NS::abs(x) < CONST_DBL_EPSILON && y < -CONST_DBL_EPSILON);
498  atan2.merge(float(-detail::__ESIMD_CONST_PI) / 2.f, 0.f, mask);
499  mask = (__ESIMD_NS::abs(x) < CONST_DBL_EPSILON && y > CONST_DBL_EPSILON);
500  atan2.merge(float(detail::__ESIMD_CONST_PI) / 2.f, mask);
501  mask = (x < -CONST_DBL_EPSILON && y < -CONST_DBL_EPSILON);
502  atan2.merge(atan - float(detail::__ESIMD_CONST_PI), mask);
503  mask = (x < -CONST_DBL_EPSILON && y >= -CONST_DBL_EPSILON);
504  atan2.merge(atan + float(detail::__ESIMD_CONST_PI), mask);
505  mask = (x > CONST_DBL_EPSILON);
506  atan2.merge(atan, mask);
507 
508  return atan2;
509 }

523  __ESIMD_NS::simd<float, N> x) {
524  __ESIMD_NS::simd<float, N> abs_x = __ESIMD_NS::abs(x);
525  __ESIMD_NS::simd<float, N> abs_y = __ESIMD_NS::abs(y);
526 
527  auto fmod_sign_mask = (y.template bit_cast_view<int32_t>()) & 0x80000000;
528 
529  __ESIMD_NS::simd<float, N> reminder =
530  abs_y - abs_x * __ESIMD_NS::trunc<float>(abs_y / abs_x);
531 
532  abs_x.merge(0.0f, reminder >= 0);
533  __ESIMD_NS::simd<float, N> fmod = reminder + abs_x;
534  __ESIMD_NS::simd<float, N> fmod_abs = __ESIMD_NS::abs(fmod);
535 
536  auto fmod_bits =
537  (fmod_abs.template bit_cast_view<int32_t>()) | fmod_sign_mask;
538  return fmod_bits.template bit_cast_view<float>();
539 }

547 // For Vector input
548 template <int N>
549 ESIMD_INLINE __ESIMD_NS::simd<float, N> sin_emu(__ESIMD_NS::simd<float, N> x) {
550  __ESIMD_NS::simd<float, N> x1;
551  __ESIMD_NS::simd<float, N> x2;
552  __ESIMD_NS::simd<float, N> t3;
553 
554  __ESIMD_NS::simd<float, N> sign;
555  __ESIMD_NS::simd<float, N> fTrig;
556  __ESIMD_NS::simd<float, N> TwoPI(float(detail::__ESIMD_CONST_PI) * 2.0f);
557  __ESIMD_NS::simd<float, N> CmpI((float)detail::__ESIMD_CONST_PI);
558  __ESIMD_NS::simd<float, N> OneP(1.0f);
559  __ESIMD_NS::simd<float, N> OneN(-1.0f);
560 
561  x = esimd::fmod(x, TwoPI);
562  x.merge(TwoPI + x, x < 0);
563 
564  x1.merge(CmpI - x, x - CmpI, (x <= float(detail::__ESIMD_CONST_PI)));
565  x1.merge(x, (x <= float(detail::__ESIMD_CONST_PI) * 0.5f));
566  x1.merge(TwoPI - x, (x > float(detail::__ESIMD_CONST_PI) * 1.5f));
567 
568  sign.merge(OneN, OneP, (x > float(detail::__ESIMD_CONST_PI)));
569 
570  x2 = x1 * x1;
571  t3 = x2 * x1 * 0.1666667f;
572 
573  fTrig =
574  x1 + t3 * (OneN + x2 * 0.05f *
575  (OneP + x2 * 0.0238095f *
576  (OneN + x2 * 0.0138889f *
577  (OneP - x2 * 0.0090909f))));
578  fTrig *= sign;
579  return fTrig;
580 }

602 template <int N>
603 ESIMD_INLINE __ESIMD_NS::simd<float, N>
604 tanh_cody_waite_impl(__ESIMD_NS::simd<float, N> x) {
605  /*
606  * 0 x_small x_medium x_large
607  * | x | rational polynomial | 1 - 2/(1 + exp(2*x)) | 1
608  *
609  * rational polynomial for single precision = x + x * (g * (p[1] * g + p[0]) /
610  * (g + q[0]) g = x^2 p0 = -0.82377 28127 E+00 p1 = -0.38310 10665 E-02 q0 =
611  * 0.24713 19654 E+01 q1 = 1.00000 00000 E+00
612  *
613  */
614 
615  constexpr float p0 = -0.8237728127E+00f;
616  constexpr float p1 = -0.3831010665E-02f;
617  constexpr float q0 = 0.2471319654E+01f;
618  constexpr float q1 = 1.0000000000E+00f;
619  constexpr float xsmall = 4.22863966691620432990E-04f;
620  constexpr float xmedium = 0.54930614433405484570f;
621  constexpr float xlarge = 8.66433975699931636772f;
622 
623  using RT = __ESIMD_NS::simd<float, N>;
624 
625  RT absX = __ESIMD_NS::abs(x);
626  RT g = absX * absX;
627 
628  RT sign;
629  sign.merge(-1.f, 1.f, x < 0.f);
630 
631  auto isLarge = absX > xlarge;
632  auto minor = absX <= xlarge;
633  auto isGtMed = minor & (absX > xmedium);
634  auto isGtSmall = (absX > xsmall) & (absX <= xmedium);
635 
636  RT res;
637  res.merge(sign, x, isLarge);
638  auto temp = __ESIMD_NS::exp(absX * 2.0f) + 1.f;
639  temp = ((temp - 2.f) / temp) * sign;
640  res.merge(temp, isGtMed);
641  res.merge((absX + absX * g * (g * p1 + p0) / (g + q0)) * sign, isGtSmall);
642 
643  return res;
644 }

646 template <int N>
647 ESIMD_INLINE __ESIMD_NS::simd<float, N>
648 tanh_impl(__ESIMD_NS::simd<float, N> x) {
649  /*
650  * 0 x_small x_large
651  * | x | ( exp(x) - exp(-x) ) / ( exp(x) + exp(-x) ) | 1
652  *
653  */
654 
655  constexpr float xsmall = 0.000045f; // same as exp(-10.0f)
656  constexpr float xlarge = 40.f;
657 
658  using RT = __ESIMD_NS::simd<float, N>;
659 
660  RT absX = __ESIMD_NS::abs(x);
661 
662  RT sign;
663  sign.merge(-1.f, 1.f, x < 0.f);
664 
665  auto isLarge = (absX > xlarge);
666  auto isLessE = (absX <= xlarge);
667 
668  RT res;
669  res.merge(sign, x, isLarge);
670 
671  RT exp;
672  exp = __ESIMD_NS::exp(absX * 2.f);
673 
674  res.merge(((exp - 1.f) / (exp + 1.f)) * sign, (absX > xsmall) & isLessE);
675 
676  return res;
677 }
