377 template <typename T, typename T0, typename T1, int N>
378 __ESIMD_API __ESIMD_NS::simd<T, N> imul_impl(__ESIMD_NS::simd<T, N> &rmd,
379  __ESIMD_NS::simd<T0, N> src0,
380  __ESIMD_NS::simd<T1, N> src1) {
381  static_assert(__ESIMD_DNS::is_dword_type<T>::value &&
382  __ESIMD_DNS::is_dword_type<T0>::value &&
383  __ESIMD_DNS::is_dword_type<T1>::value,
384  "expected 32-bit integer vector operands.");
385  using Comp32T = __ESIMD_DNS::computation_type_t<T0, T1>;
386  auto Src0 = src0.template bit_cast_view<Comp32T>();
387  auto Src1 = src1.template bit_cast_view<Comp32T>();
388 
389  // Compute the result using 64-bit multiplication operation.
390  using Comp64T =
391  std::conditional_t<std::is_signed_v<Comp32T>, int64_t, uint64_t>;
392  __ESIMD_NS::simd<Comp64T, N> Product64 = Src0;
393  Product64 *= Src1;
394 
395  // Split the 32-bit high and low parts to return them from this function.
396  auto Product32 = Product64.template bit_cast_view<T>();
397  if constexpr (N == 1) {
398  rmd = Product32[0];
399  return Product32[1];
400  } else {
401  rmd = Product32.template select<N, 2>(0);
402  return Product32.template select<N, 2>(1);
403  }
404 }

1526  * 0 x_small x_medium x_large
1527  * | x | rational polynomial | 1 - 2/(1 + exp(2*x)) | 1
1528  *
1529  * rational polynomial for single precision = x + x * (g * (p[1] * g + p[0]) /
1530  * (g + q[0]) g = x^2 p0 = -0.82377 28127 E+00 p1 = -0.38310 10665 E-02 q0 =
1531  * 0.24713 19654 E+01 q1 = 1.00000 00000 E+00
1532  *
1533  */

1536  constexpr float p1 = -0.3831010665E-02f;
1537  constexpr float q0 = 0.2471319654E+01f;
1538  constexpr float q1 = 1.0000000000E+00f;
1539  constexpr float xsmall = 4.22863966691620432990E-04f;
1540  constexpr float xmedium = 0.54930614433405484570f;
1541  constexpr float xlarge = 8.66433975699931636772f;
1542 
1543  using RT = __ESIMD_NS::simd<float, N>;
1544 
1545  RT absX = __ESIMD_NS::abs(x);
1546  RT g = absX * absX;
1547 
1548  RT sign;
1549  sign.merge(-1.f, 1.f, x < 0.f);
1550 
1551  auto isLarge = absX > xlarge;
1552  auto minor = absX <= xlarge;
1553  auto isGtMed = minor & (absX > xmedium);
1554  auto isGtSmall = (absX > xsmall) & (absX <= xmedium);
1555 
1556  RT res;
1557  res.merge(sign, x, isLarge);
1558  auto temp = __ESIMD_NS::exp(absX * 2.0f) + 1.f;
1559  temp = ((temp - 2.f) / temp) * sign;
1560  res.merge(temp, isGtMed);
1561  res.merge((absX + absX * g * (g * p1 + p0) / (g + q0)) * sign, isGtSmall);
1562
