146 event handler::finalize() {
147  // This block of code is needed only for reduction implementation.
148  // It is harmless (does nothing) for everything else.
149  if (MIsFinalized)
150  return MLastEvent;
151  MIsFinalized = true;
152 
153  // According to 4.7.6.9 of SYCL2020 spec, if a placeholder accessor is passed
154  // to a command without being bound to a command group, an exception should
155  // be thrown.
156  {
157  for (const auto &arg : MArgs) {
158  if (arg.MType != detail::kernel_param_kind_t::kind_accessor)
159  continue;
160 
161  detail::Requirement *AccImpl =
162  static_cast<detail::Requirement *>(arg.MPtr);
163  if (AccImpl->MIsPlaceH) {
164  auto It = std::find(CGData.MRequirements.begin(),
165  CGData.MRequirements.end(), AccImpl);
166  if (It == CGData.MRequirements.end())
167  throw sycl::exception(make_error_code(errc::kernel_argument),
168  "placeholder accessor must be bound by calling "
169  "handler::require() before it can be used.");
170 
171  // Check associated accessors
172  bool AccFound = false;
173  for (detail::ArgDesc &Acc : MAssociatedAccesors) {
174  if (Acc.MType == detail::kernel_param_kind_t::kind_accessor &&
175  static_cast<detail::Requirement *>(Acc.MPtr) == AccImpl) {
176  AccFound = true;
177  break;
178  }
179  }
180 
181  if (!AccFound) {
182  throw sycl::exception(make_error_code(errc::kernel_argument),
183  "placeholder accessor must be bound by calling "
184  "handler::require() before it can be used.");
185  }
186  }
187  }
188  }
189 
190  const auto &type = getType();
191  if (type == detail::CG::Kernel) {
192  // If there were uses of set_specialization_constant build the kernel_bundle
193  std::shared_ptr<detail::kernel_bundle_impl> KernelBundleImpPtr =
194  getOrInsertHandlerKernelBundle(/*Insert=*/false);
195  if (KernelBundleImpPtr) {
196  // Make sure implicit non-interop kernel bundles have the kernel
197  if (!KernelBundleImpPtr->isInterop() &&
198  !MImpl->isStateExplicitKernelBundle()) {
199  auto Dev = MGraph ? MGraph->getDevice() : MQueue->get_device();
200  kernel_id KernelID =
201  detail::ProgramManager::getInstance().getSYCLKernelID(
202  MKernelName.c_str());
203  bool KernelInserted = KernelBundleImpPtr->add_kernel(KernelID, Dev);
204  // If kernel was not inserted and the bundle is in input mode we try
205  // building it and trying to find the kernel in executable mode
206  if (!KernelInserted &&
207  KernelBundleImpPtr->get_bundle_state() == bundle_state::input) {
208  auto KernelBundle =
209  detail::createSyclObjFromImpl<kernel_bundle<bundle_state::input>>(
210  KernelBundleImpPtr);
211  kernel_bundle<bundle_state::executable> ExecKernelBundle =
212  build(KernelBundle);
213  KernelBundleImpPtr = detail::getSyclObjImpl(ExecKernelBundle);
214  setHandlerKernelBundle(KernelBundleImpPtr);
215  KernelInserted = KernelBundleImpPtr->add_kernel(KernelID, Dev);
216  }
217  // If the kernel was not found in executable mode we throw an exception
218  if (!KernelInserted)
219  throw sycl::exception(make_error_code(errc::runtime),
220  "Failed to add kernel to kernel bundle.");
221  }
222 
223  switch (KernelBundleImpPtr->get_bundle_state()) {
224  case bundle_state::input: {
225  // Underlying level expects kernel_bundle to be in executable state
226  kernel_bundle<bundle_state::executable> ExecBundle = build(
227  detail::createSyclObjFromImpl<kernel_bundle<bundle_state::input>>(
228  KernelBundleImpPtr));
229  KernelBundleImpPtr = detail::getSyclObjImpl(ExecBundle);
230  setHandlerKernelBundle(KernelBundleImpPtr);
231  break;
232  }
233  case bundle_state::executable:
234  // Nothing to do
235  break;
236  case bundle_state::object:
237  case bundle_state::ext_oneapi_source:
238  assert(0 && "Expected that the bundle is either in input or executable "
239  "states.");
240  break;
241  }
242  }

244  if (MQueue && !MGraph && !MSubgraphNode && !MQueue->getCommandGraph() &&
245  !MQueue->is_in_fusion_mode() && !CGData.MRequirements.size() &&
246  !MStreamStorage.size() &&
247  (!CGData.MEvents.size() ||
248  (MQueue->isInOrder() &&
249  detail::Scheduler::areEventsSafeForSchedulerBypass(
250  CGData.MEvents, MQueue->getContextImplPtr())))) {

293  } else {

295  ->call<detail::PiApiKind::piEnqueueKernelLaunch>(Args...);
296  }
297  }(/* queue */
298  nullptr,

300  reinterpret_cast<pi_kernel>(MHostKernel->getPtr()),

302  MNDRDesc.Dims,
303  /* global_work_offset */ &MNDRDesc.GlobalOffset[0],

305  /* local_work_size */ &MNDRDesc.LocalSize[0],
306  /* num_events_in_wait_list */ 0,
307  /* event_wait_list */ nullptr,
308  /* event */ nullptr);
309  Result = PI_SUCCESS;
310  } else {
311  Result = enqueueImpKernel(
312  MQueue, MNDRDesc, MArgs, KernelBundleImpPtr, MKernel,
313  MKernelName.c_str(), RawEvents, NewEvent, nullptr,
314  MImpl->MKernelCacheConfig, MImpl->MKernelIsCooperative);
315  }
316  }
317 #ifdef XPTI_ENABLE_INSTRUMENTATION
318  // Emit signal only when event is created
319  if (NewEvent != nullptr) {
320  detail::emitInstrumentationGeneral(
321  StreamID, InstanceID, CmdTraceEvent, xpti::trace_signal,
322  static_cast<const void *>(NewEvent->getHandleRef()));
323  }
324  detail::emitInstrumentationGeneral(StreamID, InstanceID, CmdTraceEvent,
325  xpti::trace_task_end, nullptr);
326 #endif
327  return Result;
328  };

333  bool KernelUsesAssert =

568  MImpl->MUserFacingNodeType !=
569  ext::oneapi::experimental::node_type::empty
570  ? MImpl->MUserFacingNodeType
571  : ext::oneapi::experimental::detail::getNodeTypeFromCG(MCGType);
572 
573  // Create a new node in the graph representing this command-group
574  if (MQueue->isInOrder()) {
575  // In-order queues create implicit linear dependencies between nodes.
576  // Find the last node added to the graph from this queue, so our new
577  // node can set it as a predecessor.
578  auto DependentNode = GraphImpl->getLastInorderNode(MQueue);
579 
580  NodeImpl = DependentNode
581  ? GraphImpl->add(NodeType, std::move(CommandGroup),
582  {DependentNode})
583  : GraphImpl->add(NodeType, std::move(CommandGroup));
584 
585  // If we are recording an in-order queue remember the new node, so it
586  // can be used as a dependency for any more nodes recorded from this
587  // queue.
588  GraphImpl->setLastInorderNode(MQueue, NodeImpl);
589  } else {
590  NodeImpl = GraphImpl->add(NodeType, std::move(CommandGroup));
591  }
592 
593  // Associate an event with this new node and return the event.
594  GraphImpl->addEventForNode(GraphImpl, EventImpl, NodeImpl);
595 
596  NodeImpl->MNDRangeUsed = MImpl->MNDRangeUsed;
597 
598  return detail::createSyclObjFromImpl<event>(EventImpl);
599  }
600 
601  detail::EventImplPtr Event = detail::Scheduler::getInstance().addCG(
602  std::move(CommandGroup), std::move(MQueue));
603 
604  MLastEvent = detail::createSyclObjFromImpl<event>(Event);
605  return MLastEvent;

690 void handler::processArg(void *Ptr, const detail::kernel_param_kind_t &Kind,
691  const int Size, const size_t Index, size_t &IndexShift,
692  bool IsKernelCreatedFromSource, bool IsESIMD) {
693  using detail::kernel_param_kind_t;
694 
695  switch (Kind) {
696  case kernel_param_kind_t::kind_std_layout:
697  case kernel_param_kind_t::kind_pointer: {
698  MArgs.emplace_back(Kind, Ptr, Size, Index + IndexShift);
699  break;
700  }
701  case kernel_param_kind_t::kind_stream: {
702  // Stream contains several accessors inside.
703  stream *S = static_cast<stream *>(Ptr);
704 
705  detail::AccessorBaseHost *GBufBase =
706  static_cast<detail::AccessorBaseHost *>(&S->GlobalBuf);
707  detail::AccessorImplPtr GBufImpl = detail::getSyclObjImpl(*GBufBase);
708  detail::Requirement *GBufReq = GBufImpl.get();
709  addArgsForGlobalAccessor(GBufReq, Index, IndexShift, Size,
710  IsKernelCreatedFromSource,
711  MNDRDesc.GlobalSize.size(), MArgs, IsESIMD);
712  ++IndexShift;
713  detail::AccessorBaseHost *GOffsetBase =
714  static_cast<detail::AccessorBaseHost *>(&S->GlobalOffset);
715  detail::AccessorImplPtr GOfssetImpl = detail::getSyclObjImpl(*GOffsetBase);
716  detail::Requirement *GOffsetReq = GOfssetImpl.get();
717  addArgsForGlobalAccessor(GOffsetReq, Index, IndexShift, Size,
718  IsKernelCreatedFromSource,
719  MNDRDesc.GlobalSize.size(), MArgs, IsESIMD);
720  ++IndexShift;
721  detail::AccessorBaseHost *GFlushBase =
722  static_cast<detail::AccessorBaseHost *>(&S->GlobalFlushBuf);
723  detail::AccessorImplPtr GFlushImpl = detail::getSyclObjImpl(*GFlushBase);
724  detail::Requirement *GFlushReq = GFlushImpl.get();
725 
726  size_t GlobalSize = MNDRDesc.GlobalSize.size();
727  // If work group size wasn't set explicitly then it must be recieved
728  // from kernel attribute or set to default values.
729  // For now we can't get this attribute here.
730  // So we just suppose that WG size is always default for stream.
731  // TODO adjust MNDRDesc when device image contains kernel's attribute
732  if (GlobalSize == 0) {
733  // Suppose that work group size is 1 for every dimension
734  GlobalSize = MNDRDesc.NumWorkGroups.size();
735  }
736  addArgsForGlobalAccessor(GFlushReq, Index, IndexShift, Size,
737  IsKernelCreatedFromSource, GlobalSize, MArgs,
738  IsESIMD);
739  ++IndexShift;
740  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout,
741  &S->FlushBufferSize, sizeof(S->FlushBufferSize),
742  Index + IndexShift);
743 
744  break;
745  }
746  case kernel_param_kind_t::kind_accessor: {
747  // For args kind of accessor Size is information about accessor.
748  // The first 11 bits of Size encodes the accessor target.
749  const access::target AccTarget =
750  static_cast<access::target>(Size & AccessTargetMask);
751  switch (AccTarget) {
752  case access::target::device:
753  case access::target::constant_buffer: {
754  detail::Requirement *AccImpl = static_cast<detail::Requirement *>(Ptr);
755  addArgsForGlobalAccessor(AccImpl, Index, IndexShift, Size,
756  IsKernelCreatedFromSource,
757  MNDRDesc.GlobalSize.size(), MArgs, IsESIMD);
758  break;
759  }
760  case access::target::local: {
761  detail::LocalAccessorImplHost *LAcc =
762  static_cast<detail::LocalAccessorImplHost *>(Ptr);
763 
764  range<3> &Size = LAcc->MSize;
765  const int Dims = LAcc->MDims;
766  int SizeInBytes = LAcc->MElemSize;
767  for (int I = 0; I < Dims; ++I)
768  SizeInBytes *= Size[I];
769  // Some backends do not accept zero-sized local memory arguments, so we
770  // make it a minimum allocation of 1 byte.
771  SizeInBytes = std::max(SizeInBytes, 1);
772  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout, nullptr,
773  SizeInBytes, Index + IndexShift);
774  // TODO ESIMD currently does not suport MSize field passing yet
775  // accessor::init for ESIMD-mode accessor has a single field, translated
776  // to a single kernel argument set above.
777  if (!IsESIMD && !IsKernelCreatedFromSource) {
778  ++IndexShift;
779  const size_t SizeAccField = (Dims == 0 ? 1 : Dims) * sizeof(Size[0]);
780  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout, &Size,
781  SizeAccField, Index + IndexShift);
782  ++IndexShift;
783  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout, &Size,
784  SizeAccField, Index + IndexShift);
785  ++IndexShift;
786  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout, &Size,
787  SizeAccField, Index + IndexShift);
788  }
789  break;
790  }
791  case access::target::image:
792  case access::target::image_array: {
793  detail::Requirement *AccImpl = static_cast<detail::Requirement *>(Ptr);
794  MArgs.emplace_back(Kind, AccImpl, Size, Index + IndexShift);
795  if (!IsKernelCreatedFromSource) {
796  // TODO Handle additional kernel arguments for image class
797  // if the compiler front-end adds them.
798  }
799  break;
800  }
801  case access::target::host_image:
802  case access::target::host_task:
803  case access::target::host_buffer: {
804  throw sycl::invalid_parameter_error("Unsupported accessor target case.",
805  PI_ERROR_INVALID_OPERATION);
806  break;
807  }
808  }
809  break;
810  }
811  case kernel_param_kind_t::kind_sampler: {
812  MArgs.emplace_back(kernel_param_kind_t::kind_sampler, Ptr, sizeof(sampler),
813  Index + IndexShift);
814  break;
815  }
816  case kernel_param_kind_t::kind_specialization_constants_buffer: {
817  MArgs.emplace_back(
818  kernel_param_kind_t::kind_specialization_constants_buffer, Ptr, Size,
819  Index + IndexShift);
820  break;
821  }
822  case kernel_param_kind_t::kind_invalid:
823  throw runtime_error("Invalid kernel param kind", PI_ERROR_INVALID_VALUE);
824  break;
825  }
826 }

830 // We use the worst-case estimate because the lifetime of the vector is short.
831 // In processArg the kind_stream case introduces the maximum number of
832 // additional arguments. The case adds additional 12 arguments to the currently
833 // processed argument, hence worst-case estimate is 12+1=13.
834 // TODO: the constant can be removed if the size of MArgs will be calculated at

839  assert(MKernel && "MKernel is not initialized");

841  MArgs.clear();
842 
843  std::sort(
844  UnPreparedArgs.begin(), UnPreparedArgs.end(),
845  [](const detail::ArgDesc &first, const detail::ArgDesc &second) -> bool {
846  return (first.MIndex < second.MIndex);

849  const bool IsKernelCreatedFromSource = MKernel->isCreatedFromSource();
850  MArgs.reserve(MaxNumAdditionalArgs * UnPreparedArgs.size());

854  void *Ptr = UnPreparedArgs[I].MPtr;
855  const detail::kernel_param_kind_t &Kind = UnPreparedArgs[I].MType;
856  const int &Size = UnPreparedArgs[I].MSize;
857  const int Index = UnPreparedArgs[I].MIndex;
858  processArg(Ptr, Kind, Size, Index, IndexShift, IsKernelCreatedFromSource,
859  false);
