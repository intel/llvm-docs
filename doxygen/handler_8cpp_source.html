32 // Returns a shared_ptr to kernel_bundle stored in the extended members vector.
33 // If there is no kernel_bundle created:
34 // returns newly created kernel_bundle if Insert is true
35 // returns shared_ptr(nullptr) if Insert is false
36 std::shared_ptr<detail::kernel_bundle_impl>
37 handler::getOrInsertHandlerKernelBundle(bool Insert) const {
38 
39  std::lock_guard<std::mutex> Lock(
40  detail::GlobalHandler::instance().getHandlerExtendedMembersMutex());
41 
42  assert(!MSharedPtrStorage.empty());
43 
44  std::shared_ptr<std::vector<detail::ExtendedMemberT>> ExendedMembersVec =
45  detail::convertToExtendedMembers(MSharedPtrStorage[0]);
46 
47  // Look for the kernel bundle in extended members
48  std::shared_ptr<detail::kernel_bundle_impl> KernelBundleImpPtr;
49  for (const detail::ExtendedMemberT &EMember : *ExendedMembersVec)
50  if (detail::ExtendedMembersType::HANDLER_KERNEL_BUNDLE == EMember.MType) {
51  KernelBundleImpPtr =
52  std::static_pointer_cast<detail::kernel_bundle_impl>(EMember.MData);
53  break;
54  }
55 
56  // No kernel bundle yet, create one
57  if (!KernelBundleImpPtr && Insert) {
58  KernelBundleImpPtr = detail::getSyclObjImpl(
59  get_kernel_bundle<bundle_state::input>(MQueue->get_context()));
60  if (KernelBundleImpPtr->empty()) {
61  KernelBundleImpPtr = detail::getSyclObjImpl(
62  get_kernel_bundle<bundle_state::executable>(MQueue->get_context()));
63  }
64 
65  detail::ExtendedMemberT EMember = {
66  detail::ExtendedMembersType::HANDLER_KERNEL_BUNDLE, KernelBundleImpPtr};
67 
68  ExendedMembersVec->push_back(EMember);
69  }
70 
71  return KernelBundleImpPtr;
72 }

99 event handler::finalize() {
100  // This block of code is needed only for reduction implementation.
101  // It is harmless (does nothing) for everything else.
102  if (MIsFinalized)
103  return MLastEvent;
104  MIsFinalized = true;
105 
106  // Kernel_bundles could not be used before CGType version 1
107  if (getCGTypeVersion(MCGType) >
108  static_cast<unsigned int>(detail::CG::CG_VERSION::V0)) {
109  // If there were uses of set_specialization_constant build the kernel_bundle
110  std::shared_ptr<detail::kernel_bundle_impl> KernelBundleImpPtr =
111  getOrInsertHandlerKernelBundle(/*Insert=*/false);
112  if (KernelBundleImpPtr) {
113  switch (KernelBundleImpPtr->get_bundle_state()) {
114  case bundle_state::input: {
115  // Underlying level expects kernel_bundle to be in executable state
116  kernel_bundle<bundle_state::executable> ExecBundle = build(
117  detail::createSyclObjFromImpl<kernel_bundle<bundle_state::input>>(
118  KernelBundleImpPtr));
119  setHandlerKernelBundle(detail::getSyclObjImpl(ExecBundle));
120  break;
121  }
122  case bundle_state::executable:
123  // Nothing to do
124  break;
125  case bundle_state::object:
126  assert(0 && "Expected that the bundle is either in input or executable "
127  "states.");
128  break;
129  }
130  }
131  }
132 
133  unique_ptr_class<detail::CG> CommandGroup;
134  switch (getType()) {
135  case detail::CG::KERNEL:
136  case detail::CG::RUN_ON_HOST_INTEL: {
137  CommandGroup.reset(new detail::CGExecKernel(
138  std::move(MNDRDesc), std::move(MHostKernel), std::move(MKernel),
139  std::move(MArgsStorage), std::move(MAccStorage),
140  std::move(MSharedPtrStorage), std::move(MRequirements),
141  std::move(MEvents), std::move(MArgs), std::move(MKernelName),
142  std::move(MOSModuleHandle), std::move(MStreamStorage), MCGType,
143  MCodeLoc));
144  break;
145  }
146  case detail::CG::CODEPLAY_INTEROP_TASK:
147  CommandGroup.reset(new detail::CGInteropTask(
148  std::move(MInteropTask), std::move(MArgsStorage),
149  std::move(MAccStorage), std::move(MSharedPtrStorage),
150  std::move(MRequirements), std::move(MEvents), MCGType, MCodeLoc));
151  break;
152  case detail::CG::COPY_ACC_TO_PTR:
153  case detail::CG::COPY_PTR_TO_ACC:
154  case detail::CG::COPY_ACC_TO_ACC:
155  CommandGroup.reset(new detail::CGCopy(
156  MCGType, MSrcPtr, MDstPtr, std::move(MArgsStorage),
157  std::move(MAccStorage), std::move(MSharedPtrStorage),
158  std::move(MRequirements), std::move(MEvents), MCodeLoc));
159  break;
160  case detail::CG::FILL:
161  CommandGroup.reset(new detail::CGFill(
162  std::move(MPattern), MDstPtr, std::move(MArgsStorage),
163  std::move(MAccStorage), std::move(MSharedPtrStorage),
164  std::move(MRequirements), std::move(MEvents), MCodeLoc));
165  break;
166  case detail::CG::UPDATE_HOST:
167  CommandGroup.reset(new detail::CGUpdateHost(
168  MDstPtr, std::move(MArgsStorage), std::move(MAccStorage),
169  std::move(MSharedPtrStorage), std::move(MRequirements),
170  std::move(MEvents), MCodeLoc));
171  break;
172  case detail::CG::COPY_USM:
173  CommandGroup.reset(new detail::CGCopyUSM(
174  MSrcPtr, MDstPtr, MLength, std::move(MArgsStorage),
175  std::move(MAccStorage), std::move(MSharedPtrStorage),
176  std::move(MRequirements), std::move(MEvents), MCodeLoc));
177  break;
178  case detail::CG::FILL_USM:
179  CommandGroup.reset(new detail::CGFillUSM(
180  std::move(MPattern), MDstPtr, MLength, std::move(MArgsStorage),
181  std::move(MAccStorage), std::move(MSharedPtrStorage),
182  std::move(MRequirements), std::move(MEvents), MCodeLoc));
183  break;
184  case detail::CG::PREFETCH_USM:
185  CommandGroup.reset(new detail::CGPrefetchUSM(
186  MDstPtr, MLength, std::move(MArgsStorage), std::move(MAccStorage),
187  std::move(MSharedPtrStorage), std::move(MRequirements),
188  std::move(MEvents), MCodeLoc));
189  break;
190  case detail::CG::CODEPLAY_HOST_TASK:
191  CommandGroup.reset(new detail::CGHostTask(
192  std::move(MHostTask), MQueue, MQueue->getContextImplPtr(),
193  std::move(MArgs), std::move(MArgsStorage), std::move(MAccStorage),
194  std::move(MSharedPtrStorage), std::move(MRequirements),
195  std::move(MEvents), MCGType, MCodeLoc));
196  break;
197  case detail::CG::BARRIER:
198  case detail::CG::BARRIER_WAITLIST:
199  CommandGroup.reset(new detail::CGBarrier(
200  std::move(MEventsWaitWithBarrier), std::move(MArgsStorage),
201  std::move(MAccStorage), std::move(MSharedPtrStorage),
202  std::move(MRequirements), std::move(MEvents), MCGType, MCodeLoc));
203  break;
204  case detail::CG::NONE:
205  throw runtime_error("Command group submitted without a kernel or a "
206  "explicit memory operation.",
207  PI_INVALID_OPERATION);
208  }
209 
210  if (!CommandGroup)
211  throw sycl::runtime_error(
212  "Internal Error. Command group cannot be constructed.",
213  PI_INVALID_OPERATION);
214 
215  detail::EventImplPtr Event = detail::Scheduler::getInstance().addCG(
216  std::move(CommandGroup), std::move(MQueue));
217 
218  MLastEvent = detail::createSyclObjFromImpl<event>(Event);
219  return MLastEvent;
220 }

245 void handler::processArg(void *Ptr, const detail::kernel_param_kind_t &Kind,
246  const int Size, const size_t Index, size_t &IndexShift,
247  bool IsKernelCreatedFromSource, bool IsESIMD) {
248  using detail::kernel_param_kind_t;
249 
250  switch (Kind) {
251  case kernel_param_kind_t::kind_std_layout:
252  case kernel_param_kind_t::kind_pointer: {
253  MArgs.emplace_back(Kind, Ptr, Size, Index + IndexShift);
254  break;
255  }
256  case kernel_param_kind_t::kind_accessor: {
257  // For args kind of accessor Size is information about accessor.
258  // The first 11 bits of Size encodes the accessor target.
259  const access::target AccTarget = static_cast<access::target>(Size & 0x7ff);
260  switch (AccTarget) {
261  case access::target::global_buffer:
262  case access::target::constant_buffer: {
263  detail::Requirement *AccImpl = static_cast<detail::Requirement *>(Ptr);
264 
265  // Stream implementation creates an accessor with initial size for
266  // work item. Number of work items is not available during
267  // stream construction, that is why size of the accessor is updated here
268  // using information about number of work items.
269  if (AccImpl->PerWI) {
270  AccImpl->resize(MNDRDesc.GlobalSize.size());
271  }
272  MArgs.emplace_back(Kind, AccImpl, Size, Index + IndexShift);
273 
274  // TODO ESIMD currently does not suport offset, memory and access ranges -
275  // accessor::init for ESIMD-mode accessor has a single field, translated
276  // to a single kernel argument set above.
277  if (!IsKernelCreatedFromSource && !IsESIMD) {
278  // Dimensionality of the buffer is 1 when dimensionality of the
279  // accessor is 0.
280  const size_t SizeAccField =
281  sizeof(size_t) * (AccImpl->MDims == 0 ? 1 : AccImpl->MDims);
282  ++IndexShift;
283  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout,
284  &AccImpl->MAccessRange[0], SizeAccField,
285  Index + IndexShift);
286  ++IndexShift;
287  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout,
288  &AccImpl->MMemoryRange[0], SizeAccField,
289  Index + IndexShift);
290  ++IndexShift;
291  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout,
292  &AccImpl->MOffset[0], SizeAccField,
293  Index + IndexShift);
294  }
295  break;
296  }
297  case access::target::local: {
298  detail::LocalAccessorImplHost *LAcc =
299  static_cast<detail::LocalAccessorImplHost *>(Ptr);
300 
301  range<3> &Size = LAcc->MSize;
302  const int Dims = LAcc->MDims;
303  int SizeInBytes = LAcc->MElemSize;
304  for (int I = 0; I < Dims; ++I)
305  SizeInBytes *= Size[I];
306  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout, nullptr,
307  SizeInBytes, Index + IndexShift);
308  if (!IsKernelCreatedFromSource) {
309  ++IndexShift;
310  const size_t SizeAccField = Dims * sizeof(Size[0]);
311  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout, &Size,
312  SizeAccField, Index + IndexShift);
313  ++IndexShift;
314  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout, &Size,
315  SizeAccField, Index + IndexShift);
316  ++IndexShift;
317  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout, &Size,
318  SizeAccField, Index + IndexShift);
319  }
320  break;
321  }
322  case access::target::image:
323  case access::target::image_array: {
324  detail::Requirement *AccImpl = static_cast<detail::Requirement *>(Ptr);
325  MArgs.emplace_back(Kind, AccImpl, Size, Index + IndexShift);
326  if (!IsKernelCreatedFromSource) {
327  // TODO Handle additional kernel arguments for image class
328  // if the compiler front-end adds them.
329  }
330  break;
331  }
332  case access::target::host_image:
333  case access::target::host_buffer: {
334  throw cl::sycl::invalid_parameter_error(
335  "Unsupported accessor target case.", PI_INVALID_OPERATION);
336  break;
337  }
338  }
339  break;
340  }
341  case kernel_param_kind_t::kind_sampler: {
342  MArgs.emplace_back(kernel_param_kind_t::kind_sampler, Ptr, sizeof(sampler),
343  Index + IndexShift);
344  break;
345  }
346  case kernel_param_kind_t::kind_specialization_constants_buffer: {
347  MArgs.emplace_back(
348  kernel_param_kind_t::kind_specialization_constants_buffer, Ptr, Size,
349  Index + IndexShift);
350  break;
351  }
352  }

355 void handler::extractArgsAndReqs() {
356  assert(MKernel && "MKernel is not initialized");
357  std::vector<detail::ArgDesc> UnPreparedArgs = std::move(MArgs);
358  MArgs.clear();
359 
360  std::sort(
361  UnPreparedArgs.begin(), UnPreparedArgs.end(),
362  [](const detail::ArgDesc &first, const detail::ArgDesc &second) -> bool {
363  return (first.MIndex < second.MIndex);
364  });
365 
366  const bool IsKernelCreatedFromSource = MKernel->isCreatedFromSource();
367 
368  size_t IndexShift = 0;
369  for (size_t I = 0; I < UnPreparedArgs.size(); ++I) {
370  void *Ptr = UnPreparedArgs[I].MPtr;
371  const detail::kernel_param_kind_t &Kind = UnPreparedArgs[I].MType;
372  const int &Size = UnPreparedArgs[I].MSize;
373  const int Index = UnPreparedArgs[I].MIndex;
374  processArg(Ptr, Kind, Size, Index, IndexShift, IsKernelCreatedFromSource,
375  false);
376  }
377 }

386 void handler::extractArgsAndReqsFromLambda(
387  char *LambdaPtr, size_t KernelArgsNum,
388  const detail::kernel_param_desc_t *KernelArgs, bool IsESIMD) {
389  const bool IsKernelCreatedFromSource = false;
390  size_t IndexShift = 0;
391  for (size_t I = 0; I < KernelArgsNum; ++I) {
392  void *Ptr = LambdaPtr + KernelArgs[I].offset;
393  const detail::kernel_param_kind_t &Kind = KernelArgs[I].kind;
394  const int &Size = KernelArgs[I].info;
395  if (Kind == detail::kernel_param_kind_t::kind_accessor) {
396  // For args kind of accessor Size is information about accessor.
397  // The first 11 bits of Size encodes the accessor target.
398  const access::target AccTarget =
399  static_cast<access::target>(Size & 0x7ff);
400  if ((AccTarget == access::target::global_buffer ||
401  AccTarget == access::target::constant_buffer) ||
402  (AccTarget == access::target::image ||
403  AccTarget == access::target::image_array)) {
404  detail::AccessorBaseHost *AccBase =
405  static_cast<detail::AccessorBaseHost *>(Ptr);
406  Ptr = detail::getSyclObjImpl(*AccBase).get();
407  } else if (AccTarget == access::target::local) {
408  detail::LocalAccessorBaseHost *LocalAccBase =
409  static_cast<detail::LocalAccessorBaseHost *>(Ptr);
410  Ptr = detail::getSyclObjImpl(*LocalAccBase).get();
411  }
412  }
413  processArg(Ptr, Kind, Size, I, IndexShift, IsKernelCreatedFromSource,
414  IsESIMD);
415  }
416 }
