20 event handler::finalize(const cl::sycl::detail::code_location &Payload) {
21  sycl::event EventRet;
22  unique_ptr_class<detail::CG> CommandGroup;
23  switch (MCGType) {
24  case detail::CG::KERNEL:
25  case detail::CG::RUN_ON_HOST_INTEL: {
26  CommandGroup.reset(new detail::CGExecKernel(
27  std::move(MNDRDesc), std::move(MHostKernel), std::move(MKernel),
28  std::move(MArgsStorage), std::move(MAccStorage),
29  std::move(MSharedPtrStorage), std::move(MRequirements),
30  std::move(MEvents), std::move(MArgs), std::move(MKernelName),
31  std::move(MOSModuleHandle), std::move(MStreamStorage), MCGType,
32  Payload));
33  break;
34  }
35  case detail::CG::INTEROP_TASK_CODEPLAY:
36  CommandGroup.reset(new detail::CGInteropTask(
37  std::move(MInteropTask), std::move(MArgsStorage),
38  std::move(MAccStorage), std::move(MSharedPtrStorage),
39  std::move(MRequirements), std::move(MEvents), MCGType, Payload));
40  break;
41  case detail::CG::COPY_ACC_TO_PTR:
42  case detail::CG::COPY_PTR_TO_ACC:
43  case detail::CG::COPY_ACC_TO_ACC:
44  CommandGroup.reset(new detail::CGCopy(
45  MCGType, MSrcPtr, MDstPtr, std::move(MArgsStorage),
46  std::move(MAccStorage), std::move(MSharedPtrStorage),
47  std::move(MRequirements), std::move(MEvents), Payload));
48  break;
49  case detail::CG::FILL:
50  CommandGroup.reset(new detail::CGFill(
51  std::move(MPattern), MDstPtr, std::move(MArgsStorage),
52  std::move(MAccStorage), std::move(MSharedPtrStorage),
53  std::move(MRequirements), std::move(MEvents), Payload));
54  break;
55  case detail::CG::UPDATE_HOST:
56  CommandGroup.reset(new detail::CGUpdateHost(
57  MDstPtr, std::move(MArgsStorage), std::move(MAccStorage),
58  std::move(MSharedPtrStorage), std::move(MRequirements),
59  std::move(MEvents), Payload));
60  break;
61  case detail::CG::COPY_USM:
62  CommandGroup.reset(new detail::CGCopyUSM(
63  MSrcPtr, MDstPtr, MLength, std::move(MArgsStorage),
64  std::move(MAccStorage), std::move(MSharedPtrStorage),
65  std::move(MRequirements), std::move(MEvents), Payload));
66  break;
67  case detail::CG::FILL_USM:
68  CommandGroup.reset(new detail::CGFillUSM(
69  std::move(MPattern), MDstPtr, MLength, std::move(MArgsStorage),
70  std::move(MAccStorage), std::move(MSharedPtrStorage),
71  std::move(MRequirements), std::move(MEvents), Payload));
72  break;
73  case detail::CG::PREFETCH_USM:
74  CommandGroup.reset(new detail::CGPrefetchUSM(
75  MDstPtr, MLength, std::move(MArgsStorage), std::move(MAccStorage),
76  std::move(MSharedPtrStorage), std::move(MRequirements),
77  std::move(MEvents), Payload));
78  break;
79  case detail::CG::NONE:
80  throw runtime_error("Command group submitted without a kernel or a "
81  "explicit memory operation.",
82  PI_INVALID_OPERATION);
83  default:
84  throw runtime_error("Unhandled type of command group",
85  PI_INVALID_OPERATION);
86  }
87 
88  detail::EventImplPtr Event = detail::Scheduler::getInstance().addCG(
89  std::move(CommandGroup), std::move(MQueue));
90 
91  EventRet = detail::createSyclObjFromImpl<event>(Event);
92  return EventRet;
93 }

95 void handler::processArg(void *Ptr, const detail::kernel_param_kind_t &Kind,
96  const int Size, const size_t Index, size_t &IndexShift,
97  bool IsKernelCreatedFromSource) {
98  using detail::kernel_param_kind_t;
99 
100  switch (Kind) {
101  case kernel_param_kind_t::kind_std_layout:
102  case kernel_param_kind_t::kind_pointer: {
103  MArgs.emplace_back(Kind, Ptr, Size, Index + IndexShift);
104  break;
105  }
106  case kernel_param_kind_t::kind_accessor: {
107  // For args kind of accessor Size is information about accessor.
108  // The first 11 bits of Size encodes the accessor target.
109  const access::target AccTarget = static_cast<access::target>(Size & 0x7ff);
110  switch (AccTarget) {
111  case access::target::global_buffer:
112  case access::target::constant_buffer: {
113  detail::Requirement *AccImpl = static_cast<detail::Requirement *>(Ptr);
114  MArgs.emplace_back(Kind, AccImpl, Size, Index + IndexShift);
115  if (!IsKernelCreatedFromSource) {
116  // Dimensionality of the buffer is 1 when dimensionality of the
117  // accessor is 0.
118  const size_t SizeAccField =
119  sizeof(size_t) * (AccImpl->MDims == 0 ? 1 : AccImpl->MDims);
120  ++IndexShift;
121  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout,
122  &AccImpl->MAccessRange[0], SizeAccField,
123  Index + IndexShift);
124  ++IndexShift;
125  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout,
126  &AccImpl->MMemoryRange[0], SizeAccField,
127  Index + IndexShift);
128  ++IndexShift;
129  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout,
130  &AccImpl->MOffset[0], SizeAccField,
131  Index + IndexShift);
132  }
133  break;
134  }
135  case access::target::local: {
136  detail::LocalAccessorImplHost *LAcc =
137  static_cast<detail::LocalAccessorImplHost *>(Ptr);
138  // Stream implementation creates local accessor with size per work item
139  // in work group. Number of work items is not available during stream
140  // construction, that is why size of the accessor is updated here using
141  // information about number of work items in the work group.
142  if (LAcc->PerWI)
143  LAcc->resize(MNDRDesc.LocalSize.size(), MNDRDesc.GlobalSize.size());
144  range<3> &Size = LAcc->MSize;
145  const int Dims = LAcc->MDims;
146  int SizeInBytes = LAcc->MElemSize;
147  for (int I = 0; I < Dims; ++I)
148  SizeInBytes *= Size[I];
149  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout, nullptr,
150  SizeInBytes, Index + IndexShift);
151  if (!IsKernelCreatedFromSource) {
152  ++IndexShift;
153  const size_t SizeAccField = Dims * sizeof(Size[0]);
154  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout, &Size,
155  SizeAccField, Index + IndexShift);
156  ++IndexShift;
157  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout, &Size,
158  SizeAccField, Index + IndexShift);
159  ++IndexShift;
160  MArgs.emplace_back(kernel_param_kind_t::kind_std_layout, &Size,
161  SizeAccField, Index + IndexShift);
162  }
163  break;
164  }
165  case access::target::image:
166  case access::target::image_array: {
167  detail::Requirement *AccImpl = static_cast<detail::Requirement *>(Ptr);
168  MArgs.emplace_back(Kind, AccImpl, Size, Index + IndexShift);
169  if (!IsKernelCreatedFromSource) {
170  // TODO Handle additional kernel arguments for image class
171  // if the compiler front-end adds them.
172  }
173  break;
174  }
175  case access::target::host_image:
176  case access::target::host_buffer: {
177  throw cl::sycl::invalid_parameter_error(
178  "Unsupported accessor target case.", PI_INVALID_OPERATION);
179  break;
180  }
181  }
182  break;
183  }
184  case kernel_param_kind_t::kind_sampler: {
185  MArgs.emplace_back(kernel_param_kind_t::kind_sampler, Ptr, sizeof(sampler),
186  Index + IndexShift);
187  break;
188  }
189  }
190 }

192 void handler::extractArgsAndReqs() {
193  assert(MKernel && "MKernel is not initialized");
194  std::vector<detail::ArgDesc> UnPreparedArgs = std::move(MArgs);
195  MArgs.clear();
196 
197  std::sort(
198  UnPreparedArgs.begin(), UnPreparedArgs.end(),
199  [](const detail::ArgDesc &first, const detail::ArgDesc &second) -> bool {
200  return (first.MIndex < second.MIndex);
201  });
202 
203  const bool IsKernelCreatedFromSource = MKernel->isCreatedFromSource();
204 
205  size_t IndexShift = 0;
206  for (size_t I = 0; I < UnPreparedArgs.size(); ++I) {
207  void *Ptr = UnPreparedArgs[I].MPtr;
208  const detail::kernel_param_kind_t &Kind = UnPreparedArgs[I].MType;
209  const int &Size = UnPreparedArgs[I].MSize;
210  const int Index = UnPreparedArgs[I].MIndex;
211  processArg(Ptr, Kind, Size, Index, IndexShift, IsKernelCreatedFromSource);
212  }
213 }

215 void handler::extractArgsAndReqsFromLambda(
216  char *LambdaPtr, size_t KernelArgsNum,
217  const detail::kernel_param_desc_t *KernelArgs) {
218  const bool IsKernelCreatedFromSource = false;
219  size_t IndexShift = 0;
220  for (size_t I = 0; I < KernelArgsNum; ++I) {
221  void *Ptr = LambdaPtr + KernelArgs[I].offset;
222  const detail::kernel_param_kind_t &Kind = KernelArgs[I].kind;
223  const int &Size = KernelArgs[I].info;
224  if (Kind == detail::kernel_param_kind_t::kind_accessor) {
225  // For args kind of accessor Size is information about accessor.
226  // The first 11 bits of Size encodes the accessor target.
227  const access::target AccTarget =
228  static_cast<access::target>(Size & 0x7ff);
229  if ((AccTarget == access::target::global_buffer ||
230  AccTarget == access::target::constant_buffer) ||
231  (AccTarget == access::target::image ||
232  AccTarget == access::target::image_array)) {
233  detail::AccessorBaseHost *AccBase =
234  static_cast<detail::AccessorBaseHost *>(Ptr);
235  Ptr = detail::getSyclObjImpl(*AccBase).get();
236  } else if (AccTarget == access::target::local) {
237  detail::LocalAccessorBaseHost *LocalAccBase =
238  static_cast<detail::LocalAccessorBaseHost *>(Ptr);
239  Ptr = detail::getSyclObjImpl(*LocalAccBase).get();
240  }
241  }
242  processArg(Ptr, Kind, Size, I, IndexShift, IsKernelCreatedFromSource);
243  }
244 }
