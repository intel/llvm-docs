101 template <int dimensions>
102 void group_barrier(ext::oneapi::experimental::root_group<dimensions> G,
103  memory_scope FenceScope = decltype(G)::fence_scope) {
104 #ifdef __SYCL_DEVICE_ONLY__
105  // Root group barrier synchronizes using a work group barrier if there's only
106  // one work group. This allows backends to ignore the ControlBarrier with
107  // Device scope if their maximum number of work groups is 1. This is a
108  // workaround that's not intended to reduce the bar for SPIR-V modules
109  // acceptance, but rather make a pessimistic case work until we have full
110  // support for the device barrier built-in from backends.
111  const auto ChildGroup = ext::oneapi::experimental::this_group<dimensions>();
112  if (ChildGroup.get_group_linear_range() == 1) {
113  group_barrier(ChildGroup);
114  } else {
115  detail::spirv::ControlBarrier(G, FenceScope, memory_order::seq_cst);
116  }
117 #else
118  (void)G;
119  (void)FenceScope;
120  throw sycl::runtime_error("Barriers are not supported on host device",
121  PI_ERROR_INVALID_DEVICE);
122 #endif
123 }
