99 template <int dimensions>
100 void group_barrier(ext::oneapi::experimental::root_group<dimensions> G,
101  memory_scope FenceScope = decltype(G)::fence_scope) {
102 #ifdef __SYCL_DEVICE_ONLY__
103  // Root group barrier synchronizes using a work group barrier if there's only
104  // one work group. This allows backends to ignore the ControlBarrier with
105  // Device scope if their maximum number of work groups is 1. This is a
106  // workaround that's not intended to reduce the bar for SPIR-V modules
107  // acceptance, but rather make a pessimistic case work until we have full
108  // support for the device barrier built-in from backends.
109  const auto ChildGroup = ext::oneapi::experimental::this_group<dimensions>();
110  if (ChildGroup.get_group_linear_range() == 1) {
111  group_barrier(ChildGroup);
112  } else {
113  detail::spirv::ControlBarrier(G, FenceScope, memory_order::seq_cst);
114  }
115 #else
116  (void)G;
117  (void)FenceScope;
118  throw sycl::exception(make_error_code(errc::runtime),
119  "Barriers are not supported on host");
120 #endif
121 }
