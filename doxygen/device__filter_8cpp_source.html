22 std::vector<std::string_view> tokenize(const std::string_view &Filter,
23  const std::string &Delim,
24  bool ProhibitEmptyTokens = false) {
25  std::vector<std::string_view> Tokens;
26  size_t Pos = 0;
27  size_t LastPos = 0;
28 
29  while ((Pos = Filter.find(Delim, LastPos)) != std::string::npos) {
30  std::string_view Tok(Filter.data() + LastPos, (Pos - LastPos));
31 
32  if (!Tok.empty()) {
33  Tokens.push_back(Tok);
34  } else if (ProhibitEmptyTokens) {
35  throw sycl::exception(
36  sycl::make_error_code(errc::invalid),
37  "ONEAPI_DEVICE_SELECTOR parsing error. Empty input before '" + Delim +
38  "' delimiter is not allowed.");
39  }
40  // move the search starting index
41  LastPos = Pos + 1;
42  }
43 
44  // Add remainder if any
45  if (LastPos < Filter.size()) {
46  std::string_view Tok(Filter.data() + LastPos, Filter.size() - LastPos);
47  Tokens.push_back(Tok);
48  } else if ((LastPos != 0) && ProhibitEmptyTokens) {
49  // if delimiter is the last sybmol in the string.
50  throw sycl::exception(
51  sycl::make_error_code(errc::invalid),
52  "ONEAPI_DEVICE_SELECTOR parsing error. Empty input after '" + Delim +
53  "' delimiter is not allowed.");
54  }
55  return Tokens;
56 }

61 static backend Parse_ODS_Backend(const std::string_view &BackendStr,
62  const std::string_view &FullEntry) {
63  // Check if the first entry matches with a known backend type
64  auto SyclBeMap =
65  getSyclBeMap(); // <-- std::array<std::pair<std::string, backend>>
66  // [{"level_zero", backend::level_zero}, {"*", ::all}, ...
67  auto It =
68  std::find_if(std::begin(SyclBeMap), std::end(SyclBeMap),
69  [&](auto BePair) { return BackendStr == BePair.first; });
70 
71  if (It == SyclBeMap.end()) {
72  // backend is required
73  std::stringstream ss;
74  ss << "ONEAPI_DEVICE_SELECTOR parsing error. Backend is required but "
75  "missing from \""
76  << FullEntry << "\"";
77  throw sycl::exception(sycl::make_error_code(errc::invalid), ss.str());
78  } else {
79  return It->second;
80  }
81 }

83 static void Parse_ODS_Device(ods_target &Target,
84  const std::string_view &DeviceStr) {
85  // DeviceStr will be: 'gpu', '*', '0', '0.1', 'gpu.*', '0.*', or 'gpu.2', etc.
86  std::vector<std::string_view> DeviceSubTuple =
87  tokenize(DeviceStr, ".", true /* ProhibitEmptyTokens */);
88  if (DeviceSubTuple.empty())
89  throw sycl::exception(
90  sycl::make_error_code(errc::invalid),
91  "ONEAPI_DEVICE_SELECTOR parsing error. Device must be specified.");
92 
93  std::string_view TopDeviceStr = DeviceSubTuple[0];
94 
95  // Handle explicit device type (e.g. 'gpu').
96  auto DeviceTypeMap = getSyclDeviceTypeMap<
97 #ifndef __INTEL_PREVIEW_BREAKING_CHANGES
98  true /*Enable 'acc'*/
99 #endif
100  >(); // <-- std::array<std::pair<std::string,
101  // info::device::type>>
102 
103  auto It =
104  std::find_if(std::begin(DeviceTypeMap), std::end(DeviceTypeMap),
105  [&](auto DtPair) { return TopDeviceStr == DtPair.first; });
106  if (It != DeviceTypeMap.end()) {
107  Target.DeviceType = It->second;
108  // Handle wildcard.
109  if (TopDeviceStr[0] == '*') {
110  Target.HasDeviceWildCard = true;
111  Target.DeviceType = {};
112  }
113  } else { // Only thing left is a number.
114  std::string TDS(TopDeviceStr);
115  try {
116  Target.DeviceNum = std::stoi(TDS);
117  } catch (...) {
118  throw sycl::exception(sycl::make_error_code(errc::invalid),
119  "error parsing device number: " + TDS);
120  }
121  }
122 
123  if (DeviceSubTuple.size() >= 2) {
124  // We have a subdevice.
125  // The grammar for sub-devices is ... restrictive. Neither 'gpu.0' nor
126  // 'gpu.*' are allowed. If wanting a sub-device, then the device itself must
127  // be specified by a number or a wildcard, and if by wildcard, the only
128  // allowable sub-device is another wildcard.
129 
130  if (Target.DeviceType)
131  throw sycl::exception(
132  sycl::make_error_code(errc::invalid),
133  "sub-devices can only be requested when parent device is specified "
134  "by number or wildcard, not a device type like 'gpu'");
135 
136  std::string_view SubDeviceStr = DeviceSubTuple[1];
137  // SubDeviceStr is wildcard or number.
138  if (SubDeviceStr[0] == '*') {
139  Target.HasSubDeviceWildCard = true;
140  } else {
141  // sub-device requested by number. So parent device must be a number too
142  // or it's a parsing error.
143  if (Target.HasDeviceWildCard)
144  throw sycl::exception(sycl::make_error_code(errc::invalid),
145  "sub-device can't be requested by number if "
146  "parent device is specified by a wildcard.");
147 
148  std::string SDS(SubDeviceStr);
149  try {
150  Target.SubDeviceNum = std::stoi(SDS);
151  } catch (...) {
152  throw sycl::exception(sycl::make_error_code(errc::invalid),
153  "error parsing sub-device index: " + SDS);
154  }
155  }
156  }
157  if (DeviceSubTuple.size() == 3) {
158  // We have a sub-sub-device.
159  // Similar rules for sub-sub-devices as for sub-devices above.
160 
161  std::string_view SubSubDeviceStr = DeviceSubTuple[2];
162  if (SubSubDeviceStr[0] == '*') {
163  Target.HasSubSubDeviceWildCard = true;
164  } else {
165  // sub-sub-device requested by number. So partition above must be a number
166  // too or it's a parsing error.
167  if (Target.HasSubDeviceWildCard)
168  throw sycl::exception(sycl::make_error_code(errc::invalid),
169  "sub-sub-device can't be requested by number if "
170  "sub-device before is specified by a wildcard.");
171 
172  std::string SSDS(SubSubDeviceStr);
173  try {
174  Target.SubSubDeviceNum = std::stoi(SSDS);
175  } catch (...) {
176  throw sycl::exception(sycl::make_error_code(errc::invalid),
177  "error parsing sub-sub-device index: " + SSDS);
178  }
179  }
180  } else if (DeviceSubTuple.size() > 3) {
181  std::stringstream ss;
182  ss << "error parsing " << DeviceStr
183  << " Only two levels of sub-devices supported at this time";
184  throw sycl::exception(sycl::make_error_code(errc::invalid), ss.str());
185  }
186 }

188 std::vector<ods_target>
189 Parse_ONEAPI_DEVICE_SELECTOR(const std::string &envString) {
190  // lowercase
191  std::string envStr = envString;
192  std::transform(envStr.begin(), envStr.end(), envStr.begin(), ::tolower);
193 
194  std::vector<ods_target> Result;
195  if (envStr.empty()) {
196  ods_target acceptAnything;
197  Result.push_back(acceptAnything);
198  return Result;
199  }
200 
201  std::vector<std::string_view> Entries = tokenize(envStr, ";");
202  unsigned int negative_filters = 0;
203  // Each entry: "level_zero:gpu" or "opencl:0.0,0.1" or "opencl:*" but NOT just
204  // "opencl".
205  for (const auto Entry : Entries) {
206  std::vector<std::string_view> Pair =
207  tokenize(Entry, ":", true /* ProhibitEmptyTokens */);
208 
209  if (Pair.empty()) {
210  std::stringstream ss;
211  ss << "Incomplete selector! Backend and device must be specified.";
212  throw sycl::exception(sycl::make_error_code(errc::invalid), ss.str());
213  } else if (Pair.size() == 1) {
214  std::stringstream ss;
215  ss << "Incomplete selector! Try '" << Pair[0]
216  << ":*' if all devices under the backend was original intention.";
217  throw sycl::exception(sycl::make_error_code(errc::invalid), ss.str());
218  } else if (Pair.size() == 2) {
219  backend be = Parse_ODS_Backend(Pair[0], Entry); // Pair[0] is backend.
220  std::vector<std::string_view> Targets = tokenize(Pair[1], ",");
221  for (auto TargetStr : Targets) {
222  ods_target DeviceTarget(be);
223  if (Entry[0] == '!') { // negative filter
224  DeviceTarget.IsNegativeTarget = true;
225  ++negative_filters;
226  } else { // positive filter
227  // no need to set IsNegativeTarget=false because it is so by default.
228  // ensure that no negative filter has been seen because all
229  // negative filters must come after all positive filters
230  if (negative_filters > 0) {
231  std::stringstream ss;
232  ss << "All negative(discarding) filters must appear after all "
233  "positive(accepting) filters!";
234  throw sycl::exception(sycl::make_error_code(errc::invalid),
235  ss.str());
236  }
237  }
238  Parse_ODS_Device(DeviceTarget, TargetStr);
239  Result.push_back(DeviceTarget);
240  }
241  } else if (Pair.size() > 2) {
242  std::stringstream ss;
243  ss << "Error parsing selector string \"" << Entry
244  << "\" Too many colons (:)";
245  throw sycl::exception(sycl::make_error_code(errc::invalid), ss.str());
246  }
247  }
248 
249  // This if statement handles the special case when the filter list
250  // contains at least one negative filter but no positive filters.
251  // This means that no devices will be available at all and so its as if
252  // the filter list was empty because the negative filters do not have any
253  // any effect. Hoewever, it is desirable to be able to set the
254  // ONEAPI_DEVICE_SELECTOR=!*:gpu to consider all devices except gpu
255  // devices so that we must implicitly add an acceptall target to the
256  // list of targets to make this work. So the result will be as if
257  // the filter string had the *:* string in it.
258  if (!Result.empty() && negative_filters == Result.size()) {
259  ods_target acceptAll{backend::all};
260  acceptAll.DeviceType = info::device_type::all;
261  Result.push_back(acceptAll);
262  }
263  return Result;
264 }

267  Out << Target.Backend;

272 #endif
273  >();

275  DeviceTypeMap.begin(), DeviceTypeMap.end(),
276  [&](auto Pair) { return (Pair.second == Target.DeviceType); });
277  if (Match != DeviceTypeMap.end()) {
278  Out << ":" << Match->first;
279  } else {
280  Out << ":???";
281  }
282  }

284  Out << ":*";

286  Out << ":" << Target.DeviceNum.value();
287  if (Target.HasSubDeviceWildCard)
288  Out << ".*";
289  if (Target.SubDeviceNum)
290  Out << "." << Target.SubDeviceNum.value();
291 
292  return Out;

315 device_filter::device_filter(const std::string &FilterString) {
316  std::vector<std::string_view> Tokens = tokenize(FilterString, ":");
317  size_t TripleValueID = 0;
318 
319  auto FindElement = [&](auto Element) {
320  return std::string::npos != Tokens[TripleValueID].find(Element.first);
321  };
322 
323  // Handle the optional 1st field of the filter, backend
324  // Check if the first entry matches with a known backend type
325  auto It = std::find_if(std::begin(getSyclBeMap()), std::end(getSyclBeMap()),
326  FindElement);
327  // If no match is found, set the backend type backend::all
328  // which actually means 'any backend' will be a match.
329  if (It == getSyclBeMap().end())
330  Backend = backend::all;
331  else {
332  Backend = It->second;
333  TripleValueID++;
334 
335  if (Backend == backend::host)
336  std::cerr << "WARNING: The 'host' backend type is no longer supported in "
337  "device filter."
338  << std::endl;
339  }
340 
341  // Handle the optional 2nd field of the filter - device type.
342  // Check if the 2nd entry matches with any known device type.
343  if (TripleValueID >= Tokens.size()) {
344  DeviceType = info::device_type::all;
345  } else {
346  auto Iter = std::find_if(
347  std::begin(getSyclDeviceTypeMap<true /*Enable 'acc'*/>()),
348  std::end(getSyclDeviceTypeMap<true /*Enable 'acc'*/>()), FindElement);
349  // If no match is found, set device_type 'all',
350  // which actually means 'any device_type' will be a match.
351  if (Iter == getSyclDeviceTypeMap<true /*Enable 'acc'*/>().end())
352  DeviceType = info::device_type::all;
353  else {
354  DeviceType = Iter->second;
355  TripleValueID++;
356 
357  if (DeviceType == info::device_type::host)
358  std::cerr << "WARNING: The 'host' device type is no longer supported "
359  "in device filter."
360  << std::endl;
361  }
362  }
363 
364  // Handle the optional 3rd field of the filter, device number
365  // Try to convert the remaining string to an integer.
366  // If succeessful, the converted integer is the desired device num.
367  if (TripleValueID < Tokens.size()) {
368  try {
369  DeviceNum = std::stoi(Tokens[TripleValueID].data());
370  } catch (...) {
371  std::string Message =
372  std::string("Invalid device filter: ") + FilterString +
373  "\nPossible backend values are "
374  "{opencl,level_zero,cuda,hip,*}.\n"
375  "Possible device types are {cpu,gpu,acc,*}.\n"
376  "Device number should be an non-negative integer.\n";
377  throw sycl::invalid_parameter_error(Message, PI_ERROR_INVALID_VALUE);
378  }
379  }
380 }

382 device_filter_list::device_filter_list(const std::string &FilterStr) {
383  // First, change the string in all lowercase.
384  // This means we allow the user to use both uppercase and lowercase strings.
385  std::string FilterString = FilterStr;
386  std::transform(FilterString.begin(), FilterString.end(), FilterString.begin(),
387  ::tolower);
388  // SYCL_DEVICE_FILTER can set multiple filters separated by commas.
389  // convert each filter triple string into an istance of device_filter class.
390  size_t Pos = 0;
391  while (Pos < FilterString.size()) {
392  size_t CommaPos = FilterString.find(",", Pos);
393  if (CommaPos == std::string::npos) {
394  CommaPos = FilterString.size();
395  }
396  std::string SubString = FilterString.substr(Pos, CommaPos - Pos);
397  FilterList.push_back(device_filter(SubString));
398  Pos = CommaPos + 1;
399  }
400 }
