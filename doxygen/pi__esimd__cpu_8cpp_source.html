70 // Lambda-call interface definition.
71 // 'extern "C"' is required as CM supports only C-style function calls
72 // while kernel is given as lambda function
73 //
74 #define LAMBDA_WRAPPER_TMPL(ARGTYPE, TAG, DIMS) \
75  typedef std::function<void(const ARGTYPE &)> LambdaFunction_##TAG; \
76  \
77  extern "C" struct LambdaWrapper_##TAG { \
78  LambdaFunction_##TAG Func; \
79  const sycl::range<DIMS> &LocalSize; \
80  const sycl::range<DIMS> &GlobalSize; \
81  const sycl::id<DIMS> &GlobalOffset; \
82  LambdaWrapper_##TAG(LambdaFunction_##TAG ArgFunc, \
83  const sycl::range<DIMS> &ArgLocalSize, \
84  const sycl::range<DIMS> &ArgGlobalSize, \
85  const sycl::id<DIMS> &ArgGlobalOffset) \
86  : Func(ArgFunc), LocalSize(ArgLocalSize), GlobalSize(ArgGlobalSize), \
87  GlobalOffset(ArgGlobalOffset) {} \
88  }; \
89  \
90  template <typename LambdaTy> \
91  auto makeWrapper_##TAG(LambdaTy F, const sycl::range<DIMS> &LocalSize, \
92  const sycl::range<DIMS> &GlobalSize, \
93  const sycl::id<DIMS> &GlobalOffset) { \
94  std::unique_ptr<LambdaWrapper_##TAG> Wrapper = \
95  std::make_unique<LambdaWrapper_##TAG>(LambdaWrapper_##TAG( \
96  LambdaFunction_##TAG(F), LocalSize, GlobalSize, GlobalOffset)); \
97  return Wrapper; \
98  }
