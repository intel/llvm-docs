23 kernel_impl::kernel_impl(RT::PiKernel Kernel, ContextImplPtr Context)
24  : kernel_impl(Kernel, Context,
25  std::make_shared<program_impl>(Context, Kernel),
26  /*IsCreatedFromSource*/ true) {
27  // This constructor is only called in the interoperability kernel constructor.
28  // Let the runtime caller handle native kernel retaining in other cases if
29  // it's needed.
30  getPlugin().call<PiApiKind::piKernelRetain>(MKernel);
31  // Enable USM indirect access for interoperability kernels.
32  // Some PI Plugins (like OpenCL) require this call to enable USM
33  // For others, PI will turn this into a NOP.
34  getPlugin().call<PiApiKind::piKernelSetExecInfo>(
35  MKernel, PI_USM_INDIRECT_ACCESS, sizeof(pi_bool), &PI_TRUE);
36 }

38 kernel_impl::kernel_impl(RT::PiKernel Kernel, ContextImplPtr ContextImpl,
39  ProgramImplPtr ProgramImpl,
40  bool IsCreatedFromSource)
41  : MKernel(Kernel), MContext(ContextImpl),
42  MProgramImpl(std::move(ProgramImpl)),
43  MCreatedFromSource(IsCreatedFromSource) {
44 
45  RT::PiContext Context = nullptr;
46  // Using the plugin from the passed ContextImpl
47  getPlugin().call<PiApiKind::piKernelGetInfo>(
48  MKernel, PI_KERNEL_INFO_CONTEXT, sizeof(Context), &Context, nullptr);
49  if (ContextImpl->getHandleRef() != Context)
50  throw cl::sycl::invalid_parameter_error(
51  "Input context must be the same as the context of cl_kernel",
52  PI_INVALID_CONTEXT);
53 }
