32 kernel_impl::kernel_impl(RT::PiKernel Kernel, ContextImplPtr ContextImpl,
33  ProgramImplPtr ProgramImpl,
34  bool IsCreatedFromSource)
35  : MKernel(Kernel), MContext(ContextImpl),
36  MProgramImpl(std::move(ProgramImpl)),
37  MCreatedFromSource(IsCreatedFromSource) {
38 
39  RT::PiContext Context = nullptr;
40  // Using the plugin from the passed ContextImpl
41  getPlugin().call<PiApiKind::piKernelGetInfo>(
42  MKernel, PI_KERNEL_INFO_CONTEXT, sizeof(Context), &Context, nullptr);
43  if (ContextImpl->getHandleRef() != Context)
44  throw cl::sycl::invalid_parameter_error(
45  "Input context must be the same as the context of cl_kernel",
46  PI_INVALID_CONTEXT);

151  // context Context;
152  // program Program(Context);

154  // kernel FirstKernel= Program.get_kernel<class A>();

159  return MCreatedFromSource;
