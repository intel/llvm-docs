380 // The function extends or truncates number of dimensions of objects of id
381 // or ranges classes. When extending the new values are filled with
382 // DefaultValue, truncation just removes extra values.
383 template <int NewDim, int DefaultValue, template <int> class T, int OldDim>
384 static T<NewDim> convertToArrayOfN(T<OldDim> OldObj) {
385  T<NewDim> NewObj = detail::InitializedVal<NewDim, T>::template get<0>();
386  const int CopyDims = NewDim > OldDim ? OldDim : NewDim;
387  for (int I = 0; I < CopyDims; ++I)
388  NewObj[I] = OldObj[I];
389  for (int I = CopyDims; I < NewDim; ++I)
390  NewObj[I] = DefaultValue;
391  return NewObj;
392 }

416 struct ArrayCreator<DataT, FlattenF, ArgT, ArgTN...> {
417  static constexpr auto Create(const ArgT &Arg, const ArgTN &...Args) {
