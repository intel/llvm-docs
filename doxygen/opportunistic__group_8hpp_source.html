144 inline opportunistic_group get_opportunistic_group() {
145 #ifdef __SYCL_DEVICE_ONLY__
146 #if defined(__SPIR__) || defined(__SPIRV__)
147  // TODO: It may be wiser to call the intrinsic than rely on this_group()
148  sycl::sub_group sg = sycl::ext::oneapi::experimental::this_sub_group();
149  sub_group_mask mask = sycl::ext::oneapi::group_ballot(sg, true);
150  return opportunistic_group(mask);
151 #elif defined(__NVPTX__)
152  uint32_t active_mask;
153  asm volatile("activemask.b32 %0;" : "=r"(active_mask));
154  sub_group_mask mask =
155  sycl::detail::Builder::createSubGroupMask<ext::oneapi::sub_group_mask>(
156  active_mask, 32);
157  return opportunistic_group(mask);
158 #endif
159 #else
160  throw runtime_error("Non-uniform groups are not supported on host device.",
161  PI_ERROR_INVALID_DEVICE);
162 #endif
163 }
