184 // This is the macro that plugins should all use to define their version.
185 // _PI_PLUGIN_VERSION_STRING will be printed when environment variable
186 // SYCL_PI_TRACE is set to 1. PluginVersion should be defined for each plugin
187 // in plugins/*/pi_*.hpp. PluginVersion should be incremented with each change
188 // to the plugin.
189 #define _PI_PLUGIN_VERSION_STRING(PluginVersion) \
190  _PI_TRIPLE_CONCAT(_PI_H_VERSION_MAJOR, _PI_H_VERSION_MINOR, PluginVersion)
191 
192 #define _PI_H_VERSION_STRING \
193  _PI_CONCAT(_PI_H_VERSION_MAJOR, _PI_H_VERSION_MINOR)

195 // This will be used to check the major versions of plugins versus the major
196 // versions of PI.
197 #define _PI_STRING_SUBSTITUTE(X) _PI_STRING_HELPER(X)
198 #define _PI_PLUGIN_VERSION_CHECK(PI_API_VERSION, PI_PLUGIN_VERSION) \
199  if (strncmp(PI_API_VERSION, PI_PLUGIN_VERSION, \
200  sizeof(_PI_STRING_SUBSTITUTE(_PI_H_VERSION_MAJOR))) < 0) { \
201  return PI_ERROR_INVALID_OPERATION; \
