207 // This is the macro that plugins should all use to define their version.
208 // _PI_PLUGIN_VERSION_STRING will be printed when environment variable
209 // SYCL_PI_TRACE is set to 1. PluginVersion should be defined for each plugin
210 // in plugins/*/pi_*.hpp. PluginVersion should be incremented with each change
211 // to the plugin.

213  _PI_TRIPLE_CONCAT(_PI_H_VERSION_MAJOR, _PI_H_VERSION_MINOR, PluginVersion)

218 // This will be used to check the major versions of plugins versus the major
219 // versions of PI.
220 #define _PI_STRING_SUBSTITUTE(X) _PI_STRING_HELPER(X)
221 #define _PI_PLUGIN_VERSION_CHECK(PI_API_VERSION, PI_PLUGIN_VERSION) \
222  if (strncmp(PI_API_VERSION, PI_PLUGIN_VERSION, \
