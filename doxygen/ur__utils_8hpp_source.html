22 // RAII object for keeping ownership of a UR event.
23 struct OwnedUrEvent {
24  OwnedUrEvent(const PluginPtr &Plugin)
25  : MEvent{std::nullopt}, MPlugin{Plugin} {}
26  OwnedUrEvent(ur_event_handle_t Event, const PluginPtr &Plugin,
27  bool TakeOwnership = false)
28  : MEvent(Event), MPlugin(Plugin) {
29  // If it is not instructed to take ownership, retain the event to share
30  // ownership of it.
31  if (!TakeOwnership)
32  MPlugin->call(urEventRetain, *MEvent);
33  }
34  ~OwnedUrEvent() {
35  try {
36  // Release the event if the ownership was not transferred.
37  if (MEvent.has_value())
38  MPlugin->call(urEventRelease, *MEvent);
39 
40  } catch (std::exception &e) {
41  __SYCL_REPORT_EXCEPTION_TO_STREAM("exception in ~OwnedUrEvent", e);
42  }
43  }
44 
45  OwnedUrEvent(OwnedUrEvent &&Other)
46  : MEvent(Other.MEvent), MPlugin(Other.MPlugin) {
47  Other.MEvent = std::nullopt;
48  }
49 
50  // Copy constructor explicitly deleted for simplicity as it is not currently
51  // used. Implement if needed.
52  OwnedUrEvent(const OwnedUrEvent &Other) = delete;
53 
54  operator bool() { return MEvent.has_value(); }
55 
56  ur_event_handle_t GetEvent() { return *MEvent; }
57 
58  // Transfers the ownership of the event to the caller. The destructor will
59  // no longer release the event.
60  ur_event_handle_t TransferOwnership() {
61  ur_event_handle_t Event = *MEvent;
62  MEvent = std::nullopt;
63  return Event;
64  }
65 
66 private:
67  std::optional<ur_event_handle_t> MEvent;
68  const PluginPtr &MPlugin;
69 };
