34 device_impl::device_impl(pi_native_handle InteropDeviceHandle,
35  RT::PiDevice Device, PlatformImplPtr Platform,
36  const plugin &Plugin)
37  : MDevice(Device), MIsHostDevice(false) {
38 
39  bool InteroperabilityConstructor = false;
40  if (Device == nullptr) {
41  assert(InteropDeviceHandle);
42  // Get PI device from the raw device handle.
43  Plugin.call<PiApiKind::piextDeviceCreateWithNativeHandle>(
44  InteropDeviceHandle, &MDevice);
45  InteroperabilityConstructor = true;
46  }
47 
48  // TODO catch an exception and put it to list of asynchronous exceptions
49  Plugin.call<PiApiKind::piDeviceGetInfo>(
50  MDevice, PI_DEVICE_INFO_TYPE, sizeof(RT::PiDeviceType), &MType, nullptr);
51 
52  RT::PiDevice parent = nullptr;
53  // TODO catch an exception and put it to list of asynchronous exceptions
54  Plugin.call<PiApiKind::piDeviceGetInfo>(
55  MDevice, PI_DEVICE_INFO_PARENT_DEVICE, sizeof(RT::PiDevice), &parent, nullptr);
56 
57  MIsRootDevice = (nullptr == parent);
58  if (!MIsRootDevice && !InteroperabilityConstructor) {
59  // TODO catch an exception and put it to list of asynchronous exceptions
60  // Interoperability Constructor already calls DeviceRetain in
61  // piextDeviceFromNative.
62  Plugin.call<PiApiKind::piDeviceRetain>(MDevice);
63  }
64 
65  // set MPlatform
66  if (!Platform) {
67  RT::PiPlatform plt = nullptr; // TODO catch an exception and put it to list
68  // of asynchronous exceptions
69  Plugin.call<PiApiKind::piDeviceGetInfo>(MDevice, PI_DEVICE_INFO_PLATFORM,
70  sizeof(plt), &plt, nullptr);
71  Platform = std::make_shared<platform_impl>(plt, Plugin);
72  }
73  MPlatform = Platform;
74 }

126 vector_class<device>
127 device_impl::create_sub_devices(const cl_device_partition_property *Properties,
128  size_t SubDevicesCount) const {
129 
130  vector_class<RT::PiDevice> SubDevices(SubDevicesCount);
131  pi_uint32 ReturnedSubDevices = 0;
132  const detail::plugin &Plugin = getPlugin();
133  Plugin.call<PiApiKind::piDevicePartition>(MDevice, Properties,
134  SubDevicesCount, SubDevices.data(),
135  &ReturnedSubDevices);
136  // TODO: check that returned number of sub-devices matches what was
137  // requested, otherwise this walk below is wrong.
138  //
139  // TODO: Need to describe the subdevice model. Some sub_device management
140  // may be necessary. What happens if create_sub_devices is called multiple
141  // times with the same arguments?
142  //
143  vector_class<device> res;
144  std::for_each(SubDevices.begin(), SubDevices.end(),
145  [&res, this](const RT::PiDevice &a_pi_device) {
146  device sycl_device = detail::createSyclObjFromImpl<device>(
147  std::make_shared<device_impl>(a_pi_device, MPlatform));
148  res.push_back(sycl_device);
149  });
150  return res;
151 }

153 vector_class<device>
154 device_impl::create_sub_devices(size_t ComputeUnits) const {
155 
156  if (MIsHostDevice)
157  // TODO: implement host device partitioning
158  throw runtime_error(
159  "Partitioning to subdevices of the host device is not implemented yet",
160  PI_INVALID_DEVICE);
161 
162  if (!is_partition_supported(info::partition_property::partition_equally)) {
163  throw cl::sycl::feature_not_supported();
164  }
165  size_t SubDevicesCount =
166  get_info<info::device::max_compute_units>() / ComputeUnits;
167  const cl_device_partition_property Properties[3] = {
168  CL_DEVICE_PARTITION_EQUALLY, (cl_device_partition_property)ComputeUnits,
169  0};
170  return create_sub_devices(Properties, SubDevicesCount);
171 }

173 vector_class<device>
174 device_impl::create_sub_devices(const vector_class<size_t> &Counts) const {
175 
176  if (MIsHostDevice)
177  // TODO: implement host device partitioning
178  throw runtime_error(
179  "Partitioning to subdevices of the host device is not implemented yet",
180  PI_INVALID_DEVICE);
181 
182  if (!is_partition_supported(
183  info::partition_property::partition_by_counts)) {
184  throw cl::sycl::feature_not_supported();
185  }
186  static const cl_device_partition_property P[] = {
187  CL_DEVICE_PARTITION_BY_COUNTS, CL_DEVICE_PARTITION_BY_COUNTS_LIST_END,
188  0};
189  vector_class<cl_device_partition_property> Properties(P, P + 3);
190  Properties.insert(Properties.begin() + 1, Counts.begin(), Counts.end());
191  return create_sub_devices(Properties.data(), Counts.size());
192 }

194 vector_class<device> device_impl::create_sub_devices(
195  info::partition_affinity_domain AffinityDomain) const {
196 
197  if (MIsHostDevice)
198  // TODO: implement host device partitioning
199  throw runtime_error(
200  "Partitioning to subdevices of the host device is not implemented yet",
201  PI_INVALID_DEVICE);
202 
203  if (!is_partition_supported(
204  info::partition_property::partition_by_affinity_domain) ||
205  !is_affinity_supported(AffinityDomain)) {
206  throw cl::sycl::feature_not_supported();
207  }
208  const cl_device_partition_property Properties[3] = {
209  CL_DEVICE_PARTITION_BY_AFFINITY_DOMAIN,
210  (cl_device_partition_property)AffinityDomain, 0};
211  size_t SubDevicesCount =
212  get_info<info::device::partition_max_sub_devices>();
213  return create_sub_devices(Properties, SubDevicesCount);
214 }
