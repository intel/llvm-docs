34 device_impl::device_impl(pi_native_handle InteropDeviceHandle,
35  RT::PiDevice Device, PlatformImplPtr Platform,
36  const plugin &Plugin)
37  : MDevice(Device), MIsHostDevice(false) {
38 
39  bool InteroperabilityConstructor = false;
40  if (Device == nullptr) {
41  assert(InteropDeviceHandle);
42  // Get PI device from the raw device handle.
43  // NOTE: this is for OpenCL interop only (and should go away).
44  // With SYCL-2020 BE generalization "make" functions are used instead.
45  Plugin.call<PiApiKind::piextDeviceCreateWithNativeHandle>(
46  InteropDeviceHandle, nullptr, &MDevice);
47  InteroperabilityConstructor = true;
48  }
49 
50  // TODO catch an exception and put it to list of asynchronous exceptions
51  Plugin.call<PiApiKind::piDeviceGetInfo>(
52  MDevice, PI_DEVICE_INFO_TYPE, sizeof(RT::PiDeviceType), &MType, nullptr);
53 
54  RT::PiDevice parent = nullptr;
55  // TODO catch an exception and put it to list of asynchronous exceptions
56  Plugin.call<PiApiKind::piDeviceGetInfo>(MDevice, PI_DEVICE_INFO_PARENT_DEVICE,
57  sizeof(RT::PiDevice), &parent,
58  nullptr);
59 
60  MIsRootDevice = (nullptr == parent);
61  if (!InteroperabilityConstructor) {
62  // TODO catch an exception and put it to list of asynchronous exceptions
63  // Interoperability Constructor already calls DeviceRetain in
64  // piextDeviceFromNative.
65  Plugin.call<PiApiKind::piDeviceRetain>(MDevice);
66  }
67 
68  // set MPlatform
69  if (!Platform) {
70  Platform = platform_impl::getPlatformFromPiDevice(MDevice, Plugin);
71  }
72  MPlatform = Platform;
73 }

123 vector_class<device>
124 device_impl::create_sub_devices(const cl_device_partition_property *Properties,
125  size_t SubDevicesCount) const {
126 
127  vector_class<RT::PiDevice> SubDevices(SubDevicesCount);
128  pi_uint32 ReturnedSubDevices = 0;
129  const detail::plugin &Plugin = getPlugin();
130  Plugin.call<PiApiKind::piDevicePartition>(MDevice, Properties,
131  SubDevicesCount, SubDevices.data(),
132  &ReturnedSubDevices);
133  // TODO: check that returned number of sub-devices matches what was
134  // requested, otherwise this walk below is wrong.
135  //
136  // TODO: Need to describe the subdevice model. Some sub_device management
137  // may be necessary. What happens if create_sub_devices is called multiple
138  // times with the same arguments?
139  //
140  vector_class<device> res;
141  std::for_each(SubDevices.begin(), SubDevices.end(),
142  [&res, this](const RT::PiDevice &a_pi_device) {
143  device sycl_device = detail::createSyclObjFromImpl<device>(
144  MPlatform->getOrMakeDeviceImpl(a_pi_device, MPlatform));
145  res.push_back(sycl_device);
146  });
147  return res;
148 }

150 vector_class<device>
151 device_impl::create_sub_devices(size_t ComputeUnits) const {
152 
153  if (MIsHostDevice)
154  // TODO: implement host device partitioning
155  throw runtime_error(
156  "Partitioning to subdevices of the host device is not implemented yet",
157  PI_INVALID_DEVICE);
158 
159  if (!is_partition_supported(info::partition_property::partition_equally)) {
160  throw cl::sycl::feature_not_supported();
161  }
162  size_t SubDevicesCount =
163  get_info<info::device::max_compute_units>() / ComputeUnits;
164  const cl_device_partition_property Properties[3] = {
165  CL_DEVICE_PARTITION_EQUALLY, (cl_device_partition_property)ComputeUnits,
166  0};
167  return create_sub_devices(Properties, SubDevicesCount);
168 }

170 vector_class<device>
171 device_impl::create_sub_devices(const vector_class<size_t> &Counts) const {
172 
173  if (MIsHostDevice)
174  // TODO: implement host device partitioning
175  throw runtime_error(
176  "Partitioning to subdevices of the host device is not implemented yet",
177  PI_INVALID_DEVICE);
178 
179  if (!is_partition_supported(info::partition_property::partition_by_counts)) {
180  throw cl::sycl::feature_not_supported();
181  }
182  static const cl_device_partition_property P[] = {
183  CL_DEVICE_PARTITION_BY_COUNTS, CL_DEVICE_PARTITION_BY_COUNTS_LIST_END, 0};
184  vector_class<cl_device_partition_property> Properties(P, P + 3);
185  Properties.insert(Properties.begin() + 1, Counts.begin(), Counts.end());
186  return create_sub_devices(Properties.data(), Counts.size());
187 }

189 vector_class<device> device_impl::create_sub_devices(
190  info::partition_affinity_domain AffinityDomain) const {
191 
192  if (MIsHostDevice)
193  // TODO: implement host device partitioning
194  throw runtime_error(
195  "Partitioning to subdevices of the host device is not implemented yet",
196  PI_INVALID_DEVICE);
197 
198  if (!is_partition_supported(
199  info::partition_property::partition_by_affinity_domain) ||
200  !is_affinity_supported(AffinityDomain)) {
201  throw cl::sycl::feature_not_supported();
202  }
203  const cl_device_partition_property Properties[3] = {
204  CL_DEVICE_PARTITION_BY_AFFINITY_DOMAIN,
205  (cl_device_partition_property)AffinityDomain, 0};
206  size_t SubDevicesCount = get_info<info::device::partition_max_sub_devices>();
207  return create_sub_devices(Properties, SubDevicesCount);
208 }

219 bool device_impl::has(aspect Aspect) const {
220  size_t return_size = 0;
221  pi_device_type device_type;
222 
223  switch (Aspect) {
224  case aspect::host:
225  return is_host();
226  case aspect::cpu:
227  return is_cpu();
228  case aspect::gpu:
229  return is_gpu();
230  case aspect::accelerator:
231  return is_accelerator();
232  case aspect::custom:
233  return false;
234  case aspect::fp16:
235  return has_extension("cl_khr_fp16");
236  case aspect::fp64:
237  return has_extension("cl_khr_fp64");
238  case aspect::int64_base_atomics:
239  return has_extension("cl_khr_int64_base_atomics");
240  case aspect::int64_extended_atomics:
241  return has_extension("cl_khr_int64_extended_atomics");
242  case aspect::image:
243  return get_info<info::device::image_support>();
244  case aspect::online_compiler:
245  return get_info<info::device::is_compiler_available>();
246  case aspect::online_linker:
247  return get_info<info::device::is_linker_available>();
248  case aspect::queue_profiling:
249  return get_info<info::device::queue_profiling>();
250  case aspect::usm_device_allocations:
251  return get_info<info::device::usm_device_allocations>();
252  case aspect::usm_host_allocations:
253  return get_info<info::device::usm_host_allocations>();
254  case aspect::usm_shared_allocations:
255  return get_info<info::device::usm_shared_allocations>();
256  case aspect::usm_restricted_shared_allocations:
257  return get_info<info::device::usm_restricted_shared_allocations>();
258  case aspect::usm_system_allocator:
259  return get_info<info::device::usm_system_allocator>();
260  case aspect::ext_intel_pci_address:
261  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
262  MDevice, PI_DEVICE_INFO_PCI_ADDRESS, sizeof(pi_device_type),
263  &device_type, &return_size) == PI_SUCCESS;
264  case aspect::ext_intel_gpu_eu_count:
265  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
266  MDevice, PI_DEVICE_INFO_GPU_EU_COUNT, sizeof(pi_device_type),
267  &device_type, &return_size) == PI_SUCCESS;
268  case aspect::ext_intel_gpu_eu_simd_width:
269  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
270  MDevice, PI_DEVICE_INFO_GPU_EU_SIMD_WIDTH,
271  sizeof(pi_device_type), &device_type,
272  &return_size) == PI_SUCCESS;
273  case aspect::ext_intel_gpu_slices:
274  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
275  MDevice, PI_DEVICE_INFO_GPU_SLICES, sizeof(pi_device_type),
276  &device_type, &return_size) == PI_SUCCESS;
277  case aspect::ext_intel_gpu_subslices_per_slice:
278  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
279  MDevice, PI_DEVICE_INFO_GPU_SUBSLICES_PER_SLICE,
280  sizeof(pi_device_type), &device_type,
281  &return_size) == PI_SUCCESS;
282  case aspect::ext_intel_gpu_eu_count_per_subslice:
283  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
284  MDevice, PI_DEVICE_INFO_GPU_EU_COUNT_PER_SUBSLICE,
285  sizeof(pi_device_type), &device_type,
286  &return_size) == PI_SUCCESS;
287  case aspect::ext_intel_max_mem_bandwidth:
288  // currently not supported
289  return false;
290 
291  default:
292  throw runtime_error("This device aspect has not been implemented yet.",
293  PI_INVALID_DEVICE);
294  }
