34 device_impl::device_impl(pi_native_handle InteropDeviceHandle,
35  RT::PiDevice Device, PlatformImplPtr Platform,
36  const plugin &Plugin)
37  : MDevice(Device), MIsHostDevice(false) {
38 
39  bool InteroperabilityConstructor = false;
40  if (Device == nullptr) {
41  assert(InteropDeviceHandle);
42  // Get PI device from the raw device handle.
43  // NOTE: this is for OpenCL interop only (and should go away).
44  // With SYCL-2020 BE generalization "make" functions are used instead.
45  Plugin.call<PiApiKind::piextDeviceCreateWithNativeHandle>(
46  InteropDeviceHandle, nullptr, &MDevice);
47  InteroperabilityConstructor = true;
48  }
49 
50  // TODO catch an exception and put it to list of asynchronous exceptions
51  Plugin.call<PiApiKind::piDeviceGetInfo>(
52  MDevice, PI_DEVICE_INFO_TYPE, sizeof(RT::PiDeviceType), &MType, nullptr);
53 
54  RT::PiDevice parent = nullptr;
55  // TODO catch an exception and put it to list of asynchronous exceptions
56  Plugin.call<PiApiKind::piDeviceGetInfo>(MDevice, PI_DEVICE_INFO_PARENT_DEVICE,
57  sizeof(RT::PiDevice), &parent,
58  nullptr);
59 
60  MIsRootDevice = (nullptr == parent);
61  if (!InteroperabilityConstructor) {
62  // TODO catch an exception and put it to list of asynchronous exceptions
63  // Interoperability Constructor already calls DeviceRetain in
64  // piextDeviceFromNative.
65  Plugin.call<PiApiKind::piDeviceRetain>(MDevice);
66  }
67 
68  // set MPlatform
69  if (!Platform) {
70  RT::PiPlatform plt = nullptr; // TODO catch an exception and put it to list
71  // of asynchronous exceptions
72  Plugin.call<PiApiKind::piDeviceGetInfo>(MDevice, PI_DEVICE_INFO_PLATFORM,
73  sizeof(plt), &plt, nullptr);
74  Platform = std::make_shared<platform_impl>(plt, Plugin);
75  }
76  MPlatform = Platform;
77 }

128 vector_class<device>
129 device_impl::create_sub_devices(const cl_device_partition_property *Properties,
130  size_t SubDevicesCount) const {
131 
132  vector_class<RT::PiDevice> SubDevices(SubDevicesCount);
133  pi_uint32 ReturnedSubDevices = 0;
134  const detail::plugin &Plugin = getPlugin();
135  Plugin.call<PiApiKind::piDevicePartition>(MDevice, Properties,
136  SubDevicesCount, SubDevices.data(),
137  &ReturnedSubDevices);
138  // TODO: check that returned number of sub-devices matches what was
139  // requested, otherwise this walk below is wrong.
140  //
141  // TODO: Need to describe the subdevice model. Some sub_device management
142  // may be necessary. What happens if create_sub_devices is called multiple
143  // times with the same arguments?
144  //
145  vector_class<device> res;
146  std::for_each(SubDevices.begin(), SubDevices.end(),
147  [&res, this](const RT::PiDevice &a_pi_device) {
148  device sycl_device = detail::createSyclObjFromImpl<device>(
149  std::make_shared<device_impl>(a_pi_device, MPlatform));
150  res.push_back(sycl_device);
151  });
152  return res;
153 }

155 vector_class<device>
156 device_impl::create_sub_devices(size_t ComputeUnits) const {
157 
158  if (MIsHostDevice)
159  // TODO: implement host device partitioning
160  throw runtime_error(
161  "Partitioning to subdevices of the host device is not implemented yet",
162  PI_INVALID_DEVICE);
163 
164  if (!is_partition_supported(info::partition_property::partition_equally)) {
165  throw cl::sycl::feature_not_supported();
166  }
167  size_t SubDevicesCount =
168  get_info<info::device::max_compute_units>() / ComputeUnits;
169  const cl_device_partition_property Properties[3] = {
170  CL_DEVICE_PARTITION_EQUALLY, (cl_device_partition_property)ComputeUnits,
171  0};
172  return create_sub_devices(Properties, SubDevicesCount);
173 }

175 vector_class<device>
176 device_impl::create_sub_devices(const vector_class<size_t> &Counts) const {
177 
178  if (MIsHostDevice)
179  // TODO: implement host device partitioning
180  throw runtime_error(
181  "Partitioning to subdevices of the host device is not implemented yet",
182  PI_INVALID_DEVICE);
183 
184  if (!is_partition_supported(info::partition_property::partition_by_counts)) {
185  throw cl::sycl::feature_not_supported();
186  }
187  static const cl_device_partition_property P[] = {
188  CL_DEVICE_PARTITION_BY_COUNTS, CL_DEVICE_PARTITION_BY_COUNTS_LIST_END, 0};
189  vector_class<cl_device_partition_property> Properties(P, P + 3);
190  Properties.insert(Properties.begin() + 1, Counts.begin(), Counts.end());
191  return create_sub_devices(Properties.data(), Counts.size());
192 }

194 vector_class<device> device_impl::create_sub_devices(
195  info::partition_affinity_domain AffinityDomain) const {
196 
197  if (MIsHostDevice)
198  // TODO: implement host device partitioning
199  throw runtime_error(
200  "Partitioning to subdevices of the host device is not implemented yet",
201  PI_INVALID_DEVICE);
202 
203  if (!is_partition_supported(
204  info::partition_property::partition_by_affinity_domain) ||
205  !is_affinity_supported(AffinityDomain)) {
206  throw cl::sycl::feature_not_supported();
207  }
208  const cl_device_partition_property Properties[3] = {
209  CL_DEVICE_PARTITION_BY_AFFINITY_DOMAIN,
210  (cl_device_partition_property)AffinityDomain, 0};
211  size_t SubDevicesCount = get_info<info::device::partition_max_sub_devices>();
212  return create_sub_devices(Properties, SubDevicesCount);

222 bool device_impl::has(aspect Aspect) const {
223  switch (Aspect) {
224  case aspect::host:
225  return is_host();
226  case aspect::cpu:
227  return is_cpu();
228  case aspect::gpu:
229  return is_gpu();
230  case aspect::accelerator:
231  return is_accelerator();
232  case aspect::fp16:
233  return has_extension("cl_khr_fp16");
234  case aspect::fp64:
235  return has_extension("cl_khr_fp64");
236  case aspect::int64_base_atomics:
237  return has_extension("cl_khr_int64_base_atomics");
238  case aspect::int64_extended_atomics:
239  return has_extension("cl_khr_int64_extended_atomics");
240  case aspect::image:
241  return get_info<info::device::image_support>();
242  case aspect::online_compiler:
243  return get_info<info::device::is_compiler_available>();
244  case aspect::online_linker:
245  return get_info<info::device::is_linker_available>();
246  case aspect::queue_profiling:
247  return get_info<info::device::queue_profiling>();
248  default:
249  throw runtime_error("This device aspect has not been implemented yet.",
250  PI_INVALID_DEVICE);
251  }
