36 device_impl::device_impl(pi_native_handle InteropDeviceHandle,
37  RT::PiDevice Device, PlatformImplPtr Platform,
38  const plugin &Plugin)
39  : MDevice(Device), MIsHostDevice(false) {
40 
41  bool InteroperabilityConstructor = false;
42  if (Device == nullptr) {
43  assert(InteropDeviceHandle);
44  // Get PI device from the raw device handle.
45  // NOTE: this is for OpenCL interop only (and should go away).
46  // With SYCL-2020 BE generalization "make" functions are used instead.
47  Plugin.call<PiApiKind::piextDeviceCreateWithNativeHandle>(
48  InteropDeviceHandle, nullptr, &MDevice);
49  InteroperabilityConstructor = true;
50  }
51 
52  // TODO catch an exception and put it to list of asynchronous exceptions
53  Plugin.call<PiApiKind::piDeviceGetInfo>(
54  MDevice, PI_DEVICE_INFO_TYPE, sizeof(RT::PiDeviceType), &MType, nullptr);
55 
56  // TODO catch an exception and put it to list of asynchronous exceptions
57  Plugin.call<PiApiKind::piDeviceGetInfo>(MDevice, PI_DEVICE_INFO_PARENT_DEVICE,
58  sizeof(RT::PiDevice), &MRootDevice,
59  nullptr);
60 
61  if (!InteroperabilityConstructor) {
62  // TODO catch an exception and put it to list of asynchronous exceptions
63  // Interoperability Constructor already calls DeviceRetain in
64  // piextDeviceFromNative.
65  Plugin.call<PiApiKind::piDeviceRetain>(MDevice);
66  }
67 
68  // set MPlatform
69  if (!Platform) {
70  Platform = platform_impl::getPlatformFromPiDevice(MDevice, Plugin);
71  }
72  MPlatform = Platform;
73 
74  MIsAssertFailSupported =
75  has_extension(PI_DEVICE_INFO_EXTENSION_DEVICELIB_ASSERT);
76 }

126 std::vector<device>
127 device_impl::create_sub_devices(const cl_device_partition_property *Properties,
128  size_t SubDevicesCount) const {
129 
130  std::vector<RT::PiDevice> SubDevices(SubDevicesCount);
131  pi_uint32 ReturnedSubDevices = 0;
132  const detail::plugin &Plugin = getPlugin();
133  Plugin.call<sycl::errc::invalid, PiApiKind::piDevicePartition>(
134  MDevice, Properties, SubDevicesCount, SubDevices.data(),
135  &ReturnedSubDevices);
136  // TODO: check that returned number of sub-devices matches what was
137  // requested, otherwise this walk below is wrong.
138  //
139  // TODO: Need to describe the subdevice model. Some sub_device management
140  // may be necessary. What happens if create_sub_devices is called multiple
141  // times with the same arguments?
142  //
143  std::vector<device> res;
144  std::for_each(SubDevices.begin(), SubDevices.end(),
145  [&res, this](const RT::PiDevice &a_pi_device) {
146  device sycl_device = detail::createSyclObjFromImpl<device>(
147  MPlatform->getOrMakeDeviceImpl(a_pi_device, MPlatform));
148  res.push_back(sycl_device);
149  });
150  return res;
151 }

153 std::vector<device> device_impl::create_sub_devices(size_t ComputeUnits) const {
154 
155  if (MIsHostDevice)
156  // TODO: implement host device partitioning
157  throw runtime_error(
158  "Partitioning to subdevices of the host device is not implemented yet",
159  PI_INVALID_DEVICE);
160 
161  if (!is_partition_supported(info::partition_property::partition_equally)) {
162  throw cl::sycl::feature_not_supported();
163  }
164  size_t SubDevicesCount =
165  get_info<info::device::max_compute_units>() / ComputeUnits;
166  const cl_device_partition_property Properties[3] = {
167  CL_DEVICE_PARTITION_EQUALLY, (cl_device_partition_property)ComputeUnits,
168  0};
169  return create_sub_devices(Properties, SubDevicesCount);
170 }

172 std::vector<device>
173 device_impl::create_sub_devices(const std::vector<size_t> &Counts) const {
174 
175  if (MIsHostDevice)
176  // TODO: implement host device partitioning
177  throw runtime_error(
178  "Partitioning to subdevices of the host device is not implemented yet",
179  PI_INVALID_DEVICE);
180 
181  if (!is_partition_supported(info::partition_property::partition_by_counts)) {
182  throw cl::sycl::feature_not_supported();
183  }
184  static const cl_device_partition_property P[] = {
185  CL_DEVICE_PARTITION_BY_COUNTS, CL_DEVICE_PARTITION_BY_COUNTS_LIST_END, 0};
186  std::vector<cl_device_partition_property> Properties(P, P + 3);
187  Properties.insert(Properties.begin() + 1, Counts.begin(), Counts.end());
188  return create_sub_devices(Properties.data(), Counts.size());
189 }

191 std::vector<device> device_impl::create_sub_devices(
192  info::partition_affinity_domain AffinityDomain) const {
193 
194  if (MIsHostDevice)
195  // TODO: implement host device partitioning
196  throw runtime_error(
197  "Partitioning to subdevices of the host device is not implemented yet",
198  PI_INVALID_DEVICE);
199 
200  if (!is_partition_supported(
201  info::partition_property::partition_by_affinity_domain) ||
202  !is_affinity_supported(AffinityDomain)) {
203  throw cl::sycl::feature_not_supported();
204  }
205  const pi_device_partition_property Properties[3] = {
206  PI_DEVICE_PARTITION_BY_AFFINITY_DOMAIN,
207  (pi_device_partition_property)AffinityDomain, 0};
208  size_t SubDevicesCount = get_info<info::device::partition_max_sub_devices>();
209  return create_sub_devices(Properties, SubDevicesCount);
210 }

221 bool device_impl::has(aspect Aspect) const {
222  size_t return_size = 0;
223  pi_device_type device_type;
224 
225  switch (Aspect) {
226  case aspect::host:
227  return is_host();
228  case aspect::cpu:
229  return is_cpu();
230  case aspect::gpu:
231  return is_gpu();
232  case aspect::accelerator:
233  return is_accelerator();
234  case aspect::custom:
235  return false;
236  case aspect::fp16:
237  return has_extension("cl_khr_fp16");
238  case aspect::fp64:
239  return has_extension("cl_khr_fp64");
240  case aspect::int64_base_atomics:
241  return has_extension("cl_khr_int64_base_atomics");
242  case aspect::int64_extended_atomics:
243  return has_extension("cl_khr_int64_extended_atomics");
244  case aspect::atomic64:
245  return get_info<info::device::atomic64>();
246  case aspect::image:
247  return get_info<info::device::image_support>();
248  case aspect::online_compiler:
249  return get_info<info::device::is_compiler_available>();
250  case aspect::online_linker:
251  return get_info<info::device::is_linker_available>();
252  case aspect::queue_profiling:
253  return get_info<info::device::queue_profiling>();
254  case aspect::usm_device_allocations:
255  return get_info<info::device::usm_device_allocations>();
256  case aspect::usm_host_allocations:
257  return get_info<info::device::usm_host_allocations>();
258  case aspect::usm_atomic_host_allocations:
259  return is_host() ||
260  (get_device_info<
261  pi_usm_capabilities,
262  info::device::usm_host_allocations>::get(MDevice, getPlugin()) &
263  PI_USM_CONCURRENT_ATOMIC_ACCESS);
264  case aspect::usm_shared_allocations:
265  return get_info<info::device::usm_shared_allocations>();
266  case aspect::usm_atomic_shared_allocations:
267  return is_host() ||
268  (get_device_info<
269  pi_usm_capabilities,
270  info::device::usm_shared_allocations>::get(MDevice,
271  getPlugin()) &
272  PI_USM_CONCURRENT_ATOMIC_ACCESS);
273  case aspect::usm_restricted_shared_allocations:
274  return get_info<info::device::usm_restricted_shared_allocations>();
275  case aspect::usm_system_allocations:
276  return get_info<info::device::usm_system_allocations>();
277  case aspect::ext_intel_pci_address:
278  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
279  MDevice, PI_DEVICE_INFO_PCI_ADDRESS, sizeof(pi_device_type),
280  &device_type, &return_size) == PI_SUCCESS;
281  case aspect::ext_intel_gpu_eu_count:
282  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
283  MDevice, PI_DEVICE_INFO_GPU_EU_COUNT, sizeof(pi_device_type),
284  &device_type, &return_size) == PI_SUCCESS;
285  case aspect::ext_intel_gpu_eu_simd_width:
286  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
287  MDevice, PI_DEVICE_INFO_GPU_EU_SIMD_WIDTH,
288  sizeof(pi_device_type), &device_type,
289  &return_size) == PI_SUCCESS;
290  case aspect::ext_intel_gpu_slices:
291  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
292  MDevice, PI_DEVICE_INFO_GPU_SLICES, sizeof(pi_device_type),
293  &device_type, &return_size) == PI_SUCCESS;
294  case aspect::ext_intel_gpu_subslices_per_slice:
295  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
296  MDevice, PI_DEVICE_INFO_GPU_SUBSLICES_PER_SLICE,
297  sizeof(pi_device_type), &device_type,
298  &return_size) == PI_SUCCESS;
299  case aspect::ext_intel_gpu_eu_count_per_subslice:
300  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
301  MDevice, PI_DEVICE_INFO_GPU_EU_COUNT_PER_SUBSLICE,
302  sizeof(pi_device_type), &device_type,
303  &return_size) == PI_SUCCESS;
304  case aspect::ext_intel_gpu_hw_threads_per_eu:
305  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
306  MDevice, PI_DEVICE_INFO_GPU_HW_THREADS_PER_EU,
307  sizeof(pi_device_type), &device_type,
308  &return_size) == PI_SUCCESS;
309  case aspect::ext_intel_device_info_uuid: {
310  auto Result = getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
311  MDevice, PI_DEVICE_INFO_UUID, 0, nullptr, &return_size);
312  if (Result != PI_SUCCESS) {
313  return false;
314  }
315 
316  assert(return_size <= 16);
317  unsigned char UUID[16];
318 
319  return getPlugin().call_nocheck<detail::PiApiKind::piDeviceGetInfo>(
320  MDevice, PI_DEVICE_INFO_UUID, 16 * sizeof(unsigned char), UUID,
321  nullptr) == PI_SUCCESS;
322  }
323  case aspect::ext_intel_max_mem_bandwidth:
324  // currently not supported
325  return false;
326  case aspect::ext_oneapi_srgb:
327  return get_info<info::device::ext_oneapi_srgb>();
328  case aspect::ext_oneapi_native_assert:
329  return isAssertFailSupported();
330 
331  default:
332  throw runtime_error("This device aspect has not been implemented yet.",
333  PI_INVALID_DEVICE);
334  }
335 }
