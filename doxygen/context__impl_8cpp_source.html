33 context_impl::context_impl(const vector_class<cl::sycl::device> Devices,
34  async_handler AsyncHandler, bool UseCUDAPrimaryContext)
35  : MAsyncHandler(AsyncHandler), MDevices(Devices), MContext(nullptr),
36  MPlatform(), MPluginInterop(true), MHostContext(false),
37  MUseCUDAPrimaryContext(UseCUDAPrimaryContext) {
38  MPlatform = detail::getSyclObjImpl(MDevices[0].get_platform());
39  vector_class<RT::PiDevice> DeviceIds;
40  for (const auto &D : MDevices) {
41  DeviceIds.push_back(getSyclObjImpl(D)->getHandleRef());
42  }
43 
44  if (MPlatform->is_cuda()) {
45 #if USE_PI_CUDA
46  const pi_context_properties props[] = {PI_CONTEXT_PROPERTIES_CUDA_PRIMARY,
47  UseCUDAPrimaryContext, 0};
48 
49  getPlugin().call<PiApiKind::piContextCreate>(props, DeviceIds.size(),
50  DeviceIds.data(), nullptr, nullptr, &MContext);
51 #else
52  cl::sycl::detail::pi::die("CUDA support was not enabled at compilation time");
53 #endif
54  } else {
55  getPlugin().call<PiApiKind::piContextCreate>(nullptr, DeviceIds.size(),
56  DeviceIds.data(), nullptr, nullptr, &MContext);
57  }
58 
59  MKernelProgramCache.setContextPtr(this);
60 }

62 context_impl::context_impl(RT::PiContext PiContext, async_handler AsyncHandler,
63  const plugin &Plugin)
64  : MAsyncHandler(AsyncHandler), MDevices(), MContext(PiContext), MPlatform(),
65  MPluginInterop(true), MHostContext(false) {
66 
67  vector_class<RT::PiDevice> DeviceIds;
68  size_t DevicesNum = 0;
69  // TODO catch an exception and put it to list of asynchronous exceptions
70  Plugin.call<PiApiKind::piContextGetInfo>(
71  MContext, PI_CONTEXT_INFO_NUM_DEVICES, sizeof(DevicesNum), &DevicesNum,
72  nullptr);
73  DeviceIds.resize(DevicesNum);
74  // TODO catch an exception and put it to list of asynchronous exceptions
75  Plugin.call<PiApiKind::piContextGetInfo>(MContext, PI_CONTEXT_INFO_DEVICES,
76  sizeof(RT::PiDevice) * DevicesNum,
77  &DeviceIds[0], nullptr);
78 
79  for (auto Dev : DeviceIds) {
80  MDevices.emplace_back(createSyclObjFromImpl<device>(
81  std::make_shared<device_impl>(Dev, Plugin)));
82  }
83  // TODO What if m_Devices if empty? m_Devices[0].get_platform()
84  MPlatform = detail::getSyclObjImpl(MDevices[0].get_platform());
85  // TODO catch an exception and put it to list of asynchronous exceptions
86  // getPlugin() will be the same as the Plugin passed. This should be taken
87  // care of when creating device object.
88  getPlugin().call<PiApiKind::piContextRetain>(MContext);
89  MKernelProgramCache.setContextPtr(this);
90 }
