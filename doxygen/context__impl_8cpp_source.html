33 context_impl::context_impl(const vector_class<cl::sycl::device> Devices,
34  async_handler AsyncHandler,
35  bool UseCUDAPrimaryContext)
36  : MAsyncHandler(AsyncHandler), MDevices(Devices), MContext(nullptr),
37  MPlatform(), MHostContext(false),
38  MUseCUDAPrimaryContext(UseCUDAPrimaryContext) {
39  MPlatform = detail::getSyclObjImpl(MDevices[0].get_platform());
40  vector_class<RT::PiDevice> DeviceIds;
41  for (const auto &D : MDevices) {
42  DeviceIds.push_back(getSyclObjImpl(D)->getHandleRef());
43  }
44 
45  const auto Backend = getPlugin().getBackend();
46  if (Backend == backend::cuda) {
47 #if USE_PI_CUDA
48  const pi_context_properties props[] = {
49  static_cast<pi_context_properties>(PI_CONTEXT_PROPERTIES_CUDA_PRIMARY),
50  static_cast<pi_context_properties>(UseCUDAPrimaryContext), 0};
51 
52  getPlugin().call<PiApiKind::piContextCreate>(
53  props, DeviceIds.size(), DeviceIds.data(), nullptr, nullptr, &MContext);
54 #else
55  cl::sycl::detail::pi::die("CUDA support was not enabled at compilation time");
56 #endif
57  } else {
58  getPlugin().call<PiApiKind::piContextCreate>(nullptr, DeviceIds.size(),
59  DeviceIds.data(), nullptr,
60  nullptr, &MContext);
61  }
62 
63  MKernelProgramCache.setContextPtr(this);
64 }

66 context_impl::context_impl(RT::PiContext PiContext, async_handler AsyncHandler,
67  const plugin &Plugin)
68  : MAsyncHandler(AsyncHandler), MDevices(), MContext(PiContext), MPlatform(),
69  MHostContext(false) {
70 
71  vector_class<RT::PiDevice> DeviceIds;
72  size_t DevicesNum = 0;
73  // TODO catch an exception and put it to list of asynchronous exceptions
74  Plugin.call<PiApiKind::piContextGetInfo>(
75  MContext, PI_CONTEXT_INFO_NUM_DEVICES, sizeof(DevicesNum), &DevicesNum,
76  nullptr);
77  DeviceIds.resize(DevicesNum);
78  // TODO catch an exception and put it to list of asynchronous exceptions
79  Plugin.call<PiApiKind::piContextGetInfo>(MContext, PI_CONTEXT_INFO_DEVICES,
80  sizeof(RT::PiDevice) * DevicesNum,
81  &DeviceIds[0], nullptr);
82 
83  for (auto Dev : DeviceIds) {
84  MDevices.emplace_back(createSyclObjFromImpl<device>(
85  std::make_shared<device_impl>(Dev, Plugin)));
86  }
87  // TODO What if m_Devices if empty? m_Devices[0].get_platform()
88  MPlatform = detail::getSyclObjImpl(MDevices[0].get_platform());
89  // TODO catch an exception and put it to list of asynchronous exceptions
90  // getPlugin() will be the same as the Plugin passed. This should be taken
91  // care of when creating device object.
92  getPlugin().call<PiApiKind::piContextRetain>(MContext);
93  MKernelProgramCache.setContextPtr(this);
94 }
