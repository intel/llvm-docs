35 context_impl::context_impl(const vector_class<cl::sycl::device> Devices,
36  async_handler AsyncHandler,
37  const property_list &PropList)
38  : MAsyncHandler(AsyncHandler), MDevices(Devices), MContext(nullptr),
39  MPlatform(), MPropList(PropList), MHostContext(false) {
40  MPlatform = detail::getSyclObjImpl(MDevices[0].get_platform());
41  vector_class<RT::PiDevice> DeviceIds;
42  for (const auto &D : MDevices) {
43  DeviceIds.push_back(getSyclObjImpl(D)->getHandleRef());
44  }
45 
46  const auto Backend = getPlugin().getBackend();
47  if (Backend == backend::cuda) {
48 #if USE_PI_CUDA
49  const bool UseCUDAPrimaryContext =
50  MPropList.has_property<property::context::cuda::use_primary_context>();
51  const pi_context_properties Props[] = {
52  static_cast<pi_context_properties>(
53  __SYCL_PI_CONTEXT_PROPERTIES_CUDA_PRIMARY),
54  static_cast<pi_context_properties>(UseCUDAPrimaryContext), 0};
55 
56  getPlugin().call<PiApiKind::piContextCreate>(
57  Props, DeviceIds.size(), DeviceIds.data(), nullptr, nullptr, &MContext);
58 #else
59  cl::sycl::detail::pi::die(
60  "CUDA support was not enabled at compilation time");
61 #endif
62  } else {
63  getPlugin().call<PiApiKind::piContextCreate>(nullptr, DeviceIds.size(),
64  DeviceIds.data(), nullptr,
65  nullptr, &MContext);
66  }
67 
68  MKernelProgramCache.setContextPtr(this);
69 }

71 context_impl::context_impl(RT::PiContext PiContext, async_handler AsyncHandler,
72  const plugin &Plugin)
73  : MAsyncHandler(AsyncHandler), MDevices(), MContext(PiContext), MPlatform(),
74  MHostContext(false) {
75 
76  vector_class<RT::PiDevice> DeviceIds;
77  size_t DevicesNum = 0;
78  // TODO catch an exception and put it to list of asynchronous exceptions
79  Plugin.call<PiApiKind::piContextGetInfo>(
80  MContext, PI_CONTEXT_INFO_NUM_DEVICES, sizeof(DevicesNum), &DevicesNum,
81  nullptr);
82  DeviceIds.resize(DevicesNum);
83  // TODO catch an exception and put it to list of asynchronous exceptions
84  Plugin.call<PiApiKind::piContextGetInfo>(MContext, PI_CONTEXT_INFO_DEVICES,
85  sizeof(RT::PiDevice) * DevicesNum,
86  &DeviceIds[0], nullptr);
87 
88  if (!DeviceIds.empty()) {
89  std::shared_ptr<detail::platform_impl> Platform =
90  platform_impl::getPlatformFromPiDevice(DeviceIds[0], Plugin);
91  for (RT::PiDevice Dev : DeviceIds) {
92  MDevices.emplace_back(createSyclObjFromImpl<device>(
93  Platform->getOrMakeDeviceImpl(Dev, Platform)));
94  }
95  MPlatform = Platform;
96  }
97  // TODO catch an exception and put it to list of asynchronous exceptions
98  // getPlugin() will be the same as the Plugin passed. This should be taken
99  // care of when creating device object.
100  //
101  // TODO: Move this backend-specific retain of the context to SYCL-2020 style
102  // make_context<backend::opencl> interop, when that is created.
103  if (getPlugin().getBackend() == cl::sycl::backend::opencl) {
104  getPlugin().call<PiApiKind::piContextRetain>(MContext);
105  }
106  MKernelProgramCache.setContextPtr(this);
107 }
