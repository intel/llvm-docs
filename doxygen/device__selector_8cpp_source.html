17 device device_selector::select_device() const {
18  vector_class<device> devices = device::get_devices();
19  int score = -1;
20  const device *res = nullptr;
21  for (const auto &dev : devices)
22  if (score < operator()(dev)) {
23  res = &dev;
24  score = operator()(dev);
25  }
26 
27  if (res != nullptr)
28  return *res;
29 
30  throw cl::sycl::runtime_error("No device of requested type available.",
31  PI_DEVICE_NOT_FOUND);
32 }

34 int default_selector::operator()(const device &dev) const {
35 
36  // Take note of the SYCL_BE environment variable when doing default selection
37  const char *SYCL_BE = std::getenv("SYCL_BE");
38  if (SYCL_BE) {
39  std::string backend = (SYCL_BE ? SYCL_BE : "");
40  // Taking the version information from the platform gives us more useful
41  // information than the driver_version of the device.
42  const platform platform = dev.get_info<info::device::platform>();
43  const std::string platformVersion =
44  platform.get_info<info::platform::version>();;
45  // If using PI_CUDA, don't accept a non-CUDA device
46  if (platformVersion.find("CUDA") == std::string::npos &&
47  backend == "PI_CUDA") {
48  return -1;
49  }
50  // If using PI_OPENCL, don't accept a non-OpenCL device
51  if (platformVersion.find("OpenCL") == std::string::npos &&
52  backend == "PI_OPENCL") {
53  return -1;
54  }
55  }
56 
57  if (dev.is_gpu())
58  return 500;
59 
60  if (dev.is_accelerator())
61  return 400;
62 
63  if (dev.is_cpu())
64  return 300;
65 
66  if (dev.is_host())
67  return 100;
68 
69  return -1;
70 }
