31 device device_selector::select_device() const {
32  vector_class<device> devices = device::get_devices();
33  int score = REJECT_DEVICE_SCORE;
34  const device *res = nullptr;
35 
36  for (const auto &dev : devices) {
37  int dev_score = (*this)(dev);
38 
39  if (detail::pi::trace(detail::pi::TraceLevel::PI_TRACE_ALL)) {
40  string_class PlatformVersion = dev.get_info<info::device::platform>()
41  .get_info<info::platform::version>();
42  string_class DeviceName = dev.get_info<info::device::name>();
43  std::cout << "SYCL_PI_TRACE[all]: "
44  << "select_device(): -> score = " << score
45  << ((score == REJECT_DEVICE_SCORE) ? "(REJECTED)" : " ")
46  << std::endl
47  << "SYCL_PI_TRACE[all]: "
48  << " platform: " << PlatformVersion << std::endl
49  << "SYCL_PI_TRACE[all]: "
50  << " device: " << DeviceName << std::endl;
51  }
52 
53  // Device is discarded if is marked with REJECT_DEVICE_SCORE
54  if (dev_score == REJECT_DEVICE_SCORE)
55  continue;
56 
57  // SYCL spec says: "If more than one device receives the high score then
58  // one of those tied devices will be returned, but which of the devices
59  // from the tied set is to be returned is not defined". Here we give a
60  // preference to the device of the preferred BE.
61  //
62  if ((score < dev_score) ||
63  (score == dev_score && isDeviceOfPreferredSyclBe(dev))) {
64  res = &dev;
65  score = dev_score;
66  }
67  }
68 
69  if (res != nullptr) {
70  if (detail::pi::trace(detail::pi::TraceLevel::PI_TRACE_BASIC)) {
71  string_class PlatformVersion = res->get_info<info::device::platform>()
72  .get_info<info::platform::version>();
73  string_class DeviceName = res->get_info<info::device::name>();
74  std::cout << "SYCL_PI_TRACE[all]: "
75  << "Selected device ->" << std::endl
76  << "SYCL_PI_TRACE[all]: "
77  << " platform: " << PlatformVersion << std::endl
78  << "SYCL_PI_TRACE[all]: "
79  << " device: " << DeviceName << std::endl;
80  }
81  return *res;
82  }
83 
84  throw cl::sycl::runtime_error("No device of requested type available.",
85  PI_DEVICE_NOT_FOUND);
86 }

92 int default_selector::operator()(const device &dev) const {

94  int Score = REJECT_DEVICE_SCORE;

102  Score += 1000;

112
