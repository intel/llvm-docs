60 template <int ElemsPerAddr,
61  typename = std::enable_if_t<(ElemsPerAddr == 1 || ElemsPerAddr == 2 ||
62  ElemsPerAddr == 4)>>
63 constexpr unsigned int ElemsPerAddrEncoding() {
64  // encoding requires log2 of ElemsPerAddr
65  if constexpr (ElemsPerAddr == 1)
66  return 0;
67  else if constexpr (ElemsPerAddr == 2)
68  return 1;
69  else if constexpr (ElemsPerAddr == 4)
70  return 2;
71 
72  // other cases not needed since enable_if disallows other values
73 }
74 
75 constexpr unsigned int ElemsPerAddrDecoding(unsigned int ElemsPerAddrEncoded) {
76  // encoding requires 2^ElemsPerAddrEncoded
77  return (1 << ElemsPerAddrEncoded);
78 }

83 // flat_read does flat-address gather
84 template <typename Ty, int N, int NumBlk = 0>
85 __ESIMD_INTRIN
86  __ESIMD_DNS::vector_type_t<Ty,
87  N * __ESIMD_DNS::ElemsPerAddrDecoding(NumBlk)>
88  __esimd_svm_gather(__ESIMD_DNS::vector_type_t<uint64_t, N> addrs,
89  int ElemsPerAddr = NumBlk,
90  __ESIMD_DNS::simd_mask_storage_t<N> pred = 1)
91 #ifdef __SYCL_DEVICE_ONLY__
92  ;
93 #else
94 {
95  auto NumBlkDecoded = __ESIMD_DNS::ElemsPerAddrDecoding(NumBlk);
96  __ESIMD_DNS::vector_type_t<Ty, N * __ESIMD_DNS::ElemsPerAddrDecoding(NumBlk)>
97  V = 0;
98  ElemsPerAddr = __ESIMD_DNS::ElemsPerAddrDecoding(ElemsPerAddr);
99  if (sizeof(Ty) == 2)
100  ElemsPerAddr = ElemsPerAddr / 2;
101 
102  for (int I = 0; I < N; I++) {
103  if (pred[I]) {
104  Ty *Addr = reinterpret_cast<Ty *>(addrs[I]);
105  if (sizeof(Ty) <= 2) {
106  for (int J = 0; J < NumBlkDecoded && J < ElemsPerAddr; J++)
107  V[I * NumBlkDecoded + J] = *(Addr + J);
108  } else {
109  for (int J = 0; J < NumBlkDecoded && J < ElemsPerAddr; J++)
110  V[J * N + I] = *(Addr + J);
111  }
112  }
113  }
114  return V;
115 }
116 #endif // __SYCL_DEVICE_ONLY__

201 // Reads a block of data from given surface at given offset.
202 template <typename Ty, int N, typename SurfIndAliasTy, int32_t IsModified = 0>
203 __ESIMD_INTRIN __ESIMD_DNS::vector_type_t<Ty, N>
204 __esimd_oword_ld_unaligned(SurfIndAliasTy surf_ind, uint32_t offset)
205 #ifdef __SYCL_DEVICE_ONLY__
206  ;
207 #else
208 {
209  __ESIMD_DNS::vector_type_t<Ty, N> retv;
210  sycl::detail::ESIMDDeviceInterface *I =
211  sycl::detail::getESIMDDeviceInterface();
212 
213  if (surf_ind == __ESIMD_NS::detail::SLM_BTI) {
214  // O-word/Block load for Shared Local Memory
215  // __ESIMD_NS::detail::SLM_BTI is special binding table index for SLM
216  char *SlmBase = I->__cm_emu_get_slm_ptr();
217  for (int i = 0; i < N; ++i) {
218  Ty *SlmAddr = reinterpret_cast<Ty *>(offset + SlmBase);
219  retv[i] = *SlmAddr;
220  offset += sizeof(Ty);
221  }
222  } else {
223  // O-word/Block load for regular surface indexed by surf_ind
224  char *readBase;
225  uint32_t width;
226  std::mutex *mutexLock;
227 
228  I->sycl_get_cm_buffer_params_ptr(surf_ind, &readBase, &width, &mutexLock);
229 
230  std::lock_guard<std::mutex> lock(*mutexLock);
231 
232  for (int idx = 0; idx < N; idx++) {
233  if (offset >= width) {
234  retv[idx] = 0;
235  } else {
236  retv[idx] = *((Ty *)(readBase + offset));
237  }
238  offset += (uint32_t)sizeof(Ty);
239  }
240  }
241  return retv;
242 }
243 #endif // __SYCL_DEVICE_ONLY__

349 // Low-level surface-based gather. Collects elements located at given offsets in
350 // a surface and returns them as a single \ref simd object. Element can be
351 // 1, 2 or 4-byte value, but is always returned as a 4-byte value within the
352 // resulting simd object, with upper 2 or 3 bytes undefined.
353 // Template (compile-time constant) parameters:
354 // @tparam Ty - element type; can only be a 4-byte integer or \c float,
355 // @tparam N - the number of elements
356 // @tparam SurfIndAliasTy - "surface index alias" type - internal type in the
357 // accessor used to denote the surface
358 // @tparam TySizeLog2 - Log2 of the number of bytes read per element:
359 // 0 - 1 byte, 1 - 2 bytes, 2 - 4 bytes
360 // @tparam Scale - offset scaling factor; must be zero currently
361 // @tparam L1H - L1 cache hint
362 // @tparam L3H - L3 cache hint
363 //
364 // Formal parameters:
365 // @param surf_ind - the surface index, taken from the SYCL memory object
366 // @param global_offset - offset added to each individual element's offset to
367 // compute actual memory access offset for that element
368 // @param elem_offsets - per-element offsets
369 //
370 template <typename Ty, int N, typename SurfIndAliasTy, int TySizeLog2,
371  int16_t Scale = 0>
372 __ESIMD_INTRIN __ESIMD_DNS::vector_type_t<Ty, N>
373 __esimd_gather_scaled2(SurfIndAliasTy surf_ind, uint32_t global_offset,
374  __ESIMD_DNS::vector_type_t<uint32_t, N> elem_offsets)
375 #ifdef __SYCL_DEVICE_ONLY__
376  ;
377 #else
378 {
379  static_assert(N == 1 || N == 8 || N == 16 || N == 32);
380  static_assert(TySizeLog2 <= 2 && Scale == 0);
381  static_assert(std::is_integral<Ty>::value || TySizeLog2 == 2);
382  __ESIMD_UNSUPPORTED_ON_HOST;
383 }

386 // Low-level surface-based scatter. Writes elements of a \ref simd object into a
387 // surface at given offsets. Element can be a 1, 2 or 4-byte value, but it is
388 // always represented as a 4-byte value within the input simd object,
389 // unused (not written) upper bytes are ignored.
390 // Template (compile-time constant) parameters:
391 // @tparam Ty - element type; can only be a 4-byte integer or \c float,
392 // @tparam N - the number of elements to write
393 // @tparam SurfIndAliasTy - "surface index alias" type - internal type in the
394 // accessor used to denote the surface
395 // @tparam TySizeLog2 - Log2 of the number of bytes written per element:
396 // 0 - 1 byte, 1 - 2 bytes, 2 - 4 bytes
397 // @tparam Scale - offset scale; only 0 is supported for now
398 // @tparam L1H - L1 cache hint
399 // @tparam L3H - L3 cache hint
400 //
401 // Formal parameters:
402 // @param pred - per-element predicates; elements with zero corresponding
403 // predicates are not written
404 // @param surf_ind - the surface index, taken from the SYCL memory object
405 // @param global_offset - offset added to each individual element's offset to
406 // compute actual memory access offset for that element
407 // @param elem_offsets - per-element offsets
408 // @param vals - values to write
409 //
410 template <typename Ty, int N, typename SurfIndAliasTy, int TySizeLog2,
411  int16_t Scale = 0>
412 __ESIMD_INTRIN void
413 __esimd_scatter_scaled(__ESIMD_DNS::simd_mask_storage_t<N> pred,
414  SurfIndAliasTy surf_ind, uint32_t global_offset,
415  __ESIMD_DNS::vector_type_t<uint32_t, N> elem_offsets,
416  __ESIMD_DNS::vector_type_t<Ty, N> vals)
417 #ifdef __SYCL_DEVICE_ONLY__
418  ;
419 #else
420 {
421  static_assert(N == 1 || N == 8 || N == 16 || N == 32);
422  static_assert(TySizeLog2 <= 2);
423  static_assert(std::is_integral<Ty>::value || TySizeLog2 == 2);
424 
425  // determine the original element's type size (as __esimd_scatter_scaled
426  // requires vals to be a vector of 4-byte integers)
427  constexpr size_t OrigSize = __ESIMD_DNS::ElemsPerAddrDecoding(TySizeLog2);
428  using RestoredTy = __ESIMD_DNS::uint_type_t<OrigSize>;
429 
430  sycl::detail::ESIMDDeviceInterface *I =
431  sycl::detail::getESIMDDeviceInterface();
432 
433  __ESIMD_DNS::vector_type_t<RestoredTy, N> TypeAdjustedVals;
434  if constexpr (OrigSize == 4) {
435  TypeAdjustedVals = __ESIMD_DNS::bitcast<RestoredTy, Ty, N>(vals);
436  } else {
437  static_assert(OrigSize == 1 || OrigSize == 2);
438  TypeAdjustedVals = __ESIMD_DNS::convert_vector<RestoredTy, Ty, N>(vals);
439  }
440 
441  if (surf_ind == __ESIMD_NS::detail::SLM_BTI) {
442  // Scattered-store for Shared Local Memory
443  // __ESIMD_NS::detail::SLM_BTI is special binding table index for SLM
444  assert(global_offset == 0);
445  char *SlmBase = I->__cm_emu_get_slm_ptr();
446  for (int i = 0; i < N; ++i) {
447  if (pred[i]) {
448  RestoredTy *addr =
449  reinterpret_cast<RestoredTy *>(elem_offsets[i] + SlmBase);
450  *addr = TypeAdjustedVals[i];
451  }
452  }
453  } else {
454  // Scattered-store for regular surface indexed by surf_ind
455  char *writeBase;
456  uint32_t width;
457  std::mutex *mutexLock;
458 
459  I->sycl_get_cm_buffer_params_ptr(surf_ind, &writeBase, &width, &mutexLock);
460  writeBase += global_offset;
461 
462  std::lock_guard<std::mutex> lock(*mutexLock);
463 
464  for (int idx = 0; idx < N; idx++) {
465  if (pred[idx]) {
466  RestoredTy *addr =
467  reinterpret_cast<RestoredTy *>(elem_offsets[idx] + writeBase);
468  *addr = TypeAdjustedVals[idx];
469  }
470  }
471 
472  // TODO : Optimize
473  I->cm_fence_ptr();
474  }
475 }
476 #endif // __SYCL_DEVICE_ONLY__

908 template <typename Ty, int M, int N, int Modifier, typename TACC, int Plane,
909  int BlockWidth>
910 __ESIMD_INTRIN __ESIMD_DNS::vector_type_t<Ty, M * N>
911 __esimd_media_ld(TACC handle, unsigned x, unsigned y)
912 #ifdef __SYCL_DEVICE_ONLY__
913  ;
914 #else
915 {
916  __ESIMD_DNS::vector_type_t<Ty, M * N> vals;
917  char *readBase;
918  uint32_t bpp;
919  uint32_t imgWidth;
920  uint32_t imgHeight;
921  std::mutex *mutexLock;
922 
923  assert((handle != __ESIMD_NS::detail::SLM_BTI) &&
924  "__esimd_media_ld cannot access SLM");
925 
926  sycl::detail::getESIMDDeviceInterface()->sycl_get_cm_image_params_ptr(
927  handle, &readBase, &imgWidth, &imgHeight, &bpp, &mutexLock);
928 
929  std::lock_guard<std::mutex> lock(*mutexLock);
930 
931  int x_pos_a, y_pos_a, offset, index;
932 
933  // TODO : Remove intermediate 'in' matrix
934  std::vector<std::vector<Ty>> in(M, std::vector<Ty>(N));
935  int R = M;
936  int C = N;
937  for (int i = 0; i < R; i++) {
938  for (int j = 0; j < C; j++) {
939  x_pos_a = x + j * sizeof(Ty);
940  { y_pos_a = y + i; }
941  // We should check the boundary condition based on sizeof(Ty), x_pos_a is
942  // 0-based Note: Use a signed variable; otherwise sizeof(Ty) is unsigned
943  if ((x_pos_a + sizeof(Ty)) > imgWidth) {
944  // If we're trying to read outside the boundary, limit the value of
945  // x_pos_a Assumption -- We don't this situation:
946  // x_pos_a width's boundary
947  // | |
948  // <---type(Ty)--->
949  // At most x_pos_a+sizeof(Ty) is exactly at the boundary.
950  x_pos_a = imgWidth;
951  }
952  if (y_pos_a > imgHeight - 1) {
953  y_pos_a = imgHeight - 1;
954  }
955  if (y_pos_a < 0) {
956  y_pos_a = 0;
957  }
958  {
959  if (x_pos_a < 0) {
960  // Need to align x position to bbp
961  int offset = x % bpp;
962  x_pos_a -= offset;
963  }
964  while (x_pos_a < 0) {
965  // If we're trying to read outside the left boundary, increase x_pos_a
966  x_pos_a += bpp;
967  }
968  }
969

971  {
972  x_pos_a = x_pos_a - bpp;
973  for (uint byte_count = 0; byte_count < sizeof(Ty); byte_count++) {
974  if (x_pos_a >= imgWidth) {
975  x_pos_a = x_pos_a - bpp;
976  }

979  /*
980  If destination size per element is less then or equal pixel size
981  of the surface move the pixel value accross the destination
982  elements. If destination size per element is greater then pixel
983  size of the surface replicate pixel value in the destination
984  element.
985  */
986  if (sizeof(Ty) <= bpp) {
987  for (uint bpp_count = 0; j < C && bpp_count < bpp;
988  j++, bpp_count += sizeof(Ty)) {
989  in[i][j] = *((Ty *)(readBase + offset + bpp_count));
990  }
991  j--;
992  break;
993  } else {
994  // ((unsigned char*)in.get_addr(i*C+j))[byte_count] = *((unsigned
995  // char*)((char*)buff_iter->p + offset));
996  unsigned char *pTempBase =
997  ((unsigned char *)in[i].data()) + j * sizeof(Ty);
998  pTempBase[byte_count] = *((unsigned char *)(readBase + offset));
999  }
1000 
1001  x_pos_a = x_pos_a + 1;
1002  }
1003  x_pos_a = imgWidth;
1004  }
1005  } else {
1006  offset = y_pos_a * imgWidth + x_pos_a;
1007  { in[i][j] = *((Ty *)(readBase + offset)); }
1008  }
1009  }
1010  }
