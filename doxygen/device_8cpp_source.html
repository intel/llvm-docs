19 void force_type(info::device_type &t, const info::device_type &ft) {

40 vector_class<device> device::get_devices(info::device_type deviceType) {
41  vector_class<device> devices;
42  // Host device availability should not depend on the forced type
43  const bool includeHost =
44  detail::match_types(deviceType, info::device_type::host);
45  info::device_type forced_type = detail::get_forced_type();
46  // Exclude devices which do not match requested device type
47  if (detail::match_types(deviceType, forced_type)) {
48  detail::force_type(deviceType, forced_type);
49  for (const auto &plt : platform::get_platforms()) {
50  if (includeHost && plt.is_host()) {
51  vector_class<device> host_device(
52  plt.get_devices(info::device_type::host));
53  if (!host_device.empty())
54  devices.insert(devices.end(), host_device.begin(), host_device.end());
55  } else {
56  vector_class<device> found_devices(plt.get_devices(deviceType));
57  if (!found_devices.empty())
58  devices.insert(devices.end(), found_devices.begin(),
59  found_devices.end());
60  }
61  }
62  }
63  return devices;
64 }
