17 // For Nearest Filtering mode, process float4 Coordinates and
18 // return the appropriate Pixel Coordinates based on Addressing Mode.
19 int4 getPixelCoordNearestFiltMode(float4 Coorduvw,
20  const addressing_mode SmplAddrMode,
21  const range<3> ImgRange) {
22  int4 Coordijk(0);
23  int4 Rangewhd(ImgRange[0], ImgRange[1], ImgRange[2], 0);
24  switch (SmplAddrMode) {
25  case addressing_mode::mirrored_repeat: {
26  float4 Tempuvw(0);
27  Tempuvw = 2.0f * sycl::rint(0.5f * Coorduvw);
28  Tempuvw = sycl::fabs(Coorduvw - Tempuvw);
29  Tempuvw = Tempuvw * (Rangewhd.convert<cl_float>());
30  Tempuvw = (sycl::floor(Tempuvw));
31  Coordijk = Tempuvw.convert<cl_int>();
32  Coordijk = sycl::min(Coordijk, (Rangewhd - 1));
33  // Eg:
34  // u,v,w = {2.3,1.7,0.5} // normalized coordinates.
35  // w,h,d = {9,9,9}
36  // u1=2*rint(1.15)=2
37  // v1=2*rint(0.85)=2
38  // w1=2*rint(0.5)=0
39  // u1=fabs(2.3-2)=.3
40  // v1=fabs(1.7-2)=.3
41  // w1=fabs(0.5-0)=.5
42  // u1=0.3*9=2.7
43  // v1=0.3*9=2.7
44  // w1=0.5*9=4.5
45  // i,j,k = {2,2,4}
46 
47  } break;
48  case addressing_mode::repeat: {
49 
50  float4 Tempuvw(0);
51  Tempuvw = (Coorduvw - sycl::floor(Coorduvw)) * Rangewhd.convert<cl_float>();
52  Coordijk = (sycl::floor(Tempuvw)).convert<cl_int>();
53  int4 GreaterThanEqual = (Coordijk >= Rangewhd);
54  Coordijk = sycl::select(Coordijk, (Coordijk - Rangewhd), GreaterThanEqual);
55  // Eg:
56  // u = 2.3; v = 1.5; w = 0.5; // normalized coordinates.
57  // w,h,d = {9,9,9};
58  // u1= 0.3*w;
59  // v1= 0.5*d;
60  // w1= 0.5*h;
61  // i = floor(2.7);
62  // j = floor(4.5);
63  // k = floor(4.5);
64  // if (i/j/k > w/h/d-1)
65  // // Condition is not satisfied.
66  // (This condition I think will only be satisfied if the floating point
67  // arithmetic of multiplication
68  // gave a value in u1/v1/w1 as > w/h/d)
69  // i = 2; j = 4; k = 4;
70  } break;
71  case addressing_mode::clamp_to_edge:
72  Coordijk = (sycl::floor(Coorduvw)).convert<cl_int>();
73  Coordijk = sycl::clamp(Coordijk, int4(0), (Rangewhd - 1));
74  break;
75  case addressing_mode::clamp:
76  Coordijk = (sycl::floor(Coorduvw)).convert<cl_int>();
77  Coordijk = sycl::clamp(Coordijk, int4(-1), Rangewhd);
78  break;
79  case addressing_mode::none:
80  Coordijk = (sycl::floor(Coorduvw)).convert<cl_int>();
81  break;
82  }
83  return Coordijk;
84 }

86 // For Linear Filtering mode, process Coordinates-Coord_uvw and return
87 // coordinate indexes based on Addressing Mode.
88 // The value returned contains (i0,j0,k0,0,i1,j1,k1,0).
89 // Retabc contains the values of (a,b,c,0)
90 // The caller of this function should use these values to create the 8 pixel
91 // coordinates and multiplication coefficients.
92 int8 getPixelCoordLinearFiltMode(float4 Coorduvw,
93  const addressing_mode SmplAddrMode,
94  const range<3> ImgRange, float4 &Retabc) {
95  int4 Rangewhd(ImgRange[0], ImgRange[1], ImgRange[2], 0);
96  int4 Ci0j0k0(0);
97  int4 Ci1j1k1(0);
98  int4 Int_uvwsubhalf = sycl::floor(Coorduvw - 0.5f).convert<cl_int>();
99 
100  switch (SmplAddrMode) {
101  case addressing_mode::mirrored_repeat: {
102  float4 Temp;
103  Temp = (sycl::rint(Coorduvw * 0.5f)) * 2.0f;
104  Temp = sycl::fabs(Coorduvw - Temp);
105  Coorduvw = Temp * Rangewhd.convert<cl_float>();
106  Int_uvwsubhalf = sycl::floor(Coorduvw - 0.5f).convert<cl_int>();
107 
108  Ci0j0k0 = Int_uvwsubhalf;
109  Ci1j1k1 = Ci0j0k0 + 1;
110 
111  Ci0j0k0 = sycl::max(Ci0j0k0, 0);
112  Ci1j1k1 = sycl::min(Ci1j1k1, (Rangewhd - 1));
113  } break;
114  case addressing_mode::repeat: {
115 
116  Coorduvw =
117  (Coorduvw - sycl::floor(Coorduvw)) * Rangewhd.convert<cl_float>();
118  Int_uvwsubhalf = sycl::floor(Coorduvw - 0.5f).convert<cl_int>();
119 
120  Ci0j0k0 = Int_uvwsubhalf;
121  Ci1j1k1 = Ci0j0k0 + 1;
122 
123  Ci0j0k0 = sycl::select(Ci0j0k0, (Ci0j0k0 + Rangewhd), Ci0j0k0 < int4(0));
124  Ci1j1k1 = sycl::select(Ci1j1k1, (Ci1j1k1 - Rangewhd), Ci1j1k1 >= Rangewhd);
125 
126  } break;
127  case addressing_mode::clamp_to_edge: {
128  Ci0j0k0 = sycl::clamp(Int_uvwsubhalf, int4(0), (Rangewhd - 1));
129  Ci1j1k1 = sycl::clamp((Int_uvwsubhalf + 1), int4(0), (Rangewhd - 1));
130  break;
131  }
132  case addressing_mode::clamp: {
133  Ci0j0k0 = sycl::clamp(Int_uvwsubhalf, int4(-1), Rangewhd);
134  Ci1j1k1 = sycl::clamp((Int_uvwsubhalf + 1), int4(-1), Rangewhd);
135  break;
136  }
137  case addressing_mode::none: {
138  Ci0j0k0 = Int_uvwsubhalf;
139  Ci1j1k1 = Ci0j0k0 + 1;
140  break;
141  }
142  }
143  Retabc = (Coorduvw - 0.5f) - (Int_uvwsubhalf.convert<cl_float>());
144  Retabc.w() = 0.0f;
145  return int8(Ci0j0k0, Ci1j1k1);
146 }

148 // Function returns true when PixelCoord is out of image's range.
149 // It is only valid for addressing_mode::clamp/none.
150 // Note: For addressing_mode::none , spec says outofrange access is not defined.
151 // This function handles this addressing_mode to avoid accessing out of bound
152 // memories on host.
153 bool isOutOfRange(const int4 PixelCoord, const addressing_mode SmplAddrMode,
154  const range<3> ImgRange) {
155 
156  if (SmplAddrMode != addressing_mode::clamp &&
157  SmplAddrMode != addressing_mode::none)
158  return false;
159 
160  auto CheckOutOfRange = [](cl_int Coord, cl_int Range) {
161  return ((Coord < 0) || (Coord >= Range));
162  };
163 
164  bool CheckWidth = CheckOutOfRange(PixelCoord.x(), ImgRange[0]);
165  bool CheckHeight = CheckOutOfRange(PixelCoord.y(), ImgRange[1]);
166  bool CheckDepth = CheckOutOfRange(PixelCoord.z(), ImgRange[2]);
167 
168  return (CheckWidth || CheckHeight || CheckDepth);
169 }
