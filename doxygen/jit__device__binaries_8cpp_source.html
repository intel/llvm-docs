92 sycl_device_binary_struct DeviceBinaryContainer::getPIDeviceBinary(
93  const unsigned char *BinaryStart, size_t BinarySize, const char *TargetSpec,
94  sycl_device_binary_type Format) {
95  sycl_device_binary_struct DeviceBinary;
96  DeviceBinary.Version = SYCL_DEVICE_BINARY_VERSION;
97  DeviceBinary.Kind = SYCL_DEVICE_BINARY_OFFLOAD_KIND_SYCL;
98  DeviceBinary.Format = Format;
99  DeviceBinary.CompileOptions = "";
100  DeviceBinary.LinkOptions = "";
101  DeviceBinary.ManifestStart = nullptr;
102  DeviceBinary.ManifestEnd = nullptr;
103  // It is safe to use these pointers here, as their lifetime is managed by
104  // the JITContext.
105  DeviceBinary.BinaryStart = BinaryStart;
106  DeviceBinary.BinaryEnd = BinaryStart + BinarySize;
107  DeviceBinary.DeviceTargetSpec = TargetSpec;
108  DeviceBinary.EntriesBegin = PIOffloadEntries.data();
109  DeviceBinary.EntriesEnd = PIOffloadEntries.data() + PIOffloadEntries.size();
110  DeviceBinary.PropertySetsBegin = PIPropertySets.data();
111  DeviceBinary.PropertySetsEnd = PIPropertySets.data() + PIPropertySets.size();
112  Fused = false;
113  return DeviceBinary;
114 }
