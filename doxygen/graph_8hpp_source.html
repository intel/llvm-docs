67  case UGF::sycl_kernel_bundle:

69  case UGF::sycl_ext_oneapi_kernel_properties:
70  return "sycl_ext_oneapi_kernel_properties";
71  case UGF::sycl_ext_oneapi_enqueue_barrier:

73  case UGF::sycl_ext_oneapi_memcpy2d:
