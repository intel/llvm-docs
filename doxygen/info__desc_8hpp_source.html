242 // Provide an alias to the return type for each of the info parameters

245 #define PARAM_TRAITS_SPEC(param_type, param, ret_type) \
246  template <> class param_traits<param_type, param_type::param> { \
247  public: \
248  using return_type = ret_type; \
249  };

251 #define PARAM_TRAITS_SPEC_WITH_INPUT(param_type, param, ret_type, in_type) \

254  using return_type = ret_type; \
255  using input_type = in_type; \
