443  template <typename convertT,
444  rounding_mode roundingMode = rounding_mode::automatic>
445  vec<convertT, NumElements> convert() const {
446 
447  using T = ConvertBoolAndByteT<DataT>;
448  using R = ConvertBoolAndByteT<convertT>;
449  using bfloat16 = sycl::ext::oneapi::bfloat16;
450  static_assert(std::is_integral_v<R> ||
451  detail::is_floating_point<R>::value ||
452  std::is_same_v<R, bfloat16>,
453  "Unsupported convertT");
454 
455  using OpenCLT = detail::ConvertToOpenCLType_t<T>;
456  using OpenCLR = detail::ConvertToOpenCLType_t<R>;
457  vec<convertT, NumElements> Result;
458 
459  // convertImpl can't be called with the same From and To types and therefore
460  // we need some special processing in a few cases.
461  if constexpr (std::is_same_v<DataT, convertT>) {
462  return *this;
463  } else if constexpr (std::is_same_v<OpenCLT, OpenCLR> ||
464  std::is_same_v<T, R>) {
465  for (size_t I = 0; I < NumElements; ++I)
466  Result[I] = static_cast<convertT>(getValue(I));
467  return Result;
468  } else {
469 
470 #ifdef __SYCL_DEVICE_ONLY__
471  using OpenCLVecT = OpenCLT __attribute__((ext_vector_type(NumElements)));
472  using OpenCLVecR = OpenCLR __attribute__((ext_vector_type(NumElements)));
473 
474  auto NativeVector = sycl::bit_cast<vector_t>(*this);
475  using ConvertTVecType = typename vec<convertT, NumElements>::vector_t;
476 
477  // Whole vector conversion can only be done, if:
478  constexpr bool canUseNativeVectorConvert =
479 #ifdef __NVPTX__
480  // TODO: Likely unnecessary as
481  // https://github.com/intel/llvm/issues/11840 has been closed
482  // already.
483  false &&
484 #endif
485  NumElements > 1 &&
486  // - vec storage has an equivalent OpenCL native vector it is
487  // implicitly convertible to. There are some corner cases where it
488  // is not the case with char, long and long long types.
489  std::is_convertible_v<vector_t, OpenCLVecT> &&
490  std::is_convertible_v<ConvertTVecType, OpenCLVecR> &&
491  // - it is not a signed to unsigned (or vice versa) conversion
492  // see comments within 'convertImpl' for more details;
493  !detail::is_sint_to_from_uint<T, R>::value &&
494  // - destination type is not bool. bool is stored as integer under the
495  // hood and therefore conversion to bool looks like conversion
496  // between two integer types. Since bit pattern for true and false
497  // is not defined, there is no guarantee that integer conversion
498  // yields right results here;
499  !std::is_same_v<convertT, bool>;
500 
501  if constexpr (canUseNativeVectorConvert) {
502  auto val = detail::convertImpl<T, R, roundingMode, NumElements, OpenCLVecT,
503  OpenCLVecR>(NativeVector);
504  Result.m_Data = sycl::bit_cast<decltype(Result.m_Data)>(val);
505  } else
506 #endif // __SYCL_DEVICE_ONLY__
507  {
508  // Otherwise, we fallback to per-element conversion:
509  for (size_t I = 0; I < NumElements; ++I) {
510  auto val =
511  detail::convertImpl<T, R, roundingMode, 1, OpenCLT, OpenCLR>(
512  getValue(I));
513 #ifdef __SYCL_DEVICE_ONLY__
514  // On device, we interpret BF16 as uint16.
515  if constexpr (std::is_same_v<convertT, bfloat16>)
516  Result[I] = sycl::bit_cast<convertT>(val);
517  else
518 #endif
519  Result[I] = static_cast<convertT>(val);
520  }
521  }
522  }
523  return Result;
524  }

720 // Actual computation performed on conversion or assignment operators.
721 template <typename VecT, typename OperationLeftT, typename OperationRightT,
722  template <typename> class OperationCurrentT, int... Indexes>
723 class SwizzleOp {
724  using DataT = typename VecT::element_type;
725  // Certain operators return a vector with a different element type. Also, the
726  // left and right operand types may differ. CommonDataT selects a result type
727  // based on these types to ensure that the result value can be represented.
728  //
729  // Example 1:
730  // sycl::vec<unsigned char, 4> vec{...};
731  // auto result = 300u + vec.x();
732  //
733  // CommonDataT is std::common_type_t<OperationLeftT, OperationRightT> since
734  // it's larger than unsigned char.
735  //
736  // Example 2:
737  // sycl::vec<bool, 1> vec{...};
738  // auto result = vec.template swizzle<sycl::elem::s0>() && vec;
739  //
740  // CommonDataT is DataT since operator&& returns a vector with element type
741  // int8_t, which is larger than bool.
742  //
743  // Example 3:
744  // sycl::vec<std::byte, 4> vec{...}; auto swlo = vec.lo();
745  // auto result = swlo == swlo;
746  //
747  // CommonDataT is DataT since operator== returns a vector with element type
748  // int8_t, which is the same size as std::byte. std::common_type_t<DataT, ...>
749  // can't be used here since there's no type that int8_t and std::byte can both
750  // be implicitly converted to.
751  using OpLeftDataT = typename OperationLeftT::DataT;
752  using OpRightDataT = typename OperationRightT::DataT;
753  using CommonDataT = std::conditional_t<
754  sizeof(DataT) >= sizeof(std::common_type_t<OpLeftDataT, OpRightDataT>),
755  DataT, std::common_type_t<OpLeftDataT, OpRightDataT>>;
