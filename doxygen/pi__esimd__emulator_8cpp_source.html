179 // Function template to generate entry point of kernel execution as
180 // raw function pointer. CM kernel launcher executes one instance of
181 // this function per 'NDims'
182 template <int NDims> void InvokeKernel(KernelInvocationContext<NDims> *ctx) {
183 
184  sycl::range<NDims> GroupSize{
185  sycl::detail::InitializedVal<NDims, sycl::range>::template get<0>()};
186 
187  for (int i = 0; i < NDims; ++i) {
188  GroupSize[i] = ctx->GlobalSize[i] / ctx->LocalSize[i];
189  }
190 
191  const sycl::id<NDims> LocalID = RangeBuilder<NDims>::create(
192  [](int i) { return cm_support::get_thread_idx(i); });
193 
194  const sycl::id<NDims> GroupID = RangeBuilder<NDims>::create(
195  [](int i) { return cm_support::get_group_idx(i); });
196 
197  const sycl::group<NDims> Group = IDBuilder::createGroup<NDims>(
198  ctx->GlobalSize, ctx->LocalSize, GroupSize, GroupID);
199 
200  const sycl::id<NDims> GlobalID =
201  GroupID * ctx->LocalSize + LocalID + ctx->GlobalOffset;
202 
203  const sycl::item<NDims, /*Offset=*/true> GlobalItem =
204  IDBuilder::createItem<NDims, true>(ctx->GlobalSize, GlobalID,
205  ctx->GlobalOffset);
206 
207  const sycl::item<NDims, /*Offset=*/false> LocalItem =
208  IDBuilder::createItem<NDims, false>(ctx->LocalSize, LocalID);
209 
210  const sycl::nd_item<NDims> NDItem =
211  IDBuilder::createNDItem<NDims>(GlobalItem, LocalItem, Group);
212 
213  ctx->Func(NDItem);
214 }

226  void runIterationSpace(const sycl::range<DIMS> &LocalSize,
227  const sycl::range<DIMS> &GlobalSize,
228  const sycl::id<DIMS> &GlobalOffset) {
229 
230  for (int I = 0; I < DIMS; I++) {
231  SpaceDim[I] = (uint32_t)LocalSize[I];
232  GroupDim[I] = (uint32_t)(GlobalSize[I] / LocalSize[I]);
233  }
234 
235  const auto InvokeKernelArg = KernelInvocationContext<DIMS>{
236  MKernel, LocalSize, GlobalSize, GlobalOffset};
237 
238  EsimdemuKernel{reinterpret_cast<fptrVoid>(InvokeKernel<DIMS>), GroupDim,
239  SpaceDim}
240  .launchMT(sizeof(InvokeKernelArg), &InvokeKernelArg);
241  }

327  if constexpr (NDims == 1)
328  return sycl::range<NDims>{Array[0]};
329  else if constexpr (NDims == 2)
330  return sycl::range<NDims>{Array[0], Array[1]};
331  else if constexpr (NDims == 3)

502  std::lock_guard<std::mutex> Lock(PiDeviceCacheMutex);
503 
504  if (DeviceCachePopulated) {
505  return PI_SUCCESS;
506  }
507  cm_support::CmDevice *CmDevice = nullptr;
508  // TODO FIXME Implement proper version checking and reporting:
509  // - version passed to cm_support::CreateCmDevice
510  // - CmEmuVersion
511  // - PluginVersion
512  // - ESIMDEmuPluginOpaqueData::version
513  //
514  // PI_DEVICE_INFO_DRIVER_VERSION could report the ESIMDDeviceInterface

526  // unsigned integer value = 100 * major + minor. Second from right
527  // digit in decimal must be zero as it is used as 'dot'
528  // REF - $CM_EMU/common/cm_version_defs.h - 'CURRENT_CM_VERSION'
529  // e.g. CM version 7.3 => Device version = 703

540  StrFormat << (int)(Version / 100) << "." << (int)(Version % 10);

543  new _pi_device(this, CmDevice, StrFormat.str()));
544  PiDeviceCache = std::move(Device);
