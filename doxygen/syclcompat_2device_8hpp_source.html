298  void reset() {
299  std::lock_guard<std::mutex> lock(m_mutex);
300  // The queues are shared_ptrs and the ref counts of the shared_ptrs increase
301  // only in wait_and_throw(). If there is no other thread calling
302  // wait_and_throw(), the queues will be destructed. The destructor waits for
303  // all commands executing on the queue to complete. It isn't possible to
304  // destroy a queue immediately. This is a synchronization point in SYCL.
305  _queues.clear();
306  // create new default queue.
307  _queues.push_back(
308  std::make_shared<sycl::queue>(_ctx, *this, detail::exception_handler,
309  sycl::property::queue::in_order()));
310  _saved_queue = _default_queue = _queues.front().get();
311  }
312 
313  void set_default_queue(const sycl::queue &q) {
314  std::lock_guard<std::mutex> lock(m_mutex);
315  _queues.front().get()->wait_and_throw();
316  _queues[0] = std::make_shared<sycl::queue>(q);
317  if (_saved_queue == _default_queue)
318  _saved_queue = _queues.front().get();
319  _default_queue = _queues.front().get();
320  }
321 
322  sycl::queue *default_queue() { return _default_queue; }
323 
324  void queues_wait_and_throw() {
325  std::unique_lock<std::mutex> lock(m_mutex);
326  std::vector<std::shared_ptr<sycl::queue>> current_queues(_queues);
327  lock.unlock();
328  for (const auto &q : current_queues) {
329  q->wait_and_throw();
330  }
331  // Guard the destruct of current_queues to make sure the ref count is safe.
332  lock.lock();
333  }
334  queue_ptr create_queue(bool print_on_async_exceptions = false,
335  bool in_order = true) {
336  std::lock_guard<std::mutex> lock(m_mutex);
337  sycl::property_list prop = {};
338  if (in_order) {
339  prop = {sycl::property::queue::in_order()};
340  }
341  if (print_on_async_exceptions) {
342  _queues.push_back(std::make_shared<sycl::queue>(
343  _ctx, *this, detail::exception_handler, prop));
344  } else {
345  _queues.push_back(std::make_shared<sycl::queue>(_ctx, *this, prop));
346  }
347  return _queues.back().get();
348  }
349  void destroy_queue(queue_ptr &queue) {
350  std::lock_guard<std::mutex> lock(m_mutex);
351  _queues.erase(
352  std::remove_if(_queues.begin(), _queues.end(),
353  [=](const std::shared_ptr<sycl::queue> &q) -> bool {
354  return q.get() == queue;
355  }),
356  _queues.end());
357  queue = nullptr;
358  }
359  void set_saved_queue(queue_ptr q) {
360  std::lock_guard<std::mutex> lock(m_mutex);
361  _saved_queue = q;
362  }
363  queue_ptr get_saved_queue() const {
364  std::lock_guard<std::mutex> lock(m_mutex);
365  return _saved_queue;
366  }
367  sycl::context get_context() const { return _ctx; }

370  void get_version(int &major, int &minor) const {
371  // Version string has the following format:
372  // a. OpenCL<space><major.minor><space><vendor-specific-information>
373  // b. <major.minor>
374  // c. <AmdGcnArchName> e.g gfx1030
375  std::string ver;
376  ver = get_info<sycl::info::device::version>();
377  std::string::size_type i = 0;
378  while (i < ver.size()) {
379  if (isdigit(ver[i]))
380  break;
381  i++;
382  }
383  major = std::stoi(&(ver[i]));
384  while (i < ver.size()) {
385  if (ver[i] == '.')
386  break;
387  i++;
388  }
389  if (i < ver.size()) {
390  // a. and b.
391  i++;
392  minor = std::stoi(&(ver[i]));
393  } else {
394  // c.
395  minor = 0;
396  }
397  }
398  void add_event(sycl::event event) {
399  std::lock_guard<std::mutex> lock(m_mutex);
400  _events.push_back(event);
401  }
402  friend sycl::event free_async(const std::vector<void *> &,
403  const std::vector<sycl::event> &, sycl::queue);
404  queue_ptr _default_queue;
405  queue_ptr _saved_queue;
406  sycl::context _ctx;
407  std::vector<std::shared_ptr<sycl::queue>> _queues;
408  mutable std::mutex m_mutex;
409  std::vector<sycl::event> _events;
410 };

433  std::lock_guard<std::mutex> lock(m_mutex);

437  return *_devs[_cpu_device];

439  }

441  std::lock_guard<std::mutex> lock(m_mutex);

445  unsigned int current_device_id() const {
446  std::lock_guard<std::mutex> lock(m_mutex);
447  auto it = _thread2dev_map.find(get_tid());
448  if (it != _thread2dev_map.end())
449  return it->second;
450  return _default_device_id;
451  }
452  void select_device(unsigned int id) {
453  std::lock_guard<std::mutex> lock(m_mutex);
454  check_id(id);
455  _thread2dev_map[get_tid()] = id;
456  }
457  unsigned int device_count() { return _devs.size(); }

459  unsigned int get_device_id(const sycl::device &dev) {
460  unsigned int id = 0;
461  for (auto dev_item : _devs) {
462  if (*dev_item == dev) {
463  break;
464  }
465  id++;
466  }
467  return id;
468  }

482  dev_mgr() {
483  sycl::device default_device = sycl::device(sycl::default_selector_v);
484  _devs.push_back(std::make_shared<device_ext>(default_device));
485 
486  std::vector<sycl::device> sycl_all_devs =
487  sycl::device::get_devices(sycl::info::device_type::all);
488  // Collect other devices except for the default device.
489  if (default_device.is_cpu())
490  _cpu_device = 0;
491  for (auto &dev : sycl_all_devs) {
492  if (dev == default_device) {
493  continue;
494  }
495  _devs.push_back(std::make_shared<device_ext>(dev));
496  if (_cpu_device == -1 && dev.is_cpu()) {
497  _cpu_device = _devs.size() - 1;
498  }
499  }
500  }

502  if (id >= _devs.size()) {
503  throw std::runtime_error("invalid device id");
504  }

513  int _cpu_device = -1;
