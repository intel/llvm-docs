36 inline std::vector<info::fp_config> read_fp_bitfield(pi_device_fp_config bits) {
37  std::vector<info::fp_config> result;
38  if (bits & PI_FP_DENORM)
39  result.push_back(info::fp_config::denorm);
40  if (bits & PI_FP_INF_NAN)
41  result.push_back(info::fp_config::inf_nan);
42  if (bits & PI_FP_ROUND_TO_NEAREST)
43  result.push_back(info::fp_config::round_to_nearest);
44  if (bits & PI_FP_ROUND_TO_ZERO)
45  result.push_back(info::fp_config::round_to_zero);
46  if (bits & PI_FP_ROUND_TO_INF)
47  result.push_back(info::fp_config::round_to_inf);
48  if (bits & PI_FP_FMA)
49  result.push_back(info::fp_config::fma);
50  if (bits & PI_FP_SOFT_FLOAT)
51  result.push_back(info::fp_config::soft_float);
52  if (bits & PI_FP_CORRECTLY_ROUNDED_DIVIDE_SQRT)
53  result.push_back(info::fp_config::correctly_rounded_divide_sqrt);
54  return result;
55 }

57 inline std::vector<info::partition_affinity_domain>

62  if (bits & PI_DEVICE_AFFINITY_DOMAIN_L4_CACHE)
63  result.push_back(info::partition_affinity_domain::L4_cache);
64  if (bits & PI_DEVICE_AFFINITY_DOMAIN_L3_CACHE)

66  if (bits & PI_DEVICE_AFFINITY_DOMAIN_L2_CACHE)

68  if (bits & PI_DEVICE_AFFINITY_DOMAIN_L1_CACHE)
69  result.push_back(info::partition_affinity_domain::L1_cache);
70  if (bits & PI_DEVICE_AFFINITY_DOMAIN_NEXT_PARTITIONABLE)
71  result.push_back(info::partition_affinity_domain::next_partitionable);

388 // Specialization for partition properties, variable OpenCL return size
389 template <>
390 struct get_device_info_impl<std::vector<info::partition_property>,
391  info::device::partition_properties> {
392  static std::vector<info::partition_property> get(const DeviceImplPtr &Dev) {
393  auto info_partition = PiInfoCode<info::device::partition_properties>::value;
394  const auto &Plugin = Dev->getPlugin();
395 
396  size_t resultSize;
397  Plugin->call<PiApiKind::piDeviceGetInfo>(
398  Dev->getHandleRef(), info_partition, 0, nullptr, &resultSize);
399 
400  size_t arrayLength = resultSize / sizeof(cl_device_partition_property);
401  if (arrayLength == 0) {
402  return {};
403  }
404  std::unique_ptr<cl_device_partition_property[]> arrayResult(
405  new cl_device_partition_property[arrayLength]);
406  Plugin->call<PiApiKind::piDeviceGetInfo>(Dev->getHandleRef(),
407  info_partition, resultSize,
408  arrayResult.get(), nullptr);
409 
410  std::vector<info::partition_property> result;
411  for (size_t i = 0; i < arrayLength; ++i) {
412  // OpenCL extensions may have partition_properties that
413  // are not yet defined for SYCL (eg. CL_DEVICE_PARTITION_BY_NAMES_INTEL)
414  info::partition_property pp(
415  static_cast<info::partition_property>(arrayResult[i]));
416  if (is_sycl_partition_property(pp))
417  result.push_back(pp);
418  }
419  return result;
420  }
