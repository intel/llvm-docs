68  CG(CGTYPE Type, vector_class<vector_class<char>> ArgsStorage,
69  vector_class<detail::AccessorImplPtr> AccStorage,
70  vector_class<shared_ptr_class<const void>> SharedPtrStorage,
71  vector_class<Requirement *> Requirements,
72  vector_class<detail::EventImplPtr> Events, detail::code_location loc = {})
73  : MType(Type), MArgsStorage(std::move(ArgsStorage)),
74  MAccStorage(std::move(AccStorage)),
75  MSharedPtrStorage(std::move(SharedPtrStorage)),
76  MRequirements(std::move(Requirements)), MEvents(std::move(Events)) {
77  // Capture the user code-location from Q.submit(), Q.parallel_for()
78  // etc for later use; if code location information is not available,
79  // the file name and function name members will be empty strings
80  if (loc.functionName())
81  MFunctionName = loc.functionName();
82  if (loc.fileName())
83  MFileName = loc.fileName();
84  MLine = loc.lineNumber();
85  MColumn = loc.columnNumber();
86  }
87 
88  CG(CG &&CommandGroup) = default;
89 
90  CGTYPE getType() { return MType; }
91 
92  virtual ~CG() = default;
93 
94 private:
95  CGTYPE MType;
96  // The following storages are needed to ensure that arguments won't die while
97  // we are using them.
99  vector_class<vector_class<char>> MArgsStorage;
101  vector_class<detail::AccessorImplPtr> MAccStorage;
103  vector_class<shared_ptr_class<const void>> MSharedPtrStorage;
104 
105 public:
108  vector_class<Requirement *> MRequirements;
110  vector_class<detail::EventImplPtr> MEvents;
111  // Member variables to capture the user code-location
112  // information from Q.submit(), Q.parallel_for() etc
113  // Storage for function name and source file name
114  string_class MFunctionName, MFileName;
115  // Storage for line and column of code location
116  int32_t MLine, MColumn;
117 };
