41 class interop_handler {
42  // Make accessor class friend to access the detail mem objects
43  template <typename DataT, int Dims, access::mode AccMode,
44  access::target AccTarget, access::placeholder isPlaceholder>
45  friend class accessor;
46 
47 public:
48  using QueueImplPtr = std::shared_ptr<detail::queue_impl>;
49  using ReqToMem = std::pair<detail::Requirement *, pi_mem>;
50 
51  interop_handler(std::vector<ReqToMem> MemObjs, QueueImplPtr Queue)
52  : MQueue(std::move(Queue)), MMemObjs(std::move(MemObjs)) {}
53 
54  template <backend BackendName = backend::opencl>
55  auto get_queue() const -> typename interop<BackendName, queue>::type {
56  return reinterpret_cast<typename interop<BackendName, queue>::type>(
57  GetNativeQueue());
58  }
59 
60  template <backend BackendName = backend::opencl, typename DataT, int Dims,
61  access::mode AccessMode, access::target AccessTarget,
62  access::placeholder IsPlaceholder = access::placeholder::false_t>
63  auto get_mem(accessor<DataT, Dims, AccessMode, AccessTarget,
64  access::placeholder::false_t>
65  Acc) const ->
66  typename interop<BackendName,
67  accessor<DataT, Dims, AccessMode, AccessTarget,
68  access::placeholder::false_t>>::type {
69  detail::AccessorBaseHost *AccBase = (detail::AccessorBaseHost *)&Acc;
70  return getMemImpl<BackendName, DataT, Dims, AccessMode, AccessTarget,
71  access::placeholder::false_t>(
72  detail::getSyclObjImpl(*AccBase).get());
73  }
74 
75 private:
76  QueueImplPtr MQueue;
77  std::vector<ReqToMem> MMemObjs;
78 
79  template <backend BackendName, typename DataT, int Dims,
80  access::mode AccessMode, access::target AccessTarget,
81  access::placeholder IsPlaceholder>
82  auto getMemImpl(detail::Requirement *Req) const -> typename interop<
83  BackendName,
84  accessor<DataT, Dims, AccessMode, AccessTarget, IsPlaceholder>>::type {
85  return (typename interop<BackendName,
86  accessor<DataT, Dims, AccessMode, AccessTarget,
87  IsPlaceholder>>::type)GetNativeMem(Req);
88  }

111 // The structure represents NDRange - global, local sizes, global offset and
112 // number of dimensions.
113 class NDRDescT {
114  // The method initializes all sizes for dimensions greater than the passed one
115  // to the default values, so they will not affect execution.
116  void setNDRangeLeftover(int Dims_) {
117  for (int I = Dims_; I < 3; ++I) {
118  GlobalSize[I] = 1;
119  LocalSize[I] = LocalSize[0] ? 1 : 0;
120  GlobalOffset[I] = 0;
121  NumWorkGroups[I] = 0;
122  }
123  }
124 
125 public:
126  NDRDescT()
127  : GlobalSize{0, 0, 0}, LocalSize{0, 0, 0}, NumWorkGroups{0, 0, 0} {}
128 
129  template <int Dims_> void set(sycl::range<Dims_> NumWorkItems) {
130  for (int I = 0; I < Dims_; ++I) {
131  GlobalSize[I] = NumWorkItems[I];
132  LocalSize[I] = 0;
133  GlobalOffset[I] = 0;
134  NumWorkGroups[I] = 0;
135  }
136  setNDRangeLeftover(Dims_);
137  Dims = Dims_;
138  }
139 
140  // Initializes this ND range descriptor with given range of work items and
141  // offset.
142  template <int Dims_>
143  void set(sycl::range<Dims_> NumWorkItems, sycl::id<Dims_> Offset) {
144  for (int I = 0; I < Dims_; ++I) {
145  GlobalSize[I] = NumWorkItems[I];
146  LocalSize[I] = 0;
147  GlobalOffset[I] = Offset[I];
148  NumWorkGroups[I] = 0;
149  }
150  setNDRangeLeftover(Dims_);
151  Dims = Dims_;
152  }
153 
154  template <int Dims_> void set(sycl::nd_range<Dims_> ExecutionRange) {
155  for (int I = 0; I < Dims_; ++I) {
156  GlobalSize[I] = ExecutionRange.get_global_range()[I];
157  LocalSize[I] = ExecutionRange.get_local_range()[I];
158  GlobalOffset[I] = ExecutionRange.get_offset()[I];
159  NumWorkGroups[I] = 0;
160  }
161  setNDRangeLeftover(Dims_);
162  Dims = Dims_;
163  }
164 
165  void set(int Dims_, sycl::nd_range<3> ExecutionRange) {
166  for (int I = 0; I < Dims_; ++I) {
167  GlobalSize[I] = ExecutionRange.get_global_range()[I];
168  LocalSize[I] = ExecutionRange.get_local_range()[I];
169  GlobalOffset[I] = ExecutionRange.get_offset()[I];
170  NumWorkGroups[I] = 0;
171  }
172  setNDRangeLeftover(Dims_);
173  Dims = Dims_;
174  }
175 
176  template <int Dims_> void setNumWorkGroups(sycl::range<Dims_> N) {
177  for (int I = 0; I < Dims_; ++I) {
178  GlobalSize[I] = 0;
179  // '0' is a mark to adjust before kernel launch when there is enough info:
180  LocalSize[I] = 0;
181  GlobalOffset[I] = 0;
182  NumWorkGroups[I] = N[I];
183  }
184  setNDRangeLeftover(Dims_);
185  Dims = Dims_;
186  }
187 
188  sycl::range<3> GlobalSize;
189  sycl::range<3> LocalSize;
190  sycl::id<3> GlobalOffset;
194  sycl::range<3> NumWorkGroups;
195  size_t Dims;
196 };

228 // Class which stores specific lambda object.
229 template <class KernelType, class KernelArgType, int Dims>
230 class HostKernel : public HostKernelBase {
231  using IDBuilder = sycl::detail::Builder;
232  KernelType MKernel;
233 
234 public:
235  HostKernel(KernelType Kernel) : MKernel(Kernel) {}
236  void call(const NDRDescT &NDRDesc, HostProfilingInfo *HPI) override {
237  // adjust ND range for serial host:
238  NDRDescT AdjustedRange = NDRDesc;
239 
240  if (NDRDesc.GlobalSize[0] == 0 && NDRDesc.NumWorkGroups[0] != 0) {
241  // This is a special case - NDRange information is not complete, only the
242  // desired number of work groups is set by the user. Choose work group
243  // size (LocalSize), calculate the missing NDRange characteristics
244  // needed to invoke the kernel and adjust the NDRange descriptor
245  // accordingly. For some devices the work group size selection requires
246  // access to the device's properties, hence such late "adjustment".
247  range<3> WGsize{1, 1, 1}; // no better alternative for serial host?
248  AdjustedRange.set(NDRDesc.Dims,
249  nd_range<3>(NDRDesc.NumWorkGroups * WGsize, WGsize));
250  }
251  // If local size for host is not set explicitly, let's adjust it to 1,
252  // so nd_range_error for zero local size is not thrown.
253  if (AdjustedRange.LocalSize[0] == 0)
254  for (size_t I = 0; I < AdjustedRange.Dims; ++I)
255  AdjustedRange.LocalSize[I] = 1;
256  if (HPI)
257  HPI->start();
258  runOnHost(AdjustedRange);
259  if (HPI)
260  HPI->end();
261  }
262 
263  char *getPtr() override { return reinterpret_cast<char *>(&MKernel); }
264 
265  template <class ArgT = KernelArgType>
266  typename std::enable_if<std::is_same<ArgT, void>::value>::type
267  runOnHost(const NDRDescT &) {
268  MKernel();
269  }
270 
271  template <class ArgT = KernelArgType>
272  typename std::enable_if<std::is_same<ArgT, sycl::id<Dims>>::value>::type
273  runOnHost(const NDRDescT &NDRDesc) {
274  sycl::range<Dims> Range(InitializedVal<Dims, range>::template get<0>());
275  for (int I = 0; I < Dims; ++I)
276  Range[I] = NDRDesc.GlobalSize[I];
277 
278  detail::NDLoop<Dims>::iterate(
279  Range, [&](const sycl::id<Dims> &ID) { MKernel(ID); });
280  }
281 
282  template <class ArgT = KernelArgType>
283  typename std::enable_if<
284  std::is_same<ArgT, item<Dims, /*Offset=*/false>>::value>::type
285  runOnHost(const NDRDescT &NDRDesc) {
286  sycl::id<Dims> ID;
287  sycl::range<Dims> Range(InitializedVal<Dims, range>::template get<0>());
288  for (int I = 0; I < Dims; ++I)
289  Range[I] = NDRDesc.GlobalSize[I];
290 
291  detail::NDLoop<Dims>::iterate(Range, [&](const sycl::id<Dims> ID) {
292  sycl::item<Dims, /*Offset=*/false> Item =
293  IDBuilder::createItem<Dims, false>(Range, ID);
294  MKernel(Item);
295  });
296  }
297 
298  template <class ArgT = KernelArgType>
299  typename std::enable_if<
300  std::is_same<ArgT, item<Dims, /*Offset=*/true>>::value>::type
301  runOnHost(const NDRDescT &NDRDesc) {
302  sycl::range<Dims> Range(InitializedVal<Dims, range>::template get<0>());
303  sycl::id<Dims> Offset;
304  for (int I = 0; I < Dims; ++I) {
305  Range[I] = NDRDesc.GlobalSize[I];
306  Offset[I] = NDRDesc.GlobalOffset[I];
307  }
308 
309  detail::NDLoop<Dims>::iterate(Range, [&](const sycl::id<Dims> &ID) {
310  sycl::id<Dims> OffsetID = ID + Offset;
311  sycl::item<Dims, /*Offset=*/true> Item =
312  IDBuilder::createItem<Dims, true>(Range, OffsetID, Offset);
313  MKernel(Item);
314  });
315  }
316 
317  template <class ArgT = KernelArgType>
318  typename std::enable_if<std::is_same<ArgT, nd_item<Dims>>::value>::type
319  runOnHost(const NDRDescT &NDRDesc) {
320  sycl::range<Dims> GroupSize(InitializedVal<Dims, range>::template get<0>());
321  for (int I = 0; I < Dims; ++I) {
322  if (NDRDesc.LocalSize[I] == 0 ||
323  NDRDesc.GlobalSize[I] % NDRDesc.LocalSize[I] != 0)
324  throw sycl::nd_range_error("Invalid local size for global size",
325  PI_INVALID_WORK_GROUP_SIZE);
326  GroupSize[I] = NDRDesc.GlobalSize[I] / NDRDesc.LocalSize[I];
327  }
328 
329  sycl::range<Dims> LocalSize(InitializedVal<Dims, range>::template get<0>());
330  sycl::range<Dims> GlobalSize(
331  InitializedVal<Dims, range>::template get<0>());
332  sycl::id<Dims> GlobalOffset;
333  for (int I = 0; I < Dims; ++I) {
334  GlobalOffset[I] = NDRDesc.GlobalOffset[I];
335  LocalSize[I] = NDRDesc.LocalSize[I];
336  GlobalSize[I] = NDRDesc.GlobalSize[I];
337  }
338 
339  detail::NDLoop<Dims>::iterate(GroupSize, [&](const id<Dims> &GroupID) {
340  sycl::group<Dims> Group = IDBuilder::createGroup<Dims>(
341  GlobalSize, LocalSize, GroupSize, GroupID);
342 
343  detail::NDLoop<Dims>::iterate(LocalSize, [&](const id<Dims> &LocalID) {
344  id<Dims> GlobalID = GroupID * LocalSize + LocalID + GlobalOffset;
345  const sycl::item<Dims, /*Offset=*/true> GlobalItem =
346  IDBuilder::createItem<Dims, true>(GlobalSize, GlobalID,
347  GlobalOffset);
348  const sycl::item<Dims, /*Offset=*/false> LocalItem =
349  IDBuilder::createItem<Dims, false>(LocalSize, LocalID);
350  const sycl::nd_item<Dims> NDItem =
351  IDBuilder::createNDItem<Dims>(GlobalItem, LocalItem, Group);
352  MKernel(NDItem);
353  });
354  });
355  }
356 
357  template <typename ArgT = KernelArgType>
358  enable_if_t<std::is_same<ArgT, cl::sycl::group<Dims>>::value>
359  runOnHost(const NDRDescT &NDRDesc) {
360  sycl::range<Dims> NGroups(InitializedVal<Dims, range>::template get<0>());
361 
362  for (int I = 0; I < Dims; ++I) {
363  if (NDRDesc.LocalSize[I] == 0 ||
364  NDRDesc.GlobalSize[I] % NDRDesc.LocalSize[I] != 0)
365  throw sycl::nd_range_error("Invalid local size for global size",
366  PI_INVALID_WORK_GROUP_SIZE);
367  NGroups[I] = NDRDesc.GlobalSize[I] / NDRDesc.LocalSize[I];
368  }
369 
370  sycl::range<Dims> LocalSize(InitializedVal<Dims, range>::template get<0>());
371  sycl::range<Dims> GlobalSize(
372  InitializedVal<Dims, range>::template get<0>());
373  for (int I = 0; I < Dims; ++I) {
374  LocalSize[I] = NDRDesc.LocalSize[I];
375  GlobalSize[I] = NDRDesc.GlobalSize[I];
376  }
377  detail::NDLoop<Dims>::iterate(NGroups, [&](const id<Dims> &GroupID) {
378  sycl::group<Dims> Group =
379  IDBuilder::createGroup<Dims>(GlobalSize, LocalSize, NGroups, GroupID);
380  MKernel(Group);
381  });
382  }
383 
384  ~HostKernel() = default;
385 };
