49 template <typename RT> struct satur {
50  template <typename T> static RT saturate(const T val, const int flags) {
51  if ((flags & sat_is_on) == 0) {
52  return (RT)val;
53  }
54 
55  // min/max can be macros on Windows, so wrap them into parens to avoid their
56  // expansion
57  const RT t_max = (std::numeric_limits<RT>::max)();
58  const RT t_min = (std::numeric_limits<RT>::min)();
59 
60  if (val > t_max) {
61  return t_max;
62  } else if ((val >= 0) && (t_min < 0)) {
63  // RT is "signed" if t_min < 0
64  // when comparing a signed and a unsigned variable, the signed one cast to
65  // unsigned first.
66  return (RT)val;
67  } else if (val < t_min) {
68  return t_min;
69  } else {
70  return (RT)val;
71  }
72  }
73 };

75 template <> struct satur<float> {
76  template <typename T> static float saturate(const T val, const int flags) {
77  if ((flags & sat_is_on) == 0) {
78  return (float)val;
79  }
80 
81  if (val < 0.) {
82  return 0;
83  } else if (val > 1.) {
84  return 1.;
85  } else {
