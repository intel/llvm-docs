116 void SYCLMemObjT::determineHostPtr(const ContextImplPtr &Context,
117  bool InitFromUserData, void *&HostPtr,
118  bool &HostPtrReadOnly) {
119  // The data for the allocation can be provided via either the user pointer
120  // (InitFromUserData, can be read-only) or a runtime-allocated read-write
121  // HostPtr. We can have one of these scenarios:
122  // 1. The allocation is the first one and on host. InitFromUserData == true.
123  // 2. The allocation is the first one and isn't on host. InitFromUserData
124  // varies based on unified host memory support and whether or not the data can
125  // be discarded.
126  // 3. The allocation is not the first one and is on host. InitFromUserData ==
127  // false, HostPtr == nullptr. This can only happen if the allocation command
128  // is not linked since it would be a no-op otherwise. Attempt to reuse the
129  // user pointer if it's read-write, but do not copy its contents if it's not.
130  // 4. The allocation is not the first one and not on host. InitFromUserData ==
131  // false, HostPtr is provided if the command is linked. The host pointer is
132  // guaranteed to be reused in this case.
133  if (Context->is_host() && !MOpenCLInterop && !MHostPtrReadOnly)
134  InitFromUserData = true;
135 
136  if (InitFromUserData) {
137  assert(!HostPtr && "Cannot init from user data and reuse host ptr provided "
138  "simultaneously");
139  HostPtr = getUserPtr();
140  HostPtrReadOnly = MHostPtrReadOnly;
141  } else
142  HostPtrReadOnly = false;
143 }
