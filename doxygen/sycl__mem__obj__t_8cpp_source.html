19 SYCLMemObjT::SYCLMemObjT(cl_mem MemObject, const context &SyclContext,
20  const size_t SizeInBytes, event AvailableEvent,
21  unique_ptr_class<SYCLMemObjAllocator> Allocator)
22  : MAllocator(std::move(Allocator)), MProps(),
23  MInteropEvent(detail::getSyclObjImpl(std::move(AvailableEvent))),
24  MInteropContext(detail::getSyclObjImpl(SyclContext)),
25  MInteropMemObject(MemObject), MOpenCLInterop(true),
26  MHostPtrReadOnly(false), MNeedWriteBack(true), MSizeInBytes(SizeInBytes),
27  MUserPtr(nullptr), MShadowCopy(nullptr), MUploadDataFunctor(nullptr),
28  MSharedPtrStorage(nullptr) {
29  if (MInteropContext->is_host())
30  throw cl::sycl::invalid_parameter_error(
31  "Creation of interoperability memory object using host context is "
32  "not allowed",
33  PI_INVALID_CONTEXT);
34 
35  RT::PiMem Mem = pi::cast<RT::PiMem>(MInteropMemObject);
36  RT::PiContext Context = nullptr;
37  const plugin &Plugin = getPlugin();
38  Plugin.call<PiApiKind::piMemGetInfo>(Mem, CL_MEM_CONTEXT, sizeof(Context),
39  &Context, nullptr);
40 
41  if (MInteropContext->getHandleRef() != Context)
42  throw cl::sycl::invalid_parameter_error(
43  "Input context must be the same as the context of cl_mem",
44  PI_INVALID_CONTEXT);
45  Plugin.call<PiApiKind::piMemRetain>(Mem);
46 }

53 void SYCLMemObjT::updateHostMemory(void *const Ptr) {
54  const id<3> Offset{0, 0, 0};
55  const range<3> AccessRange{MSizeInBytes, 1, 1};
56  const range<3> MemoryRange{MSizeInBytes, 1, 1};
57  const access::mode AccessMode = access::mode::read;
58  SYCLMemObjI *SYCLMemObject = this;
59  const int Dims = 1;
60  const int ElemSize = 1;
61 
62  Requirement Req(Offset, AccessRange, MemoryRange, AccessMode, SYCLMemObject,
63  Dims, ElemSize);
64  Req.MData = Ptr;
65 
66  EventImplPtr Event = Scheduler::getInstance().addCopyBack(&Req);
67  if (Event)
68  Event->wait(Event);
69 }

71 void SYCLMemObjT::updateHostMemory() {
72  if ((MUploadDataFunctor != nullptr) && MNeedWriteBack)
73  MUploadDataFunctor();
74 
75  // If we're attached to a memory record, process the deletion of the memory
76  // record. We may get detached before we do this.
77  if (MRecord)
78  Scheduler::getInstance().removeMemoryObject(this);
79  releaseHostMem(MShadowCopy);
80 
81  if (MOpenCLInterop) {
82  const plugin &Plugin = getPlugin();
83  Plugin.call<PiApiKind::piMemRelease>(
84  pi::cast<RT::PiMem>(MInteropMemObject));
85  }
