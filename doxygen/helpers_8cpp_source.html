43  Queue->getDeviceImplPtr()->getBackend() == backend::ext_oneapi_hip;
44  if (isNvidia || isHIP) {
45  auto KernelID = ProgramManager::getInstance().getSYCLKernelID(KernelName);
46  std::vector<kernel_id> KernelIds{KernelID};

51  [isNvidia](RTDeviceBinaryImage *DI) {
52  const std::string &TargetSpec = isNvidia ? std::string("llvm_nvptx64")
53  : std::string("llvm_amdgcn");
54  return DI->getFormat() == SYCL_DEVICE_BINARY_TYPE_LLVMIR_BITCODE &&
55  DI->getRawData().DeviceTargetSpec == TargetSpec;
56  });
57  if (DeviceImage == DeviceImages.end()) {
58  return {nullptr, nullptr};
59  }
60  auto ContextImpl = Queue->getContextImplPtr();
61  auto Context = detail::createSyclObjFromImpl<context>(ContextImpl);

63  auto Device = detail::createSyclObjFromImpl<device>(DeviceImpl);
64  ur_program_handle_t Program =

66  Context, Device);
67  return {*DeviceImage, Program};
68  }
