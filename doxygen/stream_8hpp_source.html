169  // Set the flags which correspond to the input stream manipulator.
170  void set_manipulator(const stream_manipulator SM) const {
171  switch (SM) {
172  case stream_manipulator::dec:
173  set_flag(detail::Dec, detail::BaseField);
174  break;
175  case stream_manipulator::hex:
176  set_flag(detail::Hex, detail::BaseField);
177  break;
178  case stream_manipulator::oct:
179  set_flag(detail::Oct, detail::BaseField);
180  break;
181  case stream_manipulator::noshowbase:
182  unset_flag(detail::ShowBase);
183  break;
184  case stream_manipulator::showbase:
185  set_flag(detail::ShowBase);
186  break;
187  case stream_manipulator::noshowpos:
188  unset_flag(detail::ShowPos);
189  break;
190  case stream_manipulator::showpos:
191  set_flag(detail::ShowPos);
192  break;
193  case stream_manipulator::fixed:
194  set_flag(detail::Fixed, detail::FloatField);
195  break;
196  case stream_manipulator::scientific:
197  set_flag(detail::Scientific, detail::FloatField);
198  break;
199  case stream_manipulator::hexfloat:
200  set_flag(detail::Fixed | detail::Scientific, detail::FloatField);
201  break;
202  case stream_manipulator::defaultfloat:
203  unset_flag(detail::FloatField);
204  break;
205  default:
206  // Unknown manipulator
207  break;
208  }

211 #ifdef __SYCL_DEVICE_ONLY__
212  void __init() {
213  // Calculate work item's global id, this should be done once, that
214  // is why this is done in _init method, call to __init method is generated
215  // by frontend. As a result each work item will write to its own section
216  // in the flush buffer
217 
218  id<1> GlobalId = __spirv::initGlobalInvocationId<1, id<1>>();
219  WIOffset = (unsigned int)GlobalId[0] * FlushBufferSize;
220  }
221 
222  void __finalize() {
223  // Flush data to global buffer if flush buffer is not empty. This could be
224  // necessary if user hasn't yet flushed data on its own and kernel execution
225  // is finished
226  // NOTE: A call to this function will be generated by compiler
227  // NOTE: In the current implementation user should explicitly flush data on
228  // the host device. Data is not flushed automatically after kernel execution
229  // because of the missing feature in scheduler.
230  if (Offset) {
231  flushBuffer(GlobalOffset, GlobalBuf, GlobalFlushBuf, WIOffset, Offset);
232  }
233  }
234 #endif

286 // Character
287 inline const stream &operator<<(const stream &Out, const char C) {
288  if (Out.Offset >= Out.FlushBufferSize ||
289  Out.WIOffset + Out.Offset + 1 > Out.GlobalFlushBuf.get_count())
290  return Out;
291  Out.GlobalFlushBuf[Out.WIOffset + Out.Offset] = C;
292  ++Out.Offset;
293  return Out;
294 }

381  const stream_manipulator RHS) {
382  switch (RHS) {
383  case stream_manipulator::endl:
384  Out << '\n';
385  flushBuffer(Out.GlobalOffset, Out.GlobalBuf, Out.GlobalFlushBuf,
