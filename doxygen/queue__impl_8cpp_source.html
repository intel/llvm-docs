110 const std::vector<event> &
111 queue_impl::getExtendDependencyList(const std::vector<event> &DepEvents,
112  std::vector<event> &MutableVec,
113  std::unique_lock<std::mutex> &QueueLock) {
114  if (!isInOrder())
115  return DepEvents;
116 
117  QueueLock.lock();
118  EventImplPtr ExtraEvent =
119  MGraph.expired() ? MLastEventPtr : MGraphLastEventPtr;
120  std::optional<event> ExternalEvent = popExternalEvent();
121 
122  if (!ExternalEvent && !ExtraEvent)
123  return DepEvents;
124 
125  MutableVec = DepEvents;
126  if (ExternalEvent)
127  MutableVec.push_back(*ExternalEvent);
128  if (ExtraEvent)
129  MutableVec.push_back(detail::createSyclObjFromImpl<event>(ExtraEvent));
130  return MutableVec;
131 }

262 event queue_impl::getLastEvent() {
263  {
264  // The external event is required to finish last if set, so it is considered
265  // the last event if present.

281 void queue_impl::addEvent(const event &Event) {
282  EventImplPtr EImpl = getSyclObjImpl(Event);
283  assert(EImpl && "Event implementation is missing");
284  auto *Cmd = static_cast<Command *>(EImpl->getCommand());
285  if (!Cmd) {
286  // if there is no command on the event, we cannot track it with MEventsWeak
287  // as that will leave it with no owner. Track in MEventsShared only if we're
288  // unable to call piQueueFinish during wait.
289  if (is_host() || MEmulateOOO)
290  addSharedEvent(Event);
291  }
292  // As long as the queue supports piQueueFinish we only need to store events
293  // for unenqueued commands and host tasks.
294  else if (is_host() || MEmulateOOO || EImpl->getHandleRef() == nullptr) {
295  std::weak_ptr<event_impl> EventWeakPtr{EImpl};
296  std::lock_guard<std::mutex> Lock{MMutex};
297  MEventsWeak.push_back(std::move(EventWeakPtr));
298  }
299 }

304 void queue_impl::addSharedEvent(const event &Event) {
305  assert(is_host() || MEmulateOOO);
306  std::lock_guard<std::mutex> Lock(MMutex);
307  // Events stored in MEventsShared are not released anywhere else aside from
308  // calls to queue::wait/wait_and_throw, which a user application might not
309  // make, and ~queue_impl(). If the number of events grows large enough,
310  // there's a good chance that most of them are already completed and ownership
311  // of them can be released.
312  const size_t EventThreshold = 128;
313  if (MEventsShared.size() >= EventThreshold) {
314  // Generally, the vector is ordered so that the oldest events are in the
315  // front and the newer events are in the end. So, search to find the first
316  // event that isn't yet complete. All the events prior to that can be
317  // erased. This could leave some few events further on that have completed
318  // not yet erased, but that is OK. This cleanup doesn't have to be perfect.
319  // This also keeps the algorithm linear rather than quadratic because it
320  // doesn't continually recheck things towards the back of the list that
321  // really haven't had time to complete.
322  MEventsShared.erase(
323  MEventsShared.begin(),
324  std::find_if(
325  MEventsShared.begin(), MEventsShared.end(), [](const event &E) {
326  return E.get_info<info::event::command_execution_status>() !=
327  info::event_command_status::complete;
328  }));
329  }
330  MEventsShared.push_back(Event);
331 }

345 template <typename HandlerFuncT, typename MemOpFuncT, typename... MemOpArgTs>
346 event queue_impl::submitMemOpHelper(const std::shared_ptr<queue_impl> &Self,
347  const std::vector<event> &DepEvents,
348  HandlerFuncT HandlerFunc,
349  MemOpFuncT MemOpFunc,
350  MemOpArgTs... MemOpArgs) {
351  // We need to submit command and update the last event under same lock if we
352  // have in-order queue.
353  {
354  std::unique_lock<std::mutex> Lock(MMutex, std::defer_lock);
355 
356  std::vector<event> MutableDepEvents;
357  const std::vector<event> &ExpandedDepEvents =
358  getExtendDependencyList(DepEvents, MutableDepEvents, Lock);
359 
360  // If we have a command graph set we need to capture the op through the
361  // handler rather than by-passing the scheduler.
362  if (MGraph.expired() && Scheduler::areEventsSafeForSchedulerBypass(
363  ExpandedDepEvents, MContext)) {
364  if (MSupportsDiscardingPiEvents) {
365  MemOpFunc(MemOpArgs..., getPIEvents(ExpandedDepEvents),
366  /*PiEvent*/ nullptr, /*EventImplPtr*/ nullptr);
367  return createDiscardedEvent();
368  }
369 
370  event ResEvent = prepareSYCLEventAssociatedWithQueue(Self);
371  auto EventImpl = detail::getSyclObjImpl(ResEvent);
372  MemOpFunc(MemOpArgs..., getPIEvents(ExpandedDepEvents),
373  &EventImpl->getHandleRef(), EventImpl);
374 
375  if (MContext->is_host())
376  return MDiscardEvents ? createDiscardedEvent() : event();
377 
378  if (isInOrder()) {
379  auto &EventToStoreIn =
380  MGraph.expired() ? MLastEventPtr : MGraphLastEventPtr;
381  EventToStoreIn = EventImpl;
382  }
383  // Track only if we won't be able to handle it with piQueueFinish.
384  if (MEmulateOOO)
385  addSharedEvent(ResEvent);
386  return discard_or_return(ResEvent);
387  }
388  }
389  return submitWithHandler(Self, DepEvents, HandlerFunc);
390 }

392 void *queue_impl::instrumentationProlog(const detail::code_location &CodeLoc,
393  std::string &Name, int32_t StreamID,
394  uint64_t &IId) {
395  void *TraceEvent = nullptr;
396  (void)CodeLoc;
397  (void)Name;
398  (void)StreamID;
399  (void)IId;
400 #ifdef XPTI_ENABLE_INSTRUMENTATION
401  constexpr uint16_t NotificationTraceType = xpti::trace_wait_begin;
402  if (!xptiCheckTraceEnabled(StreamID, NotificationTraceType))
403  return TraceEvent;
404 
405  xpti::payload_t Payload;
406  bool HasSourceInfo = false;
407  // We try to create a unique string for the wait() call by combining it with
408  // the queue address
409  xpti::utils::StringHelper NG;
410  Name = NG.nameWithAddress<queue_impl *>("queue.wait", this);
411 
412  if (CodeLoc.fileName()) {
413  // We have source code location information
414  Payload =
415  xpti::payload_t(Name.c_str(), CodeLoc.fileName(), CodeLoc.lineNumber(),
416  CodeLoc.columnNumber(), (void *)this);
417  HasSourceInfo = true;
418  } else {
419  // We have no location information, so we'll use the address of the queue
420  Payload = xpti::payload_t(Name.c_str(), (void *)this);
421  }
422  // wait() calls could be at different user-code locations; We create a new
423  // event based on the code location info and if this has been seen before, a
424  // previously created event will be returned.
425  uint64_t QWaitInstanceNo = 0;
426  xpti::trace_event_data_t *WaitEvent =
427  xptiMakeEvent(Name.c_str(), &Payload, xpti::trace_graph_event,
428  xpti_at::active, &QWaitInstanceNo);
429  IId = QWaitInstanceNo;
430  if (WaitEvent) {
431  device D = get_device();
432  std::string DevStr;
433  if (getSyclObjImpl(D)->is_host())
434  DevStr = "HOST";
435  else if (D.is_cpu())
436  DevStr = "CPU";
437  else if (D.is_gpu())
438  DevStr = "GPU";
439  else if (D.is_accelerator())
440  DevStr = "ACCELERATOR";
441  else
442  DevStr = "UNKNOWN";
443  xpti::addMetadata(WaitEvent, "sycl_device_type", DevStr);
444  if (HasSourceInfo) {
445  xpti::addMetadata(WaitEvent, "sym_function_name", CodeLoc.functionName());
446  xpti::addMetadata(WaitEvent, "sym_source_file_name", CodeLoc.fileName());
447  xpti::addMetadata(WaitEvent, "sym_line_no",
448  static_cast<int32_t>((CodeLoc.lineNumber())));
449  xpti::addMetadata(WaitEvent, "sym_column_no",
450  static_cast<int32_t>((CodeLoc.columnNumber())));
451  }
452  xptiNotifySubscribers(StreamID, xpti::trace_wait_begin, nullptr, WaitEvent,
453  QWaitInstanceNo,
454  static_cast<const void *>(Name.c_str()));
455  TraceEvent = (void *)WaitEvent;
456  }
457 #endif
458  return TraceEvent;
459 }

480 void queue_impl::wait(const detail::code_location &CodeLoc) {
481  (void)CodeLoc;
482 #ifdef XPTI_ENABLE_INSTRUMENTATION
483  void *TelemetryEvent = nullptr;
484  uint64_t IId;
485  std::string Name;
486  int32_t StreamID = xptiRegisterStream(SYCL_STREAM_NAME);
487  TelemetryEvent = instrumentationProlog(CodeLoc, Name, StreamID, IId);
488 #endif
489 
490  if (MGraph.lock()) {
491  throw sycl::exception(make_error_code(errc::invalid),
492  "wait cannot be called for a queue which is "
493  "recording to a command graph.");
494  }
495 
496  // If there is an external event set, we know we are using an in-order queue
497  // and the event is required to finish after the last event in the queue. As
498  // such, we can just wait for it and finish.
499  std::optional<event> ExternalEvent = popExternalEvent();
500  if (ExternalEvent) {
501  ExternalEvent->wait();
502 
503  // Additionally, we can clean up the event lists that we would have
504  // otherwise cleared.
505  if (!MEventsWeak.empty() || !MEventsShared.empty()) {
506  std::lock_guard<std::mutex> Lock(MMutex);
507  MEventsWeak.clear();
508  MEventsShared.clear();
509  }
510  if (!MStreamsServiceEvents.empty()) {
511  std::lock_guard<std::mutex> Lock(MStreamsServiceEventsMutex);
512  MStreamsServiceEvents.clear();
513  }
514  }
515 
516  std::vector<std::weak_ptr<event_impl>> WeakEvents;
517  std::vector<event> SharedEvents;
518  {
519  std::lock_guard<std::mutex> Lock(MMutex);
520  WeakEvents.swap(MEventsWeak);
521  SharedEvents.swap(MEventsShared);
522  }
523  // If the queue is either a host one or does not support OOO (and we use
524  // multiple in-order queues as a result of that), wait for each event
525  // directly. Otherwise, only wait for unenqueued or host task events, starting
526  // from the latest submitted task in order to minimize total amount of calls,
527  // then handle the rest with piQueueFinish.
528  const bool SupportsPiFinish = !is_host() && !MEmulateOOO;
529  for (auto EventImplWeakPtrIt = WeakEvents.rbegin();
530  EventImplWeakPtrIt != WeakEvents.rend(); ++EventImplWeakPtrIt) {
531  if (std::shared_ptr<event_impl> EventImplSharedPtr =
532  EventImplWeakPtrIt->lock()) {
533  // A nullptr PI event indicates that piQueueFinish will not cover it,
534  // either because it's a host task event or an unenqueued one.
535  if (!SupportsPiFinish || nullptr == EventImplSharedPtr->getHandleRef()) {
536  EventImplSharedPtr->wait(EventImplSharedPtr);
537  }
538  }
539  }
540  if (SupportsPiFinish) {
541  const PluginPtr &Plugin = getPlugin();
542  Plugin->call<detail::PiApiKind::piQueueFinish>(getHandleRef());
543  assert(SharedEvents.empty() && "Queues that support calling piQueueFinish "
544  "shouldn't have shared events");
545  } else {
546  for (event &Event : SharedEvents)
547  Event.wait();
548  }
549 
550  std::vector<EventImplPtr> StreamsServiceEvents;
551  {
552  std::lock_guard<std::mutex> Lock(MStreamsServiceEventsMutex);
553  StreamsServiceEvents.swap(MStreamsServiceEvents);
554  }
555  for (const EventImplPtr &Event : StreamsServiceEvents)
556  Event->wait(Event);
557 
558 #ifdef XPTI_ENABLE_INSTRUMENTATION
559  instrumentationEpilog(TelemetryEvent, Name, StreamID, IId);
560 #endif
561 }

579 bool queue_impl::ext_oneapi_empty() const {
580  // If we have in-order queue where events are not discarded then just check
581  // the status of the last event.
582  if (isInOrder() && !MDiscardEvents) {
583  std::lock_guard<std::mutex> Lock(MMutex);
584  return !MLastEventPtr ||
585  MLastEventPtr->get_info<info::event::command_execution_status>() ==
586  info::event_command_status::complete;
587  }
588 
589  // Check the status of the backend queue if this is not a host queue.
590  if (!is_host()) {
591  pi_bool IsReady = false;
592  getPlugin()->call<PiApiKind::piQueueGetInfo>(
593  MQueues[0], PI_EXT_ONEAPI_QUEUE_INFO_EMPTY, sizeof(pi_bool), &IsReady,
594  nullptr);
595  if (!IsReady)
596  return false;
597  }
598 
599  // We may have events like host tasks which are not submitted to the backend
600  // queue so we need to get their status separately.
601  std::lock_guard<std::mutex> Lock(MMutex);
602  for (event Event : MEventsShared)
603  if (Event.get_info<info::event::command_execution_status>() !=
604  info::event_command_status::complete)
605  return false;
606 
607  for (auto EventImplWeakPtrIt = MEventsWeak.begin();
608  EventImplWeakPtrIt != MEventsWeak.end(); ++EventImplWeakPtrIt)
609  if (std::shared_ptr<event_impl> EventImplSharedPtr =
610  EventImplWeakPtrIt->lock())
611  if (EventImplSharedPtr->is_host() &&
612  EventImplSharedPtr
613  ->get_info<info::event::command_execution_status>() !=
614  info::event_command_status::complete)
615  return false;
616 
617  // If we didn't exit early above then it means that all events in the queue
618  // are completed.
619  return true;
620 }

623  if (!(MDiscardEvents))
624  return Event;
625  return createDiscardedEvent();
626 }
