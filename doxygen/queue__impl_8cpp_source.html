94 void *queue_impl::instrumentationProlog(const detail::code_location &CodeLoc,
95  string_class &Name, int32_t StreamID,
96  uint64_t &IId) {
97  void *TraceEvent = nullptr;
98 #ifdef XPTI_ENABLE_INSTRUMENTATION
99  xpti::trace_event_data_t *WaitEvent = nullptr;
100  if (!xptiTraceEnabled())
101  return TraceEvent;
102 
103  xpti::payload_t Payload;
104  bool HasSourceInfo = false;
105  // We try to create a unique string for the wait() call by combining it with
106  // the queue address
107  xpti::utils::StringHelper NG;
108  Name = NG.nameWithAddress<queue_impl *>("queue.wait", this);
109 
110  if (!CodeLoc.fileName()) {
111  // We have source code location information
112  Payload =
113  xpti::payload_t(Name.c_str(), CodeLoc.fileName(), CodeLoc.lineNumber(),
114  CodeLoc.columnNumber(), (void *)this);
115  HasSourceInfo = true;
116  } else {
117  // We have no location information, so we'll use the address of the queue
118  Payload = xpti::payload_t(Name.c_str(), (void *)this);
119  }
120  // wait() calls could be at different user-code locations; We create a new
121  // event based on the code location info and if this has been seen before, a
122  // previously created event will be returned.
123  uint64_t QWaitInstanceNo = 0;
124  WaitEvent = xptiMakeEvent(Name.c_str(), &Payload, xpti::trace_graph_event,
125  xpti_at::active, &QWaitInstanceNo);
126  IId = QWaitInstanceNo;
127  if (WaitEvent) {
128  device D = get_device();
129  std::string DevStr;
130  if (D.is_host())
131  DevStr = "HOST";
132  else if (D.is_cpu())
133  DevStr = "CPU";
134  else if (D.is_gpu())
135  DevStr = "GPU";
136  else if (D.is_accelerator())
137  DevStr = "ACCELERATOR";
138  else
139  DevStr = "UNKNOWN";
140  xptiAddMetadata(WaitEvent, "sycl_device", DevStr.c_str());
141  if (HasSourceInfo) {
142  xptiAddMetadata(WaitEvent, "sym_function_name", CodeLoc.functionName());
143  xptiAddMetadata(WaitEvent, "sym_source_file_name", CodeLoc.fileName());
144  xptiAddMetadata(WaitEvent, "sym_line_no",
145  std::to_string(CodeLoc.lineNumber()).c_str());
146  }
147  xptiNotifySubscribers(StreamID, xpti::trace_wait_begin, nullptr, WaitEvent,
148  QWaitInstanceNo,
149  static_cast<const void *>(Name.c_str()));
150  TraceEvent = (void *)WaitEvent;
151  }
152 #endif
153  return TraceEvent;
154 }

156 void queue_impl::instrumentationEpilog(void *TelemetryEvent, string_class &Name,
157  int32_t StreamID, uint64_t IId) {
158 #ifdef XPTI_ENABLE_INSTRUMENTATION
159  if (!(xptiTraceEnabled() && TelemetryEvent))
160  return;
161  // Close the wait() scope
162  xpti::trace_event_data_t *TraceEvent =
163  (xpti::trace_event_data_t *)TelemetryEvent;
164  xptiNotifySubscribers(StreamID, xpti::trace_wait_end, nullptr, TraceEvent,
165  IId, static_cast<const void *>(Name.c_str()));
166 #endif

169 void queue_impl::wait(const detail::code_location &CodeLoc) {
170 #ifdef XPTI_ENABLE_INSTRUMENTATION
171  void *TelemetryEvent = nullptr;
172  uint64_t IId;
173  std::string Name;
174  int32_t StreamID = xptiRegisterStream(SYCL_STREAM_NAME);
175  TelemetryEvent = instrumentationProlog(CodeLoc, Name, StreamID, IId);
176 #endif
177 
178  std::lock_guard<mutex_class> Guard(MMutex);
179  for (auto &Event : MEvents)
180  Event.wait();
181  MEvents.clear();
182 
183 #ifdef XPTI_ENABLE_INSTRUMENTATION
184  instrumentationEpilog(TelemetryEvent, Name, StreamID, IId);
185 #endif
186 }
