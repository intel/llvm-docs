41  // The method initializes all sizes for dimensions greater than the passed one
42  // to the default values, so they will not affect execution.
43  void setNDRangeLeftover(int Dims_) {
44  for (int I = Dims_; I < 3; ++I) {
45  GlobalSize[I] = 1;
46  LocalSize[I] = LocalSize[0] ? 1 : 0;
47  GlobalOffset[I] = 0;

160 // Class which stores specific lambda object.
161 template <class KernelType, class KernelArgType, int Dims>
162 class HostKernel : public HostKernelBase {
163  using IDBuilder = sycl::detail::Builder;
164  KernelType MKernel;
165 
166 public:
167  HostKernel(KernelType Kernel) : MKernel(Kernel) {}
168  void call(const NDRDescT &NDRDesc, HostProfilingInfo *HPI) override {
169  // adjust ND range for serial host:
170  NDRDescT AdjustedRange = NDRDesc;
171 
172  if (NDRDesc.GlobalSize[0] == 0 && NDRDesc.NumWorkGroups[0] != 0) {
173  // This is a special case - NDRange information is not complete, only the
174  // desired number of work groups is set by the user. Choose work group
175  // size (LocalSize), calculate the missing NDRange characteristics
176  // needed to invoke the kernel and adjust the NDRange descriptor
177  // accordingly. For some devices the work group size selection requires
178  // access to the device's properties, hence such late "adjustment".
179  range<3> WGsize{1, 1, 1}; // no better alternative for serial host?
180  AdjustedRange.set(NDRDesc.Dims,
181  nd_range<3>(NDRDesc.NumWorkGroups * WGsize, WGsize));
182  }
183  // If local size for host is not set explicitly, let's adjust it to 1,
184  // so nd_range_error for zero local size is not thrown.
185  if (AdjustedRange.LocalSize[0] == 0)
186  for (size_t I = 0; I < AdjustedRange.Dims; ++I)
187  AdjustedRange.LocalSize[I] = 1;
188  if (HPI)
189  HPI->start();
190  runOnHost(AdjustedRange);
191  if (HPI)
192  HPI->end();
193  }
194 
195  char *getPtr() override { return reinterpret_cast<char *>(&MKernel); }

251  runOnHost(const NDRDescT &NDRDesc) {
252  sycl::range<Dims> GroupSize(InitializedVal<Dims, range>::template get<0>());
253  for (int I = 0; I < Dims; ++I) {
254  if (NDRDesc.LocalSize[I] == 0 ||
255  NDRDesc.GlobalSize[I] % NDRDesc.LocalSize[I] != 0)
256  throw sycl::nd_range_error("Invalid local size for global size",
257  PI_INVALID_WORK_GROUP_SIZE);

259  }
