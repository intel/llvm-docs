71 template <typename, typename T> struct check_fn_signature {
72  static_assert(std::integral_constant<T, false>::value,
73  "Second template parameter is required to be of function type");
74 };

76 template <typename F, typename RetT, typename... Args>
77 struct check_fn_signature<F, RetT(Args...)> {
78 private:
79  template <typename T>
80  static constexpr auto check(T *) -> typename std::is_same<
81  decltype(std::declval<T>().operator()(std::declval<Args>()...)),
82  RetT>::type;
83 
84  template <typename> static constexpr std::false_type check(...);
85 
86  using type = decltype(check<F>(0));
87 
88 public:
89  static constexpr bool value = type::value;
90 };
