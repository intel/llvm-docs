40 #ifdef __SYCL_DEVICE_ONLY__
41 template <typename FuncTy, typename... Ts>
42 auto builtin_device_rel_impl(FuncTy F, const Ts &...xs) {
43  using T = typename first_type<Ts...>::type;
44  if constexpr (detail::is_vec_or_swizzle_v<T>) {
45  // decltype(ret) is signed char ext_vector_type(N). Convert it to
46  // sycl::vec<signed char, N> first and then to the required return type of
47  // the relation builtin (vector of int16_t/int32_t/int64_t depending on the
48  // arguments' element type).
49  auto ret = F(builtins::convert_arg(xs)...);
50  vec<signed char, num_elements<T>::value> tmp{ret};
51  using res_elem_type =
52  make_type_t<get_elem_type_t<T>, type_list<int16_t, int32_t, int64_t>>;
53  static_assert(is_scalar_arithmetic_v<res_elem_type>);
54  return tmp.template convert<res_elem_type>();
55  } else if constexpr (std::is_same_v<T, half>) {
56  return bool{F(builtins::convert_arg(xs)...)};
57  } else {
58  static_assert(!detail::is_swizzle_v<T>);
59  return F(builtins::convert_arg(xs)...);
60  }
61 }
62 #endif

64 template <typename FuncTy, typename... Ts>
65 auto builtin_delegate_rel_impl(FuncTy F, const Ts &...x) {
66  using T = typename first_type<Ts...>::type;
67  if constexpr ((... || is_swizzle_v<Ts>)) {
68  return F(simplify_if_swizzle_t<T>{x}...);
69  } else if constexpr (is_vec_v<T>) {
70  // TODO: using Res{} to avoid Werror. Not sure if ok.
71  vec<same_size_signed_int_t<get_elem_type_t<T>>, T::size()> Res{};
72  detail::loop<T::size()>(
73  [&](auto idx) { Res[idx] = F(x[idx]...) ? -1 : 0; });
74  return Res;
75  } else {
76  // marray.
77  marray<bool, T::size()> Res;
78  // TODO: Can we optimize this? Note that using vector version isn't
79  // straightforward as it doesn't return booleans.
80  detail::loop<T::size()>([&](auto idx) { Res[idx] = F(x[idx]...); });
81  return Res;
82  }
83 }

264 template <typename T0, typename T1, typename T2>
265 std::enable_if_t<detail::rel_enable_select_v<T0, T1, T2>,
266  detail::simplify_if_swizzle_t<T0>>
267 select(T0 a, T1 b, T2 c) {
268  if constexpr (detail::is_marray_v<T0>) {
269  T0 ret;
270  for (size_t i = 0; i < T0::size(); ++i)
271  ret[i] = (c[i] ? b[i] : a[i]);
272  return ret;
273  } else {
274  detail::simplify_if_swizzle_t<T0> ret;
275  for (size_t i = 0; i < ret.size(); ++i)
276  ret[i] = (detail::msbIsSet(c[i]) ? b[i] : a[i]);
277  return ret;
278  }
