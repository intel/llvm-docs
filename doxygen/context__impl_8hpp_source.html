181  bool isDeviceValid(DeviceImplPtr Device) {
182  // OpenCL does not support using descendants of context members within that
183  // context yet.
184  // TODO remove once this limitation is lifted
185  if (!is_host() && Device->getBackend() == backend::opencl)
186  return hasDevice(Device);
187 
188  while (!hasDevice(Device)) {
189  if (Device->isRootDevice()) {
190  if (Device->has(aspect::ext_oneapi_is_component)) {
191  // Component devices should be implicitly usable in context created
192  // for a composite device they belong to.
193  auto CompositeDevice = Device->get_info<
194  ext::oneapi::experimental::info::device::composite_device>();
195  return hasDevice(detail::getSyclObjImpl(CompositeDevice));
196  }
197 
198  return false;
199  }
200  Device = detail::getSyclObjImpl(
201  Device->get_info<info::device::parent_device>());
202  }
203 
204  return true;
205  }

332 template <typename T, typename Capabilities>
333 void GetCapabilitiesIntersectionSet(const std::vector<sycl::device> &Devices,
334  std::vector<T> &CapabilityList) {
335  for (const sycl::device &Device : Devices) {
336  std::vector<T> NewCapabilityList;
337  std::vector<T> DeviceCapabilities = Device.get_info<Capabilities>();
338  std::set_intersection(
339  CapabilityList.begin(), CapabilityList.end(),
340  DeviceCapabilities.begin(), DeviceCapabilities.end(),
341  std::inserter(NewCapabilityList, NewCapabilityList.begin()));
342  CapabilityList = NewCapabilityList;
343  }
344  CapabilityList.shrink_to_fit();
345 }
