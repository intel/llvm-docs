27 void *alignedAllocHost(size_t Alignment, size_t Size, const context &Ctxt,
28  alloc Kind) {
29  void *RetVal = nullptr;
30  if (Size == 0)
31  return nullptr;
32  if (Ctxt.is_host()) {
33  if (!Alignment) {
34  // worst case default
35  Alignment = 128;
36  }
37 
38  aligned_allocator<char> Alloc(Alignment);
39  try {
40  RetVal = Alloc.allocate(Size);
41  } catch (const std::bad_alloc &) {
42  // Conform with Specification behavior
43  RetVal = nullptr;
44  }
45  } else {
46  std::shared_ptr<context_impl> CtxImpl = detail::getSyclObjImpl(Ctxt);
47  pi_context C = CtxImpl->getHandleRef();
48  const detail::plugin &Plugin = CtxImpl->getPlugin();
49  pi_result Error;
50 
51  switch (Kind) {
52  case alloc::host: {
53  Error = Plugin.call_nocheck<PiApiKind::piextUSMHostAlloc>(
54  &RetVal, C, nullptr, Size, Alignment);
55  break;
56  }
57  case alloc::device:
58  case alloc::shared:
59  case alloc::unknown: {
60  RetVal = nullptr;
61  Error = PI_INVALID_VALUE;
62  break;
63  }
64  }
65 
66  // Error is for debugging purposes.
67  // The spec wants a nullptr returned, not an exception.
68  if (Error != PI_SUCCESS)
69  return nullptr;
70  }
71  return RetVal;
72 }

249 alloc get_pointer_type(const void *Ptr, const context &Ctxt) {
250  if (!Ptr)
251  return alloc::unknown;
252 
253  // Everything on a host device is just system malloc so call it host
254  if (Ctxt.is_host())
255  return alloc::host;
256 
257  std::shared_ptr<detail::context_impl> CtxImpl = detail::getSyclObjImpl(Ctxt);
258  pi_context PICtx = CtxImpl->getHandleRef();
259  pi_usm_type AllocTy;
260 
261  // query type using PI function
262  const detail::plugin &Plugin = CtxImpl->getPlugin();
263  RT::PiResult Err =
264  Plugin.call_nocheck<detail::PiApiKind::piextUSMGetMemAllocInfo>(
265  PICtx, Ptr, PI_MEM_ALLOC_TYPE, sizeof(pi_usm_type), &AllocTy,
266  nullptr);
267 
268  // PI_INVALID_VALUE means USM doesn't know about this ptr
269  if (Err == PI_INVALID_VALUE)
270  return alloc::unknown;
271  // otherwise PI_SUCCESS is expected
272  if (Err != PI_SUCCESS) {
273  throw runtime_error("Error querying USM pointer: ", Err);
274  }
275 
276  alloc ResultAlloc;
277  switch (AllocTy) {
278  case PI_MEM_TYPE_HOST:
279  ResultAlloc = alloc::host;
280  break;
281  case PI_MEM_TYPE_DEVICE:
282  ResultAlloc = alloc::device;
283  break;
284  case PI_MEM_TYPE_SHARED:
285  ResultAlloc = alloc::shared;
286  break;
287  default:
288  ResultAlloc = alloc::unknown;
289  break;
290  }
291 
292  return ResultAlloc;
293 }

299 device get_pointer_device(const void *Ptr, const context &Ctxt) {
300  // Check if ptr is a valid USM pointer
301  if (get_pointer_type(Ptr, Ctxt) == alloc::unknown)
302  throw runtime_error("Ptr not a valid USM allocation!", PI_INVALID_VALUE);
303 
304  // Just return the host device in the host context
305  if (Ctxt.is_host())
306  return Ctxt.get_devices()[0];
307 
308  std::shared_ptr<detail::context_impl> CtxImpl = detail::getSyclObjImpl(Ctxt);
309 
310  // Check if ptr is a host allocation
311  if (get_pointer_type(Ptr, Ctxt) == alloc::host) {
312  auto Devs = CtxImpl->getDevices();
313  if (Devs.size() == 0)
314  throw runtime_error("No devices in passed context!", PI_INVALID_VALUE);
315 
316  // Just return the first device in the context
317  return Devs[0];
318  }
319 
320  pi_context PICtx = CtxImpl->getHandleRef();
321  pi_device DeviceId;
322 
323  // query device using PI function
324  const detail::plugin &Plugin = CtxImpl->getPlugin();
325  Plugin.call<detail::PiApiKind::piextUSMGetMemAllocInfo>(
326  PICtx, Ptr, PI_MEM_ALLOC_DEVICE, sizeof(pi_device), &DeviceId, nullptr);
327 
328  for (const device &Dev : CtxImpl->getDevices()) {
329  // Try to find the real sycl device used in the context
330  if (detail::getSyclObjImpl(Dev)->getHandleRef() == DeviceId)
331  return Dev;
332  }
333 
334  throw runtime_error("Cannot find device associated with USM allocation!",
335  PI_INVALID_OPERATION);
336 }
