43 void *alignedAllocHost(size_t Alignment, size_t Size, const context &Ctxt,
44  alloc Kind, const property_list &PropList,
45  const detail::code_location &CodeLoc) {
46 #ifdef XPTI_ENABLE_INSTRUMENTATION
47  // Stash the code location information and propagate
48  detail::tls_code_loc_t CL(CodeLoc);
49  XPTIScope PrepareNotify((void *)alignedAllocHost,
50  (uint16_t)xpti::trace_point_type_t::node_create,
51  SYCL_MEM_ALLOC_STREAM_NAME, "malloc_host");
52  PrepareNotify.addMetadata([&](auto TEvent) {
53  xpti::addMetadata(TEvent, "sycl_device_name", std::string("Host"));
54  xpti::addMetadata(TEvent, "sycl_device", 0);
55  xpti::addMetadata(TEvent, "memory_size", Size);
56  });
57  // Notify XPTI about the memset submission
58  PrepareNotify.notify();
59  // Emit a begin/end scope for this call
60  PrepareNotify.scopedNotify(
61  (uint16_t)xpti::trace_point_type_t::mem_alloc_begin);
62 #endif
63  const auto &devices = Ctxt.get_devices();
64  if (!std::any_of(devices.begin(), devices.end(), [&](const auto &device) {
65  return device.has(sycl::aspect::usm_host_allocations);
66  })) {
67  throw sycl::exception(
68  sycl::errc::feature_not_supported,
69  "No device in this context supports USM host allocations!");
70  }
71  void *RetVal = nullptr;
72  if (Size == 0)
73  return nullptr;
74 
75  std::shared_ptr<context_impl> CtxImpl = detail::getSyclObjImpl(Ctxt);
76  ur_context_handle_t C = CtxImpl->getHandleRef();
77  const PluginPtr &Plugin = CtxImpl->getPlugin();
78  ur_result_t Error = UR_RESULT_ERROR_INVALID_VALUE;
79 
80  switch (Kind) {
81  case alloc::host: {
82  ur_usm_desc_t UsmDesc{};
83  UsmDesc.align = Alignment;
84 
85  ur_usm_alloc_location_desc_t UsmLocationDesc{};
86  UsmLocationDesc.stype = UR_STRUCTURE_TYPE_USM_ALLOC_LOCATION_DESC;
87 
88  if (PropList.has_property<
89  sycl::ext::intel::experimental::property::usm::buffer_location>() &&
90  Ctxt.get_platform().has_extension(
91  "cl_intel_mem_alloc_buffer_location")) {
92  UsmLocationDesc.location = static_cast<uint32_t>(
93  PropList
94  .get_property<sycl::ext::intel::experimental::property::usm::
95  buffer_location>()
96  .get_buffer_location());
97  UsmDesc.pNext = &UsmLocationDesc;
98  }
99 
100  Error = Plugin->call_nocheck(urUSMHostAlloc, C, &UsmDesc,
101  /* pool= */ nullptr, Size, &RetVal);
102 
103  break;
104  }
105  case alloc::device:
106  case alloc::shared:
107  case alloc::unknown: {
108  RetVal = nullptr;
109  Error = UR_RESULT_ERROR_INVALID_VALUE;
110  break;
111  }
112  }
113 
114  // Error is for debugging purposes.
115  // The spec wants a nullptr returned, not an exception.
116  if (Error != UR_RESULT_SUCCESS)
117  return nullptr;
118 #ifdef XPTI_ENABLE_INSTRUMENTATION
119  xpti::addMetadata(PrepareNotify.traceEvent(), "memory_ptr",
120  reinterpret_cast<size_t>(RetVal));
121 #endif
122  return RetVal;
123 }

125 void *alignedAllocInternal(size_t Alignment, size_t Size,
126  const context_impl *CtxImpl,
127  const device_impl *DevImpl, alloc Kind,
128  const property_list &PropList) {
129  if (Kind == alloc::device &&
130  !DevImpl->has(sycl::aspect::usm_device_allocations)) {
131  throw sycl::exception(sycl::errc::feature_not_supported,
132  "Device does not support USM device allocations!");
133  }
134  if (Kind == alloc::shared &&
135  !DevImpl->has(sycl::aspect::usm_shared_allocations)) {
136  throw sycl::exception(sycl::errc::feature_not_supported,
137  "Device does not support shared USM allocations!");
138  }
139  void *RetVal = nullptr;
140  if (Size == 0)
141  return nullptr;
142 
143  ur_context_handle_t C = CtxImpl->getHandleRef();
144  const PluginPtr &Plugin = CtxImpl->getPlugin();
145  ur_result_t Error = UR_RESULT_ERROR_INVALID_VALUE;
146  ur_device_handle_t Dev;
147 
148  switch (Kind) {
149  case alloc::device: {
150  Dev = DevImpl->getHandleRef();
151 
152  ur_usm_desc_t UsmDesc{};
153  UsmDesc.align = Alignment;
154 
155  ur_usm_alloc_location_desc_t UsmLocationDesc{};
156  UsmLocationDesc.stype = UR_STRUCTURE_TYPE_USM_ALLOC_LOCATION_DESC;
157 
158  // Buffer location is only supported on FPGA devices
159  if (PropList.has_property<
160  sycl::ext::intel::experimental::property::usm::buffer_location>() &&
161  DevImpl->has_extension("cl_intel_mem_alloc_buffer_location")) {
162  UsmLocationDesc.location = static_cast<uint32_t>(
163  PropList
164  .get_property<sycl::ext::intel::experimental::property::usm::
165  buffer_location>()
166  .get_buffer_location());
167  UsmDesc.pNext = &UsmLocationDesc;
168  }
169 
170  Error = Plugin->call_nocheck(urUSMDeviceAlloc, C, Dev, &UsmDesc,
171  /*pool=*/nullptr, Size, &RetVal);
172 
173  break;
174  }
175  case alloc::shared: {
176  Dev = DevImpl->getHandleRef();
177 
178  ur_usm_desc_t UsmDesc{};
179  UsmDesc.align = Alignment;
180 
181  ur_usm_alloc_location_desc_t UsmLocationDesc{};
182  UsmLocationDesc.stype = UR_STRUCTURE_TYPE_USM_ALLOC_LOCATION_DESC;
183 
184  ur_usm_device_desc_t UsmDeviceDesc{};
185  UsmDeviceDesc.stype = UR_STRUCTURE_TYPE_USM_DEVICE_DESC;
186  UsmDeviceDesc.flags = 0;
187 
188  UsmDesc.pNext = &UsmDeviceDesc;
189 
190  if (PropList.has_property<
191  sycl::ext::oneapi::property::usm::device_read_only>()) {
192  UsmDeviceDesc.flags |= UR_USM_DEVICE_MEM_FLAG_DEVICE_READ_ONLY;
193  }
194 
195  if (PropList.has_property<
196  sycl::ext::intel::experimental::property::usm::buffer_location>() &&
197  DevImpl->has_extension("cl_intel_mem_alloc_buffer_location")) {
198  UsmLocationDesc.location = static_cast<uint32_t>(
199  PropList
200  .get_property<sycl::ext::intel::experimental::property::usm::
201  buffer_location>()
202  .get_buffer_location());
203  UsmDeviceDesc.pNext = &UsmLocationDesc;
204  }
205 
206  Error = Plugin->call_nocheck(urUSMSharedAlloc, C, Dev, &UsmDesc,
207  /*pool=*/nullptr, Size, &RetVal);
208 
209  break;
210  }
211  case alloc::host:
212  case alloc::unknown: {
213  RetVal = nullptr;
214  Error = UR_RESULT_ERROR_INVALID_VALUE;
215  break;
216  }
217  }
218 
219  // Error is for debugging purposes.
220  // The spec wants a nullptr returned, not an exception.
221  if (Error != UR_RESULT_SUCCESS)
222  return nullptr;
223  return RetVal;
224 }

538 // Pointer queries
544 alloc get_pointer_type(const void *Ptr, const context &Ctxt) {
545  if (!Ptr)
546  return alloc::unknown;
547 
548  std::shared_ptr<detail::context_impl> CtxImpl = detail::getSyclObjImpl(Ctxt);
549 
550  ur_context_handle_t URCtx = CtxImpl->getHandleRef();
551  ur_usm_type_t AllocTy;
552 
553  // query type using UR function
554  const detail::PluginPtr &Plugin = CtxImpl->getPlugin();
555  ur_result_t Err = Plugin->call_nocheck(
556  urUSMGetMemAllocInfo, URCtx, Ptr, UR_USM_ALLOC_INFO_TYPE,
557  sizeof(ur_usm_type_t), &AllocTy, nullptr);
558 
559  // UR_RESULT_ERROR_INVALID_VALUE means USM doesn't know about this ptr
560  if (Err == UR_RESULT_ERROR_INVALID_VALUE)
561  return alloc::unknown;
562  // otherwise UR_RESULT_SUCCESS is expected
563  if (Err != UR_RESULT_SUCCESS) {
564  throw detail::set_ur_error(
565  exception(make_error_code(errc::runtime), "get_pointer_type() failed"),
566  Err);
567  }
568 
569  alloc ResultAlloc;
570  switch (AllocTy) {
571  case UR_USM_TYPE_HOST:
572  ResultAlloc = alloc::host;
573  break;
574  case UR_USM_TYPE_DEVICE:
575  ResultAlloc = alloc::device;
576  break;
577  case UR_USM_TYPE_SHARED:
578  ResultAlloc = alloc::shared;
579  break;
580  default:
581  ResultAlloc = alloc::unknown;
582  break;
583  }
584 
585  return ResultAlloc;
586 }

592 device get_pointer_device(const void *Ptr, const context &Ctxt) {
593  // Check if ptr is a valid USM pointer
594  if (get_pointer_type(Ptr, Ctxt) == alloc::unknown)
595  throw exception(make_error_code(errc::invalid),
596  "Ptr not a valid USM allocation!");
597 
598  std::shared_ptr<detail::context_impl> CtxImpl = detail::getSyclObjImpl(Ctxt);
599 
600  // Check if ptr is a host allocation
601  if (get_pointer_type(Ptr, Ctxt) == alloc::host) {
602  auto Devs = CtxImpl->getDevices();
603  if (Devs.size() == 0)
604  throw exception(make_error_code(errc::invalid),
