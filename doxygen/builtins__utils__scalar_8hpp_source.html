91 // Constexpr function for checking that all types are the same, considering
92 // swizzles and vectors the same if they have the same number of elements and
93 // the same element type.
94 template <typename T, typename... Ts> constexpr bool CheckAllSameOpType() {
95  constexpr bool SameType[] = {
96  is_same_op_v<std::remove_cv_t<T>, std::remove_cv_t<Ts>>...};
97  // Replace with std::all_of with C++20.
98  for (size_t I = 0; I < sizeof...(Ts); ++I)
99  if (!SameType[I])
100  return false;
101  return true;
102 }
