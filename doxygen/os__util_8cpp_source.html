58 static int callback(struct dl_phdr_info *Info, size_t, void *Data) {
59  auto Base = reinterpret_cast<unsigned char *>(Info->dlpi_addr);
60  auto MI = reinterpret_cast<ModuleInfo *>(Data);
61  auto TestAddr = reinterpret_cast<const unsigned char *>(MI->VirtAddr);
62 
63  for (int i = 0; i < Info->dlpi_phnum; ++i) {
64  unsigned char *SegStart = Base + Info->dlpi_phdr[i].p_vaddr;
65  unsigned char *SegEnd = SegStart + Info->dlpi_phdr[i].p_memsz;
66 
67  // check if the tested address is within current segment
68  if (TestAddr >= SegStart && TestAddr < SegEnd) {
69  // ... it is - belongs to the module then
70  // dlpi_addr is zero for the executable, replace it
71  auto H = reinterpret_cast<void *>(Info->dlpi_addr);
72  MI->Handle = H ? H : reinterpret_cast<void *>(OSUtil::ExeModuleHandle);
73  MI->Name = Info->dlpi_name;
74  return 1; // non-zero tells to finish iteration via modules
75  }
76  }
77  return 0;
78 }

87 bool procMapsAddressInRange(std::istream &Stream, uintptr_t Addr) {
88  uintptr_t Start = 0, End = 0;
89  Stream >> Start;
90  assert(!Stream.fail() && Stream.peek() == '-' &&
91  "Couldn't read /proc/self/maps correctly");
92  Stream.ignore(1);
93 
94  Stream >> End;
95  assert(!Stream.fail() && Stream.peek() == ' ' &&
96  "Couldn't read /proc/self/maps correctly");
97  Stream.ignore(1);
98 
99  return Addr >= Start && Addr < End;
100 }

103 std::string OSUtil::getCurrentDSODir() {
104  // Examine /proc/self/maps and find where this function (getCurrendDSODir)
105  // comes from - this is supposed to be an absolute path to libsycl.so.
106  //
107  // File structure is the following:
108  // address perms offset dev inode pathname
109  // 00400000-00452000 r-xp 00000000 08:02 173521 /usr/bin/foo
110  // 007c2000-007c8000 r--p 001c2000 fc:05 52567930 /usr/bin/bar
111  //
112  // We need to:
113  //
114  // 1) Iterate over lines and find the line which have an address of the
115  // current function in an `address' range.
116  //
117  // 2) Check that perms have read and executable flags (since we do execute
118  // this function).
119  //
120  // 3) Skip offset, dev, inode
121  //
122  // 4) Extract an absolute path to a filename and get a dirname from it.
123  //
124  uintptr_t CurrentFunc = (uintptr_t) &getCurrentDSODir;
125  std::ifstream Stream("/proc/self/maps");
126  Stream >> std::hex;
127  while (!Stream.eof()) {
128  if (!procMapsAddressInRange(Stream, CurrentFunc)) {
129  // Skip the rest until an EOL and check the next line
130  Stream.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
131  continue;
132  }
133 
134  char Perm[4];
135  Stream.readsome(Perm, sizeof(Perm));
136  assert(Perm[0] == 'r' && Perm[2] == 'x' &&
137  "Invalid flags in /proc/self/maps");
138  assert(Stream.peek() == ' ');
139  Stream.ignore(1);
140 
141  // Read and ignore the following:
142  // offset
143  Stream.ignore(std::numeric_limits<std::streamsize>::max(), ' ');
144  Stream.ignore(1);
145  // dev major
146  Stream.ignore(std::numeric_limits<std::streamsize>::max(), ':');
147  Stream.ignore(1);
148  // dev minor
149  Stream.ignore(std::numeric_limits<std::streamsize>::max(), ' ');
150  Stream.ignore(1);
151  // inode
152  Stream.ignore(std::numeric_limits<std::streamsize>::max(), ' ');
153  Stream.ignore(1);
154 
155  // Now read the path: it is padded with whitespaces, so we skip them
156  // first.
157  while (Stream.peek() == ' ') {
158  Stream.ignore(1);
159  }
160  char Path[PATH_MAX];
161  Stream.getline(Path, PATH_MAX - 1);
162  Path[PATH_MAX - 1] = '\0';
163  return OSUtil::getDirName(Path);
164  }
165  assert(false && "Unable to find the current function in /proc/self/maps");
166  return "";
167 }

169 std::string OSUtil::getDirName(const char* Path) {
170  std::string Tmp(Path);
171  // dirname(3) needs a writable C string: a null-terminator is written where a
172  // path should split.
173  size_t TruncatedSize = strlen(dirname(const_cast<char*>(Tmp.c_str())));
174  Tmp.resize(TruncatedSize);
175  return Tmp;
176 }

179 OSModuleHandle OSUtil::getOSModuleHandle(const void *VirtAddr) {
180  HMODULE PhModule;
181  DWORD Flag = GET_MODULE_HANDLE_EX_FLAG_FROM_ADDRESS |
182  GET_MODULE_HANDLE_EX_FLAG_UNCHANGED_REFCOUNT;
183  auto LpModuleAddr = reinterpret_cast<LPCSTR>(VirtAddr);
184  if (!GetModuleHandleExA(Flag, LpModuleAddr, &PhModule)) {
185  // Expect the caller to check for zero and take
186  // necessary action
187  return 0;
188  }
189  if (PhModule == GetModuleHandleA(nullptr))
190  return OSUtil::ExeModuleHandle;
191  return reinterpret_cast<OSModuleHandle>(PhModule);
192 }

195 std::string OSUtil::getCurrentDSODir() {
196  char Path[MAX_PATH];
197  Path[0] = '\0';
198  Path[sizeof(Path) - 1] = '\0';
199  auto Handle = getOSModuleHandle(&getCurrentDSODir);
200  DWORD Ret = GetModuleFileNameA(
201  reinterpret_cast<HMODULE>(OSUtil::ExeModuleHandle == Handle ? 0 : Handle),
202  reinterpret_cast<LPSTR>(&Path), sizeof(Path));
203  assert(Ret < sizeof(Path) && "Path is longer than PATH_MAX?");
204  assert(Ret > 0 && "GetModuleFileNameA failed");
205  (void)Ret;
206 
207  BOOL RetCode = PathRemoveFileSpecA(reinterpret_cast<LPSTR>(&Path));
208  assert(RetCode && "PathRemoveFileSpecA failed");
209  (void)RetCode;
210 
211  return Path;
212 }

224 #if defined(__SYCL_RT_OS_LINUX)
225  struct sysinfo MemInfo;

227  return static_cast<size_t>(MemInfo.totalram * MemInfo.mem_unit);
