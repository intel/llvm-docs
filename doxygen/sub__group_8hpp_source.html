59 // Selects 8-bit, 16-bit or 32-bit type depending on size of T. If T doesn't
60 // maps to mentioned types, then void is returned
61 template <typename T>
62 using SelectBlockT =
63  select_apply_cl_scalar_t<T, uint8_t, uint16_t, uint32_t, void>;

65 template <typename T, access::address_space Space>

68  Space == access::address_space::global_space>;

91 template <typename T, access::address_space Space>
92 T load(const multi_ptr<T, Space> src) {
93  using BlockT = SelectBlockT<T>;
94  using PtrT =
95  sycl::detail::ConvertToOpenCLType_t<const multi_ptr<BlockT, Space>>;
96 
97  BlockT Ret =
98  __spirv_SubgroupBlockReadINTEL<BlockT>(reinterpret_cast<PtrT>(src.get()));
99 
100  return bit_cast<T>(Ret);
101 }
