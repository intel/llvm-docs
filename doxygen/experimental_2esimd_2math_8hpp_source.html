1537  * 0 x_small x_medium x_large
1538  * | x | rational polynomial | 1 - 2/(1 + exp(2*x)) | 1

1540  * rational polynomial for single precision = x + x * (g * (p[1] * g + p[0]) /
1541  * (g + q[0]) g = x^2 p0 = -0.82377 28127 E+00 p1 = -0.38310 10665 E-02 q0 =
1542  * 0.24713 19654 E+01 q1 = 1.00000 00000 E+00
1543  *
1544  */

1551  constexpr float xmedium = 0.54930614433405484570f;
1552  constexpr float xlarge = 8.66433975699931636772f;
1553  constexpr float log2E = 1.442695f; // same as esimd::log(e)
1554

1556

1558  RT g = absX * absX;
1559 
1560  RT sign;
1561  sign.merge(-1.f, 1.f, x < 0.f);
1562 
1563  auto isLarge = absX > xlarge;
1564  auto minor = absX <= xlarge;

1566  auto isGtSmall = (absX > xsmall) & (absX <= xmedium);

1571  temp = ((temp - 2.f) / temp) * sign;
1572  res.merge(temp, isGtMed);
