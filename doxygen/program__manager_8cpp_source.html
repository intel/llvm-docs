120 template <typename ExceptionT, typename RetT>
121 RetT *waitUntilBuilt(KernelProgramCache &Cache,
122  KernelProgramCache::BuildResult<RetT> *BuildResult) {
123  // any thread which will find nullptr in cache will wait until the pointer
124  // is not null anymore
125  Cache.waitUntilBuilt(*BuildResult, [BuildResult]() {
126  int State = BuildResult->State.load();
127 
128  return State == BS_Done || State == BS_Failed;
129  });
130 
131  if (BuildResult->Error.isFilledIn()) {
132  const KernelProgramCache::BuildError &Error = BuildResult->Error;
133  throw ExceptionT(Error.Msg, Error.Code);
134  }
135 
136  RetT *Result = BuildResult->Ptr.load();
137 
138  return Result;
139 }

160 template <typename RetT, typename ExceptionT, typename KeyT, typename AcquireFT,
161  typename GetCacheFT, typename BuildFT>
162 KernelProgramCache::BuildResult<RetT> *
163 getOrBuild(KernelProgramCache &KPCache, KeyT &&CacheKey, AcquireFT &&Acquire,
164  GetCacheFT &&GetCache, BuildFT &&Build) {
165  bool InsertionTookPlace;
166  KernelProgramCache::BuildResult<RetT> *BuildResult;
167 
168  {
169  auto LockedCache = Acquire(KPCache);
170  auto &Cache = GetCache(LockedCache);
171  auto Inserted =
172  Cache.emplace(std::piecewise_construct, std::forward_as_tuple(CacheKey),
173  std::forward_as_tuple(nullptr, BS_InProgress));
174 
175  InsertionTookPlace = Inserted.second;
176  BuildResult = &Inserted.first->second;
177  }
178 
179  // no insertion took place, thus some other thread has already inserted smth
180  // in the cache
181  if (!InsertionTookPlace) {
182  for (;;) {
183  RetT *Result = waitUntilBuilt<ExceptionT>(KPCache, BuildResult);
184 
185  if (Result)
186  return BuildResult;
187 
188  // Previous build is failed. There was no SYCL exception though.
189  // We might try to build once more.
190  int Expected = BS_Failed;
191  int Desired = BS_InProgress;
192 
193  if (BuildResult->State.compare_exchange_strong(Expected, Desired))
194  break; // this thread is the building thread now
195  }
196  }
197 
198  // only the building thread will run this
199  try {
200  RetT *Desired = Build();
201 
202 #ifndef NDEBUG
203  RetT *Expected = nullptr;
204 
205  if (!BuildResult->Ptr.compare_exchange_strong(Expected, Desired))
206  // We've got a funny story here
207  assert(false && "We've build an entity that is already have been built.");
208 #else
209  BuildResult->Ptr.store(Desired);
210 #endif
211 
212  {
213  // Even if shared variable is atomic, it must be modified under the mutex
214  // in order to correctly publish the modification to the waiting thread
215  std::lock_guard<std::mutex> Lock(BuildResult->MBuildResultMutex);
216  BuildResult->State.store(BS_Done);
217  }
218 
219  KPCache.notifyAllBuild(*BuildResult);
220 
221  return BuildResult;
222  } catch (const exception &Ex) {
223  BuildResult->Error.Msg = Ex.what();
224  BuildResult->Error.Code = Ex.get_cl_code();
225 
226  {
227  std::lock_guard<std::mutex> Lock(BuildResult->MBuildResultMutex);
228  BuildResult->State.store(BS_Failed);
229  }
230 
231  KPCache.notifyAllBuild(*BuildResult);
232 
233  std::rethrow_exception(std::current_exception());
234  } catch (...) {
235  {
236  std::lock_guard<std::mutex> Lock(BuildResult->MBuildResultMutex);
237  BuildResult->State.store(BS_Failed);
238  }
239 
240  KPCache.notifyAllBuild(*BuildResult);
241 
242  std::rethrow_exception(std::current_exception());
243  }
244 }

247 static bool isDeviceBinaryTypeSupported(const context &C,
248  RT::PiDeviceBinaryType Format) {
249  // All formats except PI_DEVICE_BINARY_TYPE_SPIRV are supported.
250  if (Format != PI_DEVICE_BINARY_TYPE_SPIRV)
251  return true;
252 
253  const backend ContextBackend =
254  detail::getSyclObjImpl(C)->getPlugin().getBackend();
255 
256  // The CUDA backend cannot use SPIR-V
257  if (ContextBackend == backend::ext_oneapi_cuda)
258  return false;
259 
260  std::vector<device> Devices = C.get_devices();
261 
262  // Program type is SPIR-V, so we need a device compiler to do JIT.
263  for (const device &D : Devices) {
264  if (!D.get_info<info::device::is_compiler_available>())
265  return false;
266  }
267 
268  // OpenCL 2.1 and greater require clCreateProgramWithIL
269  if (ContextBackend == backend::opencl) {
270  std::string ver = C.get_platform().get_info<info::platform::version>();
271  if (ver.find("OpenCL 1.0") == std::string::npos &&
272  ver.find("OpenCL 1.1") == std::string::npos &&
273  ver.find("OpenCL 1.2") == std::string::npos &&
274  ver.find("OpenCL 2.0") == std::string::npos)
275  return true;
276  }
277 
278  for (const device &D : Devices) {
279  // We need cl_khr_il_program extension to be present
280  // and we can call clCreateProgramWithILKHR using the extension
281  std::vector<std::string> Extensions =
282  D.get_info<info::device::extensions>();
283  if (Extensions.end() ==
284  std::find(Extensions.begin(), Extensions.end(), "cl_khr_il_program"))
285  return false;
286  }
287 
288  return true;
289 }

306 RT::PiProgram ProgramManager::createPIProgram(const RTDeviceBinaryImage &Img,
307  const context &Context,
308  const device &Device) {
309  if (DbgProgMgr > 0)
310  std::cerr << ">>> ProgramManager::createPIProgram(" << &Img << ", "
311  << getRawSyclObjImpl(Context) << ", " << getRawSyclObjImpl(Device)
312  << ")\n";
313  const pi_device_binary_struct &RawImg = Img.getRawData();
314 
315  // perform minimal sanity checks on the device image and the descriptor
316  if (RawImg.BinaryEnd < RawImg.BinaryStart) {
317  throw runtime_error("Malformed device program image descriptor",
318  PI_INVALID_VALUE);
319  }
320  if (RawImg.BinaryEnd == RawImg.BinaryStart) {
321  throw runtime_error("Invalid device program image: size is zero",
322  PI_INVALID_VALUE);
323  }
324  size_t ImgSize = Img.getSize();
325 
326  // TODO if the binary image is a part of the fat binary, the clang
327  // driver should have set proper format option to the
328  // clang-offload-wrapper. The fix depends on AOT compilation
329  // implementation, so will be implemented together with it.
330  // Img->Format can't be updated as it is inside of the in-memory
331  // OS module binary.
332  RT::PiDeviceBinaryType Format = Img.getFormat();
333 
334  if (Format == PI_DEVICE_BINARY_TYPE_NONE)
335  Format = pi::getBinaryImageFormat(RawImg.BinaryStart, ImgSize);
336  // RT::PiDeviceBinaryType Format = Img->Format;
337  // assert(Format != PI_DEVICE_BINARY_TYPE_NONE && "Image format not set");
338 
339  if (!isDeviceBinaryTypeSupported(Context, Format))
340  throw feature_not_supported(
341  "SPIR-V online compilation is not supported in this context",
342  PI_INVALID_OPERATION);
343 
344  // Get program metadata from properties
345  auto ProgMetadata = Img.getProgramMetadata();
346  std::vector<pi_device_binary_property> ProgMetadataVector{
347  ProgMetadata.begin(), ProgMetadata.end()};
348 
349  // Load the image
350  const ContextImplPtr Ctx = getSyclObjImpl(Context);
351  RT::PiProgram Res = Format == PI_DEVICE_BINARY_TYPE_SPIRV
352  ? createSpirvProgram(Ctx, RawImg.BinaryStart, ImgSize)
353  : createBinaryProgram(Ctx, Device, RawImg.BinaryStart,
354  ImgSize, ProgMetadataVector);
355 
356  {
357  std::lock_guard<std::mutex> Lock(MNativeProgramsMutex);
358  // associate the PI program with the image it was created for
359  NativePrograms[Res] = &Img;
360  }
361 
362  if (DbgProgMgr > 1)
363  std::cerr << "created program: " << Res
364  << "; image format: " << getFormatStr(Format) << "\n";
365 
366  return Res;
367 }

389  SYCLConfig<SYCL_PROGRAM_COMPILE_OPTIONS>::get();

481 RT::PiProgram ProgramManager::getBuiltPIProgram(
482  OSModuleHandle M, const ContextImplPtr &ContextImpl,
483  const DeviceImplPtr &DeviceImpl, const std::string &KernelName,
484  const program_impl *Prg, bool JITCompilationIsRequired) {
485  // TODO: Make sure that KSIds will be different for the case when the same
486  // kernel built with different options is present in the fat binary.
487  KernelSetId KSId = getKernelSetId(M, KernelName);
488 
489  using PiProgramT = KernelProgramCache::PiProgramT;
490  using ProgramCacheT = KernelProgramCache::ProgramCacheT;
491 
492  KernelProgramCache &Cache = ContextImpl->getKernelProgramCache();
493 
494  auto AcquireF = [](KernelProgramCache &Cache) {
495  return Cache.acquireCachedPrograms();
496  };
497  auto GetF = [](const Locked<ProgramCacheT> &LockedCache) -> ProgramCacheT & {
498  return LockedCache.get();
499  };
500 
501  std::string CompileOpts;
502  std::string LinkOpts;
503  if (Prg) {
504  CompileOpts = Prg->get_build_options();
505  }
506 
507  applyOptionsFromEnvironment(CompileOpts, LinkOpts);
508 
509  SerializedObj SpecConsts;
510  if (Prg)
511  Prg->stableSerializeSpecConstRegistry(SpecConsts);
512 
513  // Check if we can optimize program builds for sub-devices by using a program
514  // built for the root device
515  DeviceImplPtr RootDevImpl = DeviceImpl;
516  while (!RootDevImpl->isRootDevice()) {
517  auto ParentDev = detail::getSyclObjImpl(
518  RootDevImpl->get_info<info::device::parent_device>());
519  // Sharing is allowed within a single context only
520  if (!ContextImpl->hasDevice(ParentDev))
521  break;
522  RootDevImpl = ParentDev;
523  }
524 
525  pi_bool MustBuildOnSubdevice = PI_TRUE;
526  ContextImpl->getPlugin().call<PiApiKind::piDeviceGetInfo>(
527  RootDevImpl->getHandleRef(), PI_DEVICE_INFO_BUILD_ON_SUBDEVICE,
528  sizeof(pi_bool), &MustBuildOnSubdevice, nullptr);
529 
530  DeviceImplPtr Dev =
531  (MustBuildOnSubdevice == PI_TRUE) ? DeviceImpl : RootDevImpl;
532  auto BuildF = [this, &M, &KSId, &ContextImpl, &Dev, Prg, &CompileOpts,
533  &LinkOpts, &JITCompilationIsRequired, SpecConsts] {
534  auto Context = createSyclObjFromImpl<context>(ContextImpl);
535  auto Device = createSyclObjFromImpl<device>(Dev);
536 
537  const RTDeviceBinaryImage &Img =
538  getDeviceImage(M, KSId, Context, Device, JITCompilationIsRequired);
539 
540  applyOptionsFromImage(CompileOpts, LinkOpts, Img);
541 
542  const detail::plugin &Plugin = ContextImpl->getPlugin();
543  auto [NativePrg, DeviceCodeWasInCache] = getOrCreatePIProgram(
544  Img, Context, Device, CompileOpts + LinkOpts, SpecConsts);
545 
546  if (!DeviceCodeWasInCache) {
547  if (Prg)
548  flushSpecConstants(*Prg, NativePrg, &Img);
549  if (Img.supportsSpecConstants())
550  enableITTAnnotationsIfNeeded(NativePrg, Plugin);
551  }
552 
553  ProgramPtr ProgramManaged(
554  NativePrg, Plugin.getPiPlugin().PiFunctionTable.piProgramRelease);
555 
556  // Link a fallback implementation of device libraries if they are not
557  // supported by a device compiler.
558  // Pre-compiled programs (after AOT compilation or read from persitent
559  // cache) are supposed to be already linked.
560  // If device image is not SPIR-V, DeviceLibReqMask will be 0 which means
561  // no fallback device library will be linked.
562  uint32_t DeviceLibReqMask = 0;
563  if (!DeviceCodeWasInCache &&
564  Img.getFormat() == PI_DEVICE_BINARY_TYPE_SPIRV &&
565  !SYCLConfig<SYCL_DEVICELIB_NO_FALLBACK>::get())
566  DeviceLibReqMask = getDeviceLibReqMask(Img);
567 
568  ProgramPtr BuiltProgram =
569  build(std::move(ProgramManaged), ContextImpl, CompileOpts, LinkOpts,
570  getRawSyclObjImpl(Device)->getHandleRef(),
571  ContextImpl->getCachedLibPrograms(), DeviceLibReqMask);
572 
573  emitBuiltProgramInfo(BuiltProgram.get(), ContextImpl);
574 
575  {
576  std::lock_guard<std::mutex> Lock(MNativeProgramsMutex);
577  NativePrograms[BuiltProgram.get()] = &Img;
578  }
579 
580  // Save program to persistent cache if it is not there
581  if (!DeviceCodeWasInCache)
582  PersistentDeviceCodeCache::putItemToDisc(
583  Device, Img, SpecConsts, CompileOpts + LinkOpts, BuiltProgram.get());
584  return BuiltProgram.release();
585  };
586 
587  const RT::PiDevice PiDevice = Dev->getHandleRef();
588 
589  auto BuildResult = getOrBuild<PiProgramT, compile_program_error>(
590  Cache,
591  std::make_pair(std::make_pair(std::move(SpecConsts), KSId),
592  std::make_pair(PiDevice, CompileOpts + LinkOpts)),
593  AcquireF, GetF, BuildF);
594  // getOrBuild is not supposed to return nullptr
595  assert(BuildResult != nullptr && "Invalid build result");
596  return BuildResult->Ptr.load();
597 }

599 std::tuple<RT::PiKernel, std::mutex *, RT::PiProgram>
600 ProgramManager::getOrCreateKernel(OSModuleHandle M,
601  const ContextImplPtr &ContextImpl,
602  const DeviceImplPtr &DeviceImpl,
603  const std::string &KernelName,
604  const program_impl *Prg) {
605  if (DbgProgMgr > 0) {
606  std::cerr << ">>> ProgramManager::getOrCreateKernel(" << M << ", "
607  << ContextImpl.get() << ", " << DeviceImpl.get() << ", "
608  << KernelName << ")\n";

611  using PiKernelT = KernelProgramCache::PiKernelT;
612  using KernelCacheT = KernelProgramCache::KernelCacheT;
613  using KernelByNameT = KernelProgramCache::KernelByNameT;
614 
615  KernelProgramCache &Cache = ContextImpl->getKernelProgramCache();
616 
617  std::string CompileOpts, LinkOpts;
618  SerializedObj SpecConsts;
619  if (Prg) {
620  CompileOpts = Prg->get_build_options();
621  Prg->stableSerializeSpecConstRegistry(SpecConsts);
622  }
623  applyOptionsFromEnvironment(CompileOpts, LinkOpts);
624  const RT::PiDevice PiDevice = DeviceImpl->getHandleRef();
625 
626  auto key = std::make_tuple(std::move(SpecConsts), M, PiDevice,
627  CompileOpts + LinkOpts, KernelName);
628  auto ret_tuple = Cache.tryToGetKernelFast(key);
629  if (std::get<0>(ret_tuple))
630  return ret_tuple;
631 
632  RT::PiProgram Program =
633  getBuiltPIProgram(M, ContextImpl, DeviceImpl, KernelName, Prg);
634 
635  auto AcquireF = [](KernelProgramCache &Cache) {
636  return Cache.acquireKernelsPerProgramCache();
637  };
638  auto GetF =
639  [&Program](const Locked<KernelCacheT> &LockedCache) -> KernelByNameT & {
640  return LockedCache.get()[Program];
641  };
642  auto BuildF = [&Program, &KernelName, &ContextImpl] {
643  PiKernelT *Result = nullptr;

929 static std::vector<RT::PiProgram> getDeviceLibPrograms(
930  const ContextImplPtr Context, const RT::PiDevice &Device,
931  std::map<std::pair<DeviceLibExt, RT::PiDevice>, RT::PiProgram>
932  &CachedLibPrograms,
933  uint32_t DeviceLibReqMask) {
934  std::vector<RT::PiProgram> Programs;
935 
936  std::pair<DeviceLibExt, bool> RequiredDeviceLibExt[] = {
937  {DeviceLibExt::cl_intel_devicelib_assert,
938  /* is fallback loaded? */ false},
939  {DeviceLibExt::cl_intel_devicelib_math, false},
940  {DeviceLibExt::cl_intel_devicelib_math_fp64, false},
941  {DeviceLibExt::cl_intel_devicelib_complex, false},
942  {DeviceLibExt::cl_intel_devicelib_complex_fp64, false},
943  {DeviceLibExt::cl_intel_devicelib_cstring, false}};
944 
945  // Disable all devicelib extensions requiring fp64 support if at least
946  // one underlying device doesn't support cl_khr_fp64.
947  std::string DevExtList =
948  get_device_info<std::string, info::device::extensions>::get(
949  Device, Context->getPlugin());
950  const bool fp64Support = (DevExtList.npos != DevExtList.find("cl_khr_fp64"));
951 
952  // Load a fallback library for an extension if the device does not
953  // support it.
954  for (auto &Pair : RequiredDeviceLibExt) {
955  DeviceLibExt Ext = Pair.first;
956  bool &FallbackIsLoaded = Pair.second;
957 
958  if (FallbackIsLoaded) {
959  continue;
960  }
961 
962  if (!isDeviceLibRequired(Ext, DeviceLibReqMask)) {
963  continue;
964  }
965  if ((Ext == DeviceLibExt::cl_intel_devicelib_math_fp64 ||
966  Ext == DeviceLibExt::cl_intel_devicelib_complex_fp64) &&
967  !fp64Support) {
968  continue;
969  }
970 
971  const char *ExtStr = getDeviceLibExtensionStr(Ext);
972 
973  bool InhibitNativeImpl = false;
974  if (const char *Env = getenv("SYCL_DEVICELIB_INHIBIT_NATIVE")) {
975  InhibitNativeImpl = strstr(Env, ExtStr) != nullptr;
976  }
977 
978  bool DeviceSupports = DevExtList.npos != DevExtList.find(ExtStr);
979 
980  if (!DeviceSupports || InhibitNativeImpl) {
981  Programs.push_back(
982  loadDeviceLibFallback(Context, Ext, Device, CachedLibPrograms));
983  FallbackIsLoaded = true;
984  }
985  }
986  return Programs;
987 }

1061 static ProgramManager::KernelArgMask
1062 createKernelArgMask(const pi::ByteArray &Bytes) {
1063  const int NBytesForSize = 8;
1064  const int NBitsInElement = 8;
1065  std::uint64_t SizeInBits = 0;
1066  for (int I = 0; I < NBytesForSize; ++I)
1067  SizeInBits |= static_cast<std::uint64_t>(Bytes[I]) << I * NBitsInElement;
1068 
1069  ProgramManager::KernelArgMask Result;
1070  for (std::uint64_t I = 0; I < SizeInBits; ++I) {
1071  std::uint8_t Byte = Bytes[NBytesForSize + (I / NBitsInElement)];
1072  Result.push_back(Byte & (1 << (I % NBitsInElement)));
1073  }
1074 
1075  return Result;
1076 }

1113  createKernelArgMask(pi::DeviceBinaryProperty(Info).asByteArray());

1117  if (EntriesB != EntriesE) {
1118  std::lock_guard<std::mutex> KernelIDsGuard(m_KernelIDsMutex);
1119

1121  auto ExportedSymbols = Img->getExportedSymbols();
1122  for (const pi_device_binary_property &ExportedSymbol : ExportedSymbols)
1123  m_ExportedSymbols.insert(ExportedSymbol->Name);

1506  const context &Ctx, const std::vector<device> &Devs,
1507  bundle_state TargetState, const std::vector<kernel_id> &KernelIDs) {
1508 
1509  // Collect unique raw device images taking into account kernel ids passed
1510  // TODO: Can we avoid repacking?
1511  std::set<RTDeviceBinaryImage *> BinImages;
1512  if (!KernelIDs.empty()) {
1513  std::lock_guard<std::mutex> KernelIDsGuard(m_KernelIDsMutex);
1514  for (const kernel_id &KID : KernelIDs) {
1515  auto Range = m_KernelIDs2BinImage.equal_range(KID);
1516  for (auto It = Range.first, End = Range.second; It != End; ++It)
1517  BinImages.insert(It->second);
1518  }
1519  } else {
1520  std::lock_guard<std::mutex> Guard(Sync::getGlobalLock());
1521  for (auto &ImagesSets : m_DeviceImages) {
1522  auto &ImagesUPtrs = *ImagesSets.second.get();
1523  for (auto &ImageUPtr : ImagesUPtrs)
1524  BinImages.insert(ImageUPtr.get());
1525  }
1526  }
1527  assert(BinImages.size() > 0 && "Expected to find at least on device image");
1528 
1529  std::vector<device_image_plain> SYCLDeviceImages;
1530  for (RTDeviceBinaryImage *BinImage : BinImages) {
1531  const bundle_state ImgState = getBinImageState(BinImage);
1532 
1533  // Ignore images with incompatible state. Image is considered compatible
1534  // with a target state if an image is already in the target state or can
1535  // be brought to target state by compiling/linking/building.
1536  //
1537  // Example: an image in "executable" state is not compatible with
1538  // "input" target state - there is no operation to convert the image it
1539  // to "input" state. An image in "input" state is compatible with
1540  // "executable" target state because it can be built to get into
1541  // "executable" state.
1542  if (ImgState > TargetState)
1543  continue;

1546  if (!compatibleWithDevice(BinImage, Dev))
1547  continue;

1550  // Collect kernel names for the image
1551  {
1552  std::lock_guard<std::mutex> KernelIDsGuard(m_KernelIDsMutex);
1553  KernelIDs = m_BinImg2KernelIDs[BinImage];
1554  // If the image does not contain any non-service kernels we can skip it.
1555  if (!KernelIDs || KernelIDs->empty())
1556  continue;
1557  }
1558

1560  BinImage, Ctx, Devs, ImgState, KernelIDs, /*PIProgram=*/nullptr);
1561 
1562  SYCLDeviceImages.push_back(
1563  createSyclObjFromImpl<device_image_plain>(Impl));
1564  break;
1565  }
1566  }

1571 void ProgramManager::bringSYCLDeviceImagesToState(
1572  std::vector<device_image_plain> &DeviceImages, bundle_state TargetState) {
1573 
1574  for (device_image_plain &DevImage : DeviceImages) {
1575  const bundle_state DevImageState = getSyclObjImpl(DevImage)->get_state();
1576 
1577  switch (TargetState) {
1578  case bundle_state::input:
1579  // Do nothing since there is no state which can be upgraded to the input.
1580  assert(DevImageState == bundle_state::input);
1581  break;
1582  case bundle_state::object:
1583  if (DevImageState == bundle_state::input) {
1584  DevImage = compile(DevImage, getSyclObjImpl(DevImage)->get_devices(),
1585  /*PropList=*/{});
1586  break;
1587  }
1588  // Device image is expected to be object state then.
1589  assert(DevImageState == bundle_state::object);
1590  break;
1591  case bundle_state::executable: {
1592  switch (DevImageState) {
1593  case bundle_state::input:
1594  DevImage = build(DevImage, getSyclObjImpl(DevImage)->get_devices(),
1595  /*PropList=*/{});
1596  break;
1597  case bundle_state::object: {
1598  std::vector<device_image_plain> LinkedDevImages =
1599  link({DevImage}, getSyclObjImpl(DevImage)->get_devices(),
1600  /*PropList=*/{});
1601  // Since only one device image is passed here one output device image is
1602  // expected
1603  assert(LinkedDevImages.size() == 1 && "Expected one linked image here");
1604  DevImage = LinkedDevImages[0];
1605  break;
1606  }
1607  case bundle_state::executable:
1608  DevImage = build(DevImage, getSyclObjImpl(DevImage)->get_devices(),
1609  /*PropList=*/{});
1610  break;
1611  }
1612  break;
1613  }
1614  }
1615  }
