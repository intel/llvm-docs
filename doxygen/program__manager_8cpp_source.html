108 template <typename ExceptionT, typename RetT>
109 RetT *waitUntilBuilt(KernelProgramCache &Cache,
110  KernelProgramCache::BuildResult<RetT> *BuildResult) {
111  // any thread which will find nullptr in cache will wait until the pointer
112  // is not null anymore
113  Cache.waitUntilBuilt(*BuildResult, [BuildResult]() {
114  int State = BuildResult->State.load();
115 
116  return State == BS_Done || State == BS_Failed;
117  });
118 
119  if (BuildResult->Error.isFilledIn()) {
120  const KernelProgramCache::BuildError &Error = BuildResult->Error;
121  throw ExceptionT(Error.Msg, Error.Code);
122  }
123 
124  RetT *Result = BuildResult->Ptr.load();
125 
126  return Result;
127 }

146 template <typename RetT, typename ExceptionT, typename KeyT, typename AcquireFT,
147  typename GetCacheFT, typename BuildFT>
148 KernelProgramCache::BuildResult<RetT> *
149 getOrBuild(KernelProgramCache &KPCache, KeyT &&CacheKey, AcquireFT &&Acquire,
150  GetCacheFT &&GetCache, BuildFT &&Build) {
151  bool InsertionTookPlace;
152  KernelProgramCache::BuildResult<RetT> *BuildResult;
153 
154  {
155  auto LockedCache = Acquire(KPCache);
156  auto &Cache = GetCache(LockedCache);
157  auto Inserted =
158  Cache.emplace(std::piecewise_construct, std::forward_as_tuple(CacheKey),
159  std::forward_as_tuple(nullptr, BS_InProgress));
160 
161  InsertionTookPlace = Inserted.second;
162  BuildResult = &Inserted.first->second;
163  }
164 
165  // no insertion took place, thus some other thread has already inserted smth
166  // in the cache
167  if (!InsertionTookPlace) {
168  for (;;) {
169  RetT *Result = waitUntilBuilt<ExceptionT>(KPCache, BuildResult);
170 
171  if (Result)
172  return BuildResult;
173 
174  // Previous build is failed. There was no SYCL exception though.
175  // We might try to build once more.
176  int Expected = BS_Failed;
177  int Desired = BS_InProgress;
178 
179  if (BuildResult->State.compare_exchange_strong(Expected, Desired))
180  break; // this thread is the building thread now
181  }
182  }
183 
184  // only the building thread will run this
185  try {
186  RetT *Desired = Build();
187 
188 #ifndef NDEBUG
189  RetT *Expected = nullptr;
190 
191  if (!BuildResult->Ptr.compare_exchange_strong(Expected, Desired))
192  // We've got a funny story here
193  assert(false && "We've build an entity that is already have been built.");
194 #else
195  BuildResult->Ptr.store(Desired);
196 #endif
197 
198  {
199  // Even if shared variable is atomic, it must be modified under the mutex
200  // in order to correctly publish the modification to the waiting thread
201  std::lock_guard<std::mutex> Lock(BuildResult->MBuildResultMutex);
202  BuildResult->State.store(BS_Done);
203  }
204 
205  KPCache.notifyAllBuild(*BuildResult);
206 
207  return BuildResult;
208  } catch (const exception &Ex) {
209  BuildResult->Error.Msg = Ex.what();
210  BuildResult->Error.Code = Ex.get_cl_code();
211 
212  {
213  std::lock_guard<std::mutex> Lock(BuildResult->MBuildResultMutex);
214  BuildResult->State.store(BS_Failed);
215  }
216 
217  KPCache.notifyAllBuild(*BuildResult);
218 
219  std::rethrow_exception(std::current_exception());
220  } catch (...) {
221  {
222  std::lock_guard<std::mutex> Lock(BuildResult->MBuildResultMutex);
223  BuildResult->State.store(BS_Failed);
224  }
225 
226  KPCache.notifyAllBuild(*BuildResult);
227 
228  std::rethrow_exception(std::current_exception());
229  }
230 }

233 static bool isDeviceBinaryTypeSupported(const context &C,
234  RT::PiDeviceBinaryType Format) {
235  // All formats except PI_DEVICE_BINARY_TYPE_SPIRV are supported.
236  if (Format != PI_DEVICE_BINARY_TYPE_SPIRV)
237  return true;
238 
239  const backend ContextBackend =
240  detail::getSyclObjImpl(C)->getPlugin().getBackend();
241 
242  // The CUDA backend cannot use SPIR-V
243  if (ContextBackend == backend::cuda)
244  return false;
245 
246  vector_class<device> Devices = C.get_devices();
247 
248  // Program type is SPIR-V, so we need a device compiler to do JIT.
249  for (const device &D : Devices) {
250  if (!D.get_info<info::device::is_compiler_available>())
251  return false;
252  }
253 
254  // OpenCL 2.1 and greater require clCreateProgramWithIL
255  if (ContextBackend == backend::opencl) {
256  std::string ver = C.get_platform().get_info<info::platform::version>();
257  if (ver.find("OpenCL 1.0") == std::string::npos &&
258  ver.find("OpenCL 1.1") == std::string::npos &&
259  ver.find("OpenCL 1.2") == std::string::npos &&
260  ver.find("OpenCL 2.0") == std::string::npos)
261  return true;
262  }
263 
264  for (const device &D : Devices) {
265  // We need cl_khr_il_program extension to be present
266  // and we can call clCreateProgramWithILKHR using the extension
267  vector_class<string_class> Extensions =
268  D.get_info<info::device::extensions>();
269  if (Extensions.end() ==
270  std::find(Extensions.begin(), Extensions.end(), "cl_khr_il_program"))
271  return false;
272  }
273 
274  return true;
275 }

292 RT::PiProgram ProgramManager::createPIProgram(const RTDeviceBinaryImage &Img,
293  const context &Context,
294  const device &Device) {
295  if (DbgProgMgr > 0)
296  std::cerr << ">>> ProgramManager::createPIProgram(" << &Img << ", "
297  << getRawSyclObjImpl(Context) << ", " << getRawSyclObjImpl(Device)
298  << ")\n";
299  const pi_device_binary_struct &RawImg = Img.getRawData();
300 
301  // perform minimal sanity checks on the device image and the descriptor
302  if (RawImg.BinaryEnd < RawImg.BinaryStart) {
303  throw runtime_error("Malformed device program image descriptor",
304  PI_INVALID_VALUE);
305  }
306  if (RawImg.BinaryEnd == RawImg.BinaryStart) {
307  throw runtime_error("Invalid device program image: size is zero",
308  PI_INVALID_VALUE);
309  }
310  size_t ImgSize = Img.getSize();
311 
312  // TODO if the binary image is a part of the fat binary, the clang
313  // driver should have set proper format option to the
314  // clang-offload-wrapper. The fix depends on AOT compilation
315  // implementation, so will be implemented together with it.
316  // Img->Format can't be updated as it is inside of the in-memory
317  // OS module binary.
318  RT::PiDeviceBinaryType Format = Img.getFormat();
319 
320  if (Format == PI_DEVICE_BINARY_TYPE_NONE)
321  Format = pi::getBinaryImageFormat(RawImg.BinaryStart, ImgSize);
322  // RT::PiDeviceBinaryType Format = Img->Format;
323  // assert(Format != PI_DEVICE_BINARY_TYPE_NONE && "Image format not set");
324 
325  if (!isDeviceBinaryTypeSupported(Context, Format))
326  throw feature_not_supported(
327  "SPIR-V online compilation is not supported in this context",
328  PI_INVALID_OPERATION);
329 
330  // Load the image
331  const ContextImplPtr Ctx = getSyclObjImpl(Context);
332  RT::PiProgram Res =
333  Format == PI_DEVICE_BINARY_TYPE_SPIRV
334  ? createSpirvProgram(Ctx, RawImg.BinaryStart, ImgSize)
335  : createBinaryProgram(Ctx, Device, RawImg.BinaryStart, ImgSize);
336 
337  {
338  std::lock_guard<std::mutex> Lock(MNativeProgramsMutex);
339  // associate the PI program with the image it was created for
340  NativePrograms[Res] = &Img;
341  }
342 
343  if (DbgProgMgr > 1)
344  std::cerr << "created program: " << Res
345  << "; image format: " << getFormatStr(Format) << "\n";
346 
347  return Res;
348 }

350 RT::PiProgram ProgramManager::getBuiltPIProgram(OSModuleHandle M,
351  const context &Context,
352  const device &Device,
353  const string_class &KernelName,
354  const program_impl *Prg,
355  bool JITCompilationIsRequired) {
356  KernelSetId KSId = getKernelSetId(M, KernelName);
357 
358  const ContextImplPtr Ctx = getSyclObjImpl(Context);
359 
360  using PiProgramT = KernelProgramCache::PiProgramT;
361  using ProgramCacheT = KernelProgramCache::ProgramCacheT;
362 
363  KernelProgramCache &Cache = Ctx->getKernelProgramCache();
364 
365  auto AcquireF = [](KernelProgramCache &Cache) {
366  return Cache.acquireCachedPrograms();
367  };
368  auto GetF = [](const Locked<ProgramCacheT> &LockedCache) -> ProgramCacheT & {
369  return LockedCache.get();
370  };
371  auto BuildF = [this, &M, &KSId, &Context, &Device, Prg,
372  &JITCompilationIsRequired] {
373  const RTDeviceBinaryImage &Img =
374  getDeviceImage(M, KSId, Context, Device, JITCompilationIsRequired);
375 
376  ContextImplPtr ContextImpl = getSyclObjImpl(Context);
377  const detail::plugin &Plugin = ContextImpl->getPlugin();
378  RT::PiProgram NativePrg = createPIProgram(Img, Context, Device);
379  if (Prg)
380  flushSpecConstants(*Prg, getSyclObjImpl(Device)->getHandleRef(),
381  NativePrg, &Img);
382  ProgramPtr ProgramManaged(
383  NativePrg, Plugin.getPiPlugin().PiFunctionTable.piProgramRelease);
384 
385  // Link a fallback implementation of device libraries if they are not
386  // supported by a device compiler.
387  // Pre-compiled programs are supposed to be already linked.
388  // If device image is not SPIR-V, DeviceLibReqMask will be 0 which means
389  // no fallback device library will be linked.
390  uint32_t DeviceLibReqMask = 0;
391  // FIXME: disable the fallback device libraries online link as not all
392  // backend supports spv online link. Need to enable it when all backends
393  // support spv online link.
394  if (Img.getFormat() == PI_DEVICE_BINARY_TYPE_SPIRV &&
395  !SYCLConfig<SYCL_DEVICELIB_NO_FALLBACK>::get())
396  DeviceLibReqMask = getDeviceLibReqMask(Img);
397 
398  ProgramPtr BuiltProgram =
399  build(std::move(ProgramManaged), ContextImpl, Img.getCompileOptions(),
400  Img.getLinkOptions(), getRawSyclObjImpl(Device)->getHandleRef(),
401  ContextImpl->getCachedLibPrograms(), DeviceLibReqMask);
402 
403  {
404  std::lock_guard<std::mutex> Lock(MNativeProgramsMutex);
405  NativePrograms[BuiltProgram.get()] = &Img;
406  }
407  return BuiltProgram.release();
408  };
409 
410  SerializedObj SpecConsts;
411  if (Prg)
412  Prg->stableSerializeSpecConstRegistry(SpecConsts);
413 
414  const RT::PiDevice PiDevice = getRawSyclObjImpl(Device)->getHandleRef();
415  auto BuildResult = getOrBuild<PiProgramT, compile_program_error>(
416  Cache,
417  std::make_pair(std::make_pair(std::move(SpecConsts), KSId), PiDevice),
418  AcquireF, GetF, BuildF);
419  return BuildResult->Ptr.load();
420 }

717 static std::vector<RT::PiProgram> getDeviceLibPrograms(
718  const ContextImplPtr Context, const RT::PiDevice &Device,
719  std::map<std::pair<DeviceLibExt, RT::PiDevice>, RT::PiProgram>
720  &CachedLibPrograms,
721  uint32_t DeviceLibReqMask) {
722  std::vector<RT::PiProgram> Programs;
723 
724  std::pair<DeviceLibExt, bool> RequiredDeviceLibExt[] = {
725  {DeviceLibExt::cl_intel_devicelib_assert,
726  /* is fallback loaded? */ false},
727  {DeviceLibExt::cl_intel_devicelib_math, false},
728  {DeviceLibExt::cl_intel_devicelib_math_fp64, false},
729  {DeviceLibExt::cl_intel_devicelib_complex, false},
730  {DeviceLibExt::cl_intel_devicelib_complex_fp64, false}};
731 
732  // Disable all devicelib extensions requiring fp64 support if at least
733  // one underlying device doesn't support cl_khr_fp64.
734  std::string DevExtList =
735  get_device_info<std::string, info::device::extensions>::get(
736  Device, Context->getPlugin());
737  const bool fp64Support = (DevExtList.npos != DevExtList.find("cl_khr_fp64"));
738 
739  // Load a fallback library for an extension if the device does not
740  // support it.
741  for (auto &Pair : RequiredDeviceLibExt) {
742  DeviceLibExt Ext = Pair.first;
743  bool &FallbackIsLoaded = Pair.second;
744 
745  if (FallbackIsLoaded) {
746  continue;
747  }
748 
749  if (!isDeviceLibRequired(Ext, DeviceLibReqMask)) {
750  continue;
751  }
752  if ((Ext == DeviceLibExt::cl_intel_devicelib_math_fp64 ||
753  Ext == DeviceLibExt::cl_intel_devicelib_complex_fp64) &&
754  !fp64Support) {
755  continue;
756  }
757 
758  const char *ExtStr = getDeviceLibExtensionStr(Ext);
759 
760  bool InhibitNativeImpl = false;
761  if (const char *Env = getenv("SYCL_DEVICELIB_INHIBIT_NATIVE")) {
762  InhibitNativeImpl = strstr(Env, ExtStr) != nullptr;
763  }
764 
765  bool DeviceSupports = DevExtList.npos != DevExtList.find(ExtStr);
766 
767  if (!DeviceSupports || InhibitNativeImpl) {
768  Programs.push_back(
769  loadDeviceLibFallback(Context, Ext, Device, CachedLibPrograms));
770  FallbackIsLoaded = true;
771  }
772  }
773  return Programs;
774 }

860 static ProgramManager::KernelArgMask
861 createKernelArgMask(const pi::ByteArray &Bytes) {
862  const int NBytesForSize = 8;
863  const int NBitsInElement = 8;
864  std::uint64_t SizeInBits = 0;
865  for (int I = 0; I < NBytesForSize; ++I)
866  SizeInBits |= static_cast<std::uint64_t>(Bytes[I]) << I * NBitsInElement;
867 
868  ProgramManager::KernelArgMask Result;
869  for (std::uint64_t I = 0; I < SizeInBits; ++I) {
870  std::uint8_t Byte = Bytes[NBytesForSize + (I / NBitsInElement)];
871  Result.push_back(Byte & (1 << (I % NBitsInElement)));
872  }
873 
874  return Result;
875 }

877 void ProgramManager::addImages(pi_device_binaries DeviceBinary) {
878  std::lock_guard<std::mutex> Guard(Sync::getGlobalLock());
879 
880  for (int I = 0; I < DeviceBinary->NumDeviceBinaries; I++) {
881  pi_device_binary RawImg = &(DeviceBinary->DeviceBinaries[I]);
882  OSModuleHandle M = OSUtil::getOSModuleHandle(RawImg);
883  const _pi_offload_entry EntriesB = RawImg->EntriesBegin;
884  const _pi_offload_entry EntriesE = RawImg->EntriesEnd;
885  auto Img = make_unique_ptr<RTDeviceBinaryImage>(RawImg, M);
886 
887  // Fill the kernel argument mask map
888  const pi::DeviceBinaryImage::PropertyRange &KPOIRange =
889  Img->getKernelParamOptInfo();
890  if (KPOIRange.isAvailable()) {
891  KernelNameToArgMaskMap &ArgMaskMap =
892  m_EliminatedKernelArgMasks[Img.get()];
893  for (const auto &Info : KPOIRange)
894  ArgMaskMap[Info->Name] =
895  createKernelArgMask(pi::DeviceBinaryProperty(Info).asByteArray());
896  }
897  // Use the entry information if it's available
898  if (EntriesB != EntriesE) {
899  // The kernel sets for any pair of images are either disjoint or
900  // identical, look up the kernel set using the first kernel name...
901  StrToKSIdMap &KSIdMap = m_KernelSets[M];
902  auto KSIdIt = KSIdMap.find(EntriesB->name);
903  if (KSIdIt != KSIdMap.end()) {
904  for (_pi_offload_entry EntriesIt = EntriesB + 1; EntriesIt != EntriesE;
905  ++EntriesIt)
906  assert(KSIdMap[EntriesIt->name] == KSIdIt->second &&
907  "Kernel sets are not disjoint");
908  auto &Imgs = m_DeviceImages[KSIdIt->second];
909  assert(Imgs && "Device image vector should have been already created");
910  Imgs->push_back(std::move(Img));
911  continue;
912  }
913  // ... or create the set first if it hasn't been
914  KernelSetId KSId = getNextKernelSetId();
915  for (_pi_offload_entry EntriesIt = EntriesB; EntriesIt != EntriesE;
916  ++EntriesIt) {
917  auto Result = KSIdMap.insert(std::make_pair(EntriesIt->name, KSId));
918  (void)Result;
919  assert(Result.second && "Kernel sets are not disjoint");
920  }
921  m_DeviceImages[KSId].reset(new std::vector<RTDeviceBinaryImageUPtr>());
922  m_DeviceImages[KSId]->push_back(std::move(Img));
923  continue;
924  }
925  // Otherwise assume that the image contains all kernels associated with the
926  // module
927  KernelSetId &KSId = m_OSModuleKernelSets[M];
928  if (KSId == 0)
929  KSId = getNextKernelSetId();
930 
931  auto &Imgs = m_DeviceImages[KSId];
932  if (!Imgs)
933  Imgs.reset(new std::vector<RTDeviceBinaryImageUPtr>());
934  Imgs->push_back(std::move(Img));
935  }
936 }
