76 inline __SYCL_ALWAYS_INLINE void

79  multi_ptr<T, Space> src, size_t stride, matrix_layout MemL) {
80 #ifdef __SYCL_DEVICE_ONLY__
81  T *Ptr = src.get();
82  switch (MemL) {
83  default:

88  spv_matrix_layout_traits<Layout>::value>(
89  Ptr, stride, __spv::MatrixLayout::RowMajor,
90  spv_scope_traits<Group>::value);
91  break;
92  case matrix_layout::col_major:
93  res.spvm =
94  __spirv_JointMatrixLoadINTEL<T, NumRows, NumCols,
95  spv_matrix_layout_traits<Layout>::value>(
96  Ptr, stride, __spv::MatrixLayout::ColumnMajor,
97  spv_scope_traits<Group>::value);
98  break;
99  case matrix_layout::packed_a:
100  res.spvm =
101  __spirv_JointMatrixLoadINTEL<T, NumRows, NumCols,
102  spv_matrix_layout_traits<Layout>::value>(
103  Ptr, stride, __spv::MatrixLayout::PackedA,
104  spv_scope_traits<Group>::value);
105  break;
106  case matrix_layout::packed_b:
107  res.spvm =
108  __spirv_JointMatrixLoadINTEL<T, NumRows, NumCols,
109  spv_matrix_layout_traits<Layout>::value>(
110  Ptr, stride, __spv::MatrixLayout::PackedB,
111  spv_scope_traits<Group>::value);
112  break;
113  }
114 #else
115  (void)sg;
116  (void)res;
117  (void)src;

173 template <typename Group, typename T1, typename T2, typename T3, size_t M,
174  size_t K, size_t N, matrix_layout LayoutA, matrix_layout LayoutB,
175  matrix_layout LayoutC>
176 inline __SYCL_ALWAYS_INLINE joint_matrix<T3, M, N, LayoutC, Group>
177 joint_matrix_mad(Group sg, joint_matrix<T1, M, K, LayoutA, Group> &mA,
178  joint_matrix<T2, K, N, LayoutB, Group> &mB,
179  joint_matrix<T3, M, N, LayoutC, Group> &mC) {
180 #ifdef __SYCL_DEVICE_ONLY__
181  joint_matrix<T3, M, N, LayoutC, Group> res(sg);
182  if constexpr (std::is_same<T1, uint16_t>::value &&
183  std::is_same<T2, uint16_t>::value &&
184  std::is_same<T3, float>::value)
185  res.spvm = __spirv_JointMatrixMadINTEL(mA.spvm, mB.spvm, mC.spvm);
186  else if constexpr (std::is_unsigned<T1>::value && std::is_unsigned<T2>::value)
187  res.spvm = __spirv_JointMatrixUUMadINTEL(mA.spvm, mB.spvm, mC.spvm);
188  else if constexpr (std::is_signed<T1>::value && std::is_unsigned<T2>::value)
189  res.spvm = __spirv_JointMatrixSUMadINTEL(mA.spvm, mB.spvm, mC.spvm);
190  else if constexpr (std::is_unsigned<T1>::value && std::is_signed<T2>::value)
191  res.spvm = __spirv_JointMatrixUSMadINTEL(mA.spvm, mB.spvm, mC.spvm);
192  else
193  res.spvm = __spirv_JointMatrixMadINTEL(mA.spvm, mB.spvm, mC.spvm);
194  return res;
195 #else
196  (void)sg;
197  (void)mA;
198  (void)mB;
199  (void)mC;
200  throw runtime_error("joint matrix is not supported on host device.",
201  PI_INVALID_DEVICE);
202 #endif // __SYCL_DEVICE_ONLY__
203 }

304 // Note that similarly to the other matrix functions, uint16_t is used here to
305 // represent bf16 type. Since the AMX and DPAS implementations don't support
306 // uint16_t, this interpretation is possible. This design choice was made before
307 // the introduction of SYCL experimental bfloat16 type. Our plan is to move
308 // towards using the SYCL bfloat16. But since it is still experimental, we will
309 // probably keep both uint16 interpretation and SYCL bfloat16.
310 template <size_t NumRows, size_t NumCols, matrix_layout Layout, typename Group>
311 class wi_element<uint16_t, NumRows, NumCols, Layout, Group> {

362  // We use here the following functions for conversion (bf16=>fp32 and
363  // fp32=>bf16). This is a workaround until we are able to use
364  // __spirv_ConvertFToBF16INTEL and __spirv_ConvertBF16ToFINTEL once these are
365  // supported in the CPU backend
366  static float make_fp32(uint16_t x) {
367  unsigned int y = x;
368  y = y << 16;
369  float *res = reinterpret_cast<float *>(&y);
370  return *res;
371  }
372 
373  static uint16_t make_bf16(float x) {
374  int *res = reinterpret_cast<int *>(&x);
375  *res = *res >> 16;
376  return (uint16_t)*res;
377  }
