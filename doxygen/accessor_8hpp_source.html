26 // The file contains implementations of accessor class. Objects of accessor
27 // class define a requirement to access some SYCL memory object or local memory
28 // of the device.
29 //
30 // Basically there are 3 distinct types of accessors.
31 //
32 // One of them is an accessor to a SYCL buffer object(Buffer accessor) which has
33 // the richest interface. It supports things like accessing only a part of
34 // buffer, multidimensional access using sycl::id, conversions to various
35 // multi_ptr and atomic classes.
36 //
37 // Second type is an accessor to a SYCL image object(Image accessor) which has
38 // "image" specific methods for reading and writing.
39 //
40 // Finally, accessor to local memory(Local accessor) doesn't require access to
41 // any SYCL memory object, but asks for some local memory on device to be
42 // available. Some methods overlap with ones that "Buffer accessor" provides.
43 //
44 // Buffer and Image accessors create the requirement to access some SYCL memory
45 // object(or part of it). SYCL RT must detect when two kernels want to access
46 // the same memory objects and make sure they are executed in correct order.
47 //
48 // "accessor_common" class that contains several common methods between Buffer
49 // and Local accessors.
50 //
51 // Accessors have different representation on host and on device. On host they
52 // have non-templated base class, that is needed to safely work with any
53 // accessor type. Furhermore on host we need some additional fields in order
54 // to implement functionality required by Specification, for example during
55 // lifetime of a host accessor other operations with memory object the accessor
56 // refers to should be blocked and when all references to the host accessor are
57 // desctructed, the memory this host accessor refers to should be "written
58 // back".
59 //
60 // The scheme of inheritance for host side:
61 //
62 // +------------------+ +-----------------+ +-----------------------+
63 // | | | | | |
64 // | AccessorBaseHost | | accessor_common | | LocalAccessorBaseHost |
65 // | | | | | |
66 // +------------------+ +-----+-----------+ +--------+--------------+
67 // | | | | |
68 // | +-----------+ +----+ +---------+ +------+
69 // | | | | |
70 // v v v v v
71 // +----------------+ +-----------------+ +-------------+
72 // | | | accessor(1) | | accessor(3) |
73 // | image_accessor | +-----------------| +-------------+
74 // | | | for targets: | | for target: |
75 // +---+---+---+----+ | | | |
76 // | | | | host_buffer | | local |
77 // | | | | global_buffer | +-------------+
78 // | | | | constant_buffer |
79 // | | | +-----------------+
80 // | | |
81 // | | +------------------------------------+
82 // | | |
83 // | +----------------------+ |
84 // v v v
85 // +-----------------+ +--------------+ +-------------+
86 // | acessor(2) | | accessor(4) | | accessor(5) |
87 // +-----------------+ +--------------+ +-------------+
88 // | for targets: | | for targets: | | for target: |
89 // | | | | | |
90 // | host_image | | image | | image_array |
91 // +-----------------+ +--------------+ +-------------+
92 //
93 // For host side AccessorBaseHost/LocalAccessorBaseHost contains shared_ptr
94 // which points to AccessorImplHost/LocalAccessorImplHost object.
95 //
96 // The scheme of inheritance for device side:
97 //
98 // +-----------------+
99 // | |
100 // | accessor_common |
101 // | |
102 // +-----+-------+---+
103 // | |
104 // +----+ +-----+
105 // | |
106 // v v
107 // +----------------+ +-----------------+ +-------------+
108 // | | | accessor(1) | | accessor(3) |
109 // | image_accessor | +-----------------| +-------------+
110 // | | | for targets: | | for target: |
111 // +---+---+---+----+ | | | |
112 // | | | | host_buffer | | local |
113 // | | | | global_buffer | +-------------+
114 // | | | | constant_buffer |
115 // | | | +-----------------+
116 // | | |
117 // | | +------------------------------------+
118 // | | |
119 // | +----------------------+ |
120 // v v v
121 // +-----------------+ +--------------+ +-------------+
122 // | acessor(2) | | accessor(4) | | accessor(5) |
123 // +-----------------+ +--------------+ +-------------+
124 // | for targets: | | for targets: | | for target: |
125 // | | | | | |
126 // | host_image | | image | | image_array |
127 // +-----------------+ +--------------+ +-------------+
128 //
129 // For device side AccessorImplHost/LocalAccessorImplHost are fileds of
130 // accessor(1) and accessor(3).
131 //
132 // accessor(1) declares accessor as a template class and implements accessor
133 // class for access targets: host_buffer, global_buffer and constant_buffer.
134 //
135 // accessor(3) specializes accessor(1) for the local access target.
136 //
137 // image_accessor contains implements interfaces for access targets: host_image,
138 // image and image_array. But there are three distinct specializations of the
139 // accessor(1) (accessor(2), accessor(4), accessor(5)) that are just inherited
140 // from image_accessor.
141 //
142 // accessor_common contains several helpers common for both accessor(1) and
143 // accessor(3)

155 // The function extends or truncates number of dimensions of objects of id
156 // or ranges classes. When extending the new values are filled with
157 // DefaultValue, truncation just removes extra values.
158 template <int NewDim, int DefaultValue, template <int> class T, int OldDim>
159 static T<NewDim> convertToArrayOfN(T<OldDim> OldObj) {
160  T<NewDim> NewObj = InitializedVal<NewDim, T>::template get<0>();
161  const int CopyDims = NewDim > OldDim ? OldDim : NewDim;
162  for (int I = 0; I < CopyDims; ++I)
163  NewObj[I] = OldObj[I];
164  for (int I = CopyDims; I < NewDim; ++I)
165  NewObj[I] = DefaultValue;
166  return NewObj;
167 }

203  // The class which allows to access value of N dimensional accessor using N
204  // subscript operators, e.g. accessor[2][2][3]
205  template <int SubDims> class AccessorSubscript {
206  static constexpr int Dims = Dimensions;
207 
208  mutable id<Dims> MIDs;
209  AccType MAccessor;
210 
211  public:
212  AccessorSubscript(AccType Accessor, id<Dims> IDs)
213  : MAccessor(Accessor), MIDs(IDs) {}
214 
215  // Only accessor class is supposed to use this c'tor for the first
216  // operator[].
217  AccessorSubscript(AccType Accessor, size_t Index) : MAccessor(Accessor) {
218  MIDs[0] = Index;
219  }
220 
221  template <int CurDims = SubDims>
222  typename detail::enable_if_t<(CurDims > 1), AccessorSubscript<CurDims - 1>>
223  operator[](size_t Index) {
224  MIDs[Dims - CurDims] = Index;

226  }

228  template <int CurDims = SubDims,
229  typename = detail::enable_if_t<CurDims == 1 && IsAccessAnyWrite>>
230  RefType operator[](size_t Index) const {
231  MIDs[Dims - CurDims] = Index;

236  typename = detail::enable_if_t<CurDims == 1 && IsAccessReadOnly>>
237  DataT operator[](size_t Index) const {
238  MIDs[Dims - SubDims] = Index;

554 template <typename DataT, int Dimensions, access::mode AccessMode,
555  access::placeholder IsPlaceholder>
556 class __image_array_slice__ {
557 
558  static_assert(Dimensions < 3,
559  "Image slice cannot have more then 2 dimensions");
560 
561  constexpr static int AdjustedDims = (Dimensions == 2) ? 4 : Dimensions + 1;
562 
563  template <typename CoordT,
564  typename CoordElemType =
565  typename detail::TryToGetElementType<CoordT>::type>
566  sycl::vec<CoordElemType, AdjustedDims>
567  getAdjustedCoords(const CoordT &Coords) const {
568  CoordElemType LastCoord = 0;
569 
570  if (std::is_same<float, CoordElemType>::value) {
571  sycl::vec<int, Dimensions + 1> Size = MBaseAcc.getRangeInternal();
572  LastCoord =
573  MIdx / static_cast<float>(Size.template swizzle<Dimensions>());
574  } else {
575  LastCoord = MIdx;
576  }
577 
578  sycl::vec<CoordElemType, Dimensions> LeftoverCoords{LastCoord};
579  sycl::vec<CoordElemType, AdjustedDims> AdjustedCoords{Coords,
580  LeftoverCoords};
581  return AdjustedCoords;
582  }
583 
584 public:
585  __image_array_slice__(accessor<DataT, Dimensions, AccessMode,
586  access::target::image_array, IsPlaceholder>
587  BaseAcc,
588  size_t Idx)
589  : MBaseAcc(BaseAcc), MIdx(Idx) {}
590
