353  public:

357  // Only accessor class is supposed to use this c'tor for the first

398  if constexpr (std::is_same_v<MayBeTag1, mode_tag_t<access::mode::read>> ||
399  std::is_same_v<MayBeTag2, mode_tag_t<access::mode::read>>) {
400  return access::mode::read;
401  }
402 
403  if constexpr (std::is_same_v<MayBeTag1, mode_tag_t<access::mode::write>> ||
404  std::is_same_v<MayBeTag2, mode_tag_t<access::mode::write>>) {
405  return access::mode::write;
406  }
407 
408  if constexpr (std::is_same_v<
409  MayBeTag1,
410  mode_target_tag_t<access::mode::read,
411  access::target::constant_buffer>> ||
412  std::is_same_v<
413  MayBeTag2,
414  mode_target_tag_t<access::mode::read,
415  access::target::constant_buffer>>) {
416  return access::mode::read;
417  }
418 
419  if constexpr (std::is_same_v<MayBeTag1,
420  mode_target_tag_t<access::mode::read,
421  access::target::host_task>> ||
422  std::is_same_v<MayBeTag2,
423  mode_target_tag_t<access::mode::read,
424  access::target::host_task>>) {
425  return access::mode::read;
