220 // The function extends or truncates number of dimensions of objects of id
221 // or ranges classes. When extending the new values are filled with
222 // DefaultValue, truncation just removes extra values.
223 template <int NewDim, int DefaultValue, template <int> class T, int OldDim>
224 static T<NewDim> convertToArrayOfN(T<OldDim> OldObj) {
225  T<NewDim> NewObj = InitializedVal<NewDim, T>::template get<0>();
226  const int CopyDims = NewDim > OldDim ? OldDim : NewDim;
227  for (int I = 0; I < CopyDims; ++I)
228  NewObj[I] = OldObj[I];
229  for (int I = CopyDims; I < NewDim; ++I)
230  NewObj[I] = DefaultValue;
231  return NewObj;
232 }

268  // The class which allows to access value of N dimensional accessor using N
269  // subscript operators, e.g. accessor[2][2][3]
270  template <int SubDims> class AccessorSubscript {

275 
276  public:

278  : MAccessor(Accessor), MIDs(IDs) {}
279 
280  // Only accessor class is supposed to use this c'tor for the first
281  // operator[].
282  AccessorSubscript(AccType Accessor, size_t Index) : MAccessor(Accessor) {
283  MIDs[0] = Index;

296  MIDs[Dims - CurDims] = Index;

301  typename = detail::enable_if_t<CurDims == 1 && IsAccessReadOnly>>

638  constexpr static int AdjustedDims = (Dimensions == 2) ? 4 : Dimensions + 1;

640  template <typename CoordT,
641  typename CoordElemType =
642  typename detail::TryToGetElementType<CoordT>::type>
643  sycl::vec<CoordElemType, AdjustedDims>
644  getAdjustedCoords(const CoordT &Coords) const {
645  CoordElemType LastCoord = 0;
646 
647  if (std::is_same<float, CoordElemType>::value) {
648  sycl::vec<int, Dimensions + 1> Size = MBaseAcc.getRangeInternal();
649  LastCoord =
650  MIdx / static_cast<float>(Size.template swizzle<Dimensions>());
651  } else {
652  LastCoord = MIdx;
653  }
654 
655  sycl::vec<CoordElemType, Dimensions> LeftoverCoords{LastCoord};
656  sycl::vec<CoordElemType, AdjustedDims> AdjustedCoords{Coords,
657  LeftoverCoords};

659  }
