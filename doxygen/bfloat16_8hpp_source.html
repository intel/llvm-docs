333  // The automatic rounding mode is RTE.
334  enum SYCLRoundingMode { automatic = 0, rte = 1, rtz = 2, rtp = 3, rtn = 4 };

336  // Function to get the most significant bit position of a number.
337  template <typename Ty> static size_t get_msb_pos(const Ty &x) {
338  assert(x != 0);
339  size_t idx = 0;
340  Ty mask = ((Ty)1 << (sizeof(Ty) * 8 - 1));
341  for (idx = 0; idx < (sizeof(Ty) * 8); ++idx) {
342  if ((x & mask) == mask)
343  break;
344  mask >>= 1;
345  }
346 
347  return (sizeof(Ty) * 8 - 1 - idx);
348  }

350  // Helper function to get BF16 from float with different rounding modes.
351  // Reference:
352  // https://github.com/intel/llvm/blob/sycl/libdevice/imf_bf16.hpp#L30
353  static bfloat16
354  getBFloat16FromFloatWithRoundingMode(const float &f,
355  SYCLRoundingMode roundingMode) {
356 
357  if (roundingMode == SYCLRoundingMode::automatic ||
358  roundingMode == SYCLRoundingMode::rte) {
359  // Use the default rounding mode.
360  return bfloat16{f};
361  } else {
362  uint32_t u32_val = sycl::bit_cast<uint32_t>(f);
363  uint16_t bf16_sign = static_cast<uint16_t>((u32_val >> 31) & 0x1);
364  uint16_t bf16_exp = static_cast<uint16_t>((u32_val >> 23) & 0x7FF);
365  uint32_t f_mant = u32_val & 0x7F'FFFF;
366  uint16_t bf16_mant = static_cast<uint16_t>(f_mant >> 16);
367  // +/-infinity and NAN
368  if (bf16_exp == 0xFF) {
369  if (!f_mant)
370  return bitsToBfloat16(bf16_sign ? 0xFF80 : 0x7F80);
371  else
372  return bitsToBfloat16((bf16_sign << 15) | (bf16_exp << 7) |
373  bf16_mant);
374  }
375 
376  // +/-0
377  if (!bf16_exp && !f_mant) {
378  return bitsToBfloat16(bf16_sign ? 0x8000 : 0x0);
379  }
380 
381  uint16_t mant_discard = static_cast<uint16_t>(f_mant & 0xFFFF);
382  switch (roundingMode) {
383  case SYCLRoundingMode::rtn:
384  if (bf16_sign && mant_discard)
385  bf16_mant++;
386  break;
387  case SYCLRoundingMode::rtz:
388  break;
389  case SYCLRoundingMode::rtp:
390  if (!bf16_sign && mant_discard)
391  bf16_mant++;
392  break;
393 
394  // Should not reach here. Adding these just to suppress the warning.
395  case SYCLRoundingMode::automatic:
396  case SYCLRoundingMode::rte:
397  break;
398  }
399 
400  // if overflow happens, bf16_exp will be 0xFF and bf16_mant will be 0,
401  // infinity will be returned.
402  if (bf16_mant == 0x80) {
403  bf16_mant = 0;
404  bf16_exp++;
405  }
406 
407  return bitsToBfloat16((bf16_sign << 15) | (bf16_exp << 7) | bf16_mant);
408  }
409  }

411  // Helper function to get BF16 from unsigned integral data types
412  // with different rounding modes.
413  // Reference:
414  // https://github.com/intel/llvm/blob/sycl/libdevice/imf_bf16.hpp#L302
415  template <typename T>
416  static bfloat16
417  getBFloat16FromUIntegralWithRoundingMode(T &u,
418  SYCLRoundingMode roundingMode) {
419 
420  size_t msb_pos = get_msb_pos(u);
421  // return half representation for 1
422  if (msb_pos == 0)
423  return bitsToBfloat16(0x3F80);
424 
425  T mant = u & ((static_cast<T>(1) << msb_pos) - 1);
426  // Unsigned integral value can be represented by 1.mant * (2^msb_pos),
427  // msb_pos is also the bit number of mantissa, 0 < msb_pos < sizeof(Ty) * 8,
428  // exponent of bfloat16 precision value range is [-126, 127].
429 
430  uint16_t b_exp = msb_pos;
431  uint16_t b_mant;
432 
433  if (msb_pos <= 7) {
434  // No need to round off if we can losslessly fit the input value in
435  // mantissa of bfloat16.
436  mant <<= (7 - msb_pos);
437  b_mant = static_cast<uint16_t>(mant);
438  } else {
439  b_mant = static_cast<uint16_t>(mant >> (msb_pos - 7));
440  T mant_discard = mant & ((static_cast<T>(1) << (msb_pos - 7)) - 1);
441  T mid = static_cast<T>(1) << (msb_pos - 8);
442  switch (roundingMode) {
443  case SYCLRoundingMode::automatic:
444  case SYCLRoundingMode::rte:
445  if ((mant_discard > mid) ||
446  ((mant_discard == mid) && ((b_mant & 0x1) == 0x1)))
447  b_mant++;
448  break;
449  case SYCLRoundingMode::rtp:
450  if (mant_discard)
451  b_mant++;
452  break;
453  case SYCLRoundingMode::rtn:
454  case SYCLRoundingMode::rtz:
455  break;
456  }
457  }
458  if (b_mant == 0x80) {
459  b_exp++;
460  b_mant = 0;
461  }
462 
463  b_exp += 127;
464  return bitsToBfloat16((b_exp << 7) | b_mant);
465  }

467  // Helper function to get BF16 from signed integral data types.
468  // Reference:
469  // https://github.com/intel/llvm/blob/sycl/libdevice/imf_bf16.hpp#L353
470  template <typename T>
471  static bfloat16
472  getBFloat16FromSIntegralWithRoundingMode(T &i,
473  SYCLRoundingMode roundingMode) {
474  // Get unsigned type corresponding to T.
475  typedef typename std::make_unsigned_t<T> UTy;
476 
477  uint16_t b_sign = (i >= 0) ? 0 : 0x8000;
478  UTy ui = (i > 0) ? static_cast<UTy>(i) : static_cast<UTy>(-i);
479  size_t msb_pos = get_msb_pos<UTy>(ui);
480  if (msb_pos == 0)
481  return bitsToBfloat16(b_sign ? 0xBF80 : 0x3F80);
482  UTy mant = ui & ((static_cast<UTy>(1) << msb_pos) - 1);
483 
484  uint16_t b_exp = msb_pos;
485  uint16_t b_mant;
486  if (msb_pos <= 7) {
487  mant <<= (7 - msb_pos);
488  b_mant = static_cast<uint16_t>(mant);
489  } else {
490  b_mant = static_cast<uint16_t>(mant >> (msb_pos - 7));
491  T mant_discard = mant & ((static_cast<T>(1) << (msb_pos - 7)) - 1);
492  T mid = static_cast<T>(1) << (msb_pos - 8);
493  switch (roundingMode) {
494  case SYCLRoundingMode::automatic:
495  case SYCLRoundingMode::rte:
496  if ((mant_discard > mid) ||
497  ((mant_discard == mid) && ((b_mant & 0x1) == 0x1)))
498  b_mant++;
499  break;
500  case SYCLRoundingMode::rtp:
501  if (mant_discard && !b_sign)
502  b_mant++;
503  break;
504  case SYCLRoundingMode::rtn:
505  if (mant_discard && b_sign)
506  b_mant++;
507  case SYCLRoundingMode::rtz:
508  break;
509  }
510  }
511 
512  if (b_mant == 0x80) {
513  b_exp++;
514  b_mant = 0;
515  }
516  b_exp += 127;
517  return bitsToBfloat16(b_sign | (b_exp << 7) | b_mant);
518  }

520  // Helper function to get BF16 from double with RTE rounding modes.
521  // Reference:
522  // https://github.com/intel/llvm/blob/sycl/libdevice/imf_bf16.hpp#L79
523  static bfloat16 getBFloat16FromDoubleWithRTE(const double &d) {
524 
525  uint64_t u64_val = sycl::bit_cast<uint64_t>(d);
526  int16_t bf16_sign = (u64_val >> 63) & 0x1;
527  uint16_t fp64_exp = static_cast<uint16_t>((u64_val >> 52) & 0x7FF);
528  uint64_t fp64_mant = (u64_val & 0xF'FFFF'FFFF'FFFF);
529  uint16_t bf16_mant;
530  // handling +/-infinity and NAN for double input
531  if (fp64_exp == 0x7FF) {
532  if (!fp64_mant) {
533  return bf16_sign ? 0xFF80 : 0x7F80;
534  } else {
535  // returns a quiet NaN
536  return 0x7FC0;
537  }
538  }
539 
540  // Subnormal double precision is converted to 0
541  if (fp64_exp == 0) {
542  return bf16_sign ? 0x8000 : 0x0;
543  }
544 
545  fp64_exp -= 1023;
546  // handling overflow, convert to +/-infinity
547  if (static_cast<int16_t>(fp64_exp) > 127) {
548  return bf16_sign ? 0xFF80 : 0x7F80;
549  }
550 
551  // handling underflow
552  if (static_cast<int16_t>(fp64_exp) < -133) {
553  return bf16_sign ? 0x8000 : 0x0;
554  }
555 
556  //-133 <= fp64_exp <= 127, 1.signicand * 2^fp64_exp
557  // For these numbers, they are NOT subnormal double-precision numbers but
558  // will turn into subnormal when converting to bfloat16
559  uint64_t discard_bits;
560  if (static_cast<int16_t>(fp64_exp) < -126) {
561  fp64_mant |= 0x10'0000'0000'0000;
562  fp64_mant >>= -126 - static_cast<int16_t>(fp64_exp) - 1;
563  discard_bits = fp64_mant & 0x3FFF'FFFF'FFFF;
564  bf16_mant = static_cast<uint16_t>(fp64_mant >> 46);
565  if (discard_bits > 0x2000'0000'0000 ||
566  ((discard_bits == 0x2000'0000'0000) && ((bf16_mant & 0x1) == 0x1)))
567  bf16_mant += 1;
568  fp64_exp = 0;
569  if (bf16_mant == 0x80) {
570  bf16_mant = 0;
571  fp64_exp = 1;
572  }

574  }
575 
576  // For normal value, discard 45 bits from mantissa
577  discard_bits = fp64_mant & 0x1FFF'FFFF'FFFF;
578  bf16_mant = static_cast<uint16_t>(fp64_mant >> 45);
579  if (discard_bits > 0x1000'0000'0000 ||
580  ((discard_bits == 0x1000'0000'0000) && ((bf16_mant & 0x1) == 0x1)))
581  bf16_mant += 1;
582 
583  if (bf16_mant == 0x80) {
584  if (fp64_exp != 127) {
585  bf16_mant = 0;
586  fp64_exp++;
587  } else {
588  return bf16_sign ? 0xFF80 : 0x7F80;
589  }
590  }
591  fp64_exp += 127;
592

604

616  }

622  return getBFloat16FromFloatWithRoundingMode(static_cast<float>(a),

624  }

632  } else {
