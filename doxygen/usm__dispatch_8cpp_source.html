28 USMDispatcher::USMDispatcher(cl_platform_id platform,
29  const vector_class<RT::PiDevice> &DeviceIds) {
30  // Note: This function should be modified whenever a new BE is added.
31  // mSupported needs to be appropriately set to properly gate USM support.
32 
33  if (pi::useBackend(pi::Backend::SYCL_BE_PI_OPENCL)) {
34  GET_EXTENSION(clHostMemAllocINTEL);
35  GET_EXTENSION(clDeviceMemAllocINTEL);
36  GET_EXTENSION(clSharedMemAllocINTEL);
37  GET_EXTENSION(clMemFreeINTEL);
38  GET_EXTENSION(clGetMemAllocInfoINTEL);
39  GET_EXTENSION(clSetKernelArgMemPointerINTEL);
40  GET_EXTENSION(clEnqueueMemsetINTEL);
41  GET_EXTENSION(clEnqueueMemcpyINTEL);
42  GET_EXTENSION(clEnqueueMigrateMemINTEL);
43  GET_EXTENSION(clEnqueueMemAdviseINTEL);
44  mEmulated = !(pfn_clHostMemAllocINTEL && pfn_clDeviceMemAllocINTEL &&
45  pfn_clSharedMemAllocINTEL && pfn_clMemFreeINTEL &&
46  pfn_clSetKernelArgMemPointerINTEL &&
47  pfn_clEnqueueMemsetINTEL && pfn_clEnqueueMemcpyINTEL);
48  mEmulator.reset(new CLUSM());
49 
50  if (mEmulated) {
51  // See if every device in this context supports
52  // CL_DEVICE_SVM_FINE_GRAIN_BUFFER
53  // If not, disable USM
54 
55  if (CL_TARGET_OPENCL_VERSION >= 200) {
56  bool AnybodyNotSupportSVM = false;
57  for (const auto &D : DeviceIds) {
58  cl_device_svm_capabilities Caps;
59  cl_int Error = clGetDeviceInfo(
60  pi::cast<cl_device_id>(D), CL_DEVICE_SVM_CAPABILITIES,
61  sizeof(cl_device_svm_capabilities), &Caps, nullptr);
62  AnybodyNotSupportSVM |= ((Error != CL_SUCCESS) ||
63  (!(Caps & CL_DEVICE_SVM_FINE_GRAIN_BUFFER)));
64  }
65  mSupported = !AnybodyNotSupportSVM;
66  } else {
67  // USM isn't support on CL 1.2
68  mSupported = false;
69  }
70  } else {
71  // We support the CL Extension
72  mSupported = true;
73  }
74  } else {
75  mSupported = false;
76  }
77 }

80  cl_mem_properties_intel *Properties,
81  size_t Size, pi_uint32 Alignment,
82  pi_result *ErrcodeRet) {
83  void *RetVal = nullptr;
84 
85  if (pi::useBackend(pi::Backend::SYCL_BE_PI_OPENCL)) {
86  cl_context CLContext = pi::cast<cl_context>(Context);
87 
88  if (mEmulated) {
89  RetVal = mEmulator->hostMemAlloc(CLContext, Properties, Size, Alignment,
90  pi::cast<cl_int *>(ErrcodeRet));
91  } else {
92  RetVal = pfn_clHostMemAllocINTEL(CLContext, Properties, Size, Alignment,
93  pi::cast<cl_int *>(ErrcodeRet));
94  }
95  }
96 
97  if (ErrcodeRet && !RetVal) {
98  *ErrcodeRet = PI_INVALID_OPERATION;
99  }
100  return RetVal;
