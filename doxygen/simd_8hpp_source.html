96  if (Step == Ty())

99 #pragma unroll

101  M_data[i] = Val;

408  template <typename RTy>
409  ESIMD_INLINE void
410  writeRegion(RTy Region,
411  const detail::vector_type_t<typename RTy::element_type,
412  RTy::length> &Val) {
413  using ElemTy = typename RTy::element_type;
414  if constexpr (N * sizeof(Ty) == RTy::length * sizeof(ElemTy))
415  // update the entire vector
416  set(detail::bitcast<Ty, ElemTy, RTy::length>(Val));
417  else {
418  static_assert(!RTy::Is_2D);
419  // If element type differs, do bitcast conversion first.
420  auto Base = detail::bitcast<ElemTy, Ty, N>(data());
421  constexpr int BN = (N * sizeof(Ty)) / sizeof(ElemTy);
422  // Access the region information.
423  constexpr int M = RTy::Size_x;
424  constexpr int Stride = RTy::Stride_x;
425  uint16_t Offset = Region.M_offset_x * sizeof(ElemTy);
426 
427  // Merge and update.
428  auto Merged = __esimd_wrregion<ElemTy, BN, M,
429  /*VS*/ 0, M, Stride>(Base, Val, Offset);
430  // Convert back to the original element type, if needed.
431  set(detail::bitcast<Ty, ElemTy, BN>(Merged));
432  }
433  }

436  template <typename TR, typename UR>
437  ESIMD_INLINE void writeRegion(
438  std::pair<TR, UR> Region,
439  const detail::vector_type_t<typename TR::element_type, TR::length> &Val) {
440  // parent-region type
441  using PaTy = typename shape_type<UR>::type;
442  using ElemTy = typename TR::element_type;
443  using BT = typename PaTy::element_type;
444  constexpr int BN = PaTy::length;
445 
446  if constexpr (PaTy::Size_in_bytes == TR::Size_in_bytes) {
447  writeRegion(Region.second, detail::bitcast<BT, ElemTy, TR::length>(Val));
448  } else {
449  // Recursively read the base
450  auto Base = detail::readRegion<Ty, N>(data(), Region.second);
451  // If element type differs, do bitcast conversion first.
452  auto Base1 = detail::bitcast<ElemTy, BT, BN>(Base);
453  constexpr int BN1 = PaTy::Size_in_bytes / sizeof(ElemTy);
454 
455  if constexpr (!TR::Is_2D) {
456  // Access the region information.
457  constexpr int M = TR::Size_x;
458  constexpr int Stride = TR::Stride_x;
459  uint16_t Offset = Region.first.M_offset_x * sizeof(ElemTy);
460 
461  // Merge and update.
462  Base1 = __esimd_wrregion<ElemTy, BN1, M,
463  /*VS*/ 0, M, Stride>(Base1, Val, Offset);
464  } else {
465  static_assert(std::is_same<ElemTy, BT>::value);
466  // Read columns with non-trivial horizontal stride.
467  constexpr int M = TR::length;
468  constexpr int VS = PaTy::Size_x * TR::Stride_y;
469  constexpr int W = TR::Size_x;
470  constexpr int HS = TR::Stride_x;
471  constexpr int ParentWidth = PaTy::Size_x;
472 
473  // Compute the byte offset for the starting element.
474  uint16_t Offset = static_cast<uint16_t>(
475  (Region.first.M_offset_y * PaTy::Size_x + Region.first.M_offset_x) *
476  sizeof(ElemTy));
477 
478  // Merge and update.
479  Base1 = __esimd_wrregion<ElemTy, BN1, M, VS, W, HS, ParentWidth>(
480  Base1, Val, Offset);
481  }
482  // Convert back to the original element type, if needed.
483  auto Merged1 = detail::bitcast<BT, ElemTy, BN1>(Base1);
484  // recursively write it back to the base
485  writeRegion(Region.second, Merged1);
486  }
487  }
