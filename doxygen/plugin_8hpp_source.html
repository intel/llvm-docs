144  template <PiApiKind PiApiOffset, typename... ArgsT>
145  RT::PiResult call_nocheck(ArgsT... Args) const {
146  RT::PiFuncInfo<PiApiOffset> PiCallInfo;
147 #ifdef XPTI_ENABLE_INSTRUMENTATION
148  // Emit a function_begin trace for the PI API before the call is executed.
149  // If arguments need to be captured, then a data structure can be sent in
150  // the per_instance_user_data field.
151  const char *PIFnName = PiCallInfo.getFuncName();
152  uint64_t CorrelationID = pi::emitFunctionBeginTrace(PIFnName);
153  uint64_t CorrelationIDWithArgs = 0;
154  unsigned char *ArgsDataPtr = nullptr;
155  // TODO check if stream is observed when corresponding API is present.
156  if (xptiTraceEnabled()) {
157  auto ArgsData =
158  packCallArguments<PiApiOffset>(std::forward<ArgsT>(Args)...);
159  ArgsDataPtr = ArgsData.data();
160  CorrelationIDWithArgs = pi::emitFunctionWithArgsBeginTrace(
161  static_cast<uint32_t>(PiApiOffset), PIFnName, ArgsDataPtr, *MPlugin);
162  }
163 #endif
164  RT::PiResult R;
165  if (pi::trace(pi::TraceLevel::PI_TRACE_CALLS)) {
166  std::lock_guard<std::mutex> Guard(*TracingMutex);
167  const char *FnName = PiCallInfo.getFuncName();
168  std::cout << "---> " << FnName << "(" << std::endl;
169  RT::printArgs(Args...);
170  R = PiCallInfo.getFuncPtr(*MPlugin)(Args...);
171  std::cout << ") ---> ";
172  RT::printArgs(R);
173  RT::printOuts(Args...);
174  std::cout << std::endl;
175  } else {
176  R = PiCallInfo.getFuncPtr(*MPlugin)(Args...);
177  }
178 #ifdef XPTI_ENABLE_INSTRUMENTATION
179  // Close the function begin with a call to function end
180  pi::emitFunctionEndTrace(CorrelationID, PIFnName);
181  pi::emitFunctionWithArgsEndTrace(CorrelationIDWithArgs,
182  static_cast<uint32_t>(PiApiOffset),
183  PIFnName, ArgsDataPtr, R, *MPlugin);
184 #endif
185  return R;
186  }

209  // return the index of PiPlatforms.
210  // If not found, add it and return its index.
211  // The function is expected to be called in a thread safe manner.
212  int getPlatformId(RT::PiPlatform Platform) {
213  auto It = std::find(PiPlatforms.begin(), PiPlatforms.end(), Platform);
214  if (It != PiPlatforms.end())
215  return It - PiPlatforms.begin();
216 
217  PiPlatforms.push_back(Platform);
218  LastDeviceIds.push_back(0);
219  return PiPlatforms.size() - 1;
220  }
221 
222  // Device ids are consecutive across platforms within a plugin.
223  // We need to return the same starting index for the given platform.
224  // So, instead of returing the last device id of the given platform,
225  // return the last device id of the predecessor platform.
226  // The function is expected to be called in a thread safe manner.
227  int getStartingDeviceId(RT::PiPlatform Platform) {
228  int PlatformId = getPlatformId(Platform);
229  if (PlatformId == 0)
230  return 0;
231  return LastDeviceIds[PlatformId - 1];
232  }
