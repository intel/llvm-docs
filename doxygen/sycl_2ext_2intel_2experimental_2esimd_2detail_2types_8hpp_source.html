226 // Change the element type of a simd vector.
227 template <typename ToEltTy, typename FromEltTy, int FromN,
228  typename = csd::enable_if_t<is_vectorizable<ToEltTy>::value>>
229 ESIMD_INLINE typename csd::conditional_t<
230  std::is_same<FromEltTy, ToEltTy>::value, vector_type_t<FromEltTy, FromN>,
231  vector_type_t<ToEltTy,
232  bitcast_helper<ToEltTy, FromEltTy, FromN>::nToElems()>>
233 bitcast(vector_type_t<FromEltTy, FromN> Val) {
234  // Noop.
235  if constexpr (std::is_same<FromEltTy, ToEltTy>::value)
236  return Val;
237 
238  // Bitcast
239  constexpr int ToN = bitcast_helper<ToEltTy, FromEltTy, FromN>::nToElems();
240  using VTy = vector_type_t<ToEltTy, ToN>;
241  return reinterpret_cast<VTy>(Val);
242 }
