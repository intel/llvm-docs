48 /* Returns true if specified image should be cached on disk. It checks if
49  * cache is enabled, image has SPIRV type and matches thresholds. */
50 bool PersistentDeviceCodeCache::isImageCached(const RTDeviceBinaryImage &Img) {
51  // Cache shoould be enabled and image type should be SPIR-V
52  if (!isEnabled() || Img.getFormat() != PI_DEVICE_BINARY_TYPE_SPIRV)
53  return false;
54 
55  // Disable cache for ITT-profiled images.
56  if (SYCLConfig<INTEL_ENABLE_OFFLOAD_ANNOTATIONS>::get()) {
57  return false;
58  }
59 
60  static auto MaxImgSize = getNumParam<SYCL_CACHE_MAX_DEVICE_IMAGE_SIZE>(
61  DEFAULT_MAX_DEVICE_IMAGE_SIZE);
62  static auto MinImgSize = getNumParam<SYCL_CACHE_MIN_DEVICE_IMAGE_SIZE>(
63  DEFAULT_MIN_DEVICE_IMAGE_SIZE);
64 
65  // Make sure that image size is between caching thresholds if they are set.
66  // Zero values for threshold is treated as disabled threshold.
67  if ((MaxImgSize && (Img.getSize() > MaxImgSize)) ||
68  (MinImgSize && (Img.getSize() < MinImgSize)))
69  return false;
70 
71  return true;
72 }

74 /* Stores built program in persisten cache
75  */
76 void PersistentDeviceCodeCache::putItemToDisc(
77  const device &Device, const RTDeviceBinaryImage &Img,
78  const SerializedObj &SpecConsts, const std::string &BuildOptionsString,
79  const RT::PiProgram &NativePrg) {
80 
81  if (!isImageCached(Img))
82  return;
83 
84  std::string DirName =
85  getCacheItemPath(Device, Img, SpecConsts, BuildOptionsString);
86 
87  if (DirName.empty())
88  return;
89 
90  auto Plugin = detail::getSyclObjImpl(Device)->getPlugin();
91 
92  size_t i = 0;
93  std::string FileName;
94  do {
95  FileName = DirName + "/" + std::to_string(i++);
96  } while (OSUtil::isPathPresent(FileName + ".bin"));
97 
98  unsigned int DeviceNum = 0;
99 
100  Plugin.call<PiApiKind::piProgramGetInfo>(
101  NativePrg, PI_PROGRAM_INFO_NUM_DEVICES, sizeof(DeviceNum), &DeviceNum,
102  nullptr);
103 
104  std::vector<size_t> BinarySizes(DeviceNum);
105  Plugin.call<PiApiKind::piProgramGetInfo>(
106  NativePrg, PI_PROGRAM_INFO_BINARY_SIZES,
107  sizeof(size_t) * BinarySizes.size(), BinarySizes.data(), nullptr);
108 
109  std::vector<std::vector<char>> Result;
110  std::vector<char *> Pointers;
111  for (size_t I = 0; I < BinarySizes.size(); ++I) {
112  Result.emplace_back(BinarySizes[I]);
113  Pointers.push_back(Result[I].data());
114  }
115 
116  Plugin.call<PiApiKind::piProgramGetInfo>(NativePrg, PI_PROGRAM_INFO_BINARIES,
117  sizeof(char *) * Pointers.size(),
118  Pointers.data(), nullptr);
119 
120  try {
121  OSUtil::makeDir(DirName.c_str());
122  LockCacheItem Lock{FileName};
123  if (Lock.isOwned()) {
124  std::string FullFileName = FileName + ".bin";
125  writeBinaryDataToFile(FullFileName, Result);
126  trace("device binary has been cached: " + FullFileName);
127  writeSourceItem(FileName + ".src", Device, Img, SpecConsts,
128  BuildOptionsString);
129  }
130  } catch (...) {
131  // If a problem happens on storing cache item, do nothing
132  }
133 }

135 /* Program binaries built for one or more devices are read from persistent
136  * cache and returned in form of vector of programs. Each binary program is
137  * stored in vector of chars.
138  */
139 std::vector<std::vector<char>> PersistentDeviceCodeCache::getItemFromDisc(
140  const device &Device, const RTDeviceBinaryImage &Img,
141  const SerializedObj &SpecConsts, const std::string &BuildOptionsString) {
142 
143  if (!isImageCached(Img))
144  return {};
145 
146  std::string Path =
147  getCacheItemPath(Device, Img, SpecConsts, BuildOptionsString);
148 
149  if (Path.empty() || !OSUtil::isPathPresent(Path))
150  return {};
151 
152  int i = 0;
153 
154  std::string FileName{Path + "/" + std::to_string(i)};
155  while (OSUtil::isPathPresent(FileName + ".bin") ||
156  OSUtil::isPathPresent(FileName + ".src")) {
157 
158  if (!LockCacheItem::isLocked(FileName) &&
159  isCacheItemSrcEqual(FileName + ".src", Device, Img, SpecConsts,
160  BuildOptionsString)) {
161  try {
162  std::string FullFileName = FileName + ".bin";
163  std::vector<std::vector<char>> res =
164  readBinaryDataFromFile(FullFileName);
165  trace("using cached device binary: " + FullFileName);
166  return res; // subject for NRVO
167  } catch (...) {
168  // If read was unsuccessfull try the next item
169  }
170  }
171  FileName = Path + "/" + std::to_string(++i);
172  }
173  return {};
174 }

185 /* Write built binary to persistent cache
186  * Format: numImages, 1stImageSize, Image[, NthImageSize, NthImage...]
187  * Return on first unsuccessfull file operation
188  */
189 void PersistentDeviceCodeCache::writeBinaryDataToFile(
190  const std::string &FileName, const std::vector<std::vector<char>> &Data) {
191  std::ofstream FileStream{FileName, std::ios::binary};
192 
193  size_t Size = Data.size();
194  FileStream.write((char *)&Size, sizeof(Size));
195 
196  for (size_t i = 0; i < Data.size(); ++i) {
197  Size = Data[i].size();
198  FileStream.write((char *)&Size, sizeof(Size));
199  FileStream.write(Data[i].data(), Size);
200  }
201  FileStream.close();
202  if (FileStream.fail())
203  trace("Failed to write binary file " + FileName);
204 }

209 std::vector<std::vector<char>>
210 PersistentDeviceCodeCache::readBinaryDataFromFile(const std::string &FileName) {
211  std::ifstream FileStream{FileName, std::ios::binary};
212  size_t ImgNum = 0, ImgSize = 0;
213  FileStream.read((char *)&ImgNum, sizeof(ImgNum));
214 
215  std::vector<std::vector<char>> Res(ImgNum);
216  for (size_t i = 0; i < ImgNum; ++i) {
217  FileStream.read((char *)&ImgSize, sizeof(ImgSize));
218 
219  std::vector<char> ImgData(ImgSize);
220  FileStream.read(ImgData.data(), ImgSize);
221 
222  Res[i] = std::move(ImgData);
223  }
224  FileStream.close();
225 
226  if (FileStream.fail()) {
227  trace("Failed to read binary file from " + FileName);
228  return {};
229  }
230 
231  return Res;
232 }

238 void PersistentDeviceCodeCache::writeSourceItem(
239  const std::string &FileName, const device &Device,
240  const RTDeviceBinaryImage &Img, const SerializedObj &SpecConsts,
241  const std::string &BuildOptionsString) {
242  std::ofstream FileStream{FileName, std::ios::binary};
243 
244  std::string DeviceString{getDeviceIDString(Device)};
245  size_t Size = DeviceString.size();
246  FileStream.write((char *)&Size, sizeof(Size));
247  FileStream.write(DeviceString.data(), Size);
248 
249  Size = BuildOptionsString.size();
250  FileStream.write((char *)&Size, sizeof(Size));
251  FileStream.write(BuildOptionsString.data(), Size);
252 
253  Size = SpecConsts.size();
254  FileStream.write((char *)&Size, sizeof(Size));
255  FileStream.write((const char *)SpecConsts.data(), Size);
256 
257  Size = Img.getSize();
258  FileStream.write((char *)&Size, sizeof(Size));
259  FileStream.write((const char *)Img.getRawData().BinaryStart, Size);
260  FileStream.close();
261 
262  if (FileStream.fail()) {
263  trace("Failed to write source file to " + FileName);
264  }

267 /* Check that cache item key sources are equal to the current program.
268  * If file read operations fail cache item is treated as not equal.
269  */
270 bool PersistentDeviceCodeCache::isCacheItemSrcEqual(
271  const std::string &FileName, const device &Device,
272  const RTDeviceBinaryImage &Img, const SerializedObj &SpecConsts,
273  const std::string &BuildOptionsString) {
274  std::ifstream FileStream{FileName, std::ios::binary};
275 
276  std::string ImgString{(const char *)Img.getRawData().BinaryStart,
277  Img.getSize()};
278  std::string SpecConstsString{(const char *)SpecConsts.data(),
279  SpecConsts.size()};
280 
281  size_t Size = 0;
282  FileStream.read((char *)&Size, sizeof(Size));
283  std::string res(Size, '\0');
284  FileStream.read(&res[0], Size);
285  if (getDeviceIDString(Device).compare(res))
286  return false;
287 
288  FileStream.read((char *)&Size, sizeof(Size));
289  res.resize(Size);
290  FileStream.read(&res[0], Size);
291  if (BuildOptionsString.compare(res))
292  return false;
293 
294  FileStream.read((char *)&Size, sizeof(Size));
295  res.resize(Size);
296  FileStream.read(&res[0], Size);
297  if (SpecConstsString.compare(res))
298  return false;
299 
300  FileStream.read((char *)&Size, sizeof(Size));
301  res.resize(Size);
302  FileStream.read(&res[0], Size);
303  if (ImgString.compare(res))
304  return false;
305 
306  FileStream.close();
307 
308  if (FileStream.fail()) {
309  trace("Failed to read source file from " + FileName);
310  }
311 
312  return true;
313 }

318 std::string PersistentDeviceCodeCache::getCacheItemPath(
319  const device &Device, const RTDeviceBinaryImage &Img,
320  const SerializedObj &SpecConsts, const std::string &BuildOptionsString) {
321  static std::string cache_root{getRootDir()};
322  if (cache_root.empty()) {
323  trace("Disable persistent cache due to unconfigured cache root.");
324  return {};
325  }
326 
327  std::string ImgString{(const char *)Img.getRawData().BinaryStart,
328  Img.getSize()};
329  std::string DeviceString{getDeviceIDString(Device)};
330  std::string SpecConstsString{(const char *)SpecConsts.data(),
331  SpecConsts.size()};
332  std::hash<std::string> StringHasher{};
333 
334  return cache_root + "/" + std::to_string(StringHasher(DeviceString)) + "/" +
335  std::to_string(StringHasher(ImgString)) + "/" +
336  std::to_string(StringHasher(SpecConstsString)) + "/" +
337  std::to_string(StringHasher(BuildOptionsString));
338 }

347 // Parses persistent cache configuration and checks it for errors.
348 // Returns true if it is enabled, false otherwise.
349 static bool parsePersistentCacheConfig() {
350  constexpr bool Default = false; // default is disabled
351 
352  // Check if deprecated opt-out env var is used, then warn.
353  if (SYCLConfig<SYCL_CACHE_DISABLE_PERSISTENT>::get()) {
354  std::cerr
355  << "WARNING: " << SYCLConfig<SYCL_CACHE_DISABLE_PERSISTENT>::getName()
356  << " environment variable is deprecated "
357  << "and has no effect. By default, persistent device code caching is "
358  << (Default ? "enabled." : "disabled.") << " Use "
359  << SYCLConfig<SYCL_CACHE_PERSISTENT>::getName()
360  << "=1/0 to enable/disable.\n";
361  }
362  bool Ret = Default;
363  const char *RawVal = SYCLConfig<SYCL_CACHE_PERSISTENT>::get();
364 
365  if (RawVal) {
366  if (!std::strcmp(RawVal, "0")) {
367  Ret = false;
368  } else if (!std::strcmp(RawVal, "1")) {
369  Ret = true;
370  } else {
371  std::string Msg =
372  std::string{"Invalid value for bool configuration variable "} +
373  SYCLConfig<SYCL_CACHE_PERSISTENT>::getName() + std::string{": "} +
374  RawVal;
375  throw runtime_error(Msg, PI_INVALID_OPERATION);
376  }
377  }
378  PersistentDeviceCodeCache::trace(Ret ? "enabled" : "disabled");
379  return Ret;
380 }

389 /* Returns path for device code cache root directory
390  * If environment variables are not available return an empty string to identify
391  * that cache is not available.
392  */
393 std::string PersistentDeviceCodeCache::getRootDir() {
394  static const char *RootDir = SYCLConfig<SYCL_CACHE_DIR>::get();
395  if (RootDir)
396  return RootDir;
397 
398  constexpr char DeviceCodeCacheDir[] = "/libsycl_cache";
399 
400  // Use static to calculate directory only once per program run
401 #if defined(__SYCL_RT_OS_LINUX)
402  static const char *CacheDir = std::getenv("XDG_CACHE_HOME");
403  static const char *HomeDir = std::getenv("HOME");
404  if (!CacheDir && !HomeDir)
405  return {};
406  static std::string Res{
407  std::string(CacheDir ? CacheDir : (std::string(HomeDir) + "/.cache")) +
408  DeviceCodeCacheDir};
409 #else
410  static const char *AppDataDir = std::getenv("AppData");
411  if (!AppDataDir)
412  return {};
413  static std::string Res{std::string(AppDataDir) + DeviceCodeCacheDir};
414 #endif
415  return Res;
416 }
