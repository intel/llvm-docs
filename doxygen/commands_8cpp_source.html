190 class DispatchHostTask {
191  ExecCGCommand *MThisCmd;
192  std::vector<interop_handle::ReqToMem> MReqToMem;
193 
194  pi_result waitForEvents() const {
195  std::map<const detail::plugin *, std::vector<EventImplPtr>>
196  RequiredEventsPerPlugin;
197 
198  for (const EventImplPtr &Event : MThisCmd->MPreparedDepsEvents) {
199  const detail::plugin &Plugin = Event->getPlugin();
200  RequiredEventsPerPlugin[&Plugin].push_back(Event);
201  }
202 
203  // wait for dependency device events
204  // FIXME Current implementation of waiting for events will make the thread
205  // 'sleep' until all of dependency events are complete. We need a bit more
206  // sophisticated waiting mechanism to allow to utilize this thread for any
207  // other available job and resume once all required events are ready.
208  for (auto &PluginWithEvents : RequiredEventsPerPlugin) {
209  std::vector<RT::PiEvent> RawEvents = getPiEvents(PluginWithEvents.second);
210  try {
211  PluginWithEvents.first->call<PiApiKind::piEventsWait>(RawEvents.size(),
212  RawEvents.data());
213  } catch (const sycl::exception &E) {
214  CGHostTask &HostTask = static_cast<CGHostTask &>(MThisCmd->getCG());
215  HostTask.MQueue->reportAsyncException(std::current_exception());
216  return (pi_result)E.get_cl_code();
217  } catch (...) {
218  CGHostTask &HostTask = static_cast<CGHostTask &>(MThisCmd->getCG());
219  HostTask.MQueue->reportAsyncException(std::current_exception());
220  return PI_ERROR_UNKNOWN;
221  }
222  }
223 
224  // Wait for dependency host events.
225  // Host events can't throw exceptions so don't try to catch it.
226  for (const EventImplPtr &Event : MThisCmd->MPreparedHostDepsEvents) {
227  Event->waitInternal();
228  }
229 
230  return PI_SUCCESS;
231  }
232 
233 public:
234  DispatchHostTask(ExecCGCommand *ThisCmd,
235  std::vector<interop_handle::ReqToMem> ReqToMem)
236  : MThisCmd{ThisCmd}, MReqToMem(std::move(ReqToMem)) {}
237 
238  void operator()() const {
239  assert(MThisCmd->getCG().getType() == CG::CGTYPE::CodeplayHostTask);
240 
241  CGHostTask &HostTask = static_cast<CGHostTask &>(MThisCmd->getCG());
242 
243  pi_result WaitResult = waitForEvents();
244  if (WaitResult != PI_SUCCESS) {
245  std::exception_ptr EPtr = std::make_exception_ptr(sycl::runtime_error(
246  std::string("Couldn't wait for host-task's dependencies"),
247  WaitResult));
248  HostTask.MQueue->reportAsyncException(EPtr);
249 
250  // reset host-task's lambda and quit
251  HostTask.MHostTask.reset();
252  return;
253  }
254 
255  try {
256  // we're ready to call the user-defined lambda now
257  if (HostTask.MHostTask->isInteropTask()) {
258  interop_handle IH{MReqToMem, HostTask.MQueue,
259  HostTask.MQueue->getDeviceImplPtr(),
260  HostTask.MQueue->getContextImplPtr()};
261 
262  HostTask.MHostTask->call(IH);
263  } else
264  HostTask.MHostTask->call();
265  } catch (...) {
266  HostTask.MQueue->reportAsyncException(std::current_exception());
267  }
268 
269  HostTask.MHostTask.reset();
270 
271  // unblock user empty command here
272  EmptyCommand *EmptyCmd = MThisCmd->MEmptyCmd;
273  assert(EmptyCmd && "No empty command found");
274 
275  // Completing command's event along with unblocking enqueue readiness of
276  // empty command may lead to quick deallocation of MThisCmd by some cleanup
277  // process. Thus we'll copy deps prior to completing of event and unblocking
278  // of empty command.
279  // Also, it's possible to have record deallocated prior to enqueue process.
280  // Thus we employ read-lock of graph.
281  std::vector<Command *> ToCleanUp;
282  Scheduler &Sched = Scheduler::getInstance();
283  {
284  Scheduler::ReadLockT Lock(Sched.MGraphLock);
285 
286  std::vector<DepDesc> Deps = MThisCmd->MDeps;
287 
288  // update self-event status
289  MThisCmd->MEvent->setComplete();
290 
291  EmptyCmd->MEnqueueStatus = EnqueueResultT::SyclEnqueueReady;
292 
293  for (const DepDesc &Dep : Deps)
294  Scheduler::enqueueLeavesOfReqUnlocked(Dep.MDepRequirement, ToCleanUp);
295  }
296  Sched.cleanupCommands(ToCleanUp);
297  }
298 };

305 void Command::waitForEvents(QueueImplPtr Queue,
306  std::vector<EventImplPtr> &EventImpls,
307  RT::PiEvent &Event) {
308 
309  if (!EventImpls.empty()) {
310  if (Queue->is_host()) {
311  // Host queue can wait for events from different contexts, i.e. it may
312  // contain events with different contexts in its MPreparedDepsEvents.
313  // OpenCL 2.1 spec says that clWaitForEvents will return
314  // CL_INVALID_CONTEXT if events specified in the list do not belong to
315  // the same context. Thus we split all the events into per-context map.
316  // An example. We have two queues for the same CPU device: Q1, Q2. Thus
317  // we will have two different contexts for the same CPU device: C1, C2.
318  // Also we have default host queue. This queue is accessible via
319  // Scheduler. Now, let's assume we have three different events: E1(C1),
320  // E2(C1), E3(C2). Also, we have an EmptyCommand which is to be executed
321  // on host queue. The command's MPreparedDepsEvents will contain all three
322  // events (E1, E2, E3). Now, if piEventsWait is called for all three
323  // events we'll experience failure with CL_INVALID_CONTEXT 'cause these
324  // events refer to different contexts.
325  std::map<context_impl *, std::vector<EventImplPtr>>
326  RequiredEventsPerContext;
327 
328  for (const EventImplPtr &Event : EventImpls) {
329  ContextImplPtr Context = Event->getContextImpl();
330  assert(Context.get() &&
331  "Only non-host events are expected to be waited for here");
332  RequiredEventsPerContext[Context.get()].push_back(Event);
333  }
334 
335  for (auto &CtxWithEvents : RequiredEventsPerContext) {
336  std::vector<RT::PiEvent> RawEvents = getPiEvents(CtxWithEvents.second);
337  CtxWithEvents.first->getPlugin().call<PiApiKind::piEventsWait>(
338  RawEvents.size(), RawEvents.data());
339  }
340  } else {
341 #ifndef NDEBUG
342  for (const EventImplPtr &Event : EventImpls)
343  assert(Event->getContextImpl().get() &&
344  "Only non-host events are expected to be waited for here");
345 #endif
346 
347  std::vector<RT::PiEvent> RawEvents = getPiEvents(EventImpls);
348  flushCrossQueueDeps(EventImpls, getWorkerQueue());
349  const detail::plugin &Plugin = Queue->getPlugin();
350  Plugin.call<PiApiKind::piEnqueueEventsWait>(
351  Queue->getHandleRef(), RawEvents.size(), &RawEvents[0], &Event);
352  }
353  }
354 }

394 void Command::emitEdgeEventForCommandDependence(
395  Command *Cmd, void *ObjAddr, bool IsCommand,
396  std::optional<access::mode> AccMode) {
397 #ifdef XPTI_ENABLE_INSTRUMENTATION
398  // Bail early if either the source or the target node for the given dependency
399  // is undefined or NULL
400  if (!(xptiTraceEnabled() && MTraceEvent && Cmd && Cmd->MTraceEvent))
401  return;
402 
403  // If all the information we need for creating an edge event is available,
404  // then go ahead with creating it; if not, bail early!
405  xpti::utils::StringHelper SH;
406  std::string AddressStr = SH.addressAsString<void *>(ObjAddr);
407  std::string Prefix = AccMode ? accessModeToString(AccMode.value()) : "Event";
408  std::string TypeString = SH.nameWithAddressString(Prefix, AddressStr);
409  // Create an edge with the dependent buffer address for which a command
410  // object has been created as one of the properties of the edge
411  xpti::payload_t Payload(TypeString.c_str(), MAddress);
412  uint64_t EdgeInstanceNo;
413  xpti_td *EdgeEvent =
414  xptiMakeEvent(TypeString.c_str(), &Payload, xpti::trace_graph_event,
415  xpti_at::active, &EdgeInstanceNo);
416  if (EdgeEvent) {
417  xpti_td *SrcEvent = static_cast<xpti_td *>(Cmd->MTraceEvent);
418  xpti_td *TgtEvent = static_cast<xpti_td *>(MTraceEvent);
419  EdgeEvent->source_id = SrcEvent->unique_id;
420  EdgeEvent->target_id = TgtEvent->unique_id;
421  if (IsCommand) {
422  xpti::addMetadata(EdgeEvent, "access_mode",
423  static_cast<int>(AccMode.value()));
424  xpti::addMetadata(EdgeEvent, "memory_object",
425  reinterpret_cast<size_t>(ObjAddr));
426  } else {
427  xpti::addMetadata(EdgeEvent, "event", reinterpret_cast<size_t>(ObjAddr));
428  }
429  xptiNotifySubscribers(MStreamID, xpti::trace_edge_create,
430  detail::GSYCLGraphEvent, EdgeEvent, EdgeInstanceNo,
431  nullptr);
432  }
433  // General comment - None of these are serious errors as the instrumentation
434  // layer MUST be tolerant of errors. If we need to let the end user know, we
435  // throw exceptions in the future
436 #endif
437 }

443 void Command::emitEdgeEventForEventDependence(Command *Cmd,
444  RT::PiEvent &PiEventAddr) {
445 #ifdef XPTI_ENABLE_INSTRUMENTATION
446  // If we have failed to create an event to represent the Command, then we
447  // cannot emit an edge event. Bail early!
448  if (!(xptiTraceEnabled() && MTraceEvent))
449  return;
450 
451  if (Cmd && Cmd->MTraceEvent) {
452  // If the event is associated with a command, we use this command's trace
453  // event as the source of edge, hence modeling the control flow
454  emitEdgeEventForCommandDependence(Cmd, (void *)PiEventAddr, false);
455  return;
456  }
457  if (PiEventAddr) {
458  xpti::utils::StringHelper SH;
459  std::string AddressStr = SH.addressAsString<RT::PiEvent>(PiEventAddr);
460  // This is the case when it is a OCL event enqueued by the user or another
461  // event is registered by the runtime as a dependency The dependency on
462  // this occasion is an OCL event; so we build a virtual node in the graph
463  // with the event as the metadata for the node
464  std::string NodeName = SH.nameWithAddressString("virtual_node", AddressStr);
465  // Node name is "virtual_node[<event_addr>]"
466  xpti::payload_t VNPayload(NodeName.c_str(), MAddress);
467  uint64_t VNodeInstanceNo;
468  xpti_td *NodeEvent =
469  xptiMakeEvent(NodeName.c_str(), &VNPayload, xpti::trace_graph_event,
470  xpti_at::active, &VNodeInstanceNo);
471  // Emit the virtual node first
472  xpti::addMetadata(NodeEvent, "kernel_name", NodeName);
473  xptiNotifySubscribers(MStreamID, xpti::trace_node_create,
474  detail::GSYCLGraphEvent, NodeEvent, VNodeInstanceNo,
475  nullptr);
476  // Create a new event for the edge
477  std::string EdgeName = SH.nameWithAddressString("Event", AddressStr);
478  xpti::payload_t EdgePayload(EdgeName.c_str(), MAddress);
479  uint64_t EdgeInstanceNo;
480  xpti_td *EdgeEvent =
481  xptiMakeEvent(EdgeName.c_str(), &EdgePayload, xpti::trace_graph_event,
482  xpti_at::active, &EdgeInstanceNo);
483  if (EdgeEvent && NodeEvent) {
484  // Source node represents the event and this event needs to be completed
485  // before target node can execute
486  xpti_td *TgtEvent = static_cast<xpti_td *>(MTraceEvent);
487  EdgeEvent->source_id = NodeEvent->unique_id;
488  EdgeEvent->target_id = TgtEvent->unique_id;
489  xpti::addMetadata(EdgeEvent, "event",
490  reinterpret_cast<size_t>(PiEventAddr));
491  xptiNotifySubscribers(MStreamID, xpti::trace_edge_create,
492  detail::GSYCLGraphEvent, EdgeEvent, EdgeInstanceNo,
493  nullptr);
494  }
495  return;
496  }
497 #endif
498 }

544 Command *Command::processDepEvent(EventImplPtr DepEvent, const DepDesc &Dep,
545  std::vector<Command *> &ToCleanUp) {
546  const QueueImplPtr &WorkerQueue = getWorkerQueue();
547  const ContextImplPtr &WorkerContext = WorkerQueue->getContextImplPtr();
548 
549  // 1. Async work is not supported for host device.
550  // 2. Some types of commands do not produce PI events after they are enqueued
551  // (e.g. alloca). Note that we can't check the pi event to make that
552  // distinction since the command might still be unenqueued at this point.
553  bool PiEventExpected =
554  !DepEvent->is_host() || getType() == CommandType::HOST_TASK;
555  if (auto *DepCmd = static_cast<Command *>(DepEvent->getCommand()))
556  PiEventExpected &= DepCmd->producesPiEvent();
557 
558  if (!PiEventExpected) {
559  // call to waitInternal() is in waitForPreparedHostEvents() as it's called
560  // from enqueue process functions
561  MPreparedHostDepsEvents.push_back(DepEvent);
562  return nullptr;
563  }
564 
565  Command *ConnectionCmd = nullptr;
566 
567  // Do not add redundant event dependencies for in-order queues.
568  if (Dep.MDepCommand && Dep.MDepCommand->getWorkerQueue() == WorkerQueue &&
569  WorkerQueue->has_property<property::queue::in_order>() &&
570  getType() != CommandType::HOST_TASK)
571  return nullptr;
572 
573  ContextImplPtr DepEventContext = DepEvent->getContextImpl();
574  // If contexts don't match we'll connect them using host task
575  if (DepEventContext != WorkerContext && !WorkerContext->is_host()) {
576  Scheduler::GraphBuilder &GB = Scheduler::getInstance().MGraphBuilder;
577  ConnectionCmd = GB.connectDepEvent(this, DepEvent, Dep, ToCleanUp);
578  } else
579  MPreparedDepsEvents.push_back(std::move(DepEvent));
580 
581  return ConnectionCmd;
582 }

597 Command *Command::addDep(DepDesc NewDep, std::vector<Command *> &ToCleanUp) {
598  Command *ConnectionCmd = nullptr;
599 
600  if (NewDep.MDepCommand) {
601  ConnectionCmd =
602  processDepEvent(NewDep.MDepCommand->getEvent(), NewDep, ToCleanUp);
603  }
604  // ConnectionCmd insertion builds the following dependency structure:
605  // this -> emptyCmd (for ConnectionCmd) -> ConnectionCmd -> NewDep
606  // that means that this and NewDep are already dependent
607  if (!ConnectionCmd) {
608  MDeps.push_back(NewDep);
609  if (NewDep.MDepCommand)
610  NewDep.MDepCommand->addUser(this);
611  }
612 
613 #ifdef XPTI_ENABLE_INSTRUMENTATION
614  emitEdgeEventForCommandDependence(NewDep.MDepCommand,
615  (void *)NewDep.MDepRequirement->MSYCLMemObj,
616  true, NewDep.MDepRequirement->MAccessMode);
617 #endif
618 
619  return ConnectionCmd;
620 }

660 bool Command::enqueue(EnqueueResultT &EnqueueResult, BlockingT Blocking,
661  std::vector<Command *> &ToCleanUp) {
662  // Exit if already enqueued
663  if (MEnqueueStatus == EnqueueResultT::SyclEnqueueSuccess)
664  return true;
665 
666  // If the command is blocked from enqueueing
667  if (MIsBlockable && MEnqueueStatus == EnqueueResultT::SyclEnqueueBlocked) {
668  // Exit if enqueue type is not blocking
669  if (!Blocking) {
670  EnqueueResult = EnqueueResultT(EnqueueResultT::SyclEnqueueBlocked, this);
671  return false;
672  }
673  static bool ThrowOnBlock = getenv("SYCL_THROW_ON_BLOCK") != nullptr;
674  if (ThrowOnBlock)
675  throw sycl::runtime_error(
676  std::string("Waiting for blocked command. Block reason: ") +
677  std::string(getBlockReason()),
678  PI_INVALID_OPERATION);
679 
680 #ifdef XPTI_ENABLE_INSTRUMENTATION
681  // Scoped trace event notifier that emits a barrier begin and barrier end
682  // event, which models the barrier while enqueuing along with the blocked
683  // reason, as determined by the scheduler
684  std::string Info = "enqueue.barrier[";
685  Info += std::string(getBlockReason()) + "]";
686  emitInstrumentation(xpti::trace_barrier_begin, Info.c_str());
687 #endif
688 
689  // Wait if blocking
690  while (MEnqueueStatus == EnqueueResultT::SyclEnqueueBlocked)
691  ;
692 #ifdef XPTI_ENABLE_INSTRUMENTATION
693  emitInstrumentation(xpti::trace_barrier_end, Info.c_str());
694 #endif
695  }
696 
697  std::lock_guard<std::mutex> Lock(MEnqueueMtx);
698 
699  // Exit if the command is already enqueued
700  if (MEnqueueStatus == EnqueueResultT::SyclEnqueueSuccess)
701  return true;
702 
703 #ifdef XPTI_ENABLE_INSTRUMENTATION
704  emitInstrumentation(xpti::trace_task_begin, nullptr);
705 #endif
706 
707  if (MEnqueueStatus == EnqueueResultT::SyclEnqueueFailed) {
708  EnqueueResult = EnqueueResultT(EnqueueResultT::SyclEnqueueFailed, this);
709  return false;
710  }
711 
712  // Command status set to "failed" beforehand, so this command
713  // has already been marked as "failed" if enqueueImp throws an exception.
714  // This will avoid execution of the same failed command twice.
715  MEnqueueStatus = EnqueueResultT::SyclEnqueueFailed;
716  MShouldCompleteEventIfPossible = true;
717  cl_int Res = enqueueImp();
718 
719  if (CL_SUCCESS != Res)
720  EnqueueResult =
721  EnqueueResultT(EnqueueResultT::SyclEnqueueFailed, this, Res);
722  else {
723  if (MShouldCompleteEventIfPossible &&
724  (MEvent->is_host() || MEvent->getHandleRef() == nullptr))
725  MEvent->setComplete();
726 
727  // Consider the command is successfully enqueued if return code is
728  // CL_SUCCESS
729  MEnqueueStatus = EnqueueResultT::SyclEnqueueSuccess;
730  if (MLeafCounter == 0 && supportsPostEnqueueCleanup() &&
731  !SYCLConfig<SYCL_DISABLE_POST_ENQUEUE_CLEANUP>::get()) {
732  assert(!MPostEnqueueCleanup);
733  MPostEnqueueCleanup = true;
734  ToCleanUp.push_back(this);
735  }
736  }
737 
738  // Emit this correlation signal before the task end
739  emitEnqueuedEventSignal(MEvent->getHandleRef());
740 #ifdef XPTI_ENABLE_INSTRUMENTATION
741  emitInstrumentation(xpti::trace_task_end, nullptr);
742 #endif
743  return MEnqueueStatus == EnqueueResultT::SyclEnqueueSuccess;
744 }

752  // nodes have to be completed first before the current node can begin to
753  // execute; these edges model control flow

1025 cl_int ReleaseCommand::enqueueImp() {
1026  waitForPreparedHostEvents();
1027  std::vector<EventImplPtr> EventImpls = MPreparedDepsEvents;
1028  std::vector<RT::PiEvent> RawEvents = getPiEvents(EventImpls);
1029  bool SkipRelease = false;
1030 
1031  // On host side we only allocate memory for full buffers.
1032  // Thus, deallocating sub buffers leads to double memory freeing.
1033  SkipRelease |= MQueue->is_host() && MAllocaCmd->getType() == ALLOCA_SUB_BUF;
1034 
1035  const bool CurAllocaIsHost = MAllocaCmd->getQueue()->is_host();
1036  bool NeedUnmap = false;
1037  if (MAllocaCmd->MLinkedAllocaCmd) {
1038 
1039  // When releasing one of the "linked" allocations special rules take place:
1040  // 1. Device allocation should always be released.
1041  // 2. Host allocation should be released if host allocation is "leader".
1042  // 3. Device alloca in the pair should be in active state in order to be
1043  // correctly released.
1044 
1045 
1046  // There is no actual memory allocation if a host alloca command is created
1047  // being linked to a device allocation.
1048  SkipRelease |= CurAllocaIsHost && !MAllocaCmd->MIsLeaderAlloca;
1049 
1050  NeedUnmap |= CurAllocaIsHost == MAllocaCmd->MIsActive;
1051  }
1052 
1053  if (NeedUnmap) {
1054  const QueueImplPtr &Queue = CurAllocaIsHost
1055  ? MAllocaCmd->MLinkedAllocaCmd->getQueue()
1056  : MAllocaCmd->getQueue();
1057 
1058  EventImplPtr UnmapEventImpl(new event_impl(Queue));
1059  UnmapEventImpl->setContextImpl(Queue->getContextImplPtr());
1060  RT::PiEvent &UnmapEvent = UnmapEventImpl->getHandleRef();
1061 
1062  void *Src = CurAllocaIsHost
1063  ? MAllocaCmd->getMemAllocation()
1064  : MAllocaCmd->MLinkedAllocaCmd->getMemAllocation();
1065 
1066  void *Dst = !CurAllocaIsHost
1067  ? MAllocaCmd->getMemAllocation()
1068  : MAllocaCmd->MLinkedAllocaCmd->getMemAllocation();
1069 
1070  MemoryManager::unmap(MAllocaCmd->getSYCLMemObj(), Dst, Queue, Src,
1071  RawEvents, UnmapEvent);
1072 
1073  std::swap(MAllocaCmd->MIsActive, MAllocaCmd->MLinkedAllocaCmd->MIsActive);
1074  EventImpls.clear();
1075  EventImpls.push_back(UnmapEventImpl);
1076  }
1077  RT::PiEvent &Event = MEvent->getHandleRef();
1078  if (SkipRelease)
1079  Command::waitForEvents(MQueue, EventImpls, Event);
1080  else {
1081  MemoryManager::release(
1082  MQueue->getContextImplPtr(), MAllocaCmd->getSYCLMemObj(),
1083  MAllocaCmd->getMemAllocation(), std::move(EventImpls), Event);
1084  }
1085  return CL_SUCCESS;
1086 }

1824 // SYCL has a parallel_for_work_group variant where the only NDRange
1825 // characteristics set by a user is the number of work groups. This does not map
1826 // to the OpenCL clEnqueueNDRangeAPI, which requires global work size to be set
1827 // as well. This function determines local work size based on the device
1828 // characteristics and the number of work groups requested by the user, then
1829 // calculates the global work size.
1830 // SYCL specification (from 4.8.5.3):
1831 // The member function handler::parallel_for_work_group is parameterized by the
1832 // number of work - groups, such that the size of each group is chosen by the
1833 // runtime, or by the number of work - groups and number of work - items for
1834 // users who need more control.
1835 static void adjustNDRangePerKernel(NDRDescT &NDR, RT::PiKernel Kernel,
1836  const device_impl &DeviceImpl) {

1838  return; // GlobalSize is set - no need to adjust
1839  // check the prerequisites:
1840  assert(NDR.NumWorkGroups[0] != 0 && NDR.LocalSize[0] == 0);

1842  // avoid get_kernel_work_group_info on every kernel run
1843  range<3> WGSize = get_kernel_device_specific_info<
1844  range<3>,
1845  cl::sycl::info::kernel_device_specific::compile_work_group_size>::
1846  get(Kernel, DeviceImpl.getHandleRef(), DeviceImpl.getPlugin());

1849  WGSize = {1, 1, 1};

1854 // We have the following mapping between dimensions with SPIR-V builtins:
1855 // 1D: id[0] -> x
1856 // 2D: id[0] -> y, id[1] -> x
1857 // 3D: id[0] -> z, id[1] -> y, id[2] -> x
1858 // So in order to ensure the correctness we update all the kernel
1859 // parameters accordingly.
1860 // Initially we keep the order of NDRDescT as it provided by the user, this
1861 // simplifies overall handling and do the reverse only when
1862 // the kernel is enqueued.
