97  void set_specialization_constant_raw_value(const char *SpecName,
98  const void *Value) noexcept {
99  // Lock the mutex to prevent when one thread in the middle of writing a
100  // new value while another thread is reading the value to pass it to
101  // JIT compiler.
102  const std::lock_guard<std::mutex> SpecConstLock(MSpecConstAccessMtx);
103 
104  if (MSpecConstSymMap.count(std::string{SpecName}) == 0)
105  return;
106 
107  std::vector<SpecConstDescT> &Descs =
108  MSpecConstSymMap[std::string{SpecName}];
109  for (SpecConstDescT &Desc : Descs) {
110  Desc.IsSet = true;
111  std::memcpy(MSpecConstsBlob.data() + Desc.BlobOffset,
112  static_cast<const char *>(Value) + Desc.CompositeOffset,
113  Desc.Size);
114  }

215  void updateSpecConstSymMap() {
216  if (MBinImage) {
217  const pi::DeviceBinaryImage::PropertyRange &SCRange =
218  MBinImage->getSpecConstants();
219  using SCItTy = pi::DeviceBinaryImage::PropertyRange::ConstIterator;
220 
221  // get default values for specialization constants
222  const pi::DeviceBinaryImage::PropertyRange &SCDefValRange =
223  MBinImage->getSpecConstantsDefaultValues();
224 
225  bool HasDefaultValues = SCDefValRange.begin() != SCDefValRange.end();
226 
227  // This variable is used to calculate spec constant value offset in a
228  // flat byte array.
229  unsigned BlobOffset = 0;
230  for (SCItTy SCIt : SCRange) {
231  const char *SCName = (*SCIt)->Name;
232 
233  pi::ByteArray Descriptors =
234  pi::DeviceBinaryProperty(*SCIt).asByteArray();
235  assert(Descriptors.size() > 8 && "Unexpected property size");
236 
237  // Expected layout is vector of 3-component tuples (flattened into a
238  // vector of scalars), where each tuple consists of: ID of a scalar spec
239  // constant, (which might be a member of the composite); offset, which
240  // is used to calculate location of scalar member within the composite
241  // or zero for scalar spec constants; size of a spec constant
242  constexpr size_t NumElements = 3;
243  assert(((Descriptors.size() - 8) / sizeof(std::uint32_t)) %
244  NumElements ==
245  0 &&
246  "unexpected layout of composite spec const descriptors");
247  auto *It = reinterpret_cast<const std::uint32_t *>(&Descriptors[8]);
248  auto *End = reinterpret_cast<const std::uint32_t *>(&Descriptors[0] +
249  Descriptors.size());
250  unsigned PrevOffset = 0;
251  while (It != End) {
252  // Make sure that alignment is correct in blob.
253  BlobOffset += /*Offset*/ It[1] - PrevOffset;
254  PrevOffset = It[1];
255  // The map is not locked here because updateSpecConstSymMap() is only
256  // supposed to be called from c'tor.
257  MSpecConstSymMap[std::string{SCName}].push_back(
258  SpecConstDescT{/*ID*/ It[0], /*CompositeOffset*/ It[1],
259  /*Size*/ It[2], BlobOffset, HasDefaultValues});
260  BlobOffset += /*Size*/ It[2];
261  It += NumElements;
262  }
263  }
264  MSpecConstsBlob.resize(BlobOffset);
265 
266  if (HasDefaultValues) {
267  pi::ByteArray DefValDescriptors =
268  pi::DeviceBinaryProperty(*SCDefValRange.begin()).asByteArray();
269  std::uninitialized_copy(&DefValDescriptors[8],
270  &DefValDescriptors[8] + MSpecConstsBlob.size(),
271  MSpecConstsBlob.data());
272  }
273  }
274  }
