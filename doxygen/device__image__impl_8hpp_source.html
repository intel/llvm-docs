119  // Lock the mutex to prevent when one thread in the middle of writing a

246  void updateSpecConstSymMap() {
247  if (MBinImage) {
248  const pi::DeviceBinaryImage::PropertyRange &SCRange =
249  MBinImage->getSpecConstants();
250  using SCItTy = pi::DeviceBinaryImage::PropertyRange::ConstIterator;
251 
252  // get default values for specialization constants
253  const pi::DeviceBinaryImage::PropertyRange &SCDefValRange =
254  MBinImage->getSpecConstantsDefaultValues();
255 
256  // This variable is used to calculate spec constant value offset in a
257  // flat byte array.
258  unsigned BlobOffset = 0;
259  for (SCItTy SCIt : SCRange) {
260  const char *SCName = (*SCIt)->Name;
261 
262  pi::ByteArray Descriptors =
263  pi::DeviceBinaryProperty(*SCIt).asByteArray();
264  assert(Descriptors.size() > 8 && "Unexpected property size");
265 
266  // Expected layout is vector of 3-component tuples (flattened into a
267  // vector of scalars), where each tuple consists of: ID of a scalar spec
268  // constant, (which might be a member of the composite); offset, which
269  // is used to calculate location of scalar member within the composite
270  // or zero for scalar spec constants; size of a spec constant
271  constexpr size_t NumElements = 3;
272  assert(((Descriptors.size() - 8) / sizeof(std::uint32_t)) %
273  NumElements ==
274  0 &&
275  "unexpected layout of composite spec const descriptors");
276  auto *It = reinterpret_cast<const std::uint32_t *>(&Descriptors[8]);
277  auto *End = reinterpret_cast<const std::uint32_t *>(&Descriptors[0] +
278  Descriptors.size());
279  unsigned LocalOffset = 0;
280  while (It != End) {
281  // Make sure that alignment is correct in blob.
282  const unsigned OffsetFromLast = /*Offset*/ It[1] - LocalOffset;
283  BlobOffset += OffsetFromLast;
284  // Composites may have a special padding element at the end which
285  // should not have a descriptor. These padding elements all have max
286  // ID value.
287  if (It[0] != std::numeric_limits<std::uint32_t>::max()) {
288  // The map is not locked here because updateSpecConstSymMap() is
289  // only supposed to be called from c'tor.
290  MSpecConstSymMap[std::string{SCName}].push_back(
291  SpecConstDescT{/*ID*/ It[0], /*CompositeOffset*/ It[1],
292  /*Size*/ It[2], BlobOffset});
293  }
294  LocalOffset += OffsetFromLast + /*Size*/ It[2];
295  BlobOffset += /*Size*/ It[2];
296  It += NumElements;
297  }
298  }
299  MSpecConstsBlob.resize(BlobOffset);
300 
301  bool HasDefaultValues = SCDefValRange.begin() != SCDefValRange.end();
302 
303  if (HasDefaultValues) {
304  pi::ByteArray DefValDescriptors =
305  pi::DeviceBinaryProperty(*SCDefValRange.begin()).asByteArray();
306  assert(DefValDescriptors.size() - 8 == MSpecConstsBlob.size() &&
307  "Specialization constant default value blob do not have the "
308  "expected size.");
309  std::uninitialized_copy(&DefValDescriptors[8],
310  &DefValDescriptors[8] + MSpecConstsBlob.size(),
311  MSpecConstsBlob.data());
312  }
313  }
314  }
