68 inline float fast_length(const float *a, int len) {
69  switch (len) {
70  case 1:
71  return sycl::fast_length(a[0]);
72  case 2:
73  return sycl::fast_length(sycl::float2(a[0], a[1]));
74  case 3:
75  return sycl::fast_length(sycl::float3(a[0], a[1], a[2]));
76  case 4:
77  return sycl::fast_length(sycl::float4(a[0], a[1], a[2], a[3]));
78  case 0:
79  return 0;
80  default:
81  float f = 0;
82  for (int i = 0; i < len; ++i)
83  f += a[i] * a[i];
84  return sycl::sqrt(f);
85  }
86 }
