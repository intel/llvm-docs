87 template <typename ValueT>
88 inline auto zero_or_signed_extend(ValueT val, unsigned bit) {
89  static_assert(std::is_integral_v<ValueT>);
90  if constexpr (sizeof(ValueT) == 4) {
91  assert(bit < 64 &&
92  "When extending int32 value, bit must be smaller than 64.");
93  if constexpr (std::is_signed_v<ValueT>)
94  return int64_t(val) << (64 - bit) >> (64 - bit);
95  else
96  return int64_t(val);
97  } else if constexpr (sizeof(ValueT) == 2) {
98  assert(bit < 32 &&
99  "When extending int16 value, bit must be smaller than 32.");
100  if constexpr (std::is_signed_v<ValueT>)
101  return int32_t(val) << (32 - bit) >> (32 - bit);
102  else
103  return int32_t(val);
104  } else if constexpr (sizeof(ValueT) == 1) {
105  assert(bit < 16 &&
106  "When extending int8 value, bit must be smaller than 16.");
107  if constexpr (std::is_signed_v<ValueT>)
108  return int16_t(val) << (16 - bit) >> (16 - bit);
109  else
110  return int16_t(val);
111  } else {
112  static_assert(sizeof(ValueT) == 8);
113  assert(bit < 64 && "Cannot extend int64 value.");
114  return static_cast<int64_t>(val);
115  }
116 }

265 template <typename T>
266 inline T bfe_safe(const T source, const uint32_t bit_start,
267  const uint32_t num_bits) {
268  static_assert(std::is_integral_v<T>);
269 #if defined(__SYCL_DEVICE_ONLY__) && defined(__NVPTX__)
270  if constexpr (std::is_same_v<T, int8_t> || std::is_same_v<T, int16_t> ||
271  std::is_same_v<T, int32_t>) {
272  int32_t res{};
273  asm volatile("bfe.s32 %0, %1, %2, %3;"
274  : "=r"(res)
275  : "r"((int32_t)source), "r"(bit_start), "r"(num_bits));
276  return res;
277  } else if constexpr (std::is_same_v<T, uint8_t> ||
278  std::is_same_v<T, uint16_t> ||
279  std::is_same_v<T, uint32_t>) {
280  uint32_t res{};
281  asm volatile("bfe.u32 %0, %1, %2, %3;"
282  : "=r"(res)
283  : "r"((uint32_t)source), "r"(bit_start), "r"(num_bits));
284  return res;
285  } else if constexpr (std::is_same_v<T, int64_t>) {
286  T res{};
287  asm volatile("bfe.s64 %0, %1, %2, %3;"
288  : "=l"(res)
289  : "l"(source), "r"(bit_start), "r"(num_bits));
290  return res;
291  } else if constexpr (std::is_same_v<T, uint64_t>) {
292  T res{};
293  asm volatile("bfe.u64 %0, %1, %2, %3;"
294  : "=l"(res)
295  : "l"(source), "r"(bit_start), "r"(num_bits));
296  return res;
297  }
298 #endif
299  const uint32_t bit_width = CHAR_BIT * sizeof(T);
300  const uint32_t pos = std::min(bit_start, bit_width);
301  const uint32_t len = std::min(pos + num_bits, bit_width) - pos;
302  if constexpr (std::is_signed_v<T>) {
303  // FIXME(syclcompat-lib-reviewers): As above, catching a case whose result
304  // is undefined and likely losing perf.
305  const T mask = len >= bit_width ? T{-1} : static_cast<T>((T{1} << len) - 1);
306 
307  // Find the sign-bit, the result is padded with the sign bit of the
308  // extracted field.
309  // Note if requested num_bits==0, we return zero via sign_bit=0
310  const uint32_t sign_bit_pos = std::min(pos + len - 1, bit_width - 1);
311  const T sign_bit = num_bits != 0 && ((source >> sign_bit_pos) & 1);
312  const T sign_bit_padding = (-sign_bit & ~mask);
313  return ((source >> pos) & mask) | sign_bit_padding;
314  } else {
315  return syclcompat::detail::bfe(source, pos, len);
316  }
317 }

329 template <typename T>
330 inline T bfi(const T x, const T y, const uint32_t bit_start,
331  const uint32_t num_bits) {
332  static_assert(std::is_unsigned_v<T>);
333  constexpr unsigned bit_width = CHAR_BIT * sizeof(T);
334 
335  // if bit_start > bit_width || len == 0, should return y.
336  const T ignore_bfi = static_cast<T>(bit_start > bit_width || num_bits == 0);
337  T extract_bitfield_mask = (static_cast<T>(~T{0}) >> (bit_width - num_bits))
338  << bit_start;
339  T clean_bitfield_mask = ~extract_bitfield_mask;
340  return (y & (-ignore_bfi | clean_bitfield_mask)) |
341  (~-ignore_bfi & ((x << bit_start) & extract_bitfield_mask));
342 }
