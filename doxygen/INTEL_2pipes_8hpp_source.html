19 template <class name, class dataT, int32_t min_capacity = 0> class pipe {
20 public:
21  // Non-blocking pipes
22  // Reading from pipe is lowered to SPIR-V instruction OpReadPipe via SPIR-V
23  // friendly LLVM IR.
24  static dataT read(bool &Success) {
25 #ifdef __SYCL_DEVICE_ONLY__
26  RPipeTy<dataT> RPipe =
27  __spirv_CreatePipeFromPipeStorage_read<dataT>(&m_Storage);
28  dataT TempData;
29  Success = !static_cast<bool>(
30  __spirv_ReadPipe(RPipe, &TempData, m_Size, m_Alignment));
31  return TempData;
32 #else
33  (void)Success;
34  assert(!"Pipes are not supported on a host device!");
35 #endif // __SYCL_DEVICE_ONLY__
36  }
