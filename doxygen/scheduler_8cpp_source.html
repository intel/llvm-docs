28 void Scheduler::waitForRecordToFinish(MemObjRecord *Record) {
29 #ifdef XPTI_ENABLE_INSTRUMENTATION
30  // Will contain the list of dependencies for the Release Command
31  std::set<Command *> DepCommands;
32 #endif
33  for (Command *Cmd : Record->MReadLeaves) {
34  EnqueueResultT Res;
35  bool Enqueued = GraphProcessor::enqueueCommand(Cmd, Res);
36  if (!Enqueued && EnqueueResultT::SyclEnqueueFailed == Res.MResult)
37  throw runtime_error("Enqueue process failed.", PI_INVALID_OPERATION);
38 #ifdef XPTI_ENABLE_INSTRUMENTATION
39  // Capture the dependencies
40  DepCommands.insert(Cmd);
41 #endif
42  GraphProcessor::waitForEvent(Cmd->getEvent());
43  }
44  for (Command *Cmd : Record->MWriteLeaves) {
45  EnqueueResultT Res;
46  bool Enqueued = GraphProcessor::enqueueCommand(Cmd, Res);
47  if (!Enqueued && EnqueueResultT::SyclEnqueueFailed == Res.MResult)
48  throw runtime_error("Enqueue process failed.", PI_INVALID_OPERATION);
49 #ifdef XPTI_ENABLE_INSTRUMENTATION
50  DepCommands.insert(Cmd);
51 #endif
52  GraphProcessor::waitForEvent(Cmd->getEvent());
53  }
54  for (AllocaCommandBase *AllocaCmd : Record->MAllocaCommands) {
55  Command *ReleaseCmd = AllocaCmd->getReleaseCmd();
56  EnqueueResultT Res;
57  bool Enqueued = GraphProcessor::enqueueCommand(ReleaseCmd, Res);
58  if (!Enqueued && EnqueueResultT::SyclEnqueueFailed == Res.MResult)
59  throw runtime_error("Enqueue process failed.", PI_INVALID_OPERATION);
60 #ifdef XPTI_ENABLE_INSTRUMENTATION
61  // Report these dependencies to the Command so these dependencies can be
62  // reported as edges
63  ReleaseCmd->resolveReleaseDependencies(DepCommands);
64 #endif
65  GraphProcessor::waitForEvent(ReleaseCmd->getEvent());
66  }
67 }

69 EventImplPtr Scheduler::addCG(std::unique_ptr<detail::CG> CommandGroup,
70  QueueImplPtr Queue) {
71  EventImplPtr NewEvent = nullptr;
72  const bool IsKernel = CommandGroup->getType() == CG::KERNEL;
73  vector_class<StreamImplPtr> Streams;
74  {
75  std::unique_lock<std::shared_timed_mutex> Lock(MGraphLock, std::defer_lock);
76  lockSharedTimedMutex(Lock);
77 
78  Command *NewCmd = nullptr;
79  switch (CommandGroup->getType()) {
80  case CG::UPDATE_HOST:
81  NewCmd = MGraphBuilder.addCGUpdateHost(std::move(CommandGroup),
82  DefaultHostQueue);
83  break;
84  case CG::CODEPLAY_HOST_TASK:
85  NewCmd = MGraphBuilder.addCG(std::move(CommandGroup), DefaultHostQueue);
86  break;
87  default:
88  NewCmd = MGraphBuilder.addCG(std::move(CommandGroup), std::move(Queue));
89  }
90  NewEvent = NewCmd->getEvent();
91  }
92 
93  {
94  std::shared_lock<std::shared_timed_mutex> Lock(MGraphLock);
95 
96  Command *NewCmd = static_cast<Command *>(NewEvent->getCommand());
97  if (NewCmd) {
98  // TODO: Check if lazy mode.
99  EnqueueResultT Res;
100  bool Enqueued = GraphProcessor::enqueueCommand(NewCmd, Res);
101  if (!Enqueued && EnqueueResultT::SyclEnqueueFailed == Res.MResult)
102  throw runtime_error("Enqueue process failed.", PI_INVALID_OPERATION);
103 
104  if (IsKernel)
105  Streams = ((ExecCGCommand *)NewCmd)->getStreams();
106 
107  if (NewCmd->MDeps.size() == 0 && NewCmd->MUsers.size() == 0) {
108  NewEvent->setCommand(nullptr); // if there are no memory dependencies,
109  // decouple and free the command
110  delete NewCmd;
111  }
112  }
113  }
114 
115  for (auto StreamImplPtr : Streams) {
116  StreamImplPtr->flush();
117  }
118 
119  return NewEvent;
120 }

122 EventImplPtr Scheduler::addCopyBack(Requirement *Req) {
123  std::unique_lock<std::shared_timed_mutex> Lock(MGraphLock, std::defer_lock);
124  lockSharedTimedMutex(Lock);
125  Command *NewCmd = MGraphBuilder.addCopyBack(Req);
126  // Command was not creted because there were no operations with
127  // buffer.
128  if (!NewCmd)
129  return nullptr;
130 
131  try {
132  EnqueueResultT Res;
133  bool Enqueued = GraphProcessor::enqueueCommand(NewCmd, Res);
134  if (!Enqueued && EnqueueResultT::SyclEnqueueFailed == Res.MResult)
135  throw runtime_error("Enqueue process failed.", PI_INVALID_OPERATION);
136  } catch (...) {
137  NewCmd->getQueue()->reportAsyncException(std::current_exception());
138  }
139  return NewCmd->getEvent();
140 }

166 void Scheduler::cleanupFinishedCommands(EventImplPtr FinishedEvent) {
167  // We are going to traverse a graph of finished commands. Gather stream
168  // objects from these commands if any and deallocate buffers for these stream
169  // objects, this is needed to guarantee that streamed data is printed and
170  // resources are released.
171  std::vector<std::shared_ptr<stream_impl>> StreamsToDeallocate;
172  {
173  // Avoiding deadlock situation, where one thread is in the process of
174  // enqueueing (with a locked mutex) a currently blocked task that waits for
175  // another thread which is stuck at attempting cleanup.
176  std::unique_lock<std::shared_timed_mutex> Lock(MGraphLock,
177  std::try_to_lock);
178  if (Lock.owns_lock()) {
179  auto FinishedCmd = static_cast<Command *>(FinishedEvent->getCommand());
180  // The command might have been cleaned up (and set to nullptr) by another
181  // thread
182  if (FinishedCmd)
183  MGraphBuilder.cleanupFinishedCommands(FinishedCmd, StreamsToDeallocate);
184  }
185  }
186  deallocateStreams(StreamsToDeallocate);
187 }

189 void Scheduler::removeMemoryObject(detail::SYCLMemObjI *MemObj) {
190  // We are going to traverse a graph of finished commands. Gather stream
191  // objects from these commands if any and deallocate buffers for these stream
192  // objects, this is needed to guarantee that streamed data is printed and
193  // resources are released.
194  std::vector<std::shared_ptr<stream_impl>> StreamsToDeallocate;
195  {
196  MemObjRecord *Record = nullptr;
197  std::unique_lock<std::shared_timed_mutex> Lock(MGraphLock, std::defer_lock);
198 
199  {
200  lockSharedTimedMutex(Lock);
201 
202  Record = MGraphBuilder.getMemObjRecord(MemObj);
203  if (!Record)
204  // No operations were performed on the mem object
205  return;
206 
207  Lock.unlock();
208  }
209 
210  {
211  // This only needs a shared mutex as it only involves enqueueing and
212  // awaiting for events
213  std::shared_lock<std::shared_timed_mutex> Lock(MGraphLock);
214  waitForRecordToFinish(Record);
215  }
216 
217  {
218  lockSharedTimedMutex(Lock);
219  MGraphBuilder.decrementLeafCountersForRecord(Record);
220  MGraphBuilder.cleanupCommandsForRecord(Record, StreamsToDeallocate);
221  MGraphBuilder.removeRecordForMemObj(MemObj);
222  }
223  }
224  deallocateStreams(StreamsToDeallocate);
225 }

290 Scheduler::~Scheduler() {
291  // By specification there are several possible sync points: buffer
292  // destruction, wait() method of a queue or event. Stream doesn't introduce
293  // any synchronization point. It is guaranteed that stream is flushed and
294  // resources are released only if one of the listed sync points was used for
295  // the kernel. Otherwise resources for stream will not be released, issue a
296  // warning in this case.
297  if (pi::trace(pi::TraceLevel::PI_TRACE_BASIC)) {
298  std::lock_guard<std::mutex> lock(StreamBuffersPoolMutex);
299  if (!StreamBuffersPool.empty())
300  fprintf(
301  stderr,
302  "\nWARNING: Some commands may have not finished the execution and "
303  "not all resources were released. Please be sure that all kernels "
304  "have synchronization points.\n\n");
305  }
306 }

308 void Scheduler::lockSharedTimedMutex(
309  std::unique_lock<std::shared_timed_mutex> &Lock) {
310 #ifdef _WIN32
311  // Avoiding deadlock situation for MSVC. std::shared_timed_mutex specification
312  // does not specify a priority for shared and exclusive accesses. It will be a
313  // deadlock in MSVC's std::shared_timed_mutex implementation, if exclusive
314  // access occurs after shared access.
315  // TODO: after switching to C++17, change std::shared_timed_mutex to
316  // std::shared_mutex and use std::lock_guard here both for Windows and Linux.
317  while (!Lock.try_lock_for(std::chrono::milliseconds(10))) {
318  // Without yield while loop acts like endless while loop and occupies the
319  // whole CPU when multiple command groups are created in multiple host
320  // threads
321  std::this_thread::yield();
322  }
323 #else
324  // It is a deadlock on UNIX in implementation of lock and lock_shared, if
325  // try_lock in the loop above will be executed, so using a single lock here
326  Lock.lock();
327 #endif // _WIN32
328 }
