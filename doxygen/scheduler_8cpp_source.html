29 void Scheduler::waitForRecordToFinish(MemObjRecord *Record) {
30 #ifdef XPTI_ENABLE_INSTRUMENTATION
31  // Will contain the list of dependencies for the Release Command
32  std::set<Command *> DepCommands;
33 #endif
34  for (Command *Cmd : Record->MReadLeaves) {
35  EnqueueResultT Res;
36  bool Enqueued = GraphProcessor::enqueueCommand(Cmd, Res);
37  if (!Enqueued && EnqueueResultT::SyclEnqueueFailed == Res.MResult)
38  throw runtime_error("Enqueue process failed.", PI_INVALID_OPERATION);
39 #ifdef XPTI_ENABLE_INSTRUMENTATION
40  // Capture the dependencies
41  DepCommands.insert(Cmd);
42 #endif
43  GraphProcessor::waitForEvent(Cmd->getEvent());
44  }
45  for (Command *Cmd : Record->MWriteLeaves) {
46  EnqueueResultT Res;
47  bool Enqueued = GraphProcessor::enqueueCommand(Cmd, Res);
48  if (!Enqueued && EnqueueResultT::SyclEnqueueFailed == Res.MResult)
49  throw runtime_error("Enqueue process failed.", PI_INVALID_OPERATION);
50 #ifdef XPTI_ENABLE_INSTRUMENTATION
51  DepCommands.insert(Cmd);
52 #endif
53  GraphProcessor::waitForEvent(Cmd->getEvent());
54  }
55  for (AllocaCommandBase *AllocaCmd : Record->MAllocaCommands) {
56  Command *ReleaseCmd = AllocaCmd->getReleaseCmd();
57  EnqueueResultT Res;
58  bool Enqueued = GraphProcessor::enqueueCommand(ReleaseCmd, Res);
59  if (!Enqueued && EnqueueResultT::SyclEnqueueFailed == Res.MResult)
60  throw runtime_error("Enqueue process failed.", PI_INVALID_OPERATION);
61 #ifdef XPTI_ENABLE_INSTRUMENTATION
62  // Report these dependencies to the Command so these dependencies can be
63  // reported as edges
64  ReleaseCmd->resolveReleaseDependencies(DepCommands);
65 #endif
66  GraphProcessor::waitForEvent(ReleaseCmd->getEvent());
67  }
68 }

70 EventImplPtr Scheduler::addCG(std::unique_ptr<detail::CG> CommandGroup,
71  QueueImplPtr Queue) {
72  Command *NewCmd = nullptr;
73  const bool IsKernel = CommandGroup->getType() == CG::KERNEL;
74  {
75  std::lock_guard<std::mutex> Lock(MGraphLock);
76 
77  switch (CommandGroup->getType()) {
78  case CG::UPDATE_HOST:
79  NewCmd = MGraphBuilder.addCGUpdateHost(std::move(CommandGroup),
80  DefaultHostQueue);
81  break;
82  default:
83  NewCmd = MGraphBuilder.addCG(std::move(CommandGroup), std::move(Queue));
84  }
85 
86  // TODO: Check if lazy mode.
87  EnqueueResultT Res;
88  bool Enqueued = GraphProcessor::enqueueCommand(NewCmd, Res);
89  if (!Enqueued && EnqueueResultT::SyclEnqueueFailed == Res.MResult)
90  throw runtime_error("Enqueue process failed.", PI_INVALID_OPERATION);
91  }
92 
93  if (IsKernel)
94  ((ExecCGCommand *)NewCmd)->flushStreams();
95 
96  return NewCmd->getEvent();
97 }

99 EventImplPtr Scheduler::addCopyBack(Requirement *Req) {
100  std::lock_guard<std::mutex> lock(MGraphLock);
101  Command *NewCmd = MGraphBuilder.addCopyBack(Req);
102  // Command was not creted because there were no operations with
103  // buffer.
104  if (!NewCmd)
105  return nullptr;
106 
107  try {
108  EnqueueResultT Res;
109  bool Enqueued = GraphProcessor::enqueueCommand(NewCmd, Res);
110  if (!Enqueued && EnqueueResultT::SyclEnqueueFailed == Res.MResult)
111  throw runtime_error("Enqueue process failed.", PI_INVALID_OPERATION);
112  } catch (...) {
113  NewCmd->getQueue()->reportAsyncException(std::current_exception());
114  }
115  return NewCmd->getEvent();
116 }

151 void Scheduler::removeMemoryObject(detail::SYCLMemObjI *MemObj) {
152  std::lock_guard<std::mutex> lock(MGraphLock);
153 
154  MemObjRecord *Record = MGraphBuilder.getMemObjRecord(MemObj);
155  if (!Record)
156  // No operations were performed on the mem object
157  return;
158  waitForRecordToFinish(Record);
159  MGraphBuilder.decrementLeafCountersForRecord(Record);
160  MGraphBuilder.cleanupCommandsForRecord(Record);
161  MGraphBuilder.removeRecordForMemObj(MemObj);
162 }

164 EventImplPtr Scheduler::addHostAccessor(Requirement *Req,
165  const bool destructor) {
166  std::lock_guard<std::mutex> lock(MGraphLock);
167 
168  Command *NewCmd = MGraphBuilder.addHostAccessor(Req, destructor);
169 
170  if (!NewCmd)
171  return nullptr;
172  EnqueueResultT Res;
173  bool Enqueued = GraphProcessor::enqueueCommand(NewCmd, Res);
174  if (!Enqueued && EnqueueResultT::SyclEnqueueFailed == Res.MResult)
175  throw runtime_error("Enqueue process failed.", PI_INVALID_OPERATION);
176  return NewCmd->getEvent();
177 }

179 void Scheduler::releaseHostAccessor(Requirement *Req) {
180  Req->MBlockedCmd->MCanEnqueue = true;
181  MemObjRecord* Record = Req->MSYCLMemObj->MRecord.get();
182  auto EnqueueLeaves = [](CircularBuffer<Command *> &Leaves) {
183  for (Command *Cmd : Leaves) {
184  EnqueueResultT Res;
185  bool Enqueued = GraphProcessor::enqueueCommand(Cmd, Res);
186  if (!Enqueued && EnqueueResultT::SyclEnqueueFailed == Res.MResult)
187  throw runtime_error("Enqueue process failed.", PI_INVALID_OPERATION);
188  }
189  };
190  EnqueueLeaves(Record->MReadLeaves);
191  EnqueueLeaves(Record->MWriteLeaves);
192 }
