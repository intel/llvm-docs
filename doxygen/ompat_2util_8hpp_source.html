52 inline void matrix_mem_copy(void *to_ptr, const void *from_ptr, int to_ld,
53  int from_ld, int rows, int cols, int elem_size,
54  sycl::queue queue = syclcompat::get_default_queue(),
55  bool async = false) {
56  if (to_ptr == from_ptr && to_ld == from_ld) {
57  return;
58  }
59 
60  if (to_ld == from_ld) {
61  size_t copy_size = elem_size * ((cols - 1) * (size_t)to_ld + rows);
62  if (async)
63  detail::memcpy(queue, (void *)to_ptr, (void *)from_ptr, copy_size);
64  else
65  detail::memcpy(queue, (void *)to_ptr, (void *)from_ptr, copy_size).wait();
66  } else {
67  if (async)
68  detail::memcpy(queue, to_ptr, from_ptr, elem_size * to_ld,
69  elem_size * from_ld, elem_size * rows, cols);
70  else
71  sycl::event::wait(detail::memcpy(queue, to_ptr, from_ptr,
72  elem_size * to_ld, elem_size * from_ld,
73  elem_size * rows, cols));
74  }
75 }

123 template <typename T> inline T reverse_bits(T a) {
124  static_assert(std::is_unsigned<T>::value && std::is_integral<T>::value,
125  "unsigned integer required");
126  if (!a)
127  return 0;
128  T mask = 0;
129  size_t count = 4 * sizeof(T);
130  mask = ~mask >> count;
131  while (count) {
132  a = ((a & mask) << count) | ((a & ~mask) >> count);
133  count = count >> 1;
134  mask = mask ^ (mask << count);
135  }
136  return a;
137 }

144 inline unsigned int byte_level_permute(unsigned int a, unsigned int b,
145  unsigned int s) {
146  unsigned int ret;
147  ret =
148  ((((std::uint64_t)b << 32 | a) >> (s & 0x7) * 8) & 0xff) |
149  (((((std::uint64_t)b << 32 | a) >> ((s >> 4) & 0x7) * 8) & 0xff) << 8) |
150  (((((std::uint64_t)b << 32 | a) >> ((s >> 8) & 0x7) * 8) & 0xff) << 16) |

152  return ret;

205  int logical_sub_group_size = 32) {
206  unsigned int id = g.get_local_linear_id();
207  unsigned int end_index =
208  (id / logical_sub_group_size + 1) * logical_sub_group_size;
209  T result = sycl::shift_group_left(g, x, delta);
210  if ((id + delta) >= end_index) {
211  result = x;
212  }

290 template <int dimensions = 3>
291 inline void nd_range_barrier(
292  sycl::nd_item<dimensions> item,
293  sycl::atomic_ref<unsigned int, sycl::memory_order::acq_rel,
294  sycl::memory_scope::device,
295  sycl::access::address_space::global_space> &counter) {
296 
297  static_assert(dimensions == 3, "dimensions must be 3.");
298  constexpr unsigned int MSB32_MASK = 0x80000000;
299 
300  unsigned int num_groups = item.get_group_range(2) * item.get_group_range(1) *
301  item.get_group_range(0);
302 
303  item.barrier();
304 
305  if (item.get_local_linear_id() == 0) {
306  unsigned int inc = 1;
307  unsigned int old_arrive = 0;
308  bool is_group0 =
309  (item.get_group(2) + item.get_group(1) + item.get_group(0) == 0);
310  if (is_group0) {
311  inc = MSB32_MASK - (num_groups - 1);
312  }
313 
314  old_arrive = counter.fetch_add(inc);
315  // Synchronize all the work groups
316  while (((old_arrive ^ counter.load()) & MSB32_MASK) == 0)
317  ;
318  }
319 
320  item.barrier();
321 }

393  _g.get_local_linear_range() / _logical_group_size;

395  if (_item.get_local_linear_id() >= first_of_last_group) {
396  return _g.get_local_linear_range() - first_of_last_group;
397  } else {
398  return _logical_group_size;
399  }
400  }
402  uint32_t get_group_linear_range() const {
403  return _group_linear_range_in_parent;
404  }

442 inline int calculate_max_active_wg_per_xecore(int *num_wg, int wg_size,
443  int slm_size = 0,
444  int sg_size = 32,
445  bool used_barrier = false,
446  bool used_large_grf = false) {
447  int ret = 0;
448  const int slm_size_per_xe_core = 64 * 1024;
449  const int max_barrier_registers = 32;
450  syclcompat::device_ext &dev = syclcompat::get_current_device();
451 
452  size_t max_wg_size = dev.get_info<sycl::info::device::max_work_group_size>();
453  if (wg_size > max_wg_size) {
454  wg_size = max_wg_size;
455  ret = -1;
456  }
457 
458  int num_threads_ss = 56;
459  int max_num_wg = 56;
460  if (dev.has(sycl::aspect::ext_intel_gpu_eu_count_per_subslice) &&
461  dev.has(sycl::aspect::ext_intel_gpu_hw_threads_per_eu)) {
462  auto eu_count =
463  dev.get_info<sycl::info::device::ext_intel_gpu_eu_count_per_subslice>();
464  auto threads_count =
465  dev.get_info<sycl::ext::intel::info::device::gpu_hw_threads_per_eu>();
466  num_threads_ss = eu_count * threads_count;
467  max_num_wg = eu_count * threads_count;
468  }
469 
470  if (used_barrier) {
471  max_num_wg = max_barrier_registers;
472  }
473 
474  // Calculate num_wg_slm
475  int num_wg_slm = 0;
476  if (slm_size == 0) {
477  num_wg_slm = max_num_wg;
478  } else {
479  num_wg_slm = std::floor((float)slm_size_per_xe_core / slm_size);
480  }
481 
482  // Calculate num_wg_threads
483  if (used_large_grf)
484  num_threads_ss = num_threads_ss / 2;
485  int num_threads = std::ceil((float)wg_size / sg_size);
486  int num_wg_threads = std::floor((float)num_threads_ss / num_threads);
487 
488  // Calculate num_wg
489  *num_wg = std::min(num_wg_slm, num_wg_threads);
490  *num_wg = std::min(*num_wg, max_num_wg);
491  return ret;
492 }
