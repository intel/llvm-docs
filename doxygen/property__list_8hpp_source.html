73 // This class is used in property_list to hold properties.
74 template <class T> class PropertyHolder {
75 public:
76  PropertyHolder() = default;
77 
78  PropertyHolder(const PropertyHolder &P) {
79  if (P.isInitialized()) {
80  new (m_Mem) T(P.getProp());
81  m_Initialized = true;
82  }
83  }
84 
85  ~PropertyHolder() {
86  if (m_Initialized) {
87  T *MemPtr = reinterpret_cast<T *>(m_Mem);
88  MemPtr->~T();
89  }
90  }
91 
92  PropertyHolder &operator=(const PropertyHolder &Other) {
93  if (this != &Other) {
94  if (m_Initialized) {
95  T *MemPtr = reinterpret_cast<T *>(m_Mem);
96  MemPtr->~T();
97  m_Initialized = false;
98  }
99 
100  if (Other.m_Initialized) {
101  new (m_Mem) T(Other.getProp());
102  m_Initialized = true;
103  }
104  }
105  return *this;
106  }
107 
108  void setProp(const T &Rhs) {
109  new (m_Mem) T(Rhs);
110  m_Initialized = true;
111  }
112 
113  const T &getProp() const {
114  assert(true == m_Initialized && "Property was not set!");
115  const T *MemPtr = reinterpret_cast<const T *>(m_Mem);
116  return *MemPtr;
117  }
118  bool isInitialized() const { return m_Initialized; }
119 
120 private:
121  // Memory that is used for property allocation
122  alignas(T) unsigned char m_Mem[sizeof(T)];
123  // Indicate whether property initialized or not.
124  bool m_Initialized = false;
125 };

219  // The structs validate that all objects passed are base of PropBase class.
220  template <typename... Tail> struct AllProperties : std::true_type {};
221  template <typename T, typename... Tail>
222  struct AllProperties<T, Tail...>
223  : std::conditional<std::is_base_of<property::detail::PropBase, T>::value,
224  AllProperties<Tail...>, std::false_type>::type {};
225 
226  template <class T>
227  using PropertyHolder = cl::sycl::property::detail::PropertyHolder<T>;
228  template <property::detail::PropKind PropKindT>
229  using Property = cl::sycl::property::detail::Prop<PropKindT>;
230 
231  // The structs build tuple type that can hold all properties.
232  template <typename... Head> struct DefineTupleType {
233  using Type = std::tuple<Head...>;
234  };
235 
236  template <int Counter, typename... Head>
237  struct BuildTupleType
238  : public std::conditional<
239  (Counter < property::detail::PropKind::PropKindSize),
240  BuildTupleType<
241  Counter + 1, Head...,
242  PropertyHolder<typename Property<(property::detail::PropKind)(
243  Counter)>::FinalType>>,
244  DefineTupleType<Head...>>::type {};
245 
246 public:
247  // C'tor initialize m_PropList with properties passed by invoking ctorHelper
248  // recursively
249  template <typename... propertyTN,
250  typename = typename std::enable_if<
251  AllProperties<propertyTN...>::value>::type>
252  property_list(propertyTN... Props) {
253  ctorHelper(Props...);
254  }
255 
256  template <typename propertyT> propertyT get_property() const {
257  static_assert((int)(propertyT::getKind()) <=
258  property::detail::PropKind::PropKindSize,
259  "Invalid option passed.");
260  const auto &PropHolder = std::get<(int)(propertyT::getKind())>(m_PropsList);
261  if (PropHolder.isInitialized()) {
262  return PropHolder.getProp();
263  }
264  throw invalid_object_error();
265  }
266 
267  template <typename propertyT> bool has_property() const {
268  return std::get<(int)(propertyT::getKind())>(m_PropsList).isInitialized();
269  }
270 
271 private:
272  void ctorHelper() {}
273 
274  template <typename... propertyTN, class PropT>
275  void ctorHelper(PropT &Prop, propertyTN... props) {
276  std::get<(int)(PropT::getKind())>(m_PropsList).setProp(Prop);
277  ctorHelper(props...);
278  }
279 
280  // Tuple that able to hold all the properties.
281  BuildTupleType<0>::Type m_PropsList;
282 };
