26 ordered_queue::ordered_queue(const context &syclContext,
27  const device_selector &deviceSelector,
28  const async_handler &asyncHandler,
29  const property_list &propList) {
30 
31  const vector_class<device> Devs = syclContext.get_devices();
32 
33  auto Comp = [&deviceSelector](const device &d1, const device &d2) {
34  return deviceSelector(d1) < deviceSelector(d2);
35  };
36 
37  const device &syclDevice = *std::max_element(Devs.begin(), Devs.end(), Comp);
38  impl = std::make_shared<detail::queue_impl>(
39  detail::getSyclObjImpl(syclDevice), detail::getSyclObjImpl(syclContext),
40  asyncHandler, cl::sycl::detail::QueueOrder::Ordered, propList);
41 }

51 ordered_queue::ordered_queue(cl_command_queue clQueue,
52  const context &syclContext,
53  const async_handler &asyncHandler) {
54  cl_command_queue_properties reportedProps;
55  RT::PiQueue m_CommandQueue = detail::pi::cast<detail::RT::PiQueue>(clQueue);
56  const detail::plugin &Plugin = detail::getSyclObjImpl(syclContext)->getPlugin();
57  Plugin.call<detail::PiApiKind::piQueueGetInfo>(
58  m_CommandQueue, PI_QUEUE_INFO_DEVICE, sizeof(reportedProps),
59  &reportedProps, nullptr);
60  if (reportedProps & CL_QUEUE_OUT_OF_ORDER_EXEC_MODE_ENABLE)
61  throw runtime_error(
62  "Failed to build a sycl ordered queue from a cl OOO queue.",
63  PI_INVALID_OPERATION);
64 
65  impl = std::make_shared<detail::queue_impl>(
66  m_CommandQueue, detail::getSyclObjImpl(syclContext), asyncHandler);
67 }
