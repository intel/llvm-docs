48 template <int ElemsPerAddr,
49  typename = sycl::detail::enable_if_t<
50  (ElemsPerAddr == 1 || ElemsPerAddr == 2 || ElemsPerAddr == 4)>>
51 constexpr unsigned int ElemsPerAddrEncoding() {
52  // encoding requires log2 of ElemsPerAddr
53  if constexpr (ElemsPerAddr == 1)
54  return 0;
55  else if constexpr (ElemsPerAddr == 2)
56  return 1;
57  else if constexpr (ElemsPerAddr == 4)
58  return 2;
59 
60  // other cases not needed since enable_if disallows other values
61 }

63 constexpr unsigned int ElemsPerAddrDecoding(unsigned int ElemsPerAddrEncoded) {
64  // encoding requires 2^ElemsPerAddrEncoded
65  return (1 << ElemsPerAddrEncoded);
66 }

141 // Low-level surface-based gather. Collects elements located at given offsets in
142 // a surface and returns them as a single \ref simd object. Element can be
143 // 1, 2 or 4-byte value, but is always returned as a 4-byte value within the
144 // resulting simd object, with upper 2 or 3 bytes undefined.
145 // Template (compile-time constant) parameters:
146 // @tparam Ty - element type; can only be a 4-byte integer or \c float,
147 // @tparam N - the number of elements
148 // @tparam SurfIndAliasTy - "surface index alias" type - internal type in the
149 // accessor used to denote the surface
150 // @tparam TySizeLog2 - Log2 of the number of bytes read per element:
151 // 0 - 1 byte, 1 - 2 bytes, 2 - 4 bytes
152 // @tparam L1H - L1 cache hint
153 // @tparam L3H - L3 cache hint
154 //
155 // Formal parameters:
156 // @param scale - the scale; must be 0
157 // @param surf_ind - the surface index, taken from the SYCL memory object
158 // @param global_offset - offset added to each individual element's offset to
159 // compute actual memory access offset for that element
160 // @param elem_offsets - per-element offsets
161 //
162 template <typename Ty, int N, typename SurfIndAliasTy, int TySizeLog2,
163  __SEIEE::CacheHint L1H = __SEIEE::CacheHint::None,
164  __SEIEE::CacheHint L3H = __SEIEE::CacheHint::None>
165 SYCL_EXTERNAL SYCL_ESIMD_FUNCTION __SEIEED::vector_type_t<Ty, N>
166 __esimd_surf_read(int16_t scale, SurfIndAliasTy surf_ind,
167  uint32_t global_offset,
168  __SEIEED::vector_type_t<uint32_t, N> elem_offsets)
169 #ifdef __SYCL_DEVICE_ONLY__
170  ;
171 #else
172 {
173  static_assert(N == 1 || N == 8 || N == 16);

524 template <typename Ty, int N, int NumBlk, __SEIEE::CacheHint L1H,
525  __SEIEE::CacheHint L3H>
526 inline __SEIEED::vector_type_t<Ty, N * __SEIEED::ElemsPerAddrDecoding(NumBlk)>
527 __esimd_flat_read(__SEIEED::vector_type_t<uint64_t, N> addrs, int ElemsPerAddr,
528  __SEIEED::vector_type_t<uint16_t, N> pred) {
529  auto NumBlkDecoded = __SEIEED::ElemsPerAddrDecoding(NumBlk);
530  __SEIEED::vector_type_t<Ty, N * __SEIEED::ElemsPerAddrDecoding(NumBlk)> V;
531  ElemsPerAddr = __SEIEED::ElemsPerAddrDecoding(ElemsPerAddr);
532 
533  for (int I = 0; I < N; I++) {
534  if (pred[I]) {
535  Ty *Addr = reinterpret_cast<Ty *>(addrs[I]);
536  if (sizeof(Ty) == 2)
537  ElemsPerAddr = ElemsPerAddr / 2;
538  if (sizeof(Ty) <= 2) {
539  for (int J = 0; J < NumBlkDecoded && J < ElemsPerAddr; J++)
540  V[I * NumBlkDecoded + J] = *(Addr + J);
541  } else {
542  for (int J = 0; J < NumBlkDecoded && J < ElemsPerAddr; J++)
543  V[J * N + I] = *(Addr + J);
544  }
545  }
546  }
547  return V;
548 }

550 template <typename Ty, int N, __SEIEE::rgba_channel_mask Mask,
551  __SEIEE::CacheHint L1H, __SEIEE::CacheHint L3H>
552 inline __SEIEED::vector_type_t<Ty, N * get_num_channels_enabled(Mask)>
553 __esimd_flat_read4(__SEIEED::vector_type_t<uint64_t, N> addrs,
554  __SEIEED::vector_type_t<uint16_t, N> pred) {
555  __SEIEED::vector_type_t<Ty, N * get_num_channels_enabled(Mask)> V;
556  unsigned int Next = 0;
557 
558  if constexpr (__SEIEE::is_channel_enabled(Mask, __SEIEE::rgba_channel::R)) {
559  for (int I = 0; I < N; I++, Next++) {
560  if (pred[I]) {
561  Ty *Addr = reinterpret_cast<Ty *>(addrs[I]);
562  V[Next] = *Addr;
563  }
564  }
565  }
566 
567  if constexpr (__SEIEE::is_channel_enabled(Mask, __SEIEE::rgba_channel::G)) {
568  for (int I = 0; I < N; I++, Next++) {
569  if (pred[I]) {
570  Ty *Addr = reinterpret_cast<Ty *>(addrs[I] + sizeof(Ty));
571  V[Next] = *Addr;
572  }
573  }
574  }
575 
576  if constexpr (__SEIEE::is_channel_enabled(Mask, __SEIEE::rgba_channel::B)) {
577  for (int I = 0; I < N; I++, Next++) {

692 template <typename Ty, int M, int N, typename TACC>
693 inline __SEIEED::vector_type_t<Ty, M * N>
694 __esimd_media_block_load(unsigned modififer, TACC handle, unsigned plane,
695  unsigned width, unsigned x, unsigned y) {
696  // On host the input surface is modeled as sycl image 2d object,
697  // and the read/write access is done through accessor,
698  // which is passed in as the handle argument.
699  auto range = __SEIEED::AccessorPrivateProxy::getImageRange(handle);
700  unsigned bpp = __SEIEED::AccessorPrivateProxy::getElemSize(handle);
701  unsigned vpp = bpp / sizeof(Ty);
702  unsigned int i = x / bpp;
703  unsigned int j = y;
704 
705  assert(x % bpp == 0);
706  unsigned int xbound = range[0] - 1;
707  unsigned int ybound = range[1] - 1;
708 
709  __SEIEED::vector_type_t<Ty, M * N> vals;
710  for (int row = 0; row < M; row++) {
711  for (int col = 0; col < N; col += vpp) {
712  unsigned int xoff = (i > xbound) ? xbound : i;
713  unsigned int yoff = (j > ybound) ? ybound : j;
714  auto coords = cl::sycl::cl_int2(xoff, yoff);
715  cl::sycl::cl_uint4 data = handle.read(coords);
716 
717  __SEIEED::vector_type_t<unsigned int, 4> res;
718  for (int idx = 0; idx < 4; idx++) {
719  res[idx] = data[idx];
720  }
721 
722  constexpr int refN = sizeof(cl::sycl::cl_uint4) / sizeof(Ty);
723  unsigned int stride = sizeof(cl::sycl::cl_uint4) / bpp;
724  using refTy = __SEIEED::vector_type_t<Ty, refN>;
725  auto ref = reinterpret_cast<refTy>(res);
726 
727  unsigned int offset1 = col + row * N;
728  unsigned int offset2 = 0;
729  for (int idx = 0; idx < vpp; idx++) {
730  vals[offset1] = ref[offset2];
731  offset1++;
732  offset2 += stride;
733  }
734  i++;
735  }
736  i = x / bpp;
737  j++;
738  }
739 
740  return vals;
741 }

744 inline void __esimd_media_block_store(unsigned modififer, TACC handle,
745  unsigned plane, unsigned width,
746  unsigned x, unsigned y,
747  __SEIEED::vector_type_t<Ty, M * N> vals) {

757  for (int col = 0; col < N; col += vpp) {
758  constexpr int Sz = sizeof(cl::sycl::cl_uint4) / sizeof(Ty);
759  __SEIEED::vector_type_t<Ty, Sz> res = 0;
