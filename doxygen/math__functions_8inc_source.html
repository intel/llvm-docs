16 template <typename... Ts> struct last_int_rest_same {
17  static constexpr bool value = []() constexpr {
18  constexpr auto N = sizeof...(Ts);
19  using first_type = typename first_type<Ts...>::type;
20  if (!builtin_same_shape_v<first_type>)
21  return false;
22  int i = 0;
23  using int_type =
24  std::conditional_t<is_vec_or_swizzle_v<first_type>, int32_t, int>;
25  return (
26  (... &&
27  (++i == N
28  ? /* last */ builtin_same_shape_v<Ts> // filter out "bad" types,
29  // e.g. multi-ptr
30  && std::is_same_v<get_elem_type_t<Ts>, int_type>
31  : /* not last */ builtin_same_or_swizzle_v<first_type, Ts>)));
32  }();
33 };
34 template <typename... Ts> struct last_intptr_rest_same {
35  static constexpr bool value = []() constexpr {
36  constexpr auto N = sizeof...(Ts);
37  using first_type = typename first_type<Ts...>::type;
38  if (!builtin_same_shape_v<first_type>)
39  return false;
40  int i = 0;
41  using int_type =
42  std::conditional_t<is_vec_or_swizzle_v<first_type>, int32_t, int>;
43  return (
44  (... &&
45  (++i == N
46  ? /* last */ is_multi_ptr_v<Ts> &&
47  has_writeable_addr_space_v<Ts> &&
48  builtin_same_shape_v<first_type, get_elem_type_t<Ts>> &&
49  !is_swizzle_v<get_elem_type_t<Ts>> &&
50  std::is_same_v<get_elem_type_t<get_elem_type_t<Ts>>,
51  int_type>
52  : /* not last */ builtin_same_or_swizzle_v<first_type, Ts>)));
53  }();
54 };

216 template <typename FuncTy, typename PtrTy, typename... Ts>
217 auto builtin_delegate_ptr_impl(FuncTy F, PtrTy p, Ts... xs) {
218  using T0 = typename first_type<Ts...>::type;
219  // Simplify just incase, although most callers seem to do that on their own.
220  simplify_if_swizzle_t<T0> r{};
221 
222  // TODO: Optimize for sizes. Make not to violate ANSI-aliasing rules for the
223  // pointer argument.
224  auto p0 = [&]() {
225  if constexpr (is_multi_ptr_v<PtrTy>)
226  return address_space_cast<PtrTy::address_space,
227  get_multi_ptr_decoration_v<PtrTy>>(&(*p)[0]);
228  else
229  // Deprecated case of raw ptr/host impl.
230  return &(*p)[0];
231  }();
232 
233  constexpr auto N = T0::size();
234  if constexpr (N <= 16)
235  loop<N>([&](auto i) { r[i] = F(xs[i]..., p0 + i); });
236  else
237  for (size_t i = 0; i < N; ++i)
238  r[i] = F(xs[i]..., p0 + i);
239  return r;
240 }

393 #ifdef __SYCL_DEVICE_ONLY__
394 template <typename T0, typename T1> auto sincos_impl(T0 &x, T1 &&y) {
395  if constexpr (is_marray_v<T0>) {
396  return builtin_delegate_ptr_impl(
397  [](auto... xs) { return sincos_impl(xs...); }, y, x);
398  } else {
399  using detail::builtins::convert_arg;
400  if constexpr (use_fast_math_v<T0>) {
401  // This is a performance optimization to ensure that sincos isn't slower
402  // than a pair of sin/cos executed separately. Theoretically, calling
403  // non-native sincos might be faster than calling native::sin plus
404  // native::cos separately and we'd need some kind of cost model to make
405  // the right decision (and move this entirely to the JIT/AOT compilers).
406  // However, in practice, this simpler solution seems to work just fine and
407  // matches how sin/cos above are optimized for the fast math path.
408  *y = __spirv_ocl_native_cos(convert_arg(x));
409  return __spirv_ocl_native_sin(convert_arg(x));
410  } else {
411  return __spirv_ocl_sincos(convert_arg(x), convert_arg(y));
412  }
413  }
414 }
415 #else
416 __SYCL_EXPORT float sincos_impl(float, float *);
417 __SYCL_EXPORT double sincos_impl(double, double *);
418 __SYCL_EXPORT half sincos_impl(half, half *);
419 template <typename T0, typename T1> auto sincos_impl(T0 &x, T1 &&y) {
420  if constexpr (is_multi_ptr_v<std::remove_reference_t<T1>>) {
421  // TODO: Spec needs to be clarified, multi_ptr shouldn't be possible on
422  // host.
423  return sincos_impl(x, y.get_raw());
424  } else {
425  return builtin_delegate_ptr_impl(
426  [](auto... xs) { return sincos_impl(xs...); }, y,
427  simplify_if_swizzle_t<T0>{x});
428  }
429 }
