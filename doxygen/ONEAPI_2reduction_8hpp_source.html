906  bool IsPow2WG>
907 enable_if_t<!Reduction::has_fast_reduce && Reduction::has_fast_atomics>
908 reduCGFuncImpl(handler &CGH, KernelType KernelFunc, const nd_range<Dims> &Range,
909  Reduction &, typename Reduction::rw_accessor_type Out) {
910  size_t WGSize = Range.get_local_range().size();
911 
912  // Use local memory to reduce elements in work-groups into zero-th element.
913  // If WGSize is not power of two, then WGSize+1 elements are allocated.
914  // The additional last element is used to catch reduce elements that could
915  // otherwise be lost in the tree-reduction algorithm used in the kernel.
916  size_t NLocalElements = WGSize + (IsPow2WG ? 0 : 1);
917  auto LocalReds = Reduction::getReadWriteLocalAcc(NLocalElements, CGH);
918 
919  using Name = typename get_reduction_main_kernel_name_t<
920  KernelName, KernelType, Reduction::is_usm, IsPow2WG>::name;
921  CGH.parallel_for<Name>(Range, [=](nd_item<Dims> NDIt) {
922  // Call user's functions. Reducer.MValue gets initialized there.
923  typename Reduction::reducer_type Reducer;
924  KernelFunc(NDIt, Reducer);
925 
926  size_t WGSize = NDIt.get_local_range().size();
927  size_t LID = NDIt.get_local_linear_id();
928 
929  // Copy the element to local memory to prepare it for tree-reduction.
930  LocalReds[LID] = Reducer.MValue;
931  if (!IsPow2WG)
932  LocalReds[WGSize] = Reducer.getIdentity();
933  NDIt.barrier();
934 
935  // Tree-reduction: reduce the local array LocalReds[:] to LocalReds[0].
936  // LocalReds[WGSize] accumulates last/odd elements when the step
937  // of tree-reduction loop is not even.
938  typename Reduction::binary_operation BOp;
939  size_t PrevStep = WGSize;
940  for (size_t CurStep = PrevStep >> 1; CurStep > 0; CurStep >>= 1) {
941  if (LID < CurStep)
942  LocalReds[LID] = BOp(LocalReds[LID], LocalReds[LID + CurStep]);
943  else if (!IsPow2WG && LID == CurStep && (PrevStep & 0x1))
944  LocalReds[WGSize] = BOp(LocalReds[WGSize], LocalReds[PrevStep - 1]);
945  NDIt.barrier();
946  PrevStep = CurStep;
947  }
948 
949  if (LID == 0) {
950  Reducer.MValue =
951  IsPow2WG ? LocalReds[0] : BOp(LocalReds[0], LocalReds[WGSize]);
952  Reducer.atomic_combine(Reduction::getOutPointer(Out));
953  }
954  });
955 }
