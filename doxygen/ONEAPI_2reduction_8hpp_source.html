838  Reduction &, OutputT Out) {
839  size_t WGSize = Range.get_local_range().size();
840 
841  // Use local memory to reduce elements in work-groups into zero-th element.
842  // If WGSize is not power of two, then WGSize+1 elements are allocated.
843  // The additional last element is used to catch reduce elements that could
844  // otherwise be lost in the tree-reduction algorithm used in the kernel.
845  size_t NLocalElements = WGSize + (IsPow2WG ? 0 : 1);
846  auto LocalReds = Reduction::getReadWriteLocalAcc(NLocalElements, CGH);
847

849  KernelName, KernelType, Reduction::is_usm, IsPow2WG, OutputT>::name;

852  typename Reduction::reducer_type Reducer;
853  KernelFunc(NDIt, Reducer);
854

859  LocalReds[LID] = Reducer.MValue;

861  LocalReds[WGSize] = Reducer.getIdentity();
862  NDIt.barrier();

865  // LocalReds[WGSize] accumulates last/odd elements when the step

872  else if (!IsPow2WG && LID == CurStep && (PrevStep & 0x1))

874  NDIt.barrier();

880  IsPow2WG ? LocalReds[0] : BOp(LocalReds[0], LocalReds[WGSize]);

1420 template <typename KernelName, bool Pow2WG, bool IsOneWG, typename KernelType,
1421  int Dims, typename... Reductions, size_t... Is>
1422 void reduCGFuncImpl(handler &CGH, KernelType KernelFunc,
1423  const nd_range<Dims> &Range,
1424  std::tuple<Reductions...> &ReduTuple,
1425  std::index_sequence<Is...> ReduIndices) {
1426 
1427  size_t WGSize = Range.get_local_range().size();
1428  size_t LocalAccSize = WGSize + (Pow2WG ? 0 : 1);
1429  auto LocalAccsTuple =
1430  createReduLocalAccs<Reductions...>(LocalAccSize, CGH, ReduIndices);
1431 
1432  size_t NWorkGroups = IsOneWG ? 1 : Range.get_group_range().size();
1433  auto OutAccsTuple =
1434  createReduOutAccs<IsOneWG>(NWorkGroups, CGH, ReduTuple, ReduIndices);
1435  auto IdentitiesTuple = getReduIdentities(ReduTuple, ReduIndices);
1436  auto BOPsTuple = getReduBOPs(ReduTuple, ReduIndices);
1437 
1438  using Name = typename get_reduction_main_kernel_name_t<
1439  KernelName, KernelType, Pow2WG, IsOneWG, decltype(OutAccsTuple)>::name;
1440  CGH.parallel_for<Name>(Range, [=](nd_item<Dims> NDIt) {
1441  auto ReduIndices = std::index_sequence_for<Reductions...>();
1442  auto ReducersTuple =
1443  createReducers<Reductions...>(IdentitiesTuple, BOPsTuple, ReduIndices);
1444  // The .MValue field of each of the elements in ReducersTuple
1445  // gets initialized in this call.
1446  callReduUserKernelFunc(KernelFunc, NDIt, ReducersTuple, ReduIndices);
1447 
1448  size_t WGSize = NDIt.get_local_range().size();
1449  size_t LID = NDIt.get_local_linear_id();
1450  initReduLocalAccs<Pow2WG>(LID, WGSize, LocalAccsTuple, ReducersTuple,
1451  IdentitiesTuple, ReduIndices);
1452  NDIt.barrier();
1453 
1454  size_t PrevStep = WGSize;
1455  for (size_t CurStep = PrevStep >> 1; CurStep > 0; CurStep >>= 1) {
1456  if (LID < CurStep) {
1457  // LocalReds[LID] = BOp(LocalReds[LID], LocalReds[LID + CurStep]);
1458  reduceReduLocalAccs(LID, LID + CurStep, LocalAccsTuple, BOPsTuple,
1459  ReduIndices);
1460  } else if (!Pow2WG && LID == CurStep && (PrevStep & 0x1)) {
1461  // LocalReds[WGSize] = BOp(LocalReds[WGSize], LocalReds[PrevStep - 1]);
1462  reduceReduLocalAccs(WGSize, PrevStep - 1, LocalAccsTuple, BOPsTuple,
1463  ReduIndices);
1464  }
1465  NDIt.barrier();
1466  PrevStep = CurStep;
1467  }
1468 
1469  // Compute the partial sum/reduction for the work-group.
1470  if (LID == 0) {
1471  size_t GrID = NDIt.get_group_linear_id();
1472  // If there is only one work-group, then the original accessors need to be
1473  // updated, i.e. after the work in each work-group is done, the work-group
1474  // result is added to the original value of the read-write accessors or
1475  // USM memory.
1476  std::conditional_t<IsOneWG, IsRWReductionPredicate,
1477  EmptyReductionPredicate>
1478  Predicate;
1479  auto RWReduIndices =
1480  filterSequence<Reductions...>(Predicate, ReduIndices);
1481  writeReduSumsToOutAccs<Pow2WG>(
1482  GrID, WGSize, (std::tuple<Reductions...> *)nullptr, OutAccsTuple,
1483  LocalAccsTuple, BOPsTuple, ReduIndices, RWReduIndices);
1484  }
1485  });
