29 // TODO @Pennycook
30 // {quote}
31 // ...I'd like us to think more about what we can do to make these interfaces
32 // more user - friendly. A user providing cache hints has to provide a lot more
33 // template arguments than required.Could we make this nicer by providing the
34 // hints as tag - type arguments ?
35 // ...
36 // // Without cache hints, type and length can be deduced from offsets
37 // float* p;
38 // simd<uint32_t, 16> offsets;
39 // auto result = flat_load(p, offsets);
40 //
41 // // With cache hints as templates, verbosity increases significantly:
42 // // - Providing any cache hint forces the user to specify the type and
43 // length float* p; simd<uint32_t, 16> offsets; auto result =
44 // flat_load<uint32_t, 16, 1, CacheHint::Foo, CacheHint::Bar>(p, offsets);
45 //
46 // // With cache hints as tag types, verbosity is reduced:
47 // // - Providing a cache hint does not prevent deduction of type and length
48 // float* p;
49 // simd <uint32_t, 16> offsets;
50 // auto result = flat_load(p, offsets, CacheHint::Foo{});
51 //
52 // Note also that the templated form prevents a developer from specifying an L3
53 // hint without also explicitly specifying an L1 hint. If flat_load accepted a
54 // list of hints, it might be possible to refactor the hints to specify them in
55 // any order, and it may be more extensible to future cache hints:
56 // {/quote}
57 //
58 // TODO @keryell
59 // {quote}
60 // An approach a la https ://github.com/chriskohlhoff/propria from
61 // @chriskohlhoff would be to add a property to the pointer, such as
62 //
63 // auto result = flat_load(p, offsets);
64 // auto result = flat_load(decorate<CacheHint::Foo, CacheHint::Bar>(p),
65 // offsets);
66 // The advantage is that you do not have to change all tour API and all the uses
67 // of this decorated pointer will benefit from this. decorate is to be bikeshed
68 // accordingly.
69 // {/quote}

73 template <typename T, int n, int ElemsPerAddr = 1,
74  CacheHint L1H = CacheHint::None, CacheHint L3H = CacheHint::None>
75 ESIMD_INLINE ESIMD_NODEBUG typename sycl::detail::enable_if_t<
76  ((n == 8 || n == 16 || n == 32) &&
77  (ElemsPerAddr == 1 || ElemsPerAddr == 2 || ElemsPerAddr == 4)),
78  simd<T, n * ElemsPerAddr>>
79 gather(T *p, simd<uint32_t, n> offsets, simd<uint16_t, n> pred = 1) {
80 
81  simd<uint64_t, n> offsets_i = convert<uint64_t>(offsets);
82  simd<uint64_t, n> addrs(reinterpret_cast<uint64_t>(p));
83  addrs = addrs + offsets_i;
84 
85  if constexpr (sizeof(T) == 1 && ElemsPerAddr == 2) {
86  auto Ret =
87  __esimd_flat_read<T, n, detail::ElemsPerAddrEncoding<4>(), L1H, L3H>(
88  addrs.data(), detail::ElemsPerAddrEncoding<ElemsPerAddr>(),
89  pred.data());
90  return __esimd_rdregion<T, n * 4, n * ElemsPerAddr, /*VS*/ 4, 2, 1>(Ret, 0);
91  } else if constexpr (sizeof(T) == 1 && ElemsPerAddr == 1) {
92  auto Ret =
93  __esimd_flat_read<T, n, detail::ElemsPerAddrEncoding<4>(), L1H, L3H>(
94  addrs.data(), detail::ElemsPerAddrEncoding<ElemsPerAddr>(),
95  pred.data());
96  return __esimd_rdregion<T, n * 4, n * ElemsPerAddr, /*VS*/ 0, n, 4>(Ret, 0);
97  } else if constexpr (sizeof(T) == 2 && ElemsPerAddr == 1) {
98  auto Ret =
99  __esimd_flat_read<T, n, detail::ElemsPerAddrEncoding<2>(), L1H, L3H>(
100  addrs.data(), detail::ElemsPerAddrEncoding<2>(), pred.data());
101  return __esimd_rdregion<T, n * 2, n, /*VS*/ 0, n, 2>(Ret, 0);
102  } else if constexpr (sizeof(T) == 2)
103  return __esimd_flat_read<T, n, detail::ElemsPerAddrEncoding<ElemsPerAddr>(),
104  L1H, L3H>(
105  addrs.data(), detail::ElemsPerAddrEncoding<2 * ElemsPerAddr>(),
106  pred.data());
107  else
108  return __esimd_flat_read<T, n, detail::ElemsPerAddrEncoding<ElemsPerAddr>(),
109  L1H, L3H>(
110  addrs.data(), detail::ElemsPerAddrEncoding<ElemsPerAddr>(),
111  pred.data());
112 }

122 template <typename T, int n, int ElemsPerAddr = 1,
123  CacheHint L1H = CacheHint::None, CacheHint L3H = CacheHint::None>
124 ESIMD_INLINE ESIMD_NODEBUG typename sycl::detail::enable_if_t<
125  ((n == 8 || n == 16 || n == 32) &&
126  (ElemsPerAddr == 1 || ElemsPerAddr == 2 || ElemsPerAddr == 4)),
127  void>
128 scatter(T *p, simd<T, n * ElemsPerAddr> vals, simd<uint32_t, n> offsets,
129  simd<uint16_t, n> pred = 1) {
130  simd<uint64_t, n> offsets_i = convert<uint64_t>(offsets);
131  simd<uint64_t, n> addrs(reinterpret_cast<uint64_t>(p));
132  addrs = addrs + offsets_i;
133  if constexpr (sizeof(T) == 1 && ElemsPerAddr == 2) {
134  simd<T, n * 4> D;
135  D = __esimd_wrregion<T, n * 4, n * ElemsPerAddr, /*VS*/ 4, 2, 1>(
136  D.data(), vals.data(), 0);
137  __esimd_flat_write<T, n, detail::ElemsPerAddrEncoding<4>(), L1H, L3H>(
138  addrs.data(), D.data(), detail::ElemsPerAddrEncoding<ElemsPerAddr>(),
139  pred.data());
140  } else if constexpr (sizeof(T) == 1 && ElemsPerAddr == 1) {
141  simd<T, n * 4> D;
142  D = __esimd_wrregion<T, n * 4, n * ElemsPerAddr, /*VS*/ 0, n, 4>(
143  D.data(), vals.data(), 0);
144  __esimd_flat_write<T, n, detail::ElemsPerAddrEncoding<4>(), L1H, L3H>(
145  addrs.data(), D.data(), detail::ElemsPerAddrEncoding<ElemsPerAddr>(),
146  pred.data());
147  } else if constexpr (sizeof(T) == 2 && ElemsPerAddr == 1) {
148  simd<T, n * 2> D;
149  D = __esimd_wrregion<T, n * 2, n, /*VS*/ 0, n, 2>(D.data(), vals.data(), 0);
150  __esimd_flat_write<T, n, detail::ElemsPerAddrEncoding<2>(), L1H, L3H>(
151  addrs.data(), D.data(), detail::ElemsPerAddrEncoding<2>(), pred.data());
152  } else if constexpr (sizeof(T) == 2)
153  __esimd_flat_write<T, n, detail::ElemsPerAddrEncoding<ElemsPerAddr>(), L1H,
154  L3H>(addrs.data(), vals.data(),
155  detail::ElemsPerAddrEncoding<2 * ElemsPerAddr>(),
156  pred.data());
157  else
158  __esimd_flat_write<T, n, detail::ElemsPerAddrEncoding<ElemsPerAddr>(), L1H,
159  L3H>(addrs.data(), vals.data(),
160  detail::ElemsPerAddrEncoding<ElemsPerAddr>(),
161  pred.data());
162 }

185  return __esimd_flat_block_read_unaligned<T, n, L1H, L3H>(Addr);

245 template <typename T, int N, typename AccessorTy,
246  CacheHint L1H = CacheHint::None, CacheHint L3H = CacheHint::None>
247 ESIMD_INLINE ESIMD_NODEBUG
248  typename sycl::detail::enable_if_t<(sizeof(T) <= 4) &&
249  (N == 1 || N == 8 || N == 16) &&
250  !std::is_pointer<AccessorTy>::value,
251  simd<T, N>>
252  gather(AccessorTy acc, simd<uint32_t, N> offsets,
253  uint32_t glob_offset = 0) {
254 
255  constexpr int TypeSizeLog2 = detail::ElemsPerAddrEncoding<sizeof(T)>();
256  // TODO (performance) use hardware-supported scale once BE supports it
257  constexpr uint32_t scale = 0;
258  constexpr uint32_t t_scale = sizeof(T);
259  if constexpr (t_scale > 1) {
260  glob_offset *= t_scale;
261  offsets *= t_scale;
262  }
263 
264  if constexpr (sizeof(T) < 4) {
265  static_assert(std::is_integral<T>::value,
266  "only integral 1- & 2-byte types are supported");
267  using PromoT =
268  typename sycl::detail::conditional_t<std::is_signed<T>::value, int32_t,
269  uint32_t>;
270 #if defined(__SYCL_DEVICE_ONLY__)
271  const auto surf_ind = detail::AccessorPrivateProxy::getNativeImageObj(acc);
272  const simd<PromoT, N> promo_vals =
273  __esimd_surf_read<PromoT, N, decltype(surf_ind), TypeSizeLog2, L1H,
274  L3H>(scale, surf_ind, glob_offset, offsets);
275 #else
276  const simd<PromoT, N> promo_vals =
277  __esimd_surf_read<PromoT, N, AccessorTy, TypeSizeLog2, L1H, L3H>(
278  scale, acc, glob_offset, offsets);
279 #endif
280  return convert<T>(promo_vals);
281  } else {
282 #if defined(__SYCL_DEVICE_ONLY__)
283  const auto surf_ind = detail::AccessorPrivateProxy::getNativeImageObj(acc);
284  return __esimd_surf_read<T, N, decltype(surf_ind), TypeSizeLog2, L1H, L3H>(
285  scale, surf_ind, glob_offset, offsets);
286 #else
287  return __esimd_surf_read<T, N, AccessorTy, TypeSizeLog2, L1H, L3H>(
288  scale, acc, glob_offset, offsets);
289 #endif
290  }
291 }

415 template <EsimdAtomicOpType Op, typename T, int N, unsigned NumSrc>
416 constexpr bool check_atomic() {
417  if constexpr (!detail::isPowerOf2(N, 32)) {
418  static_assert((detail::isPowerOf2(N, 32)),
419  "Execution size 1, 2, 4, 8, 16, 32 are supported");
420  return false;
421  }
422 
423  // No source operand.
424  if constexpr (Op == EsimdAtomicOpType::ATOMIC_INC ||
425  Op == EsimdAtomicOpType::ATOMIC_DEC) {
426  if constexpr (NumSrc != 0) {
427  static_assert(NumSrc == 0, "No source operands are expected");
428  return false;
429  }
430  if constexpr (!is_type<T, uint16_t, uint32_t, uint64_t>()) {
431  static_assert((is_type<T, uint16_t, uint32_t, uint64_t>()),
432  "Type UW, UD or UQ is expected");
433  return false;
434  }
435  return true;
436  }
437 
438  // One source integer operand.
439  if constexpr (Op == EsimdAtomicOpType::ATOMIC_ADD ||
440  Op == EsimdAtomicOpType::ATOMIC_SUB ||
441  Op == EsimdAtomicOpType::ATOMIC_MIN ||
442  Op == EsimdAtomicOpType::ATOMIC_MAX ||
443  Op == EsimdAtomicOpType::ATOMIC_XCHG ||
444  Op == EsimdAtomicOpType::ATOMIC_AND ||
445  Op == EsimdAtomicOpType::ATOMIC_OR ||
446  Op == EsimdAtomicOpType::ATOMIC_XOR ||
447  Op == EsimdAtomicOpType::ATOMIC_MINSINT ||
448  Op == EsimdAtomicOpType::ATOMIC_MAXSINT) {
449  if constexpr (NumSrc != 1) {
450  static_assert(NumSrc == 1, "One source operand is expected");
451  return false;
452  }
453  if constexpr ((Op != EsimdAtomicOpType::ATOMIC_MINSINT &&
454  Op != EsimdAtomicOpType::ATOMIC_MAXSINT) &&
455  !is_type<T, uint16_t, uint32_t, uint64_t>()) {
456  static_assert((is_type<T, uint16_t, uint32_t, uint64_t>()),
457  "Type UW, UD or UQ is expected");
458  return false;
459  }
460  if constexpr ((Op == EsimdAtomicOpType::ATOMIC_MINSINT ||
461  Op == EsimdAtomicOpType::ATOMIC_MAXSINT) &&
462  !is_type<T, int16_t, int32_t, int64_t>()) {
463  static_assert((is_type<T, int16_t, int32_t, int64_t>()),
464  "Type W, D or Q is expected");
465  return false;
466  }
467  return true;
468  }
469 
470  // One source float operand.
471  if constexpr (Op == EsimdAtomicOpType::ATOMIC_FMAX ||
472  Op == EsimdAtomicOpType::ATOMIC_FMIN) {
473  if constexpr (NumSrc != 1) {
474  static_assert(NumSrc == 1, "One source operand is expected");
475  return false;
476  }
477  if constexpr (!is_type<T, float, cl::sycl::detail::half_impl::StorageT>()) {
478  static_assert(
479  (is_type<T, float, cl::sycl::detail::half_impl::StorageT>()),
480  "Type F or HF is expected");
481  return false;
482  }
483  return true;
484  }
485 
486  // Two scouce operands.
487  if constexpr (Op == EsimdAtomicOpType::ATOMIC_CMPXCHG ||
488  Op == EsimdAtomicOpType::ATOMIC_FCMPWR) {
489  if constexpr (NumSrc != 2) {
490  static_assert(NumSrc == 2, "Two source operands are expected");
491  return false;
492  }
493  if constexpr (Op == EsimdAtomicOpType::ATOMIC_CMPXCHG &&
494  !is_type<T, uint16_t, uint32_t, uint64_t>()) {
495  static_assert((is_type<T, uint16_t, uint32_t, uint64_t>()),
496  "Type UW, UD or UQ is expected");
497  return false;
498  }
499  if constexpr (Op == EsimdAtomicOpType::ATOMIC_FCMPWR &&
500  !is_type<T, float, cl::sycl::detail::half_impl::StorageT>()) {
501  static_assert(
502  (is_type<T, float, cl::sycl::detail::half_impl::StorageT>()),
503  "Type F or HF is expected");
504  return false;
505  }
506  return true;
507  }
508  // Unsupported svm atomic Op.
509  return false;
510 }
