26 template <int ElemsPerAddr,
27  typename = std::enable_if_t<(ElemsPerAddr == 1 || ElemsPerAddr == 2 ||
28  ElemsPerAddr == 4)>>
29 constexpr unsigned int ElemsPerAddrEncoding() {
30  // encoding requires log2 of ElemsPerAddr
31  if constexpr (ElemsPerAddr == 1)
32  return 0;
33  else if constexpr (ElemsPerAddr == 2)
34  return 1;
35  else if constexpr (ElemsPerAddr == 4)
36  return 2;
37 
38  // other cases not needed since enable_if disallows other values
39 }

41 // TODO @Pennycook
42 // {quote}
43 // ...I'd like us to think more about what we can do to make these interfaces
44 // more user - friendly. A user providing cache hints has to provide a lot more
45 // template arguments than required.Could we make this nicer by providing the
46 // hints as tag - type arguments ?
47 // ...
48 // // Without cache hints, type and length can be deduced from offsets
49 // float* p;
50 // simd<uint32_t, 16> offsets;
51 // auto result = flat_load(p, offsets);
52 //
53 // // With cache hints as templates, verbosity increases significantly:
54 // // - Providing any cache hint forces the user to specify the type and
55 // length float* p; simd<uint32_t, 16> offsets; auto result =
56 // flat_load<uint32_t, 16, 1, CacheHint::Foo, CacheHint::Bar>(p, offsets);
57 //
58 // // With cache hints as tag types, verbosity is reduced:
59 // // - Providing a cache hint does not prevent deduction of type and length
60 // float* p;
61 // simd <uint32_t, 16> offsets;
62 // auto result = flat_load(p, offsets, CacheHint::Foo{});
63 //
64 // Note also that the templated form prevents a developer from specifying an L3
65 // hint without also explicitly specifying an L1 hint. If flat_load accepted a
66 // list of hints, it might be possible to refactor the hints to specify them in
67 // any order, and it may be more extensible to future cache hints:
68 // {/quote}
69 //
70 // TODO @keryell
71 // {quote}
72 // An approach ŕ la https ://github.com/chriskohlhoff/propria from
73 // @chriskohlhoff would be to add a property to the pointer, such as
74 //
75 // auto result = flat_load(p, offsets);
76 // auto result = flat_load(decorate<CacheHint::Foo, CacheHint::Bar>(p),
77 // offsets);
78 // The advantage is that you do not have to change all tour API and all the uses
79 // of this decorated pointer will benefit from this. decorate is to be bikeshed
80 // accordingly.
81 // {/quote}

84 template <typename T, int n, int ElemsPerAddr = 1,
85  CacheHint L1H = CacheHint::None, CacheHint L3H = CacheHint::None>
86 ESIMD_INLINE ESIMD_NODEBUG
87  typename std::enable_if<((n == 8 || n == 16 || n == 32) &&
88  (ElemsPerAddr == 1 || ElemsPerAddr == 2 ||
89  ElemsPerAddr == 4)),
90  simd<T, n * ElemsPerAddr>>::type
91  gather(T *p, simd<uint32_t, n> offsets, simd<uint16_t, n> pred = 1) {
92 
93  simd<uint64_t, n> offsets_i = convert<uint64_t>(offsets);
94  simd<uint64_t, n> addrs(reinterpret_cast<uint64_t>(p));
95  addrs = addrs + offsets_i;
96 
97  if constexpr (sizeof(T) == 1 && ElemsPerAddr == 2) {
98  auto Ret = __esimd_flat_read<T, n, ElemsPerAddrEncoding<4>(), L1H, L3H>(
99  addrs.data(), ElemsPerAddrEncoding<ElemsPerAddr>(), pred.data());
100  return __esimd_rdregion<T, n * 4, n * ElemsPerAddr, /*VS*/ 4, 2, 1>(Ret, 0);
101  } else if constexpr (sizeof(T) == 1 && ElemsPerAddr == 1) {
102  auto Ret = __esimd_flat_read<T, n, ElemsPerAddrEncoding<4>(), L1H, L3H>(
103  addrs.data(), ElemsPerAddrEncoding<ElemsPerAddr>(), pred.data());
104  return __esimd_rdregion<T, n * 4, n * ElemsPerAddr, /*VS*/ 0, n, 4>(Ret, 0);
105  } else if constexpr (sizeof(T) == 2 && ElemsPerAddr == 1) {
106  auto Ret = __esimd_flat_read<T, n, ElemsPerAddrEncoding<2>(), L1H, L3H>(
107  addrs.data(), ElemsPerAddrEncoding<2>(), pred.data());
108  return __esimd_rdregion<T, n * 2, n, /*VS*/ 0, n, 2>(Ret, 0);
109  } else if constexpr (sizeof(T) == 2)
110  return __esimd_flat_read<T, n, ElemsPerAddrEncoding<ElemsPerAddr>(), L1H,
111  L3H>(
112  addrs.data(), ElemsPerAddrEncoding<2 * ElemsPerAddr>(), pred.data());
113  else
114  return __esimd_flat_read<T, n, ElemsPerAddrEncoding<ElemsPerAddr>(), L1H,
115  L3H>(
116  addrs.data(), ElemsPerAddrEncoding<ElemsPerAddr>(), pred.data());
117 }

120 template <typename T, int n, int ElemsPerAddr = 1,
121  CacheHint L1H = CacheHint::None, CacheHint L3H = CacheHint::None>
122 ESIMD_INLINE ESIMD_NODEBUG
123  typename std::enable_if<((n == 8 || n == 16 || n == 32) &&
124  (ElemsPerAddr == 1 || ElemsPerAddr == 2 ||
125  ElemsPerAddr == 4)),
126  void>::type
127  scatter(T *p, simd<T, n * ElemsPerAddr> vals, simd<uint32_t, n> offsets,
128  simd<uint16_t, n> pred = 1) {
129  simd<uint64_t, n> offsets_i = convert<uint64_t>(offsets);
130  simd<uint64_t, n> addrs(reinterpret_cast<uint64_t>(p));
131  addrs = addrs + offsets_i;
132  if constexpr (sizeof(T) == 1 && ElemsPerAddr == 2) {
133  simd<T, n * 4> D;
134  D = __esimd_wrregion<T, n * 4, n * ElemsPerAddr, /*VS*/ 4, 2, 1>(
135  D.data(), vals.data(), 0);
136  __esimd_flat_write<T, n, ElemsPerAddrEncoding<4>(), L1H, L3H>(
137  addrs.data(), D.data(), ElemsPerAddrEncoding<ElemsPerAddr>(),
138  pred.data());
139  } else if constexpr (sizeof(T) == 1 && ElemsPerAddr == 1) {
140  simd<T, n * 4> D;
141  D = __esimd_wrregion<T, n * 4, n * ElemsPerAddr, /*VS*/ 0, n, 4>(
142  D.data(), vals.data(), 0);
143  __esimd_flat_write<T, n, ElemsPerAddrEncoding<4>(), L1H, L3H>(
144  addrs.data(), D.data(), ElemsPerAddrEncoding<ElemsPerAddr>(),
145  pred.data());
146  } else if constexpr (sizeof(T) == 2 && ElemsPerAddr == 1) {
147  simd<T, n * 2> D;
148  D = __esimd_wrregion<T, n * 2, n, /*VS*/ 0, n, 2>(D.data(), vals.data(), 0);
149  __esimd_flat_write<T, n, ElemsPerAddrEncoding<2>(), L1H, L3H>(
150  addrs.data(), D.data(), ElemsPerAddrEncoding<2>(), pred.data());
151  } else if constexpr (sizeof(T) == 2)
152  __esimd_flat_write<T, n, ElemsPerAddrEncoding<ElemsPerAddr>(), L1H, L3H>(
153  addrs.data(), vals.data(), ElemsPerAddrEncoding<2 * ElemsPerAddr>(),
154  pred.data());
155  else
156  __esimd_flat_write<T, n, ElemsPerAddrEncoding<ElemsPerAddr>(), L1H, L3H>(
157  addrs.data(), vals.data(), ElemsPerAddrEncoding<ElemsPerAddr>(),
158  pred.data());
159 }

166 template <typename T, int n, CacheHint L1H = CacheHint::None,
167  CacheHint L3H = CacheHint::None>
168 ESIMD_INLINE ESIMD_NODEBUG simd<T, n> block_load(const T *const addr) {
169  constexpr unsigned Sz = sizeof(T) * n;
170  static_assert(Sz >= __esimd::OWORD, "block size must be at least 1 oword");
171  static_assert(Sz % __esimd::OWORD == 0,
172  "block size must be whole number of owords");
173  static_assert(__esimd::isPowerOf2(Sz / __esimd::OWORD),
174  "block must be 1, 2, 4 or 8 owords long");
175  static_assert(Sz <= 8 * __esimd::OWORD,
176  "block size must be at most 8 owords");
177 
178  uintptr_t Addr = reinterpret_cast<uintptr_t>(addr);
179  return __esimd_flat_block_read_unaligned<T, n, L1H, L3H>(Addr);
180 }

276 template <EsimdAtomicOpType Op, typename T, int N, unsigned NumSrc>
277 constexpr bool check_atomic() {
278  if constexpr (!__esimd::isPowerOf2(N, 32)) {
279  static_assert((__esimd::isPowerOf2(N, 32)),
280  "Execution size 1, 2, 4, 8, 16, 32 are supported");
281  return false;
282  }
283 
284  // No source operand.
285  if constexpr (Op == EsimdAtomicOpType::ATOMIC_INC ||
286  Op == EsimdAtomicOpType::ATOMIC_DEC) {
287  if constexpr (NumSrc != 0) {
288  static_assert(NumSrc == 0, "No source operands are expected");
289  return false;
290  }
291  if constexpr (!is_type<T, uint16_t, uint32_t, uint64_t>()) {
292  static_assert((is_type<T, uint16_t, uint32_t, uint64_t>()),
293  "Type UW, UD or UQ is expected");
294  return false;
295  }
296  return true;
297  }
298 
299  // One source integer operand.
300  if constexpr (Op == EsimdAtomicOpType::ATOMIC_ADD ||
301  Op == EsimdAtomicOpType::ATOMIC_SUB ||
302  Op == EsimdAtomicOpType::ATOMIC_MIN ||
303  Op == EsimdAtomicOpType::ATOMIC_MAX ||
304  Op == EsimdAtomicOpType::ATOMIC_XCHG ||
305  Op == EsimdAtomicOpType::ATOMIC_AND ||
306  Op == EsimdAtomicOpType::ATOMIC_OR ||
307  Op == EsimdAtomicOpType::ATOMIC_XOR ||
308  Op == EsimdAtomicOpType::ATOMIC_MINSINT ||
309  Op == EsimdAtomicOpType::ATOMIC_MAXSINT) {
310  if constexpr (NumSrc != 1) {
311  static_assert(NumSrc == 1, "One source operand is expected");
312  return false;
313  }
314  if constexpr ((Op != EsimdAtomicOpType::ATOMIC_MINSINT &&
315  Op != EsimdAtomicOpType::ATOMIC_MAXSINT) &&
316  !is_type<T, uint16_t, uint32_t, uint64_t>()) {
317  static_assert((is_type<T, uint16_t, uint32_t, uint64_t>()),
318  "Type UW, UD or UQ is expected");
319  return false;
320  }
321  if constexpr ((Op == EsimdAtomicOpType::ATOMIC_MINSINT ||
322  Op == EsimdAtomicOpType::ATOMIC_MAXSINT) &&
323  !is_type<T, int16_t, int32_t, int64_t>()) {
324  static_assert((is_type<T, int16_t, int32_t, int64_t>()),
325  "Type W, D or Q is expected");
326  return false;
327  }
328  return true;
329  }
330 
331  // One source float operand.
332  if constexpr (Op == EsimdAtomicOpType::ATOMIC_FMAX ||
333  Op == EsimdAtomicOpType::ATOMIC_FMIN) {
334  if constexpr (NumSrc != 1) {
335  static_assert(NumSrc == 1, "One source operand is expected");
336  return false;
337  }
338  if constexpr (!is_type<T, float, cl::sycl::detail::half_impl::StorageT>()) {
339  static_assert(
340  (is_type<T, float, cl::sycl::detail::half_impl::StorageT>()),
341  "Type F or HF is expected");
342  return false;
343  }
344  return true;
345  }
346 
347  // Two scouce operands.
348  if constexpr (Op == EsimdAtomicOpType::ATOMIC_CMPXCHG ||
349  Op == EsimdAtomicOpType::ATOMIC_FCMPWR) {
350  if constexpr (NumSrc != 2) {
351  static_assert(NumSrc == 2, "Two source operands are expected");
352  return false;
353  }
354  if constexpr (Op == EsimdAtomicOpType::ATOMIC_CMPXCHG &&
355  !is_type<T, uint16_t, uint32_t, uint64_t>()) {
356  static_assert((is_type<T, uint16_t, uint32_t, uint64_t>()),
357  "Type UW, UD or UQ is expected");
358  return false;
359  }
360  if constexpr (Op == EsimdAtomicOpType::ATOMIC_FCMPWR &&
361  !is_type<T, float, cl::sycl::detail::half_impl::StorageT>()) {
362  static_assert(
363  (is_type<T, float, cl::sycl::detail::half_impl::StorageT>()),
364  "Type F or HF is expected");
365  return false;
366  }
367  return true;
368  }
369  // Unsupported svm atomic Op.
370  return false;
371 }
