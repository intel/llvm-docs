26 template <int ElemsPerAddr,
27  typename = sycl::detail::enable_if_t<
28  (ElemsPerAddr == 1 || ElemsPerAddr == 2 || ElemsPerAddr == 4)>>
29 constexpr unsigned int ElemsPerAddrEncoding() {
30  // encoding requires log2 of ElemsPerAddr
31  if constexpr (ElemsPerAddr == 1)
32  return 0;
33  else if constexpr (ElemsPerAddr == 2)
34  return 1;
35  else if constexpr (ElemsPerAddr == 4)
36  return 2;
37 
38  // other cases not needed since enable_if disallows other values
39 }

41 // TODO @Pennycook
42 // {quote}
43 // ...I'd like us to think more about what we can do to make these interfaces
44 // more user - friendly. A user providing cache hints has to provide a lot more
45 // template arguments than required.Could we make this nicer by providing the
46 // hints as tag - type arguments ?
47 // ...
48 // // Without cache hints, type and length can be deduced from offsets
49 // float* p;
50 // simd<uint32_t, 16> offsets;
51 // auto result = flat_load(p, offsets);
52 //
53 // // With cache hints as templates, verbosity increases significantly:
54 // // - Providing any cache hint forces the user to specify the type and
55 // length float* p; simd<uint32_t, 16> offsets; auto result =
56 // flat_load<uint32_t, 16, 1, CacheHint::Foo, CacheHint::Bar>(p, offsets);
57 //
58 // // With cache hints as tag types, verbosity is reduced:
59 // // - Providing a cache hint does not prevent deduction of type and length
60 // float* p;
61 // simd <uint32_t, 16> offsets;
62 // auto result = flat_load(p, offsets, CacheHint::Foo{});
63 //
64 // Note also that the templated form prevents a developer from specifying an L3
65 // hint without also explicitly specifying an L1 hint. If flat_load accepted a
66 // list of hints, it might be possible to refactor the hints to specify them in
67 // any order, and it may be more extensible to future cache hints:
68 // {/quote}
69 //
70 // TODO @keryell
71 // {quote}
72 // An approach a la https ://github.com/chriskohlhoff/propria from
73 // @chriskohlhoff would be to add a property to the pointer, such as
74 //
75 // auto result = flat_load(p, offsets);
76 // auto result = flat_load(decorate<CacheHint::Foo, CacheHint::Bar>(p),
77 // offsets);
78 // The advantage is that you do not have to change all tour API and all the uses
79 // of this decorated pointer will benefit from this. decorate is to be bikeshed
80 // accordingly.

85 template <typename T, int n, int ElemsPerAddr = 1,
86  CacheHint L1H = CacheHint::None, CacheHint L3H = CacheHint::None>
87 ESIMD_INLINE ESIMD_NODEBUG typename sycl::detail::enable_if_t<
88  ((n == 8 || n == 16 || n == 32) &&
89  (ElemsPerAddr == 1 || ElemsPerAddr == 2 || ElemsPerAddr == 4)),
90  simd<T, n * ElemsPerAddr>>
91 gather(T *p, simd<uint32_t, n> offsets, simd<uint16_t, n> pred = 1) {
92 
93  simd<uint64_t, n> offsets_i = convert<uint64_t>(offsets);
94  simd<uint64_t, n> addrs(reinterpret_cast<uint64_t>(p));
95  addrs = addrs + offsets_i;
96 
97  if constexpr (sizeof(T) == 1 && ElemsPerAddr == 2) {
98  auto Ret = __esimd_flat_read<T, n, ElemsPerAddrEncoding<4>(), L1H, L3H>(
99  addrs.data(), ElemsPerAddrEncoding<ElemsPerAddr>(), pred.data());
100  return __esimd_rdregion<T, n * 4, n * ElemsPerAddr, /*VS*/ 4, 2, 1>(Ret, 0);
101  } else if constexpr (sizeof(T) == 1 && ElemsPerAddr == 1) {
102  auto Ret = __esimd_flat_read<T, n, ElemsPerAddrEncoding<4>(), L1H, L3H>(
103  addrs.data(), ElemsPerAddrEncoding<ElemsPerAddr>(), pred.data());
104  return __esimd_rdregion<T, n * 4, n * ElemsPerAddr, /*VS*/ 0, n, 4>(Ret, 0);
105  } else if constexpr (sizeof(T) == 2 && ElemsPerAddr == 1) {
106  auto Ret = __esimd_flat_read<T, n, ElemsPerAddrEncoding<2>(), L1H, L3H>(
107  addrs.data(), ElemsPerAddrEncoding<2>(), pred.data());
108  return __esimd_rdregion<T, n * 2, n, /*VS*/ 0, n, 2>(Ret, 0);
109  } else if constexpr (sizeof(T) == 2)
110  return __esimd_flat_read<T, n, ElemsPerAddrEncoding<ElemsPerAddr>(), L1H,
111  L3H>(
112  addrs.data(), ElemsPerAddrEncoding<2 * ElemsPerAddr>(), pred.data());
113  else
114  return __esimd_flat_read<T, n, ElemsPerAddrEncoding<ElemsPerAddr>(), L1H,
115  L3H>(
116  addrs.data(), ElemsPerAddrEncoding<ElemsPerAddr>(), pred.data());
117 }

127 template <typename T, int n, int ElemsPerAddr = 1,
128  CacheHint L1H = CacheHint::None, CacheHint L3H = CacheHint::None>
129 ESIMD_INLINE ESIMD_NODEBUG typename sycl::detail::enable_if_t<
130  ((n == 8 || n == 16 || n == 32) &&
131  (ElemsPerAddr == 1 || ElemsPerAddr == 2 || ElemsPerAddr == 4)),
132  void>
133 scatter(T *p, simd<T, n * ElemsPerAddr> vals, simd<uint32_t, n> offsets,
134  simd<uint16_t, n> pred = 1) {
135  simd<uint64_t, n> offsets_i = convert<uint64_t>(offsets);
136  simd<uint64_t, n> addrs(reinterpret_cast<uint64_t>(p));
137  addrs = addrs + offsets_i;
138  if constexpr (sizeof(T) == 1 && ElemsPerAddr == 2) {
139  simd<T, n * 4> D;
140  D = __esimd_wrregion<T, n * 4, n * ElemsPerAddr, /*VS*/ 4, 2, 1>(
141  D.data(), vals.data(), 0);
142  __esimd_flat_write<T, n, ElemsPerAddrEncoding<4>(), L1H, L3H>(
143  addrs.data(), D.data(), ElemsPerAddrEncoding<ElemsPerAddr>(),
144  pred.data());
145  } else if constexpr (sizeof(T) == 1 && ElemsPerAddr == 1) {
146  simd<T, n * 4> D;
147  D = __esimd_wrregion<T, n * 4, n * ElemsPerAddr, /*VS*/ 0, n, 4>(
148  D.data(), vals.data(), 0);
149  __esimd_flat_write<T, n, ElemsPerAddrEncoding<4>(), L1H, L3H>(
150  addrs.data(), D.data(), ElemsPerAddrEncoding<ElemsPerAddr>(),
151  pred.data());
152  } else if constexpr (sizeof(T) == 2 && ElemsPerAddr == 1) {
153  simd<T, n * 2> D;
154  D = __esimd_wrregion<T, n * 2, n, /*VS*/ 0, n, 2>(D.data(), vals.data(), 0);
155  __esimd_flat_write<T, n, ElemsPerAddrEncoding<2>(), L1H, L3H>(
156  addrs.data(), D.data(), ElemsPerAddrEncoding<2>(), pred.data());
157  } else if constexpr (sizeof(T) == 2)
158  __esimd_flat_write<T, n, ElemsPerAddrEncoding<ElemsPerAddr>(), L1H, L3H>(
159  addrs.data(), vals.data(), ElemsPerAddrEncoding<2 * ElemsPerAddr>(),
160  pred.data());
161  else
162  __esimd_flat_write<T, n, ElemsPerAddrEncoding<ElemsPerAddr>(), L1H, L3H>(
163  addrs.data(), vals.data(), ElemsPerAddrEncoding<ElemsPerAddr>(),
164  pred.data());
165 }

173 template <typename T, int n, CacheHint L1H = CacheHint::None,
174  CacheHint L3H = CacheHint::None>
175 ESIMD_INLINE ESIMD_NODEBUG simd<T, n> block_load(const T *const addr) {
176  constexpr unsigned Sz = sizeof(T) * n;
177  static_assert(Sz >= __esimd::OWORD, "block size must be at least 1 oword");
178  static_assert(Sz % __esimd::OWORD == 0,
179  "block size must be whole number of owords");
180  static_assert(__esimd::isPowerOf2(Sz / __esimd::OWORD),
181  "block must be 1, 2, 4 or 8 owords long");
182  static_assert(Sz <= 8 * __esimd::OWORD,
183  "block size must be at most 8 owords");
184 
185  uintptr_t Addr = reinterpret_cast<uintptr_t>(addr);
186  return __esimd_flat_block_read_unaligned<T, n, L1H, L3H>(Addr);
187 }

267 template <typename T, int N, typename AccessorTy,
268  CacheHint L1H = CacheHint::None, CacheHint L3H = CacheHint::None>
269 ESIMD_INLINE ESIMD_NODEBUG
270  typename sycl::detail::enable_if_t<(sizeof(T) <= 4) &&
271  (N == 1 || N == 8 || N == 16) &&
272  !std::is_pointer<AccessorTy>::value,
273  simd<T, N>>
274  gather(AccessorTy acc, simd<uint32_t, N> offsets,
275  uint32_t glob_offset = 0) {
276 
277  constexpr int TypeSizeLog2 =
278  sycl::INTEL::gpu::ElemsPerAddrEncoding<sizeof(T)>();
279  // TODO (performance) use hardware-supported scale once BE supports it
280  constexpr uint32_t scale = 0;
281  constexpr uint32_t t_scale = sizeof(T);
282  if constexpr (t_scale > 1) {
283  glob_offset *= t_scale;
284  offsets *= t_scale;
285  }
286 
287  if constexpr (sizeof(T) < 4) {
288  static_assert(std::is_integral<T>::value,
289  "only integral 1- & 2-byte types are supported");
290  using PromoT =
291  typename sycl::detail::conditional_t<std::is_signed<T>::value, int32_t,
292  uint32_t>;
293 #if defined(__SYCL_DEVICE_ONLY__)
294  const auto surf_ind = AccessorPrivateProxy::getNativeImageObj(acc);
295  const simd<PromoT, N> promo_vals =
296  __esimd_surf_read<PromoT, N, decltype(surf_ind), TypeSizeLog2, L1H,
297  L3H>(scale, surf_ind, glob_offset, offsets);
298 #else
299  const simd<PromoT, N> promo_vals =
300  __esimd_surf_read<PromoT, N, AccessorTy, TypeSizeLog2, L1H, L3H>(
301  scale, acc, glob_offset, offsets);
302 #endif
303  return sycl::INTEL::gpu::convert<T>(promo_vals);
304  } else {
305 #if defined(__SYCL_DEVICE_ONLY__)
306  const auto surf_ind = AccessorPrivateProxy::getNativeImageObj(acc);
307  return __esimd_surf_read<T, N, decltype(surf_ind), TypeSizeLog2, L1H, L3H>(
308  scale, surf_ind, glob_offset, offsets);
309 #else
310  return __esimd_surf_read<T, N, AccessorTy, TypeSizeLog2, L1H, L3H>(
311  scale, acc, glob_offset, offsets);
312 #endif
313  }
314 }

438 template <EsimdAtomicOpType Op, typename T, int N, unsigned NumSrc>
439 constexpr bool check_atomic() {
440  if constexpr (!__esimd::isPowerOf2(N, 32)) {
441  static_assert((__esimd::isPowerOf2(N, 32)),
442  "Execution size 1, 2, 4, 8, 16, 32 are supported");
443  return false;
444  }
445 
446  // No source operand.
447  if constexpr (Op == EsimdAtomicOpType::ATOMIC_INC ||
448  Op == EsimdAtomicOpType::ATOMIC_DEC) {
449  if constexpr (NumSrc != 0) {
450  static_assert(NumSrc == 0, "No source operands are expected");
451  return false;
452  }
453  if constexpr (!is_type<T, uint16_t, uint32_t, uint64_t>()) {
454  static_assert((is_type<T, uint16_t, uint32_t, uint64_t>()),
455  "Type UW, UD or UQ is expected");
456  return false;
457  }
458  return true;
459  }
460 
461  // One source integer operand.
462  if constexpr (Op == EsimdAtomicOpType::ATOMIC_ADD ||
463  Op == EsimdAtomicOpType::ATOMIC_SUB ||
464  Op == EsimdAtomicOpType::ATOMIC_MIN ||
465  Op == EsimdAtomicOpType::ATOMIC_MAX ||
466  Op == EsimdAtomicOpType::ATOMIC_XCHG ||
467  Op == EsimdAtomicOpType::ATOMIC_AND ||
468  Op == EsimdAtomicOpType::ATOMIC_OR ||
469  Op == EsimdAtomicOpType::ATOMIC_XOR ||
470  Op == EsimdAtomicOpType::ATOMIC_MINSINT ||
471  Op == EsimdAtomicOpType::ATOMIC_MAXSINT) {
472  if constexpr (NumSrc != 1) {
473  static_assert(NumSrc == 1, "One source operand is expected");
474  return false;
475  }
476  if constexpr ((Op != EsimdAtomicOpType::ATOMIC_MINSINT &&
477  Op != EsimdAtomicOpType::ATOMIC_MAXSINT) &&
478  !is_type<T, uint16_t, uint32_t, uint64_t>()) {
479  static_assert((is_type<T, uint16_t, uint32_t, uint64_t>()),
480  "Type UW, UD or UQ is expected");
481  return false;
482  }
483  if constexpr ((Op == EsimdAtomicOpType::ATOMIC_MINSINT ||
484  Op == EsimdAtomicOpType::ATOMIC_MAXSINT) &&
485  !is_type<T, int16_t, int32_t, int64_t>()) {
486  static_assert((is_type<T, int16_t, int32_t, int64_t>()),
487  "Type W, D or Q is expected");
488  return false;
489  }
490  return true;
491  }
492 
493  // One source float operand.
494  if constexpr (Op == EsimdAtomicOpType::ATOMIC_FMAX ||
495  Op == EsimdAtomicOpType::ATOMIC_FMIN) {
496  if constexpr (NumSrc != 1) {
497  static_assert(NumSrc == 1, "One source operand is expected");
498  return false;
499  }
500  if constexpr (!is_type<T, float, cl::sycl::detail::half_impl::StorageT>()) {
501  static_assert(
502  (is_type<T, float, cl::sycl::detail::half_impl::StorageT>()),
503  "Type F or HF is expected");
504  return false;
505  }
506  return true;
507  }
508 
509  // Two scouce operands.
510  if constexpr (Op == EsimdAtomicOpType::ATOMIC_CMPXCHG ||
511  Op == EsimdAtomicOpType::ATOMIC_FCMPWR) {
512  if constexpr (NumSrc != 2) {
513  static_assert(NumSrc == 2, "Two source operands are expected");
514  return false;
515  }
516  if constexpr (Op == EsimdAtomicOpType::ATOMIC_CMPXCHG &&
517  !is_type<T, uint16_t, uint32_t, uint64_t>()) {
518  static_assert((is_type<T, uint16_t, uint32_t, uint64_t>()),
519  "Type UW, UD or UQ is expected");
520  return false;
521  }
522  if constexpr (Op == EsimdAtomicOpType::ATOMIC_FCMPWR &&
523  !is_type<T, float, cl::sycl::detail::half_impl::StorageT>()) {
524  static_assert(
525  (is_type<T, float, cl::sycl::detail::half_impl::StorageT>()),
526  "Type F or HF is expected");
527  return false;
528  }
529  return true;
530  }
531  // Unsupported svm atomic Op.
532  return false;
