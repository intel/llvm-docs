69 // Aligns the pointer down to the specified alignment
70 // (e.g. returns 8 for Size = 13, Alignment = 8)
71 static void *AlignPtrDown(void *Ptr, const size_t Alignment) {
72  return reinterpret_cast<void *>((reinterpret_cast<size_t>(Ptr)) &
73  (~(Alignment - 1)));
74 }
75 
76 // Aligns the pointer up to the specified alignment
77 // (e.g. returns 16 for Size = 13, Alignment = 8)
78 static void *AlignPtrUp(void *Ptr, const size_t Alignment) {
79  void *AlignedPtr = AlignPtrDown(Ptr, Alignment);
80  // Special case when the pointer is already aligned
81  if (Ptr == AlignedPtr) {
82  return Ptr;
83  }
84  return static_cast<char *>(AlignedPtr) + Alignment;
85 }
86 
87 // Aligns the value up to the specified alignment
88 // (e.g. returns 16 for Size = 13, Alignment = 8)
89 static size_t AlignUp(size_t Val, size_t Alignment) {
90  assert(Alignment > 0);
91  return (Val + Alignment - 1) & (~(Alignment - 1));
92 }

261  // Use the first free chunk index as a hint for the search.
262  auto It = std::find_if(Chunks.begin() + FirstFreeChunkIdx, Chunks.end(),

271 void *Slab::getFreeChunk() {
272  assert(NumAllocated != Chunks.size());
273 
274  const size_t ChunkIdx = FindFirstAvailableChunkIdx();
275  // Free chunk must exist, otherwise we would have allocated another slab
276  assert(ChunkIdx != (static_cast<size_t>(-1)));
277 
278  void *const FreeChunk =
279  (static_cast<uint8_t *>(getPtr())) + ChunkIdx * getChunkSize();
280  Chunks[ChunkIdx] = true;
281  NumAllocated += 1;
282 
283  // Use the found index as the next hint
284  FirstFreeChunkIdx = ChunkIdx;
285 
286  return FreeChunk;
287 }

302 void Slab::unregSlabByAddr(void *Addr, Slab &Slab) {
303  auto &Lock = Slab.getBucket().getUsmAllocCtx().getKnownSlabsMapLock();
304  auto &Map = Slab.getBucket().getUsmAllocCtx().getKnownSlabs();
305 
306  std::lock_guard<std::shared_timed_mutex> Lg(Lock);
307 
308  auto Slabs = Map.equal_range(Addr);
309  // At least the must get the current slab from the map.
310  assert(Slabs.first != Slabs.second && "Slab is not found");
311 
312  for (auto It = Slabs.first; It != Slabs.second; ++It) {
313  if (It->second == Slab) {
314  Map.erase(It);
315  return;
316  }
317  }
318 
319  assert(false && "Slab is not found");
320 }

338 void Slab::freeChunk(void *Ptr) {
339  // This method should be called through bucket(since we might remove the slab
340  // as a result), therefore all locks are done on that level.
341 
342  // Make sure that we're in the right slab
343  assert(Ptr >= getPtr() && Ptr < getEnd());
344 
345  // Even if the pointer p was previously aligned, it's still inside the
346  // corresponding chunk, so we get the correct index here.
347  auto ChunkIdx =
348  (static_cast<char *>(Ptr) - static_cast<char *>(MemPtr)) / getChunkSize();
349 
350  // Make sure that the chunk was allocated
351  assert(Chunks[ChunkIdx] && "double free detected");
352 
353  Chunks[ChunkIdx] = false;
354  NumAllocated -= 1;
355 
356  if (ChunkIdx < FirstFreeChunkIdx)
357  FirstFreeChunkIdx = ChunkIdx;
358 }

397 // The lock must be acquired before calling this method
398 void Bucket::onFreeChunk(Slab &Slab) {
399  // In case if the slab was previously full and now has 1 available
400  // chunk, it should be moved to the list of available slabs
401  if (Slab.getNumAllocated() == (Slab.getNumChunks() - 1)) {
402  auto SlabIter = Slab.getIterator();
403  assert(SlabIter != UnavailableSlabs.end());
404 
405  auto It =
406  AvailableSlabs.insert(AvailableSlabs.begin(), std::move(*SlabIter));
407  UnavailableSlabs.erase(SlabIter);
408 
409  (*It)->setIterator(It);
410  }
411 
412  // Remove the slab when all the chunks from it are deallocated
413  // Note: since the slab is stored as unique_ptr, just remove it from
414  // the list to remove the list to destroy the object
415  if (Slab.getNumAllocated() == 0) {
416  auto It = Slab.getIterator();
417  assert(It != AvailableSlabs.end());
418 
419  AvailableSlabs.erase(It);
420  }
421 }

435 void *USMAllocContext::USMAllocImpl::allocate(size_t Size, size_t Alignment) {
436  if (Size == 0)
437  return nullptr;
438 
439  if (Alignment <= 1)
440  return allocate(Size);
441 
442  size_t AlignedSize = (Size > 1) ? AlignUp(Size, Alignment) : Alignment;
443 
444  // Check if our largest chunk is able to fit aligned size.
445  // If not, just request aligned pointer from the system.
446  if (AlignedSize > settings::CutOff) {
447  return getMemHandle().allocate(Size, Alignment);
448  }
449 
450  auto *Ptr = findBucket(AlignedSize).getChunk();
451  return AlignPtrUp(Ptr, Alignment);
452 }

466 void USMAllocContext::USMAllocImpl::deallocate(void *Ptr) {
467  auto *SlabPtr = AlignPtrDown(Ptr, settings::SlabSize);
468 
469  // Lock the map on read
470  std::shared_lock<std::shared_timed_mutex> Lk(getKnownSlabsMapLock());
471 
472  auto Slabs = getKnownSlabs().equal_range(SlabPtr);
473  if (Slabs.first == Slabs.second) {
474  Lk.unlock();
475  getMemHandle().deallocate(Ptr);
476  return;
477  }
478 
479  for (auto It = Slabs.first; It != Slabs.second; ++It) {
480  // The slab object won't be deleted until it's removed from the map which is
481  // protected by the lock, so it's safe to access it here.
482  auto &Slab = It->second;
483  if (Ptr >= Slab.getPtr() && Ptr < Slab.getEnd()) {
484 
485  // Unlock the map before freeing the chunk, it may be locked on write
486  // there
487  Lk.unlock();
488  auto &Bucket = Slab.getBucket();
489  Bucket.freeChunk(Ptr, Slab);
490  return;
491  }
492  }
493 
494  Lk.unlock();
495  // There is a rare case when we have a pointer from system allocation next
496  // to some slab with an entry in the map. So we find a slab
497  // but the range checks fail.
498  getMemHandle().deallocate(Ptr);
499 }
