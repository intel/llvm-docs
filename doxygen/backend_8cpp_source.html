32 static const PluginPtr &getPlugin(backend Backend) {
33  switch (Backend) {
34  case backend::opencl:
35  return ur::getPlugin<backend::opencl>();
36  case backend::ext_oneapi_level_zero:
37  return ur::getPlugin<backend::ext_oneapi_level_zero>();
38  case backend::ext_oneapi_cuda:
39  return ur::getPlugin<backend::ext_oneapi_cuda>();

43  throw sycl::exception(
44  sycl::make_error_code(sycl::errc::runtime),
45  "getPlugin: Unsupported backend " +
46  detail::codeToString(UR_RESULT_ERROR_INVALID_OPERATION));
47  }

94 __SYCL_EXPORT context make_context(ur_native_handle_t NativeHandle,
95  const async_handler &Handler,
96  backend Backend, bool KeepOwnership,
97  const std::vector<device> &DeviceList) {
98  const auto &Plugin = getPlugin(Backend);
99 
100  ur_context_handle_t UrContext = nullptr;
101  ur_context_native_properties_t Properties{};
102  Properties.stype = UR_STRUCTURE_TYPE_CONTEXT_NATIVE_PROPERTIES;
103  Properties.isNativeHandleOwned = false;
104  std::vector<ur_device_handle_t> DeviceHandles;
105  for (const auto &Dev : DeviceList) {
106  DeviceHandles.push_back(detail::getSyclObjImpl(Dev)->getHandleRef());
107  }
108  Plugin->call(urContextCreateWithNativeHandle, NativeHandle,
109  Plugin->getUrAdapter(), DeviceHandles.size(),
110  DeviceHandles.data(), &Properties, &UrContext);
111  // Construct the SYCL context from UR context.
112  return detail::createSyclObjFromImpl<context>(std::make_shared<context_impl>(
113  UrContext, Handler, Plugin, DeviceList, !KeepOwnership));
114 }

116 __SYCL_EXPORT queue make_queue(ur_native_handle_t NativeHandle,
117  int32_t NativeHandleDesc, const context &Context,
118  const device *Device, bool KeepOwnership,
119  const property_list &PropList,
120  const async_handler &Handler, backend Backend) {
121  ur_device_handle_t UrDevice =
122  Device ? getSyclObjImpl(*Device)->getHandleRef() : nullptr;
123  const auto &Plugin = getPlugin(Backend);
124  const auto &ContextImpl = getSyclObjImpl(Context);
125 
126  if (PropList.has_property<ext::intel::property::queue::compute_index>()) {
127  throw sycl::exception(
128  make_error_code(errc::invalid),
129  "Queue create using make_queue cannot have compute_index property.");
130  }
131 
132  ur_queue_native_desc_t Desc{};
133  Desc.stype = UR_STRUCTURE_TYPE_QUEUE_NATIVE_DESC;
134  Desc.pNativeData = &NativeHandleDesc;
135 
136  ur_queue_properties_t Properties{};
137  Properties.stype = UR_STRUCTURE_TYPE_QUEUE_PROPERTIES;
138  Properties.flags = queue_impl::createUrQueueFlags(
139  PropList, PropList.has_property<property::queue::in_order>()
140  ? QueueOrder::Ordered
141  : QueueOrder::OOO);
142 
143  ur_queue_native_properties_t NativeProperties{};
144  NativeProperties.stype = UR_STRUCTURE_TYPE_QUEUE_NATIVE_PROPERTIES;
145  NativeProperties.isNativeHandleOwned = !KeepOwnership;
146 
147  Properties.pNext = &Desc;
148  NativeProperties.pNext = &Properties;
149 
150  // Create UR queue first.
151  ur_queue_handle_t UrQueue = nullptr;
152 
153  Plugin->call(urQueueCreateWithNativeHandle, NativeHandle,
154  ContextImpl->getHandleRef(), UrDevice, &NativeProperties,
155  &UrQueue);
156  // Construct the SYCL queue from UR queue.
157  return detail::createSyclObjFromImpl<queue>(
158  std::make_shared<queue_impl>(UrQueue, ContextImpl, Handler, PropList));
159 }

187 std::shared_ptr<detail::kernel_bundle_impl>
188 make_kernel_bundle(ur_native_handle_t NativeHandle,
189  const context &TargetContext, bool KeepOwnership,
190  bundle_state State, backend Backend) {
191  const auto &Plugin = getPlugin(Backend);
192  const auto &ContextImpl = getSyclObjImpl(TargetContext);
193 
194  ur_program_handle_t UrProgram = nullptr;
195  ur_program_native_properties_t Properties{};
196  Properties.stype = UR_STRUCTURE_TYPE_PROGRAM_NATIVE_PROPERTIES;
197  Properties.isNativeHandleOwned = !KeepOwnership;
198 
199  Plugin->call(urProgramCreateWithNativeHandle, NativeHandle,
200  ContextImpl->getHandleRef(), &Properties, &UrProgram);
201  if (ContextImpl->getBackend() == backend::opencl)
202  Plugin->call(urProgramRetain, UrProgram);
203 
204  std::vector<ur_device_handle_t> ProgramDevices;
205  uint32_t NumDevices = 0;
206 
207  Plugin->call(urProgramGetInfo, UrProgram, UR_PROGRAM_INFO_NUM_DEVICES,
208  sizeof(NumDevices), &NumDevices, nullptr);
209  ProgramDevices.resize(NumDevices);
210  Plugin->call(urProgramGetInfo, UrProgram, UR_PROGRAM_INFO_DEVICES,
211  sizeof(ur_device_handle_t) * NumDevices, ProgramDevices.data(),
212  nullptr);
213 
214  for (auto &Dev : ProgramDevices) {
215  ur_program_binary_type_t BinaryType;
216  Plugin->call(urProgramGetBuildInfo, UrProgram, Dev,
217  UR_PROGRAM_BUILD_INFO_BINARY_TYPE,
218  sizeof(ur_program_binary_type_t), &BinaryType, nullptr);
219  switch (BinaryType) {
220  case (UR_PROGRAM_BINARY_TYPE_NONE):
221  if (State == bundle_state::object) {
222  auto Res = Plugin->call_nocheck(urProgramCompileExp, UrProgram, 1, &Dev,
223  nullptr);
224  if (Res == UR_RESULT_ERROR_UNSUPPORTED_FEATURE) {
225  Res = Plugin->call_nocheck(urProgramCompile,
226  ContextImpl->getHandleRef(), UrProgram,
227  nullptr);
228  }
229  Plugin->checkUrResult<errc::build>(Res);
230  }
231 
232  else if (State == bundle_state::executable) {
233  auto Res = Plugin->call_nocheck(urProgramBuildExp, UrProgram, 1, &Dev,
234  nullptr);
235  if (Res == UR_RESULT_ERROR_UNSUPPORTED_FEATURE) {
236  Res = Plugin->call_nocheck(
237  urProgramBuild, ContextImpl->getHandleRef(), UrProgram, nullptr);
238  }
239  Plugin->checkUrResult<errc::build>(Res);
240  }
241 
242  break;
243  case (UR_PROGRAM_BINARY_TYPE_COMPILED_OBJECT):
244  case (UR_PROGRAM_BINARY_TYPE_LIBRARY):
245  if (State == bundle_state::input)
246  throw sycl::exception(
247  sycl::make_error_code(sycl::errc::runtime),
248  "Program and kernel_bundle state mismatch " +
249  detail::codeToString(UR_RESULT_ERROR_INVALID_VALUE));
250  if (State == bundle_state::executable) {
251  ur_program_handle_t UrLinkedProgram = nullptr;
252  auto Res =
253  Plugin->call_nocheck(urProgramLinkExp, ContextImpl->getHandleRef(),
254  1, &Dev, 1, &UrProgram, nullptr, &UrLinkedProgram);
255  if (Res == UR_RESULT_ERROR_UNSUPPORTED_FEATURE) {
256  Res = Plugin->call_nocheck(urProgramLink, ContextImpl->getHandleRef(),
257  1, &UrProgram, nullptr, &UrLinkedProgram);
258  }
259  Plugin->checkUrResult<errc::build>(Res);
260  if (UrLinkedProgram != nullptr) {
261  UrProgram = UrLinkedProgram;
262  }
263  }
264  break;
265  case (UR_PROGRAM_BINARY_TYPE_EXECUTABLE):
266  if (State == bundle_state::input || State == bundle_state::object)
267  throw sycl::exception(
268  sycl::make_error_code(sycl::errc::runtime),
269  "Program and kernel_bundle state mismatch " +
270  detail::codeToString(UR_RESULT_ERROR_INVALID_VALUE));
271  break;
272  default:
273  break;
274  }
275  }
276 
277  std::vector<device> Devices;
278  Devices.reserve(ProgramDevices.size());
279  std::transform(
280  ProgramDevices.begin(), ProgramDevices.end(), std::back_inserter(Devices),
281  [&Plugin](const auto &Dev) {
282  auto Platform =
283  detail::platform_impl::getPlatformFromUrDevice(Dev, Plugin);
284  auto DeviceImpl = Platform->getOrMakeDeviceImpl(Dev, Platform);
285  return createSyclObjFromImpl<device>(DeviceImpl);
286  });
287 
288  // Unlike SYCL, other backends, like OpenCL or Level Zero, may not support
289  // getting kernel IDs before executable is built. The SYCL Runtime workarounds
290  // this by pre-building the device image and extracting kernel info. We can't
291  // do the same to user images, since they may contain references to undefined
292  // symbols (e.g. when kernel_bundle is supposed to be joined with another).
293  auto KernelIDs = std::make_shared<std::vector<kernel_id>>();
294  auto DevImgImpl = std::make_shared<device_image_impl>(
295  nullptr, TargetContext, Devices, State, KernelIDs, UrProgram);
296  device_image_plain DevImg{DevImgImpl};
297 
298  return std::make_shared<kernel_bundle_impl>(TargetContext, Devices, DevImg);
299 }

309 kernel make_kernel(const context &TargetContext,
310  const kernel_bundle<bundle_state::executable> &KernelBundle,
311  ur_native_handle_t NativeHandle, bool KeepOwnership,
312  backend Backend) {
313  const auto &Plugin = getPlugin(Backend);
314  const auto &ContextImpl = getSyclObjImpl(TargetContext);
315  const auto KernelBundleImpl = getSyclObjImpl(KernelBundle);
316 
317  // For Level-Zero expect exactly one device image in the bundle. This is
318  // natural for interop kernel to get created out of a single native
319  // program/module. This way we don't need to search the exact device image for
320  // the kernel, which may not be trivial.
321  //
322  // Other backends don't need UR program.
323  //
324  ur_program_handle_t UrProgram = nullptr;
325  if (Backend == backend::ext_oneapi_level_zero) {
326  if (KernelBundleImpl->size() != 1)
327  throw sycl::exception(
328  sycl::make_error_code(sycl::errc::runtime),
329  "make_kernel: kernel_bundle must have single program image " +
330  detail::codeToString(UR_RESULT_ERROR_INVALID_PROGRAM));
331 
332  const device_image<bundle_state::executable> &DeviceImage =
333  *KernelBundle.begin();
334  const auto &DeviceImageImpl = getSyclObjImpl(DeviceImage);
335  UrProgram = DeviceImageImpl->get_ur_program_ref();
336  }
337 
338  // Create UR kernel first.
339  ur_kernel_handle_t UrKernel = nullptr;
340  ur_kernel_native_properties_t Properties{};
341  Properties.stype = UR_STRUCTURE_TYPE_KERNEL_NATIVE_PROPERTIES;
342  Properties.isNativeHandleOwned = !KeepOwnership;
343  Plugin->call(urKernelCreateWithNativeHandle, NativeHandle,
344  ContextImpl->getHandleRef(), UrProgram, &Properties, &UrKernel);
345 
346  if (Backend == backend::opencl)
347  Plugin->call(urKernelRetain, UrKernel);
348 
349  // Construct the SYCL queue from UR queue.
350  return detail::createSyclObjFromImpl<kernel>(
351  std::make_shared<kernel_impl>(UrKernel, ContextImpl, KernelBundleImpl));
352 }
