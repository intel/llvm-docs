106 std::shared_ptr<detail::kernel_bundle_impl>
107 make_kernel_bundle(pi_native_handle NativeHandle, const context &TargetContext,
108  bundle_state State, backend Backend) {
109  const auto &Plugin = getPlugin(Backend);
110  const auto &ContextImpl = getSyclObjImpl(TargetContext);
111 
112  pi::PiProgram PiProgram = nullptr;
113  Plugin.call<PiApiKind::piextProgramCreateWithNativeHandle>(
114  NativeHandle, ContextImpl->getHandleRef(), &PiProgram);
115 
116  std::vector<pi::PiDevice> ProgramDevices;
117  size_t NumDevices = 0;
118 
119  Plugin.call<PiApiKind::piProgramGetInfo>(
120  PiProgram, PI_PROGRAM_INFO_NUM_DEVICES, sizeof(size_t), &NumDevices,
121  nullptr);
122  ProgramDevices.resize(NumDevices);
123  Plugin.call<PiApiKind::piProgramGetInfo>(PiProgram, PI_PROGRAM_INFO_DEVICES,
124  sizeof(pi::PiDevice) * NumDevices,
125  ProgramDevices.data(), nullptr);
126 
127  for (const auto &Dev : ProgramDevices) {
128  size_t BinaryType = 0;
129  Plugin.call<PiApiKind::piProgramGetBuildInfo>(
130  PiProgram, Dev, PI_PROGRAM_BUILD_INFO_BINARY_TYPE, sizeof(size_t),
131  &BinaryType, nullptr);
132  switch (BinaryType) {
133  case (PI_PROGRAM_BINARY_TYPE_NONE):
134  if (State == bundle_state::object)
135  Plugin.call<PiApiKind::piProgramCompile>(
136  PiProgram, 1, &Dev, nullptr, 0, nullptr, nullptr, nullptr, nullptr);
137  else if (State == bundle_state::executable)
138  Plugin.call<PiApiKind::piProgramBuild>(PiProgram, 1, &Dev, nullptr,
139  nullptr, nullptr);
140  break;
141  case (PI_PROGRAM_BINARY_TYPE_COMPILED_OBJECT):
142  case (PI_PROGRAM_BINARY_TYPE_LIBRARY):
143  if (State == bundle_state::input)
144  // TODO SYCL2020 exception
145  throw sycl::runtime_error("Program and kernel_bundle state mismatch",
146  PI_INVALID_VALUE);
147  if (State == bundle_state::executable)
148  Plugin.call<PiApiKind::piProgramLink>(ContextImpl->getHandleRef(), 1,
149  &Dev, nullptr, 1, &PiProgram,
150  nullptr, nullptr, &PiProgram);
151  break;
152  case (PI_PROGRAM_BINARY_TYPE_EXECUTABLE):
153  if (State == bundle_state::input || State == bundle_state::object)
154  // TODO SYCL2020 exception
155  throw sycl::runtime_error("Program and kernel_bundle state mismatch",
156  PI_INVALID_VALUE);
157  break;
158  }
159  }
160 
161  std::vector<device> Devices;
162  Devices.reserve(ProgramDevices.size());
163  std::transform(
164  ProgramDevices.begin(), ProgramDevices.end(), std::back_inserter(Devices),
165  [&Plugin](const auto &Dev) {
166  auto Platform =
167  detail::platform_impl::getPlatformFromPiDevice(Dev, Plugin);
168  auto DeviceImpl = Platform->getOrMakeDeviceImpl(Dev, Platform);
169  return createSyclObjFromImpl<device>(DeviceImpl);
170  });
171 
172  // Unlike SYCL, other backends, like OpenCL or Level Zero, may not support
173  // getting kernel IDs before executable is built. The SYCL Runtime workarounds
174  // this by pre-building the device image and extracting kernel info. We can't
175  // do the same to user images, since they may contain references to undefined
176  // symbols (e.g. when kernel_bundle is supposed to be joined with another).
177  std::vector<kernel_id> KernelIDs{};
178  auto DevImgImpl = std::make_shared<device_image_impl>(
179  nullptr, TargetContext, Devices, State, KernelIDs, PiProgram);
180  device_image_plain DevImg{DevImgImpl};
181 
182  return std::make_shared<kernel_bundle_impl>(TargetContext, Devices, DevImg);
183 }
184 kernel make_kernel(pi_native_handle NativeHandle, const context &TargetContext,
185  backend Backend) {
186  const auto &Plugin = getPlugin(Backend);
187  const auto &ContextImpl = getSyclObjImpl(TargetContext);
188  // Create PI kernel first.
189  pi::PiKernel PiKernel = nullptr;
190  Plugin.call<PiApiKind::piextKernelCreateWithNativeHandle>(
191  NativeHandle, ContextImpl->getHandleRef(), false, &PiKernel);
192 
193  if (Backend == backend::opencl)
194  Plugin.call<PiApiKind::piKernelRetain>(PiKernel);
195 
196  // Construct the SYCL queue from PI queue.
197  return detail::createSyclObjFromImpl<kernel>(
198  std::make_shared<kernel_impl>(PiKernel, ContextImpl));
199 }
