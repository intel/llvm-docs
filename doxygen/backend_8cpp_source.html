119 std::shared_ptr<detail::kernel_bundle_impl>
120 make_kernel_bundle(pi_native_handle NativeHandle, const context &TargetContext,
121  bool KeepOwnership, bundle_state State, backend Backend) {
122  const auto &Plugin = getPlugin(Backend);
123  const auto &ContextImpl = getSyclObjImpl(TargetContext);
124 
125  pi::PiProgram PiProgram = nullptr;
126  Plugin.call<PiApiKind::piextProgramCreateWithNativeHandle>(
127  NativeHandle, ContextImpl->getHandleRef(), KeepOwnership, &PiProgram);
128 
129  std::vector<pi::PiDevice> ProgramDevices;
130  size_t NumDevices = 0;
131 
132  Plugin.call<PiApiKind::piProgramGetInfo>(
133  PiProgram, PI_PROGRAM_INFO_NUM_DEVICES, sizeof(size_t), &NumDevices,
134  nullptr);
135  ProgramDevices.resize(NumDevices);
136  Plugin.call<PiApiKind::piProgramGetInfo>(PiProgram, PI_PROGRAM_INFO_DEVICES,
137  sizeof(pi::PiDevice) * NumDevices,
138  ProgramDevices.data(), nullptr);
139 
140  for (const auto &Dev : ProgramDevices) {
141  size_t BinaryType = 0;
142  Plugin.call<PiApiKind::piProgramGetBuildInfo>(
143  PiProgram, Dev, PI_PROGRAM_BUILD_INFO_BINARY_TYPE, sizeof(size_t),
144  &BinaryType, nullptr);
145  switch (BinaryType) {
146  case (PI_PROGRAM_BINARY_TYPE_NONE):
147  if (State == bundle_state::object)
148  Plugin.call<errc::build, PiApiKind::piProgramCompile>(
149  PiProgram, 1, &Dev, nullptr, 0, nullptr, nullptr, nullptr, nullptr);
150  else if (State == bundle_state::executable)
151  Plugin.call<errc::build, PiApiKind::piProgramBuild>(
152  PiProgram, 1, &Dev, nullptr, nullptr, nullptr);
153  break;
154  case (PI_PROGRAM_BINARY_TYPE_COMPILED_OBJECT):
155  case (PI_PROGRAM_BINARY_TYPE_LIBRARY):
156  if (State == bundle_state::input)
157  // TODO SYCL2020 exception
158  throw sycl::runtime_error(errc::invalid,
159  "Program and kernel_bundle state mismatch",
160  PI_INVALID_VALUE);
161  if (State == bundle_state::executable)
162  Plugin.call<errc::build, PiApiKind::piProgramLink>(
163  ContextImpl->getHandleRef(), 1, &Dev, nullptr, 1, &PiProgram,
164  nullptr, nullptr, &PiProgram);
165  break;
166  case (PI_PROGRAM_BINARY_TYPE_EXECUTABLE):
167  if (State == bundle_state::input || State == bundle_state::object)
168  // TODO SYCL2020 exception
169  throw sycl::runtime_error(errc::invalid,
170  "Program and kernel_bundle state mismatch",
171  PI_INVALID_VALUE);
172  break;
173  }
174  }
175 
176  std::vector<device> Devices;
177  Devices.reserve(ProgramDevices.size());
178  std::transform(
179  ProgramDevices.begin(), ProgramDevices.end(), std::back_inserter(Devices),
180  [&Plugin](const auto &Dev) {
181  auto Platform =
182  detail::platform_impl::getPlatformFromPiDevice(Dev, Plugin);
183  auto DeviceImpl = Platform->getOrMakeDeviceImpl(Dev, Platform);
184  return createSyclObjFromImpl<device>(DeviceImpl);
185  });
186 
187  // Unlike SYCL, other backends, like OpenCL or Level Zero, may not support
188  // getting kernel IDs before executable is built. The SYCL Runtime workarounds
189  // this by pre-building the device image and extracting kernel info. We can't
190  // do the same to user images, since they may contain references to undefined
191  // symbols (e.g. when kernel_bundle is supposed to be joined with another).
192  auto KernelIDs = std::make_shared<std::vector<kernel_id>>();
193  auto DevImgImpl = std::make_shared<device_image_impl>(
194  nullptr, TargetContext, Devices, State, KernelIDs, PiProgram);
195  device_image_plain DevImg{DevImgImpl};
196 
197  return std::make_shared<kernel_bundle_impl>(TargetContext, Devices, DevImg);
198 }

207 kernel make_kernel(const context &TargetContext,
208  const kernel_bundle<bundle_state::executable> &KernelBundle,
209  pi_native_handle NativeHandle, bool KeepOwnership,
210  backend Backend) {
211  const auto &Plugin = getPlugin(Backend);
212  const auto &ContextImpl = getSyclObjImpl(TargetContext);
213  const auto KernelBundleImpl = getSyclObjImpl(KernelBundle);
214 
215  // For Level-Zero expect exactly one device image in the bundle. This is
216  // natural for interop kernel to get created out of a single native
217  // program/module. This way we don't need to search the exact device image for
218  // the kernel, which may not be trivial.
219  //
220  // Other backends don't need PI program.
221  //
222  pi::PiProgram PiProgram = nullptr;
223  if (Backend == backend::ext_oneapi_level_zero) {
224  if (KernelBundleImpl->size() != 1)
225  throw sycl::runtime_error{
226  "make_kernel: kernel_bundle must have single program image",
227  PI_INVALID_PROGRAM};
228 
229  const device_image<bundle_state::executable> &DeviceImage =
230  *KernelBundle.begin();
231  const auto &DeviceImageImpl = getSyclObjImpl(DeviceImage);
232  PiProgram = DeviceImageImpl->get_program_ref();
233  }
234 
235  // Create PI kernel first.
236  pi::PiKernel PiKernel = nullptr;
237  Plugin.call<PiApiKind::piextKernelCreateWithNativeHandle>(
238  NativeHandle, ContextImpl->getHandleRef(), PiProgram, KeepOwnership,
239  &PiKernel);
240 
241  if (Backend == backend::opencl)
242  Plugin.call<PiApiKind::piKernelRetain>(PiKernel);
243 
244  // Construct the SYCL queue from PI queue.
245  return detail::createSyclObjFromImpl<kernel>(
246  std::make_shared<kernel_impl>(PiKernel, ContextImpl, KernelBundleImpl));
247 }
