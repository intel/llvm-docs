431  bool isContiguousRegion(const id<2> &offset, const range<2> &newRange,
432  const range<2> &parentRange) {
433  // For 2D sub buffer there are 2 cases:
434  // 1) Offset {Any, Any} | a piece of any line of a buffer
435  // Range {1, Any} |
436  // 2) Offset {Any, 0 } | any number of full lines
437  // Range {Any, Col} |
438  // where Col is a number of columns of original buffer
439  if (offset[1])
440  return newRange[0] == 1;
441  return newRange[1] == parentRange[1];
442  }
443 
444  bool isContiguousRegion(const id<3> &offset, const range<3> &newRange,
445  const range<3> &parentRange) {
446  // For 3D sub buffer there are 3 cases:
447  // 1) Offset {Any, Any, Any} | a piece of any line in any slice of a buffer
448  // Range {1, 1, Any} |
449  // 2) Offset {Any, Any, 0 } | any number of full lines in any slice
450  // Range {1, Any, Col} |
451  // 3) Offset {Any, 0, 0 } | any number of slices
452  // Range {Any, Row, Col} |
453  // where Row and Col are numbers of rows and columns of original buffer
454  if (offset[2])
455  return newRange[0] == 1 && newRange[1] == 1;
456  if (offset[1])
457  return newRange[0] == 1 && newRange[2] == parentRange[2];
458  return newRange[1] == parentRange[1] && newRange[2] == parentRange[2];
459  }
