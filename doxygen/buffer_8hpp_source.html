288  buffer(buffer<T, dimensions, AllocatorT> &b, const id<dimensions> &baseIndex,
289  const range<dimensions> &subRange,
290  const detail::code_location CodeLoc = detail::code_location::current())
291  : impl(b.impl), Range(subRange),
292  OffsetInBytes(getOffsetInBytes<T>(baseIndex, b.Range)),
293  IsSubBuffer(true) {
294  impl->constructorNotification(CodeLoc, (void *)impl.get(), impl.get(),
295  (const void *)typeid(T).name(), dimensions,
296  sizeof(T), rangeToArray(Range).data());
297 
298  if (b.is_sub_buffer())
299  throw cl::sycl::invalid_object_error(
300  "Cannot create sub buffer from sub buffer.", PI_INVALID_VALUE);
301  if (isOutOfBounds(baseIndex, subRange, b.Range))
302  throw cl::sycl::invalid_object_error(
303  "Requested sub-buffer size exceeds the size of the parent buffer",
304  PI_INVALID_VALUE);
305  if (!isContiguousRegion(baseIndex, subRange, b.Range))
306  throw cl::sycl::invalid_object_error(
307  "Requested sub-buffer region is not contiguous", PI_INVALID_VALUE);
308  }

487  template <typename ReinterpretT, int ReinterpretDim = dimensions>
488  typename std::enable_if<
489  (ReinterpretDim == 1) && ((dimensions != ReinterpretDim) ||
490  (sizeof(ReinterpretT) != sizeof(T))),
491  buffer<ReinterpretT, ReinterpretDim, AllocatorT>>::type
492  reinterpret() const {
493  long sz = byte_size();
494  if (sz % sizeof(ReinterpretT) != 0)

501  impl, range<1>{sz / sizeof(ReinterpretT)}, OffsetInBytes, IsSubBuffer);

512 protected:
513  bool isOutOfBounds(const id<dimensions> &offset,
514  const range<dimensions> &newRange,
515  const range<dimensions> &parentRange) {
516  bool outOfBounds = false;
517  for (int i = 0; i < dimensions; ++i)
518  outOfBounds |= newRange[i] + offset[i] > parentRange[i];
519 
520  return outOfBounds;
521  }

586  // 1) Offset {Any, Any} | a piece of any line of a buffer

597  const range<3> &parentRange) {
598  // For 3D sub buffer there are 3 cases:
599  // 1) Offset {Any, Any, Any} | a piece of any line in any slice of a buffer
600  // Range {1, 1, Any} |
601  // 2) Offset {Any, Any, 0 } | any number of full lines in any slice
602  // Range {1, Any, Col} |
603  // 3) Offset {Any, 0, 0 } | any number of slices
604  // Range {Any, Row, Col} |
605  // where Row and Col are numbers of rows and columns of original buffer
606  if (offset[2])

608  if (offset[1])
