72 // {quote}
73 // ...I'd like us to think more about what we can do to make these interfaces
74 // more user - friendly. A user providing cache hints has to provide a lot more
75 // template arguments than required.Could we make this nicer by providing the
76 // hints as tag - type arguments ?
77 // ...
78 // // Without cache hints, type and length can be deduced from offsets
79 // float* p;
80 // simd<uint32_t, 16> offsets;
81 // auto result = flat_load(p, offsets);
82 //
83 // // With cache hints as templates, verbosity increases significantly:
84 // // - Providing any cache hint forces the user to specify the type and
85 // length float* p; simd<uint32_t, 16> offsets; auto result =
86 // flat_load<uint32_t, 16, 1, CacheHint::Foo, CacheHint::Bar>(p, offsets);
87 //
88 // // With cache hints as tag types, verbosity is reduced:
89 // // - Providing a cache hint does not prevent deduction of type and length
90 // float* p;
91 // simd <uint32_t, 16> offsets;
92 // auto result = flat_load(p, offsets, CacheHint::Foo{});
93 //
94 // Note also that the templated form prevents a developer from specifying an L3
95 // hint without also explicitly specifying an L1 hint. If flat_load accepted a
96 // list of hints, it might be possible to refactor the hints to specify them in
97 // any order, and it may be more extensible to future cache hints:
98 // {/quote}
99 //
100 // TODO @keryell
101 // {quote}
102 // An approach a la https ://github.com/chriskohlhoff/propria from
103 // @chriskohlhoff would be to add a property to the pointer, such as
104 //
105 // auto result = flat_load(p, offsets);
106 // auto result = flat_load(decorate<CacheHint::Foo, CacheHint::Bar>(p),
107 // offsets);
108 // The advantage is that you do not have to change all tour API and all the uses
109 // of this decorated pointer will benefit from this. decorate is to be bikeshed

129 template <typename Tx, int N, class T = detail::__raw_t<Tx>>
130 __ESIMD_API std::enable_if_t<detail::isPowerOf2(N, 32), simd<Tx, N>>
131 gather(const Tx *p, simd<uint32_t, N> offsets, simd_mask<N> mask = 1) {
132  simd<uint64_t, N> offsets_i = convert<uint64_t>(offsets);
133  simd<uint64_t, N> addrs(reinterpret_cast<uint64_t>(p));
134  addrs = addrs + offsets_i;
135 
136  if constexpr (sizeof(T) == 1) {
137  auto Ret = __esimd_svm_gather<T, N, detail::ElemsPerAddrEncoding<4>()>(
138  addrs.data(), detail::ElemsPerAddrEncoding<1>(), mask.data());
139  return __esimd_rdregion<T, N * 4, N, /*VS*/ 0, N, 4>(Ret, 0);
140  } else if constexpr (sizeof(T) == 2) {
141  auto Ret = __esimd_svm_gather<T, N, detail::ElemsPerAddrEncoding<2>()>(
142  addrs.data(), detail::ElemsPerAddrEncoding<2>(), mask.data());
143  return __esimd_rdregion<T, N * 2, N, /*VS*/ 0, N, 2>(Ret, 0);
144  } else
145  return __esimd_svm_gather<T, N, detail::ElemsPerAddrEncoding<1>()>(
146  addrs.data(), detail::ElemsPerAddrEncoding<1>(), mask.data());
147 }

162 template <typename Tx, int N, class T = detail::__raw_t<Tx>>
163 __ESIMD_API std::enable_if_t<detail::isPowerOf2(N, 32)>
164 scatter(Tx *p, simd<uint32_t, N> offsets, simd<Tx, N> vals,
165  simd_mask<N> mask = 1) {
166  simd<uint64_t, N> offsets_i = convert<uint64_t>(offsets);
167  simd<uint64_t, N> addrs(reinterpret_cast<uint64_t>(p));
168  addrs = addrs + offsets_i;
169  if constexpr (sizeof(T) == 1) {
170  simd<T, N * 4> D;
171  D = __esimd_wrregion<T, N * 4, N, /*VS*/ 0, N, 4>(D.data(), vals.data(), 0);
172  __esimd_svm_scatter<T, N, detail::ElemsPerAddrEncoding<4>()>(
173  addrs.data(), D.data(), detail::ElemsPerAddrEncoding<1>(), mask.data());
174  } else if constexpr (sizeof(T) == 2) {
175  simd<T, N * 2> D;
176  D = __esimd_wrregion<T, N * 2, N, /*VS*/ 0, N, 2>(D.data(), vals.data(), 0);
177  __esimd_svm_scatter<T, N, detail::ElemsPerAddrEncoding<2>()>(
178  addrs.data(), D.data(), detail::ElemsPerAddrEncoding<2>(), mask.data());
179  } else
180  __esimd_svm_scatter<T, N, detail::ElemsPerAddrEncoding<1>()>(
181  addrs.data(), vals.data(), detail::ElemsPerAddrEncoding<1>(),
182  mask.data());
183 }

199  class T = detail::__raw_t<Tx>,
200  typename = std::enable_if_t<is_simd_flag_type_v<Flags>>>
201 __ESIMD_API simd<Tx, N> block_load(const Tx *addr, Flags = {}) {
202  constexpr unsigned Sz = sizeof(T) * N;
203  static_assert(Sz >= detail::OperandSize::OWORD,
204  "block size must be at least 1 oword");
205  static_assert(Sz % detail::OperandSize::OWORD == 0,
206  "block size must be whole number of owords");
207  static_assert(detail::isPowerOf2(Sz / detail::OperandSize::OWORD),
208  "block must be 1, 2, 4 or 8 owords long");
209  static_assert(Sz <= 8 * detail::OperandSize::OWORD,
210  "block size must be at most 8 owords");
211 
212  uintptr_t Addr = reinterpret_cast<uintptr_t>(addr);
213  if constexpr (Flags::template alignment<simd<T, N>> >=
214  detail::OperandSize::OWORD) {
215  return __esimd_svm_block_ld<T, N>(Addr);
216  } else {
217  return __esimd_svm_block_ld_unaligned<T, N>(Addr);
218  }

323 template <typename T, int N, typename AccessorTy>
324 ESIMD_INLINE
325  ESIMD_NODEBUG std::enable_if_t<(sizeof(T) <= 4) &&
326  (N == 1 || N == 8 || N == 16 || N == 32) &&
327  !std::is_pointer<AccessorTy>::value>
328  scatter_impl(AccessorTy acc, simd<T, N> vals, simd<uint32_t, N> offsets,
329  uint32_t glob_offset, simd_mask<N> mask) {
330 
331  constexpr int TypeSizeLog2 = detail::ElemsPerAddrEncoding<sizeof(T)>();
332  // TODO (performance) use hardware-supported scale once BE supports it
333  constexpr int16_t scale = 0;
334  const auto si = __ESIMD_GET_SURF_HANDLE(acc);
335 
336  if constexpr (sizeof(T) < 4) {
337  using Tint = std::conditional_t<std::is_integral_v<T>, T,
338  detail::uint_type_t<sizeof(T)>>;
339  using Treal = __raw_t<T>;
340  simd<Tint, N> vals_int = bitcast<Tint, Treal, N>(std::move(vals).data());
341  using PromoT =
342  typename sycl::detail::conditional_t<std::is_signed<Tint>::value,
343  int32_t, uint32_t>;
344  const simd<PromoT, N> promo_vals = convert<PromoT>(std::move(vals_int));
345  __esimd_scatter_scaled<PromoT, N, decltype(si), TypeSizeLog2, scale>(
346  mask.data(), si, glob_offset, offsets.data(), promo_vals.data());
347  } else {
348  __esimd_scatter_scaled<T, N, decltype(si), TypeSizeLog2, scale>(
349  mask.data(), si, glob_offset, offsets.data(), vals.data());
350  }
351 }

353 template <typename T, int N, typename AccessorTy>
354 ESIMD_INLINE ESIMD_NODEBUG std::enable_if_t<
355  (sizeof(T) <= 4) && (N == 1 || N == 8 || N == 16 || N == 32) &&
356  !std::is_pointer<AccessorTy>::value,
357  simd<T, N>>
358 gather_impl(AccessorTy acc, simd<uint32_t, N> offsets, uint32_t glob_offset,
359  simd_mask<N> mask) {
360 
361  constexpr int TypeSizeLog2 = detail::ElemsPerAddrEncoding<sizeof(T)>();
362  // TODO (performance) use hardware-supported scale once BE supports it
363  constexpr uint32_t scale = 0;
364  const auto si = get_surface_index(acc);
365 
366  if constexpr (sizeof(T) < 4) {
367  using Tint = std::conditional_t<std::is_integral_v<T>, T,
368  detail::uint_type_t<sizeof(T)>>;
369  using Treal = __raw_t<T>;
370  static_assert(std::is_integral<Tint>::value,
371  "only integral 1- & 2-byte types are supported");
372  using PromoT =
373  typename sycl::detail::conditional_t<std::is_signed<Tint>::value,
374  int32_t, uint32_t>;
375  const simd<PromoT, N> promo_vals =
376  __esimd_gather_masked_scaled2<PromoT, N, decltype(si), TypeSizeLog2,
377  scale>(si, glob_offset, offsets.data(),
378  mask.data());
379  auto Res = convert<Tint>(promo_vals);
380 
381  if constexpr (!std::is_same_v<Tint, T>) {
382  return detail::bitcast<Treal, Tint, N>(Res.data());
383  } else {
384  return Res;
385  }
386  } else {
387  return __esimd_gather_masked_scaled2<T, N, decltype(si), TypeSizeLog2,
388  scale>(si, glob_offset, offsets.data(),
389  mask.data());
390  }
391 }

565 template <atomic_op Op, typename T, int N, unsigned NumSrc>
566 constexpr bool check_atomic() {
567  if constexpr (!detail::isPowerOf2(N, 32)) {
568  static_assert((detail::isPowerOf2(N, 32)),
569  "Execution size 1, 2, 4, 8, 16, 32 are supported");
570  return false;
571  }
572 
573  // No source operands.
574  if constexpr (Op == atomic_op::inc || Op == atomic_op::dec) {
575  if constexpr (NumSrc != 0) {
576  static_assert(NumSrc == 0, "No source operands are expected");
577  return false;
578  }
579  if constexpr (!is_type<T, uint16_t, uint32_t, uint64_t>()) {
580  static_assert((is_type<T, uint16_t, uint32_t, uint64_t>()),
581  "Type UW, UD or UQ is expected");
582  return false;
583  }
584  return true;
585  }
586 
587  // One source integer operand.
588  if constexpr (Op == atomic_op::add || Op == atomic_op::sub ||
589  Op == atomic_op::min || Op == atomic_op::max ||
590  Op == atomic_op::xchg || Op == atomic_op::bit_and ||
591  Op == atomic_op::bit_or || Op == atomic_op::bit_xor ||
592  Op == atomic_op::minsint || Op == atomic_op::maxsint) {
593  if constexpr (NumSrc != 1) {
594  static_assert(NumSrc == 1, "One source operand is expected");
595  return false;
596  }
597  if constexpr ((Op != atomic_op::minsint && Op != atomic_op::maxsint) &&
598  !is_type<T, uint16_t, uint32_t, uint64_t>()) {
599  static_assert((is_type<T, uint16_t, uint32_t, uint64_t>()),
600  "Type UW, UD or UQ is expected");
601  return false;
602  }
603  if constexpr ((Op == atomic_op::minsint || Op == atomic_op::maxsint) &&
604  !is_type<T, int16_t, int32_t, int64_t>()) {
605  static_assert((is_type<T, int16_t, int32_t, int64_t>()),
606  "Type W, D or Q is expected");
607  return false;
608  }
609  return true;
610  }
611 
612  // One source float operand.
613  if constexpr (Op == atomic_op::fmax || Op == atomic_op::fmin ||
614  Op == atomic_op::fadd || Op == atomic_op::fsub) {
615  if constexpr (NumSrc != 1) {
616  static_assert(NumSrc == 1, "One source operand is expected");
617  return false;
618  }
619  if constexpr (!is_type<T, float, sycl::half>()) {
620  static_assert((is_type<T, float, sycl::half>()),
621  "Type F or HF is expected");
622  return false;
623  }
624  return true;
625  }
626 
627  // Two source operands.
628  if constexpr (Op == atomic_op::cmpxchg || Op == atomic_op::fcmpwr) {
629  if constexpr (NumSrc != 2) {
630  static_assert(NumSrc == 2, "Two source operands are expected");
631  return false;
632  }
633  if constexpr (Op == atomic_op::cmpxchg &&
634  !is_type<T, uint16_t, uint32_t, uint64_t>()) {
635  static_assert((is_type<T, uint16_t, uint32_t, uint64_t>()),
636  "Type UW, UD or UQ is expected");
637  return false;
638  }
639  if constexpr (Op == atomic_op::fcmpwr && !is_type<T, float, sycl::half>()) {
640  static_assert((is_type<T, float, sycl::half>()),
641  "Type F or HF is expected");
642  return false;
643  }
644  return true;
645  }
646  // Unsupported svm atomic Op.
647  return false;
648 }
