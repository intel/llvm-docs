122 // Functionality for any atomic of type T, reused by partial specializations
123 template <typename T, memory_order DefaultOrder, memory_scope DefaultScope,
124  access::address_space AddressSpace>
125 class atomic_ref_base {
126  static_assert(
127  detail::IsValidAtomicType<T>::value,
128  "Invalid atomic type. Valid types are arithmetic and pointer types");
129  static_assert(!std::is_same<T, bool>::value,
130  "ONEAPI::atomic_ref does not support bool type");
131  static_assert(!(std::is_same<T, char>::value ||
132  std::is_same<T, signed char>::value ||
133  std::is_same<T, unsigned char>::value),
134  "ONEAPI::atomic_ref does not support char type");
135  static_assert(!(std::is_same<T, short>::value ||
136  std::is_same<T, unsigned short>::value),
137  "ONEAPI::atomic_ref does not support short type");
138  static_assert(detail::IsValidAtomicAddressSpace<AddressSpace>::value,
139  "Invalid atomic address_space. Valid address spaces are: "
140  "global_space, local_space, global_device_space");
141  static_assert(
142  detail::IsValidDefaultOrder<DefaultOrder>::value,
143  "Invalid default memory_order for atomics. Valid defaults are: "
144  "relaxed, acq_rel, seq_cst");
145 
146 public:
147  using value_type = T;
148  static constexpr size_t required_alignment = sizeof(T);
149  static constexpr bool is_always_lock_free =
150  detail::IsValidAtomicType<T>::value;
151  static constexpr memory_order default_read_order =
152  detail::memory_order_traits<DefaultOrder>::read_order;
153  static constexpr memory_order default_write_order =
154  detail::memory_order_traits<DefaultOrder>::write_order;
155  static constexpr memory_order default_read_modify_write_order = DefaultOrder;
156  static constexpr memory_scope default_scope = DefaultScope;
157 
158  bool is_lock_free() const noexcept {
159  return detail::IsValidAtomicType<T>::value;
160  }
161 
162 #ifdef __SYCL_DEVICE_ONLY__
163  explicit atomic_ref_base(T &ref) : ptr(multi_ptr<T, AddressSpace>(&ref)) {}
164 #else
165  // FIXME: This reinterpret_cast is UB, but happens to work for now
166  explicit atomic_ref_base(T &ref)
167  : ptr(reinterpret_cast<std::atomic<T> *>(&ref)) {}
168 #endif
169  // Our implementation of copy constructor could be trivial
170  // Defined this way for consistency with standard atomic_ref
171  atomic_ref_base(const atomic_ref_base &ref) noexcept { ptr = ref.ptr; };

211  bool compare_exchange_strong(T &expected, T desired, memory_order success,
212  memory_order failure,
213  memory_scope scope = default_scope) const
214  noexcept {
215 #ifdef __SYCL_DEVICE_ONLY__
216  T value = detail::spirv::AtomicCompareExchange(ptr, scope, success, failure,
217  desired, expected);
218  bool succeeded = detail::bit_equal<T>()(value, expected);
219  if (!succeeded) {
220  expected = value;
221  }
222  return succeeded;
223 #else
224  (void)scope;
225  return ptr->compare_exchange_strong(expected, desired,
226  detail::getStdMemoryOrder(success),
227  detail::getStdMemoryOrder(failure));
228 #endif
229  }
