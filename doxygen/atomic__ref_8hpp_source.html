80  case memory_order_acquire:
81  case memory_order::__consume_unsupported:
82  case memory_order_acq_rel:

122 // Functionality for any atomic of type T, reused by partial specializations
123 template <typename T, memory_order DefaultOrder, memory_scope DefaultScope,
124  access::address_space AddressSpace>
125 class atomic_ref_base {
126  static_assert(
127  detail::IsValidAtomicType<T>::value,
128  "Invalid atomic type. Valid types are arithmetic and pointer types");
129  static_assert(!std::is_same<T, bool>::value,
130  "intel::atomic_ref does not support bool type");
131  static_assert(!(std::is_same<T, char>::value ||
132  std::is_same<T, signed char>::value ||
133  std::is_same<T, unsigned char>::value),
134  "intel::atomic_ref does not support char type");
135  static_assert(!(std::is_same<T, short>::value ||
136  std::is_same<T, unsigned short>::value),
137  "intel::atomic_ref does not support short type");
138  static_assert(!std::is_pointer<T>::value,
139  "intel::atomic_ref does not yet support pointer types");
140  static_assert(detail::IsValidAtomicAddressSpace<AddressSpace>::value,
141  "Invalid atomic address_space. Valid address spaces are: "
142  "global_space, local_space, global_device_space");
143  static_assert(
144  detail::IsValidDefaultOrder<DefaultOrder>::value,
145  "Invalid default memory_order for atomics. Valid defaults are: "
146  "relaxed, acq_rel, seq_cst");
147 
148 public:
149  using value_type = T;
150  static constexpr size_t required_alignment = sizeof(T);
151  static constexpr bool is_always_lock_free =
152  detail::IsValidAtomicType<T>::value;
153  static constexpr memory_order default_read_order =
154  detail::memory_order_traits<DefaultOrder>::read_order;
155  static constexpr memory_order default_write_order =
156  detail::memory_order_traits<DefaultOrder>::write_order;
157  static constexpr memory_order default_read_modify_write_order = DefaultOrder;
158  static constexpr memory_scope default_scope = DefaultScope;
159 
160  bool is_lock_free() const noexcept {
161  return detail::IsValidAtomicType<T>::value;
162  }
163 
164 #ifdef __SYCL_DEVICE_ONLY__
165  explicit atomic_ref_base(T &ref) : ptr(multi_ptr<T, AddressSpace>(&ref)) {}
166 #else
167  // FIXME: This reinterpret_cast is UB, but happens to work for now
168  explicit atomic_ref_base(T &ref)
169  : ptr(reinterpret_cast<std::atomic<T> *>(&ref)) {}
170 #endif
171  // Our implementation of copy constructor could be trivial
172  // Defined this way for consistency with standard atomic_ref
173  atomic_ref_base(const atomic_ref_base &ref) noexcept { ptr = ref.ptr; };
174  atomic_ref_base &operator=(const atomic_ref_base &) = delete;

213  bool compare_exchange_strong(T &expected, T desired, memory_order success,
214  memory_order failure,
215  memory_scope scope = default_scope) const
216  noexcept {
217 #ifdef __SYCL_DEVICE_ONLY__
218  T value = detail::spirv::AtomicCompareExchange(ptr, scope, success, failure,
219  desired, expected);
220  bool succeeded = detail::bit_equal<T>()(value, expected);
221  if (!succeeded) {
222  expected = value;
223  }
224  return succeeded;
225 #else
226  (void)scope;
227  return ptr->compare_exchange_strong(expected, desired,
228  detail::getStdMemoryOrder(success),
229  detail::getStdMemoryOrder(failure));
230 #endif
231  }
