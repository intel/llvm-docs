60 // Compute the region type of a simd_view type.
61 //
62 // A region type could be either
63 // - region1d_t
64 // - region2d_t
65 // - a pair (top_region_type, base_region_type)
66 //
67 // This is a recursive definition to capture the following rvalue region stack:
68 //
69 // simd<int 16> v;
70 // v.format<int, 4, 4>().select<1, 0, 4, 1>(0, 0).format<short>() = 0;
71 //
72 // The LHS will be represented as a rvalue
73 //
74 // simd_view({v, { region1d_t<short, 8, 1>(0, 0),
75 // { region2d_t<int, 1, 1, 4, 1>(0, 0),
76 // region2d_t<int, 4, 1, 4, 1>(0, 0)
77 // }}})
78 //
79 template <typename Ty> struct shape_type {
80  using element_type = Ty;
81  using type = void;
82 };
83
