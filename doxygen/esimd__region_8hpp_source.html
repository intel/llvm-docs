62 // Compute the region type of a simd_view type.
63 //
64 // A region type could be either
65 // - region1d_t
66 // - region2d_t
67 // - a pair (top_region_type, base_region_type)
68 //
69 // This is a recursive definition to capture the following rvalue region stack:
70 //
71 // simd<int 16> v;
72 // v.format<int, 4, 4>().select<1, 0, 4, 1>(0, 0).format<short>() = 0;
73 //
74 // The LHS will be represented as a rvalue
75 //
76 // simd_view({v, { region1d_t<short, 8, 1>(0, 0),
77 // { region2d_t<int, 1, 1, 4, 1>(0, 0),
78 // region2d_t<int, 4, 1, 4, 1>(0, 0)
79 // }}})
80 //
81 template <typename Ty> struct shape_type {
82  using element_type = Ty;
83  using type = void;
84 };
