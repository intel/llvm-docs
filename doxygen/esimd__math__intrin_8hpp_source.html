249 template <typename T>
250 T extract(const uint32_t &width, const uint32_t &offset, uint32_t src,
251  uint32_t &sign_extend) {
252  uint32_t mask = ((1 << width) - 1) << offset;
253  T ret = (src & mask) >> offset;
254  if (sign_extend) {
255  if ((src >> (offset + width - 1)) & 0x1) {
256  uint32_t sign_extend = ((1 << (32 - width)) - 1) << width;
257  ret = ret | sign_extend;
258  }
259  }
260 
261  return ret;
262 }
