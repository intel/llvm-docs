24 inline sycl::detail::pi::PiVirtualAccessFlags AccessModeToVirtualAccessFlags(
25  ext::oneapi::experimental::address_access_mode Mode) {
26  switch (Mode) {
27  case ext::oneapi::experimental::address_access_mode::read:
28  return PI_VIRTUAL_ACCESS_FLAG_READ_ONLY;
29  case ext::oneapi::experimental::address_access_mode::read_write:
30  return PI_VIRTUAL_ACCESS_FLAG_RW;
31  case ext::oneapi::experimental::address_access_mode::none:
32  return 0;
33  }
34  throw sycl::exception(make_error_code(errc::invalid),
35  "Invalid address_access_mode.");
36 }
