190  // Maps specialization constant symbolic name to its integer ID.
191  // NOTE: a key in this map is a pointer into some existing device binary image
192  // and it is invalidated once the encompassing OS module gets unloaded.
193  using SpecConstMapTy =
194  std::unordered_map<const char *, spec_constant_impl, HashCStr, CmpCStr>;
195 
196  // Keeps specialization constant map for each operating system module.
197  // The base approach is that there is a global name->id spec constant map for
198  // each OS module. This map is populated from the name->id pairs found in
199  // device binary image upon registration at startup for this OS module.
200  // If multiple images beloning to the same OS module have a specialization
201  // constant with the same name, its corresponding numeric id must also match.
202  // It is modified (populated) once at startup, so no locks are necessary.
203  std::unordered_map<OSModuleHandle, SpecConstMapTy> SpecConstRegistry;
204 
205  // Keeps track of pi_program to image correspondence. Needed for:
206  // - knowing which specialization constants are used in the program and
207  // injecting their current values before compiling the SPIRV; the binary
208  // image object has info about all spec constants used in the module
209  // NOTE: using RTDeviceBinaryImage raw pointers is OK, since they are not
210  // referenced from outside SYCL runtime and RTDeviceBinaryImage object
211  // lifetime matches program manager's one.
212  // NOTE: keys in the map can be invalid (reference count went to zero and
213  // the underlying program disposed of), so the map can't be used in any way
214  // other than binary image lookup with known live PiProgram as the key.
215  // NOTE: access is synchronized via the same lock as program cache
216  std::unordered_map<pi::PiProgram, const RTDeviceBinaryImage *> NativePrograms;
