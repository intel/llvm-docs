44 // SYCL 1.2.1rev5, section "4.8.5.3 Parallel For hierarchical invoke":
45 // Quote:
46 // ... To guarantee use of private per-work-item memory, the private_memory
47 // class can be used to wrap the data. This class very simply constructs
48 // private data for a given group across the entire group.The id of the
49 // current work-item is passed to any access to grab the correct data.
50 template <typename T, int Dimensions = 1> class private_memory {
51 public:
52  // Construct based directly off the number of work-items
53  private_memory(const group<Dimensions> &G) {
54 #ifndef __SYCL_DEVICE_ONLY__
55  // serial host => one instance per work-group - allocate space for each WI
56  // in the group:
57  Val.reset(new T[G.get_local_range().size()]);
58 #endif // __SYCL_DEVICE_ONLY__
59  (void)G;
60  }
61 
62  // Access the instance for the current work-item
63  T &operator()(const h_item<Dimensions> &Id) {
64 #ifndef __SYCL_DEVICE_ONLY__
65  // Calculate the linear index of current WI and return reference to the
66  // corresponding spot in the value array:
67  size_t Ind = Id.get_physical_local().get_linear_id();
68  return Val.get()[Ind];
69 #else
70  (void)Id;
71  return Val;
72 #endif // __SYCL_DEVICE_ONLY__
73  }
74 
75 private:
76 #ifdef __SYCL_DEVICE_ONLY__
77  // On SYCL device private_memory<T> instance is created per physical WI, so
78  // there is 1:1 correspondence betwen this class instances and per-WI memory.
79  T Val;
80 #else
81  // On serial host there is one private_memory<T> instance per work group, so
82  // it must have space to hold separate value per WI in the group.
83  std::unique_ptr<T[]> Val;
84 #endif // #ifdef __SYCL_DEVICE_ONLY__
85 };

134  // SYCL specification 1.2.1rev5, section 4.7.6.5 "Buffer accessor":
135  // Whenever a multi-dimensional index is passed to a SYCL accessor the
136  // linear index is calculated based on the index {id1, id2, id3} provided
137  // and the range of the SYCL accessor {r1, r2, r3} according to row-major
138  // ordering as follows:
139  // id3 + (id2 · r3) + (id1 · r3 · r2) (4.3)
140  // section 4.8.1.8 "group class":
141  // size_t get_linear_id()const
142  // Get a linearized version of the work-group id. Calculating a linear
143  // work-group id from a multi-dimensional index follows the equation 4.3.
144  template <int dims = Dimensions>
145  typename std::enable_if<(dims == 3), size_t>::type get_linear_id() const {
146  return (index[0] * groupRange[1] * groupRange[2]) +
147  (index[1] * groupRange[2]) + index[2];
148  }

150  template <typename WorkItemFunctionT>
151  void parallel_for_work_item(WorkItemFunctionT Func) const {
152  // need barriers to enforce SYCL semantics for the work item loop -
153  // compilers are expected to optimize when possible
154  detail::workGroupBarrier();
155 #ifdef __SYCL_DEVICE_ONLY__
156  range<Dimensions> GlobalSize{
157  __spirv::initGlobalSize<Dimensions, range<Dimensions>>()};
158  range<Dimensions> LocalSize{
159  __spirv::initWorkgroupSize<Dimensions, range<Dimensions>>()};
160  id<Dimensions> GlobalId{
161  __spirv::initGlobalInvocationId<Dimensions, id<Dimensions>>()};
162  id<Dimensions> LocalId{
163  __spirv::initLocalInvocationId<Dimensions, id<Dimensions>>()};
164 
165  // no 'iterate' in the device code variant, because
166  // (1) this code is already invoked by each work item as a part of the
167  // enclosing parallel_for_work_group kernel
168  // (2) the range this pfwi iterates over matches work group size exactly
169  item<Dimensions, false> GlobalItem =
170  detail::Builder::createItem<Dimensions, false>(GlobalSize, GlobalId);
171  item<Dimensions, false> LocalItem =
172  detail::Builder::createItem<Dimensions, false>(LocalSize, LocalId);
173  h_item<Dimensions> HItem =
174  detail::Builder::createHItem<Dimensions>(GlobalItem, LocalItem);
175 
176  Func(HItem);
177 #else
178  id<Dimensions> GroupStartID = index * localRange;
179 
180  // ... host variant needs explicit 'iterate' because it is serial
181  detail::NDLoop<Dimensions>::iterate(
182  localRange, [&](const id<Dimensions> &LocalID) {
183  item<Dimensions, false> GlobalItem =
184  detail::Builder::createItem<Dimensions, false>(
185  globalRange, GroupStartID + LocalID);
186  item<Dimensions, false> LocalItem =
187  detail::Builder::createItem<Dimensions, false>(localRange,
188  LocalID);
189  h_item<Dimensions> HItem =
190  detail::Builder::createHItem<Dimensions>(GlobalItem, LocalItem);
191  Func(HItem);
192  });
193 #endif // __SYCL_DEVICE_ONLY__
194  // Need both barriers here - before and after the parallel_for_work_item
195  // (PFWI). There can be work group scope code after the PFWI which reads
196  // work group local data written within this PFWI. Back Ends are expected to
197  // optimize away unneeded barriers (e.g. two barriers in a row).
198  detail::workGroupBarrier();
199  }

201  template <typename WorkItemFunctionT>
202  void parallel_for_work_item(range<Dimensions> flexibleRange,
203  WorkItemFunctionT Func) const {
204  detail::workGroupBarrier();
205 #ifdef __SYCL_DEVICE_ONLY__
206  range<Dimensions> GlobalSize{
207  __spirv::initGlobalSize<Dimensions, range<Dimensions>>()};
208  range<Dimensions> LocalSize{
209  __spirv::initWorkgroupSize<Dimensions, range<Dimensions>>()};
210  id<Dimensions> GlobalId{
211  __spirv::initGlobalInvocationId<Dimensions, id<Dimensions>>()};
212  id<Dimensions> LocalId{
213  __spirv::initLocalInvocationId<Dimensions, id<Dimensions>>()};
214 
215  item<Dimensions, false> GlobalItem =
216  detail::Builder::createItem<Dimensions, false>(GlobalSize, GlobalId);
217  item<Dimensions, false> LocalItem =
218  detail::Builder::createItem<Dimensions, false>(LocalSize, LocalId);
219  h_item<Dimensions> HItem = detail::Builder::createHItem<Dimensions>(
220  GlobalItem, LocalItem, flexibleRange);
221 
222  // iterate over flexible range with work group size stride; each item
223  // performs flexibleRange/LocalSize iterations (if the former is divisible
224  // by the latter)
225  detail::NDLoop<Dimensions>::iterate(
226  LocalId, LocalSize, flexibleRange,
227  [&](const id<Dimensions> &LogicalLocalID) {
228  HItem.setLogicalLocalID(LogicalLocalID);
229  Func(HItem);
230  });
231 #else
232  id<Dimensions> GroupStartID = index * localRange;
233 
234  detail::NDLoop<Dimensions>::iterate(
235  localRange, [&](const id<Dimensions> &LocalID) {
236  item<Dimensions, false> GlobalItem =
237  detail::Builder::createItem<Dimensions, false>(
238  globalRange, GroupStartID + LocalID);
239  item<Dimensions, false> LocalItem =
240  detail::Builder::createItem<Dimensions, false>(localRange,
241  LocalID);
242  h_item<Dimensions> HItem = detail::Builder::createHItem<Dimensions>(
243  GlobalItem, LocalItem, flexibleRange);
244 
245  detail::NDLoop<Dimensions>::iterate(
246  LocalID, localRange, flexibleRange,
247  [&](const id<Dimensions> &LogicalLocalID) {
248  HItem.setLogicalLocalID(LogicalLocalID);
249  Func(HItem);
250  });
251  });
252 #endif // __SYCL_DEVICE_ONLY__
253  detail::workGroupBarrier();
254  }
