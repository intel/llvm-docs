39 bool visitNodeDepthFirst(
40  std::shared_ptr<node_impl> Node,
41  std::set<std::shared_ptr<node_impl>> &VisitedNodes,
42  std::deque<std::shared_ptr<node_impl>> &NodeStack,
43  std::function<bool(std::shared_ptr<node_impl> &,
44  std::deque<std::shared_ptr<node_impl>> &)>
45  NodeFunc) {
46  auto EarlyReturn = NodeFunc(Node, NodeStack);
47  if (EarlyReturn) {
48  return true;
49  }
50  NodeStack.push_back(Node);
51  Node->MVisited = true;
52  VisitedNodes.emplace(Node);
53  for (auto &Successor : Node->MSuccessors) {
54  if (visitNodeDepthFirst(Successor.lock(), VisitedNodes, NodeStack,
55  NodeFunc)) {
56  return true;
57  }
58  }
59  NodeStack.pop_back();
60  return false;
61 }

70  std::list<std::shared_ptr<node_impl>> &Schedule,
71  bool PartitionBounded = false) {
72  for (auto &Succ : NodeImpl->MSuccessors) {
73  auto NextNode = Succ.lock();

76  continue;

81  sortTopological(NextNode, Schedule, PartitionBounded);
82  }
83  }
84

183 void exec_graph_impl::makePartitions() {
184  int CurrentPartition = -1;
185  std::list<std::shared_ptr<node_impl>> HostTaskList;
186  // find all the host-tasks in the graph
187  for (auto &Node : MNodeStorage) {
188  if (Node->MCGType == sycl::detail::CG::CodeplayHostTask) {
189  HostTaskList.push_back(Node);
190  }
191  }
192 
193  // Annotate nodes
194  // The first step in graph partitioning is to annotate all nodes of the graph
195  // with a temporary partition or group number. This step allows us to group
196  // the graph nodes into sets of nodes with kind of meta-dependencies that must
197  // be enforced by the runtime. For example, Group 2 depends on Groups 0 and 1,
198  // which means that we should not try to run Group 2 before Groups 0 and 1
199  // have finished executing. Since host-tasks are currently the only tasks that
200  // require runtime dependency handling, groups of nodes are created from
201  // host-task nodes. We therefore loop over all the host-task nodes, and for
202  // each node:
203  // - Its predecessors are assigned to group number `n-1`
204  // - The node itself constitutes a group, group number `n`
205  // - Its successors are assigned to group number `n+1`
206  // Since running multiple partitions slows down the whole graph execution, we
207  // then try to reduce the number of partitions by merging them when possible.
208  // Typically, the grouping algorithm can create two successive partitions
209  // of target nodes in the following case:
210  // A host-task `A` is added to the graph. Later, another host task `B` is
211  // added to the graph. Consequently, the node `A` is stored before the node
212  // `B` in the node storage vector. Now, if `A` is placed as a successor of `B`
213  // (using make_edge function to make node `A` dependent on node `B`.) In this
214  // case, the host-task node `A` must be reprocessed after the node `B` and the
215  // group that includes the predecessor of `B` can be merged with the group of
216  // the predecessors of the node `A`.
217  while (HostTaskList.size() > 0) {
218  auto Node = HostTaskList.front();
219  HostTaskList.pop_front();
220  CurrentPartition++;
221  for (auto &Predecessor : Node->MPredecessors) {
222  propagatePartitionUp(Predecessor.lock(), CurrentPartition);
223  }
224  CurrentPartition++;
225  Node->MPartitionNum = CurrentPartition;
226  CurrentPartition++;
227  auto TmpSize = HostTaskList.size();
228  for (auto &Successor : Node->MSuccessors) {
229  propagatePartitionDown(Successor.lock(), CurrentPartition, HostTaskList);
230  }
231  if (HostTaskList.size() > TmpSize) {
232  // At least one HostTask has been re-numbered so group merge opportunities
233  for (const auto &HT : HostTaskList) {
234  auto HTPartitionNum = HT->MPartitionNum;
235  if (HTPartitionNum != -1) {
236  // can merge predecessors of node `Node` with predecessors of node
237  // `HT` (HTPartitionNum-1) since HT must be reprocessed
238  for (const auto &NodeImpl : MNodeStorage) {
239  if (NodeImpl->MPartitionNum == Node->MPartitionNum - 1) {
240  NodeImpl->MPartitionNum = HTPartitionNum - 1;
241  }
242  }
243  } else {
244  break;
245  }
246  }
247  }
248  }
249 
250  // Create partitions
251  int PartitionFinalNum = 0;
252  for (int i = -1; i <= CurrentPartition; i++) {
253  const std::shared_ptr<partition> &Partition = std::make_shared<partition>();
254  for (auto &Node : MNodeStorage) {
255  if (Node->MPartitionNum == i) {
256  MPartitionNodes[Node] = PartitionFinalNum;
257  if (isPartitionRoot(Node)) {
258  Partition->MRoots.insert(Node);
259  }
260  }
261  }
262  if (Partition->MRoots.size() > 0) {
263  Partition->schedule();
264  Partition->MIsInOrderGraph = Partition->checkIfGraphIsSinglePath();
265  MPartitions.push_back(Partition);
266  PartitionFinalNum++;
267  }
268  }
269 
270  // Add an empty partition if there is no partition, i.e. empty graph
271  if (MPartitions.size() == 0) {
272  MPartitions.push_back(std::make_shared<partition>());
273  }
274 
275  // Make global schedule list
276  for (const auto &Partition : MPartitions) {
277  MSchedule.insert(MSchedule.end(), Partition->MSchedule.begin(),
278  Partition->MSchedule.end());
279  }
280 
281  // Compute partition dependencies
282  for (const auto &Partition : MPartitions) {
283  for (auto const &Root : Partition->MRoots) {
284  auto RootNode = Root.lock();
285  for (const auto &Dep : RootNode->MPredecessors) {
286  auto NodeDep = Dep.lock();
287  Partition->MPredecessors.push_back(
288  MPartitions[MPartitionNodes[NodeDep]]);
289  }
290  }
291  }
292 
293  // Reset node groups (if node have to be re-processed - e.g. subgraph)
294  for (auto &Node : MNodeStorage) {
295  Node->MPartitionNum = -1;
296  }
297 }

307  const std::shared_ptr<graph_impl> &Impl,
308  const std::list<std::shared_ptr<node_impl>> &NodeList) {
309  // Find all input and output nodes from the node list
310  std::vector<std::shared_ptr<node_impl>> Inputs;
311  std::vector<std::shared_ptr<node_impl>> Outputs;
312  for (auto &NodeImpl : NodeList) {

322  // successors

325  for (auto &Input : Inputs) {
326  NodeImpl->registerSuccessor(Input, NodeImpl);
327  }
328  }
329  }

332  for (auto &Node : NodeList) {
333  MNodeStorage.push_back(Node);

335  }

433 std::shared_ptr<node_impl>
434 graph_impl::add(node_type NodeType,
435  std::unique_ptr<sycl::detail::CG> CommandGroup,
436  const std::vector<std::shared_ptr<node_impl>> &Dep) {
437  // Copy deps so we can modify them
438  auto Deps = Dep;
439 
440  // A unique set of dependencies obtained by checking requirements and events
441  std::set<std::shared_ptr<node_impl>> UniqueDeps;
442  const auto &Requirements = CommandGroup->getRequirements();
443  if (!MAllowBuffers && Requirements.size()) {
444  throw sycl::exception(make_error_code(errc::invalid),
445  "Cannot use buffers in a graph without passing the "
446  "assume_buffer_outlives_graph property on "
447  "Graph construction.");
448  }
449 
450  for (auto &Req : Requirements) {
451  // Track and mark the memory objects being used by the graph.
452  auto MemObj = static_cast<sycl::detail::SYCLMemObjT *>(Req->MSYCLMemObj);
453  bool WasInserted = MMemObjs.insert(MemObj).second;
454  if (WasInserted) {
455  MemObj->markBeingUsedInGraph();
456  }
457  // Look through the graph for nodes which share this requirement
458  for (auto &Node : MNodeStorage) {
459  if (Node->hasRequirementDependency(Req)) {
460  bool ShouldAddDep = true;
461  // If any of this node's successors have this requirement then we skip
462  // adding the current node as a dependency.
463  for (auto &Succ : Node->MSuccessors) {
464  if (Succ.lock()->hasRequirementDependency(Req)) {
465  ShouldAddDep = false;
466  break;
467  }
468  }
469  if (ShouldAddDep) {
470  UniqueDeps.insert(Node);
471  }
472  }
473  }
474  }
475 
476  // Add any nodes specified by event dependencies into the dependency list
477  for (auto &Dep : CommandGroup->getEvents()) {
478  if (auto NodeImpl = MEventsMap.find(Dep); NodeImpl != MEventsMap.end()) {
479  UniqueDeps.insert(NodeImpl->second);
480  } else {
481  throw sycl::exception(sycl::make_error_code(errc::invalid),
482  "Event dependency from handler::depends_on does "
483  "not correspond to a node within the graph");
484  }
485  }
486  // Add any deps determined from requirements and events into the dependency
487  // list
488  Deps.insert(Deps.end(), UniqueDeps.begin(), UniqueDeps.end());
489 
490  // Add any deps from the extra dependencies vector
491  Deps.insert(Deps.end(), MExtraDependencies.begin(), MExtraDependencies.end());
492 
493  const std::shared_ptr<node_impl> &NodeImpl =
494  std::make_shared<node_impl>(NodeType, std::move(CommandGroup));
495  MNodeStorage.push_back(NodeImpl);
496 
497  addDepsToNode(NodeImpl, Deps);
498 
499  // Set barrier nodes as prerequisites (new start points) for subsequent nodes
500  if (NodeImpl->MCGType == sycl::detail::CG::Barrier) {
501  MExtraDependencies.push_back(NodeImpl);
502  }
503 
504  return NodeImpl;
505 }

542 bool graph_impl::checkForCycles() {
543  // Using a depth-first search and checking if we vist a node more than once in
544  // the current path to identify if there are cycles.
545  bool CycleFound = false;
546  auto CheckFunc = [&](std::shared_ptr<node_impl> &Node,
547  std::deque<std::shared_ptr<node_impl>> &NodeStack) {
548  // If the current node has previously been found in the current path through
549  // the graph then we have a cycle and we end the search early.
550  if (std::find(NodeStack.begin(), NodeStack.end(), Node) !=
551  NodeStack.end()) {
552  CycleFound = true;
553  return true;
554  }
555  return false;
556  };
557  searchDepthFirst(CheckFunc);
558  return CycleFound;
559 }

561 void graph_impl::makeEdge(std::shared_ptr<node_impl> Src,
562  std::shared_ptr<node_impl> Dest) {
563  throwIfGraphRecordingQueue("make_edge()");
564  if (Src == Dest) {
565  throw sycl::exception(
566  make_error_code(sycl::errc::invalid),
567  "make_edge() cannot be called when Src and Dest are the same.");
568  }
569 
570  bool SrcFound = false;
571  bool DestFound = false;
572  for (const auto &Node : MNodeStorage) {
573 
574  SrcFound |= Node == Src;
575  DestFound |= Node == Dest;
576 
577  if (SrcFound && DestFound) {
578  break;
579  }
580  }
581 
582  if (!SrcFound) {
583  throw sycl::exception(make_error_code(sycl::errc::invalid),
584  "Src must be a node inside the graph.");
585  }
586  if (!DestFound) {
587  throw sycl::exception(make_error_code(sycl::errc::invalid),
588  "Dest must be a node inside the graph.");
589  }
590 
591  // We need to add the edges first before checking for cycles
592  Src->registerSuccessor(Dest, Src);
593 
594  // We can skip cycle checks if either Dest has no successors (cycle not
595  // possible) or cycle checks have been disabled with the no_cycle_check
596  // property;
597  if (Dest->MSuccessors.empty() || !MSkipCycleChecks) {
598  bool CycleFound = checkForCycles();
599 
600  if (CycleFound) {
601  // Remove the added successor and predecessor
602  Src->MSuccessors.pop_back();
603  Dest->MPredecessors.pop_back();
604 
605  throw sycl::exception(make_error_code(sycl::errc::invalid),
606  "Command graphs cannot contain cycles.");
607  }
608  }
609  removeRoot(Dest); // remove receiver from root node list
610 }

816 sycl::event
817 exec_graph_impl::enqueue(const std::shared_ptr<sycl::detail::queue_impl> &Queue,
818  sycl::detail::CG::StorageInitHelper CGData) {
819  WriteLock Lock(MMutex);
820 
821  // Map of the partitions to their execution events
822  std::unordered_map<std::shared_ptr<partition>, sycl::detail::EventImplPtr>
823  PartitionsExecutionEvents;
824 
825  auto CreateNewEvent([&]() {
826  auto NewEvent = std::make_shared<sycl::detail::event_impl>(Queue);
827  NewEvent->setContextImpl(Queue->getContextImplPtr());
828  NewEvent->setStateIncomplete();
829  return NewEvent;
830  });
831 
832  sycl::detail::EventImplPtr NewEvent;
833  std::vector<sycl::detail::EventImplPtr> BackupCGDataMEvents;
834  if (MPartitions.size() > 1) {
835  BackupCGDataMEvents = CGData.MEvents;
836  }
837  for (uint32_t currentPartitionsNum = 0;
838  currentPartitionsNum < MPartitions.size(); currentPartitionsNum++) {
839  auto CurrentPartition = MPartitions[currentPartitionsNum];
840  // restore initial MEvents to add only needed additional depenencies
841  if (currentPartitionsNum > 0) {
842  CGData.MEvents = BackupCGDataMEvents;
843  }
844 
845  for (auto const &DepPartition : CurrentPartition->MPredecessors) {
846  CGData.MEvents.push_back(PartitionsExecutionEvents[DepPartition]);
847  }
848 
849  auto CommandBuffer =
850  CurrentPartition->MPiCommandBuffers[Queue->get_device()];
851 
852  if (CommandBuffer) {
853  // if previous submissions are incompleted, we automatically
854  // add completion events of previous submissions as dependencies.
855  // With Level-Zero backend we cannot resubmit a command-buffer until the
856  // previous one has already completed.
857  // Indeed, since a command-list does not accept a list a dependencies at
858  // submission, we circumvent this lack by adding a barrier that waits on a
859  // specific event and then define the conditions to signal this event in
860  // another command-list. Consequently, if a second submission is
861  // performed, the signal conditions of this single event are redefined by
862  // this second submission. Thus, this can lead to an undefined behaviour
863  // and potential hangs. We have therefore to expliclty wait in the host
864  // for previous submission to complete before resubmitting the
865  // command-buffer for level-zero backend.
866  // TODO : add a check to release this constraint and allow multiple
867  // concurrent submissions if the exec_graph has been updated since the
868  // last submission.
869  for (std::vector<sycl::detail::EventImplPtr>::iterator It =
870  MExecutionEvents.begin();
871  It != MExecutionEvents.end();) {
872  auto Event = *It;
873  if (!Event->isCompleted()) {
874  if (Queue->get_device().get_backend() ==
875  sycl::backend::ext_oneapi_level_zero) {
876  Event->wait(Event);
877  } else {
878  auto &AttachedEventsList = Event->getPostCompleteEvents();
879  CGData.MEvents.reserve(AttachedEventsList.size() + 1);
880  CGData.MEvents.push_back(Event);
881  // Add events of the previous execution of all graph partitions.
882  for (auto &AttachedEvent : AttachedEventsList) {
883  CGData.MEvents.push_back(AttachedEvent);
884  }
885  }
886  ++It;
887  } else {
888  // Remove completed events
889  It = MExecutionEvents.erase(It);
890  }
891  }
892 
893  NewEvent = CreateNewEvent();
894  sycl::detail::pi::PiEvent *OutEvent = &NewEvent->getHandleRef();
895  // Merge requirements from the nodes into requirements (if any) from the
896  // handler.
897  CGData.MRequirements.insert(CGData.MRequirements.end(),
898  MRequirements.begin(), MRequirements.end());
899  CGData.MAccStorage.insert(CGData.MAccStorage.end(), MAccessors.begin(),
900  MAccessors.end());
901 
902  // If we have no requirements or dependent events for the command buffer,
903  // enqueue it directly
904  if (CGData.MRequirements.empty() && CGData.MEvents.empty()) {
905  if (NewEvent != nullptr)
906  NewEvent->setHostEnqueueTime();
907  pi_result Res =
908  Queue->getPlugin()
909  ->call_nocheck<
910  sycl::detail::PiApiKind::piextEnqueueCommandBuffer>(
911  CommandBuffer, Queue->getHandleRef(), 0, nullptr, OutEvent);
912  if (Res == pi_result::PI_ERROR_INVALID_QUEUE_PROPERTIES) {
913  throw sycl::exception(
914  make_error_code(errc::invalid),
915  "Graphs cannot be submitted to a queue which uses "
916  "immediate command lists. Use "
917  "sycl::ext::intel::property::queue::no_immediate_"
918  "command_list to disable them.");
919  } else if (Res != pi_result::PI_SUCCESS) {
920  throw sycl::exception(
921  errc::event,
922  "Failed to enqueue event for command buffer submission");
923  }
924  } else {
925  std::unique_ptr<sycl::detail::CG> CommandGroup =
926  std::make_unique<sycl::detail::CGExecCommandBuffer>(
927  CommandBuffer, nullptr, std::move(CGData));
928 
929  NewEvent = sycl::detail::Scheduler::getInstance().addCG(
930  std::move(CommandGroup), Queue);
931  }
932  NewEvent->setEventFromSubmittedExecCommandBuffer(true);
933  } else if ((CurrentPartition->MSchedule.size() > 0) &&
934  (CurrentPartition->MSchedule.front()->MCGType ==
935  sycl::detail::CG::CGTYPE::CodeplayHostTask)) {
936  auto NodeImpl = CurrentPartition->MSchedule.front();
937  // Schedule host task
938  NodeImpl->MCommandGroup->getEvents().insert(
939  NodeImpl->MCommandGroup->getEvents().end(), CGData.MEvents.begin(),
940  CGData.MEvents.end());
941  // HostTask CG stores the Queue on which the task was submitted.
942  // In case of graph, this queue may differ from the actual execution
943  // queue. We therefore overload this Queue before submitting the task.
944  static_cast<sycl::detail::CGHostTask &>(*NodeImpl->MCommandGroup.get())
945  .MQueue = Queue;
946 
947  NewEvent = sycl::detail::Scheduler::getInstance().addCG(
948  NodeImpl->getCGCopy(), Queue);
949  } else {
950  std::vector<std::shared_ptr<sycl::detail::event_impl>> ScheduledEvents;
951  for (auto &NodeImpl : CurrentPartition->MSchedule) {
952  std::vector<sycl::detail::pi::PiEvent> RawEvents;
953 
954  // If the node has no requirements for accessors etc. then we skip the
955  // scheduler and enqueue directly.
956  if (NodeImpl->MCGType == sycl::detail::CG::Kernel &&
957  NodeImpl->MCommandGroup->getRequirements().size() +
958  static_cast<sycl::detail::CGExecKernel *>(
959  NodeImpl->MCommandGroup.get())
960  ->MStreams.size() ==
961  0) {
962  sycl::detail::CGExecKernel *CG =
963  static_cast<sycl::detail::CGExecKernel *>(
964  NodeImpl->MCommandGroup.get());
965  auto OutEvent = CreateNewEvent();
966  pi_int32 Res = sycl::detail::enqueueImpKernel(
967  Queue, CG->MNDRDesc, CG->MArgs, CG->MKernelBundle,
968  CG->MSyclKernel, CG->MKernelName, RawEvents, OutEvent,
969  // TODO: Pass accessor mem allocations
970  nullptr,
971  // TODO: Extract from handler
972  PI_EXT_KERNEL_EXEC_INFO_CACHE_DEFAULT, CG->MKernelIsCooperative);
973  if (Res != pi_result::PI_SUCCESS) {
974  throw sycl::exception(
975  sycl::make_error_code(sycl::errc::kernel),
976  "Error during emulated graph command group submission.");
977  }
978  ScheduledEvents.push_back(NewEvent);
979  } else if (!NodeImpl->isEmpty()) {
980  // Empty nodes are node processed as other nodes, but only their
981  // dependencies are propagated in findRealDeps
982  sycl::detail::EventImplPtr EventImpl =
983  sycl::detail::Scheduler::getInstance().addCG(
984  NodeImpl->getCGCopy(), Queue);
985 
986  ScheduledEvents.push_back(EventImpl);
987  }
988  }
989  // Create an event which has all kernel events as dependencies
990  NewEvent = std::make_shared<sycl::detail::event_impl>(Queue);
991  NewEvent->setStateIncomplete();
992  NewEvent->getPreparedDepsEvents() = ScheduledEvents;
993  }
994  PartitionsExecutionEvents[CurrentPartition] = NewEvent;
995  }
996 
997  // Keep track of this execution event so we can make sure it's completed in
998  // the destructor.
999  MExecutionEvents.push_back(NewEvent);
1000  // Attach events of previous partitions to ensure that when the returned event
1001  // is complete all execution associated with the graph have been completed.
1002  for (auto const &Elem : PartitionsExecutionEvents) {
1003  if (Elem.second != NewEvent) {
1004  NewEvent->attachEventToComplete(Elem.second);
1005  }
1006  }
1007  NewEvent->setProfilingEnabled(MEnableProfiling);
1008  sycl::event QueueEvent =
1009  sycl::detail::createSyclObjFromImpl<sycl::event>(NewEvent);
1010  return QueueEvent;
1011 }

1013 void exec_graph_impl::duplicateNodes() {
1014  // Map of original modifiable nodes (keys) to new duplicated nodes (values)
1015  std::map<std::shared_ptr<node_impl>, std::shared_ptr<node_impl>> NodesMap;
1016 
1017  const std::vector<std::shared_ptr<node_impl>> &ModifiableNodes =
1018  MGraphImpl->MNodeStorage;
1019  std::deque<std::shared_ptr<node_impl>> NewNodes;
1020 
1021  for (size_t i = 0; i < ModifiableNodes.size(); i++) {
1022  auto OriginalNode = ModifiableNodes[i];
1023  std::shared_ptr<node_impl> NodeCopy =
1024  std::make_shared<node_impl>(*OriginalNode);
1025 
1026  // Associate the ID of the original node with the node copy for later quick
1027  // access
1028  MIDCache.insert(std::make_pair(OriginalNode->MID, NodeCopy));
1029 
1030  // Clear edges between nodes so that we can replace with new ones
1031  NodeCopy->MSuccessors.clear();
1032  NodeCopy->MPredecessors.clear();
1033  // Push the new node to the front of the stack
1034  NewNodes.push_back(NodeCopy);
1035  // Associate the new node with the old one for updating edges
1036  NodesMap.insert({OriginalNode, NodeCopy});
1037  }
1038 
1039  // Now that all nodes have been copied rebuild edges on new nodes. This must
1040  // be done as a separate step since successors may be out of order.
1041  for (size_t i = 0; i < ModifiableNodes.size(); i++) {
1042  auto OriginalNode = ModifiableNodes[i];
1043  auto NodeCopy = NewNodes[i];
1044  // Look through all the original node successors, find their copies and
1045  // register those as successors with the current copied node
1046  for (auto &NextNode : OriginalNode->MSuccessors) {
1047  auto Successor = NodesMap.at(NextNode.lock());
1048  NodeCopy->registerSuccessor(Successor, NodeCopy);
1049  }
1050  }
1051 
1052  // Subgraph nodes need special handling, we extract all subgraph nodes and
1053  // merge them into the main node list
1054 
1055  for (auto NewNodeIt = NewNodes.rbegin(); NewNodeIt != NewNodes.rend();
1056  ++NewNodeIt) {
1057  auto NewNode = *NewNodeIt;
1058  if (NewNode->MNodeType != node_type::subgraph) {
1059  continue;
1060  }
1061  const std::vector<std::shared_ptr<node_impl>> &SubgraphNodes =
1062  NewNode->MSubGraphImpl->MNodeStorage;
1063  std::deque<std::shared_ptr<node_impl>> NewSubgraphNodes{};
1064 
1065  // Map of original subgraph nodes (keys) to new duplicated nodes (values)
1066  std::map<std::shared_ptr<node_impl>, std::shared_ptr<node_impl>>
1067  SubgraphNodesMap;
1068 
1069  // Copy subgraph nodes
1070  for (size_t i = 0; i < SubgraphNodes.size(); i++) {
1071  auto SubgraphNode = SubgraphNodes[i];
1072  auto NodeCopy = std::make_shared<node_impl>(*SubgraphNode);
1073  // Associate the ID of the original subgraph node with all extracted node
1074  // copies for future quick access.
1075  MIDCache.insert(std::make_pair(SubgraphNode->MID, NodeCopy));
1076 
1077  NewSubgraphNodes.push_back(NodeCopy);
1078  SubgraphNodesMap.insert({SubgraphNode, NodeCopy});
1079  NodeCopy->MSuccessors.clear();
1080  NodeCopy->MPredecessors.clear();
1081  }
1082 
1083  // Rebuild edges for new subgraph nodes
1084  for (size_t i = 0; i < SubgraphNodes.size(); i++) {
1085  auto SubgraphNode = SubgraphNodes[i];
1086  auto NodeCopy = NewSubgraphNodes[i];
1087 
1088  for (auto &NextNode : SubgraphNode->MSuccessors) {
1089  auto Successor = SubgraphNodesMap.at(NextNode.lock());
1090  NodeCopy->registerSuccessor(Successor, NodeCopy);
1091  }
1092  }
1093 
1094  // Collect input and output nodes for the subgraph
1095  std::vector<std::shared_ptr<node_impl>> Inputs;
1096  std::vector<std::shared_ptr<node_impl>> Outputs;
1097  for (auto &NodeImpl : NewSubgraphNodes) {
1098  if (NodeImpl->MPredecessors.size() == 0) {
1099  Inputs.push_back(NodeImpl);
1100  }
1101  if (NodeImpl->MSuccessors.size() == 0) {
1102  Outputs.push_back(NodeImpl);
1103  }
1104  }
1105 
1106  // Update the predecessors and successors of the nodes which reference the
1107  // original subgraph node
1108 
1109  // Predecessors
1110  for (auto &PredNodeWeak : NewNode->MPredecessors) {
1111  auto PredNode = PredNodeWeak.lock();
1112  auto &Successors = PredNode->MSuccessors;
1113 
1114  // Remove the subgraph node from this nodes successors
1115  Successors.erase(std::remove_if(Successors.begin(), Successors.end(),
1116  [NewNode](auto WeakNode) {
1117  return WeakNode.lock() == NewNode;
1118  }),
1119  Successors.end());
1120 
1121  // Add all input nodes from the subgraph as successors for this node
1122  // instead
1123  for (auto &Input : Inputs) {
1124  PredNode->registerSuccessor(Input, PredNode);
1125  }
1126  }
1127 
1128  // Successors
1129  for (auto &SuccNodeWeak : NewNode->MSuccessors) {
1130  auto SuccNode = SuccNodeWeak.lock();
1131  auto &Predecessors = SuccNode->MPredecessors;
1132 
1133  // Remove the subgraph node from this nodes successors
1134  Predecessors.erase(std::remove_if(Predecessors.begin(),
1135  Predecessors.end(),
1136  [NewNode](auto WeakNode) {
1137  return WeakNode.lock() == NewNode;
1138  }),
1139  Predecessors.end());
1140 
1141  // Add all Output nodes from the subgraph as predecessors for this node
1142  // instead
1143  for (auto &Output : Outputs) {
1144  Output->registerSuccessor(SuccNode, Output);
1145  }
1146  }
1147 
1148  // Remove single subgraph node and add all new individual subgraph nodes
1149  // to the node storage in its place
1150  auto OldPositionIt =
1151  NewNodes.erase(std::find(NewNodes.begin(), NewNodes.end(), NewNode));
1152  // Also set the iterator to the newly added nodes so we can continue
1153  // iterating over all remaining nodes
1154  auto InsertIt = NewNodes.insert(OldPositionIt, NewSubgraphNodes.begin(),
1155  NewSubgraphNodes.end());
1156  // Since the new reverse_iterator will be at i - 1 we need to advance it
1157  // when constructing
1158  NewNodeIt = std::make_reverse_iterator(std::next(InsertIt));
1159  }
1160 
1161  // Store all the new nodes locally
1162  MNodeStorage.insert(MNodeStorage.begin(), NewNodes.begin(), NewNodes.end());
1163 }

1215 void exec_graph_impl::update(
1216  const std::vector<std::shared_ptr<node_impl>> Nodes) {
1217 
1218  if (!MIsUpdatable) {
1219  throw sycl::exception(sycl::make_error_code(errc::invalid),
1220  "update() cannot be called on a executable graph "
1221  "which was not created with property::updatable");
1222  }
1223 
1224  // If there are any accessor requirements, we have to update through the
1225  // scheduler to ensure that any allocations have taken place before trying to
1226  // update.
1227  bool NeedScheduledUpdate = false;
1228  std::vector<sycl::detail::AccessorImplHost *> UpdateRequirements;
1229  // At worst we may have as many requirements as there are for the entire graph
1230  // for updating.
1231  UpdateRequirements.reserve(MRequirements.size());
1232  for (auto &Node : Nodes) {
1233  // Check if node(s) derived from this modifiable node exists in this graph
1234  if (MIDCache.count(Node->getID()) == 0) {
1235  throw sycl::exception(
1236  sycl::make_error_code(errc::invalid),
1237  "Node passed to update() is not part of the graph.");
1238  }
1239  if (Node->MCGType != sycl::detail::CG::Kernel) {
1240  throw sycl::exception(errc::invalid, "Cannot update non-kernel nodes");
1241  }
1242 
1243  if (Node->MCommandGroup->getRequirements().size() == 0) {
1244  continue;
1245  }
1246  NeedScheduledUpdate = true;
1247 
1248  UpdateRequirements.insert(UpdateRequirements.end(),
1249  Node->MCommandGroup->getRequirements().begin(),

1398  }
1399  case access::mode::write:
1400  case access::mode::discard_write: {
1401  MemObjData.mem_access = PI_ACCESS_WRITE_ONLY;
1402  break;
1403  }
1404  default: {
1405  MemObjData.mem_access = PI_ACCESS_READ_WRITE;
1406  break;
1407  }
1408  }
1409  MemObjData.type = PI_KERNEL_ARG_MEM_OBJ_ACCESS;
1410  MemobjDescs.push_back(pi_ext_command_buffer_update_memobj_arg_desc_t{
1411  static_cast<uint32_t>(NodeArg.MIndex), &MemObjData,
1412  static_cast<pi_mem>(Req->MData)});
1413 
1414  } break;
1415 
1416  default:
1417  break;
1418  }
1419  }

1422  UpdateDesc.mem_obj_arg_list = MemobjDescs.data();

1429  UpdateDesc.global_work_size = &NDRDesc.GlobalSize[0];
1430  UpdateDesc.local_work_size = LocalSize;
1431  UpdateDesc.num_work_dim = NDRDesc.Dims;
1432

1436  // associated with a single key, once those node types are supported for
