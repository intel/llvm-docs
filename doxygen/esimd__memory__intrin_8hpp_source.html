45 template <int ElemsPerAddr,
46  typename = sycl::detail::enable_if_t<
47  (ElemsPerAddr == 1 || ElemsPerAddr == 2 || ElemsPerAddr == 4)>>
48 constexpr unsigned int ElemsPerAddrEncoding() {
49  // encoding requires log2 of ElemsPerAddr
50  if constexpr (ElemsPerAddr == 1)
51  return 0;
52  else if constexpr (ElemsPerAddr == 2)
53  return 1;
54  else if constexpr (ElemsPerAddr == 4)
55  return 2;
56 
57  // other cases not needed since enable_if disallows other values
58 }

60 constexpr unsigned int ElemsPerAddrDecoding(unsigned int ElemsPerAddrEncoded) {
61  // encoding requires 2^ElemsPerAddrEncoded
62  return (1 << ElemsPerAddrEncoded);
63 }

133 // Low-level surface-based gather. Collects elements located at given offsets in
134 // a surface and returns them as a single \ref simd object. Element can be
135 // 1, 2 or 4-byte value, but is always returned as a 4-byte value within the
136 // resulting simd object, with upper 2 or 3 bytes undefined.
137 // Template (compile-time constant) parameters:
138 // @tparam Ty - element type; can only be a 4-byte integer or \c float,
139 // @tparam N - the number of elements
140 // @tparam SurfIndAliasTy - "surface index alias" type - internal type in the
141 // accessor used to denote the surface
142 // @tparam TySizeLog2 - Log2 of the number of bytes read per element:
143 // 0 - 1 byte, 1 - 2 bytes, 2 - 4 bytes
144 // @tparam L1H - L1 cache hint

147 // Formal parameters:
148 // @param scale - the scale; must be 0
149 // @param surf_ind - the surface index, taken from the SYCL memory object
150 // @param global_offset - offset added to each individual element's offset to
151 // compute actual memory access offset for that element
152 // @param elem_offsets - per-element offsets

503 template <typename Ty, int N, int NumBlk, sycl::INTEL::gpu::CacheHint L1H,
504  sycl::INTEL::gpu::CacheHint L3H>
505 inline __SIGD::vector_type_t<Ty, N * __SIGD::ElemsPerAddrDecoding(NumBlk)>
506 __esimd_flat_read(__SIGD::vector_type_t<uint64_t, N> addrs, int ElemsPerAddr,
507  __SIGD::vector_type_t<uint16_t, N> pred) {
508  auto NumBlkDecoded = __SIGD::ElemsPerAddrDecoding(NumBlk);
509  __SIGD::vector_type_t<Ty, N * __SIGD::ElemsPerAddrDecoding(NumBlk)> V;
510  ElemsPerAddr = __SIGD::ElemsPerAddrDecoding(ElemsPerAddr);
511 
512  for (int I = 0; I < N; I++) {
513  if (pred[I]) {
514  Ty *Addr = reinterpret_cast<Ty *>(addrs[I]);
515  if (sizeof(Ty) == 2)
516  ElemsPerAddr = ElemsPerAddr / 2;
517  if (sizeof(Ty) <= 2) {
518  for (int J = 0; J < NumBlkDecoded && J < ElemsPerAddr; J++)
519  V[I * NumBlkDecoded + J] = *(Addr + J);
520  } else {
521  for (int J = 0; J < NumBlkDecoded && J < ElemsPerAddr; J++)
522  V[J * N + I] = *(Addr + J);
523  }
524  }
525  }
526  return V;
527 }

529 template <typename Ty, int N, sycl::INTEL::gpu::ChannelMaskType Mask,
530  sycl::INTEL::gpu::CacheHint L1H, sycl::INTEL::gpu::CacheHint L3H>
531 inline __SIGD::vector_type_t<Ty, N * NumChannels(Mask)>
532 __esimd_flat_read4(__SIGD::vector_type_t<uint64_t, N> addrs,
533  __SIGD::vector_type_t<uint16_t, N> pred) {
534  __SIGD::vector_type_t<Ty, N * NumChannels(Mask)> V;
535  unsigned int Next = 0;
536 
537  if constexpr (HasR(Mask)) {
538  for (int I = 0; I < N; I++, Next++) {
539  if (pred[I]) {
540  Ty *Addr = reinterpret_cast<Ty *>(addrs[I]);
541  V[Next] = *Addr;
542  }
543  }
544  }
545 
546  if constexpr (HasG(Mask)) {
547  for (int I = 0; I < N; I++, Next++) {
548  if (pred[I]) {
549  Ty *Addr = reinterpret_cast<Ty *>(addrs[I] + sizeof(Ty));
550  V[Next] = *Addr;
551  }
552  }
553  }
554 
555  if constexpr (HasB(Mask)) {
556  for (int I = 0; I < N; I++, Next++) {
557  if (pred[I]) {
558  Ty *Addr = reinterpret_cast<Ty *>(addrs[I] + sizeof(Ty) + sizeof(Ty));
559  V[Next] = *Addr;
560  }
561  }
562  }
563 
564  if constexpr (HasA(Mask)) {
565  for (int I = 0; I < N; I++, Next++) {
566  if (pred[I]) {
567  Ty *Addr = reinterpret_cast<Ty *>(addrs[I] + sizeof(Ty) + sizeof(Ty) +
568  sizeof(Ty));
569  V[Next] = *Addr;
570  }
571  }
572  }
573 
574  return V;
575 }

672 template <typename Ty, int M, int N, typename TACC>
673 inline __SIGD::vector_type_t<Ty, M * N>
674 __esimd_media_block_load(unsigned modififer, TACC handle, unsigned plane,
675  unsigned width, unsigned x, unsigned y) {
676  // On host the input surface is modeled as sycl image 2d object,
677  // and the read/write access is done through accessor,
678  // which is passed in as the handle argument.
679  auto range = __SIGD::AccessorPrivateProxy::getImageRange(handle);
680  unsigned bpp = __SIGD::AccessorPrivateProxy::getElemSize(handle);
681  unsigned vpp = bpp / sizeof(Ty);
682  unsigned int i = x / bpp;
683  unsigned int j = y;
684 
685  assert(x % bpp == 0);
686  unsigned int xbound = range[0] - 1;
687  unsigned int ybound = range[1] - 1;
688 
689  __SIGD::vector_type_t<Ty, M * N> vals;
690  for (int row = 0; row < M; row++) {
691  for (int col = 0; col < N; col += vpp) {
692  unsigned int xoff = (i > xbound) ? xbound : i;
693  unsigned int yoff = (j > ybound) ? ybound : j;
694  auto coords = cl::sycl::cl_int2(xoff, yoff);
695  cl::sycl::cl_uint4 data = handle.read(coords);
696 
697  __SIGD::vector_type_t<unsigned int, 4> res;
698  for (int idx = 0; idx < 4; idx++) {
699  res[idx] = data[idx];
700  }
701 
702  constexpr int refN = sizeof(cl::sycl::cl_uint4) / sizeof(Ty);
703  unsigned int stride = sizeof(cl::sycl::cl_uint4) / bpp;
704  using refTy = __SIGD::vector_type_t<Ty, refN>;
705  auto ref = reinterpret_cast<refTy>(res);
706 
707  unsigned int offset1 = col + row * N;
708  unsigned int offset2 = 0;
709  for (int idx = 0; idx < vpp; idx++) {
710  vals[offset1] = ref[offset2];
711  offset1++;
712  offset2 += stride;
713  }
714  i++;
715  }
716  i = x / bpp;
717  j++;
718  }
719 
720  return vals;
721 }
