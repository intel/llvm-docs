572  struct arguments {
573  static constexpr size_t MAX_PARAM_BYTES = 4000u;
574  using args_t = std::array<char, MAX_PARAM_BYTES>;
575  using args_size_t = std::vector<size_t>;
576  using args_index_t = std::vector<void *>;
577  args_t storage_;
578  args_size_t paramSizes_;
579  args_index_t indices_;
580  args_size_t offsetPerIndex_;
581 
582  std::uint32_t implicitOffsetArgs_[3] = {0, 0, 0};
583 
584  arguments() {
585  // Place the implicit offset index at the end of the indicies collection
586  indices_.emplace_back(&implicitOffsetArgs_);
587  }
588 
594  void add_arg(size_t index, size_t size, const void *arg,
595  size_t localSize = 0) {
596  if (index + 2 > indices_.size()) {
597  // Move implicit offset argument index with the end
598  indices_.resize(index + 2, indices_.back());
599  // Ensure enough space for the new argument
600  paramSizes_.resize(index + 1);
601  offsetPerIndex_.resize(index + 1);
602  }
603  paramSizes_[index] = size;
604  // calculate the insertion point on the array
605  size_t insertPos = std::accumulate(std::begin(paramSizes_),
606  std::begin(paramSizes_) + index, 0);
607  // Update the stored value for the argument
608  std::memcpy(&storage_[insertPos], arg, size);
609  indices_[index] = &storage_[insertPos];
610  offsetPerIndex_[index] = localSize;
611  }
612 
613  void add_local_arg(size_t index, size_t size) {
614  size_t localOffset = this->get_local_size();
615 
616  // maximum required alignment is the size of the largest vector type
617  const size_t max_alignment = sizeof(double) * 16;
618 
619  // for arguments smaller than the maximum alignment simply align to the
620  // size of the argument
621  const size_t alignment = std::min(max_alignment, size);
622 
623  // align the argument
624  size_t alignedLocalOffset = localOffset;
625  if (localOffset % alignment != 0) {
626  alignedLocalOffset += alignment - (localOffset % alignment);
627  }
628 
629  add_arg(index, sizeof(size_t), (const void *)&(alignedLocalOffset),
630  size + (alignedLocalOffset - localOffset));
631  }
632 
633  void set_implicit_offset(size_t size, std::uint32_t *implicitOffset) {
634  assert(size == sizeof(std::uint32_t) * 3);
635  std::memcpy(implicitOffsetArgs_, implicitOffset, size);
636  }
637 
638  void clear_local_size() {
639  std::fill(std::begin(offsetPerIndex_), std::end(offsetPerIndex_), 0);
640  }
641 
642  args_index_t get_indices() const noexcept { return indices_; }
643 
644  pi_uint32 get_local_size() const {
645  return std::accumulate(std::begin(offsetPerIndex_),
646  std::end(offsetPerIndex_), 0);
647  }
648  } args_;
