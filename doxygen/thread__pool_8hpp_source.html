25 class ThreadPool {
26  std::vector<std::thread> MLaunchedThreads;
27 
28  size_t MThreadCount;
29  std::queue<std::function<void()>> MJobQueue;
30  std::mutex MJobQueueMutex;
31  std::condition_variable MDoSmthOrStop;
32  std::atomic_bool MStop;
33 
34  void worker() {
35  std::unique_lock<std::mutex> Lock(MJobQueueMutex);
36 
37  while (true) {
38  MDoSmthOrStop.wait(
39  Lock, [this]() { return !MJobQueue.empty() || MStop.load(); });
40 
41  if (MStop.load())
42  break;
43 
44  std::function<void()> Job = std::move(MJobQueue.front());
45  MJobQueue.pop();
46  Lock.unlock();
47 
48  Job();
49 
50  Lock.lock();
51  }
52  }
53 
54 public:
55  ThreadPool(unsigned int ThreadCount = 1) : MThreadCount(ThreadCount) {}
56 
57  ~ThreadPool() { finishAndWait(); }
58 
59  void start() {
60  MLaunchedThreads.reserve(MThreadCount);
61 
62  MStop.store(false);
63 
64  for (size_t Idx = 0; Idx < MThreadCount; ++Idx)
65  MLaunchedThreads.emplace_back([this] { worker(); });
66  }
67 
68  void finishAndWait() {
69  MStop.store(true);
70 
71  MDoSmthOrStop.notify_all();
72 
73  for (std::thread &Thread : MLaunchedThreads)
74  if (Thread.joinable())
75  Thread.join();
76  }
77 
78  template <typename T> void submit(T &&Func) {
79  {
80  std::lock_guard<std::mutex> Lock(MJobQueueMutex);
81  MJobQueue.emplace([F = std::move(Func)]() { F(); });
82  }
83 
84  MDoSmthOrStop.notify_one();
85  }
86 
87  void submit(std::function<void()> &&Func) {
88  {
89  std::lock_guard<std::mutex> Lock(MJobQueueMutex);
90  MJobQueue.emplace(Func);
91  }
92 
93  MDoSmthOrStop.notify_one();
94  }
95 };
