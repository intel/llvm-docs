86 HOST_IMPL(bitselect, [](auto x, auto y, auto z) {
87  using T0 = decltype(x);
88  using T1 = decltype(y);
89  using T2 = decltype(z);
90  constexpr size_t N = sizeof(T0) * 8;
91  using bitset = std::bitset<N>;
92 
93  static_assert(std::is_same_v<T0, T1> && std::is_same_v<T1, T2> &&
94  detail::is_scalar_arithmetic_v<T0>);
95 
96  using utype = detail::make_type_t<
97  T0, detail::type_list<unsigned char, unsigned short, unsigned int,
98  unsigned long, unsigned long long>>;
99  static_assert(sizeof(utype) == sizeof(T0));
100  bitset bx(bit_cast<utype>(x)), by(bit_cast<utype>(y)), bz(bit_cast<utype>(z));
101  bitset res = (bz & by) | (~bz & bx);
102  unsigned long long ures = res.to_ullong();
103  assert((ures & std::numeric_limits<utype>::max()) == ures);
104  return bit_cast<T0>(static_cast<utype>(ures));
105 })
