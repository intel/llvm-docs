150 // Helper function for extracting implementation from SYCL's interface objects.
151 // Note! This function relies on the fact that all SYCL interface classes
152 // contain "impl" field that points to implementation object. "impl" field
153 // should be accessible from this function.
154 //
155 // Note that due to a bug in MSVC compilers (including MSVC2019 v19.20), it
156 // may not recognize the usage of this function in friend member declarations
157 // if the template parameter name there is not equal to the name used here,
158 // i.e. 'Obj'. For example, using 'Obj' here and 'T' in such declaration
159 // would trigger that error in MSVC:
160 // template <class T>
161 // friend decltype(T::impl) detail::getSyclObjImpl(const T &SyclObject);
162 template <class Obj> decltype(Obj::impl) getSyclObjImpl(const Obj &SyclObject) {
163  return SyclObject.impl;
164 }

208  NDLoopIterateImpl(const LoopIndexTy<NDIMS> &LowerBound,
209  const LoopBoundTy<NDIMS> &Stride,
210  const LoopBoundTy<NDIMS> &UpperBound, FuncTy f,
211  LoopIndexTy<NDIMS> &Index) {
212  constexpr size_t AdjIdx = NDIMS - 1 - DIM;
