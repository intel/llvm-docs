61 std::string generateSemiUniqueId() {
62  auto Now = std::chrono::high_resolution_clock::now();
63  auto Milliseconds = std::chrono::duration_cast<std::chrono::milliseconds>(
64  Now.time_since_epoch());
65 
66  // Generate random number between 10'000 and 99'900.
67  std::random_device RD;
68  std::mt19937 Gen(RD());
69  std::uniform_int_distribution<int> Distrib(10'000, 99'999);
70  int RandomNumber = Distrib(Gen);
71 
72  // Combine time and random number into a string.
73  std::stringstream Ss;
74  Ss << Milliseconds.count() << "_" << std::setfill('0') << std::setw(5)
75  << RandomNumber;
76 
77  return Ss.str();
78 }

93  }

148  "Failed to open .cpp file for write: " +

193 int invokeCommand(const std::string &command, std::string &output) {
194 #ifdef _WIN32
195  FILE *pipe = _popen(command.c_str(), "r");
196 #else
197  FILE *pipe = popen(command.c_str(), "r");
198 #endif
199  if (!pipe) {
200  return -1;
201  }
202 
203  char buffer[1024];
204  while (!feof(pipe)) {
205  if (fgets(buffer, sizeof(buffer), pipe) != NULL) {
206  output += buffer;
207  }
208  }
209 
210 #ifdef _WIN32
211  _pclose(pipe);
212 #else
213  pclose(pipe);
214 #endif
215 
216  return 0;
217 }

219 std::string invokeCompiler(const std::filesystem::path &FPath,
220  const std::filesystem::path &DPath,
221  const std::string &Id,
222  const std::vector<std::string> &UserArgs,
223  std::string *LogPtr) {
224 
225  std::filesystem::path FilePath(FPath);
226  std::filesystem::path ParentDir(DPath);
227  std::filesystem::path TargetPath = ParentDir / (Id + ".bin");
228  std::filesystem::path LogPath = ParentDir / "compilation_log.txt";
229  std::string Compiler = getCompilerPath().make_preferred().string();
230 
231  std::string Command =
232  Compiler + " -fsycl -o " + TargetPath.make_preferred().string() + " " +
233  userArgsAsString(UserArgs) +
234  " -fno-sycl-dead-args-optimization -fsycl-dump-device-code=" +
235  ParentDir.make_preferred().string() + " " +
236  FilePath.make_preferred().string() + " 2>&1";
237 
238  std::string CompileLog;
239  int Result = invokeCommand(Command, CompileLog);
240 
241  if (LogPtr != nullptr) {
242  LogPtr->append(CompileLog);
243  }
244 
245  // There is little chance of Result being non-zero.
246  // Actual compilation failure is not detected by error code,
247  // but by missing .spv files.
248  if (Result != 0) {
249  throw sycl::exception(sycl::errc::build,
250  "Compile failure: " + std::to_string(Result) + " " +
251  CompileLog);
252  }
253  return CompileLog;
254 }

256 std::filesystem::path findSpv(const std::filesystem::path &ParentDir,
257  const std::string &Id, std::string &CompileLog) {
258  std::regex PatternRegex(Id + R"(.*\.spv)");
259 
260  // Iterate through all files in the directory matching the pattern.
261  for (const auto &Entry : std::filesystem::directory_iterator(ParentDir)) {
262  if (Entry.is_regular_file() &&
263  std::regex_match(Entry.path().filename().string(), PatternRegex)) {
264  return Entry.path(); // Return the path if it matches the SPV pattern.
265  }
266  }
267 
268  // Missing .spv file indicates there was a compilation failure.
269  throw sycl::exception(sycl::errc::build, "Compile failure: " + CompileLog);
270 }

272 spirv_vec_t loadSpvFromFile(const std::filesystem::path &FileName) {
273  std::ifstream SpvStream(FileName, std::ios::binary);
274  SpvStream.seekg(0, std::ios::end);
275  size_t Size = SpvStream.tellg();
276  SpvStream.seekg(0);
277  spirv_vec_t Spv(Size);
278  SpvStream.read(reinterpret_cast<char *>(Spv.data()), Size);
279 
280  return Spv;
281 }

283 spirv_vec_t
284 SYCL_to_SPIRV(const std::string &SYCLSource, include_pairs_t IncludePairs,
285  const std::vector<std::string> &UserArgs, std::string *LogPtr,
286  const std::vector<std::string> &RegisteredKernelNames) {
287  // clang-format off
288  const std::string id = generateSemiUniqueId();
289  const std::filesystem::path ParentDir = prepareWS(id);
290  std::filesystem::path FilePath = outputCpp(ParentDir, id, SYCLSource, UserArgs, RegisteredKernelNames);
291  outputIncludeFiles(ParentDir, IncludePairs);
292  std::string CompileLog = invokeCompiler(FilePath, ParentDir, id, UserArgs, LogPtr);
293  std::filesystem::path SpvPath = findSpv(ParentDir, id, CompileLog);
294  spirv_vec_t Spv = loadSpvFromFile(SpvPath);
295  deleteWS(ParentDir);
296  return Spv;
297  // clang-format on
298 }
