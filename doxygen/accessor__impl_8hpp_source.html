24 // The class describes a requirement to access a SYCL memory object such as
25 // sycl::buffer and sycl::image. For example, each accessor used in a kernel,
26 // except one with access target "local", adds such requirement for the command
27 // group.

29 template <int Dims> class AccessorImplDevice {
30 public:
31  AccessorImplDevice() = default;
32  AccessorImplDevice(id<Dims> Offset, range<Dims> AccessRange,
33  range<Dims> MemoryRange)
34  : Offset(Offset), AccessRange(AccessRange), MemRange(MemoryRange) {}
35 
36  id<Dims> Offset;
37  range<Dims> AccessRange;
38  range<Dims> MemRange;
39 
40  bool operator==(const AccessorImplDevice &Rhs) const {
41  return (Offset == Rhs.Offset && AccessRange == Rhs.AccessRange &&
42  MemRange == Rhs.MemRange);
43  }
44 };
