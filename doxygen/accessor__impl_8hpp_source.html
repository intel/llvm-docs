133 public:
134  LocalAccessorImplHost(sycl::range<3> Size, int Dims, int ElemSize)
135  : MSize(Size), MDims(Dims), MElemSize(ElemSize),
136  MMem(Size[0] * Size[1] * Size[2] * ElemSize) {}

138  sycl::range<3> MSize;
139  int MDims;
140  int MElemSize;
141  std::vector<char> MMem;
142 
143  bool PerWI = false;
144  size_t LocalMemSize;
145  size_t MaxWGSize;
146  void resize(size_t LocalSize, size_t GlobalSize) {
147  if (GlobalSize != 1 && LocalSize != 1) {
148  // If local size is not specified then work group size is chosen by
149  // runtime. That is why try to allocate based on max work group size or
150  // global size. In the worst case allocate 80% of local memory.
151  size_t MinEstWGSize = LocalSize ? LocalSize : GlobalSize;
152  MinEstWGSize = MinEstWGSize > MaxWGSize ? MaxWGSize : MinEstWGSize;
153  size_t NewSize = MinEstWGSize * MSize[0];
154  MSize[0] =
155  NewSize > 8 * LocalMemSize / 10 ? 8 * LocalMemSize / 10 : NewSize;
156  MMem.resize(NewSize * MElemSize);
157  }
158  }
159 };
