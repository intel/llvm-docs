221 // The rdregion intrinsics computes a result vector using following algorithm:
222 //
223 // \code{.cpp}
224 // uint16_t EltOffset = Offset / sizeof(T);
225 // assert(Offset % sizeof(T) == 0);
226 //
227 // int NumRows = M / Width;
228 // assert(M % Width == 0);
229 //
230 // int Index = 0;
231 // for (int i = 0; i < NumRows; ++i) {
232 // for (int j = 0; j < Width; ++j) {
233 // Result[Index++] = Input[i * VStride + j * Stride +
234 // EltOffset];
235 // }
236 // }
237 // \endcode
238 // Hence the checks are to prevent reading beyond the input vector.
239 template <int N, int M, int VStride, int Width, int Stride>
240 constexpr void check_rdregion_params() {
241  static_assert(Width > 0 && M % Width == 0, "Malformed RHS region.");
242  static_assert(Width == M ||
243  ((M / Width) - 1) * VStride + (Width - 1) * Stride < N,
244  "Malformed RHS region - too big vertical and/or "
245  "horizontal stride.");
246 }

255 // The wrregion intrinsics computes a result vector using following algorithm:
256 //
257 // \code{.cpp}
258 // uint16_t EltOffset = Offset / sizeof(T);
259 // assert(Offset % sizeof(T) == 0);
260 //
261 // int NumRows = M / Width;
262 // assert(M % Width == 0);
263 //
264 // Result = OldValue;
265 // int Index = 0;
266 // for (int i = 0; i < NumRows; ++i) {
267 // for (int j = 0; j < Width; ++j) {
268 // if (Mask[Index])
269 // Result[i * VStride + j * Stride + EltOffset] = NewVal[Index];
270 // ++Index;
271 // }
272 // }
273 // \endcode
274 // Hence the checks are to prevent reading beyond the input array and prevent
275 // writing beyond destination vector.
276 template <int N, int M, int VStride, int Width, int Stride>
277 constexpr void check_wrregion_params() {
278  static_assert(M <= N, "Attempt to access beyond viewed area: The "
279  "viewed object in LHS does not fit RHS.");
280  static_assert((M - 1) * Stride < N, "Malformed RHS region - too big stride.");
281  check_rdregion_params<N, M, VStride, Width, Stride>();
282 }
