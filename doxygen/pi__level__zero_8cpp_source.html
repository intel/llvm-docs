184 ze_result_t
185 _pi_context::getFreeSlotInExistingOrNewPool(ze_event_pool_handle_t &ZePool,
186  size_t &Index) {
187  // Maximum number of events that can be present in an event ZePool is captured
188  // here. Setting it to 256 gave best possible performance for several
189  // benchmarks.
190  static const pi_uint32 MaxNumEventsPerPool = [] {
191  const auto MaxNumEventsPerPoolEnv =
192  std::getenv("ZE_MAX_NUMBER_OF_EVENTS_PER_EVENT_POOL");
193  return MaxNumEventsPerPoolEnv ? std::atoi(MaxNumEventsPerPoolEnv) : 256;
194  }();
195 
196  if (MaxNumEventsPerPool == 0) {
197  zePrint("Zero size can't be specified in the "
198  "ZE_MAX_NUMBER_OF_EVENTS_PER_EVENT_POOL\n");
199  return ZE_RESULT_ERROR_INVALID_SIZE;
200  }
201 
202  Index = 0;
203  // Create one event ZePool per MaxNumEventsPerPool events
204  if ((ZeEventPool == nullptr) ||
205  (NumEventsAvailableInEventPool[ZeEventPool] == 0)) {
206  // Creation of the new ZePool with record in NumEventsAvailableInEventPool
207  // and initialization of the record in NumEventsLiveInEventPool must be done
208  // atomically. Otherwise it is possible that decrementAliveEventsInPool will
209  // be called for the record in NumEventsLiveInEventPool before its
210  std::lock(NumEventsAvailableInEventPoolMutex,
211  NumEventsLiveInEventPoolMutex);
212  std::lock_guard<std::mutex> NumEventsAvailableInEventPoolGuard(
213  NumEventsAvailableInEventPoolMutex, std::adopt_lock);
214  std::lock_guard<std::mutex> NumEventsLiveInEventPoolGuard(
215  NumEventsLiveInEventPoolMutex, std::adopt_lock);
216 
217  ze_event_pool_desc_t ZeEventPoolDesc;
218  ZeEventPoolDesc.count = MaxNumEventsPerPool;
219  ZeEventPoolDesc.flags = ZE_EVENT_POOL_FLAG_TIMESTAMP;
220  ZeEventPoolDesc.version = ZE_EVENT_POOL_DESC_VERSION_CURRENT;
221 
222  ze_device_handle_t ZeDevice = Device->ZeDevice;
223  if (ze_result_t ZeRes =
224  zeEventPoolCreate(Device->Platform->ZeDriver, &ZeEventPoolDesc, 1,
225  &ZeDevice, &ZeEventPool))
226  return ZeRes;
227  NumEventsAvailableInEventPool[ZeEventPool] = MaxNumEventsPerPool - 1;
228  NumEventsLiveInEventPool[ZeEventPool] = MaxNumEventsPerPool;
229  } else {
230  std::lock_guard<std::mutex> NumEventsAvailableInEventPoolGuard(
231  NumEventsAvailableInEventPoolMutex);
232  Index = MaxNumEventsPerPool - NumEventsAvailableInEventPool[ZeEventPool];
233  --NumEventsAvailableInEventPool[ZeEventPool];
234  }
235  ZePool = ZeEventPool;
236  return ZE_RESULT_SUCCESS;
237 }

470 pi_result piPlatformsGet(pi_uint32 NumEntries, pi_platform *Platforms,
471  pi_uint32 *NumPlatforms) {
472 
473  static const char *DebugMode = std::getenv("ZE_DEBUG");
474  if (DebugMode)
475  ZeDebug = true;
476 
477  static const char *SerializeMode = std::getenv("ZE_SERIALIZE");
478  static const pi_uint32 SerializeModeValue =
479  SerializeMode ? std::atoi(SerializeMode) : 0;
480  ZeSerialize = SerializeModeValue;
481 
482  if (NumEntries == 0 && Platforms != nullptr) {
483  return PI_INVALID_VALUE;
484  }
485  if (Platforms == nullptr && NumPlatforms == nullptr) {
486  return PI_INVALID_VALUE;
487  }
488 
489  // Cache pi_platforms for reuse in the future
490  // It solves two problems;
491  // 1. sycl::device equality issue; we always return the same pi_device.
492  // 2. performance; we can save time by immediately return from cache.
493  static std::vector<pi_platform> PiPlatformsCache;
494  static std::mutex PiPlatformsCacheMutex;
495 
496  // This is a good time to initialize Level Zero.
497  // TODO: We can still safely recover if something goes wrong during the init.
498  // Implement handling segfault using sigaction.
499  // TODO: We should not call zeInit multiples times ever, so
500  // this code should be changed.
501  ze_result_t ZeResult = ZE_CALL_NOCHECK(zeInit(ZE_INIT_FLAG_NONE));
502 
503  // Absorb the ZE_RESULT_ERROR_UNINITIALIZED and just return 0 Platforms.
504  if (ZeResult == ZE_RESULT_ERROR_UNINITIALIZED) {
505  assert(NumPlatforms != 0);
506  *NumPlatforms = 0;
507  return PI_SUCCESS;
508  }
509 
510  if (ZeResult != ZE_RESULT_SUCCESS) {
511  zePrint("zeInit: Level Zero initialization failure\n");
512  return mapError(ZeResult);
513  }
514 
515  // Level Zero does not have concept of Platforms, but Level Zero driver is the
516  // closest match.
517  if (Platforms && NumEntries > 0) {
518  uint32_t ZeDriverCount = 0;
519  ZE_CALL(zeDriverGet(&ZeDriverCount, nullptr));
520  if (ZeDriverCount == 0) {
521  assert(NumPlatforms != 0);
522  *NumPlatforms = 0;
523  return PI_SUCCESS;
524  }
525 
526  ze_driver_handle_t ZeDriver;
527  assert(ZeDriverCount == 1);
528  ZE_CALL(zeDriverGet(&ZeDriverCount, &ZeDriver));
529 
530  std::lock_guard<std::mutex> Lock(PiPlatformsCacheMutex);
531  for (const pi_platform CachedPlatform : PiPlatformsCache) {
532  if (CachedPlatform->ZeDriver == ZeDriver) {
533  Platforms[0] = CachedPlatform;
534  // if the caller sent a valid NumPlatforms pointer, set it here
535  if (NumPlatforms)
536  *NumPlatforms = 1;
537 
538  return PI_SUCCESS;
539  }
540  }
541 
542  try {
543  // TODO: figure out how/when to release this memory
544  *Platforms = new _pi_platform(ZeDriver);
545 
546  // Cache driver properties
547  ze_driver_properties_t ZeDriverProperties;
548  ZE_CALL(zeDriverGetProperties(ZeDriver, &ZeDriverProperties));
549  uint32_t ZeDriverVersion = ZeDriverProperties.driverVersion;
550  // Intel Level-Zero GPU driver stores version as:
551  // | 31 - 24 | 23 - 16 | 15 - 0 |
552  // | Major | Minor | Build |
553  std::string VersionMajor =
554  std::to_string((ZeDriverVersion & 0xFF000000) >> 24);
555  std::string VersionMinor =
556  std::to_string((ZeDriverVersion & 0x00FF0000) >> 16);
557  std::string VersionBuild = std::to_string(ZeDriverVersion & 0x0000FFFF);
558  Platforms[0]->ZeDriverVersion = VersionMajor + std::string(".") +
559  VersionMinor + std::string(".") +
560  VersionBuild;
561 
562  ze_api_version_t ZeApiVersion;
563  ZE_CALL(zeDriverGetApiVersion(ZeDriver, &ZeApiVersion));
564  Platforms[0]->ZeDriverApiVersion =
565  std::to_string(ZE_MAJOR_VERSION(ZeApiVersion)) + std::string(".") +
566  std::to_string(ZE_MINOR_VERSION(ZeApiVersion));
567 
568  // save a copy in the cache for future uses
569  PiPlatformsCache.push_back(Platforms[0]);
570  } catch (const std::bad_alloc &) {
571  return PI_OUT_OF_HOST_MEMORY;
572  } catch (...) {
573  return PI_ERROR_UNKNOWN;
574  }
575  }
576 
577  if (NumPlatforms)
578  *NumPlatforms = 1;
579 
580  return PI_SUCCESS;

654 pi_result piDevicesGet(pi_platform Platform, pi_device_type DeviceType,
655  pi_uint32 NumEntries, pi_device *Devices,
656  pi_uint32 *NumDevices) {
657 
658  assert(Platform);
659  ze_driver_handle_t ZeDriver = Platform->ZeDriver;
660 
661  // Get number of devices supporting Level Zero
662  uint32_t ZeDeviceCount = 0;
663  std::lock_guard<std::mutex> Lock(Platform->PiDevicesCacheMutex);
664  ZeDeviceCount = Platform->PiDevicesCache.size();
665 
666  const bool AskingForGPU = (DeviceType & PI_DEVICE_TYPE_GPU);
667  const bool AskingForDefault = (DeviceType == PI_DEVICE_TYPE_DEFAULT);
668 
669  if (ZeDeviceCount == 0) {
670  ZE_CALL(zeDeviceGet(ZeDriver, &ZeDeviceCount, nullptr));
671  }
672 
673  if (ZeDeviceCount == 0 || !(AskingForGPU || AskingForDefault)) {
674  if (NumDevices)
675  *NumDevices = 0;
676  return PI_SUCCESS;
677  }
678 
679  if (NumDevices)
680  *NumDevices = ZeDeviceCount;
681 
682  if (NumEntries == 0) {
683  assert(Devices == nullptr &&
684  "Devices should be nullptr when querying the number of devices");
685  return PI_SUCCESS;
686  }
687 
688  // if devices are already captured in cache, return them from the cache.
689  for (const pi_device CachedDevice : Platform->PiDevicesCache) {
690  *Devices++ = CachedDevice;
691  }
692  if (!Platform->PiDevicesCache.empty()) {
693  return PI_SUCCESS;
694  }
695 
696  try {
697  std::vector<ze_device_handle_t> ZeDevices(ZeDeviceCount);
698  ZE_CALL(zeDeviceGet(ZeDriver, &ZeDeviceCount, ZeDevices.data()));
699 
700  for (uint32_t I = 0; I < ZeDeviceCount; ++I) {
701  if (I < NumEntries) {
702  Devices[I] = new _pi_device(ZeDevices[I], Platform);
703  pi_result Result = Devices[I]->initialize();
704  if (Result != PI_SUCCESS) {
705  return Result;
706  }
707  // save a copy in the cache for future uses.
708  Platform->PiDevicesCache.push_back(Devices[I]);
709  }
710  }
711  } catch (const std::bad_alloc &) {
712  return PI_OUT_OF_HOST_MEMORY;
713  } catch (...) {
714  return PI_ERROR_UNKNOWN;
715  }
716  return PI_SUCCESS;
717 }

1957 pi_result piProgramLink(pi_context Context, pi_uint32 NumDevices,
1958  const pi_device *DeviceList, const char *Options,
1959  pi_uint32 NumInputPrograms,
1960  const pi_program *InputPrograms,
1961  void (*PFnNotify)(pi_program Program, void *UserData),
1962  void *UserData, pi_program *RetProgram) {
1963

1965  assert(NumDevices == 1);
1966  assert(DeviceList && DeviceList[0] == Context->Device);
1967  assert(!PFnNotify && !UserData);
1968

1970  if (NumInputPrograms == 0 || InputPrograms == nullptr)
1971  return PI_INVALID_VALUE;
1972  for (pi_uint32 I = 0; I < NumInputPrograms; I++) {
1973  if (InputPrograms[I]->State != _pi_program::Object) {
1974  return PI_INVALID_OPERATION;
1975  }
1976  assert(InputPrograms[I]->ZeModule);
1977  }
1978 
1979  // Linking modules on Level Zero is different from OpenCL. With Level Zero,
1980  // each input object module already has native code loaded onto the device.
1981  // Linking two modules together causes the importing module to be changed
1982  // such that its native code points to an address in the exporting module.
1983  // As a result, a module that imports symbols can only be linked into one
1984  // executable at a time. By contrast, modules that export symbols are not
1985  // changed, so they can be safely linked into multiple executables
1986  // simultaneously.
1987  //
1988  // Level Zero linking also differs from OpenCL because a link operation does
1989  // not create a new module that represents the linked executable. Instead,
1990  // we must keep track of all the input modules and refer to the entire list
1991  // whenever we want to know something about the executable.
1992 
1993  // This vector hold the Level Zero modules that we will actually link
1994  // together. This may be different from "InputPrograms" because some of
1995  // those modules may import symbols and already be linked into other
1996  // executables. In such a case, we must make a copy of the module before we
1997  // can link it again.
1998  std::vector<_pi_program::LinkedReleaser> Inputs;
1999  try {
2000  Inputs.reserve(NumInputPrograms);
2001 
2002  // We do several things in this loop.
2003  //
2004  // 1. We identify any modules that need to be copied because they import
2005  // symbols and are already linked into some other program.
2006  // 2. For any module that does not need to be copied, we bump its reference
2007  // count because we will hold a reference to it.
2008  // 3. We create a vector of Level Zero modules, which we can pass to the
2009  // zeModuleDynamicLink() API.
2010  std::vector<ze_module_handle_t> ZeHandles;
2011  ZeHandles.reserve(NumInputPrograms);
2012  for (pi_uint32 I = 0; I < NumInputPrograms; I++) {
2013  pi_program Input = InputPrograms[I];
2014  if (Input->HasImports) {
2015  std::unique_lock<std::mutex> Guard(Input->MutexHasImportsAndIsLinked);
2016  if (!Input->HasImportsAndIsLinked) {
2017  // This module imports symbols, but it isn't currently linked with
2018  // any other module. Grab the flag to indicate that it is now
2019  // linked.
2020  piProgramRetain(Input);
2021  Input->HasImportsAndIsLinked = true;
2022  } else {
2023  // This module imports symbols and is also linked with another module
2024  // already, so it needs to be copied. We expect this to be quite
2025  // rare since linking is mostly used to link against libraries which
2026  // only export symbols.
2027  Guard.unlock();
2028  ze_module_handle_t ZeModule;
2029  pi_result res =
2030  copyModule(Context->Device->ZeDevice, Input->ZeModule, &ZeModule);
2031  if (res != PI_SUCCESS) {
2032  return res;
2033  }
2034  Input = new _pi_program(Input->Context, ZeModule, _pi_program::Object,
2035  Input->HasImports);
2036  Input->HasImportsAndIsLinked = true;
2037  }
2038  } else {
2039  piProgramRetain(Input);
2040  }
2041  Inputs.emplace_back(Input);
2042  ZeHandles.push_back(Input->ZeModule);
2043  }
2044 
2045  // Link all the modules together. If this fails (or if we catch an
2046  // exception below), we need to release the reference counts on the input
2047  // modules, delete any copies, etc.
2048  ze_module_build_log_handle_t ZeBuildLog;
2049  ZE_CALL(
2050  zeModuleDynamicLink(ZeHandles.size(), ZeHandles.data(), &ZeBuildLog));
2051 
2052  // Construct a new program object to represent the linked executable. This
2053  // new object holds a reference to all the input programs.
2054  *RetProgram = new _pi_program(Context, std::move(Inputs), ZeBuildLog);
2055  } catch (const std::bad_alloc &) {
2056  return PI_OUT_OF_HOST_MEMORY;
2057  } catch (...) {
2058  return PI_ERROR_UNKNOWN;
2059  }
2060  return PI_SUCCESS;
2061 }
