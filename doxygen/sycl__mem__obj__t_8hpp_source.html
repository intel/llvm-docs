158  template <typename Destination>
159  EnableIfOutputIteratorT<Destination> set_final_data(Destination FinalData) {
160  MUploadDataFunctor = [this, FinalData]() {
161  using DestinationValueT = iterator_value_type_t<Destination>;
162  // TODO if Destination is ContiguousIterator then don't create
163  // ContiguousStorage. updateHostMemory works only with pointer to
164  // continuous data.
165  const size_t Size = MSizeInBytes / sizeof(DestinationValueT);
166  vector_class<DestinationValueT> ContiguousStorage(Size);
167  updateHostMemory(ContiguousStorage.data());
168  std::copy(ContiguousStorage.cbegin(), ContiguousStorage.cend(),
169  FinalData);
170  };

193  void handleHostData(void *HostPtr, const size_t RequiredAlign) {
194  if (!MHostPtrReadOnly)
195  set_final_data(reinterpret_cast<char *>(HostPtr));
196 
197  if (canReuseHostPtr(HostPtr, RequiredAlign)) {
198  MUserPtr = HostPtr;
199  } else {
200  setAlign(RequiredAlign);
201  MShadowCopy = allocateHostMem();
202  MUserPtr = MShadowCopy;
203  std::memcpy(MUserPtr, HostPtr, MSizeInBytes);
204  }
205  }

210  }

212  template <typename T>
213  void handleHostData(const shared_ptr_class<T> &HostPtr,
214  const size_t RequiredAlign) {
215  MSharedPtrStorage = HostPtr;
216  MHostPtrReadOnly = std::is_const<T>::value;
217  if (HostPtr) {
218  if (!MHostPtrReadOnly)
219  set_final_data_from_storage();
220 
221  if (canReuseHostPtr(HostPtr.get(), RequiredAlign))
222  MUserPtr = HostPtr.get();
223  else {
224  setAlign(RequiredAlign);
225  MShadowCopy = allocateHostMem();
226  MUserPtr = MShadowCopy;
227  std::memcpy(MUserPtr, HostPtr.get(), MSizeInBytes);
228  }
229  }

232  template <class InputIterator>
233  void handleHostData(InputIterator First, InputIterator Last,
234  const size_t RequiredAlign) {
235  MHostPtrReadOnly = iterator_to_const_type_t<InputIterator>::value;
236  setAlign(RequiredAlign);
237  if (useHostPtr())
238  throw runtime_error(
239  "Buffer constructor from a pair of iterator values does not support "
240  "use_host_ptr property.",
241  PI_INVALID_OPERATION);
242 
243  setAlign(RequiredAlign);
244  MShadowCopy = allocateHostMem();
245  MUserPtr = MShadowCopy;
246 
247  // We need to cast MUserPtr to pointer to the iteration type to get correct
248  // offset in std::copy when it will increment destination pointer.
249  using IteratorValueType = iterator_value_type_t<InputIterator>;
250  using IteratorNonConstValueType = remove_const_t<IteratorValueType>;
251  using IteratorPointerToNonConstValueType =
252  add_pointer_t<IteratorNonConstValueType>;
253  std::copy(First, Last,
254  static_cast<IteratorPointerToNonConstValueType>(MUserPtr));
255  }
