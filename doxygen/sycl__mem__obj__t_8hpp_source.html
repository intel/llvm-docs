177  template <typename Destination>

188  template <typename Destination>
189  __SYCL_DLL_LOCAL EnableIfOutputIteratorT<Destination>
190  set_final_data(Destination FinalData) {
191  MUploadDataFunctor = [this, FinalData]() {
192  using DestinationValueT = iterator_value_type_t<Destination>;
193  // TODO if Destination is ContiguousIterator then don't create
194  // ContiguousStorage. updateHostMemory works only with pointer to
195  // continuous data.
196  const size_t Size = MSizeInBytes / sizeof(DestinationValueT);
197  std::unique_ptr<DestinationValueT[]> ContiguousStorage(
198  new DestinationValueT[Size]);
199  updateHostMemory(ContiguousStorage.get());
200  std::copy(ContiguousStorage.get(), ContiguousStorage.get() + Size,
201  FinalData);
202  };
203  }

228  __SYCL_DLL_LOCAL void handleHostData(void *HostPtr,
229  const size_t RequiredAlign) {
230  if (!MHostPtrReadOnly)
231  set_final_data(reinterpret_cast<char *>(HostPtr));
232 
233  if (canReuseHostPtr(HostPtr, RequiredAlign)) {
234  MUserPtr = HostPtr;
235  } else {
236  setAlign(RequiredAlign);
237  MShadowCopy = allocateHostMem();
238  MUserPtr = MShadowCopy;
239  std::memcpy(MUserPtr, HostPtr, MSizeInBytes);
240  }
241  }

247  }

249  template <typename T>
250  __SYCL_DLL_LOCAL void handleHostData(const std::shared_ptr<T> &HostPtr,
251  const size_t RequiredAlign) {
252  MSharedPtrStorage = HostPtr;
253  MHostPtrReadOnly = std::is_const<T>::value;
254  if (HostPtr) {
255  if (!MHostPtrReadOnly)
256  set_final_data_from_storage();
257 
258  if (canReuseHostPtr(HostPtr.get(), RequiredAlign))
259  MUserPtr = HostPtr.get();
260  else {
261  setAlign(RequiredAlign);
262  MShadowCopy = allocateHostMem();
263  MUserPtr = MShadowCopy;
264  std::memcpy(MUserPtr, HostPtr.get(), MSizeInBytes);
265  }
266  }
267  }

269  template <class InputIterator>
270  __SYCL_DLL_LOCAL void handleHostData(InputIterator First, InputIterator Last,
271  const size_t RequiredAlign) {
272  MHostPtrReadOnly = iterator_to_const_type_t<InputIterator>::value;
273  setAlign(RequiredAlign);
274  if (useHostPtr())
275  throw runtime_error(
276  "Buffer constructor from a pair of iterator values does not support "
277  "use_host_ptr property.",
278  PI_INVALID_OPERATION);
279 
280  setAlign(RequiredAlign);
281  MShadowCopy = allocateHostMem();
282  MUserPtr = MShadowCopy;
283 
284  // We need to cast MUserPtr to pointer to the iteration type to get correct
285  // offset in std::copy when it will increment destination pointer.
286  using IteratorValueType = iterator_value_type_t<InputIterator>;
287  using IteratorNonConstValueType = remove_const_t<IteratorValueType>;
288  using IteratorPointerToNonConstValueType =
289  add_pointer_t<IteratorNonConstValueType>;
290  std::copy(First, Last,
291  static_cast<IteratorPointerToNonConstValueType>(MUserPtr));
292  }
