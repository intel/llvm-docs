184  getOrInsertProgram(const ProgramCacheKeyT &CacheKey) {
185  auto LockedCache = acquireCachedPrograms();
186  auto &ProgCache = LockedCache.get();
187  auto [It, DidInsert] = ProgCache.Cache.try_emplace(CacheKey, nullptr);
188  if (DidInsert) {
189  It->second = std::make_shared<ProgramBuildResult>(getPlugin());
190  // Save reference between the common key and the full key.
191  CommonProgramKeyT CommonKey =
192  std::make_pair(CacheKey.first.second, CacheKey.second);
193  ProgCache.KeyMap.emplace(CommonKey, CacheKey);
194  }
195  return std::make_pair(It->second, DidInsert);

283  template <errc Errc, typename GetCachedBuildFT, typename BuildFT>
284  auto getOrBuild(GetCachedBuildFT &&GetCachedBuild, BuildFT &&Build) {
285  using BuildState = KernelProgramCache::BuildState;
286  constexpr size_t MaxAttempts = 2;
287  for (size_t AttemptCounter = 0;; ++AttemptCounter) {
288  auto Res = GetCachedBuild();
289  auto &BuildResult = Res.first;
290  BuildState Expected = BuildState::BS_Initial;
291  BuildState Desired = BuildState::BS_InProgress;
292  if (!BuildResult->State.compare_exchange_strong(Expected, Desired)) {
293  // no insertion took place, thus some other thread has already inserted
294  // smth in the cache
295  BuildState NewState = BuildResult->waitUntilTransition();
296 
297  // Build succeeded.
298  if (NewState == BuildState::BS_Done)
299  return BuildResult;
300 
301  // Build failed, or this is the last attempt.
302  if (NewState == BuildState::BS_Failed ||
303  AttemptCounter + 1 == MaxAttempts) {
304  if (BuildResult->Error.isFilledIn())
305  throw detail::set_ur_error(
306  exception(make_error_code(Errc), BuildResult->Error.Msg),
307  BuildResult->Error.Code);
308  else
309  throw exception();
310  }
311 
312  // NewState == BuildState::BS_Initial
313  // Build state was set back to the initial state,
314  // which means to go back to the beginning of the
315  // loop and try again.
316  continue;
317  }
318 
319  // only the building thread will run this
320  try {
321  BuildResult->Val = Build();
322 
323  BuildResult->updateAndNotify(BuildState::BS_Done);
324  return BuildResult;
325  } catch (const exception &Ex) {
326  BuildResult->Error.Msg = Ex.what();
327  BuildResult->Error.Code = detail::get_ur_error(Ex);
328  if (Ex.code() == errc::memory_allocation ||
329  BuildResult->Error.Code == UR_RESULT_ERROR_OUT_OF_RESOURCES ||
330  BuildResult->Error.Code == UR_RESULT_ERROR_OUT_OF_HOST_MEMORY) {
331  reset();
332  BuildResult->updateAndNotify(BuildState::BS_Initial);
333  continue;
334  }
335 
336  BuildResult->updateAndNotify(BuildState::BS_Failed);
337  std::rethrow_exception(std::current_exception());
338  } catch (...) {
339  BuildResult->updateAndNotify(BuildState::BS_Initial);
340  std::rethrow_exception(std::current_exception());
341  }
342  }
343  }
