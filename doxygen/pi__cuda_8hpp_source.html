559  struct arguments {
560  static constexpr size_t MAX_PARAM_BYTES = 4000u;
561  using args_t = std::array<char, MAX_PARAM_BYTES>;
562  using args_size_t = std::vector<size_t>;
563  using args_index_t = std::vector<void *>;
564  args_t storage_;
565  args_size_t paramSizes_;
566  args_index_t indices_;
567  args_size_t offsetPerIndex_;
568 
569  std::uint32_t implicitOffsetArgs_[3] = {0, 0, 0};
570 
571  arguments() {
572  // Place the implicit offset index at the end of the indicies collection
573  indices_.emplace_back(&implicitOffsetArgs_);
574  }
575 
581  void add_arg(size_t index, size_t size, const void *arg,
582  size_t localSize = 0) {
583  if (index + 2 > indices_.size()) {
584  // Move implicit offset argument index with the end
585  indices_.resize(index + 2, indices_.back());
586  // Ensure enough space for the new argument
587  paramSizes_.resize(index + 1);
588  offsetPerIndex_.resize(index + 1);
589  }
590  paramSizes_[index] = size;
591  // calculate the insertion point on the array
592  size_t insertPos = std::accumulate(std::begin(paramSizes_),
593  std::begin(paramSizes_) + index, 0);
594  // Update the stored value for the argument
595  std::memcpy(&storage_[insertPos], arg, size);
596  indices_[index] = &storage_[insertPos];
597  offsetPerIndex_[index] = localSize;
598  }
599 
600  void add_local_arg(size_t index, size_t size) {
601  size_t localOffset = this->get_local_size();
602  add_arg(index, sizeof(size_t), (const void *)&(localOffset), size);
603  }
604 
605  void set_implicit_offset(size_t size, std::uint32_t *implicitOffset) {
606  assert(size == sizeof(std::uint32_t) * 3);
607  std::memcpy(implicitOffsetArgs_, implicitOffset, size);
608  }
609 
610  void clear_local_size() {
611  std::fill(std::begin(offsetPerIndex_), std::end(offsetPerIndex_), 0);
612  }
613 
614  args_index_t get_indices() const noexcept { return indices_; }
615 
616  pi_uint32 get_local_size() const {
617  return std::accumulate(std::begin(offsetPerIndex_),
618  std::end(offsetPerIndex_), 0);
619  }
620  } args_;
