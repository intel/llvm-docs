474  struct arguments {
475  static constexpr size_t MAX_PARAM_BYTES = 4000u;
476  using args_t = std::array<char, MAX_PARAM_BYTES>;
477  using args_size_t = std::vector<size_t>;
478  using args_index_t = std::vector<void *>;
479  args_t storage_;
480  args_size_t paramSizes_;
481  args_index_t indices_;
482  args_size_t offsetPerIndex_;
483 
488  void add_arg(size_t index, size_t size, const void *arg,
489  size_t localSize = 0) {
490  if (index + 1 > indices_.size()) {
491  indices_.resize(index + 1);
492  // Ensure enough space for the new argument
493  paramSizes_.resize(index + 1);
494  offsetPerIndex_.resize(index + 1);
495  }
496  paramSizes_[index] = size;
497  // calculate the insertion point on the array
498  size_t insertPos = std::accumulate(std::begin(paramSizes_),
499  std::begin(paramSizes_) + index, 0);
500  // Update the stored value for the argument
501  std::memcpy(&storage_[insertPos], arg, size);
502  indices_[index] = &storage_[insertPos];
503  offsetPerIndex_[index] = localSize;
504  }
505 
506  void add_local_arg(size_t index, size_t size) {
507  size_t localOffset = this->get_local_size();
508  add_arg(index, sizeof(size_t), (const void *)&(localOffset), size);
509  }
510 
511  void clear_local_size() {
512  std::fill(std::begin(offsetPerIndex_), std::end(offsetPerIndex_), 0);
513  }
514 
515  args_index_t get_indices() const noexcept { return indices_; }
516 
517  pi_uint32 get_local_size() const {
518  return std::accumulate(std::begin(offsetPerIndex_),
519  std::end(offsetPerIndex_), 0);
520  }
521  } args_;
522
