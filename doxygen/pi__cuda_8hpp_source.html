611  arguments() {
612  // Place the implicit offset index at the end of the indicies collection
613  indices_.emplace_back(&implicitOffsetArgs_);
614  }
615 
621  void add_arg(size_t index, size_t size, const void *arg,
622  size_t localSize = 0) {
623  if (index + 2 > indices_.size()) {
624  // Move implicit offset argument index with the end
625  indices_.resize(index + 2, indices_.back());
626  // Ensure enough space for the new argument
627  paramSizes_.resize(index + 1);
628  offsetPerIndex_.resize(index + 1);
629  }
630  paramSizes_[index] = size;
631  // calculate the insertion point on the array
632  size_t insertPos = std::accumulate(std::begin(paramSizes_),
633  std::begin(paramSizes_) + index, 0);
634  // Update the stored value for the argument
635  std::memcpy(&storage_[insertPos], arg, size);
636  indices_[index] = &storage_[insertPos];
637  offsetPerIndex_[index] = localSize;
638  }
639 
640  void add_local_arg(size_t index, size_t size) {
641  size_t localOffset = this->get_local_size();
642 
643  // maximum required alignment is the size of the largest vector type
644  const size_t max_alignment = sizeof(double) * 16;
645 
646  // for arguments smaller than the maximum alignment simply align to the
647  // size of the argument
648  const size_t alignment = std::min(max_alignment, size);
649 
650  // align the argument
651  size_t alignedLocalOffset = localOffset;
652  if (localOffset % alignment != 0) {
653  alignedLocalOffset += alignment - (localOffset % alignment);
654  }
655 
656  add_arg(index, sizeof(size_t), (const void *)&(alignedLocalOffset),
657  size + (alignedLocalOffset - localOffset));
658  }
659 
660  void set_implicit_offset(size_t size, std::uint32_t *implicitOffset) {
661  assert(size == sizeof(std::uint32_t) * 3);
662  std::memcpy(implicitOffsetArgs_, implicitOffset, size);
663  }
664 
665  void clear_local_size() {
666  std::fill(std::begin(offsetPerIndex_), std::end(offsetPerIndex_), 0);
667  }
668 
669  const args_index_t &get_indices() const noexcept { return indices_; }
670 
671  pi_uint32 get_local_size() const {
672  return std::accumulate(std::begin(offsetPerIndex_),
673  std::end(offsetPerIndex_), 0);
674  }
