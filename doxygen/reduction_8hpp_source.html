455 template <typename KernelName, typename KernelType, int Dims, class Reduction>
456 enable_if_t<Reduction::has_fast_reduce && Reduction::has_fast_atomics>
457 reduCGFunc(handler &CGH, KernelType KernelFunc, const nd_range<Dims> &Range,
458  Reduction &Redu, typename Reduction::rw_accessor_type &Out) {
459 
460  size_t NWorkItems = Range.get_global_range().size();
461  size_t WGSize = Range.get_local_range().size();
462  size_t NWorkGroups = Range.get_group_range().size();
463 
464  if (NWorkGroups * WGSize == NWorkItems) {
465  // Efficient case: all work-groups are uniform.
466  CGH.parallel_for<KernelName>(Range, [=](nd_item<Dims> NDIt) {
467  // Call user's function. Reducer.MValue gets initialized there.
468  typename Reduction::reducer_type Reducer;
469  KernelFunc(NDIt, Reducer);
470 
471  typename Reduction::binary_operation BOp;
472  Reducer.MValue = intel::reduce(NDIt.get_group(), Reducer.MValue, BOp);
473  if (NDIt.get_local_linear_id() == 0)
474  Reducer.atomic_combine(Out.get_pointer().get());
475  });
476  } else {
477  // Inefficient case: non-uniform work groups.
478  using AuxName =
479  typename get_reduction_main_2nd_kernel_name_t<KernelName,
480  KernelType>::name;
481  CGH.parallel_for<AuxName>(Range, [=](nd_item<Dims> NDIt) {
482  // Call user's functions. Reducer.MValue gets initialized there.
483  typename Reduction::reducer_type Reducer;
484  KernelFunc(NDIt, Reducer);
485 
486  typename Reduction::binary_operation BOp;
487  size_t GID = NDIt.get_global_linear_id();
488  auto Val = (GID < NWorkItems) ? Reducer.MValue : Reducer.getIdentity();
489  Reducer.MValue = intel::reduce(NDIt.get_group(), Val, BOp);
490  if (NDIt.get_local_linear_id() == 0)
491  Reducer.atomic_combine(Out.get_pointer().get());
492  });
493  }
494 }

504 template <typename KernelName, typename KernelType, int Dims, class Reduction>
505 enable_if_t<!Reduction::has_fast_reduce && Reduction::has_fast_atomics>
506 reduCGFunc(handler &CGH, KernelType KernelFunc, const nd_range<Dims> &Range,
507  Reduction &Redu, typename Reduction::rw_accessor_type &Out) {
508 
509  size_t NWorkItems = Range.get_global_range().size();
510  size_t WGSize = Range.get_local_range().size();
511  size_t NWorkGroups = Range.get_group_range().size();
512 
513  // Use local memory to reduce elements in work-groups into zero-th element.
514  // If WGSize is not power of two, then WGSize+1 elements are allocated.
515  // The additional last element is used to catch reduce elements that could
516  // otherwise be lost in the tree-reduction algorithm used in the kernel.
517  bool HasNonUniformWG = (NWorkGroups * WGSize - NWorkItems) != 0;
518  bool IsEfficientCase = !HasNonUniformWG && (WGSize & (WGSize - 1)) == 0;
519  size_t NLocalElements = WGSize + (IsEfficientCase ? 0 : 1);
520  auto LocalReds = Redu.getReadWriteLocalAcc(NLocalElements, CGH);
521 
522  if (IsEfficientCase) {
523  // Efficient case: uniform work-groups and WGSize is power of two.
524  CGH.parallel_for<KernelName>(Range, [=](nd_item<Dims> NDIt) {
525  // Call user's functions. Reducer.MValue gets initialized there.
526  typename Reduction::reducer_type Reducer;
527  KernelFunc(NDIt, Reducer);
528 
529  size_t LID = NDIt.get_local_linear_id();
530  // Copy the element to local memory to prepare it for tree-reduction.
531  LocalReds[LID] = Reducer.MValue;
532  NDIt.barrier();
533 
534  // Tree-reduction: reduce the local array LocalReds[:] to LocalReds[0].
535  typename Reduction::binary_operation BOp;
536  size_t WGSize = NDIt.get_local_range().size();
537  for (size_t CurStep = WGSize >> 1; CurStep > 0; CurStep >>= 1) {
538  if (LID < CurStep)
539  LocalReds[LID] = BOp(LocalReds[LID], LocalReds[LID + CurStep]);
540  NDIt.barrier();
541  }
542 
543  if (LID == 0) {
544  Reducer.MValue = LocalReds[0];
545  Reducer.atomic_combine(Out.get_pointer().get());
546  }
547  });
548  } else {
549  // Inefficient case: non-uniform work-groups or WGSize is not power of two.
550  // These two inefficient cases are handled by one kernel, which
551  // can be split later into two separate kernels, if there are users who
552  // really need more efficient code for them.
553  using AuxName =
554  typename get_reduction_main_2nd_kernel_name_t<KernelName,
555  KernelType>::name;
556  CGH.parallel_for<AuxName>(Range, [=](nd_item<Dims> NDIt) {
557  // Call user's functions. Reducer.MValue gets initialized there.
558  typename Reduction::reducer_type Reducer;
559  KernelFunc(NDIt, Reducer);
560 
561  size_t WGSize = NDIt.get_local_range().size();
562  size_t LID = NDIt.get_local_linear_id();
563  size_t GID = NDIt.get_global_linear_id();
564  // Copy the element to local memory to prepare it for tree-reduction.
565  auto ReduIdentity = Reducer.getIdentity();
566  LocalReds[LID] = (GID < NWorkItems) ? Reducer.MValue : ReduIdentity;
567  LocalReds[WGSize] = ReduIdentity;
568  NDIt.barrier();
569 
570  // Tree-reduction: reduce the local array LocalReds[:] to LocalReds[0].
571  // LocalReds[WGSize] accumulates last/odd elements when the step
572  // of tree-reduction loop is not even.
573  typename Reduction::binary_operation BOp;
574  size_t PrevStep = WGSize;
575  for (size_t CurStep = PrevStep >> 1; CurStep > 0; CurStep >>= 1) {
576  if (LID < CurStep)
577  LocalReds[LID] = BOp(LocalReds[LID], LocalReds[LID + CurStep]);
578  else if (LID == CurStep && (PrevStep & 0x1))
579  LocalReds[WGSize] = BOp(LocalReds[WGSize], LocalReds[PrevStep - 1]);
580  NDIt.barrier();
581  PrevStep = CurStep;
582  }
583 
584  if (LID == 0) {
585  Reducer.MValue = BOp(LocalReds[0], LocalReds[WGSize]);
586  Reducer.atomic_combine(Out.get_pointer().get());
587  }
588  });
589  }
590 }

662 template <typename KernelName, typename KernelType, int Dims, class Reduction>
663 enable_if_t<!Reduction::has_fast_reduce && !Reduction::has_fast_atomics>
664 reduCGFunc(handler &CGH, KernelType KernelFunc, const nd_range<Dims> &Range,
665  Reduction &Redu) {
666 
667  size_t NWorkItems = Range.get_global_range().size();
668  size_t WGSize = Range.get_local_range().size();
669  size_t NWorkGroups = Range.get_group_range().size();
670 
671  // The last work-group may be not fully loaded with work, or the work group
672  // size may be not power of two. Those two cases considered inefficient
673  // as they require additional code and checks in the kernel.
674  bool HasNonUniformWG = NWorkGroups * WGSize != NWorkItems;
675  bool IsEfficientCase = !HasNonUniformWG && ((WGSize & (WGSize - 1)) == 0);
676 
677  bool IsUpdateOfUserAcc =
678  Reduction::accessor_mode == access::mode::read_write && NWorkGroups == 1;
679 
680  // Use local memory to reduce elements in work-groups into 0-th element.
681  // If WGSize is not power of two, then WGSize+1 elements are allocated.
682  // The additional last element is used to catch elements that could
683  // otherwise be lost in the tree-reduction algorithm.
684  size_t NumLocalElements = WGSize + (IsEfficientCase ? 0 : 1);
685  auto LocalReds = Redu.getReadWriteLocalAcc(NumLocalElements, CGH);
686 
687  auto Out = Redu.getWriteAccForPartialReds(NWorkGroups, 0, CGH);
688  auto ReduIdentity = Redu.getIdentity();
689  if (IsEfficientCase) {
690  // Efficient case: work-groups are uniform and WGSize is is power of two.
691  CGH.parallel_for<KernelName>(Range, [=](nd_item<Dims> NDIt) {
692  // Call user's functions. Reducer.MValue gets initialized there.
693  typename Reduction::reducer_type Reducer(ReduIdentity);
694  KernelFunc(NDIt, Reducer);
695 
696  // Copy the element to local memory to prepare it for tree-reduction.
697  size_t LID = NDIt.get_local_linear_id();
698  LocalReds[LID] = Reducer.MValue;
699  NDIt.barrier();
700 
701  // Tree-reduction: reduce the local array LocalReds[:] to LocalReds[0].
702  typename Reduction::binary_operation BOp;
703  size_t WGSize = NDIt.get_local_range().size();
704  for (size_t CurStep = WGSize >> 1; CurStep > 0; CurStep >>= 1) {
705  if (LID < CurStep)
706  LocalReds[LID] = BOp(LocalReds[LID], LocalReds[LID + CurStep]);
707  NDIt.barrier();
708  }
709 
710  // Compute the partial sum/reduction for the work-group.
711  if (LID == 0)
712  Out.get_pointer().get()[NDIt.get_group_linear_id()] =
713  IsUpdateOfUserAcc ? BOp(*(Out.get_pointer()), LocalReds[0])
714  : LocalReds[0];
715  });
716  } else {
717  // Inefficient case: work-groups are non uniform or WGSize is not power
718  // of two, which requires more conditional, read and write operations.
719  // These two inefficient cases are handled by one kernel, which
720  // can be split later into two separate kernels, if there are users who
721  // really need more efficient code for them.
722  using AuxName =
723  typename get_reduction_main_2nd_kernel_name_t<KernelName,
724  KernelType>::name;
725  CGH.parallel_for<AuxName>(Range, [=](nd_item<Dims> NDIt) {
726  // Call user's functions. Reducer.MValue gets initialized there.
727  typename Reduction::reducer_type Reducer(ReduIdentity);
728  KernelFunc(NDIt, Reducer);
729 
730  size_t WGSize = NDIt.get_local_range().size();
731  size_t LID = NDIt.get_local_linear_id();
732  size_t GID = NDIt.get_global_linear_id();
733  // Copy the element to local memory to prepare it for tree-reduction.
734  LocalReds[LID] = (GID < NWorkItems) ? Reducer.MValue : ReduIdentity;
735  LocalReds[WGSize] = ReduIdentity;
736  NDIt.barrier();
737 
738  // Tree-reduction: reduce the local array LocalReds[:] to LocalReds[0]
739  // LocalReds[WGSize] accumulates last/odd elements when the step
740  // of tree-reduction loop is not even.
741  typename Reduction::binary_operation BOp;
742  size_t PrevStep = WGSize;
743  for (size_t CurStep = PrevStep >> 1; CurStep > 0; CurStep >>= 1) {
744  if (LID < CurStep)
745  LocalReds[LID] = BOp(LocalReds[LID], LocalReds[LID + CurStep]);
746  else if (LID == CurStep && (PrevStep & 0x1))
747  LocalReds[WGSize] = BOp(LocalReds[WGSize], LocalReds[PrevStep - 1]);
748  NDIt.barrier();
749  PrevStep = CurStep;
750  }
751 
752  // Compute the partial sum/reduction for the work-group.
753  if (LID == 0) {
754  size_t GrID = NDIt.get_group_linear_id();
755  auto V = BOp(LocalReds[0], LocalReds[WGSize]);
756  Out.get_pointer().get()[GrID] =
757  IsUpdateOfUserAcc ? BOp(*(Out.get_pointer()), V) : V;
758  }
759  });
760  }
761 }
