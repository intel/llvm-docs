29 using IsZeroIdentityOp = bool_constant<
30  ((is_geninteger8bit<T>::value || is_geninteger16bit<T>::value ||
31  is_geninteger32bit<T>::value || is_geninteger64bit<T>::value) &&
32  (std::is_same<BinaryOperation, intel::plus<T>>::value ||
33  std::is_same<BinaryOperation, intel::bit_or<T>>::value ||
34  std::is_same<BinaryOperation, intel::bit_xor<T>>::value)) ||
35  ((std::is_same<T, float>::value || std::is_same<T, double>::value) &&
36  std::is_same<BinaryOperation, intel::plus<T>>::value)>;

41  (is_geninteger8bit<T>::value || is_geninteger16bit<T>::value ||
42  is_geninteger32bit<T>::value || is_geninteger64bit<T>::value ||
43  std::is_same<T, float>::value || std::is_same<T, double>::value) &&
44  std::is_same<BinaryOperation, std::multiplies<T>>::value>;

287  // TODO: support true 1-Dimensional accessors/reductions (get_count() > 1).
288  // (get_count() == 1) is checked in the constructor of reduction_impl.
289  static_assert(Dims <= 1 && IsPlaceholder == access::placeholder::false_t,

318  template <
319  typename _T = T, class _BinaryOperation = BinaryOperation,
320  enable_if_t<IsKnownIdentityOp<_T, _BinaryOperation>::value> * = nullptr>
321  reduction_impl(accessor_type &Acc, const T &Identity)
322  : MAcc(Acc), MIdentity(Identity) {
323  assert(Acc.get_count() == 1 &&
324  "Only scalar/1-element reductions are supported now.");
325  // For operations with known identity value the operator == is defined.
326  // It is sort of dilemma here: from one point of view - user may set
327  // such identity that would be enough for his data, i.e. identity=100 for
328  // min operation if user knows all data elements are less than 100.
329  // From another point of view - it is the source of unexpected errors,
330  // when the input data changes.
331  // Let's be strict for now and emit an error if identity is not proper.
332  assert(Identity == getIdentity() && "Unexpected Identity parameter value.");
333  }

414 template <typename KernelName, typename KernelType, int Dims, class Reduction>
415 void reduCGFunc(handler &CGH, KernelType KernelFunc,
416  const nd_range<Dims> &Range, Reduction &Redu) {
417 
418  size_t NWorkItems = Range.get_global_range().size();
419  size_t WGSize = Range.get_local_range().size();
420  size_t NWorkGroups = Range.get_group_range().size();
421 
422  // The last work-group may be not fully loaded with work, or the work group
423  // size may be not power of two. Those two cases considered inefficient
424  // as they require additional code and checks in the kernel.
425  bool HasNonUniformWG = (NWorkGroups * WGSize - NWorkItems) != 0;
426  bool IsEfficientCase = !HasNonUniformWG && ((WGSize & (WGSize - 1)) == 0);
427 
428  bool IsUpdateOfUserAcc =
429  Reduction::accessor_mode == access::mode::read_write && NWorkGroups == 1;
430 
431  // Use local memory to reduce elements in work-groups into 0-th element.
432  // If WGSize is not power of two, then WGSize+1 elements are allocated.
433  // The additional last element is used to catch elements that could
434  // otherwise be lost in the tree-reduction algorithm.
435  size_t NumLocalElements = WGSize + (IsEfficientCase ? 0 : 1);
436  auto LocalReds = Redu.getReadWriteLocalAcc(NumLocalElements, CGH);
437 
438  auto Out = Redu.getWriteAccForPartialReds(NWorkGroups, 0, CGH);
439  auto ReduIdentity = Redu.getIdentity();
440  if (IsEfficientCase) {
441  // Efficient case: work-groups are uniform and WGSize is is power of two.
442  CGH.parallel_for<KernelName>(Range, [=](nd_item<Dims> NDIt) {
443  // Call user's functions. Reducer.MValue gets initialized there.
444  typename Reduction::reducer_type Reducer(ReduIdentity);
445  KernelFunc(NDIt, Reducer);
446 
447  // Copy the element to local memory to prepare it for tree-reduction.
448  size_t LID = NDIt.get_local_linear_id();
449  LocalReds[LID] = Reducer.MValue;
450  NDIt.barrier();
451 
452  // Tree-reduction: reduce the local array LocalReds[:] to LocalReds[0].
453  typename Reduction::binary_operation BOp;
454  size_t WGSize = NDIt.get_local_range().size();
455  for (size_t CurStep = WGSize >> 1; CurStep > 0; CurStep >>= 1) {
456  if (LID < CurStep)
457  LocalReds[LID] = BOp(LocalReds[LID], LocalReds[LID + CurStep]);
458  NDIt.barrier();
459  }
460 
461  // Compute the partial sum/reduction for the work-group.
462  if (LID == 0)
463  Out.get_pointer().get()[NDIt.get_group_linear_id()] =
464  IsUpdateOfUserAcc ? BOp(*(Out.get_pointer()), LocalReds[0])
465  : LocalReds[0];
466  });
467  } else {
468  // Inefficient case: work-groups are non uniform or WGSize is not power
469  // of two, which requires more conditional, read and write operations.
470  // These two inefficient cases are handled by one kernel, which
471  // can be split later into two separate kernels, if there are users who
472  // really need more efficient code for them.
473  using AuxName =
474  typename get_reduction_main_2nd_kernel_name_t<KernelName,
475  KernelType>::name;
476  CGH.parallel_for<AuxName>(Range, [=](nd_item<Dims> NDIt) {
477  // Call user's functions. Reducer.MValue gets initialized there.
478  typename Reduction::reducer_type Reducer(ReduIdentity);
479  KernelFunc(NDIt, Reducer);
480 
481  size_t WGSize = NDIt.get_local_range().size();
482  size_t LID = NDIt.get_local_linear_id();
483  size_t GID = NDIt.get_global_linear_id();
484  // Copy the element to local memory to prepare it for tree-reduction.
485  LocalReds[LID] = (GID < NWorkItems) ? Reducer.MValue : ReduIdentity;
486  LocalReds[WGSize] = ReduIdentity;
487  NDIt.barrier();
488 
489  // Tree-reduction: reduce the local array LocalReds[:] to LocalReds[0]
490  // LocalReds[WGSize] accumulates last/odd elements when the step
491  // of tree-reduction loop is not even.
492  typename Reduction::binary_operation BOp;
493  size_t PrevStep = WGSize;
494  for (size_t CurStep = PrevStep >> 1; CurStep > 0; CurStep >>= 1) {
495  if (LID < CurStep)
496  LocalReds[LID] = BOp(LocalReds[LID], LocalReds[LID + CurStep]);
497  else if (LID == CurStep && (PrevStep & 0x1))
498  LocalReds[WGSize] = BOp(LocalReds[WGSize], LocalReds[PrevStep - 1]);
499  NDIt.barrier();
500  PrevStep = CurStep;
501  }
502 
503  // Compute the partial sum/reduction for the work-group.
504  if (LID == 0) {
505  size_t GrID = NDIt.get_group_linear_id();
506  auto V = BOp(LocalReds[0], LocalReds[WGSize]);
507  Out.get_pointer().get()[GrID] =
508  IsUpdateOfUserAcc ? BOp(*(Out.get_pointer()), V) : V;
509  }
510  });
511  }
512 }

521 template <typename KernelName, typename KernelType, int Dims, class Reduction>
522 void reduAuxCGFunc(handler &CGH, const nd_range<Dims> &Range, size_t NWorkItems,
523  size_t KernelRun, Reduction &Redu) {
524  size_t WGSize = Range.get_local_range().size();
525  size_t NWorkGroups = Range.get_group_range().size();
526 
527  // The last work-group may be not fully loaded with work, or the work group
528  // size may be not power of two. Those two cases considered inefficient
529  // as they require additional code and checks in the kernel.
530  bool HasNonUniformWG = NWorkGroups * WGSize != NWorkItems;
531  bool IsEfficientCase = !HasNonUniformWG && (WGSize & (WGSize - 1)) == 0;
532 
533  bool IsUpdateOfUserAcc =
534  Reduction::accessor_mode == access::mode::read_write && NWorkGroups == 1;
535 
536  // Use local memory to reduce elements in work-groups into 0-th element.
537  // If WGSize is not power of two, then WGSize+1 elements are allocated.
538  // The additional last element is used to catch elements that could
539  // otherwise be lost in the tree-reduction algorithm.
540  size_t NumLocalElements = WGSize + (IsEfficientCase ? 0 : 1);
541  auto LocalReds = Redu.getReadWriteLocalAcc(NumLocalElements, CGH);
542 
543  // Get read accessor to the buffer that was used as output
544  // in the previous kernel. After that create new output buffer if needed
545  // and get accessor to it (or use reduction's accessor if the kernel
546  // is the last one).
547  auto In = Redu.getReadAccToPreviousPartialReds(CGH);
548  auto Out = Redu.getWriteAccForPartialReds(NWorkGroups, KernelRun, CGH);
549 
550  if (IsEfficientCase) {
551  // Efficient case: work-groups are fully loaded and work-group size
552  // is power of two.
553  using AuxName =
554  typename get_reduction_aux_1st_kernel_name_t<KernelName,
555  KernelType>::name;
556  CGH.parallel_for<AuxName>(Range, [=](nd_item<Dims> NDIt) {
557  // Copy the element to local memory to prepare it for tree-reduction.
558  size_t LID = NDIt.get_local_linear_id();
559  size_t GID = NDIt.get_global_linear_id();
560  LocalReds[LID] = In[GID];
561  NDIt.barrier();
562 
563  // Tree-reduction: reduce the local array LocalReds[:] to LocalReds[0]
564  typename Reduction::binary_operation BOp;
565  size_t WGSize = NDIt.get_local_range().size();
566  for (size_t CurStep = WGSize >> 1; CurStep > 0; CurStep >>= 1) {
567  if (LID < CurStep)
568  LocalReds[LID] = BOp(LocalReds[LID], LocalReds[LID + CurStep]);
569  NDIt.barrier();
570  }
571 
572  // Compute the partial sum/reduction for the work-group.
573  if (LID == 0)
574  Out.get_pointer().get()[NDIt.get_group_linear_id()] =
575  IsUpdateOfUserAcc ? BOp(*(Out.get_pointer()), LocalReds[0])
576  : LocalReds[0];
577  });
578  } else {
579  // Inefficient case: work-groups are not fully loaded
580  // or WGSize is not power of two.
581  // These two inefficient cases are handled by one kernel, which
582  // can be split later into two separate kernels, if there are users
583  // who really need more efficient code for them.
584  using AuxName =
585  typename get_reduction_aux_2nd_kernel_name_t<KernelName,
586  KernelType>::name;
587  auto ReduIdentity = Redu.getIdentity();
588  CGH.parallel_for<AuxName>(Range, [=](nd_item<Dims> NDIt) {
589  size_t WGSize = NDIt.get_local_range().size();
590  size_t LID = NDIt.get_local_linear_id();
591  size_t GID = NDIt.get_global_linear_id();
592  // Copy the element to local memory to prepare it for tree-reduction
593  LocalReds[LID] = (GID < NWorkItems) ? In[GID] : ReduIdentity;
594  LocalReds[WGSize] = ReduIdentity;
595  NDIt.barrier();
596 
597  // Tree-reduction: reduce the local array LocalReds[:] to LocalReds[0]
598  // LocalReds[WGSize] accumulates last/odd elements when the step
599  // of tree-reduction loop is not even.
600  typename Reduction::binary_operation BOp;
601  size_t PrevStep = WGSize;
602  for (size_t CurStep = PrevStep >> 1; CurStep > 0; CurStep >>= 1) {
603  if (LID < CurStep)
604  LocalReds[LID] = BOp(LocalReds[LID], LocalReds[LID + CurStep]);
605  else if (LID == CurStep && (PrevStep & 0x1))
606  LocalReds[WGSize] = BOp(LocalReds[WGSize], LocalReds[PrevStep - 1]);
607  NDIt.barrier();
608  PrevStep = CurStep;
609  }
610 
611  // Compute the partial sum/reduction for the work-group.
612  if (LID == 0) {
613  size_t GrID = NDIt.get_group_linear_id();
614  auto V = BOp(LocalReds[0], LocalReds[WGSize]);
615  Out.get_pointer().get()[GrID] =
616  IsUpdateOfUserAcc ? BOp(*(Out.get_pointer()), V) : V;
617  }
618  });
619  }
