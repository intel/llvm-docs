608  bool UniformWG, typename OutputT>
609 enable_if_t<Reduction::has_fast_reduce && Reduction::has_fast_atomics>
610 reduCGFuncImpl(handler &CGH, KernelType KernelFunc, const nd_range<Dims> &Range,
611  Reduction &, OutputT Out) {
612  size_t NWorkItems = Range.get_global_range().size();
613  using Name = typename get_reduction_main_kernel_name_t<
614  KernelName, KernelType, Reduction::is_usm, UniformWG, OutputT>::name;
615  CGH.parallel_for<Name>(Range, [=](nd_item<Dims> NDIt) {
616  // Call user's function. Reducer.MValue gets initialized there.
617  typename Reduction::reducer_type Reducer;
618  KernelFunc(NDIt, Reducer);
619 
620  typename Reduction::binary_operation BOp;
621  typename Reduction::result_type Val =
622  (UniformWG || NDIt.get_global_linear_id() < NWorkItems)
623  ? Reducer.MValue
624  : Reducer.getIdentity();
625  Reducer.MValue = intel::reduce(NDIt.get_group(), Val, BOp);
626  if (NDIt.get_local_linear_id() == 0)
627  Reducer.atomic_combine(Reduction::getOutPointer(Out));
628  });
