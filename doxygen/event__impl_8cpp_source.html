61 void event_impl::waitInternal(bool *Success) {
62  if (!MHostEvent && MEvent) {
63  // Wait for the native event
64  sycl::detail::pi::PiResult Err =
65  getPlugin()->call_nocheck<PiApiKind::piEventsWait>(1, &MEvent);
66  // TODO drop the PI_ERROR_UKNOWN from here once the UR counterpart to
67  // PI_ERROR_EXEC_STATUS_ERROR_FOR_EVENTS_IN_WAIT_LIST is added:
68  // https://github.com/oneapi-src/unified-runtime/issues/1459
69  if (Success != nullptr &&
70  (Err == PI_ERROR_UNKNOWN ||
71  Err == PI_ERROR_EXEC_STATUS_ERROR_FOR_EVENTS_IN_WAIT_LIST))
72  *Success = false;
73  else {
74  getPlugin()->checkPiResult(Err);
75  if (Success != nullptr)
76  *Success = true;
77  }
78  } else if (MState == HES_Discarded) {
79  // Waiting for the discarded event is invalid
80  throw sycl::exception(
81  make_error_code(errc::invalid),
82  "waitInternal method cannot be used for a discarded event.");
83  } else if (MState != HES_Complete) {
84  // Wait for the host event
85  std::unique_lock<std::mutex> lock(MMutex);
86  cv.wait(lock, [this] { return MState == HES_Complete; });
87  }
88 
89  // Wait for connected events(e.g. streams prints)
90  for (const EventImplPtr &Event : MPostCompleteEvents)
91  Event->wait(Event);
92 }

94 void event_impl::setComplete() {
95  if (MHostEvent || !MEvent) {
96  {
97  std::unique_lock<std::mutex> lock(MMutex);
98 #ifndef NDEBUG
99  int Expected = HES_NotComplete;
100  int Desired = HES_Complete;
101 
102  bool Succeeded = MState.compare_exchange_strong(Expected, Desired);
103 
104  assert(Succeeded && "Unexpected state of event");
105 #else
106  MState.store(static_cast<int>(HES_Complete));
107 #endif
108  }
109  cv.notify_all();
110  return;
111  }
112 
113  assert(false && "setComplete is not supported for non-host event");
114 }

176  MQueue = Queue;

193 void *event_impl::instrumentationProlog(std::string &Name, int32_t StreamID,
194  uint64_t &IId) const {
195  void *TraceEvent = nullptr;
196 #ifdef XPTI_ENABLE_INSTRUMENTATION
197  constexpr uint16_t NotificationTraceType = xpti::trace_wait_begin;
198  if (!xptiCheckTraceEnabled(StreamID, NotificationTraceType))
199  return TraceEvent;
200  // Use a thread-safe counter to get a unique instance ID for the wait() on the
201  // event
202  static std::atomic<uint64_t> InstanceID = {1};
203  xpti::trace_event_data_t *WaitEvent = nullptr;
204 
205  // Create a string with the event address so it
206  // can be associated with other debug data
207  xpti::utils::StringHelper SH;
208  Name = SH.nameWithAddress<sycl::detail::pi::PiEvent>("event.wait", MEvent);
209 
210  // We can emit the wait associated with the graph if the
211  // event does not have a command object or associated with
212  // the command object, if it exists
213  if (MCommand) {
214  Command *Cmd = (Command *)MCommand;
215  WaitEvent = Cmd->MTraceEvent ? static_cast<xpti_td *>(Cmd->MTraceEvent)
216  : GSYCLGraphEvent;
217  } else
218  WaitEvent = GSYCLGraphEvent;
219 
220  // Record the current instance ID for use by Epilog
221  IId = InstanceID++;
222  xptiNotifySubscribers(StreamID, NotificationTraceType, nullptr, WaitEvent,
223  IId, static_cast<const void *>(Name.c_str()));
224  TraceEvent = (void *)WaitEvent;
225 #endif
226  return TraceEvent;
227 }

245 void event_impl::wait(std::shared_ptr<sycl::detail::event_impl> Self,
246  bool *Success) {
247  if (MState == HES_Discarded)
248  throw sycl::exception(make_error_code(errc::invalid),
249  "wait method cannot be used for a discarded event.");
250 
251  if (!MGraph.expired()) {
252  throw sycl::exception(make_error_code(errc::invalid),
253  "wait method cannot be used for an event associated "
254  "with a command graph.");
255  }
256 
257 #ifdef XPTI_ENABLE_INSTRUMENTATION
258  void *TelemetryEvent = nullptr;
259  uint64_t IId = 0;
260  std::string Name;
261  int32_t StreamID = xptiRegisterStream(SYCL_STREAM_NAME);
262  TelemetryEvent = instrumentationProlog(Name, StreamID, IId);
263 #endif
264 
265  if (MEvent)
266  // presence of MEvent means the command has been enqueued, so no need to
267  // go via the slow path event waiting in the scheduler
268  waitInternal(Success);
269  else if (MCommand)
270  detail::Scheduler::getInstance().waitForEvent(Self, Success);
271 
272 #ifdef XPTI_ENABLE_INSTRUMENTATION
273  instrumentationEpilog(TelemetryEvent, Name, StreamID, IId);
274 #endif
275 }

301 template <>
302 uint64_t
303 event_impl::get_profiling_info<info::event_profiling::command_submit>() {
304  checkProfilingPreconditions();
305 
306  // The delay between the submission and the actual start of a CommandBuffer
307  // can be short. Consequently, the submission time, which is based on
308  // an estimated clock and not on the real device clock, may be ahead of the
309  // start time, which is based on the actual device clock.
310  // MSubmitTime is set in a critical performance path.
311  // Force reading the device clock when setting MSubmitTime may deteriorate
312  // the performance.
313  // Since submit time is an estimated time, we implement this little hack
314  // that allows all profiled time to be meaningful.
315  // (Note that the observed time deviation between the estimated clock and
316  // the real device clock is typically less than 0.5ms. The approximation we
317  // made by forcing the re-sync of submit time to start time is less than
318  // 0.5ms. These timing values were obtained empirically using an integrated
319  // Intel GPU).
320  if (MEventFromSubmittedExecCommandBuffer && !MHostEvent && MEvent) {
321  uint64_t StartTime =
322  get_event_profiling_info<info::event_profiling::command_start>(
323  this->getHandleRef(), this->getPlugin());
324  if (StartTime < MSubmitTime)
325  MSubmitTime = StartTime;
326  }
327  return MSubmitTime;
328 }

330 template <>
331 uint64_t
332 event_impl::get_profiling_info<info::event_profiling::command_start>() {
333  checkProfilingPreconditions();
334 
335  // For nop command start time is equal to submission time.
336  if (isNOP() && MSubmitTime)
337  return MSubmitTime;
338 
339  if (!MHostEvent) {
340  if (MEvent) {
341  auto StartTime =
342  get_event_profiling_info<info::event_profiling::command_start>(
343  this->getHandleRef(), this->getPlugin());
344  if (!MFallbackProfiling) {
345  return StartTime;
346  } else {
347  auto DeviceBaseTime =
348  get_event_profiling_info<info::event_profiling::command_submit>(
349  this->getHandleRef(), this->getPlugin());
350  return MHostBaseTime - DeviceBaseTime + StartTime;
351  }
352  }
353  return 0;
354  }
355  if (!MHostProfilingInfo)
356  throw sycl::exception(
357  sycl::make_error_code(sycl::errc::invalid),
358  "Profiling info is not available. " +
359  codeToString(PI_ERROR_PROFILING_INFO_NOT_AVAILABLE));
360  return MHostProfilingInfo->getStartTime();
361 }

517 void event_impl::flushIfNeeded(const QueueImplPtr &UserQueue) {
518  // Some events might not have a native handle underneath even at this point,
519  // e.g. those produced by memset with 0 size (no PI call is made).
520  if (MIsFlushed || !MEvent)
521  return;
522 
523  QueueImplPtr Queue = MQueue.lock();
524  // If the queue has been released, all of the commands have already been
525  // implicitly flushed by piQueueRelease.
526  if (!Queue) {
527  MIsFlushed = true;
528  return;
529  }
530  if (Queue == UserQueue)
531  return;
532 
533  // Check if the task for this event has already been submitted.
534  pi_event_status Status = PI_EVENT_QUEUED;
535  getPlugin()->call<PiApiKind::piEventGetInfo>(
536  MEvent, PI_EVENT_INFO_COMMAND_EXECUTION_STATUS, sizeof(pi_int32), &Status,
537  nullptr);
538  if (Status == PI_EVENT_QUEUED) {
539  getPlugin()->call<PiApiKind::piQueueFlush>(Queue->getHandleRef());
540  }
541  MIsFlushed = true;
542 }
