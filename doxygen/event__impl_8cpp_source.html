62 void event_impl::setComplete() {

97 event_impl::event_impl(RT::PiEvent Event, const context &SyclContext)
98  : MEvent(Event), MContext(detail::getSyclObjImpl(SyclContext)),
99  MOpenCLInterop(true), MHostEvent(false), MState(HES_Complete) {
100 
101  if (MContext->is_host()) {
102  throw cl::sycl::invalid_parameter_error(
103  "The syclContext must match the OpenCL context associated with the "
104  "clEvent.",
105  PI_INVALID_CONTEXT);
106  }
107 
108  RT::PiContext TempContext;
109  getPlugin().call<PiApiKind::piEventGetInfo>(MEvent, PI_EVENT_INFO_CONTEXT,
110  sizeof(RT::PiContext),
111  &TempContext, nullptr);
112  if (MContext->getHandleRef() != TempContext) {
113  throw cl::sycl::invalid_parameter_error(
114  "The syclContext must match the OpenCL context associated with the "
115  "clEvent.",
116  PI_INVALID_CONTEXT);
117  }
118 
119  getPlugin().call<PiApiKind::piEventRetain>(MEvent);
120 }

138 void *event_impl::instrumentationProlog(string_class &Name, int32_t StreamID,
139  uint64_t &IId) const {
140  void *TraceEvent = nullptr;
141 #ifdef XPTI_ENABLE_INSTRUMENTATION
142  if (!xptiTraceEnabled())
143  return TraceEvent;
144  // Use a thread-safe counter to get a unique instance ID for the wait() on the
145  // event
146  static std::atomic<uint64_t> InstanceID = {1};
147  xpti::trace_event_data_t *WaitEvent = nullptr;
148 
149  // Create a string with the event address so it
150  // can be associated with other debug data
151  xpti::utils::StringHelper SH;
152  Name = SH.nameWithAddress<RT::PiEvent>("event.wait", MEvent);
153 
154  // We can emit the wait associated with the graph if the
155  // event does not have a command object or associated with
156  // the command object, if it exists
157  if (MCommand) {
158  Command *Cmd = (Command *)MCommand;
159  WaitEvent = Cmd->MTraceEvent ? static_cast<xpti_td *>(Cmd->MTraceEvent)
160  : GSYCLGraphEvent;
161  } else
162  WaitEvent = GSYCLGraphEvent;
163 
164  // Record the current instance ID for use by Epilog
165  IId = InstanceID++;
166  xptiNotifySubscribers(StreamID, xpti::trace_wait_begin, nullptr, WaitEvent,
167  IId, static_cast<const void *>(Name.c_str()));
168  TraceEvent = (void *)WaitEvent;
169 #endif
170  return TraceEvent;
171 }

173 void event_impl::instrumentationEpilog(void *TelemetryEvent,
174  const string_class &Name,
175  int32_t StreamID, uint64_t IId) const {
176 #ifdef XPTI_ENABLE_INSTRUMENTATION
177  if (!(xptiTraceEnabled() && TelemetryEvent))
178  return;
179  // Close the wait() scope
180  xpti::trace_event_data_t *TraceEvent =
181  (xpti::trace_event_data_t *)TelemetryEvent;
182  xptiNotifySubscribers(StreamID, xpti::trace_wait_end, nullptr, TraceEvent,
183  IId, static_cast<const void *>(Name.c_str()));
184 #endif
185 }

187 void event_impl::wait(
188  std::shared_ptr<cl::sycl::detail::event_impl> Self) const {
189 #ifdef XPTI_ENABLE_INSTRUMENTATION
190  void *TelemetryEvent = nullptr;
191  uint64_t IId;
192  std::string Name;
193  int32_t StreamID = xptiRegisterStream(SYCL_STREAM_NAME);
194  TelemetryEvent = instrumentationProlog(Name, StreamID, IId);
195 #endif
196 
197  if (MEvent)
198  // presence of MEvent means the command has been enqueued, so no need to
199  // go via the slow path event waiting in the scheduler
200  waitInternal();
201  else if (MCommand)
202  detail::Scheduler::getInstance().waitForEvent(Self);
203  if (MCommand && !SYCLConfig<SYCL_DISABLE_EXECUTION_GRAPH_CLEANUP>::get())
204  detail::Scheduler::getInstance().cleanupFinishedCommands(std::move(Self));
205 
206 #ifdef XPTI_ENABLE_INSTRUMENTATION
207  instrumentationEpilog(TelemetryEvent, Name, StreamID, IId);
208 #endif
209 }

211 void event_impl::wait_and_throw(
212  std::shared_ptr<cl::sycl::detail::event_impl> Self) {
213  wait(Self);
214  for (auto &EventImpl :
215  detail::Scheduler::getInstance().getWaitList(std::move(Self))) {
216  Command *Cmd = (Command *)EventImpl->getCommand();
217  if (Cmd)
218  Cmd->getQueue()->throw_asynchronous();
219  }
220  Command *Cmd = (Command *)getCommand();
221  if (Cmd)
222  Cmd->getQueue()->throw_asynchronous();
223 }
