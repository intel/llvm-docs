107 void *event_impl::instrumentationProlog(string_class &Name, int32_t StreamID,
108  uint64_t &IId) const {
109  void *TraceEvent = nullptr;
110 #ifdef XPTI_ENABLE_INSTRUMENTATION
111  if (!xptiTraceEnabled())
112  return TraceEvent;
113  // Use a thread-safe counter to get a unique instance ID for the wait() on the
114  // event
115  static std::atomic<uint64_t> InstanceID = {1};
116  xpti::trace_event_data_t *WaitEvent = nullptr;
117 
118  // Create a string with the event address so it
119  // can be associated with other debug data
120  xpti::utils::StringHelper SH;
121  Name = SH.nameWithAddress<RT::PiEvent>("event.wait", MEvent);
122 
123  // We can emit the wait associated with the graph if the
124  // event does not have a command object or associated with
125  // the command object, if it exists
126  if (MCommand) {
127  Command *Cmd = (Command *)MCommand;
128  WaitEvent = Cmd->MTraceEvent ? static_cast<xpti_td *>(Cmd->MTraceEvent)
129  : GSYCLGraphEvent;
130  } else
131  WaitEvent = GSYCLGraphEvent;
132 
133  // Record the current instance ID for use by Epilog
134  IId = InstanceID++;
135  xptiNotifySubscribers(StreamID, xpti::trace_wait_begin, nullptr, WaitEvent,
136  IId, static_cast<const void *>(Name.c_str()));
137  TraceEvent = (void *)WaitEvent;
138 #endif
139  return TraceEvent;
140 }

142 void event_impl::instrumentationEpilog(void *TelemetryEvent,
143  const string_class &Name,
144  int32_t StreamID, uint64_t IId) const {
145 #ifdef XPTI_ENABLE_INSTRUMENTATION
146  if (!(xptiTraceEnabled() && TelemetryEvent))
147  return;
148  // Close the wait() scope
149  xpti::trace_event_data_t *TraceEvent =
150  (xpti::trace_event_data_t *)TelemetryEvent;
151  xptiNotifySubscribers(StreamID, xpti::trace_wait_end, nullptr, TraceEvent,
152  IId, static_cast<const void *>(Name.c_str()));
153 #endif
154 }

156 void event_impl::wait(
157  std::shared_ptr<cl::sycl::detail::event_impl> Self) const {
158 #ifdef XPTI_ENABLE_INSTRUMENTATION
159  void *TelemetryEvent = nullptr;
160  uint64_t IId;
161  std::string Name;
162  int32_t StreamID = xptiRegisterStream(SYCL_STREAM_NAME);
163  TelemetryEvent = instrumentationProlog(Name, StreamID, IId);
164 #endif
165 
166  if (MEvent)
167  // presence of MEvent means the command has been enqueued, so no need to
168  // go via the slow path event waiting in the scheduler
169  waitInternal();
170  else if (MCommand)
171  detail::Scheduler::getInstance().waitForEvent(Self);
172  if (MCommand && !SYCLConfig<SYCL_DISABLE_EXECUTION_GRAPH_CLEANUP>::get())
173  detail::Scheduler::getInstance().cleanupFinishedCommands(std::move(Self));
174 
175 #ifdef XPTI_ENABLE_INSTRUMENTATION
176  instrumentationEpilog(TelemetryEvent, Name, StreamID, IId);
177 #endif
178 }

180 void event_impl::wait_and_throw(
181  std::shared_ptr<cl::sycl::detail::event_impl> Self) {
182  wait(Self);
183  for (auto &EventImpl :
184  detail::Scheduler::getInstance().getWaitList(std::move(Self))) {
185  Command *Cmd = (Command *)EventImpl->getCommand();
186  if (Cmd)
187  Cmd->getQueue()->throw_asynchronous();
188  }
189  QueueImplPtr Queue = MQueue.lock();
190  if (Queue)
191  Queue->throw_asynchronous();
192 }
