27 template <typename T> constexpr dpas_argument_type dpas_precision_from_type() {
28  if constexpr (std::is_same_v<T,
29  sycl::ext::intel::experimental::esimd::tfloat32>)
30  return dpas_argument_type::tf32;
31  else if constexpr (std::is_same_v<T, sycl::half>)
32  return dpas_argument_type::fp16;
33  else if constexpr (std::is_same_v<T, sycl::ext::oneapi::bfloat16>)
34  return dpas_argument_type::bf16;
35  else if constexpr (std::is_same_v<T, unsigned char>)
36  return dpas_argument_type::u8;
37  else if constexpr (__ESIMD_DNS::is_type<T, char, signed char>())
38  return dpas_argument_type::s8;
39  else
40  return dpas_argument_type::Invalid;
41 }

43 template <dpas_argument_type T> constexpr int dpas_bitsize_from_precision() {
44  if constexpr (T == dpas_argument_type::u2 || T == dpas_argument_type::s2)
45  return 2;
46  else if constexpr (T == dpas_argument_type::u4 || T == dpas_argument_type::s4)
47  return 4;
48  else if constexpr (T == dpas_argument_type::u8 || T == dpas_argument_type::s8)
49  return 8;
50  else if constexpr (T == dpas_argument_type::bf16 ||
51  T == dpas_argument_type::fp16)
52  return 16;
53  else if constexpr (T == dpas_argument_type::tf32)
54  return 32;
55  else
56  return -1;
57 }

59 template <int RepeatCount, int AElemBitSize, int BElemBitSize, bool IsDPASW>
60 constexpr void verify_repeat_count() {
61  static_assert(RepeatCount >= 1 && RepeatCount <= 8,
62  "Repeat count must be within 1 to 8 range");
63 
64  if constexpr (IsDPASW && RepeatCount != 8) {
65  static_assert(!(AElemBitSize == 2 && BElemBitSize > 4),
66  "Unsupported repeat count for DPASW operation");
67 
68  static_assert(
69  RepeatCount == 4 ||
70  (AElemBitSize != 2 && (AElemBitSize != 4 || BElemBitSize <= 4)),
71  "Unsupported repeat count for DPASW operation");
72  }

75 template <int SystolicDepth, int RepeatCount, typename T, typename CT,
76  typename BT, typename AT, dpas_argument_type BPrecision,
77  dpas_argument_type APrecision, int BN, int AN, bool IsDPASW = false>
78 constexpr int verify_parameters_and_deduce_exec_size() {
79 
80  static_assert(SystolicDepth == 8, "Systolic depth must be equal to 8");
81  static_assert(
82  APrecision != dpas_argument_type::Invalid &&
83  BPrecision != dpas_argument_type::Invalid,
84  "The types of dpas arguments are either incorrect or cannot be deduced."
85  "Fix the types and/or explicitly specify them.");
86 
87  constexpr int AElemBitSize = dpas_bitsize_from_precision<APrecision>();
88  constexpr int BElemBitSize = dpas_bitsize_from_precision<BPrecision>();
89  static_assert(AElemBitSize != -1 && BElemBitSize != -1,
90  "Cannot deduce element size of input arguments");
91  verify_repeat_count<RepeatCount, AElemBitSize, BElemBitSize, IsDPASW>();
92 
93  constexpr int MaxElemBitSize =
94  AElemBitSize > BElemBitSize ? AElemBitSize : BElemBitSize;
95  constexpr int MaxElemsInDword = 32 / MaxElemBitSize;
96  constexpr int OpsPerChannel =
97  MaxElemsInDword > 8 ? 8 : (MaxElemsInDword < 1 ? 1 : MaxElemsInDword);
98 
99  // A(_Mx_K) * B(_Kx_N) + C(_Mx_N)
100  // where:
101  // _M = RepeatCount;
102  // _K = SystolicDepth * OpsPerChannel;
103  // _N = ExecutionSize (unknown, but deducible), must be 8 or 16.
104  constexpr int _M = RepeatCount;
105  constexpr int _K = SystolicDepth * OpsPerChannel;
106 
107  // Compute _N (aka ExecutionSize) from the matrix B.
108  // It has _K*_N elements of BPrecision type, and BN elements of BT type
109  // hold those _K*_N*BPrecision bits, which let's us compute _N.
110  constexpr int BMatrixBitSize = sizeof(BT) * BN * 8;
111  constexpr int BNumElems = BMatrixBitSize / BElemBitSize;
112  constexpr int _N = BNumElems / _K;
113  static_assert(_K * _N == BNumElems, "Cannot deduce the execution size.");
114 
115  // Now verify that AN elements of AT type hold exactly _M*_K elements
116  // of APrecision type/size. Similarly for B: BN elements of BT type must
117  // hold _K*_N elements of BPrecision type/size.
118  // DPASW accepts 2x less expected AN elements than regular DPAS.
119  constexpr int AFactorForDPASW = IsDPASW ? 2 : 1;
120  static_assert(_M * _K * AElemBitSize == AN * sizeof(AT) * 8 * AFactorForDPASW,
121  "The first matrix multiplier has wrong size.");
122  static_assert(_K * _N * BElemBitSize == BN * sizeof(BT) * 8,
123  "The second matrix multiplier has wrong size.");
124 
125  // Execution size may be 8 or 16 depending on the target device.
126  // User must check if used execution size is supported before calling DPAS.
127  constexpr int ExecutionSize = _N;
128 
129  static_assert(ExecutionSize == 8 || (!IsDPASW && ExecutionSize == 16),
130  "Execution size must be 8 or 16 for DPAS and 8 for DPASW.");
131 
132  if constexpr (APrecision == dpas_argument_type::fp16 ||
133  BPrecision == dpas_argument_type::fp16) {
134  if constexpr (ExecutionSize == 8) {
135  static_assert(APrecision == BPrecision &&
136  __ESIMD_DNS::is_type<T, float>() &&
137  __ESIMD_DNS::is_type<CT, float>(),
138  "Unsupported DPAS types! The supported types are:\n"
139  " Result | C | B | A \n"
140  " f | f | hf | hf \n");
141  } else {
142  static_assert(APrecision == BPrecision &&
143  __ESIMD_DNS::is_type<T, float, sycl::half>() &&
144  __ESIMD_DNS::is_type<CT, float, sycl::half>(),
145  "Unsupported DPAS types! The supported types are:\n"
146  " Result | C | B | A \n"
147  " f, hf | f, hf | hf | hf \n");
148  }
149  } else if constexpr (APrecision == dpas_argument_type::bf16 ||
150  BPrecision == dpas_argument_type::bf16) {
151  using bfloat16 = sycl::ext::oneapi::bfloat16;
152  if constexpr (ExecutionSize == 8) {
153  static_assert(APrecision == BPrecision &&
154  __ESIMD_DNS::is_type<T, float, bfloat16>() &&
155  __ESIMD_DNS::is_type<CT, float, bfloat16>(),
156  "Unsupported DPAS types! The supported types are:\n"
157  " Result | C | B | A \n"
158  " f | f | bf | bf \n");
159  } else {
160  static_assert(APrecision == BPrecision &&
161  __ESIMD_DNS::is_type<T, float, bfloat16>() &&
162  __ESIMD_DNS::is_type<CT, float, bfloat16>(),
163  "Unsupported DPAS types! The supported types are:\n"
164  " Result | C | B | A \n"
165  " f, bf | f, bf | bf | bf \n");
166  }
167  } else if constexpr (APrecision == dpas_argument_type::tf32 ||
168  BPrecision == dpas_argument_type::tf32) {
169  static_assert(ExecutionSize == 16,
170  "tf32 type can be used only with ExecutionSize=16");
171  static_assert(APrecision == BPrecision && std::is_same_v<T, float> &&
172  std::is_same_v<CT, float>,
173  "Unsupported DPAS types! The supported types are:\n"
174  " Result | C | B | A \n"
175  " f | f | tf32 | tf32 \n");
176  } else {
177  static_assert((APrecision == dpas_argument_type::u2 ||
178  APrecision == dpas_argument_type::s2 ||
179  APrecision == dpas_argument_type::u4 ||
180  APrecision == dpas_argument_type::s4 ||
181  APrecision == dpas_argument_type::u8 ||
182  APrecision == dpas_argument_type::s8) &&
183  (BPrecision == dpas_argument_type::u2 ||
184  BPrecision == dpas_argument_type::s2 ||
185  BPrecision == dpas_argument_type::u4 ||
186  BPrecision == dpas_argument_type::s4 ||
187  BPrecision == dpas_argument_type::u8 ||
188  BPrecision == dpas_argument_type::s8),
189  "Unsupported DPAS types! The supported types are:\n"
190  " Result | C | B | A \n"
191  " ud, d | ud, d | ub,b,u4,s4,u2,s2 | ub,b,u4,s4,u2,s2 \n");
192  }
193  return ExecutionSize;
194 }

242 template <
243  int SystolicDepth, int RepeatCount, typename T, typename BT, typename AT,
244  dpas_argument_type BPrecision = detail::dpas_precision_from_type<BT>(),
245  dpas_argument_type APrecision = detail::dpas_precision_from_type<AT>(),
246  int BN, int AN>
247 auto dpas(__ESIMD_NS::simd<BT, BN> B, __ESIMD_NS::simd<AT, AN> A) {
248 
249  constexpr int ExecutionSize =
250  detail::verify_parameters_and_deduce_exec_size<SystolicDepth, RepeatCount,
251  T, T, BT, AT, BPrecision,
252  APrecision, BN, AN>();
253  // Result(_Mx_N) = A(_Mx_K) * B(_Kx_N)
254  // where:
255  // _M = RepeatCount;
256  // _K = SystolicDepth * OpsPerChannel;
257  // _N = ExecutionSize (unknown, but deducible), must be 8 or 16.
258  constexpr int ResultN = RepeatCount * ExecutionSize;
259 
260  using MsgT = int;
261  constexpr int ANCasted = AN * sizeof(AT) / sizeof(MsgT);
262  constexpr int BNCasted = BN * sizeof(BT) / sizeof(MsgT);
263  __ESIMD_NS::simd<MsgT, ANCasted> ACasted = A.template bit_cast_view<MsgT>();
264  __ESIMD_NS::simd<MsgT, BNCasted> BCasted = B.template bit_cast_view<MsgT>();
265 
266  constexpr int Info = (RepeatCount << 24) + (SystolicDepth << 16) +
267  ((int)APrecision << 8) + (int)BPrecision;
268  using RawT = typename __ESIMD_NS::simd<T, ResultN>::raw_element_type;
269  __ESIMD_NS::simd<T, ResultN> Result =
270  __esimd_dpas_nosrc0<Info, RawT, MsgT, MsgT, ResultN, BNCasted, ANCasted>(
271  BCasted.data(), ACasted.data());
272  return Result;
273 }
