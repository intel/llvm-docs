24 bool oclHandleInvalidWorkGroupSize(const device_impl &DeviceImpl,
25  pi_kernel Kernel, const NDRDescT &NDRDesc) {
26  const bool HasLocalSize = (NDRDesc.LocalSize[0] != 0);
27 
28  const plugin &Plugin = DeviceImpl.getPlugin();
29  RT::PiDevice Device = DeviceImpl.getHandleRef();
30 
31  size_t VerSize = 0;
32  Plugin.call<PiApiKind::piDeviceGetInfo>(Device, PI_DEVICE_INFO_VERSION, 0,
33  nullptr, &VerSize);
34  assert(VerSize >= 10 &&
35  "Unexpected device version string"); // strlen("OpenCL X.Y")
36  string_class VerStr(VerSize, '\0');
37  Plugin.call<PiApiKind::piDeviceGetInfo>(Device, PI_DEVICE_INFO_VERSION,
38  VerSize, &VerStr.front(), nullptr);
39  const char *Ver = &VerStr[7]; // strlen("OpenCL ")
40 
41  size_t CompileWGSize[3] = {0};
42  Plugin.call<PiApiKind::piKernelGetGroupInfo>(
43  Kernel, Device, PI_KERNEL_GROUP_INFO_COMPILE_WORK_GROUP_SIZE,
44  sizeof(size_t) * 3, CompileWGSize, nullptr);
45 
46  if (CompileWGSize[0] != 0) {
47  // OpenCL 1.x && 2.0:
48  // PI_INVALID_WORK_GROUP_SIZE if local_work_size is NULL and the
49  // reqd_work_group_size attribute is used to declare the work-group size
50  // for kernel in the program source.
51  if (!HasLocalSize && (Ver[0] == '1' || (Ver[0] == '2' && Ver[2] == '0')))
52  throw sycl::nd_range_error(
53  "OpenCL 1.x and 2.0 requires to pass local size argument even if "
54  "required work-group size was specified in the program source",
55  PI_INVALID_WORK_GROUP_SIZE);
56 
57  // Any OpenCL version:
58  // PI_INVALID_WORK_GROUP_SIZE if local_work_size is specified and does not
59  // match the required work-group size for kernel in the program source.
60  if (NDRDesc.LocalSize[0] != CompileWGSize[0] ||
61  NDRDesc.LocalSize[1] != CompileWGSize[1] ||
62  NDRDesc.LocalSize[2] != CompileWGSize[2])
63  throw sycl::nd_range_error(
64  "Specified local size doesn't match the required work-group size "
65  "specified in the program source",
66  PI_INVALID_WORK_GROUP_SIZE);
67  }
68 
69  if (Ver[0] == '1') {
70  // OpenCL 1.x:
71  // PI_INVALID_WORK_GROUP_SIZE if local_work_size is specified and the
72  // total number of work-items in the work-group computed as
73  // local_work_size[0] * ... * local_work_size[work_dim – 1] is greater
74  // than the value specified by PI_DEVICE_MAX_WORK_GROUP_SIZE in
75  // table 4.3
76  size_t MaxWGSize = 0;
77  Plugin.call<PiApiKind::piDeviceGetInfo>(
78  Device, PI_DEVICE_INFO_MAX_WORK_GROUP_SIZE, sizeof(size_t), &MaxWGSize,
79  nullptr);
80  const size_t TotalNumberOfWIs =
81  NDRDesc.LocalSize[0] * NDRDesc.LocalSize[1] * NDRDesc.LocalSize[2];
82  if (TotalNumberOfWIs > MaxWGSize)
83  throw sycl::nd_range_error(
84  "Total number of work-items in a work-group cannot exceed "
85  "info::device::max_work_group_size which is equal to " +
86  std::to_string(MaxWGSize),
87  PI_INVALID_WORK_GROUP_SIZE);
88  } else {
89  // OpenCL 2.x:
90  // PI_INVALID_WORK_GROUP_SIZE if local_work_size is specified and the
91  // total number of work-items in the work-group computed as
92  // local_work_size[0] * ... * local_work_size[work_dim – 1] is greater
93  // than the value specified by PI_KERNEL_GROUP_INFO_WORK_GROUP_SIZE in
94  // table 5.21.
95  size_t KernelWGSize = 0;
96  Plugin.call<PiApiKind::piKernelGetGroupInfo>(
97  Kernel, Device, PI_KERNEL_GROUP_INFO_WORK_GROUP_SIZE, sizeof(size_t),
98  &KernelWGSize, nullptr);
99  const size_t TotalNumberOfWIs =
100  NDRDesc.LocalSize[0] * NDRDesc.LocalSize[1] * NDRDesc.LocalSize[2];
101  if (TotalNumberOfWIs > KernelWGSize)
102  throw sycl::nd_range_error(
103  "Total number of work-items in a work-group cannot exceed "
104  "info::kernel_work_group::work_group_size which is equal to " +
105  std::to_string(KernelWGSize) + " for this kernel",
106  PI_INVALID_WORK_GROUP_SIZE);
107  }
108 
109  if (HasLocalSize) {
110  const bool NonUniformWGs =
111  (NDRDesc.LocalSize[0] != 0 &&
112  NDRDesc.GlobalSize[0] % NDRDesc.LocalSize[0] != 0) ||
113  (NDRDesc.LocalSize[1] != 0 &&
114  NDRDesc.GlobalSize[1] % NDRDesc.LocalSize[1] != 0) ||
115  (NDRDesc.LocalSize[2] != 0 &&
116  NDRDesc.GlobalSize[2] % NDRDesc.LocalSize[2] != 0);
117 
118  if (Ver[0] == '1') {
119  // OpenCL 1.x:
120  // PI_INVALID_WORK_GROUP_SIZE if local_work_size is specified and
121  // number of workitems specified by global_work_size is not evenly
122  // divisible by size of work-group given by local_work_size
123 
124  if (NonUniformWGs)
125  throw sycl::nd_range_error(
126  "Non-uniform work-groups are not supported by the target device",
127  PI_INVALID_WORK_GROUP_SIZE);
128  } else {
129  // OpenCL 2.x:
130  // PI_INVALID_WORK_GROUP_SIZE if the program was compiled with
131  // –cl-uniform-work-group-size and the number of work-items specified
132  // by global_work_size is not evenly divisible by size of work-group
133  // given by local_work_size
134 
135  pi_program Program = nullptr;
136  Plugin.call<PiApiKind::piKernelGetInfo>(
137  Kernel, PI_KERNEL_INFO_PROGRAM, sizeof(pi_program), &Program, nullptr);
138  size_t OptsSize = 0;
139  Plugin.call<PiApiKind::piProgramGetBuildInfo>(
140  Program, Device, PI_PROGRAM_BUILD_INFO_OPTIONS, 0, nullptr, &OptsSize);
141  string_class Opts(OptsSize, '\0');
142  Plugin.call<PiApiKind::piProgramGetBuildInfo>(
143  Program, Device, PI_PROGRAM_BUILD_INFO_OPTIONS, OptsSize, &Opts.front(),
144  nullptr);
145  if (NonUniformWGs) {
146  const bool HasStd20 = Opts.find("-cl-std=CL2.0") != string_class::npos;
147  if (!HasStd20)
148  throw sycl::nd_range_error(
149  "Non-uniform work-groups are not allowed by default. Underlying "
150  "OpenCL 2.x implementation supports this feature and to enable "
151  "it, build device program with -cl-std=CL2.0",
152  PI_INVALID_WORK_GROUP_SIZE);
153  else
154  throw sycl::nd_range_error(
155  "Non-uniform work-groups are not allowed by default. Underlying "
156  "OpenCL 2.x implementation supports this feature, but it is "
157  "disabled by -cl-uniform-work-group-size build flag",
158  PI_INVALID_WORK_GROUP_SIZE);
159  }
160  }
161  }
162 
163  // TODO: required number of sub-groups, OpenCL 2.1:
164  // PI_INVALID_WORK_GROUP_SIZE if local_work_size is specified and is not
165  // consistent with the required number of sub-groups for kernel in the
166  // program source.
167 
168  // Fallback
169  constexpr pi_result Error = PI_INVALID_WORK_GROUP_SIZE;
170  throw runtime_error(
171  "OpenCL API failed. OpenCL API returns: " + codeToString(Error), Error);
172 }

174 bool handleInvalidWorkGroupSize(const device_impl &DeviceImpl, pi_kernel Kernel,
175  const NDRDescT &NDRDesc) {
176  const bool HasLocalSize = (NDRDesc.LocalSize[0] != 0);
177 
178  const plugin &Plugin = DeviceImpl.getPlugin();
179  RT::PiDevice Device = DeviceImpl.getHandleRef();
180 
181  if (HasLocalSize) {
182  size_t MaxThreadsPerBlock[3] = {};
183  Plugin.call<PiApiKind::piDeviceGetInfo>(
184  Device, PI_DEVICE_INFO_MAX_WORK_ITEM_SIZES, sizeof(MaxThreadsPerBlock),
185  MaxThreadsPerBlock, nullptr);
186 
187  for (size_t I = 0; I < 3; ++I) {
188  if (MaxThreadsPerBlock[I] < NDRDesc.LocalSize[I]) {
189  throw sycl::nd_range_error(
190  "The number of work-items in each dimension of a work-group cannot "
191  "exceed info::device::max_work_item_sizes which is {" +
192  std::to_string(MaxThreadsPerBlock[0]) + ", " +
193  std::to_string(MaxThreadsPerBlock[1]) + ", " +
194  std::to_string(MaxThreadsPerBlock[2]) + "} for this device",
195  PI_INVALID_WORK_GROUP_SIZE);
196  }
197  }
198  }
199 
200  // Backend specific invalid work group size handing
201  // TODO: Find a better way to determine the backend
202  std::string PlatformName =
203  DeviceImpl.get_platform().get_info<info::platform::name>();
204  if (PlatformName.find("OpenCL") != std::string::npos) {
205  return oclHandleInvalidWorkGroupSize(DeviceImpl, Kernel, NDRDesc);
206  }
207 
208  // Fallback
209  constexpr pi_result Error = PI_INVALID_WORK_GROUP_SIZE;
210  throw runtime_error(
211  "PI backend failed. PI backend returns: " + codeToString(Error), Error);
212 }

214 bool handleInvalidWorkItemSize(const device_impl &DeviceImpl,
215  const NDRDescT &NDRDesc) {
216 
217  const plugin &Plugin = DeviceImpl.getPlugin();
218  RT::PiDevice Device = DeviceImpl.getHandleRef();
219 
220  size_t MaxWISize[] = {0, 0, 0};
221 
222  Plugin.call<PiApiKind::piDeviceGetInfo>(
223  Device, PI_DEVICE_INFO_MAX_WORK_ITEM_SIZES, sizeof(MaxWISize), &MaxWISize,
224  nullptr);
225  for (unsigned I = 0; I < NDRDesc.Dims; I++) {
226  if (NDRDesc.LocalSize[I] > MaxWISize[I])
227  throw sycl::nd_range_error(
228  "Number of work-items in a work-group exceed limit for dimension " +
229  std::to_string(I) + " : " + std::to_string(NDRDesc.LocalSize[I]) +
230  " > " + std::to_string(MaxWISize[I]),
231  PI_INVALID_WORK_ITEM_SIZE);
232  }
233  return 0;
234 }

236 bool handleError(pi_result Error, const device_impl &DeviceImpl,
237  pi_kernel Kernel, const NDRDescT &NDRDesc) {
238  assert(Error != PI_SUCCESS &&
239  "Success is expected to be handled on caller side");
240  switch (Error) {
241  case PI_INVALID_WORK_GROUP_SIZE:
242  return handleInvalidWorkGroupSize(DeviceImpl, Kernel, NDRDesc);
243 
244  case PI_INVALID_KERNEL_ARGS:
245  throw sycl::nd_range_error(
246  "The kernel argument values have not been specified "
247  " OR "
248  "a kernel argument declared to be a pointer to a type.",
249  PI_INVALID_KERNEL_ARGS);
250 
251  case PI_INVALID_WORK_ITEM_SIZE:
252  return handleInvalidWorkItemSize(DeviceImpl, NDRDesc);
253 
254  case PI_IMAGE_FORMAT_NOT_SUPPORTED:
255  throw sycl::nd_range_error(
256  "image object is specified as an argument value"
257  " and the image format is not supported by device associated"
258  " with queue",
259  PI_IMAGE_FORMAT_NOT_SUPPORTED);
260 
261  case PI_MISALIGNED_SUB_BUFFER_OFFSET:
262  throw sycl::nd_range_error(
263  "a sub-buffer object is specified as the value for an argument "
264  " that is a buffer object and the offset specified "
265  "when the sub-buffer object is created is not aligned "
266  "to CL_DEVICE_MEM_BASE_ADDR_ALIGN value for device associated"
267  " with queue",
268  PI_MISALIGNED_SUB_BUFFER_OFFSET);
269 
270  case PI_MEM_OBJECT_ALLOCATION_FAILURE:
271  throw sycl::nd_range_error(
272  "failure to allocate memory for data store associated with image"
273  " or buffer objects specified as arguments to kernel",
274  PI_MEM_OBJECT_ALLOCATION_FAILURE);
275 
276  case PI_INVALID_IMAGE_SIZE:
277  throw sycl::nd_range_error(
278  "image object is specified as an argument value and the image "
279  "dimensions (image width, height, specified or compute row and/or "
280  "slice pitch) are not supported by device associated with queue",
281  PI_INVALID_IMAGE_SIZE);
282 
283  // TODO: Handle other error codes
284 
285  default:
286  throw runtime_error(
287  "OpenCL API failed. OpenCL API returns: " + codeToString(Error), Error);
288  }
