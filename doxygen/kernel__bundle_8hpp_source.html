507 template <sycl::bundle_state State>
508 sycl::kernel_bundle<State>
509 join(const std::vector<sycl::kernel_bundle<State>> &Bundles) {
510  // Convert kernel_bundle<State> to impls to abstract template parameter away
511  std::vector<detail::KernelBundleImplPtr> KernelBundleImpls;
512  KernelBundleImpls.reserve(Bundles.size());
513  for (const sycl::kernel_bundle<State> &Bundle : Bundles)
514  KernelBundleImpls.push_back(detail::getSyclObjImpl(Bundle));
515 
516  std::shared_ptr<detail::kernel_bundle_impl> Impl =
517  detail::join_impl(KernelBundleImpls);
518  return detail::createSyclObjFromImpl<kernel_bundle<State>>(Impl);
519 }
