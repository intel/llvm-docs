32 queue::queue(const context &syclContext, const device_selector &deviceSelector,
33  const async_handler &asyncHandler, const property_list &propList) {
34 
35  const vector_class<device> Devs = syclContext.get_devices();
36 
37  auto Comp = [&deviceSelector](const device &d1, const device &d2) {
38  return deviceSelector(d1) < deviceSelector(d2);
39  };
40 
41  const device &syclDevice = *std::max_element(Devs.begin(), Devs.end(), Comp);
42 
43  impl = std::make_shared<detail::queue_impl>(
44  detail::getSyclObjImpl(syclDevice), detail::getSyclObjImpl(syclContext),
45  asyncHandler, detail::getQueueOrder(propList), propList);
46 }
