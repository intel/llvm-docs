25 template <info::kernel Param> struct get_kernel_info<string_class, Param> {
26  static string_class get(RT::PiKernel Kernel, const plugin &Plugin) {
27  size_t ResultSize;
28 
29  // TODO catch an exception and put it to list of asynchronous exceptions
30  Plugin.call<PiApiKind::piKernelGetInfo>(Kernel, pi_kernel_info(Param), 0,
31  nullptr, &ResultSize);
32  if (ResultSize == 0) {
33  return "";
34  }
35  vector_class<char> Result(ResultSize);
36  // TODO catch an exception and put it to list of asynchronous exceptions
37  Plugin.call<PiApiKind::piKernelGetInfo>(Kernel, pi_kernel_info(Param),
38  ResultSize, Result.data(), nullptr);
39  return string_class(Result.data());
40  }
41 };
