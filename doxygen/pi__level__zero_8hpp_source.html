177 // The wrapper for immutable Level-Zero data.
178 // The data is initialized only once at first access (via ->) with the
179 // initialization function provided in Init. All subsequent access to
180 // the data just returns the already stored data.
181 //
182 template <class T> struct ZeCache : private T {
183  // The initialization function takes a reference to the data
184  // it is going to initialize, since it is private here in
185  // order to disallow access other than through "->".
186  //
187  typedef std::function<void(T &)> InitFunctionType;
188  InitFunctionType Compute;
189  bool Computed{false};
190 
191  ZeCache() : T{} {}
192 
193  // Access to the fields of the original T data structure.
194  T *operator->() {
195  if (!Computed) {
196  Compute(*this);
197  Computed = true;
198  }
199  return this;
200  }

513 struct _pi_context : _pi_object {
514  _pi_context(ze_context_handle_t ZeContext, pi_uint32 NumDevices,
515  const pi_device *Devs, bool OwnZeContext)
516  : ZeContext{ZeContext}, OwnZeContext{OwnZeContext},
517  Devices{Devs, Devs + NumDevices}, ZeCommandListInit{nullptr} {
518  // NOTE: one must additionally call initialize() to complete
519  // PI context creation.
520 
521  // Create USM allocator context for each pair (device, context).
522  for (uint32_t I = 0; I < NumDevices; I++) {
523  pi_device Device = Devs[I];
524  SharedMemAllocContexts.emplace(
525  std::piecewise_construct, std::make_tuple(Device),
526  std::make_tuple(std::unique_ptr<SystemMemory>(
527  new USMSharedMemoryAlloc(this, Device))));
528  SharedReadOnlyMemAllocContexts.emplace(
529  std::piecewise_construct, std::make_tuple(Device),
530  std::make_tuple(std::unique_ptr<SystemMemory>(
531  new USMSharedReadOnlyMemoryAlloc(this, Device))));
532  DeviceMemAllocContexts.emplace(
533  std::piecewise_construct, std::make_tuple(Device),
534  std::make_tuple(std::unique_ptr<SystemMemory>(
535  new USMDeviceMemoryAlloc(this, Device))));
536  }
537  // Create USM allocator context for host. Device and Shared USM allocations
538  // are device-specific. Host allocations are not device-dependent therefore
539  // we don't need a map with device as key.
540  HostMemAllocContext = std::make_unique<USMAllocContext>(
541  std::unique_ptr<SystemMemory>(new USMHostMemoryAlloc(this)));
542 
543  if (NumDevices == 1) {
544  SingleRootDevice = Devices[0];
545  } else {
546 
547  // Check if we have context with subdevices of the same device (context
548  // may include root device itself as well)
549  SingleRootDevice =
550  Devices[0]->RootDevice ? Devices[0]->RootDevice : Devices[0];
551 
552  // For context with sub subdevices, the SingleRootDevice might still
553  // not be the root device.
554  // Check whether the SingleRootDevice is the subdevice or root device.
555  if (SingleRootDevice->isSubDevice()) {
556  SingleRootDevice = SingleRootDevice->RootDevice;
557  }
558 
559  for (auto &Device : Devices) {
560  if ((!Device->RootDevice && Device != SingleRootDevice) ||
561  (Device->RootDevice && Device->RootDevice != SingleRootDevice)) {
562  SingleRootDevice = nullptr;
563  break;
564  }
565  }
566  }
567 
568  // We may allocate memory to this root device so create allocators.
569  if (SingleRootDevice && DeviceMemAllocContexts.find(SingleRootDevice) ==
570  DeviceMemAllocContexts.end()) {
571  SharedMemAllocContexts.emplace(
572  std::piecewise_construct, std::make_tuple(SingleRootDevice),
573  std::make_tuple(std::unique_ptr<SystemMemory>(
574  new USMSharedMemoryAlloc(this, SingleRootDevice))));
575  DeviceMemAllocContexts.emplace(
576  std::piecewise_construct, std::make_tuple(SingleRootDevice),
577  std::make_tuple(std::unique_ptr<SystemMemory>(
578  new USMDeviceMemoryAlloc(this, SingleRootDevice))));
579  }
580  }
