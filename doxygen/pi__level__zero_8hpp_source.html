482 struct _pi_program : _pi_object {
483  // Possible states of a program.
484  typedef enum {
485  // The program has been created from intermediate language (SPIR-v), but it
486  // is not yet compiled.
487  IL,
488 
489  // The program has been created by loading native code, but it has not yet
490  // been built. This is equivalent to an OpenCL "program executable" that
491  // is loaded via clCreateProgramWithBinary().
492  Native,
493 
494  // The program consists of native code (typically compiled from SPIR-v),
495  // but it has unresolved external dependencies which need to be resolved
496  // by linking with other Object state program(s). Programs in this state
497  // have a single Level Zero module.
498  Object,
499 
500  // The program consists of native code with no external dependencies.
501  // Programs in this state have a single Level Zero module, but no linking
502  // is needed in order to run kernels.
503  Exe,
504 
505  // The program consists of several Level Zero modules, each of which
506  // contains native code. Some modules may import external symbols and
507  // other modules may export definitions of those external symbols. All of
508  // the modules have been linked together, so the imported references are
509  // resolved by the exported definitions.
510  //
511  // Module linking in Level Zero is quite different from program linking in
512  // OpenCL. OpenCL statically links several program objects together to
513  // form a new program that contains the linked result. Level Zero is more
514  // similar to shared libraries. When several Level Zero modules are linked
515  // together, each module is modified "in place" such that external
516  // references from one are linked to external definitions in another.
517  // Linking in Level Zero does not produce a new Level Zero module that
518  // represents the linked result, therefore a program in LinkedExe state
519  // holds a list of all the pi_programs that were linked together. Queries
520  // about the linked program need to query all the pi_programs in this list.
521  LinkedExe
522  } state;
523 
524  // This is a wrapper class used for programs in LinkedExe state. Such a
525  // program contains a list of pi_programs in Object state that have been
526  // linked together. The program in LinkedExe state increments the reference
527  // counter for each of the Object state programs, thus "retaining" a
528  // reference to them, and it may also set the "HasImportsAndIsLinked" flag
529  // in these Object state programs. The purpose of this wrapper is to
530  // decrement the reference count and clear the flag when the LinkedExe
531  // program is destroyed, so all the interesting code is in the wrapper's
532  // destructor.
533  //
534  // In order to ensure that the reference count is never decremented more
535  // than once, the wrapper has no copy constructor or copy assignment
536  // operator. Instead, we only allow move semantics for the wrapper.
537  class LinkedReleaser {
538  public:
539  LinkedReleaser(pi_program Prog) : Prog(Prog) {}
540  LinkedReleaser(LinkedReleaser &&Other) {
541  Prog = Other.Prog;
542  Other.Prog = nullptr;
543  }
544  LinkedReleaser(const LinkedReleaser &Other) = delete;
545  LinkedReleaser &operator=(LinkedReleaser &&Other) {
546  std::swap(Prog, Other.Prog);
547  return *this;
548  }
549  LinkedReleaser &operator=(const LinkedReleaser &Other) = delete;
550  ~LinkedReleaser();
551 
552  pi_program operator->() const { return Prog; }
553 
554  private:
555  pi_program Prog;
556  };
557 
558  // A utility class that iterates over the Level Zero modules contained by
559  // the program. This helps hide the difference between programs in Object
560  // or Exe state (which have one module) and programs in LinkedExe state
561  // (which have several modules).
562  class ModuleIterator {
563  public:
564  ModuleIterator(pi_program Prog)
565  : Prog(Prog), It(Prog->LinkedPrograms.begin()) {
566  if (Prog->State == LinkedExe) {
567  NumMods = Prog->LinkedPrograms.size();
568  IsDone = (It == Prog->LinkedPrograms.end());
569  Mod = IsDone ? nullptr : (*It)->ZeModule;
570  } else if (Prog->State == IL || Prog->State == Native) {
571  NumMods = 0;
572  IsDone = true;
573  Mod = nullptr;
574  } else {
575  NumMods = 1;
576  IsDone = false;
577  Mod = Prog->ZeModule;
578  }
579  }
580 
581  bool Done() const { return IsDone; }
582  size_t Count() const { return NumMods; }
583  ze_module_handle_t operator*() const { return Mod; }
584 
585  void operator++(int) {
586  if (!IsDone && (Prog->State == LinkedExe) &&
587  (++It != Prog->LinkedPrograms.end())) {
588  Mod = (*It)->ZeModule;
589  } else {
590  Mod = nullptr;
591  IsDone = true;
592  }
593  }
594 
595  private:
596  pi_program Prog;
597  ze_module_handle_t Mod;
598  size_t NumMods;
599  bool IsDone;
600  std::vector<LinkedReleaser>::iterator It;
601  };
