18 inline backend_return_t<backend::ext_oneapi_hip, device>
19 get_native<backend::ext_oneapi_hip, device>(const device &Obj) {
20  // TODO swap with SYCL 2020 exception when in ABI-break window
21  if (Obj.get_backend() != backend::ext_oneapi_hip) {
22  throw sycl::runtime_error(errc::backend_mismatch, "Backends mismatch",
23  PI_ERROR_INVALID_OPERATION);
24  }
25  // HIP uses a 32-bit int instead of an opaque pointer like other backends,
26  // so we need a specialization with static_cast instead of reinterpret_cast.
27  return static_cast<backend_return_t<backend::ext_oneapi_hip, device>>(
28  Obj.getNative());
29 }

31 template <>
32 inline device make_device<backend::ext_oneapi_hip>(
33  const backend_input_t<backend::ext_oneapi_hip, device> &BackendObject) {
34  auto devs = device::get_devices(info::device_type::gpu);
35  for (auto &dev : devs) {
36  if (dev.get_backend() == backend::ext_oneapi_hip &&
37  BackendObject == get_native<backend::ext_oneapi_hip>(dev)) {
38  return dev;
39  }
40  }
41  // The ext_oneapi_hip platform(s) adds all n available devices where n
42  // is returned from call to `hipGetDeviceCount`.
43  // Hence if this code is reached then the requested device ordinal must
44  // not be visible to the driver.
45  throw sycl::exception(make_error_code(errc::invalid),
46  "Native device has an invalid ordinal.");
