226 __ESIMD_API void named_barrier_signal(uint8_t barrier_id,
227  uint8_t producer_consumer_mode,
228  uint32_t num_producers,
229  uint32_t num_consumers) {
230  constexpr uint32_t gateway = 3;
231  constexpr uint32_t barrier = 4;
232  constexpr uint32_t descriptor = 1 << 25 | // Message length: 1 register
233  0 << 12 | // Fence Data Ports: No fence
234  barrier; // Barrier subfunction
235 
236  __ESIMD_DNS::vector_type_t<uint32_t, 8> payload = 0;
237  payload[2] = (num_consumers & 0xff) << 24 | (num_producers & 0xff) << 16 |
238  producer_consumer_mode << 14 | (barrier_id & 0b11111) << 0;
239 
240  __esimd_raw_send_nbarrier_signal<uint32_t, 8>(
241  0 /*sendc*/, gateway, descriptor, payload, 1 /*pred*/);
242 }

254 template <typename T, int NBlocks, int Height, int Width, bool Transposed,
255  bool Transformed>

257  if (Transformed)

260  return Width * Height * NBlocks;
