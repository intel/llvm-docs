174 std::string memFlagsToString(pi_mem_flags Flags) {
175  std::stringstream Sstream;
176  bool FoundFlag = false;
177 
178  auto FlagSeparator = [](bool FoundFlag) { return FoundFlag ? "|" : ""; };
179 
180  pi_mem_flags ValidFlags[] = {PI_MEM_FLAGS_ACCESS_RW,
181  PI_MEM_FLAGS_HOST_PTR_USE,
182  PI_MEM_FLAGS_HOST_PTR_COPY};
183 
184  if (Flags == 0u) {
185  Sstream << "pi_mem_flags(0)";
186  } else {
187  for (const auto Flag : ValidFlags) {
188  if (Flag & Flags) {
189  Sstream << FlagSeparator(FoundFlag) << memFlagToString(Flag);
190  FoundFlag = true;
191  }
192  }
193 
194  std::bitset<64> UnkownBits(Flags & ~(PI_MEM_FLAGS_ACCESS_RW |
195  PI_MEM_FLAGS_HOST_PTR_USE |
196  PI_MEM_FLAGS_HOST_PTR_COPY));
197  if (UnkownBits.any()) {
198  Sstream << FlagSeparator(FoundFlag)
199  << "unknown pi_mem_flags bits == " << UnkownBits;
200  }
201  }
202 
203  return Sstream.str();
204 }

258 // Initializes all available Plugins.
259 const vector_class<plugin> &initialize() {
260  static std::once_flag PluginsInitDone;
261  static vector_class<plugin> *Plugins = nullptr;
262 
263  std::call_once(PluginsInitDone, []() {
264  // The memory for "Plugins" is intentionally leaked because the application
265  // may call into the SYCL runtime from a global destructor, and such a call
266  // could eventually call down to initialize(). Therefore, there is no safe
267  // time when "Plugins" could be deleted.
268  Plugins = new vector_class<plugin>;
269  initializePlugins(Plugins);
270  });
271 
272  return *Plugins;
273 }

275 static void initializePlugins(vector_class<plugin> *Plugins) {
276  vector_class<std::pair<std::string, backend>> PluginNames;
277  findPlugins(PluginNames);
278 
279  if (PluginNames.empty() && trace(PI_TRACE_ALL))
280  std::cerr << "SYCL_PI_TRACE[all]: "
281  << "No Plugins Found." << std::endl;
282 
283  PiPlugin PluginInformation{_PI_H_VERSION_STRING, _PI_H_VERSION_STRING,
284  nullptr};
285 
286  for (unsigned int I = 0; I < PluginNames.size(); I++) {
287  void *Library = loadPlugin(PluginNames[I].first);
288 
289  if (!Library) {
290  if (trace(PI_TRACE_ALL)) {
291  std::cerr << "SYCL_PI_TRACE[all]: "
292  << "Check if plugin is present. "
293  << "Failed to load plugin: " << PluginNames[I].first
294  << std::endl;
295  }
296  continue;
297  }
298 
299  if (!bindPlugin(Library, &PluginInformation)) {
300  if (trace(PI_TRACE_ALL)) {
301  std::cerr << "SYCL_PI_TRACE[all]: "
302  << "Failed to bind PI APIs to the plugin: "
303  << PluginNames[I].first << std::endl;
304  }
305  continue;
306  }
307  backend *BE = SYCLConfig<SYCL_BE>::get();
308  // Use OpenCL as the default interoperability plugin.
309  // This will go away when we make backend interoperability selection
310  // explicit in SYCL-2020.
311  backend InteropBE = BE ? *BE : backend::opencl;
312 
313  if (InteropBE == backend::opencl &&
314  PluginNames[I].first.find("opencl") != std::string::npos) {
315  // Use the OpenCL plugin as the GlobalPlugin
316  GlobalPlugin =
317  std::make_shared<plugin>(PluginInformation, backend::opencl);
318  } else if (InteropBE == backend::cuda &&
319  PluginNames[I].first.find("cuda") != std::string::npos) {
320  // Use the CUDA plugin as the GlobalPlugin
321  GlobalPlugin = std::make_shared<plugin>(PluginInformation, backend::cuda);
322  }
323  Plugins->emplace_back(plugin(PluginInformation, PluginNames[I].second));
324  if (trace(TraceLevel::PI_TRACE_BASIC))
325  std::cerr << "SYCL_PI_TRACE[basic]: "
326  << "Plugin found and successfully loaded: "
327  << PluginNames[I].first << std::endl;
328  }
329 
330 #ifdef XPTI_ENABLE_INSTRUMENTATION
331  if (!(xptiTraceEnabled() && !XPTIInitDone))
332  return;
333  // Not sure this is the best place to initialize the framework; SYCL runtime
334  // team needs to advise on the right place, until then we piggy-back on the
335  // initialization of the PI layer.
336 
337  // Initialize the global events just once, in the case pi::initialize() is
338  // called multiple times
339  XPTIInitDone = true;
340  // Registers a new stream for 'sycl' and any plugin that wants to listen to
341  // this stream will register itself using this string or stream ID for this
342  // string.
343  uint8_t StreamID = xptiRegisterStream(SYCL_STREAM_NAME);
344  // Let all tool plugins know that a stream by the name of 'sycl' has been
345  // initialized and will be generating the trace stream.
346  //
347  // +--- Minor version #
348  // Major version # ------+ | Version string
349  // | | |
350  // v v v
351  xptiInitialize(SYCL_STREAM_NAME, GMajVer, GMinVer, GVerStr);
352  // Create a tracepoint to indicate the graph creation
353  xpti::payload_t GraphPayload("application_graph");
354  uint64_t GraphInstanceNo;
355  GSYCLGraphEvent =
356  xptiMakeEvent("application_graph", &GraphPayload, xpti::trace_graph_event,
357  xpti_at::active, &GraphInstanceNo);
358  if (GSYCLGraphEvent) {
359  // The graph event is a global event and will be used as the parent for
360  // all nodes (command groups)
361  xptiNotifySubscribers(StreamID, xpti::trace_graph_create, nullptr,
362  GSYCLGraphEvent, GraphInstanceNo, nullptr);
363  }
364 
365  // Let subscribers know a new stream is being initialized
366  xptiInitialize(SYCL_PICALL_STREAM_NAME, GMajVer, GMinVer, GVerStr);
367  xpti::payload_t PIPayload("Plugin Interface Layer");
368  uint64_t PiInstanceNo;
369  GPICallEvent =
370  xptiMakeEvent("PI Layer", &PIPayload, xpti::trace_algorithm_event,
371  xpti_at::active, &PiInstanceNo);
372 #endif
373 }
