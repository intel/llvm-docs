162 constexpr int is_channel_enabled(rgba_channel_mask M, rgba_channel Ch) {
163  int Pos = static_cast<int>(Ch);
164  return (static_cast<int>(M) & (1 << Pos)) >> Pos;
165 }
166 
167 constexpr int get_num_channels_enabled(rgba_channel_mask M) {
168  return is_channel_enabled(M, rgba_channel::R) +
169  is_channel_enabled(M, rgba_channel::G) +
170  is_channel_enabled(M, rgba_channel::B) +
171  is_channel_enabled(M, rgba_channel::A);
172 }
