150 template <int Dims, typename T>
151 typename detail::enable_if_t<std::is_same<T, range<Dims>>::value ||
152  std::is_same<T, id<Dims>>::value>
153 checkValueRange(const T &V) {
154 #if __SYCL_ID_QUERIES_FIT_IN_INT__
155  for (size_t Dim = 0; Dim < Dims; ++Dim)
156  checkValueRangeImpl<T>(V[Dim]);
157 
158  {
159  unsigned long long Product = 1;
160  for (size_t Dim = 0; Dim < Dims; ++Dim) {
161  Product *= V[Dim];
162  // check value now to prevent product overflow in the end
163  checkValueRangeImpl<T>(Product);
164  }
165  }
166 #else
167  (void)V;
168 #endif

724  template <typename KernelName, typename KernelType, int Dims>
725  void parallel_for_lambda_impl(range<Dims> NumWorkItems,
726  KernelType KernelFunc) {
727  throwIfActionIsCreated();
728  using LambdaArgType = sycl::detail::lambda_arg_type<KernelType, item<Dims>>;
729 
730  // If 1D kernel argument is an integral type, convert it to sycl::item<1>
731  using TransformedArgType =
732  typename std::conditional<std::is_integral<LambdaArgType>::value &&
733  Dims == 1,
734  item<Dims>, LambdaArgType>::type;
735  using NameT =
736  typename detail::get_kernel_name_t<KernelName, KernelType>::name;
737 
738  // Range rounding can be disabled by the user.
739  // Range rounding is not done on the host device.
740  // Range rounding is supported only for newer SYCL standards.
741 #if !defined(__SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING__) && \
742  !defined(DPCPP_HOST_DEVICE_OPENMP) && \
743  !defined(DPCPP_HOST_DEVICE_PERF_NATIVE) && SYCL_LANGUAGE_VERSION >= 202001
744  // Range should be a multiple of this for reasonable performance.
745  size_t MinFactorX = 16;
746  // Range should be a multiple of this for improved performance.
747  size_t GoodFactorX = 32;
748  // Range should be at least this to make rounding worthwhile.
749  size_t MinRangeX = 1024;
750 
751  // Parse optional parameters of this form:
752  // MinRound:PreferredRound:MinRange
753  char *RoundParams = getenv("SYCL_PARALLEL_FOR_RANGE_ROUNDING_PARAMS");
754  if (RoundParams != nullptr) {
755  std::string Params(RoundParams);
756  size_t Pos = Params.find(':');
757  if (Pos != std::string::npos) {
758  MinFactorX = std::stoi(Params.substr(0, Pos));
759  Params.erase(0, Pos + 1);
760  Pos = Params.find(':');
761  if (Pos != std::string::npos) {
762  GoodFactorX = std::stoi(Params.substr(0, Pos));
763  Params.erase(0, Pos + 1);
764  MinRangeX = std::stoi(Params);
765  }
766  }
767  }
768 
769  // Disable the rounding-up optimizations under these conditions:
770  // 1. The env var SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING is set.
771  // 2. The string SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING is in
772  // the kernel name.
773  // 3. The kernel is provided via an interoperability method.
774  // 4. The API "this_item" is used inside the kernel.
775  // 5. The range is already a multiple of the rounding factor.
776  //
777  // Cases 3 and 4 could be supported with extra effort.
778  // As an optimization for the common case it is an
779  // implementation choice to not support those scenarios.
780  // Note that "this_item" is a free function, i.e. not tied to any
781  // specific id or item. When concurrent parallel_fors are executing
782  // on a device it is difficult to tell which parallel_for the call is
783  // being made from. One could replicate portions of the
784  // call-graph to make this_item calls kernel-specific but this is
785  // not considered worthwhile.
786 
787  // Get the kernal name to check condition 3.
788  std::string KName = typeid(NameT *).name();
789  using KI = detail::KernelInfo<KernelName>;
790  bool DisableRounding =
791  (getenv("SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING") != nullptr) ||
792  (KName.find("SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING") !=
793  std::string::npos) ||
794  (KI::getName() == nullptr || KI::getName()[0] == '\0') ||
795  (KI::callsThisItem());
796 
797  // Perform range rounding if rounding-up is enabled
798  // and there are sufficient work-items to need rounding
799  // and the user-specified range is not a multiple of a "good" value.
800  if (!DisableRounding && (NumWorkItems[0] >= MinRangeX) &&
801  (NumWorkItems[0] % MinFactorX != 0)) {
802  // It is sufficient to round up just the first dimension.
803  // Multiplying the rounded-up value of the first dimension
804  // by the values of the remaining dimensions (if any)
805  // will yield a rounded-up value for the total range.
806  size_t NewValX =
807  ((NumWorkItems[0] + GoodFactorX - 1) / GoodFactorX) * GoodFactorX;
808  using NameWT = typename detail::get_kernel_wrapper_name_t<NameT>::name;
809  if (getenv("SYCL_PARALLEL_FOR_RANGE_ROUNDING_TRACE") != nullptr)
810  std::cout << "parallel_for range adjusted from " << NumWorkItems[0]
811  << " to " << NewValX << std::endl;
812 
813  auto Wrapper = getRangeRoundedKernelLambda<TransformedArgType, Dims>(
814  KernelFunc, NumWorkItems);
815 
816  range<Dims> AdjustedRange = NumWorkItems;
817  AdjustedRange.set_range_dim0(NewValX);
818 #ifdef __SYCL_DEVICE_ONLY__
819  kernel_parallel_for_wrapper<NameWT, TransformedArgType>(Wrapper);
820 #else
821  detail::checkValueRange<Dims>(AdjustedRange);
822  MNDRDesc.set(std::move(AdjustedRange));
823  StoreLambda<NameWT, decltype(Wrapper), Dims, TransformedArgType>(
824  std::move(Wrapper));
825  MCGType = detail::CG::KERNEL_V1;
826 #endif
827  } else
828 #endif // !__SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING__ && \
829  // !DPCPP_HOST_DEVICE_OPENMP && !DPCPP_HOST_DEVICE_PERF_NATIVE && \
830  // SYCL_LANGUAGE_VERSION >= 202001
831  {
832 #ifdef __SYCL_DEVICE_ONLY__
833  (void)NumWorkItems;
834  kernel_parallel_for_wrapper<NameT, TransformedArgType>(KernelFunc);
835 #else
836  detail::checkValueRange<Dims>(NumWorkItems);
837  MNDRDesc.set(std::move(NumWorkItems));
838  StoreLambda<NameT, KernelType, Dims, TransformedArgType>(
839  std::move(KernelFunc));
840  MCGType = detail::CG::KERNEL_V1;
841 #endif
842  }
843  }

1369  template <typename KernelName = detail::auto_name, typename KernelType,
1370  int Dims, typename Reduction>
1371  detail::enable_if_t<!Reduction::has_fast_atomics>
1372  parallel_for(nd_range<Dims> Range, Reduction Redu,
1373  _KERNELFUNCPARAM(KernelFunc)) {
1374  // This parallel_for() is lowered to the following sequence:
1375  // 1) Call a kernel that a) call user's lambda function and b) performs
1376  // one iteration of reduction, storing the partial reductions/sums
1377  // to either a newly created global buffer or to user's reduction
1378  // accessor. So, if the original 'Range' has totally
1379  // N1 elements and work-group size is W, then after the first iteration
1380  // there will be N2 partial sums where N2 = N1 / W.
1381  // If (N2 == 1) then the partial sum is written to user's accessor.
1382  // Otherwise, a new global buffer is created and partial sums are written
1383  // to it.
1384  // 2) Call an aux kernel (if necessary, i.e. if N2 > 1) as many times as
1385  // necessary to reduce all partial sums into one final sum.
1386 
1387  // Before running the kernels, check that device has enough local memory
1388  // to hold local arrays that may be required for the reduction algorithm.
1389  // TODO: If the work-group-size is limited by the local memory, then
1390  // a special version of the main kernel may be created. The one that would
1391  // not use local accessors, which means it would not do the reduction in
1392  // the main kernel, but simply generate Range.get_global_range.size() number
1393  // of partial sums, leaving the reduction work to the additional/aux
1394  // kernels.
1395  constexpr bool HFR = Reduction::has_fast_reduce;
1396  size_t OneElemSize = HFR ? 0 : sizeof(typename Reduction::result_type);
1397  // TODO: currently the maximal work group size is determined for the given
1398  // queue/device, while it may be safer to use queries to the kernel compiled
1399  // for the device.
1400  size_t MaxWGSize = ONEAPI::detail::reduGetMaxWGSize(MQueue, OneElemSize);
1401  if (Range.get_local_range().size() > MaxWGSize)
1402  throw sycl::runtime_error("The implementation handling parallel_for with"
1403  " reduction requires work group size not bigger"
1404  " than " +
1405  std::to_string(MaxWGSize),
1406  PI_INVALID_WORK_GROUP_SIZE);
1407 
1408  // 1. Call the kernel that includes user's lambda function.
1409  ONEAPI::detail::reduCGFunc<KernelName>(*this, KernelFunc, Range, Redu);
1410  shared_ptr_class<detail::queue_impl> QueueCopy = MQueue;
1411  this->finalize();
1412 
1413  // 2. Run the additional kernel as many times as needed to reduce
1414  // all partial sums into one scalar.
1415 
1416  // TODO: Create a special slow/sequential version of the kernel that would
1417  // handle the reduction instead of reporting an assert below.
1418  if (MaxWGSize <= 1)
1419  throw sycl::runtime_error("The implementation handling parallel_for with "
1420  "reduction requires the maximal work group "
1421  "size to be greater than 1 to converge. "
1422  "The maximal work group size depends on the "
1423  "device and the size of the objects passed to "
1424  "the reduction.",
1425  PI_INVALID_WORK_GROUP_SIZE);
1426  size_t NWorkItems = Range.get_group_range().size();
1427  while (NWorkItems > 1) {
1428  handler AuxHandler(QueueCopy, MIsHost);
1429  AuxHandler.saveCodeLoc(MCodeLoc);
1430 
1431  NWorkItems = ONEAPI::detail::reduAuxCGFunc<KernelName, KernelType>(
1432  AuxHandler, NWorkItems, MaxWGSize, Redu);
1433  MLastEvent = AuxHandler.finalize();
1434  } // end while (NWorkItems > 1)
1435 
1436  if (Reduction::is_usm || Redu.hasUserDiscardWriteAccessor()) {
1437  handler CopyHandler(QueueCopy, MIsHost);
1438  CopyHandler.saveCodeLoc(MCodeLoc);
1439  ONEAPI::detail::reduSaveFinalResultToUserMem<KernelName>(CopyHandler,
1440  Redu);
1441  MLastEvent = CopyHandler.finalize();
1442  }
1443  }

1445  // This version of parallel_for may handle one or more reductions packed in
1446  // \p Rest argument. Note thought that the last element in \p Rest pack is
1447  // the kernel function.
1448  // TODO: this variant is currently enabled for 2+ reductions only as the
1449  // versions handling 1 reduction variable are more efficient right now.
1450  //
1451  // Algorithm:
1452  // 1) discard_write accessor (DWAcc), InitializeToIdentity = true:
1453  // a) Create uninitialized buffer and read_write accessor (RWAcc).
1454  // b) discard-write partial sums to RWAcc.
1455  // c) Repeat the steps (a) and (b) to get one final sum.
1456  // d) Copy RWAcc to DWAcc.
1457  // 2) read_write accessor (RWAcc), InitializeToIdentity = false:
1458  // a) Create new uninitialized buffer (if #work-groups > 1) and RWAcc or
1459  // re-use user's RWAcc (if #work-groups is 1).
1460  // b) discard-write to RWAcc (#WG > 1), or update-write (#WG == 1).
1461  // c) Repeat the steps (a) and (b) to get one final sum.
1462  // 3) read_write accessor (RWAcc), InitializeToIdentity = true:
1463  // a) Create new uninitialized buffer (if #work-groups > 1) and RWAcc or
1464  // re-use user's RWAcc (if #work-groups is 1).
1465  // b) discard-write to RWAcc.
1466  // c) Repeat the steps (a) and (b) to get one final sum.
1467  // 4) USM pointer, InitializeToIdentity = false:
1468  // a) Create new uninitialized buffer (if #work-groups > 1) and RWAcc or
1469  // re-use user's USM pointer (if #work-groups is 1).
1470  // b) discard-write to RWAcc (#WG > 1) or
1471  // update-write to USM pointer (#WG == 1).
1472  // c) Repeat the steps (a) and (b) to get one final sum.
1473  // 5) USM pointer, InitializeToIdentity = true:
1474  // a) Create new uninitialized buffer (if #work-groups > 1) and RWAcc or
1475  // re-use user's USM pointer (if #work-groups is 1).
1476  // b) discard-write to RWAcc (#WG > 1) or
1477  // discard-write to USM pointer (#WG == 1).
1478  // c) Repeat the steps (a) and (b) to get one final sum.

1493  // TODO: currently the maximal work group size is determined for the given
1494  // queue/device, while it is safer to use queries to the kernel compiled
1495  // for the device.
1496  size_t MaxWGSize =
1497  ONEAPI::detail::reduGetMaxWGSize(MQueue, LocalMemPerWorkItem);
1498  if (Range.get_local_range().size() > MaxWGSize)
1499  throw sycl::runtime_error("The implementation handling parallel_for with"
1500  " reduction requires work group size not bigger"
1501  " than " +
1502  std::to_string(MaxWGSize),
1503  PI_INVALID_WORK_GROUP_SIZE);
1504 
1505  ONEAPI::detail::reduCGFunc<KernelName>(*this, KernelFunc, Range, ReduTuple,
