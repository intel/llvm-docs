150 template <int Dims, typename T>
151 typename detail::enable_if_t<std::is_same<T, range<Dims>>::value ||
152  std::is_same<T, id<Dims>>::value>
153 checkValueRange(const T &V) {
154 #if __SYCL_ID_QUERIES_FIT_IN_INT__
155  for (size_t Dim = 0; Dim < Dims; ++Dim)
156  checkValueRangeImpl<T>(V[Dim]);
157 
158  {
159  unsigned long long Product = 1;
160  for (size_t Dim = 0; Dim < Dims; ++Dim) {
161  Product *= V[Dim];
162  // check value now to prevent product overflow in the end
163  checkValueRangeImpl<T>(Product);
164  }
165  }
166 #else
167  (void)V;
168 #endif
169 }

951  template <typename KernelName, typename KernelType, int Dims>
952  void parallel_for_lambda_impl(range<Dims> NumWorkItems,
953  KernelType KernelFunc) {
954  throwIfActionIsCreated();
955  using LambdaArgType = sycl::detail::lambda_arg_type<KernelType, item<Dims>>;
956 
957  // If 1D kernel argument is an integral type, convert it to sycl::item<1>
958  // If user type is convertible from sycl::item/sycl::nd_item, use
959  // sycl::item/sycl::nd_item to transport item information
960  using TransformedArgType = typename std::conditional<
961  std::is_integral<LambdaArgType>::value && Dims == 1, item<Dims>,
962  typename TransformUserItemType<Dims, LambdaArgType>::type>::type;
963 
964  using NameT =
965  typename detail::get_kernel_name_t<KernelName, KernelType>::name;
966 
967  verifyUsedKernelBundle(detail::KernelInfo<NameT>::getName());
968 
969  // Range rounding can be disabled by the user.
970  // Range rounding is not done on the host device.
971  // Range rounding is supported only for newer SYCL standards.
972 #if !defined(__SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING__) && \
973  !defined(DPCPP_HOST_DEVICE_OPENMP) && \
974  !defined(DPCPP_HOST_DEVICE_PERF_NATIVE) && SYCL_LANGUAGE_VERSION >= 202001
975  // Range should be a multiple of this for reasonable performance.
976  size_t MinFactorX = 16;
977  // Range should be a multiple of this for improved performance.
978  size_t GoodFactorX = 32;
979  // Range should be at least this to make rounding worthwhile.
980  size_t MinRangeX = 1024;
981 
982  // Check if rounding parameters have been set through environment:
983  // SYCL_PARALLEL_FOR_RANGE_ROUNDING_PARAMS=MinRound:PreferredRound:MinRange
984  this->GetRangeRoundingSettings(MinFactorX, GoodFactorX, MinRangeX);
985 
986  // Disable the rounding-up optimizations under these conditions:
987  // 1. The env var SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING is set.
988  // 2. The kernel is provided via an interoperability method.
989  // 3. The range is already a multiple of the rounding factor.
990  //
991  // Cases 2 and 3 could be supported with extra effort.
992  // As an optimization for the common case it is an
993  // implementation choice to not support those scenarios.
994  // Note that "this_item" is a free function, i.e. not tied to any
995  // specific id or item. When concurrent parallel_fors are executing
996  // on a device it is difficult to tell which parallel_for the call is
997  // being made from. One could replicate portions of the
998  // call-graph to make this_item calls kernel-specific but this is
999  // not considered worthwhile.
1000 
1001  // Get the kernel name to check condition 2.
1002  std::string KName = typeid(NameT *).name();
1003  using KI = detail::KernelInfo<KernelName>;
1004  bool DisableRounding =
1005  this->DisableRangeRounding() ||
1006  (KI::getName() == nullptr || KI::getName()[0] == '\0');
1007 
1008  // Perform range rounding if rounding-up is enabled
1009  // and there are sufficient work-items to need rounding
1010  // and the user-specified range is not a multiple of a "good" value.
1011  if (!DisableRounding && (NumWorkItems[0] >= MinRangeX) &&
1012  (NumWorkItems[0] % MinFactorX != 0)) {
1013  // It is sufficient to round up just the first dimension.
1014  // Multiplying the rounded-up value of the first dimension
1015  // by the values of the remaining dimensions (if any)
1016  // will yield a rounded-up value for the total range.
1017  size_t NewValX =
1018  ((NumWorkItems[0] + GoodFactorX - 1) / GoodFactorX) * GoodFactorX;
1019  if (this->RangeRoundingTrace())
1020  std::cout << "parallel_for range adjusted from " << NumWorkItems[0]
1021  << " to " << NewValX << std::endl;
1022 
1023  using NameWT = typename detail::get_kernel_wrapper_name_t<NameT>::name;
1024  auto Wrapper =
1025  getRangeRoundedKernelLambda<NameWT, TransformedArgType, Dims>(
1026  KernelFunc, NumWorkItems);
1027 
1028  using KName = std::conditional_t<std::is_same<KernelType, NameT>::value,
1029  decltype(Wrapper), NameWT>;
1030 
1031  range<Dims> AdjustedRange = NumWorkItems;
1032  AdjustedRange.set_range_dim0(NewValX);
1033  kernel_parallel_for_wrapper<KName, TransformedArgType>(Wrapper);
1034 #ifndef __SYCL_DEVICE_ONLY__
1035  detail::checkValueRange<Dims>(AdjustedRange);
1036  MNDRDesc.set(std::move(AdjustedRange));
1037  StoreLambda<KName, decltype(Wrapper), Dims, TransformedArgType>(
1038  std::move(Wrapper));
1039  setType(detail::CG::Kernel);
1040 #endif
1041  } else
1042 #endif // !__SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING__ &&
1043  // !DPCPP_HOST_DEVICE_OPENMP && !DPCPP_HOST_DEVICE_PERF_NATIVE &&
1044  // SYCL_LANGUAGE_VERSION >= 202001
1045  {
1046  (void)NumWorkItems;
1047  kernel_parallel_for_wrapper<NameT, TransformedArgType>(KernelFunc);
1048 #ifndef __SYCL_DEVICE_ONLY__
1049  detail::checkValueRange<Dims>(NumWorkItems);
1050  MNDRDesc.set(std::move(NumWorkItems));
1051  StoreLambda<NameT, KernelType, Dims, TransformedArgType>(
1052  std::move(KernelFunc));
1053  setType(detail::CG::Kernel);
1054 #endif
1055  }
1056  }

1735  template <typename KernelName, typename KernelType, int Dims,
1736  typename Reduction>
1737  detail::enable_if_t<!Reduction::has_fast_atomics>
1738  parallel_for_Impl(nd_range<Dims> Range, Reduction Redu,
1739  KernelType KernelFunc) {
1740  // This parallel_for() is lowered to the following sequence:
1741  // 1) Call a kernel that a) call user's lambda function and b) performs
1742  // one iteration of reduction, storing the partial reductions/sums
1743  // to either a newly created global buffer or to user's reduction
1744  // accessor. So, if the original 'Range' has totally
1745  // N1 elements and work-group size is W, then after the first iteration
1746  // there will be N2 partial sums where N2 = N1 / W.
1747  // If (N2 == 1) then the partial sum is written to user's accessor.
1748  // Otherwise, a new global buffer is created and partial sums are written
1749  // to it.
1750  // 2) Call an aux kernel (if necessary, i.e. if N2 > 1) as many times as
1751  // necessary to reduce all partial sums into one final sum.
1752 
1753  // Before running the kernels, check that device has enough local memory
1754  // to hold local arrays that may be required for the reduction algorithm.
1755  // TODO: If the work-group-size is limited by the local memory, then
1756  // a special version of the main kernel may be created. The one that would
1757  // not use local accessors, which means it would not do the reduction in
1758  // the main kernel, but simply generate Range.get_global_range.size() number
1759  // of partial sums, leaving the reduction work to the additional/aux
1760  // kernels.
1761  constexpr bool HFR = Reduction::has_fast_reduce;
1762  size_t OneElemSize = HFR ? 0 : sizeof(typename Reduction::result_type);
1763  // TODO: currently the maximal work group size is determined for the given
1764  // queue/device, while it may be safer to use queries to the kernel compiled
1765  // for the device.
1766  size_t MaxWGSize =
1767  ext::oneapi::detail::reduGetMaxWGSize(MQueue, OneElemSize);
1768  if (Range.get_local_range().size() > MaxWGSize)
1769  throw sycl::runtime_error("The implementation handling parallel_for with"
1770  " reduction requires work group size not bigger"
1771  " than " +
1772  std::to_string(MaxWGSize),
1773  PI_INVALID_WORK_GROUP_SIZE);
1774 
1775  // 1. Call the kernel that includes user's lambda function.
1776  ext::oneapi::detail::reduCGFunc<KernelName>(*this, KernelFunc, Range, Redu);
1777  std::shared_ptr<detail::queue_impl> QueueCopy = MQueue;
1778  this->finalize();
1779 
1780  // 2. Run the additional kernel as many times as needed to reduce
1781  // all partial sums into one scalar.
1782 
1783  // TODO: Create a special slow/sequential version of the kernel that would
1784  // handle the reduction instead of reporting an assert below.
1785  if (MaxWGSize <= 1)
1786  throw sycl::runtime_error("The implementation handling parallel_for with "
1787  "reduction requires the maximal work group "
1788  "size to be greater than 1 to converge. "
1789  "The maximal work group size depends on the "
1790  "device and the size of the objects passed to "
1791  "the reduction.",
1792  PI_INVALID_WORK_GROUP_SIZE);
1793  size_t NWorkItems = Range.get_group_range().size();
1794  while (NWorkItems > 1) {
1795  handler AuxHandler(QueueCopy, MIsHost);
1796  AuxHandler.saveCodeLoc(MCodeLoc);
1797 
1798  NWorkItems = ext::oneapi::detail::reduAuxCGFunc<KernelName, KernelType>(
1799  AuxHandler, NWorkItems, MaxWGSize, Redu);
1800  MLastEvent = AuxHandler.finalize();
1801  } // end while (NWorkItems > 1)
1802 
1803  if (Reduction::is_usm || Redu.hasUserDiscardWriteAccessor()) {
1804  handler CopyHandler(QueueCopy, MIsHost);
1805  CopyHandler.saveCodeLoc(MCodeLoc);
1806  ext::oneapi::detail::reduSaveFinalResultToUserMem<KernelName>(CopyHandler,
1807  Redu);
1808  MLastEvent = CopyHandler.finalize();

1812  // This version of parallel_for may handle one or more reductions packed in
1813  // \p Rest argument. Note thought that the last element in \p Rest pack is
1814  // the kernel function.
1815  // TODO: this variant is currently enabled for 2+ reductions only as the
1816  // versions handling 1 reduction variable are more efficient right now.
1817  //
1818  // Algorithm:
1819  // 1) discard_write accessor (DWAcc), InitializeToIdentity = true:
1820  // a) Create uninitialized buffer and read_write accessor (RWAcc).
1821  // b) discard-write partial sums to RWAcc.
1822  // c) Repeat the steps (a) and (b) to get one final sum.
1823  // d) Copy RWAcc to DWAcc.
1824  // 2) read_write accessor (RWAcc), InitializeToIdentity = false:
1825  // a) Create new uninitialized buffer (if #work-groups > 1) and RWAcc or
1826  // re-use user's RWAcc (if #work-groups is 1).
1827  // b) discard-write to RWAcc (#WG > 1), or update-write (#WG == 1).
1828  // c) Repeat the steps (a) and (b) to get one final sum.
1829  // 3) read_write accessor (RWAcc), InitializeToIdentity = true:
1830  // a) Create new uninitialized buffer (if #work-groups > 1) and RWAcc or
1831  // re-use user's RWAcc (if #work-groups is 1).
1832  // b) discard-write to RWAcc.
1833  // c) Repeat the steps (a) and (b) to get one final sum.
1834  // 4) USM pointer, InitializeToIdentity = false:
1835  // a) Create new uninitialized buffer (if #work-groups > 1) and RWAcc or
1836  // re-use user's USM pointer (if #work-groups is 1).
1837  // b) discard-write to RWAcc (#WG > 1) or
1838  // update-write to USM pointer (#WG == 1).
1839  // c) Repeat the steps (a) and (b) to get one final sum.
1840  // 5) USM pointer, InitializeToIdentity = true:
1841  // a) Create new uninitialized buffer (if #work-groups > 1) and RWAcc or
1842  // re-use user's USM pointer (if #work-groups is 1).
1843  // b) discard-write to RWAcc (#WG > 1) or
1844  // discard-write to USM pointer (#WG == 1).
1845  // c) Repeat the steps (a) and (b) to get one final sum.
1846  template <typename KernelName = detail::auto_name, int Dims,
1847  typename... RestT>
1848  std::enable_if_t<
1849  (sizeof...(RestT) >= 3 &&

1863  // for the device.

1866  if (Range.get_local_range().size() > MaxWGSize)
