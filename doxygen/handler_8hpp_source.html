331 template <int Dims> class RoundedRangeIDGenerator {
332  id<Dims> Id;
333  id<Dims> InitId;
334  range<Dims> UserRange;
335  range<Dims> RoundedRange;
336  bool Done = false;
337 
338 public:
339  RoundedRangeIDGenerator(const id<Dims> &Id, const range<Dims> &UserRange,
340  const range<Dims> &RoundedRange)
341  : Id(Id), InitId(Id), UserRange(UserRange), RoundedRange(RoundedRange) {
342  for (int i = 0; i < Dims; ++i)
343  if (Id[i] >= UserRange[i])
344  Done = true;
345  }
346 
347  explicit operator bool() { return !Done; }
348 
349  void updateId() {
350  for (int i = 0; i < Dims; ++i) {
351  Id[i] += RoundedRange[i];
352  if (Id[i] < UserRange[i])
353  return;
354  Id[i] = InitId[i];
355  }
356  Done = true;
357  }
358 
359  id<Dims> getId() { return Id; }

361  template <typename KernelType> auto getItem() {
362  if constexpr (std::is_invocable_v<KernelType, item<Dims> &> ||
363  std::is_invocable_v<KernelType, item<Dims> &, kernel_handler>)
364  return detail::Builder::createItem<Dims, true>(UserRange, getId(), {});
365  else {

367  std::is_invocable_v<KernelType, item<Dims, false> &,
368  kernel_handler>,
369  "Kernel must be invocable with an item!");
370  return detail::Builder::createItem<Dims, false>(UserRange, getId());
371  }

1160  template <int Dims>
1161  std::tuple<range<Dims>, bool> getRoundedRange(range<Dims> UserRange) {
1162  range<Dims> RoundedRange = UserRange;
1163  // Disable the rounding-up optimizations under these conditions:
1164  // 1. The env var SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING is set.
1165  // 2. The kernel is provided via an interoperability method (this uses a
1166  // different code path).
1167  // 3. The range is already a multiple of the rounding factor.
1168  //
1169  // Cases 2 and 3 could be supported with extra effort.
1170  // As an optimization for the common case it is an
1171  // implementation choice to not support those scenarios.
1172  // Note that "this_item" is a free function, i.e. not tied to any
1173  // specific id or item. When concurrent parallel_fors are executing
1174  // on a device it is difficult to tell which parallel_for the call is
1175  // being made from. One could replicate portions of the
1176  // call-graph to make this_item calls kernel-specific but this is
1177  // not considered worthwhile.
1178 
1179  // Perform range rounding if rounding-up is enabled.
1180  if (this->DisableRangeRounding())
1181  return {range<Dims>{}, false};
1182 
1183  // Range should be a multiple of this for reasonable performance.
1184  size_t MinFactorX = 16;
1185  // Range should be a multiple of this for improved performance.
1186  size_t GoodFactor = 32;
1187  // Range should be at least this to make rounding worthwhile.
1188  size_t MinRangeX = 1024;
1189 
1190  // Check if rounding parameters have been set through environment:
1191  // SYCL_PARALLEL_FOR_RANGE_ROUNDING_PARAMS=MinRound:PreferredRound:MinRange
1192  this->GetRangeRoundingSettings(MinFactorX, GoodFactor, MinRangeX);
1193 
1194  // In SYCL, each dimension of a global range size is specified by
1195  // a size_t, which can be up to 64 bits. All backends should be
1196  // able to accept a kernel launch with a 32-bit global range size
1197  // (i.e. do not throw an error). The OpenCL CPU backend will
1198  // accept every 64-bit global range, but the GPU backends will not
1199  // generally accept every 64-bit global range. So, when we get a
1200  // non-32-bit global range, we wrap the old kernel in a new kernel
1201  // that has each work item peform multiple invocations the old
1202  // kernel in a 32-bit global range.
1203  id<Dims> MaxNWGs = [&] {
1204  auto [MaxWGs, HasMaxWGs] = getMaxWorkGroups_v2();
1205  if (!HasMaxWGs) {
1206  id<Dims> Default;
1207  for (int i = 0; i < Dims; ++i)
1208  Default[i] = (std::numeric_limits<int32_t>::max)();
1209  return Default;
1210  }
1211 
1212  id<Dims> IdResult;
1213  size_t Limit = (std::numeric_limits<int>::max)();
1214  for (int i = 0; i < Dims; ++i)
1215  IdResult[i] = (std::min)(Limit, MaxWGs[Dims - i - 1]);
1216  return IdResult;
1217  }();
1218  auto M = (std::numeric_limits<uint32_t>::max)();
1219  range<Dims> MaxRange;
1220  for (int i = 0; i < Dims; ++i) {
1221  auto DesiredSize = MaxNWGs[i] * GoodFactor;
1222  MaxRange[i] =
1223  DesiredSize <= M ? DesiredSize : (M / GoodFactor) * GoodFactor;
1224  }
1225 
1226  bool DidAdjust = false;
1227  auto Adjust = [&](int Dim, size_t Value) {
1228  if (this->RangeRoundingTrace())
1229  std::cout << "parallel_for range adjusted at dim " << Dim << " from "
1230  << RoundedRange[Dim] << " to " << Value << std::endl;
1231  RoundedRange[Dim] = Value;
1232  DidAdjust = true;
1233  };
1234 
1235 #ifdef __SYCL_EXP_PARALLEL_FOR_RANGE_ROUNDING__
1236  size_t GoodExpFactor = 1;
1237  switch (Dims) {
1238  case 1:
1239  GoodExpFactor = 32; // Make global range multiple of {32}
1240  break;
1241  case 2:
1242  GoodExpFactor = 16; // Make global range multiple of {16, 16}
1243  break;
1244  case 3:
1245  GoodExpFactor = 8; // Make global range multiple of {8, 8, 8}
1246  break;
1247  }
1248 
1249  // Check if rounding parameters have been set through environment:
1250  // SYCL_PARALLEL_FOR_RANGE_ROUNDING_PARAMS=MinRound:PreferredRound:MinRange
1251  this->GetRangeRoundingSettings(MinFactorX, GoodExpFactor, MinRangeX);
1252 
1253  for (auto i = 0; i < Dims; ++i)
1254  if (UserRange[i] % GoodExpFactor) {
1255  Adjust(i, ((UserRange[i] / GoodExpFactor) + 1) * GoodExpFactor);
1256  }
1257 #else
1258  // Perform range rounding if there are sufficient work-items to
1259  // need rounding and the user-specified range is not a multiple of
1260  // a "good" value.
1261  if (RoundedRange[0] % MinFactorX != 0 && RoundedRange[0] >= MinRangeX) {
1262  // It is sufficient to round up just the first dimension.
1263  // Multiplying the rounded-up value of the first dimension
1264  // by the values of the remaining dimensions (if any)
1265  // will yield a rounded-up value for the total range.
