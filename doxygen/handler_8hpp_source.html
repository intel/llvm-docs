275  std::is_same_v<T, id<Dims>>>
276 checkValueRange(const T &V) {
277 #if __SYCL_ID_QUERIES_FIT_IN_INT__
278  for (size_t Dim = 0; Dim < Dims; ++Dim)
279  checkValueRangeImpl<T>(V[Dim]);

325 template <int Dims> class RoundedRangeIDGenerator {
326  id<Dims> Id;
327  id<Dims> InitId;
328  range<Dims> UserRange;
329  range<Dims> RoundedRange;
330  bool Done = false;
331 
332 public:
333  RoundedRangeIDGenerator(const id<Dims> &Id, const range<Dims> &UserRange,
334  const range<Dims> &RoundedRange)
335  : Id(Id), InitId(Id), UserRange(UserRange), RoundedRange(RoundedRange) {
336  for (int i = 0; i < Dims; ++i)
337  if (Id[i] >= UserRange[i])
338  Done = true;
339  }
340 
341  explicit operator bool() { return !Done; }
342 
343  void updateId() {
344  for (int i = 0; i < Dims; ++i) {
345  Id[i] += RoundedRange[i];
346  if (Id[i] < UserRange[i])
347  return;
348  Id[i] = InitId[i];
349  }
350  Done = true;
351  }
352 
353  id<Dims> getId() { return Id; }

1161  template <int Dims>

1163  range<Dims> RoundedRange = UserRange;
1164  // Disable the rounding-up optimizations under these conditions:
1165  // 1. The env var SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING is set.
1166  // 2. The kernel is provided via an interoperability method (this uses a
1167  // different code path).
1168  // 3. The range is already a multiple of the rounding factor.
1169  //
1170  // Cases 2 and 3 could be supported with extra effort.
1171  // As an optimization for the common case it is an
1172  // implementation choice to not support those scenarios.
1173  // Note that "this_item" is a free function, i.e. not tied to any
1174  // specific id or item. When concurrent parallel_fors are executing
1175  // on a device it is difficult to tell which parallel_for the call is
1176  // being made from. One could replicate portions of the
1177  // call-graph to make this_item calls kernel-specific but this is
1178  // not considered worthwhile.
1179 
1180  // Perform range rounding if rounding-up is enabled.
1181  if (this->DisableRangeRounding())
1182  return {range<Dims>{}, false};
1183 
1184  // Range should be a multiple of this for reasonable performance.
1185  size_t MinFactorX = 16;
1186  // Range should be a multiple of this for improved performance.
1187  size_t GoodFactor = 32;
1188  // Range should be at least this to make rounding worthwhile.
1189  size_t MinRangeX = 1024;
1190 
1191  // Check if rounding parameters have been set through environment:
1192  // SYCL_PARALLEL_FOR_RANGE_ROUNDING_PARAMS=MinRound:PreferredRound:MinRange
1193  this->GetRangeRoundingSettings(MinFactorX, GoodFactor, MinRangeX);
1194 
1195  // In SYCL, each dimension of a global range size is specified by
1196  // a size_t, which can be up to 64 bits. All backends should be
1197  // able to accept a kernel launch with a 32-bit global range size
1198  // (i.e. do not throw an error). The OpenCL CPU backend will
1199  // accept every 64-bit global range, but the GPU backends will not
1200  // generally accept every 64-bit global range. So, when we get a
1201  // non-32-bit global range, we wrap the old kernel in a new kernel
1202  // that has each work item peform multiple invocations the old
1203  // kernel in a 32-bit global range.
1204  id<Dims> MaxNWGs = [&] {
1205  auto [MaxWGs, HasMaxWGs] = getMaxWorkGroups_v2();
1206  if (!HasMaxWGs) {
1207  id<Dims> Default;
1208  for (int i = 0; i < Dims; ++i)
1209  Default[i] = (std::numeric_limits<int32_t>::max)();
1210  return Default;
1211  }
1212 
1213  id<Dims> IdResult;
1214  size_t Limit = (std::numeric_limits<int>::max)();
1215  for (int i = 0; i < Dims; ++i)
1216  IdResult[i] = (std::min)(Limit, MaxWGs[Dims - i - 1]);
1217  return IdResult;
1218  }();
1219  auto M = (std::numeric_limits<uint32_t>::max)();
1220  range<Dims> MaxRange;
1221  for (int i = 0; i < Dims; ++i) {
1222  auto DesiredSize = MaxNWGs[i] * GoodFactor;
1223  MaxRange[i] =
1224  DesiredSize <= M ? DesiredSize : (M / GoodFactor) * GoodFactor;
1225  }

1255  if (UserRange[i] % GoodExpFactor) {
1256  Adjust(i, ((UserRange[i] / GoodExpFactor) + 1) * GoodExpFactor);
1257  }
1258 #else

1260  // need rounding and the user-specified range is not a multiple of
1261  // a "good" value.
1262  if (RoundedRange[0] % MinFactorX != 0 && RoundedRange[0] >= MinRangeX) {
1263  // It is sufficient to round up just the first dimension.
1264  // Multiplying the rounded-up value of the first dimension
1265  // by the values of the remaining dimensions (if any)
1266  // will yield a rounded-up value for the total range.
1267  Adjust(0, ((RoundedRange[0] + GoodFactor - 1) / GoodFactor) * GoodFactor);
1268  }
1269 #endif // __SYCL_EXP_PARALLEL_FOR_RANGE_ROUNDING__
1270 #ifdef __SYCL_FORCE_PARALLEL_FOR_RANGE_ROUNDING__
1271  // If we are forcing range rounding kernels to be used, we always want the
1272  // rounded range kernel to be generated, even if rounding isn't needed
1273  DidAdjust = true;
1274 #endif // __SYCL_FORCE_PARALLEL_FOR_RANGE_ROUNDING__
1275 
1276  for (int i = 0; i < Dims; ++i)
1277  if (RoundedRange[i] > MaxRange[i])
1278  Adjust(i, MaxRange[i]);
1279 
1280  if (!DidAdjust)
1281  return {range<Dims>{}, false};
1282  return {RoundedRange, true};
