1028  template <typename KernelName = detail::auto_name, typename KernelType,
1029  int Dims, typename Reduction>
1030  detail::enable_if_t<!Reduction::has_fast_atomics>
1031  parallel_for(nd_range<Dims> Range, Reduction Redu, KernelType KernelFunc) {
1032  size_t NWorkGroups = Range.get_group_range().size();
1033 
1034  // This parallel_for() is lowered to the following sequence:
1035  // 1) Call a kernel that a) call user's lambda function and b) performs
1036  // one iteration of reduction, storing the partial reductions/sums
1037  // to either a newly created global buffer or to user's reduction
1038  // accessor. So, if the original 'Range' has totally
1039  // N1 elements and work-group size is W, then after the first iteration
1040  // there will be N2 partial sums where N2 = N1 / W.
1041  // If (N2 == 1) then the partial sum is written to user's accessor.
1042  // Otherwise, a new global buffer is created and partial sums are written
1043  // to it.
1044  // 2) Call an aux kernel (if necessary, i.e. if N2 > 1) as many times as
1045  // necessary to reduce all partial sums into one final sum.
1046 
1047  // 1. Call the kernel that includes user's lambda function.
1048  if (Reduction::is_usm && NWorkGroups == 1)
1049  Redu.associateWithHandler(*this);
1050  intel::detail::reduCGFunc<KernelName>(*this, KernelFunc, Range, Redu);
1051  shared_ptr_class<detail::queue_impl> QueueCopy = MQueue;
1052  this->finalize();
1053 
1054  // 2. Run the additional aux kernel as many times as needed to reduce
1055  // all partial sums into one scalar.
1056 
1057  // TODO: user's nd_range and the work-group size specified there must
1058  // be honored only for the main kernel that calls user's lambda functions.
1059  // There is no need in using the same work-group size in these additional
1060  // kernels. Thus, the better strategy here is to make the work-group size
1061  // as big as possible to converge/reduce the partial sums into the last
1062  // sum faster.
1063  size_t WGSize = Range.get_local_range().size();
1064  size_t NWorkItems = NWorkGroups;
1065  while (NWorkItems > 1) {
1066  WGSize = std::min(WGSize, NWorkItems);
1067  NWorkGroups = NWorkItems / WGSize;
1068  // The last group may be not fully loaded. Still register it as a group.
1069  if ((NWorkItems % WGSize) != 0)
1070  ++NWorkGroups;
1071  nd_range<1> Range(range<1>(WGSize * NWorkGroups), range<1>(WGSize));
1072 
1073  handler AuxHandler(QueueCopy, MIsHost);
1074  AuxHandler.saveCodeLoc(MCodeLoc);
1075 
1076  // The last kernel DOES write to reduction's accessor.
1077  // Associate it with handler manually.
1078  if (NWorkGroups == 1)
1079  Redu.associateWithHandler(AuxHandler);
1080  intel::detail::reduAuxCGFunc<KernelName, KernelType>(AuxHandler, Range,
1081  NWorkItems, Redu);
1082  MLastEvent = AuxHandler.finalize();
1083 
1084  NWorkItems = NWorkGroups;
1085  } // end while (NWorkItems > 1)
1086 
1087  // Submit non-blocking copy from reduction accessor to user's reduction
1088  // variable.
1089  if (Reduction::is_usm) {
1090  handler CopyHandler(QueueCopy, MIsHost);
1091  CopyHandler.saveCodeLoc(MCodeLoc);
1092  Redu.associateWithHandler(CopyHandler);
1093  CopyHandler.copy(Redu.getUserAccessor(), Redu.getUSMPointer());
1094  MLastEvent = CopyHandler.finalize();
1095  }
1096  }
