185 template <int Dims, typename T>
186 typename detail::enable_if_t<std::is_same<T, range<Dims>>::value ||
187  std::is_same<T, id<Dims>>::value>
188 checkValueRange(const T &V) {
189 #if __SYCL_ID_QUERIES_FIT_IN_INT__
190  for (size_t Dim = 0; Dim < Dims; ++Dim)
191  checkValueRangeImpl<T>(V[Dim]);
192 
193  {
194  unsigned long long Product = 1;
195  for (size_t Dim = 0; Dim < Dims; ++Dim) {
196  Product *= V[Dim];
197  // check value now to prevent product overflow in the end
198  checkValueRangeImpl<T>(Product);
199  }
200  }
201 #else
202  (void)V;
203 #endif
204 }

752  template <typename KernelName, typename KernelType, int Dims>
753  void parallel_for_lambda_impl(range<Dims> NumWorkItems,
754  KernelType KernelFunc) {
755  throwIfActionIsCreated();
756  using LambdaArgType = sycl::detail::lambda_arg_type<KernelType, item<Dims>>;
757 
758  // If 1D kernel argument is an integral type, convert it to sycl::item<1>
759  using TransformedArgType =
760  typename std::conditional<std::is_integral<LambdaArgType>::value &&
761  Dims == 1,
762  item<Dims>, LambdaArgType>::type;
763  using NameT =
764  typename detail::get_kernel_name_t<KernelName, KernelType>::name;
765 
766  // The work group size preferred by this device.
767  // A reasonable choice for rounding up the range is 32.
768  constexpr size_t GoodLocalSizeX = 32;
769 
770  // Disable the rounding-up optimizations under these conditions:
771  // 1. The env var SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING is set.
772  // 2. The string SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING is in
773  // the kernel name.
774  // 3. The kernel is provided via an interoperability method.
775  // 4. The API "this_item" is used inside the kernel.
776  // 5. The range is already a multiple of the rounding factor.
777  //
778  // Cases 3 and 4 could be supported with extra effort.
779  // As an optimization for the common case it is an
780  // implementation choice to not support those scenarios.
781  // Note that "this_item" is a free function, i.e. not tied to any
782  // specific id or item. When concurrent parallel_fors are executing
783  // on a device it is difficult to tell which parallel_for the call is
784  // being made from. One could replicate portions of the
785  // call-graph to make this_item calls kernel-specific but this is
786  // not considered worthwhile.
787 
788  // Get the kernal name to check condition 3.
789  std::string KName = typeid(NameT *).name();
790  using KI = detail::KernelInfo<KernelName>;
791  bool DisableRounding =
792  (getenv("SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING") != nullptr) ||
793  (KName.find("SYCL_DISABLE_PARALLEL_FOR_RANGE_ROUNDING") !=
794  std::string::npos) ||
795  (KI::getName() == nullptr || KI::getName()[0] == '\0') ||
796  (KI::callsThisItem());
797 
798  // Perform range rounding if rounding-up is enabled
799  // and the user-specified range is not a multiple of a "good" value.
800  if (!DisableRounding && NumWorkItems[0] % GoodLocalSizeX != 0) {
801  // It is sufficient to round up just the first dimension.
802  // Multiplying the rounded-up value of the first dimension
803  // by the values of the remaining dimensions (if any)
804  // will yield a rounded-up value for the total range.
805  size_t NewValX =
806  ((NumWorkItems[0] + GoodLocalSizeX - 1) / GoodLocalSizeX) *
807  GoodLocalSizeX;
808  using NameWT = typename detail::get_kernel_wrapper_name_t<NameT>::name;
809  if (getenv("SYCL_PARALLEL_FOR_RANGE_ROUNDING_TRACE") != nullptr)
810  std::cout << "parallel_for range adjusted from " << NumWorkItems[0]
811  << " to " << NewValX << std::endl;
812  auto Wrapper = [=](TransformedArgType Arg) {
813  if (Arg[0] >= NumWorkItems[0])
814  return;
815  Arg.set_allowed_range(NumWorkItems);
816  KernelFunc(Arg);
817  };
818 
819  range<Dims> AdjustedRange = NumWorkItems;
820  AdjustedRange.set_range_dim0(NewValX);
821 #ifdef __SYCL_DEVICE_ONLY__
822  kernel_parallel_for<NameWT, TransformedArgType>(Wrapper);
823 #else
824  detail::checkValueRange<Dims>(AdjustedRange);
825  MNDRDesc.set(std::move(AdjustedRange));
826  StoreLambda<NameWT, decltype(Wrapper), Dims, TransformedArgType>(
827  std::move(Wrapper));
828  MCGType = detail::CG::KERNEL;
829 #endif
830  } else {
831 #ifdef __SYCL_DEVICE_ONLY__
832  (void)NumWorkItems;
833  kernel_parallel_for<NameT, TransformedArgType>(KernelFunc);
834 #else
835  detail::checkValueRange<Dims>(NumWorkItems);
836  MNDRDesc.set(std::move(NumWorkItems));
837  StoreLambda<NameT, KernelType, Dims, TransformedArgType>(
838  std::move(KernelFunc));
839  MCGType = detail::CG::KERNEL;
840 #endif
841  }

1239  template <typename KernelName = detail::auto_name, typename KernelType,
1240  int Dims, typename Reduction>
1241  detail::enable_if_t<!Reduction::has_fast_atomics>
1242  parallel_for(nd_range<Dims> Range, Reduction Redu,
1243  _KERNELFUNCPARAM(KernelFunc)) {
1244  // This parallel_for() is lowered to the following sequence:
1245  // 1) Call a kernel that a) call user's lambda function and b) performs
1246  // one iteration of reduction, storing the partial reductions/sums
1247  // to either a newly created global buffer or to user's reduction
1248  // accessor. So, if the original 'Range' has totally
1249  // N1 elements and work-group size is W, then after the first iteration
1250  // there will be N2 partial sums where N2 = N1 / W.
1251  // If (N2 == 1) then the partial sum is written to user's accessor.
1252  // Otherwise, a new global buffer is created and partial sums are written
1253  // to it.
1254  // 2) Call an aux kernel (if necessary, i.e. if N2 > 1) as many times as
1255  // necessary to reduce all partial sums into one final sum.
1256 
1257  // Before running the kernels, check that device has enough local memory
1258  // to hold local arrays that may be required for the reduction algorithm.
1259  // TODO: If the work-group-size is limited by the local memory, then
1260  // a special version of the main kernel may be created. The one that would
1261  // not use local accessors, which means it would not do the reduction in
1262  // the main kernel, but simply generate Range.get_global_range.size() number
1263  // of partial sums, leaving the reduction work to the additional/aux
1264  // kernels.
1265  constexpr bool HFR = Reduction::has_fast_reduce;
1266  size_t OneElemSize = HFR ? 0 : sizeof(typename Reduction::result_type);
1267  // TODO: currently the maximal work group size is determined for the given
1268  // queue/device, while it may be safer to use queries to the kernel compiled
1269  // for the device.
1270  size_t MaxWGSize = ONEAPI::detail::reduGetMaxWGSize(MQueue, OneElemSize);
1271  if (Range.get_local_range().size() > MaxWGSize)
1272  throw sycl::runtime_error("The implementation handling parallel_for with"
1273  " reduction requires work group size not bigger"
1274  " than " +
1275  std::to_string(MaxWGSize),
1276  PI_INVALID_WORK_GROUP_SIZE);
1277 
1278  // 1. Call the kernel that includes user's lambda function.
1279  ONEAPI::detail::reduCGFunc<KernelName>(*this, KernelFunc, Range, Redu);
1280  shared_ptr_class<detail::queue_impl> QueueCopy = MQueue;
1281  this->finalize();
1282 
1283  // 2. Run the additional kernel as many times as needed to reduce
1284  // all partial sums into one scalar.
1285 
1286  // TODO: Create a special slow/sequential version of the kernel that would
1287  // handle the reduction instead of reporting an assert below.
1288  if (MaxWGSize <= 1)
1289  throw sycl::runtime_error("The implementation handling parallel_for with "
1290  "reduction requires the maximal work group "
1291  "size to be greater than 1 to converge. "
1292  "The maximal work group size depends on the "
1293  "device and the size of the objects passed to "
1294  "the reduction.",
1295  PI_INVALID_WORK_GROUP_SIZE);
1296  size_t NWorkItems = Range.get_group_range().size();
1297  while (NWorkItems > 1) {
1298  handler AuxHandler(QueueCopy, MIsHost);
1299  AuxHandler.saveCodeLoc(MCodeLoc);
1300 
1301  NWorkItems = ONEAPI::detail::reduAuxCGFunc<KernelName, KernelType>(
1302  AuxHandler, NWorkItems, MaxWGSize, Redu);
1303  MLastEvent = AuxHandler.finalize();
1304  } // end while (NWorkItems > 1)
1305  }
