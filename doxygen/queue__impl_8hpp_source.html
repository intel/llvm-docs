209  void throw_asynchronous() {
210  std::unique_lock<mutex_class> lock(MMutex);
211 
212  if (MAsyncHandler && MExceptions.size()) {
213  exception_list Exceptions;
214 
215  std::swap(MExceptions, Exceptions);
216 
217  // Unlock the mutex before calling user-provided handler to avoid
218  // potential deadlock if the same queue is somehow referenced in the
219  // handler.
220  lock.unlock();
221 
222  MAsyncHandler(std::move(Exceptions));
223  }

230  RT::PiQueue createQueue(QueueOrder Order) {
231  RT::PiQueueProperties CreationFlags = 0;
232 
233  if (Order == QueueOrder::OOO) {
234  CreationFlags = PI_QUEUE_OUT_OF_ORDER_EXEC_MODE_ENABLE;
235  }
236  if (MPropList.has_property<property::queue::enable_profiling>()) {
237  CreationFlags |= PI_QUEUE_PROFILING_ENABLE;
238  }
239  RT::PiQueue Queue;
240  RT::PiContext Context = MContext->getHandleRef();
241  RT::PiDevice Device = MDevice->getHandleRef();
242  const detail::plugin &Plugin = getPlugin();
243 
244  assert(Plugin.getBackend() == MDevice->getPlugin().getBackend());
245  RT::PiResult Error = Plugin.call_nocheck<PiApiKind::piQueueCreate>(
246  Context, Device, CreationFlags, &Queue);
247 
248  // If creating out-of-order queue failed and this property is not
249  // supported (for example, on FPGA), it will return
250  // CL_INVALID_QUEUE_PROPERTIES and will try to create in-order queue.
251  if (MSupportOOO && Error == PI_INVALID_QUEUE_PROPERTIES) {
252  MSupportOOO = false;
253  Queue = createQueue(QueueOrder::Ordered);
254  } else {
255  Plugin.checkPiResult(Error);
256  }
257 
258  return Queue;
259  }

263  RT::PiQueue &getExclusiveQueueHandleRef() {
264  std::lock_guard<mutex_class> Guard(MMutex);
265 
266  // To achieve parallelism for FPGA with in order execution model with
267  // possibility of two kernels to share data with each other we shall
268  // create a queue for every kernel enqueued.
269  if (MQueues.size() < MaxNumQueues) {
270  MQueues.push_back(createQueue(QueueOrder::Ordered));
271  return MQueues.back();
272  }
273 
274  // If the limit of OpenCL queues is going to be exceeded - take the
275  // earliest used queue, wait until it finished and then reuse it.
276  MQueueNumber %= MaxNumQueues;
277  size_t FreeQueueNum = MQueueNumber++;
278 
279  getPlugin().call<PiApiKind::piQueueFinish>(MQueues[FreeQueueNum]);
280  return MQueues[FreeQueueNum];
281  }

285  RT::PiQueue &getHandleRef() {
286  if (MSupportOOO) {
287  return MCommandQueue;
288  }
289 
290  {
291  // Reduce the scope since this mutex is also
292  // locked inside of getExclusiveQueueHandleRef()
293  std::lock_guard<mutex_class> Guard(MMutex);
294 
295  if (MQueues.empty()) {
296  MQueues.push_back(MCommandQueue);
297  return MCommandQueue;
298  }
299  }
300 
301  return getExclusiveQueueHandleRef();
302  }
