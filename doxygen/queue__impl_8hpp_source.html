214  if (has_property<ext::oneapi::property::queue::discard_events>() &&

225  // TODO catch an exception and put it to list of asynchronous exceptions
226  Plugin->call(urQueueGetInfo, MQueues[0], UR_QUEUE_INFO_DEVICE,
227  sizeof(DeviceUr), &DeviceUr, nullptr);
228  MDevice = MContext->findMatchingDeviceImpl(DeviceUr);

240  static_cast<uint16_t>(xpti::trace_point_type_t::queue_create);

242  SYCL_STREAM_NAME, MQueueID, "queue_create");

245  // Cache the trace event, stream id and instance IDs for the destructor

469  void throw_asynchronous() {
470  if (!MAsyncHandler)
471  return;
472 
473  exception_list Exceptions;
474  {
475  std::lock_guard<std::mutex> Lock(MMutex);
476  std::swap(Exceptions, MExceptions);
477  }
478  // Unlock the mutex before calling user-provided handler to avoid
479  // potential deadlock if the same queue is somehow referenced in the
480  // handler.
481  if (Exceptions.size())
482  MAsyncHandler(std::move(Exceptions));
483  }

490  static ur_queue_flags_t createUrQueueFlags(const property_list &PropList,
491  QueueOrder Order) {
492  ur_queue_flags_t CreationFlags = 0;
493 
494  if (Order == QueueOrder::OOO) {
495  CreationFlags = UR_QUEUE_FLAG_OUT_OF_ORDER_EXEC_MODE_ENABLE;
496  }
497  if (PropList.has_property<property::queue::enable_profiling>()) {
498  CreationFlags |= UR_QUEUE_FLAG_PROFILING_ENABLE;
499  }
500  if (PropList.has_property<
501  ext::oneapi::cuda::property::queue::use_default_stream>()) {
502  CreationFlags |= UR_QUEUE_FLAG_USE_DEFAULT_STREAM;
503  }
504  if (PropList.has_property<ext::oneapi::property::queue::discard_events>()) {
505  // Pass this flag to the Level Zero plugin to be able to check it from
506  // queue property.
507  CreationFlags |= UR_QUEUE_FLAG_DISCARD_EVENTS;
508  }
509  // Track that priority settings are not ambiguous.
510  bool PrioritySeen = false;
511  if (PropList
512  .has_property<ext::oneapi::property::queue::priority_normal>()) {
513  // Normal is the default priority, don't pass anything.
514  PrioritySeen = true;
515  }
516  if (PropList.has_property<ext::oneapi::property::queue::priority_low>()) {
517  if (PrioritySeen) {
518  throw sycl::exception(
519  make_error_code(errc::invalid),
520  "Queue cannot be constructed with different priorities.");
521  }
522  CreationFlags |= UR_QUEUE_FLAG_PRIORITY_LOW;
523  PrioritySeen = true;
524  }
525  if (PropList.has_property<ext::oneapi::property::queue::priority_high>()) {
526  if (PrioritySeen) {
527  throw sycl::exception(
528  make_error_code(errc::invalid),
529  "Queue cannot be constructed with different priorities.");
530  }
531  CreationFlags |= UR_QUEUE_FLAG_PRIORITY_HIGH;
532  }
533  // Track that submission modes do not conflict.
534  bool SubmissionSeen = false;
535  if (PropList.has_property<
536  ext::intel::property::queue::no_immediate_command_list>()) {
537  SubmissionSeen = true;
538  CreationFlags |= UR_QUEUE_FLAG_SUBMISSION_BATCHED;
539  }
540  if (PropList.has_property<
541  ext::intel::property::queue::immediate_command_list>()) {
542  if (SubmissionSeen) {
543  throw sycl::exception(
544  make_error_code(errc::invalid),
545  "Queue cannot be constructed with different submission modes.");
546  }
547  SubmissionSeen = true;
548  CreationFlags |= UR_QUEUE_FLAG_SUBMISSION_IMMEDIATE;
549  }
550  return CreationFlags;
551  }

557  ur_queue_handle_t createQueue(QueueOrder Order) {
558  ur_queue_handle_t Queue{};
559  ur_context_handle_t Context = MContext->getHandleRef();
560  ur_device_handle_t Device = MDevice->getHandleRef();
561  const PluginPtr &Plugin = getPlugin();
562  /*
563  sycl::detail::pi::PiQueueProperties Properties[] = {
564  PI_QUEUE_FLAGS, createPiQueueProperties(MPropList, Order), 0, 0, 0};
565  */
566  ur_queue_properties_t Properties = {UR_STRUCTURE_TYPE_QUEUE_PROPERTIES,
567  nullptr, 0};
568  Properties.flags = createUrQueueFlags(MPropList, Order);
569  ur_queue_index_properties_t IndexProperties = {
570  UR_STRUCTURE_TYPE_QUEUE_INDEX_PROPERTIES, nullptr, 0};
571  if (has_property<ext::intel::property::queue::compute_index>()) {
572  IndexProperties.computeIndex =
573  get_property<ext::intel::property::queue::compute_index>()
574  .get_index();
575  Properties.pNext = &IndexProperties;
576  }
577  ur_result_t Error = Plugin->call_nocheck(urQueueCreate, Context, Device,
578  &Properties, &Queue);
579 
580  // If creating out-of-order queue failed and this property is not
581  // supported (for example, on FPGA), it will return
582  // UR_RESULT_ERROR_INVALID_QUEUE_PROPERTIES and will try to create in-order
583  // queue.
584  if (!MEmulateOOO && Error == UR_RESULT_ERROR_INVALID_QUEUE_PROPERTIES) {
585  MEmulateOOO = true;
586  Queue = createQueue(QueueOrder::Ordered);
587  } else {
588  Plugin->checkUrResult(Error);
589  }
590 
591  return Queue;
592  }

596  ur_queue_handle_t &getExclusiveUrQueueHandleRef() {
597  ur_queue_handle_t *PIQ = nullptr;
598  bool ReuseQueue = false;
599  {
600  std::lock_guard<std::mutex> Lock(MMutex);
601 
602  // To achieve parallelism for FPGA with in order execution model with
603  // possibility of two kernels to share data with each other we shall
604  // create a queue for every kernel enqueued.
605  if (MQueues.size() < MaxNumQueues) {
606  MQueues.push_back({});
607  PIQ = &MQueues.back();
608  } else {
609  // If the limit of OpenCL queues is going to be exceeded - take the
610  // earliest used queue, wait until it finished and then reuse it.
611  PIQ = &MQueues[MNextQueueIdx];
612  MNextQueueIdx = (MNextQueueIdx + 1) % MaxNumQueues;
613  ReuseQueue = true;
614  }
615  }
616 
617  if (!ReuseQueue)
618  *PIQ = createQueue(QueueOrder::Ordered);
619  else
620  getPlugin()->call(urQueueFinish, *PIQ);
621 
622  return *PIQ;
623  }

805  // template is needed for proper unit testing
806  template <typename HandlerType = handler>
807  void finalizeHandler(HandlerType &Handler, event &EventRet) {
808  if (MIsInorder) {
809  // Accessing and changing of an event isn't atomic operation.
810  // Hence, here is the lock for thread-safety.
811  std::lock_guard<std::mutex> Lock{MMutex};
812 
813  auto &EventToBuildDeps = MGraph.expired() ? MDefaultGraphDeps.LastEventPtr
814  : MExtGraphDeps.LastEventPtr;
815 
816  // This dependency is needed for the following purposes:
817  // - host tasks are handled by the runtime and cannot be implicitly
818  // synchronized by the backend.
819  // - to prevent the 2nd kernel enqueue when the 1st kernel is blocked
820  // by a host task. This dependency allows to build the enqueue order in
821  // the RT but will not be passed to the backend. See getPIEvents in
822  // Command.
823  if (EventToBuildDeps) {
824  // In the case where the last event was discarded and we are to run a
825  // host_task, we insert a barrier into the queue and use the resulting
826  // event as the dependency for the host_task.
827  // Note that host_task events can never be discarded, so this will not
828  // insert barriers between host_task enqueues.
829  if (EventToBuildDeps->isDiscarded() &&
830  getSyclObjImpl(Handler)->MCGType == CGType::CodeplayHostTask)
831  EventToBuildDeps = insertHelperBarrier(Handler);
832 
833  if (!EventToBuildDeps->isDiscarded())
834  Handler.depends_on(EventToBuildDeps);
835  }
836 
837  // If there is an external event set, add it as a dependency and clear it.
838  // We do not need to hold the lock as MLastEventMtx will ensure the last
839  // event reflects the corresponding external event dependence as well.
840  std::optional<event> ExternalEvent = popExternalEvent();
841  if (ExternalEvent)
842  Handler.depends_on(*ExternalEvent);
843 
844  EventRet = Handler.finalize();
845  EventToBuildDeps = getSyclObjImpl(EventRet);
846  } else {
847  const CGType Type = getSyclObjImpl(Handler)->MCGType;
848  std::lock_guard<std::mutex> Lock{MMutex};
849  // The following code supports barrier synchronization if host task is
850  // involved in the scenario. Native barriers cannot handle host task
851  // dependency so in the case where some commands were not enqueued
852  // (blocked), we track them to prevent barrier from being enqueued
853  // earlier.
854  {
855  std::lock_guard<std::mutex> RequestLock(MMissedCleanupRequestsMtx);
856  for (auto &UpdatedGraph : MMissedCleanupRequests)
857  doUnenqueuedCommandCleanup(UpdatedGraph);
858  MMissedCleanupRequests.clear();
859  }
860  auto &Deps = MGraph.expired() ? MDefaultGraphDeps : MExtGraphDeps;
861  if (Type == CGType::Barrier && !Deps.UnenqueuedCmdEvents.empty()) {
862  Handler.depends_on(Deps.UnenqueuedCmdEvents);
863  }
864  if (Deps.LastBarrier)
865  Handler.depends_on(Deps.LastBarrier);
866  EventRet = Handler.finalize();
867  EventImplPtr EventRetImpl = getSyclObjImpl(EventRet);
868  if (Type == CGType::CodeplayHostTask)
869  Deps.UnenqueuedCmdEvents.push_back(EventRetImpl);
870  else if (!EventRetImpl->isEnqueued()) {
871  if (Type == CGType::Barrier || Type == CGType::BarrierWaitlist) {
872  Deps.LastBarrier = EventRetImpl;
873  Deps.UnenqueuedCmdEvents.clear();
874  } else
875  Deps.UnenqueuedCmdEvents.push_back(EventRetImpl);
876  }
877  }
878  }
