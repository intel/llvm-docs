223  void throw_asynchronous() {
224  if (!MAsyncHandler)
225  return;
226 
227  exception_list Exceptions;
228  {
229  std::lock_guard<std::mutex> Lock(MMutex);
230  std::swap(Exceptions, MExceptions);

232  // Unlock the mutex before calling user-provided handler to avoid
233  // potential deadlock if the same queue is somehow referenced in the
234  // handler.
235  if (Exceptions.size())
236  MAsyncHandler(std::move(Exceptions));

243  RT::PiQueue createQueue(QueueOrder Order) {
244  RT::PiQueueProperties CreationFlags = 0;
245 
246  if (Order == QueueOrder::OOO) {
247  CreationFlags = PI_QUEUE_OUT_OF_ORDER_EXEC_MODE_ENABLE;
248  }
249  if (MPropList.has_property<property::queue::enable_profiling>()) {
250  CreationFlags |= PI_QUEUE_PROFILING_ENABLE;
251  }
252  if (MPropList.has_property<property::queue::cuda::use_default_stream>()) {
253  CreationFlags |= __SYCL_PI_CUDA_USE_DEFAULT_STREAM;
254  }
255  RT::PiQueue Queue{};
256  RT::PiContext Context = MContext->getHandleRef();
257  RT::PiDevice Device = MDevice->getHandleRef();
258  const detail::plugin &Plugin = getPlugin();
259 
260  assert(Plugin.getBackend() == MDevice->getPlugin().getBackend());
261  RT::PiResult Error = Plugin.call_nocheck<PiApiKind::piQueueCreate>(
262  Context, Device, CreationFlags, &Queue);
263 
264  // If creating out-of-order queue failed and this property is not
265  // supported (for example, on FPGA), it will return
266  // CL_INVALID_QUEUE_PROPERTIES and will try to create in-order queue.
267  if (MSupportOOO && Error == PI_INVALID_QUEUE_PROPERTIES) {
268  MSupportOOO = false;
269  Queue = createQueue(QueueOrder::Ordered);
270  } else {
271  Plugin.checkPiResult(Error);
272  }
273 
274  return Queue;
275  }

279  RT::PiQueue &getExclusiveQueueHandleRef() {
280  RT::PiQueue *PIQ = nullptr;
281  bool ReuseQueue = false;
282  {
283  std::lock_guard<std::mutex> Lock(MMutex);
284 
285  // To achieve parallelism for FPGA with in order execution model with
286  // possibility of two kernels to share data with each other we shall
287  // create a queue for every kernel enqueued.
288  if (MQueues.size() < MaxNumQueues) {
289  MQueues.push_back({});
290  PIQ = &MQueues.back();
291  } else {
292  // If the limit of OpenCL queues is going to be exceeded - take the
293  // earliest used queue, wait until it finished and then reuse it.
294  PIQ = &MQueues[MNextQueueIdx];
295  MNextQueueIdx = (MNextQueueIdx + 1) % MaxNumQueues;
296  ReuseQueue = true;
297  }
298  }
299 
300  if (!ReuseQueue)
301  *PIQ = createQueue(QueueOrder::Ordered);
302  else
303  getPlugin().call<PiApiKind::piQueueFinish>(*PIQ);
304 
305  return *PIQ;
306  }
