222  void throw_asynchronous() {
223  if (!MAsyncHandler)
224  return;

226  exception_list Exceptions;
227  {

234  if (Exceptions.size())
235  MAsyncHandler(std::move(Exceptions));

242  RT::PiQueue createQueue(QueueOrder Order) {
243  RT::PiQueueProperties CreationFlags = 0;
244 
245  if (Order == QueueOrder::OOO) {
246  CreationFlags = PI_QUEUE_OUT_OF_ORDER_EXEC_MODE_ENABLE;
247  }
248  if (MPropList.has_property<property::queue::enable_profiling>()) {
249  CreationFlags |= PI_QUEUE_PROFILING_ENABLE;
250  }
251  RT::PiQueue Queue{};
252  RT::PiContext Context = MContext->getHandleRef();
253  RT::PiDevice Device = MDevice->getHandleRef();
254  const detail::plugin &Plugin = getPlugin();
255 
256  assert(Plugin.getBackend() == MDevice->getPlugin().getBackend());
257  RT::PiResult Error = Plugin.call_nocheck<PiApiKind::piQueueCreate>(
258  Context, Device, CreationFlags, &Queue);
259 
260  // If creating out-of-order queue failed and this property is not
261  // supported (for example, on FPGA), it will return
262  // CL_INVALID_QUEUE_PROPERTIES and will try to create in-order queue.
263  if (MSupportOOO && Error == PI_INVALID_QUEUE_PROPERTIES) {
264  MSupportOOO = false;
265  Queue = createQueue(QueueOrder::Ordered);
266  } else {
267  Plugin.checkPiResult(Error);
268  }
269 
270  return Queue;
271  }

275  RT::PiQueue &getExclusiveQueueHandleRef() {
276  RT::PiQueue *PIQ = nullptr;
277  bool ReuseQueue = false;
278  {
279  std::lock_guard<mutex_class> Lock(MMutex);
280 
281  // To achieve parallelism for FPGA with in order execution model with
282  // possibility of two kernels to share data with each other we shall
283  // create a queue for every kernel enqueued.
284  if (MQueues.size() < MaxNumQueues) {
285  MQueues.push_back({});
286  PIQ = &MQueues.back();
287  } else {
288  // If the limit of OpenCL queues is going to be exceeded - take the
289  // earliest used queue, wait until it finished and then reuse it.
290  PIQ = &MQueues[MNextQueueIdx];
291  MNextQueueIdx = (MNextQueueIdx + 1) % MaxNumQueues;
292  ReuseQueue = true;
293  }
294  }
295 
296  if (!ReuseQueue)
297  *PIQ = createQueue(QueueOrder::Ordered);
298  else
299  getPlugin().call<PiApiKind::piQueueFinish>(*PIQ);
300 
301  return *PIQ;
302  }
