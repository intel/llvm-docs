53 context::context(const vector_class<device> &DeviceList,
54  async_handler AsyncHandler, const property_list &PropList) {
55  if (DeviceList.empty()) {
56  throw invalid_parameter_error("DeviceList is empty.", PI_INVALID_VALUE);
57  }
58  auto NonHostDeviceIter = std::find_if_not(
59  DeviceList.begin(), DeviceList.end(),
60  [&](const device &CurrentDevice) { return CurrentDevice.is_host(); });
61  if (NonHostDeviceIter == DeviceList.end())
62  impl = std::make_shared<detail::context_impl>(DeviceList[0], AsyncHandler,
63  PropList);
64  else {
65  const device &NonHostDevice = *NonHostDeviceIter;
66  const auto &NonHostPlatform =
67  detail::getSyclObjImpl(NonHostDevice.get_platform())->getHandleRef();
68  if (std::any_of(DeviceList.begin(), DeviceList.end(),
69  [&](const device &CurrentDevice) {
70  return (
71  CurrentDevice.is_host() ||
72  (detail::getSyclObjImpl(CurrentDevice.get_platform())
73  ->getHandleRef() != NonHostPlatform));
74  }))
75  throw invalid_parameter_error(
76  "Can't add devices across platforms to a single context.",
77  PI_INVALID_DEVICE);
78  else
79  impl = std::make_shared<detail::context_impl>(DeviceList, AsyncHandler,
80  PropList);
81  }
82 }
