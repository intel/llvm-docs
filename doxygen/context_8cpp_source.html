40 context::context(const vector_class<device> &DeviceList,
41  async_handler AsyncHandler, bool UsePrimaryContext) {
42  if (DeviceList.empty()) {
43  throw invalid_parameter_error("DeviceList is empty.", PI_INVALID_VALUE);
44  }
45  auto NonHostDeviceIter = std::find_if_not(
46  DeviceList.begin(), DeviceList.end(),
47  [&](const device &CurrentDevice) { return CurrentDevice.is_host(); });
48  if (NonHostDeviceIter == DeviceList.end())
49  impl =
50  std::make_shared<detail::context_impl>(DeviceList[0], AsyncHandler,
51  UsePrimaryContext);
52  else {
53  const device &NonHostDevice = *NonHostDeviceIter;
54  const auto &NonHostPlatform = NonHostDevice.get_platform().get();
55  if (std::any_of(DeviceList.begin(), DeviceList.end(),
56  [&](const device &CurrentDevice) {
57  return (CurrentDevice.is_host() ||
58  (CurrentDevice.get_platform().get() !=
59  NonHostPlatform));
60  }))
61  throw invalid_parameter_error(
62  "Can't add devices across platforms to a single context.",
63  PI_INVALID_DEVICE);
64  else
65  impl = std::make_shared<detail::context_impl>(DeviceList, AsyncHandler,
66  UsePrimaryContext);
67  }
68 }
