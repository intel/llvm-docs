39 static void initValue(const char *Key, const char *Value) {
40 #define CONFIG(Name, MaxSize, CompileTimeDef) \
41  if (0 == strncmp(Key, SYCLConfigBase<Name>::MConfigName, MAX_CONFIG_NAME)) { \
42  strncpy(SYCLConfigBase<Name>::MStorage, Value, MaxSize); \
43  SYCLConfigBase<Name>::MStorage[MaxSize] = '\0'; \
44  SYCLConfigBase<Name>::MValueFromFile = SYCLConfigBase<Name>::MStorage; \
45  return; \
46  }
47 #include "detail/config.def"
48 #undef CONFIG

51 void readConfig() {
52  static bool Initialized = false;
53  if (Initialized)
54  return;
55 
56  std::fstream File;
57  if (const char *ConfigFile = getenv("SYCL_CONFIG_FILE_NAME"))
58  File.open(ConfigFile, std::ios::in);
59  else {
60  const std::string LibSYCLDir = sycl::detail::OSUtil::getCurrentDSODir();
61  File.open(LibSYCLDir + sycl::detail::OSUtil::DirSep + SYCL_CONFIG_FILE_NAME,
62  std::ios::in);
63  }
64 
65  if (File.is_open()) {
66  // TODO: Use max size from macro instead of 256
67  char Key[MAX_CONFIG_NAME] = {0}, Value[256] = {0};
68  while (!File.eof()) {
69  // Expected fromat:
70  // ConfigName=Value\r
71  // ConfigName=Value
72  // TODO: Skip spaces before and after '='
73  File.getline(Key, sizeof(Key), '=');
74  if (File.fail()) {
75  // Fail to process the line. Skip it completely and try next one.
76  // Do we want to restore here? Or just throw an exception?
77  File.clear(File.rdstate() & ~std::ios_base::failbit);
78  File.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
79  continue;
80  }
81  File.getline(Value, sizeof(Value), '\n');
82 
83  if (File.fail()) {
84  // Fail to process the value while config name is OK. It's likely that
85  // value is too long. Currently just deal what we have got and ignore
86  // remaining characters on the line.
87  // Do we want to restore here? Or just throw an exception?
88  File.clear(File.rdstate() & ~std::ios_base::failbit);
89  File.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
90  }
91 
92  // Handle '\r' by nullifying it
93  const std::streamsize ReadSybmols = File.gcount();
94  if (ReadSybmols > 1 && '\r' == Value[ReadSybmols - 2])
95  Value[ReadSybmols - 2] = '\0';
96 
97  initValue(Key, Value);
98  }
99  }
100  Initialized = true;
101 }
