228 // Change the element type of a simd vector.
229 template <typename ToEltTy, typename FromEltTy, int FromN,
230  typename = csd::enable_if_t<is_vectorizable<ToEltTy>::value>>
231 ESIMD_INLINE typename std::conditional<
232  std::is_same<FromEltTy, ToEltTy>::value, vector_type_t<FromEltTy, FromN>,
233  vector_type_t<ToEltTy,
234  bitcast_helper<ToEltTy, FromEltTy, FromN>::nToElems()>>::type
235 bitcast(vector_type_t<FromEltTy, FromN> Val) {
236  // Noop.
237  if constexpr (std::is_same<FromEltTy, ToEltTy>::value)
238  return Val;
239 
240  // Bitcast
241  constexpr int ToN = bitcast_helper<ToEltTy, FromEltTy, FromN>::nToElems();
242  using VTy = vector_type_t<ToEltTy, ToN>;
243  return reinterpret_cast<VTy>(Val);
244 }
