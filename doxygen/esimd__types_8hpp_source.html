209 template <typename ToEltTy, typename FromEltTy, int FromN,

214  constexpr int R2 = sizeof(FromEltTy) / sizeof(ToEltTy);

220 // Change the element type of a simd vector.
221 template <typename ToEltTy, typename FromEltTy, int FromN,
222  typename = csd::enable_if_t<is_vectorizable<ToEltTy>::value>>
223 ESIMD_INLINE typename csd::conditional_t<
224  std::is_same<FromEltTy, ToEltTy>::value, vector_type_t<FromEltTy, FromN>,
225  vector_type_t<ToEltTy,
226  bitcast_helper<ToEltTy, FromEltTy, FromN>::nToElems()>>
227 bitcast(vector_type_t<FromEltTy, FromN> Val) {
228  // Noop.
229  if constexpr (std::is_same<FromEltTy, ToEltTy>::value)
230  return Val;
231 
232  // Bitcast
233  constexpr int ToN = bitcast_helper<ToEltTy, FromEltTy, FromN>::nToElems();
234  using VTy = vector_type_t<ToEltTy, ToN>;
235  return reinterpret_cast<VTy>(Val);
236 }
