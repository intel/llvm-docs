81  std::vector<kernel_id> get_kernel_ids() const {
82  // Collect kernel ids from all device images, then remove duplicates
83 
84  std::vector<kernel_id> Result;
85  for (const device_image_plain &DeviceImage : MDeviceImages) {
86  const std::vector<kernel_id> &KernelIDs =
87  getSyclObjImpl(DeviceImage)->get_kernel_ids();
88 
89  Result.insert(Result.end(), KernelIDs.begin(), KernelIDs.end());
90  }
91  std::sort(Result.begin(), Result.end(),
92  [](const kernel_id &LHS, const kernel_id &RHS) {
93  return detail::getSyclObjImpl(LHS) <
94  detail::getSyclObjImpl(RHS);
95  });
96  auto LastIt = std::unique(Result.begin(), Result.end());
97  Result.erase(LastIt, Result.end());
