49  // Matches sycl::build and sycl::compile
50  // Have one constructor because sycl::build and sycl::compile have the same
51  // signature
52  kernel_bundle_impl(const kernel_bundle<bundle_state::input> &InputBundle,
53  std::vector<device> Devs, const property_list &PropList,
54  bundle_state TargetState)
55  : MContext(InputBundle.get_context()), MDevices(std::move(Devs)) {
56 
57  for (const device_image_plain &DeviceImage : InputBundle) {
58  // Skip images which are not compatible with devices provided
59  if (std::none_of(
60  MDevices.begin(), MDevices.end(),
61  [&DeviceImage](const device &Dev) {
62  return getSyclObjImpl(DeviceImage)->compatible_with_device(Dev);
63  }))
64  continue;
65 
66  switch (TargetState) {
67  case bundle_state::object:
68  MDeviceImages.push_back(detail::ProgramManager::getInstance().compile(
69  DeviceImage, MDevices, PropList));
70  break;
71  case bundle_state::executable:
72  MDeviceImages.push_back(detail::ProgramManager::getInstance().build(
73  DeviceImage, MDevices, PropList));
74  break;
75  case bundle_state::input:
76  throw sycl::runtime_error(
77  "Internal error. The target state should not be input",
78  PI_INVALID_OPERATION);
79  break;
80  }
81  }
82  }

191  auto It = std::find_if(MDeviceImages.begin(), MDeviceImages.end(),

198  RT::PiKernel Kernel = nullptr;

202  DeviceImageImpl->get_program_ref());

205  Kernel, detail::getSyclObjImpl(MContext), DeviceImageImpl, Self);
206 
207  return detail::createSyclObjFromImpl<kernel>(KernelImpl);
