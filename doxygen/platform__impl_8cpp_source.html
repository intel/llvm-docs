68 static bool IsBannedPlatform(platform Platform) {
69  // The NVIDIA OpenCL platform is currently not compatible with DPC++
70  // since it is only 1.2 but gets selected by default in many systems
71  // There is also no support on the PTX backend for OpenCL consumption,
72  // and there have been some internal reports.
73  // To avoid problems on default users and deployment of DPC++ on platforms
74  // where CUDA is available, the OpenCL support is disabled.
75  //
76  auto IsNVIDIAOpenCL = [](platform Platform) {
77  if (Platform.is_host())
78  return false;
79 
80  const bool HasCUDA = Platform.get_info<info::platform::name>().find(
81  "NVIDIA CUDA") != std::string::npos;
82  const auto Backend =
83  detail::getSyclObjImpl(Platform)->getPlugin().getBackend();
84  const bool IsCUDAOCL = (HasCUDA && Backend == backend::opencl);
85  if (detail::pi::trace(detail::pi::TraceLevel::PI_TRACE_ALL) && IsCUDAOCL) {
86  std::cout << "SYCL_PI_TRACE[all]: "
87  << "NVIDIA CUDA OpenCL platform found but is not compatible."
88  << std::endl;
89  }
90  return IsCUDAOCL;
91  };
92  return IsNVIDIAOpenCL(Platform);
93 }

95 vector_class<platform> platform_impl::get_platforms() {
96  vector_class<platform> Platforms;
97  const vector_class<plugin> &Plugins = RT::initialize();
98 
99  info::device_type ForcedType = detail::get_forced_type();
100  for (unsigned int i = 0; i < Plugins.size(); i++) {
101 
102  pi_uint32 NumPlatforms = 0;
103  Plugins[i].call<PiApiKind::piPlatformsGet>(0, nullptr, &NumPlatforms);
104 
105  if (NumPlatforms) {
106  vector_class<RT::PiPlatform> PiPlatforms(NumPlatforms);
107  Plugins[i].call<PiApiKind::piPlatformsGet>(NumPlatforms,
108  PiPlatforms.data(), nullptr);
109 
110  for (const auto &PiPlatform : PiPlatforms) {
111  platform Platform = detail::createSyclObjFromImpl<platform>(
112  getOrMakePlatformImpl(PiPlatform, Plugins[i]));
113  // Skip platforms which do not contain requested device types
114  if (!Platform.get_devices(ForcedType).empty() &&
115  !IsBannedPlatform(Platform))
116  Platforms.push_back(Platform);
117  }
118  }
119  }
120 
121  // The host platform should always be available.
122  Platforms.emplace_back(platform());
123 
124  return Platforms;
125 }

127 std::string getValue(const std::string &AllowList, size_t &Pos,
128  unsigned long int Size) {
129  size_t Prev = Pos;
130  if ((Pos = AllowList.find("{{", Pos)) == std::string::npos) {
131  throw sycl::runtime_error("Malformed syntax in SYCL_DEVICE_ALLOWLIST",
132  PI_INVALID_VALUE);
133  }
134  if (Pos > Prev + Size) {
135  throw sycl::runtime_error("Malformed syntax in SYCL_DEVICE_ALLOWLIST",
136  PI_INVALID_VALUE);
137  }
138 
139  Pos = Pos + 2;
140  size_t Start = Pos;
141  if ((Pos = AllowList.find("}}", Pos)) == std::string::npos) {
142  throw sycl::runtime_error("Malformed syntax in SYCL_DEVICE_ALLOWLIST",
143  PI_INVALID_VALUE);
144  }
145  std::string Value = AllowList.substr(Start, Pos - Start);
146  Pos = Pos + 2;
147  return Value;
148 }

157 static std::vector<DevDescT> getAllowListDesc() {
158  std::string AllowList(SYCLConfig<SYCL_DEVICE_ALLOWLIST>::get());
159  if (AllowList.empty())
160  return {};
161 
162  std::string DeviceName("DeviceName:");
163  std::string DriverVersion("DriverVersion:");
164  std::string PlatformName("PlatformName:");
165  std::string PlatformVersion("PlatformVersion:");
166  std::vector<DevDescT> DecDescs;
167  DecDescs.emplace_back();
168 
169  size_t Pos = 0;
170  while (Pos < AllowList.size()) {
171  if ((AllowList.compare(Pos, DeviceName.size(), DeviceName)) == 0) {
172  DecDescs.back().DevName = getValue(AllowList, Pos, DeviceName.size());
173  if (AllowList[Pos] == ',') {
174  Pos++;
175  }
176  }
177 
178  else if ((AllowList.compare(Pos, DriverVersion.size(), DriverVersion)) ==
179  0) {
180  DecDescs.back().DevDriverVer =
181  getValue(AllowList, Pos, DriverVersion.size());
182  if (AllowList[Pos] == ',') {
183  Pos++;
184  }
185  }
186 
187  else if ((AllowList.compare(Pos, PlatformName.size(), PlatformName)) == 0) {
188  DecDescs.back().PlatName = getValue(AllowList, Pos, PlatformName.size());
189  if (AllowList[Pos] == ',') {
190  Pos++;
191  }
192  }
193 
194  else if ((AllowList.compare(Pos, PlatformVersion.size(),
195  PlatformVersion)) == 0) {
196  DecDescs.back().PlatVer =
197  getValue(AllowList, Pos, PlatformVersion.size());
198  } else if (AllowList.find('|', Pos) != std::string::npos) {
199  Pos = AllowList.find('|') + 1;
200  while (AllowList[Pos] == ' ') {
201  Pos++;
202  }
203  DecDescs.emplace_back();
204  }
205 
206  else {
207  throw sycl::runtime_error("Unrecognized key in device allowlist",
208  PI_INVALID_VALUE);
209  }
210  } // while (Pos <= AllowList.size())
211  return DecDescs;
212 }

216 static void filterAllowList(vector_class<RT::PiDevice> &PiDevices,
217  RT::PiPlatform PiPlatform, const plugin &Plugin) {
218  const std::vector<DevDescT> AllowList(getAllowListDesc());
219  if (AllowList.empty())
220  return;
221 
222  FilterState DevNameState = FilterState::ALLOWED;
223  FilterState DevVerState = FilterState::ALLOWED;
224  FilterState PlatNameState = FilterState::ALLOWED;
225  FilterState PlatVerState = FilterState::ALLOWED;
226 
227  const string_class PlatformName =
228  sycl::detail::get_platform_info<string_class, info::platform::name>::get(
229  PiPlatform, Plugin);
230 
231  const string_class PlatformVer =
232  sycl::detail::get_platform_info<string_class,
233  info::platform::version>::get(PiPlatform,
234  Plugin);
235 
236  int InsertIDx = 0;
237  for (RT::PiDevice Device : PiDevices) {
238  const string_class DeviceName =
239  sycl::detail::get_device_info<string_class, info::device::name>::get(
240  Device, Plugin);
241 
242  const string_class DeviceDriverVer = sycl::detail::get_device_info<
243  string_class, info::device::driver_version>::get(Device, Plugin);
244 
245  for (const DevDescT &Desc : AllowList) {
246  if (!Desc.PlatName.empty()) {
247  if (!std::regex_match(PlatformName, std::regex(Desc.PlatName))) {
248  PlatNameState = FilterState::DENIED;
249  continue;
250  }
251  }
252 
253  if (!Desc.PlatVer.empty()) {
254  if (!std::regex_match(PlatformVer, std::regex(Desc.PlatVer))) {
255  PlatVerState = FilterState::DENIED;
256  continue;
257  }
258  }
259 
260  if (!Desc.DevName.empty()) {
261  if (!std::regex_match(DeviceName, std::regex(Desc.DevName))) {
262  DevNameState = FilterState::DENIED;
263  continue;
264  }
265  }
266 
267  if (!Desc.DevDriverVer.empty()) {
268  if (!std::regex_match(DeviceDriverVer, std::regex(Desc.DevDriverVer))) {
269  DevVerState = FilterState::DENIED;
270  continue;
271  }
272  }
273 
274  if (DevNameState == FilterState::ALLOWED &&
275  DevVerState == FilterState::ALLOWED &&
276  PlatNameState == FilterState::ALLOWED &&
277  PlatVerState == FilterState::ALLOWED)
278  PiDevices[InsertIDx++] = Device;
279  break;
280  }
281  }
282  PiDevices.resize(InsertIDx);
283 }

305 vector_class<device>
306 platform_impl::get_devices(info::device_type DeviceType) const {
307  vector_class<device> Res;
308  if (is_host() && (DeviceType == info::device_type::host ||
309  DeviceType == info::device_type::all)) {
310  Res.push_back(device());
311  }
312 
313  // If any DeviceType other than host was requested for host platform,
314  // an empty vector will be returned.
315  if (is_host() || DeviceType == info::device_type::host)
316  return Res;
317 
318  pi_uint32 NumDevices = 0;
319  const detail::plugin &Plugin = getPlugin();
320  Plugin.call<PiApiKind::piDevicesGet>(
321  MPlatform, pi::cast<RT::PiDeviceType>(DeviceType), 0,
322  pi::cast<RT::PiDevice *>(nullptr), &NumDevices);
323 
324  if (NumDevices == 0)
325  return Res;
326 
327  vector_class<RT::PiDevice> PiDevices(NumDevices);
328  // TODO catch an exception and put it to list of asynchronous exceptions
329  Plugin.call<PiApiKind::piDevicesGet>(MPlatform,
330  pi::cast<RT::PiDeviceType>(DeviceType),
331  NumDevices, PiDevices.data(), nullptr);
332 
333  // Filter out devices that are not present in the allowlist
334  if (SYCLConfig<SYCL_DEVICE_ALLOWLIST>::get())
335  filterAllowList(PiDevices, MPlatform, this->getPlugin());
336 
337  PlatformImplPtr PlatformImpl = getOrMakePlatformImpl(MPlatform, *MPlugin);
338  std::transform(
339  PiDevices.begin(), PiDevices.end(), std::back_inserter(Res),
340  [PlatformImpl](const RT::PiDevice &PiDevice) -> device {
341  return detail::createSyclObjFromImpl<device>(
342  PlatformImpl->getOrMakeDeviceImpl(PiDevice, PlatformImpl));
343  });
344 
345  return Res;
346 }
