23 vector_class<platform> platform_impl::get_platforms() {
24  vector_class<platform> Platforms;
25  vector_class<plugin> Plugins = RT::initialize();
26 
27  info::device_type ForcedType = detail::get_forced_type();
28  for (unsigned int i = 0; i < Plugins.size(); i++) {
29 
30  pi_uint32 NumPlatforms = 0;
31  Plugins[i].call<PiApiKind::piPlatformsGet>(0, nullptr, &NumPlatforms);
32 
33  if (NumPlatforms) {
34  vector_class<RT::PiPlatform> PiPlatforms(NumPlatforms);
35  Plugins[i].call<PiApiKind::piPlatformsGet>(NumPlatforms,
36  PiPlatforms.data(), nullptr);
37 
38  for (const auto &PiPlatform : PiPlatforms) {
39  platform Platform = detail::createSyclObjFromImpl<platform>(
40  std::make_shared<platform_impl>(PiPlatform, Plugins[i]));
41  // Skip platforms which do not contain requested device types
42  if (!Platform.get_devices(ForcedType).empty())
43  Platforms.push_back(Platform);
44  }
45  }
46  }
47 
48  // The host platform should always be available.
49  Platforms.emplace_back(platform());
50 
51  return Platforms;
52 }

68 static std::vector<DevDescT> getAllowListDesc() {
69  const char *str = SYCLConfig<SYCL_DEVICE_ALLOWLIST>::get();
70  if (!str)
71  return {};
72 
73  std::vector<DevDescT> decDescs;
74  const char devNameStr[] = "DeviceName";
75  const char driverVerStr[] = "DriverVersion";
76  const char platformNameStr[] = "PlatformName";
77  const char platformVerStr[] = "PlatformVersion";
78  decDescs.emplace_back();
79  while ('\0' != *str) {
80  const char **valuePtr = nullptr;
81  int *size = nullptr;
82 
83  // -1 to avoid comparing null terminator
84  if (0 == strncmp(devNameStr, str, sizeof(devNameStr) - 1)) {
85  valuePtr = &decDescs.back().devName;
86  size = &decDescs.back().devNameSize;
87  str += sizeof(devNameStr) - 1;
88  } else if (0 ==
89  strncmp(platformNameStr, str, sizeof(platformNameStr) - 1)) {
90  valuePtr = &decDescs.back().platformName;
91  size = &decDescs.back().platformNameSize;
92  str += sizeof(platformNameStr) - 1;
93  } else if (0 == strncmp(platformVerStr, str, sizeof(platformVerStr) - 1)) {
94  valuePtr = &decDescs.back().platformVer;
95  size = &decDescs.back().platformVerSize;
96  str += sizeof(platformVerStr) - 1;
97  } else if (0 == strncmp(driverVerStr, str, sizeof(driverVerStr) - 1)) {
98  valuePtr = &decDescs.back().devDriverVer;
99  size = &decDescs.back().devDriverVerSize;
100  str += sizeof(driverVerStr) - 1;
101  }
102 
103  if (':' != *str)
104  throw sycl::runtime_error("Malformed device allowlist", PI_INVALID_VALUE);
105 
106  // Skip ':'
107  str += 1;
108 
109  if ('{' != *str || '{' != *(str + 1))
110  throw sycl::runtime_error("Malformed device allowlist", PI_INVALID_VALUE);
111 
112  // Skip opening sequence "{{"
113  str += 2;
114 
115  *valuePtr = str;
116 
117  // Increment until closing sequence is encountered
118  while (('\0' != *str) && ('}' != *str || '}' != *(str + 1)))
119  ++str;
120 
121  if ('\0' == *str)
122  throw sycl::runtime_error("Malformed device allowlist", PI_INVALID_VALUE);
123 
124  *size = str - *valuePtr;
125 
126  // Skip closing sequence "}}"
127  str += 2;
128 
129  if ('\0' == *str)
130  break;
131 
132  // '|' means that the is another filter
133  if ('|' == *str)
134  decDescs.emplace_back();
135  else if (',' != *str)
136  throw sycl::runtime_error("Malformed device allowlist", PI_INVALID_VALUE);
137 
138  ++str;
139  }
140 
141  return decDescs;
142 }

144 static void filterAllowList(vector_class<RT::PiDevice> &PiDevices,
145  RT::PiPlatform PiPlatform, const plugin &Plugin) {
146  const std::vector<DevDescT> AllowList(getAllowListDesc());
147  if (AllowList.empty())
148  return;
149 
150  const string_class PlatformName =
151  sycl::detail::get_platform_info<string_class, info::platform::name>::get(
152  PiPlatform, Plugin);
153 
154  const string_class PlatformVer =
155  sycl::detail::get_platform_info<string_class,
156  info::platform::version>::get(PiPlatform,
157  Plugin);
158 
159  int InsertIDx = 0;
160  for (RT::PiDevice Device : PiDevices) {
161  const string_class DeviceName =
162  sycl::detail::get_device_info<string_class, info::device::name>::get(
163  Device, Plugin);
164 
165  const string_class DeviceDriverVer = sycl::detail::get_device_info<
166  string_class, info::device::driver_version>::get(Device, Plugin);
167 
168  for (const DevDescT &Desc : AllowList) {
169  if (nullptr != Desc.platformName &&
170  !std::regex_match(PlatformName,
171  std::regex(std::string(Desc.platformName,
172  Desc.platformNameSize))))
173  continue;
174 
175  if (nullptr != Desc.platformVer &&
176  !std::regex_match(
177  PlatformVer,
178  std::regex(std::string(Desc.platformVer, Desc.platformVerSize))))
179  continue;
180 
181  if (nullptr != Desc.devName &&
182  !std::regex_match(DeviceName, std::regex(std::string(
183  Desc.devName, Desc.devNameSize))))
184  continue;
185 
186  if (nullptr != Desc.devDriverVer &&
187  !std::regex_match(DeviceDriverVer,
188  std::regex(std::string(Desc.devDriverVer,
189  Desc.devDriverVerSize))))
190  continue;
191 
192  PiDevices[InsertIDx++] = Device;
193  break;
194  }
195  }
196  PiDevices.resize(InsertIDx);
197 }

199 vector_class<device>
200 platform_impl::get_devices(info::device_type DeviceType) const {
201  vector_class<device> Res;
202  if (is_host() && (DeviceType == info::device_type::host ||
203  DeviceType == info::device_type::all)) {
204  Res.resize(1); // default device constructor creates host device
205  }
206 
207  // If any DeviceType other than host was requested for host platform,
208  // an empty vector will be returned.
209  if (is_host() || DeviceType == info::device_type::host)
210  return Res;
211 
212  pi_uint32 NumDevices;
213  const detail::plugin &Plugin = getPlugin();
214  Plugin.call<PiApiKind::piDevicesGet>(
215  MPlatform, pi::cast<RT::PiDeviceType>(DeviceType), 0,
216  pi::cast<RT::PiDevice *>(nullptr), &NumDevices);
217 
218  if (NumDevices == 0)
219  return Res;
220 
221  vector_class<RT::PiDevice> PiDevices(NumDevices);
222  // TODO catch an exception and put it to list of asynchronous exceptions
223  Plugin.call<PiApiKind::piDevicesGet>(MPlatform,
224  pi::cast<RT::PiDeviceType>(DeviceType),
225  NumDevices, PiDevices.data(), nullptr);
226 
227  // Filter out devices that are not present in the allowlist
228  if (SYCLConfig<SYCL_DEVICE_ALLOWLIST>::get())
229  filterAllowList(PiDevices, MPlatform, this->getPlugin());
230 
231  std::transform(PiDevices.begin(), PiDevices.end(), std::back_inserter(Res),
232  [this](const RT::PiDevice &PiDevice) -> device {
233  return detail::createSyclObjFromImpl<device>(
234  std::make_shared<device_impl>(
235  PiDevice, std::make_shared<platform_impl>(*this)));
236  });
237 
238  return Res;
239 }
