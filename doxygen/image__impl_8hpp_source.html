80  void setPitches() {
81  size_t WHD[3] = {1, 1, 1}; // Width, Height, Depth.
82  for (int I = 0; I < Dimensions; I++)
83  WHD[I] = MRange[I];
84  MRowPitch = MElementSize * WHD[0];
85  MSlicePitch = MRowPitch * WHD[1];
86  BaseT::MSizeInBytes = MSlicePitch * WHD[2];
87  }
88 
89  template <bool B = (Dimensions > 1)>
90  void setPitches(const EnableIfPitchT<B> Pitch) {
91  MRowPitch = Pitch[0];
92  MSlicePitch =
93  (Dimensions == 3) ? Pitch[1] : MRowPitch; // Dimensions will be 2/3.
94  // NumSlices is depth when dim==3, and height when dim==2.
95  size_t NumSlices =
96  (Dimensions == 3) ? MRange[2] : MRange[1]; // Dimensions will be 2/3.
97  BaseT::MSizeInBytes = MSlicePitch * NumSlices;
98  }
