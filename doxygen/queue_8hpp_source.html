374  // having _TWO_ mid-param #ifdefs makes the functions very difficult to read.
375  // Here we simplify the &CodeLoc declaration to be _CODELOCPARAM(&CodeLoc) and
376  // _CODELOCARG(&CodeLoc) Similarly, the KernelFunc param is simplified to be
377  // _KERNELFUNCPARAM(KernelFunc) Once the queue kernel functions are defined,
378  // these macros are #undef immediately.
379 
380  // replace _CODELOCPARAM(&CodeLoc) with nothing
381  // or : , const detail::code_location &CodeLoc =
382  // detail::code_location::current()
383  // replace _CODELOCARG(&CodeLoc) with nothing
384  // or : const detail::code_location &CodeLoc = {}
385 
386 #ifndef DISABLE_SYCL_INSTRUMENTATION_METADATA
387 #define _CODELOCPARAM(a) \
388  , const detail::code_location a = detail::code_location::current()
389

391 #else
392 #define _CODELOCPARAM(a)
393 
394 #define _CODELOCARG(a) const detail::code_location a = {}
395 #endif
