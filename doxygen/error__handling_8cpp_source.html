23 void handleOutOfResources(const device_impl &DeviceImpl,
24  ur_kernel_handle_t Kernel, const NDRDescT &NDRDesc) {
25  sycl::platform Platform = DeviceImpl.get_platform();
26  sycl::backend Backend = Platform.get_backend();
27  if (Backend == sycl::backend::ext_oneapi_cuda) {
28  // CUDA:
29  // UR_RESULT_ERROR_INVALID_WORK_GROUP_SIZE is returned when the kernel
30  // registers required for the launch config exceeds the maximum number of
31  // registers per block (UR_DEVICE_INFO_MAX_REGISTERS_PER_WORK_GROUP).
32  // This is if local_work_size[0] * ... * local_work_size[work_dim - 1]
33  // multiplied by UR_KERNEL_INFO_NUM_REGS is greater than the value of
34  // UR_DEVICE_INFO_MAX_REGISTERS_PER_WORK_GROUP. See Table 15: Technical
35  // Specifications per Compute Capability, for limitations.
36  const size_t TotalNumberOfWIs =
37  NDRDesc.LocalSize[0] * NDRDesc.LocalSize[1] * NDRDesc.LocalSize[2];
38 
39  const PluginPtr &Plugin = DeviceImpl.getPlugin();
40  uint32_t NumRegisters = 0;
41  Plugin->call(urKernelGetInfo, Kernel, UR_KERNEL_INFO_NUM_REGS,
42  sizeof(NumRegisters), &NumRegisters, nullptr);
43 
44  uint32_t MaxRegistersPerBlock =
45  DeviceImpl.get_info<ext::codeplay::experimental::info::device::
46  max_registers_per_work_group>();
47 
48  const bool HasExceededAvailableRegisters =
49  TotalNumberOfWIs * NumRegisters > MaxRegistersPerBlock;
50 
51  if (HasExceededAvailableRegisters) {
52  std::string message(
53  "Exceeded the number of registers available on the hardware.\n");
54  throw sycl::exception(
55  sycl::make_error_code(sycl::errc::nd_range),
56  // Additional information which can be helpful to the user.
57  message.append(
58  "\tThe number registers per work-group cannot exceed " +
59  std::to_string(MaxRegistersPerBlock) +
60  " for this kernel on this device.\n"
61  "\tThe kernel uses " +
62  std::to_string(NumRegisters) +
63  " registers per work-item for a total of " +
64  std::to_string(TotalNumberOfWIs) +
65  " work-items per work-group.\n"));
66  }
67  }
68  // Fallback
69  constexpr ur_result_t Error = UR_RESULT_ERROR_OUT_OF_RESOURCES;
70  throw sycl::exception(sycl::make_error_code(sycl::errc::runtime),
71  "UR backend failed. UR backend returns:" +
72  codeToString(Error));
73 }

75 void handleInvalidWorkGroupSize(const device_impl &DeviceImpl,
76  ur_kernel_handle_t Kernel,
77  const NDRDescT &NDRDesc) {
78  sycl::platform Platform = DeviceImpl.get_platform();
79 
80  // Some of the error handling below is special for particular OpenCL
81  // versions. If this is an OpenCL backend, get the version.
82  bool IsOpenCL = false; // Backend is any OpenCL version
83  bool IsOpenCLV1x = false; // Backend is OpenCL 1.x
84  bool IsOpenCLVGE20 = false; // Backend is Greater or Equal to OpenCL 2.0
85  bool IsLevelZero = false; // Backend is any OneAPI Level 0 version
86  auto Backend = Platform.get_backend();
87  if (Backend == sycl::backend::opencl) {
88  std::string VersionString =
89  DeviceImpl.get_info<info::device::version>().substr(7, 3);
90  IsOpenCL = true;
91  IsOpenCLV1x = (VersionString.find("1.") == 0);
92  IsOpenCLVGE20 =
93  (VersionString.find("2.") == 0) || (VersionString.find("3.") == 0);
94  } else if (Backend == sycl::backend::ext_oneapi_level_zero) {
95  IsLevelZero = true;
96  }
97 
98  const PluginPtr &Plugin = DeviceImpl.getPlugin();
99  ur_device_handle_t Device = DeviceImpl.getHandleRef();
100 
101  size_t CompileWGSize[3] = {0};
102  Plugin->call(urKernelGetGroupInfo, Kernel, Device,
103  UR_KERNEL_GROUP_INFO_COMPILE_WORK_GROUP_SIZE, sizeof(size_t) * 3,
104  CompileWGSize, nullptr);
105 
106  size_t MaxWGSize = 0;
107  Plugin->call(urDeviceGetInfo, Device, UR_DEVICE_INFO_MAX_WORK_GROUP_SIZE,
108  sizeof(size_t), &MaxWGSize, nullptr);
109 
110  const bool HasLocalSize = (NDRDesc.LocalSize[0] != 0);
111 
112  if (CompileWGSize[0] != 0) {
113  if (CompileWGSize[0] > MaxWGSize || CompileWGSize[1] > MaxWGSize ||
114  CompileWGSize[2] > MaxWGSize)
115  throw sycl::exception(
116  make_error_code(errc::kernel_not_supported),
117  "Submitting a kernel decorated with reqd_work_group_size attribute "
118  "to a device that does not support this work group size is invalid.");
119 
120  // OpenCL 1.x && 2.0:
121  // UR_RESULT_ERROR_INVALID_WORK_GROUP_SIZE if local_work_size is NULL and
122  // the reqd_work_group_size attribute is used to declare the work-group size
123  // for kernel in the program source.
124  if (!HasLocalSize && (IsOpenCLV1x || IsOpenCLVGE20)) {
125  throw sycl::exception(
126  make_error_code(errc::nd_range),
127  "OpenCL 1.x and 2.0 requires to pass local size argument even if "
128  "required work-group size was specified in the program source");
129  }
130  // UR_RESULT_ERROR_INVALID_WORK_GROUP_SIZE if local_work_size is specified
131  // and does not match the required work-group size for kernel in the program
132  // source.
133  if (NDRDesc.LocalSize[0] != CompileWGSize[0] ||
134  NDRDesc.LocalSize[1] != CompileWGSize[1] ||
135  NDRDesc.LocalSize[2] != CompileWGSize[2])
136  throw sycl::exception(
137  make_error_code(errc::nd_range),
138  "The specified local size {" + std::to_string(NDRDesc.LocalSize[2]) +
139  ", " + std::to_string(NDRDesc.LocalSize[1]) + ", " +
140  std::to_string(NDRDesc.LocalSize[0]) +
141  "} doesn't match the required work-group size specified "
142  "in the program source {" +
143  std::to_string(CompileWGSize[2]) + ", " +
144  std::to_string(CompileWGSize[1]) + ", " +
145  std::to_string(CompileWGSize[0]) + "}");
146  }
147 
148  if (HasLocalSize) {
149  size_t MaxThreadsPerBlock[3] = {};
150  Plugin->call(urDeviceGetInfo, Device, UR_DEVICE_INFO_MAX_WORK_ITEM_SIZES,
151  sizeof(MaxThreadsPerBlock), MaxThreadsPerBlock, nullptr);
152 
153  for (size_t I = 0; I < 3; ++I) {
154  if (MaxThreadsPerBlock[I] < NDRDesc.LocalSize[I]) {
155  throw sycl::exception(make_error_code(errc::nd_range),
156  "The number of work-items in each dimension of a "
157  "work-group cannot exceed {" +
158  std::to_string(MaxThreadsPerBlock[0]) + ", " +
159  std::to_string(MaxThreadsPerBlock[1]) + ", " +
160  std::to_string(MaxThreadsPerBlock[2]) +
161  "} for this device");
162  }
163  }
164  }
165 
166  if (IsOpenCLV1x) {
167  // OpenCL 1.x:
168  // UR_RESULT_ERROR_INVALID_WORK_GROUP_SIZE if local_work_size is specified
169  // and the total number of work-items in the work-group computed as
170  // local_work_size[0] * ... * local_work_size[work_dim - 1] is greater
171  // than the value specified by UR_DEVICE_INFO_MAX_WORK_GROUP_SIZE in
172  // table 4.3
173  const size_t TotalNumberOfWIs =
174  NDRDesc.LocalSize[0] * NDRDesc.LocalSize[1] * NDRDesc.LocalSize[2];
175  if (TotalNumberOfWIs > MaxWGSize)
176  throw sycl::exception(
177  make_error_code(errc::nd_range),
178  "Total number of work-items in a work-group cannot exceed " +
179  std::to_string(MaxWGSize));
180  } else if (IsOpenCLVGE20 || IsLevelZero) {
181  // OpenCL 2.x or OneAPI Level Zero:
182  // UR_RESULT_ERROR_INVALID_WORK_GROUP_SIZE if local_work_size is specified
183  // and the total number of work-items in the work-group computed as
184  // local_work_size[0] * ... * local_work_size[work_dim - 1] is greater
185  // than the value specified by UR_KERNEL_GROUP_INFO_WORK_GROUP_SIZE in
186  // table 5.21.
187  size_t KernelWGSize = 0;
188  Plugin->call(urKernelGetGroupInfo, Kernel, Device,
189  UR_KERNEL_GROUP_INFO_WORK_GROUP_SIZE, sizeof(size_t),
190  &KernelWGSize, nullptr);
191  const size_t TotalNumberOfWIs =
192  NDRDesc.LocalSize[0] * NDRDesc.LocalSize[1] * NDRDesc.LocalSize[2];
193  if (TotalNumberOfWIs > KernelWGSize)
194  throw sycl::exception(
195  make_error_code(errc::nd_range),
196  "Total number of work-items in a work-group cannot exceed " +
197  std::to_string(KernelWGSize) + " for this kernel");
198  } else {
199  // TODO: Should probably have something similar for the other backends
200  }
201 
202  if (HasLocalSize) {
203  // Is the global range size evenly divisible by the local workgroup size?
204  const bool NonUniformWGs =
205  (NDRDesc.LocalSize[0] != 0 &&
206  NDRDesc.GlobalSize[0] % NDRDesc.LocalSize[0] != 0) ||
207  (NDRDesc.LocalSize[1] != 0 &&
208  NDRDesc.GlobalSize[1] % NDRDesc.LocalSize[1] != 0) ||
209  (NDRDesc.LocalSize[2] != 0 &&
210  NDRDesc.GlobalSize[2] % NDRDesc.LocalSize[2] != 0);
211  // Is the local size of the workgroup greater than the global range size in
212  // any dimension?
213  if (IsOpenCL) {
214  const bool LocalExceedsGlobal =
215  NonUniformWGs && (NDRDesc.LocalSize[0] > NDRDesc.GlobalSize[0] ||
216  NDRDesc.LocalSize[1] > NDRDesc.GlobalSize[1] ||
217  NDRDesc.LocalSize[2] > NDRDesc.GlobalSize[2]);
218 
219  if (NonUniformWGs) {
220  if (IsOpenCLV1x) {
221  // OpenCL 1.x:
222  // UR_RESULT_ERROR_INVALID_WORK_GROUP_SIZE if local_work_size is
223  // specified and number of workitems specified by global_work_size is
224  // not evenly divisible by size of work-group given by local_work_size
225  if (LocalExceedsGlobal)
226  throw sycl::exception(make_error_code(errc::nd_range),
227  "Local workgroup size cannot be greater than "
228  "global range in any dimension");
229  else
230  throw sycl::exception(make_error_code(errc::nd_range),
231  "Global_work_size must be evenly divisible "
232  "by local_work_size. Non-uniform work-groups "
233  "are not supported by the target device");
234  } else {
235  // OpenCL 2.x:
236  // UR_RESULT_ERROR_INVALID_WORK_GROUP_SIZE if the program was compiled
237  // with –cl-uniform-work-group-size and the number of work-items
238  // specified by global_work_size is not evenly divisible by size of
239  // work-group given by local_work_size
240 
241  ur_program_handle_t Program = nullptr;
242  Plugin->call(urKernelGetInfo, Kernel, UR_KERNEL_INFO_PROGRAM,
243  sizeof(ur_program_handle_t), &Program, nullptr);
244  size_t OptsSize = 0;
245  Plugin->call(urProgramGetBuildInfo, Program, Device,
246  UR_PROGRAM_BUILD_INFO_OPTIONS, 0, nullptr, &OptsSize);
247  std::string Opts(OptsSize, '\0');
248  Plugin->call(urProgramGetBuildInfo, Program, Device,
249  UR_PROGRAM_BUILD_INFO_OPTIONS, OptsSize, &Opts.front(),
250  nullptr);
251  const bool HasStd20 = Opts.find("-cl-std=CL2.0") != std::string::npos;
252  const bool RequiresUniformWGSize =
253  Opts.find("-cl-uniform-work-group-size") != std::string::npos;
254  std::string LocalWGSize = std::to_string(NDRDesc.LocalSize[0]) +
255  ", " +
256  std::to_string(NDRDesc.LocalSize[1]) +
257  ", " + std::to_string(NDRDesc.LocalSize[2]);
258  std::string GlobalWGSize =
259  std::to_string(NDRDesc.GlobalSize[0]) + ", " +
260  std::to_string(NDRDesc.GlobalSize[1]) + ", " +
261  std::to_string(NDRDesc.GlobalSize[2]);
262  std::string message =
263  LocalExceedsGlobal
264  ? "Local work-group size {" + LocalWGSize +
265  "} is greater than global range size {" + GlobalWGSize +
266  "}. "
267  : "Global work size {" + GlobalWGSize +
268  "} is not evenly divisible by local work-group size {" +
269  LocalWGSize + "}. ";
270  if (!HasStd20)
271  throw sycl::exception(
272  make_error_code(errc::nd_range),
273  message.append("Non-uniform work-groups are not allowed by "
274  "default. Underlying OpenCL 2.x implementation "
275  "supports this feature and to enable it, build "
276  "device program with -cl-std=CL2.0"));
277  else if (RequiresUniformWGSize)
278  throw sycl::exception(
279  make_error_code(errc::nd_range),
280  message.append("Non-uniform work-groups are not allowed when "
281  "-cl-uniform-work-group-size flag is used. "
282  "Underlying OpenCL 2.x implementation supports "
283  "this feature, but it is being disabled by "
284  "-cl-uniform-work-group-size build flag"));
285  // else unknown. fallback (below)
286  }
287  }
288  } else {
289  // TODO: Decide what checks (if any) we need for the other backends
290  }
291  throw sycl::exception(
292  make_error_code(errc::nd_range),
293  "Non-uniform work-groups are not supported by the target device");
294  }
295  // TODO: required number of sub-groups, OpenCL 2.1:
296  // UR_RESULT_ERROR_INVALID_WORK_GROUP_SIZE if local_work_size is specified and
297  // is not consistent with the required number of sub-groups for kernel in the
298  // program source.
299 
300  throw exception(make_error_code(errc::nd_range),
301  "internal error: expected HasLocalSize");
302 }

304 void handleInvalidWorkItemSize(const device_impl &DeviceImpl,
305  const NDRDescT &NDRDesc) {
306 
307  const PluginPtr &Plugin = DeviceImpl.getPlugin();
308  ur_device_handle_t Device = DeviceImpl.getHandleRef();
309 
310  size_t MaxWISize[] = {0, 0, 0};
311 
312  Plugin->call(urDeviceGetInfo, Device, UR_DEVICE_INFO_MAX_WORK_ITEM_SIZES,
313  sizeof(MaxWISize), &MaxWISize, nullptr);
314  for (unsigned I = 0; I < NDRDesc.Dims; I++) {
315  if (NDRDesc.LocalSize[I] > MaxWISize[I])
316  throw sycl::exception(
317  make_error_code(errc::nd_range),
318  "Number of work-items in a work-group exceed limit for dimension " +
319  std::to_string(I) + " : " + std::to_string(NDRDesc.LocalSize[I]) +
320  " > " + std::to_string(MaxWISize[I]));
321  }
322 }

324 void handleInvalidValue(const device_impl &DeviceImpl,
325  const NDRDescT &NDRDesc) {
326  const PluginPtr &Plugin = DeviceImpl.getPlugin();
327  ur_device_handle_t Device = DeviceImpl.getHandleRef();
328 
329  size_t MaxNWGs[] = {0, 0, 0};
330  Plugin->call(urDeviceGetInfo, Device, UR_DEVICE_INFO_MAX_WORK_GROUPS_3D,
331  sizeof(MaxNWGs), &MaxNWGs, nullptr);
332  for (unsigned int I = 0; I < NDRDesc.Dims; I++) {
333  size_t NWgs = NDRDesc.GlobalSize[I] / NDRDesc.LocalSize[I];
334  if (NWgs > MaxNWGs[I])
335  throw sycl::exception(
336  make_error_code(errc::nd_range),
337  "Number of work-groups exceed limit for dimension " +
338  std::to_string(I) + " : " + std::to_string(NWgs) + " > " +
339  std::to_string(MaxNWGs[I]));
340  }
341 
342  // fallback
343  throw exception(make_error_code(errc::nd_range), "unknown internal error");
344 }

346 void handleErrorOrWarning(ur_result_t Error, const device_impl &DeviceImpl,
347  ur_kernel_handle_t Kernel, const NDRDescT &NDRDesc) {
348  assert(Error != UR_RESULT_SUCCESS &&
349  "Success is expected to be handled on caller side");
350  switch (Error) {
351  case UR_RESULT_ERROR_OUT_OF_DEVICE_MEMORY:
352  case UR_RESULT_ERROR_OUT_OF_RESOURCES:
353  return handleOutOfResources(DeviceImpl, Kernel, NDRDesc);
354 
355  case UR_RESULT_ERROR_INVALID_WORK_GROUP_SIZE:
356  return handleInvalidWorkGroupSize(DeviceImpl, Kernel, NDRDesc);
357 
358  case UR_RESULT_ERROR_INVALID_KERNEL_ARGS:
359  throw detail::set_ur_error(
360  sycl::exception(
361  make_error_code(errc::kernel_argument),
362  "The kernel argument values have not been specified OR a kernel "
363  "argument declared to be a pointer to a type."),
364  UR_RESULT_ERROR_INVALID_KERNEL_ARGS);
365 
366  case UR_RESULT_ERROR_INVALID_WORK_ITEM_SIZE:
367  return handleInvalidWorkItemSize(DeviceImpl, NDRDesc);
368 
369  case UR_RESULT_ERROR_UNSUPPORTED_IMAGE_FORMAT:
370  throw detail::set_ur_error(
371  sycl::exception(
372  make_error_code(errc::feature_not_supported),
373  "image object is specified as an argument value and the image "
374  "format is not supported by device associated with queue"),
375  UR_RESULT_ERROR_UNSUPPORTED_IMAGE_FORMAT);
376 
377  case UR_RESULT_ERROR_MISALIGNED_SUB_BUFFER_OFFSET:
378  throw detail::set_ur_error(
379  sycl::exception(make_error_code(errc::invalid),
380  "a sub-buffer object is specified as the value for an "
381  "argument that is a buffer object and the offset "
382  "specified when the sub-buffer object is created is "
383  "not aligned to CL_DEVICE_MEM_BASE_ADDR_ALIGN value "
384  "for device associated with queue"),
385  UR_RESULT_ERROR_MISALIGNED_SUB_BUFFER_OFFSET);
386 
387  case UR_RESULT_ERROR_MEM_OBJECT_ALLOCATION_FAILURE:
388  throw detail::set_ur_error(
389  sycl::exception(
390  make_error_code(errc::memory_allocation),
391  "failure to allocate memory for data store associated with image "
392  "or buffer objects specified as arguments to kernel"),
393  UR_RESULT_ERROR_MEM_OBJECT_ALLOCATION_FAILURE);
394 
395  case UR_RESULT_ERROR_INVALID_IMAGE_SIZE:
396  throw detail::set_ur_error(
397  sycl::exception(
398  make_error_code(errc::invalid),
399  "image object is specified as an argument value and the image "
400  "dimensions (image width, height, specified or compute row and/or "
401  "slice pitch) are not supported by device associated with queue"),
402  UR_RESULT_ERROR_INVALID_IMAGE_SIZE);
403 
404  case UR_RESULT_ERROR_INVALID_VALUE:
405  return handleInvalidValue(DeviceImpl, NDRDesc);
406 
407  case UR_RESULT_ERROR_ADAPTER_SPECIFIC:
408  // checkUrResult does all the necessary handling for
409  // UR_RESULT_ERROR_ADAPTER_SPECIFIC_ERROR, making sure an error is thrown or
410  // not, depending on whether UR_RESULT_ERROR_ADAPTER_SPECIFIC_ERROR contains
411  // an error or a warning. It also ensures that the contents of the error
412  // message buffer (used only by UR_RESULT_ERROR_ADAPTER_SPECIFIC_ERROR) get
413  // handled correctly.
414  return DeviceImpl.getPlugin()->checkUrResult(Error);
415 
416  // TODO: Handle other error codes
417 
418  default:
419  throw detail::set_ur_error(
420  exception(make_error_code(errc::runtime), "UR error"), Error);
421  }
422 }
