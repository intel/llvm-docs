44 program_impl::program_impl(
45  vector_class<shared_ptr_class<program_impl>> ProgramList,
46  string_class LinkOptions, const property_list &PropList)
47  : MState(program_state::linked), MPropList(PropList),
48  MLinkOptions(LinkOptions), MBuildOptions(LinkOptions) {
49  // Verify arguments
50  if (ProgramList.empty()) {
51  throw runtime_error("Non-empty vector of programs expected",
52  PI_INVALID_VALUE);
53  }
54 
55  // Sort the programs to avoid deadlocks due to locking multiple mutexes &
56  // verify that all programs are unique.
57  std::sort(ProgramList.begin(), ProgramList.end());
58  auto It = std::unique(ProgramList.begin(), ProgramList.end());
59  if (It != ProgramList.end()) {
60  throw runtime_error("Attempting to link a program with itself",
61  PI_INVALID_PROGRAM);
62  }
63 
64  MContext = ProgramList[0]->MContext;
65  if (MContext->getDevices().size() > 1) {
66  throw feature_not_supported(
67  "multiple devices within a context are not supported with "
68  "sycl::program and sycl::kernel",
69  PI_INVALID_OPERATION);
70  }
71  MDevices = ProgramList[0]->MDevices;
72  vector_class<device> DevicesSorted;
73  if (!is_host()) {
74  DevicesSorted = sort_devices_by_cl_device_id(MDevices);
75  }
76  check_device_feature_support<info::device::is_linker_available>(MDevices);
77  std::list<std::lock_guard<std::mutex>> Locks;
78  for (const auto &Prg : ProgramList) {
79  Locks.emplace_back(Prg->MMutex);
80  Prg->throw_if_state_is_not(program_state::compiled);
81  if (Prg->MContext != MContext) {
82  throw invalid_object_error(
83  "Not all programs are associated with the same context",
84  PI_INVALID_PROGRAM);
85  }
86  if (!is_host()) {
87  vector_class<device> PrgDevicesSorted =
88  sort_devices_by_cl_device_id(Prg->MDevices);
89  if (PrgDevicesSorted != DevicesSorted) {
90  throw invalid_object_error(
91  "Not all programs are associated with the same devices",
92  PI_INVALID_PROGRAM);
93  }
94  }
95  }
96 
97  if (!is_host()) {
98  vector_class<RT::PiDevice> Devices(get_pi_devices());
99  vector_class<RT::PiProgram> Programs;
100  bool NonInterOpToLink = false;
101  for (const auto &Prg : ProgramList) {
102  if (!Prg->MLinkable && NonInterOpToLink)
103  continue;
104  NonInterOpToLink |= !Prg->MLinkable;
105  Programs.push_back(Prg->MProgram);
106  }
107  const detail::plugin &Plugin = getPlugin();
108  RT::PiResult Err = Plugin.call_nocheck<PiApiKind::piProgramLink>(
109  MContext->getHandleRef(), Devices.size(), Devices.data(),
110  LinkOptions.c_str(), Programs.size(), Programs.data(), nullptr, nullptr,
111  &MProgram);
112  Plugin.checkPiResult<compile_program_error>(Err);
113  }
114 }

120 program_impl::program_impl(ContextImplPtr Context,
121  pi_native_handle InteropProgram,
122  RT::PiProgram Program)
123  : MProgram(Program), MContext(Context), MLinkable(true) {
124 
125  if (Context->getDevices().size() > 1) {
126  throw feature_not_supported(
127  "multiple devices within a context are not supported with "
128  "sycl::program and sycl::kernel",
129  PI_INVALID_OPERATION);
130  }
131 
132  const detail::plugin &Plugin = getPlugin();
133  if (MProgram == nullptr) {
134  assert(InteropProgram &&
135  "No InteropProgram/PiProgram defined with piextProgramFromNative");
136  // Translate the raw program handle into PI program.
137  Plugin.call<PiApiKind::piextProgramCreateWithNativeHandle>(
138  InteropProgram, MContext->getHandleRef(), &MProgram);
139  } else
140  Plugin.call<PiApiKind::piProgramRetain>(Program);
141 
142  // TODO handle the case when cl_program build is in progress
143  pi_uint32 NumDevices;
144  Plugin.call<PiApiKind::piProgramGetInfo>(
145  MProgram, PI_PROGRAM_INFO_NUM_DEVICES, sizeof(pi_uint32), &NumDevices,
146  nullptr);
147  vector_class<RT::PiDevice> PiDevices(NumDevices);
148  Plugin.call<PiApiKind::piProgramGetInfo>(MProgram, PI_PROGRAM_INFO_DEVICES,
149  sizeof(RT::PiDevice) * NumDevices,
150  PiDevices.data(), nullptr);
151  vector_class<device> SyclContextDevices =
152  MContext->get_info<info::context::devices>();
153 
154  // Keep only the subset of the devices (associated with context) that
155  // were actually used to create the program.
156  // This is possible when clCreateProgramWithBinary is used.
157  auto NewEnd = std::remove_if(
158  SyclContextDevices.begin(), SyclContextDevices.end(),
159  [&PiDevices](const sycl::device &Dev) {
160  return PiDevices.end() ==
161  std::find(PiDevices.begin(), PiDevices.end(),
162  detail::getSyclObjImpl(Dev)->getHandleRef());
163  });
164  SyclContextDevices.erase(NewEnd, SyclContextDevices.end());
165  MDevices = SyclContextDevices;
166  RT::PiDevice Device = getSyclObjImpl(MDevices[0])->getHandleRef();
167  assert(!MDevices.empty() && "No device found for this program");
168  // TODO check build for each device instead
169  cl_program_binary_type BinaryType;
170  Plugin.call<PiApiKind::piProgramGetBuildInfo>(
171  MProgram, Device, CL_PROGRAM_BINARY_TYPE, sizeof(cl_program_binary_type),
172  &BinaryType, nullptr);
173  size_t Size = 0;
174  Plugin.call<PiApiKind::piProgramGetBuildInfo>(
175  MProgram, Device, CL_PROGRAM_BUILD_OPTIONS, 0, nullptr, &Size);
176  std::vector<char> OptionsVector(Size);
177  Plugin.call<PiApiKind::piProgramGetBuildInfo>(MProgram, Device,
178  CL_PROGRAM_BUILD_OPTIONS, Size,
179  OptionsVector.data(), nullptr);
180  string_class Options(OptionsVector.begin(), OptionsVector.end());
181  switch (BinaryType) {
182  case CL_PROGRAM_BINARY_TYPE_NONE:
183  MState = program_state::none;
184  break;
185  case CL_PROGRAM_BINARY_TYPE_COMPILED_OBJECT:
186  MState = program_state::compiled;
187  MCompileOptions = Options;
188  MBuildOptions = Options;
189  break;
190  case CL_PROGRAM_BINARY_TYPE_LIBRARY:
191  case CL_PROGRAM_BINARY_TYPE_EXECUTABLE:
192  MState = program_state::linked;
193  MLinkOptions = "";
194  MBuildOptions = Options;
195  }
196 }

425 RT::PiKernel program_impl::get_pi_kernel(const string_class &KernelName) const {
426  RT::PiKernel Kernel = nullptr;
427 
428  if (is_cacheable()) {
429  std::tie(Kernel, std::ignore) =
430  ProgramManager::getInstance().getOrCreateKernel(
431  MProgramModuleHandle, get_context(), get_devices()[0], KernelName,
432  this);
433  getPlugin().call<PiApiKind::piKernelRetain>(Kernel);
434  } else {
435  const detail::plugin &Plugin = getPlugin();
436  RT::PiResult Err = Plugin.call_nocheck<PiApiKind::piKernelCreate>(
437  MProgram, KernelName.c_str(), &Kernel);
438  if (Err == PI_INVALID_KERNEL_NAME) {
439  throw invalid_object_error(
440  "This instance of program does not contain the kernel requested",
441  Err);
442  }
443  Plugin.checkPiResult(Err);
444 
445  // Some PI Plugins (like OpenCL) require this call to enable USM
446  // For others, PI will turn this into a NOP.
447  Plugin.call<PiApiKind::piKernelSetExecInfo>(Kernel, PI_USM_INDIRECT_ACCESS,
448  sizeof(pi_bool), &PI_TRUE);
449  }
450 
451  return Kernel;
452 }

521 void program_impl::flush_spec_constants(const RTDeviceBinaryImage &Img,
522  RT::PiProgram NativePrg) const {
523  // iterate via all specialization constants the program's image depends on,
524  // and set each to current runtime value (if any)
525  const pi::DeviceBinaryImage::PropertyRange &SCRange = Img.getSpecConstants();
526  ContextImplPtr Ctx = getSyclObjImpl(get_context());
527  using SCItTy = pi::DeviceBinaryImage::PropertyRange::ConstIterator;
528 
529  auto LockGuard = Ctx->getKernelProgramCache().acquireCachedPrograms();
530 
531  for (SCItTy SCIt : SCRange) {
532  const char *SCName = (*SCIt)->Name;
533  auto SCEntry = SpecConstRegistry.find(SCName);
534  if (SCEntry == SpecConstRegistry.end())
535  // spec constant has not been set in user code - SPIR-V will use default
536  continue;
537  const spec_constant_impl &SC = SCEntry->second;
538  assert(SC.isSet() && "uninitialized spec constant");
539  pi_device_binary_property SCProp = *SCIt;
540  pi_uint32 ID = pi::DeviceBinaryProperty(SCProp).asUint32();
541  NativePrg = NativePrg ? NativePrg : getHandleRef();
542  Ctx->getPlugin().call<PiApiKind::piextProgramSetSpecializationConstant>(
543  NativePrg, ID, SC.getSize(), SC.getValuePtr());
544  }
545 }
