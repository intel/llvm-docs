45 template <int Dim>
46 sycl::nd_range<3> transform_nd_range(const sycl::nd_range<Dim> &range) {
47  sycl::range<Dim> global_range = range.get_global_range();
48  sycl::range<Dim> local_range = range.get_local_range();
49  if constexpr (Dim == 3) {
50  return range;
51  } else if constexpr (Dim == 2) {
52  return sycl::nd_range<3>{{1, global_range[0], global_range[1]},
53  {1, local_range[0], local_range[1]}};
54  }
55  return sycl::nd_range<3>{{1, 1, global_range[0]}, {1, 1, local_range[0]}};
56 }

60 launch(const sycl::nd_range<3> &range, sycl::queue q, Args... args) {
61  static_assert(detail::getArgumentCount(F) == sizeof...(args),
62  "Wrong number of arguments to SYCL kernel");

67  return q.parallel_for(

73 template <int Dim>
74 inline sycl::nd_range<Dim> compute_nd_range(sycl::range<Dim> global_size_in,
75  sycl::range<Dim> work_group_size) {
76 
77  if (global_size_in.size() == 0 || work_group_size.size() == 0) {
78  throw std::invalid_argument("Global or local size is zero!");
79  }
80  for (size_t i = 0; i < Dim; ++i) {
81  if (global_size_in[i] < work_group_size[i])
82  throw std::invalid_argument("Work group size larger than global size");
83  }
84 
85  auto global_size =
86  ((global_size_in + work_group_size - 1) / work_group_size) *
87  work_group_size;
88  return {global_size, work_group_size};
89 }

127 template <auto F, typename LaunchPolicy, typename... Args>
128 sycl::event launch(LaunchPolicy launch_policy, sycl::queue q, Args... args) {
129  static_assert(syclcompat::args_compatible<LaunchPolicy, F, Args...>,
130  "Mismatch between device function signature and supplied "
131  "arguments. Have you correctly handled local memory/char*?");
132 
133  sycl_exp::launch_config config(launch_policy.get_range(),
134  launch_policy.get_launch_properties());
135 
136  return sycl_exp::submit_with_event(q, [&](sycl::handler &cgh) {
137  auto KernelFunctor = build_kernel_functor<F>(cgh, launch_policy, args...);
138  if constexpr (syclcompat::detail::is_range_v<
139  typename LaunchPolicy::RangeT>) {
140  parallel_for(cgh, config, KernelFunctor);
141  } else {
142  static_assert(
143  syclcompat::detail::is_nd_range_v<typename LaunchPolicy::RangeT>);
144  nd_launch(cgh, config, KernelFunctor);
145  }
146  });
147 }
