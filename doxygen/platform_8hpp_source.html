59  using ParamNoRef = std::remove_reference_t<ParamT>;
60  if constexpr (std::is_same_v<ParamNoRef, std::string>) {
61  return detail::string{Info};
62  } else if constexpr (std::is_same_v<ParamNoRef, std::vector<std::string>>) {
