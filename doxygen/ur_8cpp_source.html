246 // Reads an integer value from ELF data.
247 template <typename ResT>
248 static ResT readELFValue(const unsigned char *Data, size_t NumBytes,
249  bool IsBigEndian) {
250  assert(NumBytes <= sizeof(ResT));
251  ResT Result = 0;
252  if (IsBigEndian) {
253  for (size_t I = 0; I < NumBytes; ++I) {
254  Result = (Result << 8) | static_cast<ResT>(Data[I]);
255  }
256  } else {
257  std::copy(Data, Data + NumBytes, reinterpret_cast<char *>(&Result));
258  }
259  return Result;
260 }

262 // Checks if an ELF image contains a section with a specified name.
263 static bool checkELFSectionPresent(const std::string &ExpectedSectionName,
264  const unsigned char *ImgData,
265  size_t ImgSize) {
266  // Check for 64bit and big-endian.
267  bool Is64bit = ImgData[4] == 2;
268  bool IsBigEndian = ImgData[5] == 2;
269 
270  // Make offsets based on whether the ELF file is 64bit or not.
271  size_t SectionHeaderOffsetInfoOffset = Is64bit ? 0x28 : 0x20;
272  size_t SectionHeaderSizeInfoOffset = Is64bit ? 0x3A : 0x2E;
273  size_t SectionHeaderNumInfoOffset = Is64bit ? 0x3C : 0x30;
274  size_t SectionStringsHeaderIndexInfoOffset = Is64bit ? 0x3E : 0x32;
275 
276  // if the image doesn't contain enough data for the header values, end early.
277  if (ImgSize < SectionStringsHeaderIndexInfoOffset + 2)
278  return false;
279 
280  // Read the e_shoff, e_shentsize, e_shnum, and e_shstrndx entries in the
281  // header.
282  uint64_t SectionHeaderOffset = readELFValue<uint64_t>(
283  ImgData + SectionHeaderOffsetInfoOffset, Is64bit ? 8 : 4, IsBigEndian);
284  uint16_t SectionHeaderSize = readELFValue<uint16_t>(
285  ImgData + SectionHeaderSizeInfoOffset, 2, IsBigEndian);
286  uint16_t SectionHeaderNum = readELFValue<uint16_t>(
287  ImgData + SectionHeaderNumInfoOffset, 2, IsBigEndian);
288  uint16_t SectionStringsHeaderIndex = readELFValue<uint16_t>(
289  ImgData + SectionStringsHeaderIndexInfoOffset, 2, IsBigEndian);
290 
291  // End early if we do not have the expected number of section headers or
292  // if the read section string header index is out-of-range.
293  if (ImgSize < SectionHeaderOffset + SectionHeaderNum * SectionHeaderSize ||
294  SectionStringsHeaderIndex >= SectionHeaderNum)
295  return false;
296 
297  // Get the location of the section string data.
298  size_t SectionStringsInfoOffset = Is64bit ? 0x18 : 0x10;
299  const unsigned char *SectionStringsHeaderData =
300  ImgData + SectionHeaderOffset +
301  SectionStringsHeaderIndex * SectionHeaderSize;
302  uint64_t SectionStrings = readELFValue<uint64_t>(
303  SectionStringsHeaderData + SectionStringsInfoOffset, Is64bit ? 8 : 4,
304  IsBigEndian);
305  const unsigned char *SectionStringsData = ImgData + SectionStrings;
306 
307  // For each section, check the name against the expected section and return
308  // true if we find it.
309  for (size_t I = 0; I < SectionHeaderNum; ++I) {
310  // Get the offset into the section string data of this sections name.
311  const unsigned char *HeaderData =
312  ImgData + SectionHeaderOffset + I * SectionHeaderSize;
313  uint32_t SectionNameOffset =
314  readELFValue<uint32_t>(HeaderData, 4, IsBigEndian);
315 
316  // Read the section name and check if it is the same as the name we are
317  // looking for.
318  const char *SectionName =
319  reinterpret_cast<const char *>(SectionStringsData + SectionNameOffset);
320  if (SectionName == ExpectedSectionName)
321  return true;
322  }
323  return false;
324 }

327 static uint16_t getELFHeaderType(const unsigned char *ImgData, size_t ImgSize) {
328  (void)ImgSize;
329  assert(ImgSize >= 18 && "Not enough bytes to have an ELF header type.");

335 sycl_device_binary_type getBinaryImageFormat(const unsigned char *ImgData,
336  size_t ImgSize) {
337  // Top-level magic numbers for the recognized binary image formats.
338  auto MatchMagicNumber = [&](auto Number) {
339  return ImgSize >= sizeof(Number) &&
340  std::memcmp(ImgData, &Number, sizeof(Number)) == 0;
341  };
342 
343  if (MatchMagicNumber(uint32_t{0x07230203}))
344  return SYCL_DEVICE_BINARY_TYPE_SPIRV;
345 
346  if (MatchMagicNumber(uint32_t{0xDEC04342}))
347  return SYCL_DEVICE_BINARY_TYPE_LLVMIR_BITCODE;
348 
349  if (MatchMagicNumber(uint32_t{0x43544E49}))
350  // 'I', 'N', 'T', 'C' ; Intel native
351  return SYCL_DEVICE_BINARY_TYPE_LLVMIR_BITCODE;
352 
353  // Check for ELF format, size requirements include data we'll read in case of
354  // succesful match.
355  if (ImgSize >= 18 && MatchMagicNumber(uint32_t{0x464c457F})) {
356  uint16_t ELFHdrType = getELFHeaderType(ImgData, ImgSize);
357  if (ELFHdrType == 0xFF04)
358  // OpenCL executable.
359  return SYCL_DEVICE_BINARY_TYPE_NATIVE;
360 
361  if (ELFHdrType == 0xFF12)
362  // ZEBIN executable.
363  return SYCL_DEVICE_BINARY_TYPE_NATIVE;
364 
365  // Newer ZEBIN format does not have a special header type, but can instead
366  // be identified by having a required .ze_info section.
367  if (checkELFSectionPresent(".ze_info", ImgData, ImgSize))
368  return SYCL_DEVICE_BINARY_TYPE_NATIVE;
369  }
370 
371  if (MatchMagicNumber(std::array{'!', '<', 'a', 'r', 'c', 'h', '>', '\n'}))
372  // "ar" format is used to pack binaries for multiple devices, e.g. via
373  //
374  // -Xsycl-target-backend=spir64_gen "-device acm-g10,acm-g11"
375  //
376  // option.
377  return SYCL_DEVICE_BINARY_TYPE_NATIVE;
378 
379  return SYCL_DEVICE_BINARY_TYPE_NONE;
380 }

382 ur_program_metadata_t mapDeviceBinaryPropertyToProgramMetadata(
383  const sycl_device_binary_property &DeviceBinaryProperty) {
384  ur_program_metadata_t URMetadata{};
385  URMetadata.pName = DeviceBinaryProperty->Name;
386  URMetadata.size = DeviceBinaryProperty->ValSize;
387  switch (DeviceBinaryProperty->Type) {
388  case SYCL_PROPERTY_TYPE_UINT32:
389  URMetadata.type = UR_PROGRAM_METADATA_TYPE_UINT32;
390  URMetadata.value.data32 = DeviceBinaryProperty->ValSize;
391  break;
392  case SYCL_PROPERTY_TYPE_BYTE_ARRAY:
393  URMetadata.type = UR_PROGRAM_METADATA_TYPE_BYTE_ARRAY;
394  URMetadata.value.pData = DeviceBinaryProperty->ValAddr;
395  break;
396  case SYCL_PROPERTY_TYPE_STRING:
397  URMetadata.type = UR_PROGRAM_METADATA_TYPE_STRING;
398  URMetadata.value.pString =
399  reinterpret_cast<char *>(DeviceBinaryProperty->ValAddr);
400  break;
401  default:
402  break;
403  }
404  return URMetadata;
405 }
