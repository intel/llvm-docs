56 constexpr bool checkMutuallyExclusive(const fp_mode &mode) {
57  // Check that either none of the flags is set or only one flag is set.
58  fp_mode roundMask = fp_mode::round_to_nearest | fp_mode::round_upward |
59  fp_mode::round_downward | fp_mode::round_toward_zero;
60  bool isCorrectRoundingMode = ((mode & roundMask) == fp_mode(0)) ||
61  (isSet(mode, fp_mode::round_to_nearest) ^
62  isSet(mode, fp_mode::round_upward) ^
63  isSet(mode, fp_mode::round_downward) ^
64  isSet(mode, fp_mode::round_toward_zero));
65  // Check that if denorm_ftz is set then other denorm flags are not set.
66  fp_mode denormAllowMask = fp_mode::denorm_hf_allow | fp_mode::denorm_f_allow |
67  fp_mode::denorm_d_allow;
68  bool isCorrectDenormMode = !isSet(mode, fp_mode::denorm_ftz) ||
69  ((mode & denormAllowMask) == fp_mode(0));
70  return isCorrectRoundingMode && isCorrectDenormMode;
71 }

73 constexpr fp_mode setDefaultValuesIfNeeded(fp_mode mode) {
74  fp_mode roundMask = fp_mode::round_to_nearest | fp_mode::round_upward |
75  fp_mode::round_downward | fp_mode::round_toward_zero;
76  if ((mode & roundMask) == fp_mode(0))
77  mode = mode | fp_mode::round_to_nearest;
78 
79  auto denormMask = fp_mode::denorm_ftz | fp_mode::denorm_hf_allow |
80  fp_mode::denorm_f_allow | fp_mode::denorm_d_allow;
81  if ((mode & denormMask) == fp_mode(0))
82  mode = mode | fp_mode::denorm_ftz;
83 
84  return mode;
85 }
