226  class ConstIterator

228  ValTy, // value_type
229  ptrdiff_t, // difference_type
230  const pi_device_binary_property, // pointer
231  pi_device_binary_property> // reference
232  {
233  pi_device_binary_property Cur;
234 
235  public:
236  ConstIterator(pi_device_binary_property Cur = nullptr) : Cur(Cur) {}
237  ConstIterator &operator++() {
238  Cur++;
239  return *this;
240  }
241  ConstIterator operator++(int) {
242  ConstIterator Ret = *this;
243  ++(*this);
244  return Ret;
245  }
246  bool operator==(ConstIterator Other) const { return Cur == Other.Cur; }
247  bool operator!=(ConstIterator Other) const { return !(*this == Other); }
248  reference operator*() const { return Cur; }
