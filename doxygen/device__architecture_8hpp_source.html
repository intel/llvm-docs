1085 // Call the callable object "fn" only when this code runs on a device which
1086 // has a certain set of aspects or a particular architecture.
1087 //
1088 // Condition is a parameter pack of int's that define a simple expression

1100 // The "Condition" parameter pack above is a sequence of int's that define an
1101 // expression tree. Each node represents a boolean subexpression:
1102 //
1103 // ConditionAspect - Next int is a value from "enum aspect". The
1104 // subexpression is true if the device has this
1105 // aspect.
1106 // ConditionArchitecture - Next int is a value from "enum architecture". The
1107 // subexpression is true if the device has this
1108 // architecture.
1109 // ConditionNot - Next int is the root of another subexpression S1.
1110 // This subexpression is true if S1 is false.
1111 // ConditionAnd - Next int is the root of another subexpression S1.
1112 // The int following that subexpression is the root
1113 // of another subexpression S2. This subexpression
1114 // is true if both S1 and S2 are true.
1115 // ConditionOr - Next int is the root of another subexpression S1.
1116 // The int following that subexpression is the root
1117 // of another subexpression S2. This subexpression
1118 // is true if either S1 or S2 are true.
1119 //
1120 // These values are stored in the application's executable, so they are
1121 // effectively part of the ABI. Therefore, any change to an existing value
1122 // is an ABI break.
1123 //
1124 // There is no programmatic reason for the values to be negative. They are
1125 // negative only by convention to make it easier for humans to distinguish them
1126 // from aspect or architecture values (which are positive).
1127 static constexpr int ConditionAspect = -1;
1128 static constexpr int ConditionArchitecture = -2;
1129 static constexpr int ConditionNot = -3;
1130 static constexpr int ConditionAnd = -4;
1131 static constexpr int ConditionOr = -5;

1133 // Metaprogramming helper to construct a ConditionOr expression for a sequence
1134 // of architectures. "ConditionAnyArchitectureBuilder<Archs...>::seq" is an
1135 // "std::integer_sequence" representing the expression.
1136 template <architecture... Archs> struct ConditionAnyArchitectureBuilder;
1137 
1138 template <architecture Arch, architecture... Archs>
1139 struct ConditionAnyArchitectureBuilder<Arch, Archs...> {
1140  template <int I1, int I2, int I3, int... Is>
1141  static auto append(std::integer_sequence<int, Is...>) {
1142  return std::integer_sequence<int, I1, I2, I3, Is...>{};
1143  }
1144  using rest = typename ConditionAnyArchitectureBuilder<Archs...>::seq;
1145  static constexpr int arch = static_cast<int>(Arch);
1146  using seq =
1147  decltype(append<ConditionOr, ConditionArchitecture, arch>(rest{}));
1148 };
1149 
1150 template <architecture Arch> struct ConditionAnyArchitectureBuilder<Arch> {
1151  static constexpr int arch = static_cast<int>(Arch);
1152  using seq = std::integer_sequence<int, ConditionArchitecture, arch>;
1153 };

1155 // Metaprogramming helper to construct a ConditionNot expression.
1156 // ConditionNotBuilder<Exp>::seq" is an "std::integer_sequence" representing
1157 // the expression.
1158 template <typename Exp> struct ConditionNotBuilder {
1159  template <int I, int... Is>
1160  static auto append(std::integer_sequence<int, Is...>) {
1161  return std::integer_sequence<int, I, Is...>{};
1162  }
1163  using rest = typename Exp::seq;
1164  using seq = decltype(append<ConditionNot>(rest{}));
1165 };

1167 // Metaprogramming helper to construct a ConditionAnd expression.
1168 // "ConditionAndBuilder<Exp1, Exp2>::seq" is an "std::integer_sequence"
1169 // representing the expression.
1170 template <typename Exp1, typename Exp2> struct ConditionAndBuilder {
1171  template <int I, int... I1s, int... I2s>
1172  static auto append(std::integer_sequence<int, I1s...>,
1173  std::integer_sequence<int, I2s...>) {
1174  return std::integer_sequence<int, I, I1s..., I2s...>{};
1175  }
1176  using rest1 = typename Exp1::seq;
1177  using rest2 = typename Exp2::seq;
1178  using seq = decltype(append<ConditionAnd>(rest1{}, rest2{}));
1179 };

1181 // Metaprogramming helper to construct a ConditionOr expression.
1182 // "ConditionOrBuilder<Exp1, Exp2>::seq" is an "std::integer_sequence"
1183 // representing the expression.
1184 template <typename Exp1, typename Exp2> struct ConditionOrBuilder {
1185  template <int I, int... I1s, int... I2s>
1186  static auto append(std::integer_sequence<int, I1s...>,
1187  std::integer_sequence<int, I2s...>) {
1188  return std::integer_sequence<int, I, I1s..., I2s...>{};
1189  }
1190  using rest1 = typename Exp1::seq;
1191  using rest2 = typename Exp2::seq;
1192  using seq = decltype(append<ConditionOr>(rest1{}, rest2{}));
1193 };
