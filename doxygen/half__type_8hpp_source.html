90 // Several aliases are defined below:
91 // - StorageT: actual representation of half data type. It is used by scalar
92 // half values and by 'cl::sycl::vec' class. On device side, it points to some
93 // native half data type, while on host some custom data type is used to
94 // emulate operations of 16-bit floating-point values
95 //
96 // - BIsRepresentationT: data type which is used by built-in functions. It is
97 // distinguished from StorageT, because on host, we can still operate on the
98 // wrapper itself and there is no sense in direct usage of underlying data
99 // type (too many changes required for BIs implementation without any
100 // foreseeable profits)
101 //
102 // - VecNStorageT - representation of N-element vector of halfs. Follows the
103 // same logic as StorageT
104 #ifdef __SYCL_DEVICE_ONLY__
105  using StorageT = _Float16;
106  using BIsRepresentationT = _Float16;
107 
108  using Vec2StorageT = StorageT __attribute__((ext_vector_type(2)));
109  using Vec3StorageT = StorageT __attribute__((ext_vector_type(3)));
110  using Vec4StorageT = StorageT __attribute__((ext_vector_type(4)));
111  using Vec8StorageT = StorageT __attribute__((ext_vector_type(8)));
112  using Vec16StorageT = StorageT __attribute__((ext_vector_type(16)));
113 #else
114  using StorageT = detail::host_half_impl::half;
115  // No need to extract underlying data type for built-in functions operating on
116  // host
117  using BIsRepresentationT = half;
118 
119  // On the host side we cannot use OpenCL cl_half# types as an underlying type
120  // for vec because they are actually defined as an integer type under the
121  // hood. As a result half values will be converted to the integer and passed
122  // as a kernel argument which is expected to be floating point number.
123  template <int NumElements> struct half_vec {
124  alignas(detail::vector_alignment<StorageT, NumElements>::value)
125  std::array<StorageT, NumElements> s;
126  };
127 
128  using Vec2StorageT = half_vec<2>;
129  using Vec3StorageT = half_vec<3>;
130  using Vec4StorageT = half_vec<4>;
131  using Vec8StorageT = half_vec<8>;
132  using Vec16StorageT = half_vec<16>;
133 #endif

205 // Accroding to C++ standard math functions from cmath/math.h should work only
206 // on arithmetic types. We can't specify half type as arithmetic/floating
207 // point(via std::is_floating_point) since only float, double and long double
208 // types are "floating point" according to the standard. In order to use half
209 // type with these math functions we cast half to float using template
210 // function helper.
211 template <typename T> inline T cast_if_host_half(T val) { return val; }
