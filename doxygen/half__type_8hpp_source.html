100 // Several aliases are defined below:
101 // - StorageT: actual representation of half data type. It is used by scalar
102 // half values and by 'cl::sycl::vec' class. On device side, it points to some
103 // native half data type, while on host some custom data type is used to
104 // emulate operations of 16-bit floating-point values
105 //
106 // - BIsRepresentationT: data type which is used by built-in functions. It is
107 // distinguished from StorageT, because on host, we can still operate on the
108 // wrapper itself and there is no sense in direct usage of underlying data
109 // type (too many changes required for BIs implementation without any
110 // foreseeable profits)
111 //
112 // - VecNStorageT - representation of N-element vector of halfs. Follows the
113 // same logic as StorageT
114 #ifdef __SYCL_DEVICE_ONLY__
115  using StorageT = _Float16;
116  using BIsRepresentationT = _Float16;
117 
118  using Vec2StorageT = StorageT __attribute__((ext_vector_type(2)));
119  using Vec3StorageT = StorageT __attribute__((ext_vector_type(3)));
120  using Vec4StorageT = StorageT __attribute__((ext_vector_type(4)));
121  using Vec8StorageT = StorageT __attribute__((ext_vector_type(8)));
122  using Vec16StorageT = StorageT __attribute__((ext_vector_type(16)));
123 #else
124  using StorageT = detail::host_half_impl::half;
125  // No need to extract underlying data type for built-in functions operating on
126  // host
127  using BIsRepresentationT = half;
128 
129  // On the host side we cannot use OpenCL cl_half# types as an underlying type
130  // for vec because they are actually defined as an integer type under the
131  // hood. As a result half values will be converted to the integer and passed
132  // as a kernel argument which is expected to be floating point number.
133  template <int NumElements> struct half_vec {
134  alignas(detail::vector_alignment<StorageT, NumElements>::value)
135  std::array<StorageT, NumElements> s;
136  };
137 
138  using Vec2StorageT = half_vec<2>;
139  using Vec3StorageT = half_vec<3>;
140  using Vec4StorageT = half_vec<4>;
141  using Vec8StorageT = half_vec<8>;
142  using Vec16StorageT = half_vec<16>;
143 #endif

222 // Accroding to C++ standard math functions from cmath/math.h should work only
223 // on arithmetic types. We can't specify half type as arithmetic/floating
224 // point(via std::is_floating_point) since only float, double and long double
225 // types are "floating point" according to the standard. In order to use half
226 // type with these math functions we cast half to float using template
227 // function helper.
