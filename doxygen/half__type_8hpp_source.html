105 // Several aliases are defined below:
106 // - StorageT: actual representation of half data type. It is used by scalar
107 // half values and by 'cl::sycl::vec' class. On device side, it points to some
108 // native half data type, while on host some custom data type is used to
109 // emulate operations of 16-bit floating-point values
110 //
111 // - BIsRepresentationT: data type which is used by built-in functions. It is
112 // distinguished from StorageT, because on host, we can still operate on the
113 // wrapper itself and there is no sense in direct usage of underlying data
114 // type (too many changes required for BIs implementation without any
115 // foreseeable profits)
116 //
117 // - VecNStorageT - representation of N-element vector of halfs. Follows the
118 // same logic as StorageT
119 #ifdef __SYCL_DEVICE_ONLY__
120  using StorageT = _Float16;
121  using BIsRepresentationT = _Float16;
122 
123  using Vec2StorageT = StorageT __attribute__((ext_vector_type(2)));
124  using Vec3StorageT = StorageT __attribute__((ext_vector_type(3)));
125  using Vec4StorageT = StorageT __attribute__((ext_vector_type(4)));
126  using Vec8StorageT = StorageT __attribute__((ext_vector_type(8)));
127  using Vec16StorageT = StorageT __attribute__((ext_vector_type(16)));
128 #else
129  using StorageT = detail::host_half_impl::half;
130  // No need to extract underlying data type for built-in functions operating on
131  // host
132  using BIsRepresentationT = half;
133 
134  // On the host side we cannot use OpenCL cl_half# types as an underlying type
135  // for vec because they are actually defined as an integer type under the
136  // hood. As a result half values will be converted to the integer and passed
137  // as a kernel argument which is expected to be floating point number.
138  template <int NumElements> struct half_vec {
139  alignas(detail::vector_alignment<StorageT, NumElements>::value)
140  std::array<StorageT, NumElements> s;
141  };
142 
143  using Vec2StorageT = half_vec<2>;
144  using Vec3StorageT = half_vec<3>;
145  using Vec4StorageT = half_vec<4>;
146  using Vec8StorageT = half_vec<8>;
147  using Vec16StorageT = half_vec<16>;
148 #endif

227 // Accroding to C++ standard math functions from cmath/math.h should work only
228 // on arithmetic types. We can't specify half type as arithmetic/floating
229 // point(via std::is_floating_point) since only float, double and long double
230 // types are "floating point" according to the standard. In order to use half
231 // type with these math functions we cast half to float using template
232 // function helper.
