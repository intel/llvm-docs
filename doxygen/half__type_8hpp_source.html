91 // Several aliases are defined below:
92 // - StorageT: actual representation of half data type. It is used by scalar
93 // half values and by 'cl::sycl::vec' class. On device side, it points to some
94 // native half data type, while on host some custom data type is used to
95 // emulate operations of 16-bit floating-point values
96 //
97 // - BIsRepresentationT: data type which is used by built-in functions. It is
98 // distinguished from StorageT, because on host, we can still operate on the
99 // wrapper itself and there is no sense in direct usage of underlying data
100 // type (too many changes required for BIs implementation without any
101 // foreseeable profits)
102 //
103 // - VecNStorageT - representation of N-element vector of halfs. Follows the
104 // same logic as StorageT
105 #ifdef __SYCL_DEVICE_ONLY__
106  using StorageT = _Float16;
107  using BIsRepresentationT = _Float16;
108 
109  using Vec2StorageT = StorageT __attribute__((ext_vector_type(2)));
110  using Vec3StorageT = StorageT __attribute__((ext_vector_type(3)));
111  using Vec4StorageT = StorageT __attribute__((ext_vector_type(4)));
112  using Vec8StorageT = StorageT __attribute__((ext_vector_type(8)));
113  using Vec16StorageT = StorageT __attribute__((ext_vector_type(16)));
114 #else
115  using StorageT = detail::host_half_impl::half;
116  // No need to extract underlying data type for built-in functions operating on
117  // host
118  using BIsRepresentationT = half;
119 
120  // On the host side we cannot use OpenCL cl_half# types as an underlying type
121  // for vec because they are actually defined as an integer type under the
122  // hood. As a result half values will be converted to the integer and passed
123  // as a kernel argument which is expected to be floating point number.
124  template <int NumElements> struct half_vec {
125  alignas(detail::vector_alignment<StorageT, NumElements>::value)
126  std::array<StorageT, NumElements> s;
127  };
128 
129  using Vec2StorageT = half_vec<2>;
130  using Vec3StorageT = half_vec<3>;
131  using Vec4StorageT = half_vec<4>;
132  using Vec8StorageT = half_vec<8>;
133  using Vec16StorageT = half_vec<16>;
134 #endif

206 // Accroding to C++ standard math functions from cmath/math.h should work only
207 // on arithmetic types. We can't specify half type as arithmetic/floating
208 // point(via std::is_floating_point) since only float, double and long double
209 // types are "floating point" according to the standard. In order to use half
210 // type with these math functions we cast half to float using template
211 // function helper.
212 template <typename T> inline T cast_if_host_half(T val) { return val; }
213
