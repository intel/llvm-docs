221 template <int N> struct helper {
222  template <typename Res, typename Op, typename T1>
223  inline void run_1v(Res &r, Op op, T1 x) {
224  helper<N - 1>().run_1v(r, op, x);
225  r.template swizzle<N>() = op(x.template swizzle<N>());
226  }
