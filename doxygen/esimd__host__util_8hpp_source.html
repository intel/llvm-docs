41 template <typename RT> struct satur {
42  template <typename T> static RT saturate(const T val, const int flags) {
43  if ((flags & sat_is_on) == 0) {
44  return (RT)val;
45  }
46 
47  // min/max can be macros on Windows, so wrap them into parens to avoid their
48  // expansion
49  const RT t_max = (std::numeric_limits<RT>::max)();
50  const RT t_min = (std::numeric_limits<RT>::min)();
51 
52  if (val > t_max) {
53  return t_max;
54  } else if ((val >= 0) && (t_min < 0)) {
55  // RT is "signed" if t_min < 0
56  // when comparing a signed and a unsigned variable, the signed one cast to
57  // unsigned first.
58  return (RT)val;
59  } else if (val < t_min) {
60  return t_min;
61  } else {
62  return (RT)val;
63  }
64  }
65 };

69  if ((flags & sat_is_on) == 0) {
70  return (float)val;
