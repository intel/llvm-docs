41 template <typename RT> struct satur {
42  template <typename T> static RT saturate(const T val, const int flags) {
43  if ((flags & sat_is_on) == 0) {
44  return (RT)val;
45  }
46 
47 #ifdef max
48 #undef max
49 #endif
50 #ifdef min
51 #undef min
52 #endif
53  const RT t_max = std::numeric_limits<RT>::max();
54  const RT t_min = std::numeric_limits<RT>::min();
55 
56  if (val > t_max) {
57  return t_max;
58  } else if ((val >= 0) && (t_min < 0)) {
59  // RT is "signed" if t_min < 0
60  // when comparing a signed and a unsigned variable, the signed one cast to
61  // unsigned first.
62  return (RT)val;
63  } else if (val < t_min) {
64  return t_min;
65  } else {
66  return (RT)val;
67  }
68  }
69 };
