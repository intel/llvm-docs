234 // float to int
235 template <typename T, typename R, rounding_mode roundingMode>
236 detail::enable_if_t<is_float_to_int<T, R>::value, R> convertImpl(T Value) {
237 #ifndef __SYCL_DEVICE_ONLY__
238  switch (roundingMode) {
239  // Round to nearest even is default rounding mode for floating-point types
240  case rounding_mode::automatic:
241  // Round to nearest even.
242  case rounding_mode::rte: {
243  int OldRoundingDirection = std::fegetround();
244  int Err = std::fesetround(FE_TONEAREST);
245  if (Err)
246  throw runtime_error("Unable to set rounding mode to FE_TONEAREST",
247  PI_ERROR_UNKNOWN);
248  R Result = std::rint(Value);
249  Err = std::fesetround(OldRoundingDirection);
250  if (Err)
251  throw runtime_error("Unable to restore rounding mode.", PI_ERROR_UNKNOWN);
252  return Result;
253  }
254  // Round toward zero.
255  case rounding_mode::rtz:
256  return std::trunc(Value);
257  // Round toward positive infinity.
258  case rounding_mode::rtp:
259  return std::ceil(Value);
260  // Round toward negative infinity.
261  case rounding_mode::rtn:
262  return std::floor(Value);
263  default:
264  assert(!"Unsupported rounding mode!");
265  return static_cast<R>(Value);
266  };
267 #else
268  // TODO implement device side conversion.
269  return static_cast<R>(Value);
270 #endif
271 }
