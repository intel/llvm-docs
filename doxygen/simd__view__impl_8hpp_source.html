195 #define DEF_BINOP(BINOP, OPASSIGN) \
196  ESIMD_INLINE friend auto operator BINOP(const simd_view_impl &X, \
197  const value_type &Y) { \
198  using ComputeTy = detail::compute_type_t<value_type>; \
199  auto V0 = \
200  detail::convert<typename ComputeTy::vector_type>(X.read().data()); \
201  auto V1 = detail::convert<typename ComputeTy::vector_type>(Y.data()); \
202  auto V2 = V0 BINOP V1; \
203  return ComputeTy(V2); \
204  } \
205  ESIMD_INLINE friend auto operator BINOP(const value_type &X, \
206  const simd_view_impl &Y) { \
207  using ComputeTy = detail::compute_type_t<value_type>; \
208  auto V0 = detail::convert<typename ComputeTy::vector_type>(X.data()); \
209  auto V1 = \
210  detail::convert<typename ComputeTy::vector_type>(Y.read().data()); \
211  auto V2 = V0 BINOP V1; \
212  return ComputeTy(V2); \
213  } \
214  ESIMD_INLINE friend auto operator BINOP(const simd_view_impl &X, \

218  simd_view_impl &operator OPASSIGN(const value_type &RHS) { \
219  using ComputeTy = detail::compute_type_t<value_type>; \
220  auto V0 = detail::convert<typename ComputeTy::vector_type>(read().data()); \
221  auto V1 = detail::convert<typename ComputeTy::vector_type>(RHS.data()); \
222  auto V2 = V0 BINOP V1; \

224  write(V3); \
225  return *this; \
226  } \

228  return (*this OPASSIGN RHS.read()); \
229  }
