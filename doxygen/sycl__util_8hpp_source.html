53 template <sycl::access::mode Mode, accessor_mode_cap_val_t Cap>
54 constexpr bool accessor_mode_has_capability() {
55  static_assert(Cap == accessor_mode_cap::can_read ||
56  Cap == accessor_mode_cap::can_write,
57  "unsupported capability");
58 
59  if constexpr (Mode == sycl::access::mode::atomic ||
60  Mode == sycl::access::mode::read_write ||
61  Mode == sycl::access::mode::discard_read_write)
62  return true; // atomic and *read_write accessors can read/write
63 
64  return (Cap == accessor_mode_cap::can_read) ==
65  (Mode == sycl::access::mode::read);
66 }
