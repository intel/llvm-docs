23 static void waitForEvents(const std::vector<EventImplPtr> &Events) {
24  // Assuming all events will be on the same device or
25  // devices associated with the same Backend.
26  if (!Events.empty()) {
27  const detail::plugin &Plugin = Events[0]->getPlugin();
28  std::vector<RT::PiEvent> PiEvents(Events.size());
29  std::transform(Events.begin(), Events.end(), PiEvents.begin(),
30  [](const EventImplPtr &EventImpl) {
31  return EventImpl->getHandleRef();
32  });
33  Plugin.call<PiApiKind::piEventsWait>(PiEvents.size(), &PiEvents[0]);
34  }
35 }

84 // Creates an image1d buffer wrapper object around given memory object.
85 void *MemoryManager::wrapIntoImageBuffer(ContextImplPtr TargetContext,
86  void *MemBuf, SYCLMemObjI *MemObj) {
87  // Image format: 1 channel per pixel, each pixel 8 bit, Size pixels occupies
88  // Size bytes.
89  pi_image_format Format = {PI_IMAGE_CHANNEL_ORDER_R,
90  PI_IMAGE_CHANNEL_TYPE_UNSIGNED_INT8};
91 
92  // Image descriptor - request wrapper image1d creation.
93  pi_image_desc Desc = {};
94  Desc.image_type = PI_MEM_TYPE_IMAGE1D_BUFFER;
95  Desc.image_width = MemObj->getSize();
96  Desc.buffer = reinterpret_cast<pi_mem>(MemBuf);
97 
98  // Create the image object.
99  const detail::plugin &Plugin = TargetContext->getPlugin();
100  pi_mem Res = nullptr;
101  pi_mem_flags Flags = 0;
102  // Do not ref count the context handle, as it is not captured by the call.
103  Plugin.call<PiApiKind::piMemImageCreate>(TargetContext->getHandleRef(), Flags,
104  &Format, &Desc, nullptr, &Res);
105  return Res;
106 }

108 void *MemoryManager::allocateHostMemory(SYCLMemObjI *MemObj, void *UserPtr,
109  bool HostPtrReadOnly, size_t Size,
110  const sycl::property_list &) {
111  // Can return user pointer directly if it points to writable memory.
112  if (UserPtr && HostPtrReadOnly == false)
113  return UserPtr;
114 
115  void *NewMem = MemObj->allocateHostMem();
116 
117  // Need to initialize new memory if user provides pointer to read only
118  // memory.
119  if (UserPtr && HostPtrReadOnly == true)
120  std::memcpy((char *)NewMem, (char *)UserPtr, Size);
121  return NewMem;

217 void *MemoryManager::allocateMemSubBuffer(ContextImplPtr TargetContext,
218  void *ParentMemObj, size_t ElemSize,
219  size_t Offset, range<3> Range,
220  std::vector<EventImplPtr> DepEvents,
221  RT::PiEvent &OutEvent) {
222  waitForEvents(DepEvents);
223  OutEvent = nullptr;
224 
225  if (TargetContext->is_host())
226  return static_cast<void *>(static_cast<char *>(ParentMemObj) + Offset);
227 
228  size_t SizeInBytes = ElemSize;
229  for (size_t I = 0; I < 3; ++I)
230  SizeInBytes *= Range[I];
231 
232  RT::PiResult Error = PI_SUCCESS;
233  pi_buffer_region_struct Region{Offset, SizeInBytes};
234  RT::PiMem NewMem;
235  const detail::plugin &Plugin = TargetContext->getPlugin();
236  Error = Plugin.call_nocheck<PiApiKind::piMemBufferPartition>(
237  pi::cast<RT::PiMem>(ParentMemObj), PI_MEM_FLAGS_ACCESS_RW,
238  PI_BUFFER_CREATE_TYPE_REGION, &Region, &NewMem);
239  if (Error == PI_MISALIGNED_SUB_BUFFER_OFFSET)
240  throw invalid_object_error(
241  "Specified offset of the sub-buffer being constructed is not a "
242  "multiple of the memory base address alignment",
243  PI_INVALID_VALUE);
244  return NewMem;
245 }

247 void copyH2D(SYCLMemObjI *SYCLMemObj, char *SrcMem, QueueImplPtr,
248  unsigned int DimSrc, sycl::range<3> SrcSize,
249  sycl::range<3> SrcAccessRange, sycl::id<3> SrcOffset,
250  unsigned int SrcElemSize, RT::PiMem DstMem, QueueImplPtr TgtQueue,
251  unsigned int DimDst, sycl::range<3> DstSize,
252  sycl::range<3> DstAccessRange, sycl::id<3> DstOffset,
253  unsigned int DstElemSize, std::vector<RT::PiEvent> DepEvents,
254  RT::PiEvent &OutEvent) {
255  assert(SYCLMemObj && "The SYCLMemObj is nullptr");
256 
257  const RT::PiQueue Queue = TgtQueue->getHandleRef();
258  // Adjust first dimension of copy range and offset as OpenCL expects size in
259  // bytes.
260  DstSize[0] *= DstElemSize;
261  const detail::plugin &Plugin = TgtQueue->getPlugin();
262  if (SYCLMemObj->getType() == detail::SYCLMemObjI::MemObjType::BUFFER) {
263  DstOffset[0] *= DstElemSize;
264  SrcOffset[0] *= SrcElemSize;
265  SrcAccessRange[0] *= SrcElemSize;
266  DstAccessRange[0] *= DstElemSize;
267  SrcSize[0] *= SrcElemSize;
268 
269  if (1 == DimDst && 1 == DimSrc) {
270  Plugin.call<PiApiKind::piEnqueueMemBufferWrite>(
271  Queue, DstMem,
272  /*blocking_write=*/CL_FALSE, DstOffset[0], DstAccessRange[0],
273  SrcMem + SrcOffset[0], DepEvents.size(), &DepEvents[0], &OutEvent);
274  } else {
275  size_t BufferRowPitch = (1 == DimDst) ? 0 : DstSize[0];
276  size_t BufferSlicePitch = (3 == DimDst) ? DstSize[0] * DstSize[1] : 0;
277  size_t HostRowPitch = (1 == DimSrc) ? 0 : SrcSize[0];
278  size_t HostSlicePitch = (3 == DimSrc) ? SrcSize[0] * SrcSize[1] : 0;
279  Plugin.call<PiApiKind::piEnqueueMemBufferWriteRect>(
280  Queue, DstMem,
281  /*blocking_write=*/CL_FALSE, &DstOffset[0], &SrcOffset[0],
282  &DstAccessRange[0], BufferRowPitch, BufferSlicePitch, HostRowPitch,
283  HostSlicePitch, SrcMem, DepEvents.size(), &DepEvents[0], &OutEvent);
284  }
285  } else {
286  size_t InputRowPitch = (1 == DimDst) ? 0 : DstSize[0];
287  size_t InputSlicePitch = (3 == DimDst) ? DstSize[0] * DstSize[1] : 0;
288  Plugin.call<PiApiKind::piEnqueueMemImageWrite>(
289  Queue, DstMem,
290  /*blocking_write=*/CL_FALSE, &DstOffset[0], &DstAccessRange[0],
291  InputRowPitch, InputSlicePitch, SrcMem, DepEvents.size(), &DepEvents[0],
292  &OutEvent);
293  }
294 }

386 static void copyH2H(SYCLMemObjI *, char *SrcMem, QueueImplPtr,
387  unsigned int DimSrc, sycl::range<3> SrcSize,
388  sycl::range<3> SrcAccessRange, sycl::id<3> SrcOffset,
389  unsigned int SrcElemSize, char *DstMem, QueueImplPtr,
390  unsigned int DimDst, sycl::range<3> DstSize,
391  sycl::range<3> DstAccessRange, sycl::id<3> DstOffset,
392  unsigned int DstElemSize, std::vector<RT::PiEvent>,
393  RT::PiEvent &) {
394  if ((DimSrc != 1 || DimDst != 1) &&
395  (SrcOffset != id<3>{0, 0, 0} || DstOffset != id<3>{0, 0, 0} ||
396  SrcSize != SrcAccessRange || DstSize != DstAccessRange)) {
397  throw runtime_error("Not supported configuration of memcpy requested",

401  SrcMem += SrcOffset[0] * SrcElemSize;
402  DstMem += DstOffset[0] * DstElemSize;
403 
404  if (SrcMem == DstMem)
405  return;
406 
407  size_t BytesToCopy =
408  SrcAccessRange[0] * SrcElemSize * SrcAccessRange[1] * SrcAccessRange[2];
409  std::memcpy(DstMem, SrcMem, BytesToCopy);
410 }

480 void *MemoryManager::map(SYCLMemObjI *, void *Mem, QueueImplPtr Queue,
481  access::mode AccessMode, unsigned int, sycl::range<3>,
482  sycl::range<3> AccessRange, sycl::id<3> AccessOffset,
483  unsigned int ElementSize,
484  std::vector<RT::PiEvent> DepEvents,
485  RT::PiEvent &OutEvent) {
486  if (Queue->is_host()) {
487  throw runtime_error("Not supported configuration of map requested",
488  PI_INVALID_OPERATION);
489  }
490 
491  cl_map_flags Flags = 0;
492 
493  switch (AccessMode) {
494  case access::mode::read:
495  Flags |= CL_MAP_READ;
496  break;
497  case access::mode::write:
498  Flags |= CL_MAP_WRITE;
499  break;
500  case access::mode::read_write:
501  case access::mode::atomic:
502  Flags = CL_MAP_WRITE | CL_MAP_READ;
503  break;
504  case access::mode::discard_write:
505  case access::mode::discard_read_write:
506  Flags |= CL_MAP_WRITE_INVALIDATE_REGION;
507  break;
508  }
509 
510  AccessOffset[0] *= ElementSize;
511  AccessRange[0] *= ElementSize;
512 
513  // TODO: Handle offset
514  assert(AccessOffset[0] == 0 && "Handle offset");
515 
516  void *MappedPtr = nullptr;
517  const size_t BytesToMap = AccessRange[0] * AccessRange[1] * AccessRange[2];
518  const detail::plugin &Plugin = Queue->getPlugin();
519  Plugin.call<PiApiKind::piEnqueueMemBufferMap>(
