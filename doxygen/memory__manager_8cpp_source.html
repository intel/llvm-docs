120 static void waitForEvents(const std::vector<EventImplPtr> &Events) {
121  // Assuming all events will be on the same device or
122  // devices associated with the same Backend.
123  if (!Events.empty()) {
124  const PluginPtr &Plugin = Events[0]->getPlugin();
125  std::vector<sycl::detail::pi::PiEvent> PiEvents(Events.size());
126  std::transform(Events.begin(), Events.end(), PiEvents.begin(),
127  [](const EventImplPtr &EventImpl) {
128  return EventImpl->getHandleRef();
129  });
130  Plugin->call<PiApiKind::piEventsWait>(PiEvents.size(), &PiEvents[0]);
131  }

134 void memBufferCreateHelper(const PluginPtr &Plugin, pi_context Ctx,
135  pi_mem_flags Flags, size_t Size, void *HostPtr,
136  pi_mem *RetMem, const pi_mem_properties *Props) {
137 #ifdef XPTI_ENABLE_INSTRUMENTATION
138  uint64_t CorrID = 0;
139 #endif
140  // We only want to instrument piMemBufferCreate
141  {
142 #ifdef XPTI_ENABLE_INSTRUMENTATION
143  CorrID =
144  emitMemAllocBeginTrace(0 /* mem object */, Size, 0 /* guard zone */);
145  xpti::utils::finally _{[&] {
146  // C-style cast is required for MSVC
147  uintptr_t MemObjID = (uintptr_t)(*RetMem);
148  pi_native_handle Ptr = 0;
149  // Always use call_nocheck here, because call may throw an exception,
150  // and this lambda will be called from destructor, which in combination
151  // rewards us with UB.
152  // When doing buffer interop we don't know what device the memory should
153  // be resident on, so pass nullptr for Device param. Buffer interop may
154  // not be supported by all backends.
155  Plugin->call_nocheck<PiApiKind::piextMemGetNativeHandle>(
156  *RetMem, /*Dev*/ nullptr, &Ptr);
157  emitMemAllocEndTrace(MemObjID, (uintptr_t)(Ptr), Size, 0 /* guard zone */,
158  CorrID);
159  }};
160 #endif
161  if (Size)
162  Plugin->call<PiApiKind::piMemBufferCreate>(Ctx, Flags, Size, HostPtr,
163  RetMem, Props);
164  }

167 void memReleaseHelper(const PluginPtr &Plugin, pi_mem Mem) {
168  // FIXME piMemRelease does not guarante memory release. It is only true if
169  // reference counter is 1. However, SYCL runtime currently only calls
170  // piMemRetain only for OpenCL interop
171 #ifdef XPTI_ENABLE_INSTRUMENTATION
172  uint64_t CorrID = 0;
173  // C-style cast is required for MSVC
174  uintptr_t MemObjID = (uintptr_t)(Mem);
175  uintptr_t Ptr = 0;
176  // Do not make unnecessary PI calls without instrumentation enabled
177  if (xptiTraceEnabled()) {
178  pi_native_handle PtrHandle = 0;
179  // When doing buffer interop we don't know what device the memory should be
180  // resident on, so pass nullptr for Device param. Buffer interop may not be
181  // supported by all backends.
182  Plugin->call<PiApiKind::piextMemGetNativeHandle>(Mem, /*Dev*/ nullptr,
183  &PtrHandle);
184  Ptr = (uintptr_t)(PtrHandle);
185  }
186 #endif
187  // We only want to instrument piMemRelease
188  {
189 #ifdef XPTI_ENABLE_INSTRUMENTATION
190  CorrID = emitMemReleaseBeginTrace(MemObjID, Ptr);
191  xpti::utils::finally _{
192  [&] { emitMemReleaseEndTrace(MemObjID, Ptr, CorrID); }};
193 #endif
194  Plugin->call<PiApiKind::piMemRelease>(Mem);
195  }
196 }

351  bool HostPtrReadOnly, const size_t Size,

353  sycl::detail::pi::PiMemFlags CreationFlags =
354  getMemObjCreationFlags(UserPtr, HostPtrReadOnly);

357  CreationFlags |= PI_MEM_FLAGS_HOST_PTR_ALLOC;

359  sycl::detail::pi::PiMem NewMem = nullptr;
360  const PluginPtr &Plugin = TargetContext->getPlugin();
361 
362  std::vector<pi_mem_properties> AllocProps;
363 
364  if (PropsList.has_property<property::buffer::detail::buffer_location>() &&
365  TargetContext->isBufferLocationSupported()) {
366  auto Location =
367  PropsList.get_property<property::buffer::detail::buffer_location>()
368  .get_buffer_location();
369  AllocProps.reserve(AllocProps.size() + 2);
370  AllocProps.push_back(PI_MEM_PROPERTIES_ALLOC_BUFFER_LOCATION);
371  AllocProps.push_back(Location);
372  }
373 
374  if (PropsList.has_property<property::buffer::mem_channel>()) {
375  auto Channel =
376  PropsList.get_property<property::buffer::mem_channel>().get_channel();
377  AllocProps.reserve(AllocProps.size() + 2);
378  AllocProps.push_back(PI_MEM_PROPERTIES_CHANNEL);
379  AllocProps.push_back(Channel);
380  }
381 
382  pi_mem_properties *AllocPropsPtr = nullptr;
383  if (!AllocProps.empty()) {
384  // If there are allocation properties, push an end to the list and update
385  // the properties pointer.
386  AllocProps.push_back(0);
387  AllocPropsPtr = AllocProps.data();
388  }
389 
390  memBufferCreateHelper(Plugin, TargetContext->getHandleRef(), CreationFlags,
391  Size, UserPtr, &NewMem, AllocPropsPtr);
392  return NewMem;
393 }

473 void prepTermPositions(TermPositions &pos, int Dimensions,
474  detail::SYCLMemObjI::MemObjType Type) {
475  // For buffers, the offsets/ranges coming from accessor are always
476  // id<3>/range<3> But their organization varies by dimension:
477  // 1 ==> {width, 1, 1}
478  // 2 ==> {height, width, 1}
479  // 3 ==> {depth, height, width}
480  // Some callers schedule 0 as DimDst/DimSrc.
481 
482  if (Type == detail::SYCLMemObjI::MemObjType::Buffer) {
483  if (Dimensions == 3) {
484  pos.XTerm = 2, pos.YTerm = 1, pos.ZTerm = 0;
485  } else if (Dimensions == 2) {
486  pos.XTerm = 1, pos.YTerm = 0, pos.ZTerm = 2;
487  } else { // Dimension is 1 or 0
488  pos.XTerm = 0, pos.YTerm = 1, pos.ZTerm = 2;
489  }
490  } else { // While range<>/id<> use by images is different than buffers, it's
491  // consistent with their accessors.
492  pos.XTerm = 0;
493  pos.YTerm = 1;
494  pos.ZTerm = 2;
495  }
496 }

498 void copyH2D(SYCLMemObjI *SYCLMemObj, char *SrcMem, QueueImplPtr,
499  unsigned int DimSrc, sycl::range<3> SrcSize,
500  sycl::range<3> SrcAccessRange, sycl::id<3> SrcOffset,
501  unsigned int SrcElemSize, sycl::detail::pi::PiMem DstMem,
502  QueueImplPtr TgtQueue, unsigned int DimDst, sycl::range<3> DstSize,
503  sycl::range<3> DstAccessRange, sycl::id<3> DstOffset,
504  unsigned int DstElemSize,
505  std::vector<sycl::detail::pi::PiEvent> DepEvents,
506  sycl::detail::pi::PiEvent &OutEvent,
507  const detail::EventImplPtr &OutEventImpl) {
508  (void)SrcAccessRange;
509  assert(SYCLMemObj && "The SYCLMemObj is nullptr");
510 
511  const sycl::detail::pi::PiQueue Queue = TgtQueue->getHandleRef();
512  const PluginPtr &Plugin = TgtQueue->getPlugin();
513 
514  detail::SYCLMemObjI::MemObjType MemType = SYCLMemObj->getType();
515  TermPositions SrcPos, DstPos;
516  prepTermPositions(SrcPos, DimSrc, MemType);
517  prepTermPositions(DstPos, DimDst, MemType);
518 
519  size_t DstXOffBytes = DstOffset[DstPos.XTerm] * DstElemSize;
520  size_t SrcXOffBytes = SrcOffset[SrcPos.XTerm] * SrcElemSize;
521  size_t DstAccessRangeWidthBytes = DstAccessRange[DstPos.XTerm] * DstElemSize;
522  size_t DstSzWidthBytes = DstSize[DstPos.XTerm] * DstElemSize;
523  size_t SrcSzWidthBytes = SrcSize[SrcPos.XTerm] * SrcElemSize;
524 
525  if (MemType == detail::SYCLMemObjI::MemObjType::Buffer) {
526  if (1 == DimDst && 1 == DimSrc) {
527  if (OutEventImpl != nullptr)
528  OutEventImpl->setHostEnqueueTime();
529  Plugin->call<PiApiKind::piEnqueueMemBufferWrite>(
530  Queue, DstMem,
531  /*blocking_write=*/PI_FALSE, DstXOffBytes, DstAccessRangeWidthBytes,
532  SrcMem + SrcXOffBytes, DepEvents.size(), DepEvents.data(), &OutEvent);
533  } else {
534  size_t BufferRowPitch = (1 == DimDst) ? 0 : DstSzWidthBytes;
535  size_t BufferSlicePitch =
536  (3 == DimDst) ? DstSzWidthBytes * DstSize[DstPos.YTerm] : 0;
537  size_t HostRowPitch = (1 == DimSrc) ? 0 : SrcSzWidthBytes;
538  size_t HostSlicePitch =
539  (3 == DimSrc) ? SrcSzWidthBytes * SrcSize[SrcPos.YTerm] : 0;
540 
541  pi_buff_rect_offset_struct BufferOffset{
542  DstXOffBytes, DstOffset[DstPos.YTerm], DstOffset[DstPos.ZTerm]};
543  pi_buff_rect_offset_struct HostOffset{
544  SrcXOffBytes, SrcOffset[SrcPos.YTerm], SrcOffset[SrcPos.ZTerm]};
545  pi_buff_rect_region_struct RectRegion{DstAccessRangeWidthBytes,
546  DstAccessRange[DstPos.YTerm],
547  DstAccessRange[DstPos.ZTerm]};
548  if (OutEventImpl != nullptr)
549  OutEventImpl->setHostEnqueueTime();
550  Plugin->call<PiApiKind::piEnqueueMemBufferWriteRect>(
551  Queue, DstMem,
552  /*blocking_write=*/PI_FALSE, &BufferOffset, &HostOffset, &RectRegion,
553  BufferRowPitch, BufferSlicePitch, HostRowPitch, HostSlicePitch,
554  SrcMem, DepEvents.size(), DepEvents.data(), &OutEvent);
555  }
556  } else {
557  size_t InputRowPitch = (1 == DimDst) ? 0 : DstSzWidthBytes;
558  size_t InputSlicePitch =
559  (3 == DimDst) ? DstSzWidthBytes * DstSize[DstPos.YTerm] : 0;
560 
561  pi_image_offset_struct Origin{DstOffset[DstPos.XTerm],
562  DstOffset[DstPos.YTerm],
563  DstOffset[DstPos.ZTerm]};

1086  const void *SrcMem, size_t SrcPitch, QueueImplPtr Queue, void *DstMem,
1087  size_t DstPitch, size_t Width, size_t Height,
1088  std::vector<sycl::detail::pi::PiEvent> DepEvents,
1089  sycl::detail::pi::PiEvent *OutEvent,
1090  const detail::EventImplPtr &OutEventImpl) {
1091  assert(!Queue->getContextImplPtr()->is_host() &&
1092  "Host queue not supported in copy_2d_usm.");
1093

1095  // no-op, but ensure DepEvents will still be waited on
1096  if (!DepEvents.empty()) {
1097  if (OutEventImpl != nullptr)
1098  OutEventImpl->setHostEnqueueTime();
1099  Queue->getPlugin()->call<PiApiKind::piEnqueueEventsWait>(
1100  Queue->getHandleRef(), DepEvents.size(), DepEvents.data(), OutEvent);
1101  }
1102  return;
1103  }
1104 
1105  if (!DstMem || !SrcMem)
1106  throw sycl::exception(sycl::make_error_code(errc::invalid),
1107  "NULL pointer argument in 2D memory copy operation.");
1108

1110 
1111  pi_bool SupportsUSMMemcpy2D = false;
1112  Plugin->call<detail::PiApiKind::piContextGetInfo>(
1113  Queue->getContextImplPtr()->getHandleRef(),
1114  PI_EXT_ONEAPI_CONTEXT_INFO_USM_MEMCPY2D_SUPPORT, sizeof(pi_bool),
1115  &SupportsUSMMemcpy2D, nullptr);
1116 
1117  if (SupportsUSMMemcpy2D) {
1118  if (OutEventImpl != nullptr)
1119  OutEventImpl->setHostEnqueueTime();
1120  // Direct memcpy2D is supported so we use this function.
1121  Plugin->call<PiApiKind::piextUSMEnqueueMemcpy2D>(
1122  Queue->getHandleRef(), /*blocking=*/PI_FALSE, DstMem, DstPitch, SrcMem,
1123  SrcPitch, Width, Height, DepEvents.size(), DepEvents.data(), OutEvent);
1124  return;
1125  }
1126 
1127  // Otherwise we allow the special case where the copy is to or from host.
1128 #ifndef NDEBUG
1129  context Ctx = createSyclObjFromImpl<context>(Queue->getContextImplPtr());
1130  usm::alloc SrcAllocType = get_pointer_type(SrcMem, Ctx);
1131  usm::alloc DstAllocType = get_pointer_type(DstMem, Ctx);
1132  bool SrcIsHost =
1133  SrcAllocType == usm::alloc::unknown || SrcAllocType == usm::alloc::host;
1134  bool DstIsHost =
1135  DstAllocType == usm::alloc::unknown || DstAllocType == usm::alloc::host;
1136  assert((SrcIsHost || DstIsHost) && "In fallback path for copy_2d_usm either "
1137  "source or destination must be on host.");
1138 #endif // NDEBUG
1139 
1140  // The fallback in this case is to insert a copy per row.
1141  std::vector<OwnedPiEvent> CopyEventsManaged;
1142  CopyEventsManaged.reserve(Height);
1143  // We'll need continuous range of events for a wait later as well.
1144  std::vector<sycl::detail::pi::PiEvent> CopyEvents(Height);
1145  if (OutEventImpl != nullptr)
1146  OutEventImpl->setHostEnqueueTime();
1147  for (size_t I = 0; I < Height; ++I) {
1148  char *DstItBegin = static_cast<char *>(DstMem) + I * DstPitch;
1149  const char *SrcItBegin = static_cast<const char *>(SrcMem) + I * SrcPitch;
1150  Plugin->call<PiApiKind::piextUSMEnqueueMemcpy>(
1151  Queue->getHandleRef(), /* blocking */ PI_FALSE, DstItBegin, SrcItBegin,
1152  Width, DepEvents.size(), DepEvents.data(), CopyEvents.data() + I);
1153  CopyEventsManaged.emplace_back(CopyEvents[I], Plugin,
1154  /*TakeOwnership=*/true);
1155  }
1156  if (OutEventImpl != nullptr)
1157  OutEventImpl->setHostEnqueueTime();
1158  // Then insert a wait to coalesce the copy events.
1159  Queue->getPlugin()->call<PiApiKind::piEnqueueEventsWait>(
1160  Queue->getHandleRef(), CopyEvents.size(), CopyEvents.data(), OutEvent);
1161 }
