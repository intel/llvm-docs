19 static uint16_t float2Half(const float &Val) {
20  const uint32_t Bits = *reinterpret_cast<const uint32_t *>(&Val);
21 
22  // Extract the sign from the float value
23  const uint16_t Sign = (Bits & 0x80000000) >> 16;
24  // Extract the fraction from the float value
25  const uint32_t Frac32 = Bits & 0x7fffff;
26  // Extract the exponent from the float value
27  const uint8_t Exp32 = (Bits & 0x7f800000) >> 23;
28  const int16_t Exp32Diff = Exp32 - 127;
29 
30  // intialize to 0, covers the case for 0 and small numbers
31  uint16_t Exp16 = 0, Frac16 = 0;
32 
33  if (__builtin_expect(Exp32Diff > 15, 0)) {
34  // Infinity and big numbers convert to infinity
35  Exp16 = 0x1f;
36  } else if (__builtin_expect(Exp32Diff > -14, 0)) {
37  // normal range for half type
38  Exp16 = Exp32Diff + 15;
39  // convert 23-bit mantissa to 10-bit mantissa.
40  Frac16 = Frac32 >> 13;
41  // Round the mantissa as given in OpenCL spec section : 6.1.1.1 The half
42  // data type.
43  if (Frac32 >> 12 & 0x01)
44  Frac16 += 1;
45  } else if (__builtin_expect(Exp32Diff > -24, 0)) {
46  // subnormals
47  Frac16 = (Frac32 | (uint32_t(1) << 23)) >> (-Exp32Diff - 1);
48  }
49 
50  if (__builtin_expect(Exp32 == 0xff && Frac32 != 0, 0)) {
51  // corner case: FP32 is NaN
52  Exp16 = 0x1F;
53  Frac16 = 0x200;
54  }
55 
56  // Compose the final FP16 binary
57  uint16_t Ret = 0;
58  Ret |= Sign;
59  Ret |= Exp16 << 10;
60  Ret += Frac16; // Add the carry bit from operation Frac16 += 1;
61 
62  return Ret;
63 }

65 static float half2Float(const uint16_t &Val) {
66  // Extract the sign from the bits
67  const uint32_t Sign = static_cast<uint32_t>(Val & 0x8000) << 16;
68  // Extract the exponent from the bits
69  const uint8_t Exp16 = (Val & 0x7c00) >> 10;
70  // Extract the fraction from the bits
71  uint16_t Frac16 = Val & 0x3ff;
72 
73  uint32_t Exp32 = 0;
74  if (__builtin_expect(Exp16 == 0x1f, 0)) {
75  Exp32 = 0xff;
76  } else if (__builtin_expect(Exp16 == 0, 0)) {
77  Exp32 = 0;
78  } else {
79  Exp32 = static_cast<uint32_t>(Exp16) + 112;
80  }
81 
82  // corner case: subnormal -> normal
83  // The denormal number of FP16 can be represented by FP32, therefore we need
84  // to recover the exponent and recalculate the fration.
85  if (__builtin_expect(Exp16 == 0 && Frac16 != 0, 0)) {
86  uint8_t OffSet = 0;
87  do {
88  ++OffSet;
89  Frac16 <<= 1;
90  } while ((Frac16 & 0x400) != 0x400);
91  // mask the 9th bit
92  Frac16 &= 0x3ff;
93  Exp32 = 113 - OffSet;
94  }
95 
96  uint32_t Frac32 = Frac16 << 13;
97 
98  // Compose the final FP32 binary
99  uint32_t Bits = 0;
100 
101  Bits |= Sign;
102  Bits |= (Exp32 << 23);
103  Bits |= Frac32;
104 
105  float Result;
106  std::memcpy(&Result, &Bits, sizeof(Result));
107  return Result;
108 }
