118 template <typename T>
119 typename sycl::detail::enable_if_t<d::is_sgenfloat<T>::value,
120  T> inline __bitselect(T a, T b, T c) {
121  databitset<T> ba;
122  ba.f = a;
123  databitset<T> bb;
124  bb.f = b;
125  databitset<T> bc;
126  bc.f = c;
127  databitset<T> br;
128  br.f = 0;
129  br.i = ((ba.i & ~bc.i) | (bb.i & bc.i));
130  return br.f;
131 }
