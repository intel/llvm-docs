508  WriteLockT acquireWriteLock() {
509 #ifdef _WIN32
510  WriteLockT Lock(MGraphLock, std::defer_lock);
511  while (!Lock.try_lock_for(std::chrono::milliseconds(10))) {
512  // Without yield while loop acts like endless while loop and occupies the
513  // whole CPU when multiple command groups are created in multiple host
514  // threads
515  std::this_thread::yield();
516  }
517 #else
518  WriteLockT Lock(MGraphLock);
519  // It is a deadlock on UNIX in implementation of lock and lock_shared, if
520  // try_lock in the loop above will be executed, so using a single lock here
521 #endif // _WIN32
522  return Lock;
523  }
524 
527  WriteLockT acquireFusionWriteLock() {
