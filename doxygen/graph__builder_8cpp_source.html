30 // The function checks whether two requirements overlaps or not. This
31 // information can be used to prove that executing two kernels that
32 // work on different parts of the memory object in parallel is legal.
33 static bool doOverlap(const Requirement *LHS, const Requirement *RHS) {
34  return (LHS->MOffsetInBytes + LHS->MAccessRange.size() * LHS->MElemSize >=
35  RHS->MOffsetInBytes) ||
36  (RHS->MOffsetInBytes + RHS->MAccessRange.size() * RHS->MElemSize >=
37  LHS->MOffsetInBytes);
38 }

40 static bool sameCtx(const ContextImplPtr &LHS, const ContextImplPtr &RHS) {
41  // Consider two different host contexts to be the same to avoid additional
42  // allocation on the host
43  return LHS == RHS || (LHS->is_host() && RHS->is_host());
44 }

51 // Checks if the required access mode is allowed under the current one
52 static bool isAccessModeAllowed(access::mode Required, access::mode Current) {
53  switch (Current) {
54  case access::mode::read:
55  return (Required == Current);
56  case access::mode::write:
57  assert(false && "Write only access is expected to be mapped as read_write");
58  return (Required == Current || Required == access::mode::discard_write);
59  case access::mode::read_write:
60  case access::mode::atomic:
61  case access::mode::discard_write:
62  case access::mode::discard_read_write:
63  return true;
64  }
65  assert(false);
66  return false;
67 }

150 // Helper function which removes all values in Cmds from Leaves
151 void Scheduler::GraphBuilder::updateLeaves(const std::set<Command *> &Cmds,
152  MemObjRecord *Record,
153  access::mode AccessMode) {
154 
155  const bool ReadOnlyReq = AccessMode == access::mode::read;
156  if (ReadOnlyReq)
157  return;
158 
159  for (Command *Cmd : Cmds) {
160  auto NewEnd = std::remove(Record->MReadLeaves.begin(),
161  Record->MReadLeaves.end(), Cmd);
162  Cmd->MLeafCounter -= std::distance(NewEnd, Record->MReadLeaves.end());
163  Record->MReadLeaves.erase(NewEnd, Record->MReadLeaves.end());
164 
165  NewEnd = std::remove(Record->MWriteLeaves.begin(),
166  Record->MWriteLeaves.end(), Cmd);
167  Cmd->MLeafCounter -= std::distance(NewEnd, Record->MWriteLeaves.end());
168  Record->MWriteLeaves.erase(NewEnd, Record->MWriteLeaves.end());
169  }
170 }

172 void Scheduler::GraphBuilder::addNodeToLeaves(MemObjRecord *Record,
173  Command *Cmd,
174  access::mode AccessMode) {
175  CircularBuffer<Command *> &Leaves{AccessMode == access::mode::read
176  ? Record->MReadLeaves
177  : Record->MWriteLeaves};
178  if (Leaves.full()) {
179  Command *OldLeaf = Leaves.front();
180  // TODO this is a workaround for duplicate leaves, remove once fixed
181  if (OldLeaf == Cmd)
182  return;
183  // Add the old leaf as a dependency for the new one by duplicating one of
184  // the requirements for the current record
185  DepDesc Dep = findDepForRecord(Cmd, Record);
186  Dep.MDepCommand = OldLeaf;
187  Cmd->addDep(Dep);
188  OldLeaf->addUser(Cmd);
189  --(OldLeaf->MLeafCounter);
190  }
191  Leaves.push_back(Cmd);
192  ++(Cmd->MLeafCounter);
193 }

195 UpdateHostRequirementCommand *Scheduler::GraphBuilder::insertUpdateHostReqCmd(
196  MemObjRecord *Record, Requirement *Req, const QueueImplPtr &Queue) {
197  AllocaCommandBase *AllocaCmd =
198  findAllocaForReq(Record, Req, Queue->getContextImplPtr());
199  assert(AllocaCmd && "There must be alloca for requirement!");
200  UpdateHostRequirementCommand *UpdateCommand =
201  new UpdateHostRequirementCommand(Queue, *Req, AllocaCmd, &Req->MData);
202  // Need copy of requirement because after host accessor destructor call
203  // dependencies become invalid if requirement is stored by pointer.
204  const Requirement *StoredReq = UpdateCommand->getRequirement();
205 
206  std::set<Command *> Deps =
207  findDepsForReq(Record, Req, Queue->getContextImplPtr());
208  for (Command *Dep : Deps) {
209  UpdateCommand->addDep(DepDesc{Dep, StoredReq, AllocaCmd});
210  Dep->addUser(UpdateCommand);
211  }
212  updateLeaves(Deps, Record, Req->MAccessMode);
213  addNodeToLeaves(Record, UpdateCommand, Req->MAccessMode);
214  return UpdateCommand;
215 }

217 // Takes linked alloca commands. Makes AllocaCmdDst command active using map
218 // or unmap operation.
219 static Command *insertMapUnmapForLinkedCmds(AllocaCommandBase *AllocaCmdSrc,
220  AllocaCommandBase *AllocaCmdDst,
221  access::mode MapMode) {
222  assert(AllocaCmdSrc->MLinkedAllocaCmd == AllocaCmdDst &&
223  "Expected linked alloca commands");
224  assert(AllocaCmdSrc->MIsActive &&
225  "Expected source alloca command to be active");
226 
227  if (AllocaCmdSrc->getQueue()->is_host()) {
228  UnMapMemObject *UnMapCmd = new UnMapMemObject(
229  AllocaCmdDst, *AllocaCmdDst->getRequirement(),
230  &AllocaCmdSrc->MMemAllocation, AllocaCmdDst->getQueue());
231 
232  std::swap(AllocaCmdSrc->MIsActive, AllocaCmdDst->MIsActive);
233 
234  return UnMapCmd;
235  }
236 
237  MapMemObject *MapCmd = new MapMemObject(
238  AllocaCmdSrc, *AllocaCmdSrc->getRequirement(),
239  &AllocaCmdDst->MMemAllocation, AllocaCmdSrc->getQueue(), MapMode);
240 
241  std::swap(AllocaCmdSrc->MIsActive, AllocaCmdDst->MIsActive);
242 
243  return MapCmd;
244 }

246 Command *Scheduler::GraphBuilder::insertMemoryMove(MemObjRecord *Record,
247  Requirement *Req,
248  const QueueImplPtr &Queue) {
249 
250  AllocaCommandBase *AllocaCmdDst = getOrCreateAllocaForReq(Record, Req, Queue);
251  if (!AllocaCmdDst)
252  throw runtime_error("Out of host memory", PI_OUT_OF_HOST_MEMORY);
253 
254  std::set<Command *> Deps =
255  findDepsForReq(Record, Req, Queue->getContextImplPtr());
256  Deps.insert(AllocaCmdDst);
257  // Get parent allocation of sub buffer to perform full copy of whole buffer
258  if (IsSuitableSubReq(Req)) {
259  if (AllocaCmdDst->getType() == Command::CommandType::ALLOCA_SUB_BUF)
260  AllocaCmdDst =
261  static_cast<AllocaSubBufCommand *>(AllocaCmdDst)->getParentAlloca();
262  }
263 
264  AllocaCommandBase *AllocaCmdSrc =
265  findAllocaForReq(Record, Req, Record->MCurContext);
266  if (!AllocaCmdSrc && IsSuitableSubReq(Req)) {
267  // Since no alloca command for the sub buffer requirement was found in the
268  // current context, need to find a parent alloca command for it (it must be
269  // there)
270  auto IsSuitableAlloca = [Record, Req](AllocaCommandBase *AllocaCmd) {
271  bool Res = sameCtx(AllocaCmd->getQueue()->getContextImplPtr(),
272  Record->MCurContext) &&
273  // Looking for a parent buffer alloca command
274  AllocaCmd->getType() == Command::CommandType::ALLOCA;
275  return Res;
276  };
277  const auto It =
278  std::find_if(Record->MAllocaCommands.begin(),
279  Record->MAllocaCommands.end(), IsSuitableAlloca);
280  AllocaCmdSrc = (Record->MAllocaCommands.end() != It) ? *It : nullptr;
281  }
282  if (!AllocaCmdSrc)
283  throw runtime_error("Cannot find buffer allocation", PI_INVALID_VALUE);
284  // Get parent allocation of sub buffer to perform full copy of whole buffer
285  if (IsSuitableSubReq(Req)) {
286  if (AllocaCmdSrc->getType() == Command::CommandType::ALLOCA_SUB_BUF)
287  AllocaCmdSrc =
288  static_cast<AllocaSubBufCommand *>(AllocaCmdSrc)->getParentAlloca();
289  else if (AllocaCmdSrc->getSYCLMemObj() != Req->MSYCLMemObj)
290  assert(!"Inappropriate alloca command.");
291  }
292 
293  Command *NewCmd = nullptr;
294 
295  if (AllocaCmdSrc->MLinkedAllocaCmd == AllocaCmdDst) {
296  // Map write only as read-write
297  access::mode MapMode = Req->MAccessMode;
298  if (MapMode == access::mode::write)
299  MapMode = access::mode::read_write;
300  NewCmd = insertMapUnmapForLinkedCmds(AllocaCmdSrc, AllocaCmdDst, MapMode);
301  Record->MHostAccess = MapMode;
302  } else {
303 
304  // Full copy of buffer is needed to avoid loss of data that may be caused
305  // by copying specific range from host to device and backwards.
306  NewCmd =
307  new MemCpyCommand(*AllocaCmdSrc->getRequirement(), AllocaCmdSrc,
308  *AllocaCmdDst->getRequirement(), AllocaCmdDst,
309  AllocaCmdSrc->getQueue(), AllocaCmdDst->getQueue());
310  }
311 
312  for (Command *Dep : Deps) {
313  NewCmd->addDep(DepDesc{Dep, NewCmd->getRequirement(), AllocaCmdDst});
314  Dep->addUser(NewCmd);
315  }
316  updateLeaves(Deps, Record, access::mode::read_write);
317  addNodeToLeaves(Record, NewCmd, access::mode::read_write);
318  Record->MCurContext = Queue->getContextImplPtr();
319  return NewCmd;
320 }

322 Command *Scheduler::GraphBuilder::remapMemoryObject(
323  MemObjRecord *Record, Requirement *Req, AllocaCommandBase *HostAllocaCmd) {
324  assert(HostAllocaCmd->getQueue()->is_host() &&
325  "Host alloca command expected");
326  assert(HostAllocaCmd->MIsActive && "Active alloca command expected");
327 
328  AllocaCommandBase *LinkedAllocaCmd = HostAllocaCmd->MLinkedAllocaCmd;
329  assert(LinkedAllocaCmd && "Linked alloca command expected");
330 
331  std::set<Command *> Deps = findDepsForReq(Record, Req, Record->MCurContext);
332 
333  UnMapMemObject *UnMapCmd = new UnMapMemObject(
334  LinkedAllocaCmd, *LinkedAllocaCmd->getRequirement(),
335  &HostAllocaCmd->MMemAllocation, LinkedAllocaCmd->getQueue());
336 
337  // Map write only as read-write
338  access::mode MapMode = Req->MAccessMode;
339  if (MapMode == access::mode::write)
340  MapMode = access::mode::read_write;
341  MapMemObject *MapCmd = new MapMemObject(
342  LinkedAllocaCmd, *LinkedAllocaCmd->getRequirement(),
343  &HostAllocaCmd->MMemAllocation, LinkedAllocaCmd->getQueue(), MapMode);
344 
345  for (Command *Dep : Deps) {
346  UnMapCmd->addDep(DepDesc{Dep, UnMapCmd->getRequirement(), LinkedAllocaCmd});
347  Dep->addUser(UnMapCmd);
348  }
349 
350  MapCmd->addDep(DepDesc{UnMapCmd, MapCmd->getRequirement(), HostAllocaCmd});
351  UnMapCmd->addUser(MapCmd);
352 
353  updateLeaves(Deps, Record, access::mode::read_write);
354  addNodeToLeaves(Record, MapCmd, access::mode::read_write);
355  Record->MHostAccess = MapMode;
356  return MapCmd;
357 }

359 // The function adds copy operation of the up to date'st memory to the memory
360 // pointed by Req.
361 Command *Scheduler::GraphBuilder::addCopyBack(Requirement *Req) {
362 
363  QueueImplPtr HostQueue = Scheduler::getInstance().getDefaultHostQueue();
364  SYCLMemObjI *MemObj = Req->MSYCLMemObj;
365  MemObjRecord *Record = getMemObjRecord(MemObj);
366  if (Record && MPrintOptionsArray[BeforeAddCopyBack])
367  printGraphAsDot("before_addCopyBack");
368 
369  // Do nothing if there were no or only read operations with the memory object.
370  if (nullptr == Record || !Record->MMemModified)
371  return nullptr;
372 
373  std::set<Command *> Deps =
374  findDepsForReq(Record, Req, HostQueue->getContextImplPtr());
375  AllocaCommandBase *SrcAllocaCmd =
376  findAllocaForReq(Record, Req, Record->MCurContext);
377 
378  std::unique_ptr<MemCpyCommandHost> MemCpyCmdUniquePtr(new MemCpyCommandHost(
379  *SrcAllocaCmd->getRequirement(), SrcAllocaCmd, *Req, &Req->MData,
380  SrcAllocaCmd->getQueue(), std::move(HostQueue)));
381 
382  if (!MemCpyCmdUniquePtr)
383  throw runtime_error("Out of host memory", PI_OUT_OF_HOST_MEMORY);
384 
385  MemCpyCommandHost *MemCpyCmd = MemCpyCmdUniquePtr.release();
386  for (Command *Dep : Deps) {
387  MemCpyCmd->addDep(DepDesc{Dep, MemCpyCmd->getRequirement(), SrcAllocaCmd});
388  Dep->addUser(MemCpyCmd);
389  }
390 
391  updateLeaves(Deps, Record, Req->MAccessMode);
392  addNodeToLeaves(Record, MemCpyCmd, Req->MAccessMode);
393  if (MPrintOptionsArray[AfterAddCopyBack])
394  printGraphAsDot("after_addCopyBack");
395  return MemCpyCmd;
396 }

398 // The function implements SYCL host accessor logic: host accessor
399 // should provide access to the buffer in user space.
400 Command *Scheduler::GraphBuilder::addHostAccessor(Requirement *Req,
401  const bool destructor) {
402 
403  const QueueImplPtr &HostQueue = getInstance().getDefaultHostQueue();
404 
405  MemObjRecord *Record = getOrInsertMemObjRecord(HostQueue, Req);
406  if (MPrintOptionsArray[BeforeAddHostAcc])
407  printGraphAsDot("before_addHostAccessor");
408  markModifiedIfWrite(Record, Req);
409 
410  AllocaCommandBase *HostAllocaCmd =
411  getOrCreateAllocaForReq(Record, Req, HostQueue);
412 
413  if (sameCtx(HostAllocaCmd->getQueue()->getContextImplPtr(),
414  Record->MCurContext)) {
415  if (!isAccessModeAllowed(Req->MAccessMode, Record->MHostAccess))
416  remapMemoryObject(Record, Req, HostAllocaCmd);
417  } else
418  insertMemoryMove(Record, Req, HostQueue);
419 
420  Command *UpdateHostAccCmd = insertUpdateHostReqCmd(Record, Req, HostQueue);
421 
422  // Need empty command to be blocked until host accessor is destructed
423  EmptyCommand *EmptyCmd = new EmptyCommand(HostQueue, *Req);
424  EmptyCmd->addDep(
425  DepDesc{UpdateHostAccCmd, EmptyCmd->getRequirement(), HostAllocaCmd});
426  UpdateHostAccCmd->addUser(EmptyCmd);
427 
428  EmptyCmd->MIsBlockable = true;
429  EmptyCmd->MCanEnqueue = false;
430  EmptyCmd->MBlockReason = "A Buffer is locked by the host accessor";
431 
432  updateLeaves({UpdateHostAccCmd}, Record, Req->MAccessMode);
433  addNodeToLeaves(Record, EmptyCmd, Req->MAccessMode);
434 
435  Req->MBlockedCmd = EmptyCmd;
436 
437  if (MPrintOptionsArray[AfterAddHostAcc])
438  printGraphAsDot("after_addHostAccessor");
439 
440  return UpdateHostAccCmd;
441 }

453 // The functions finds dependencies for the requirement. It starts searching
454 // from list of "leaf" commands for the record and check if the examining
455 // command can be executed in parallel with new one with regard to the memory
456 // object. If can, then continue searching through dependencies of that
457 // command. There are several rules used:
458 //
459 // 1. New and examined commands only read -> can bypass
460 // 2. New and examined commands has non-overlapping requirements -> can bypass
461 // 3. New and examined commands has different contexts -> cannot bypass
462 std::set<Command *>
463 Scheduler::GraphBuilder::findDepsForReq(MemObjRecord *Record, Requirement *Req,
464  const ContextImplPtr &Context) {
465  std::set<Command *> RetDeps;
466  std::set<Command *> Visited;
467  const bool ReadOnlyReq = Req->MAccessMode == access::mode::read;
468 
469  std::vector<Command *> ToAnalyze{Record->MWriteLeaves.begin(),
470  Record->MWriteLeaves.end()};
471 
472  if (!ReadOnlyReq)
473  ToAnalyze.insert(ToAnalyze.begin(), Record->MReadLeaves.begin(),
474  Record->MReadLeaves.end());
475 
476  while (!ToAnalyze.empty()) {
477  Command *DepCmd = ToAnalyze.back();
478  ToAnalyze.pop_back();
479 
480  std::vector<Command *> NewAnalyze;
481 
482  for (const DepDesc &Dep : DepCmd->MDeps) {
483  if (Dep.MDepRequirement->MSYCLMemObj != Req->MSYCLMemObj)
484  continue;
485 
486  bool CanBypassDep = false;
487  // If both only read
488  CanBypassDep |=
489  Dep.MDepRequirement->MAccessMode == access::mode::read && ReadOnlyReq;
490 
491  // If not overlap
492  CanBypassDep |= !doOverlap(Dep.MDepRequirement, Req);
493 
494  // Going through copying memory between contexts is not supported.
495  if (Dep.MDepCommand)
496  CanBypassDep &=
497  sameCtx(Context, Dep.MDepCommand->getQueue()->getContextImplPtr());
498 
499  if (!CanBypassDep) {
500  RetDeps.insert(DepCmd);
501  // No need to analyze deps of examining command as it's dependency
502  // itself.
503  NewAnalyze.clear();
504  break;
505  }
506 
507  if (Visited.insert(Dep.MDepCommand).second)
508  NewAnalyze.push_back(Dep.MDepCommand);
509  }
510  ToAnalyze.insert(ToAnalyze.end(), NewAnalyze.begin(), NewAnalyze.end());
511  }
512  return RetDeps;
513 }

528 // The function searches for the alloca command matching context and
529 // requirement.
530 AllocaCommandBase *Scheduler::GraphBuilder::findAllocaForReq(
531  MemObjRecord *Record, Requirement *Req, const ContextImplPtr &Context) {
532  auto IsSuitableAlloca = [&Context, Req](AllocaCommandBase *AllocaCmd) {
533  bool Res = sameCtx(AllocaCmd->getQueue()->getContextImplPtr(), Context);
534  if (IsSuitableSubReq(Req)) {
535  const Requirement *TmpReq = AllocaCmd->getRequirement();
536  Res &= TmpReq->MOffsetInBytes == Req->MOffsetInBytes;
537  Res &= TmpReq->MSYCLMemObj->getSize() == Req->MSYCLMemObj->getSize();
538  }
539  return Res;
540  };
541  const auto It = std::find_if(Record->MAllocaCommands.begin(),
542  Record->MAllocaCommands.end(), IsSuitableAlloca);
543  return (Record->MAllocaCommands.end() != It) ? *It : nullptr;
544 }

546 // The function searches for the alloca command matching context and
547 // requirement. If none exists, new allocation command is created.
548 // Note, creation of new allocation command can lead to the current context
549 // (Record->MCurContext) change.
550 AllocaCommandBase *Scheduler::GraphBuilder::getOrCreateAllocaForReq(
551  MemObjRecord *Record, Requirement *Req, QueueImplPtr Queue) {
552 
553  AllocaCommandBase *AllocaCmd =
554  findAllocaForReq(Record, Req, Queue->getContextImplPtr());
555 
556  if (!AllocaCmd) {
557  if (IsSuitableSubReq(Req)) {
558  // Get parent requirement. It's hard to get right parents' range
559  // so full parent requirement has range represented in bytes
560  range<3> ParentRange{Req->MSYCLMemObj->getSize(), 1, 1};
561  Requirement ParentRequirement(/*Offset*/ {0, 0, 0}, ParentRange,
562  ParentRange, access::mode::read_write,
563  Req->MSYCLMemObj, /*Dims*/ 1,
564  /*Working with bytes*/ sizeof(char));
565 
566  auto *ParentAlloca =
567  getOrCreateAllocaForReq(Record, &ParentRequirement, Queue);
568  AllocaCmd = new AllocaSubBufCommand(Queue, *Req, ParentAlloca);
569  } else {
570 
571  const Requirement FullReq(/*Offset*/ {0, 0, 0}, Req->MMemoryRange,
572  Req->MMemoryRange, access::mode::read_write,
573  Req->MSYCLMemObj, Req->MDims, Req->MElemSize);
574  // Can reuse user data for the first allocation
575  const bool InitFromUserData = Record->MAllocaCommands.empty();
576 
577  AllocaCommandBase *LinkedAllocaCmd = nullptr;
578  // If it is not the first allocation, try to setup a link
579  // FIXME: Temporary limitation, linked alloca commands for an image is not
580  // supported because map operation is not implemented for an image.
581  if (!Record->MAllocaCommands.empty() &&
582  Req->MSYCLMemObj->getType() == SYCLMemObjI::MemObjType::BUFFER)
583  // Current limitation is to setup link between current allocation and
584  // new one. There could be situations when we could setup link with
585  // "not" current allocation, but it will require memory copy.
586  // Can setup link between cl and host allocations only
587  if (Queue->is_host() != Record->MCurContext->is_host()) {
588 
589  AllocaCommandBase *LinkedAllocaCmdCand =
590  findAllocaForReq(Record, Req, Record->MCurContext);
591 
592  // Cannot setup link if candidate is linked already
593  if (LinkedAllocaCmdCand && !LinkedAllocaCmdCand->MLinkedAllocaCmd)
594  LinkedAllocaCmd = LinkedAllocaCmdCand;
595  }
596 
597  AllocaCmd =
598  new AllocaCommand(Queue, FullReq, InitFromUserData, LinkedAllocaCmd);
599 
600  // Update linked command
601  if (LinkedAllocaCmd) {
602  AllocaCmd->addDep(DepDesc{LinkedAllocaCmd, AllocaCmd->getRequirement(),
603  LinkedAllocaCmd});
604  LinkedAllocaCmd->addUser(AllocaCmd);
605  LinkedAllocaCmd->MLinkedAllocaCmd = AllocaCmd;
606 
607  // To ensure that the leader allocation is removed first
608  AllocaCmd->getReleaseCmd()->addDep(
609  DepDesc(LinkedAllocaCmd->getReleaseCmd(), AllocaCmd->getRequirement(),
610  LinkedAllocaCmd));
611 
612  // Device allocation takes ownership of the host ptr during
613  // construction, host allocation doesn't. So, device allocation should
614  // always be active here. Also if the "follower" command is a device one
615  // we have to change current context to the device one.
616  if (Queue->is_host()) {
617  AllocaCmd->MIsActive = false;
618  } else {
619  LinkedAllocaCmd->MIsActive = false;
620  Record->MCurContext = Queue->getContextImplPtr();
621  }
622  }
623  }
624 
625  Record->MAllocaCommands.push_back(AllocaCmd);
626  Record->MWriteLeaves.push_back(AllocaCmd);
627  ++(AllocaCmd->MLeafCounter);
628  }
629  return AllocaCmd;

647 Command *
648 Scheduler::GraphBuilder::addCG(std::unique_ptr<detail::CG> CommandGroup,
649  QueueImplPtr Queue) {
650  const std::vector<Requirement *> &Reqs = CommandGroup->MRequirements;
651  const std::vector<detail::EventImplPtr> &Events = CommandGroup->MEvents;
652  std::unique_ptr<ExecCGCommand> NewCmd(
653  new ExecCGCommand(std::move(CommandGroup), Queue));
654  if (!NewCmd)
655  throw runtime_error("Out of host memory", PI_OUT_OF_HOST_MEMORY);
656 
657  if (MPrintOptionsArray[BeforeAddCG])
658  printGraphAsDot("before_addCG");
659 
660  for (Requirement *Req : Reqs) {
661  MemObjRecord *Record = getOrInsertMemObjRecord(Queue, Req);
662  markModifiedIfWrite(Record, Req);
663 
664  AllocaCommandBase *AllocaCmd = getOrCreateAllocaForReq(Record, Req, Queue);
665  // If there is alloca command we need to check if the latest memory is in
666  // required context.
667  if (sameCtx(Queue->getContextImplPtr(), Record->MCurContext)) {
668  // If the memory is already in the required host context, check if the
669  // required access mode is valid, remap if not.
670  if (Record->MCurContext->is_host() &&
671  !isAccessModeAllowed(Req->MAccessMode, Record->MHostAccess))
672  remapMemoryObject(Record, Req, AllocaCmd);
673  } else {
674  // Cannot directly copy memory from OpenCL device to OpenCL device -
675  // create two copies: device->host and host->device.
676  if (!Queue->is_host() && !Record->MCurContext->is_host())
677  insertMemoryMove(Record, Req,
678  Scheduler::getInstance().getDefaultHostQueue());
679  insertMemoryMove(Record, Req, Queue);
680  }
681  std::set<Command *> Deps =
682  findDepsForReq(Record, Req, Queue->getContextImplPtr());
683 
684  for (Command *Dep : Deps)
685  NewCmd->addDep(DepDesc{Dep, Req, AllocaCmd});
686  }
687 
688  // Set new command as user for dependencies and update leaves.
689  for (DepDesc &Dep : NewCmd->MDeps) {
690  Dep.MDepCommand->addUser(NewCmd.get());
691  const Requirement *Req = Dep.MDepRequirement;
692  MemObjRecord *Record = getMemObjRecord(Req->MSYCLMemObj);
693  updateLeaves({Dep.MDepCommand}, Record, Req->MAccessMode);
694  addNodeToLeaves(Record, NewCmd.get(), Req->MAccessMode);
695  }
696 
697  // Register all the events as dependencies
698  for (detail::EventImplPtr e : Events) {
699  NewCmd->addDep(e);
700  }
701 
702  if (MPrintOptionsArray[AfterAddCG])
703  printGraphAsDot("after_addCG");
704  return NewCmd.release();
705 }

717 void Scheduler::GraphBuilder::cleanupCommandsForRecord(MemObjRecord *Record) {
718  std::vector<AllocaCommandBase *> &AllocaCommands = Record->MAllocaCommands;
719  if (AllocaCommands.empty())
720  return;
721 
722  std::queue<Command *> ToVisit;
723  std::set<Command *> Visited;
724  std::vector<Command *> CmdsToDelete;
725  // First, mark all allocas for deletion and their direct users for traversal
726  // Dependencies of the users will be cleaned up during the traversal
727  for (Command *AllocaCmd : AllocaCommands) {
728  Visited.insert(AllocaCmd);
729  for (Command *UserCmd : AllocaCmd->MUsers)
730  ToVisit.push(UserCmd);
731  CmdsToDelete.push_back(AllocaCmd);
732  // These commands will be deleted later, clear users now to avoid
733  // updating them during edge removal
734  AllocaCmd->MUsers.clear();
735  }
736 
737  // Traverse the graph using BFS
738  while (!ToVisit.empty()) {
739  Command *Cmd = ToVisit.front();
740  ToVisit.pop();
741 
742  if (!Visited.insert(Cmd).second)
743  continue;
744 
745  for (Command *UserCmd : Cmd->MUsers)
746  ToVisit.push(UserCmd);
747 
748  // Delete all dependencies on any allocations being removed
749  // Track which commands should have their users updated
750  std::map<Command *, bool> ShouldBeUpdated;
751  auto NewEnd = std::remove_if(
752  Cmd->MDeps.begin(), Cmd->MDeps.end(), [&](const DepDesc &Dep) {
753  if (std::find(AllocaCommands.begin(), AllocaCommands.end(),
754  Dep.MAllocaCmd) != AllocaCommands.end()) {
755  ShouldBeUpdated.insert({Dep.MDepCommand, true});
756  return true;
757  }
758  ShouldBeUpdated[Dep.MDepCommand] = false;
759  return false;
760  });
761  Cmd->MDeps.erase(NewEnd, Cmd->MDeps.end());
762 
763  // Update users of removed dependencies
764  for (auto DepCmdIt : ShouldBeUpdated) {
765  if (!DepCmdIt.second)
766  continue;
767  DepCmdIt.first->MUsers.erase(Cmd);
768  }
769 
770  // If all dependencies have been removed this way, mark the command for
771  // deletion
772  if (Cmd->MDeps.empty()) {
773  CmdsToDelete.push_back(Cmd);
774  Cmd->MUsers.clear();
775  }
776  }
777 
778  for (Command *Cmd : CmdsToDelete) {
779  Cmd->getEvent()->setCommand(nullptr);
780  delete Cmd;
781  }
782 }

784 void Scheduler::GraphBuilder::cleanupFinishedCommands(Command *FinishedCmd) {
785  std::queue<Command *> CmdsToVisit({FinishedCmd});
786  std::set<Command *> Visited;
787 
788  // Traverse the graph using BFS
789  while (!CmdsToVisit.empty()) {
790  Command *Cmd = CmdsToVisit.front();
791  CmdsToVisit.pop();
792 
793  if (!Visited.insert(Cmd).second)
794  continue;
795 
796  for (const DepDesc &Dep : Cmd->MDeps) {
797  if (Dep.MDepCommand)
798  CmdsToVisit.push(Dep.MDepCommand);
799  }
800 
801  // Do not clean up the node if it is a leaf for any memory object
802  if (Cmd->MLeafCounter > 0)
803  continue;
804  // Do not clean up allocation commands
805  Command::CommandType CmdT = Cmd->getType();
806  if (CmdT == Command::ALLOCA || CmdT == Command::ALLOCA_SUB_BUF)
807  continue;
808 
809  for (Command *UserCmd : Cmd->MUsers) {
810  for (DepDesc &Dep : UserCmd->MDeps) {
811  // Link the users of the command to the alloca command(s) instead
812  if (Dep.MDepCommand == Cmd) {
813  Dep.MDepCommand = Dep.MAllocaCmd;
814  Dep.MDepCommand->MUsers.insert(UserCmd);
815  }
816  }
817  }
818  // Update dependency users
819  for (DepDesc &Dep : Cmd->MDeps) {
820  Command *DepCmd = Dep.MDepCommand;
821  DepCmd->MUsers.erase(Cmd);
822  }
823  Cmd->getEvent()->setCommand(nullptr);
824  delete Cmd;
825  }
826 }
