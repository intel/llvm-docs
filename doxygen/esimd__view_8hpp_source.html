26 template <typename BaseTy, typename RegionTy> class simd_view {
27 public:
28  static_assert(!is_simd_view_v<BaseTy>::value);
29  // Deduce the corresponding value type from its region type.
30  using ShapeTy = typename shape_type<RegionTy>::type;

39  // The region type of this class.

42  // The element type of this class, which could be different from the element
43  // type of the base object type.
44  using element_type = typename ShapeTy::element_type;

107  // Read this simd_view object.
108  value_type read() const {
109  using BT = typename BaseTy::element_type;

114  // Write to this simd_view object.

199 #define DEF_BINOP(BINOP, OPASSIGN) \
200  auto operator BINOP(const value_type &RHS) const { \
201  using ComputeTy = compute_type_t<value_type>; \
202  auto V0 = convert<typename ComputeTy::vector_type>(read().data()); \
203  auto V1 = convert<typename ComputeTy::vector_type>(RHS.data()); \
204  auto V2 = V0 BINOP V1; \
205  return ComputeTy(V2); \
206  } \
207  simd_view &operator OPASSIGN(const value_type &RHS) { \
208  using ComputeTy = compute_type_t<value_type>; \
209  auto V0 = convert<typename ComputeTy::vector_type>(read().data()); \
210  auto V1 = convert<typename ComputeTy::vector_type>(RHS.data()); \
211  auto V2 = V0 BINOP V1; \
212  auto V3 = convert<vector_type>(V2); \
213  write(V3); \
214  return *this; \
215  }

225  simd<uint16_t, length> operator RELOP(const simd_view &RHS) const { \
