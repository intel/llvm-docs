701 // bit_cast Color to a type the backend is known to accept
702 template <typename DataT> constexpr auto convert_color(DataT Color) {
703  constexpr size_t dataSize = sizeof(DataT);
704  static_assert(
705  dataSize == 1 || dataSize == 2 || dataSize == 4 || dataSize == 8 ||
706  dataSize == 16,
707  "Expected input data type to be of size 1, 2, 4, 8, or 16 bytes.");
708 
709  if constexpr (dataSize == 1) {
710  return sycl::bit_cast<uint8_t>(Color);
711  } else if constexpr (dataSize == 2) {
712  return sycl::bit_cast<uint16_t>(Color);
713  } else if constexpr (dataSize == 4) {
714  return sycl::bit_cast<uint32_t>(Color);
715  } else if constexpr (dataSize == 8) {
716  return sycl::bit_cast<sycl::vec<uint32_t, 2>>(Color);
717  } else { // dataSize == 16
718  return sycl::bit_cast<sycl::vec<uint32_t, 4>>(Color);
719  }
720 }

722 // assert coords or elements of coords is of an integer type
723 template <typename CoordT> constexpr void assert_unsampled_coords() {
724  if constexpr (std::is_scalar_v<CoordT>) {
725  static_assert(std::is_same_v<CoordT, int>,
726  "Expected integer coordinate data type");
727  } else {
728  static_assert(is_vec_v<CoordT>, "Expected sycl::vec coordinates");
729  static_assert(std::is_same_v<typename CoordT::element_type, int>,
730  "Expected integer coordinates data type");
731  }
732 }

736  if constexpr (std::is_scalar_v<CoordT>) {
737  static_assert(std::is_same_v<CoordT, float>,
738  "Expected float coordinate data type");
739  } else {

741  static_assert(std::is_same_v<typename CoordT::element_type, float>,
742  "Expected float coordinates data type");

746 template <typename DataT> constexpr bool is_data_size_valid() {
747  return (sizeof(DataT) == 1) || (sizeof(DataT) == 2) || (sizeof(DataT) == 4) ||
748  (sizeof(DataT) == 8) || (sizeof(DataT) == 16);
749 }

751 template <typename DataT> constexpr bool is_recognized_standard_type() {
752  return is_data_size_valid<DataT>() &&
753  (is_vec_v<DataT> || std::is_scalar_v<DataT> ||
754  std::is_floating_point_v<DataT> || std::is_same_v<DataT, sycl::half>);
755 }

807 template <typename DataT, typename HintT = DataT, typename CoordT>
808 DataT fetch_image(const unsampled_image_handle &imageHandle [[maybe_unused]],
809  const CoordT &coords [[maybe_unused]]) {
810  detail::assert_unsampled_coords<CoordT>();
811  constexpr size_t coordSize = detail::coord_size<CoordT>();
812  static_assert(coordSize == 1 || coordSize == 2 || coordSize == 3,
813  "Expected input coordinate to be have 1, 2, or 3 components "
814  "for 1D, 2D and 3D images, respectively.");
815 
816 #ifdef __SYCL_DEVICE_ONLY__
817  if constexpr (detail::is_recognized_standard_type<DataT>()) {
818  return __invoke__ImageRead<DataT>(imageHandle.raw_handle, coords);
819  } else {
820  static_assert(sizeof(HintT) == sizeof(DataT),
821  "When trying to read a user-defined type, HintT must be of "
822  "the same size as the user-defined DataT.");
823  static_assert(detail::is_recognized_standard_type<HintT>(),
824  "HintT must always be a recognized standard type");
825  return sycl::bit_cast<DataT>(
826  __invoke__ImageRead<HintT>(imageHandle.raw_handle, coords));
827  }
828 #else
829  assert(false); // Bindless images not yet implemented on host
830 #endif
831 }
