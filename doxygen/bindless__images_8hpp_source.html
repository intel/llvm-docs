704 // bit_cast Color to a type the backend is known to accept
705 template <typename DataT> constexpr auto convert_color(DataT Color) {
706  constexpr size_t dataSize = sizeof(DataT);
707  static_assert(
708  dataSize == 1 || dataSize == 2 || dataSize == 4 || dataSize == 8 ||
709  dataSize == 16,
710  "Expected input data type to be of size 1, 2, 4, 8, or 16 bytes.");
711 
712  if constexpr (dataSize == 1) {
713  return sycl::bit_cast<uint8_t>(Color);
714  } else if constexpr (dataSize == 2) {
715  return sycl::bit_cast<uint16_t>(Color);
716  } else if constexpr (dataSize == 4) {
717  return sycl::bit_cast<uint32_t>(Color);
718  } else if constexpr (dataSize == 8) {
719  return sycl::bit_cast<sycl::vec<uint32_t, 2>>(Color);
720  } else { // dataSize == 16
721  return sycl::bit_cast<sycl::vec<uint32_t, 4>>(Color);
722  }

881 template <typename DataT, typename HintT = DataT, typename CoordT>
882 DataT fetch_image(const unsampled_image_handle &imageHandle [[maybe_unused]],
883  const CoordT &coords [[maybe_unused]]) {
884  detail::assert_fetch_coords<CoordT>();
885  constexpr size_t coordSize = detail::coord_size<CoordT>();
886  static_assert(coordSize == 1 || coordSize == 2 || coordSize == 3,
887  "Expected input coordinate to be have 1, 2, or 3 components "
888  "for 1D, 2D and 3D images, respectively.");
889 
890 #ifdef __SYCL_DEVICE_ONLY__
891  if constexpr (detail::is_recognized_standard_type<DataT>()) {
892  return FETCH_UNSAMPLED_IMAGE(
893  DataT,
894  CONVERT_HANDLE_TO_IMAGE(imageHandle.raw_handle,
895  detail::OCLImageTyRead<coordSize>),
896  coords);
897 
898  } else {
899  static_assert(sizeof(HintT) == sizeof(DataT),
900  "When trying to read a user-defined type, HintT must be of "
901  "the same size as the user-defined DataT.");
902  static_assert(detail::is_recognized_standard_type<HintT>(),
903  "HintT must always be a recognized standard type");
904  return sycl::bit_cast<DataT>(FETCH_UNSAMPLED_IMAGE(
905  HintT,
906  CONVERT_HANDLE_TO_IMAGE(imageHandle.raw_handle,
907  detail::OCLImageTyRead<coordSize>),
908  coords));
909  }
910 #else
911  assert(false); // Bindless images not yet implemented on host
912 #endif
913 }
