128 void shutdown() {
129  // First, release resources, that may access plugins.
130  GlobalHandler::instance().MScheduler.reset(nullptr);
131  GlobalHandler::instance().MProgramManager.reset(nullptr);
132  GlobalHandler::instance().MPlatformCache.reset(nullptr);
133 
134  // Call to GlobalHandler::instance().getPlugins() initializes plugins. If
135  // user application has loaded SYCL runtime, and never called any APIs,
136  // there's no need to load and unload plugins.
137  if (GlobalHandler::instance().MPlugins) {
138  for (plugin &Plugin : GlobalHandler::instance().getPlugins()) {
139  // PluginParameter is reserved for future use that can control
140  // some parameters in the plugin tear-down process.
141  // Currently, it is not used.
142  void *PluginParameter = nullptr;
143  Plugin.call_nocheck<PiApiKind::piTearDown>(PluginParameter);
144  Plugin.unload();
145  }
146  GlobalHandler::instance().MPlugins.reset(nullptr);
147  }
148 
149  // Release the rest of global resources.
150  delete &GlobalHandler::instance();
151 }

154 BOOL WINAPI DllMain(HINSTANCE hinstDLL, DWORD fdwReason, LPVOID lpReserved) {
155  // Perform actions based on the reason for calling.
156  switch (fdwReason) {

160  case DLL_PROCESS_ATTACH:
161  case DLL_THREAD_ATTACH:
162  case DLL_THREAD_DETACH:

166 }

168 // Setting low priority on destructor ensures it runs after all other global

170 // value 110 allows SYCL users to run their destructors after runtime library
171 // deinitialization.
