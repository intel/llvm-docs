237  // Returns all guarantees that are either equal to guarantee or weaker than
238  // it. E.g if guarantee == parallel, it returns the vector {weakly_parallel,
239  // parallel}.
240  template <typename ReturnT>
241  static ReturnT getProgressGuaranteesUpTo(
242  ext::oneapi::experimental::forward_progress_guarantee guarantee) {
243  const int forwardProgressGuaranteeSize = 3;
244  int guaranteeVal = static_cast<int>(guarantee);
245  ReturnT res;
246  res.reserve(forwardProgressGuaranteeSize - guaranteeVal);
247  for (int currentGuarantee = forwardProgressGuaranteeSize - 1;
248  currentGuarantee >= guaranteeVal; --currentGuarantee) {
249  res.emplace_back(
250  static_cast<ext::oneapi::experimental::forward_progress_guarantee>(
251  currentGuarantee));
252  }
253  return res;
254  }
