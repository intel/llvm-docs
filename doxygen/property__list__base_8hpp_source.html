118  void delete_accessor_property_helper(const PropWithDataKind &Kind) {
119  auto It = MPropsWithData.begin();
120  for (; It != MPropsWithData.end(); ++It) {
121  if ((*It)->isSame(Kind))
122  break;
123  }
124  if (It != MPropsWithData.end()) {
