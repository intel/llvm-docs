28 template <typename T0, typename T1, int SZ>
29 ESIMD_NODEBUG ESIMD_INLINE simd<T0, SZ> esimd_sat(simd<T1, SZ> src) {
30  if constexpr (std::is_floating_point<T0>::value)
31  return __esimd_satf<T0, T1, SZ>(src.data());
32  else if constexpr (std::is_floating_point<T1>::value) {
33  if constexpr (std::is_unsigned<T0>::value)
34  return __esimd_fptoui_sat<T0, T1, SZ>(src.data());
35  else
36  return __esimd_fptosi_sat<T0, T1, SZ>(src.data());
37  } else if constexpr (std::is_unsigned<T0>::value) {
38  if constexpr (std::is_unsigned<T1>::value)
39  return __esimd_uutrunc_sat<T0, T1, SZ>(src.data());
40  else
41  return __esimd_ustrunc_sat<T0, T1, SZ>(src.data());
42  } else {
43  if constexpr (std::is_signed<T1>::value)
44  return __esimd_sutrunc_sat<T0, T1, SZ>(src.data());
45  else
46  return __esimd_sstrunc_sat<T0, T1, SZ>(src.data());
47  }
48 }

1743 template <int N>
1744 ESIMD_INLINE simd<float, N> esimd_tanh_cody_waite_impl(simd<float, N> x) {
1745  /*
1746  * 0 x_small x_medium x_large
1747  * | x | rational polynomial | 1 - 2/(1 + exp(2*x)) | 1
1748  *
1749  * rational polynomial for single precision = x + x * (g * (p[1] * g + p[0]) /
1750  * (g + q[0]) g = x^2 p0 = −0.82377 28127 E+00 p1 = −0.38310 10665 E-02 q0 =
1751  * 0.24713 19654 E+01 q1 = 1.00000 00000 E+00
1752  *
1753  */
1754 
1755  constexpr float p0 = -0.8237728127E+00f;
1756  constexpr float p1 = -0.3831010665E-02f;
1757  constexpr float q0 = 0.2471319654E+01f;
1758  constexpr float q1 = 1.0000000000E+00f;
1759  constexpr float xsmall = 4.22863966691620432990E-04f;
1760  constexpr float xmedium = 0.54930614433405484570f;
1761  constexpr float xlarge = 8.66433975699931636772f;
1762  constexpr float log2E = 1.442695f; // same as esimd_log(e)
1763 
1764  using RT = simd<float, N>;
1765 
1766  RT absX = esimd_abs(x);
1767  RT g = absX * absX;
1768 
1769  RT sign;
1770  sign.merge(-1.f, 1.f, x < 0.f);
1771 
1772  auto isLarge = absX > xlarge;
1773  auto minor = absX <= xlarge;
1774  auto isGtMed = minor & (absX > xmedium);
1775  auto isGtSmall = (absX > xsmall) & (absX <= xmedium);
1776 
1777  RT res;
1778  res.merge(sign, x, isLarge);
1779  auto temp = esimd_exp(absX * 2.0f * log2E) + 1.f;
1780  temp = ((temp - 2.f) / temp) * sign;
1781  res.merge(temp, isGtMed);
1782  res.merge((absX + absX * g * (g * p1 + p0) / (g + q0)) * sign, isGtSmall);
1783 
1784  return res;
1785 }
