108 template <> class SYCLConfig<SYCL_BE> {
109  using BaseT = SYCLConfigBase<SYCL_BE>;
110 
111 public:
112  static backend *get() {
113  static bool Initialized = false;
114  static backend *BackendPtr = nullptr;
115 
116  // Configuration parameters are processed only once, like reading a string
117  // from environment and converting it into a typed object.
118  if (Initialized)
119  return BackendPtr;
120 
121  const char *ValStr = BaseT::getRawValue();
122  const std::array<std::pair<std::string, backend>, 4> SyclBeMap = {
123  {{"PI_OPENCL", backend::opencl},
124  {"PI_LEVEL_ZERO", backend::level_zero},
125  {"PI_LEVEL0", backend::level_zero}, // for backward compatibility
126  {"PI_CUDA", backend::cuda}}};
127  if (ValStr) {
128  auto It = std::find_if(
129  std::begin(SyclBeMap), std::end(SyclBeMap),
130  [&ValStr](const std::pair<std::string, backend> &element) {
131  return element.first == ValStr;
132  });
133  if (It == SyclBeMap.end())
134  pi::die("Invalid backend. "
135  "Valid values are PI_OPENCL/PI_LEVEL_ZERO/PI_CUDA");
136  static backend Backend = It->second;
137  BackendPtr = &Backend;
138  }
139  Initialized = true;
140  return BackendPtr;
141  }
