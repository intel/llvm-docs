75 template <ConfigID Config> class SYCLConfig {
76  using BaseT = SYCLConfigBase<Config>;
77 
78 public:
79  static const char *get() {
80  const char *ValStr = getRawValue();
81  return ValStr;
82  }
83 
84 private:
85  static const char *getRawValue() {
86  if (ConfigFromEnvEnabled)
87  if (const char *ValStr = getenv(BaseT::MConfigName))
88  return ValStr;
89 
90  if (ConfigFromFileEnabled) {
91  readConfig();
92  if (BaseT::MValueFromFile)
93  return BaseT::MValueFromFile;
94  }
95 
96  if (ConfigFromCompileDefEnabled && BaseT::MCompileTimeDef)
97  return BaseT::MCompileTimeDef;
98 
99  return nullptr;
100  }
101 };
