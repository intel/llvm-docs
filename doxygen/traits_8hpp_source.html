137 // tuple_template_index is a trait helper which finds the index of a class
138 // template in a std::tuple<Ts...>. During template argument deduction for
139 // launch, this enables us to search the tuple for e.g. `kernel_properties`
140 // without knowing the concrete type (e.g. kernel_properties<KProps>) A compile
141 // time error is raised if the class template is found more than once. If not
142 // found, returns the tuple size (i.e. this is not an error).
143 template <template <typename TT> typename PropertyContainer, typename Tuple>
144 struct tuple_template_index {
145  static constexpr std::size_t value =
146  tuple_template_index_helper<PropertyContainer, Tuple>::value;
147 };
