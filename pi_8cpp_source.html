74 std::string memFlagToString(pi_mem_flags Flag) {
75  assertion(((Flag == 0u) || ((Flag & (Flag - 1)) == 0)) &&
76  "More than one bit set");
77 
78  std::stringstream Sstream;
79 
80  switch (Flag) {
81  case pi_mem_flags{0}:
82  Sstream << "pi_mem_flags(0)";
83  break;
84  case PI_MEM_FLAGS_ACCESS_RW:
85  Sstream << "PI_MEM_FLAGS_ACCESS_RW";
86  break;
87  case PI_MEM_FLAGS_HOST_PTR_USE:
88  Sstream << "PI_MEM_FLAGS_HOST_PTR_USE";
89  break;
90  case PI_MEM_FLAGS_HOST_PTR_COPY:
91  Sstream << "PI_MEM_FLAGS_HOST_PTR_COPY";
92  break;
93  default:
94  Sstream << "unknown pi_mem_flags bit == " << Flag;
95  }
96 
97  return Sstream.str();
98 }

100 std::string memFlagsToString(pi_mem_flags Flags) {
101  std::stringstream Sstream;
102  bool FoundFlag = false;
103 
104  auto FlagSeparator = [](bool FoundFlag) { return FoundFlag ? "|" : ""; };
105 
106  pi_mem_flags ValidFlags[] = {PI_MEM_FLAGS_ACCESS_RW,
107  PI_MEM_FLAGS_HOST_PTR_USE,
108  PI_MEM_FLAGS_HOST_PTR_COPY};
109 
110  if (Flags == 0u) {
111  Sstream << "pi_mem_flags(0)";
112  } else {
113  for (const auto Flag : ValidFlags) {
114  if (Flag & Flags) {
115  Sstream << FlagSeparator(FoundFlag) << memFlagToString(Flag);
116  FoundFlag = true;
117  }
118  }
119 
120  std::bitset<64> UnkownBits(Flags & ~(PI_MEM_FLAGS_ACCESS_RW |
121  PI_MEM_FLAGS_HOST_PTR_USE |
122  PI_MEM_FLAGS_HOST_PTR_COPY));
123  if (UnkownBits.any()) {
124  Sstream << FlagSeparator(FoundFlag)
125  << "unknown pi_mem_flags bits == " << UnkownBits;
126  }
127  }
128 
129  return Sstream.str();
130 }

198 // Load the plugin based on SYCL_BE.
199 // TODO: Currently only accepting OpenCL and CUDA plugins. Edit it to identify
200 // and load other kinds of plugins, do the required changes in the
201 // findPlugins, loadPlugin and bindPlugin functions.
202 vector_class<plugin> initialize() {
203  vector_class<plugin> Plugins;
204 
205  if (!useBackend(SYCL_BE_PI_OPENCL) && !useBackend(SYCL_BE_PI_CUDA)) {
206  die("Unknown SYCL_BE");
207  }
208 
209  bool EnableTrace = (std::getenv("SYCL_PI_TRACE") != nullptr);
210 
211  vector_class<std::string> PluginNames;
212  findPlugins(PluginNames);
213 
214  if (PluginNames.empty() && EnableTrace)
215  std::cerr << "No Plugins Found." << std::endl;
216 
217  PiPlugin PluginInformation; // TODO: include.
218  for (unsigned int I = 0; I < PluginNames.size(); I++) {
219  void *Library = loadPlugin(PluginNames[I]);
220  if (!Library && EnableTrace) {
221  std::cerr << "Check if plugin is present. Failed to load plugin: "
222  << PluginNames[I] << std::endl;
223  }
224 
225  if (!bindPlugin(Library, &PluginInformation) && EnableTrace) {
226  std::cerr << "Failed to bind PI APIs to the plugin: " << PluginNames[I]
227  << std::endl;
228  }
229  if (useBackend(SYCL_BE_PI_OPENCL) &&
230  PluginNames[I].find("opencl") != std::string::npos) {
231  // Use the OpenCL plugin as the GlobalPlugin
232  GlobalPlugin = std::make_shared<plugin>(PluginInformation);
233  }
234  if (useBackend(SYCL_BE_PI_CUDA) &&
235  PluginNames[I].find("cuda") != std::string::npos) {
236  // Use the CUDA plugin as the GlobalPlugin
237  GlobalPlugin = std::make_shared<plugin>(PluginInformation);
238  }
239  Plugins.push_back(plugin(PluginInformation));
240  }
241 
242 #ifdef XPTI_ENABLE_INSTRUMENTATION
243  if (!(xptiTraceEnabled() && !XPTIInitDone))
244  return Plugins;
245  // Not sure this is the best place to initialize the framework; SYCL runtime
246  // team needs to advise on the right place, until then we piggy-back on the
247  // initialization of the PI layer.
248 
249  // Initialize the global events just once, in the case pi::initialize() is
250  // called multiple times
251  XPTIInitDone = true;
252  // Registers a new stream for 'sycl' and any plugin that wants to listen to
253  // this stream will register itself using this string or stream ID for this
254  // string.
255  uint8_t StreamID = xptiRegisterStream(SYCL_STREAM_NAME);
256  // Let all tool plugins know that a stream by the name of 'sycl' has been
257  // initialized and will be generating the trace stream.
258  //
259  // +--- Minor version #
260  // Major version # ------+ | Version string
261  // | | |
262  // v v v
263  xptiInitialize(SYCL_STREAM_NAME, GMajVer, GMinVer, GVerStr);
264  // Create a tracepoint to indicate the graph creation
265  xpti::payload_t GraphPayload("application_graph");
266  uint64_t GraphInstanceNo;
267  GSYCLGraphEvent =
268  xptiMakeEvent("application_graph", &GraphPayload, xpti::trace_graph_event,
269  xpti_at::active, &GraphInstanceNo);
270  if (GSYCLGraphEvent) {
271  // The graph event is a global event and will be used as the parent for
272  // all nodes (command groups)
273  xptiNotifySubscribers(StreamID, xpti::trace_graph_create, nullptr,
274  GSYCLGraphEvent, GraphInstanceNo, nullptr);
275  }
276 
277  xpti::payload_t PIPayload("Plugin Interface Layer");
278  uint64_t PiInstanceNo;
279  GPICallEvent =
280  xptiMakeEvent("PI Layer", &PIPayload, xpti::trace_algorithm_event,
281  xpti_at::active, &PiInstanceNo);
282 #endif
283 
284  return Plugins;
285 }
