59 // Selects 8-bit, 16-bit or 32-bit type depending on size of T. If T doesn't
60 // maps to mentioned types, then void is returned

86 template <typename T, access::address_space Space>
87 T load(const multi_ptr<T, Space> src) {
88  using BlockT = SelectBlockT<T>;
89  using PtrT = detail::ConvertToOpenCLType_t<const multi_ptr<BlockT, Space>>;
90 
91  BlockT Ret =
92  __spirv_SubgroupBlockReadINTEL<BlockT>(reinterpret_cast<PtrT>(src.get()));
93 
94  return bit_cast<T>(Ret);
95 }

97 template <int N, typename T, access::address_space Space>
98 vec<T, N> load(const multi_ptr<T, Space> src) {
99  using BlockT = SelectBlockT<T>;
100  using VecT = detail::ConvertToOpenCLType_t<vec<BlockT, N>>;
101  using PtrT = detail::ConvertToOpenCLType_t<const multi_ptr<BlockT, Space>>;
102 
103  VecT Ret =
104  __spirv_SubgroupBlockReadINTEL<VecT>(reinterpret_cast<PtrT>(src.get()));
105 
106  return bit_cast<typename vec<T, N>::vector_t>(Ret);
