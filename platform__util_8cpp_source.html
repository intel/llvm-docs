23 // Used by methods that duplicate OpenCL behaviour in order to get CPU info
24 static void cpuid(uint32_t *CPUInfo, uint32_t Type, uint32_t SubType = 0) {
25 #if defined(SYCL_RT_OS_LINUX)
26  __cpuid_count(Type, SubType, CPUInfo[0], CPUInfo[1], CPUInfo[2], CPUInfo[3]);
27 #elif defined(SYCL_RT_OS_WINDOWS)
28  __cpuidex(reinterpret_cast<int *>(CPUInfo), Type, SubType);
29 #endif
30 }

32 uint32_t PlatformUtil::getMaxClockFrequency() {
33  throw runtime_error(
34  "max_clock_frequency parameter is not supported for host device",
35  PI_INVALID_DEVICE);
36  uint32_t CPUInfo[4];
37  string_class Buff(sizeof(CPUInfo) * 3 + 1, 0);
38  size_t Offset = 0;
39 
40  for (uint32_t i = 0x80000002; i <= 0x80000004; i++) {
41  cpuid(CPUInfo, i);
42  std::copy(reinterpret_cast<char *>(CPUInfo),
43  reinterpret_cast<char *>(CPUInfo) + sizeof(CPUInfo),
44  Buff.begin() + Offset);
45  Offset += sizeof(CPUInfo);
46  }
47  std::size_t Found = Buff.rfind("Hz");
48  // Bail out if frequency is not found in CPUID string
49  if (Found == std::string::npos)
50  return 0;
51 
52  Buff = Buff.substr(0, Found);
53  uint32_t Freq = 0;
54  switch (Buff[Buff.size() - 1]) {
55  case 'M':
56  Freq = 1;
57  break;
58  case 'G':
59  Freq = 1000;
60  break;
61  }
62  Buff = Buff.substr(Buff.rfind(' '), Buff.length());
63  Freq *= std::stod(Buff);
64  return Freq;
65 }

79 uint32_t PlatformUtil::getNativeVectorWidth(PlatformUtil::TypeIndex TIndex) {
80  // SSE4.2 has 16 byte (XMM) registers
81  static constexpr uint32_t VECTOR_WIDTH_SSE42[] = {16, 8, 4, 2, 4, 2, 0};
82  // AVX supports 32 byte (YMM) registers only for floats and doubles
83  static constexpr uint32_t VECTOR_WIDTH_AVX[] = {16, 8, 4, 2, 8, 4, 0};
84  // AVX2 has a full set of 32 byte (YMM) registers
85  static constexpr uint32_t VECTOR_WIDTH_AVX2[] = {32, 16, 8, 4, 8, 4, 0};
86  // AVX512 has 64 byte (ZMM) registers
87  static constexpr uint32_t VECTOR_WIDTH_AVX512[] = {64, 32, 16, 8, 16, 8, 0};
88 
89  uint32_t Index = static_cast<uint32_t>(TIndex);
90 
91 #if defined(SYCL_RT_OS_LINUX)
92  if (__builtin_cpu_supports("avx512f"))
93  return VECTOR_WIDTH_AVX512[Index];
94  if (__builtin_cpu_supports("avx2"))
95  return VECTOR_WIDTH_AVX2[Index];
96  if (__builtin_cpu_supports("avx"))
97  return VECTOR_WIDTH_AVX[Index];
98 #elif defined(SYCL_RT_OS_WINDOWS)
99 
100  uint32_t Info[4];
101 
102  // Check that CPUID func number 7 is available.
103  cpuid(Info, 0);
104  if (Info[0] >= 7) {
105  // avx512f = CPUID.7.EBX[16]
106  cpuid(Info, 7);
107  if (Info[1] & (1 << 16))
108  return VECTOR_WIDTH_AVX512[Index];
109 
110  // avx2 = CPUID.7.EBX[5]
111  if (Info[1] & (1 << 5))
112  return VECTOR_WIDTH_AVX2[Index];
113  }
114  // It is assumed that CPUID func number 1 is always available.
115  // avx = CPUID.1.ECX[28]
116  cpuid(Info, 1);
117  if (Info[2] & (1 << 28))
118  return VECTOR_WIDTH_AVX[Index];
119 #endif
120 
121  return VECTOR_WIDTH_SSE42[Index];
122 }

124 void PlatformUtil::prefetch(const char *Ptr, size_t NumBytes) {
125  if (!Ptr)
126  return;
127 
128  // The current implementation assumes 64-byte x86 cache lines.
129  const size_t CacheLineSize = 64;
130  const size_t CacheLineMask = ~(CacheLineSize - 1);
131  const char *PtrEnd = Ptr + NumBytes;
132 
133  // Set the pointer to the beginning of the current cache line.
134  Ptr = reinterpret_cast<const char *>(
135  reinterpret_cast<size_t>(Ptr) & CacheLineMask);
136  for (; Ptr < PtrEnd; Ptr += CacheLineSize) {
137 #if defined(SYCL_RT_OS_LINUX)
138  __builtin_prefetch(Ptr);
139 #elif defined(SYCL_RT_OS_WINDOWS)
140  _mm_prefetch(Ptr, _MM_HINT_T0);
141 #endif
142  }
143 }
