48 static RT::PiDevice getFirstDevice(const ContextImplPtr &Context) {
49  pi_uint32 NumDevices = 0;
50  const detail::plugin &Plugin = Context->getPlugin();
51  Plugin.call<PiApiKind::piContextGetInfo>(Context->getHandleRef(),
52  PI_CONTEXT_INFO_NUM_DEVICES,
53  sizeof(NumDevices), &NumDevices,
54  /*param_value_size_ret=*/nullptr);
55  assert(NumDevices > 0 && "Context without devices?");
56 
57  vector_class<RT::PiDevice> Devices(NumDevices);
58  size_t ParamValueSize = 0;
59  Plugin.call<PiApiKind::piContextGetInfo>(
60  Context->getHandleRef(), PI_CONTEXT_INFO_DEVICES,
61  sizeof(cl_device_id) * NumDevices, &Devices[0], &ParamValueSize);
62  assert(ParamValueSize == sizeof(cl_device_id) * NumDevices &&
63  "Number of CL_CONTEXT_DEVICES should match CL_CONTEXT_NUM_DEVICES.");
64  return Devices[0];
65 }

67 static RT::PiProgram createBinaryProgram(const ContextImplPtr Context,
68  const unsigned char *Data,
69  size_t DataLen) {
70  // FIXME: we don't yet support multiple devices with a single binary.
71  const detail::plugin &Plugin = Context->getPlugin();
72 #ifndef _NDEBUG
73  pi_uint32 NumDevices = 0;
74  Plugin.call<PiApiKind::piContextGetInfo>(Context->getHandleRef(),
75  PI_CONTEXT_INFO_NUM_DEVICES,
76  sizeof(NumDevices), &NumDevices,
77  /*param_value_size_ret=*/nullptr);
78  assert(NumDevices > 0 &&
79  "Only a single device is supported for AOT compilation");
80 #endif
81 
82  RT::PiProgram Program;
83 
84  bool IsCUDA = false;
85 
86  // TODO: Implement `piProgramCreateWithBinary` to not require extra logic for
87  // the CUDA backend.
88 #if USE_PI_CUDA
89  // All devices in a context are from the same platform.
90  RT::PiDevice Device = getFirstDevice(Context);
91  RT::PiPlatform Platform = nullptr;
92  Plugin.call<PiApiKind::piDeviceGetInfo>(Device, PI_DEVICE_INFO_PLATFORM, sizeof(Platform),
93  &Platform, nullptr);
94  size_t PlatformNameSize = 0u;
95  Plugin.call<PiApiKind::piPlatformGetInfo>(Platform, PI_PLATFORM_INFO_NAME, 0u, nullptr,
96  &PlatformNameSize);
97  std::vector<char> PlatformName(PlatformNameSize, '\0');
98  Plugin.call<PiApiKind::piPlatformGetInfo>(Platform, PI_PLATFORM_INFO_NAME,
99  PlatformName.size(), PlatformName.data(), nullptr);
100  if (PlatformNameSize > 0u &&
101  std::strncmp(PlatformName.data(), "NVIDIA CUDA", PlatformNameSize) == 0) {
102  IsCUDA = true;
103  }
104 #endif // USE_PI_CUDA
105 
106  if (IsCUDA) {
107  // TODO: Reemplace CreateWithSource with CreateWithBinary in CUDA backend
108  const char *SignedData = reinterpret_cast<const char *>(Data);
109  Plugin.call<PiApiKind::piclProgramCreateWithSource>(Context->getHandleRef(), 1 /*one binary*/, &SignedData,
110  &DataLen, &Program);
111  } else {
112  RT::PiDevice Device = getFirstDevice(Context);
113  pi_int32 BinaryStatus = CL_SUCCESS;
114  Plugin.call<PiApiKind::piclProgramCreateWithBinary>(Context->getHandleRef(), 1 /*one binary*/, &Device,
115  &DataLen, &Data, &BinaryStatus,
116  &Program);
117  }
118 
119  return Program;
120 }

143 template <typename ExceptionT, typename RetT>
144 RetT *waitUntilBuilt(KernelProgramCache &Cache,
145  KernelProgramCache::BuildResult<RetT> *BuildResult) {
146  // any thread which will find nullptr in cache will wait until the pointer
147  // is not null anymore
148  Cache.waitUntilBuilt([BuildResult]() {
149  int State = BuildResult->State.load();
150 
151  return State == BS_Done || State == BS_Failed;
152  });
153 
154  if (BuildResult->Error.isFilledIn()) {
155  const KernelProgramCache::BuildError &Error = BuildResult->Error;
156  throw ExceptionT(Error.Msg, Error.Code);
157  }
158 
159  RetT *Result = BuildResult->Ptr.load();
160 
161  return Result;
162 }

181 template <typename RetT, typename ExceptionT, typename KeyT, typename AcquireFT,
182  typename GetCacheFT, typename BuildFT>
183 RetT *getOrBuild(KernelProgramCache &KPCache, const KeyT &CacheKey,
184  AcquireFT &&Acquire, GetCacheFT &&GetCache, BuildFT &&Build) {
185  bool InsertionTookPlace;
186  KernelProgramCache::BuildResult<RetT> *BuildResult;
187 
188  {
189  auto LockedCache = Acquire(KPCache);
190  auto &Cache = GetCache(LockedCache);
191  auto Inserted =
192  Cache.emplace(std::piecewise_construct, std::forward_as_tuple(CacheKey),
193  std::forward_as_tuple(nullptr, BS_InProgress));
194 
195  InsertionTookPlace = Inserted.second;
196  BuildResult = &Inserted.first->second;
197  }
198 
199  // no insertion took place, thus some other thread has already inserted smth
200  // in the cache
201  if (!InsertionTookPlace) {
202  for (;;) {
203  RetT *Result = waitUntilBuilt<ExceptionT>(KPCache, BuildResult);
204 
205  if (Result)
206  return Result;
207 
208  // Previous build is failed. There was no SYCL exception though.
209  // We might try to build once more.
210  int Expected = BS_Failed;
211  int Desired = BS_InProgress;
212 
213  if (BuildResult->State.compare_exchange_strong(Expected, Desired))
214  break; // this thread is the building thread now
215  }
216  }
217 
218  // only the building thread will run this
219  try {
220  RetT *Desired = Build();
221 
222 #ifndef NDEBUG
223  RetT *Expected = nullptr;
224 
225  if (!BuildResult->Ptr.compare_exchange_strong(Expected, Desired))
226  // We've got a funny story here
227  assert(false && "We've build an entity that is already have been built.");
228 #else
229  BuildResult->Ptr.store(Desired);
230 #endif
231 
232  BuildResult->State.store(BS_Done);
233 
234  KPCache.notifyAllBuild();
235 
236  return Desired;
237  } catch (const exception &Ex) {
238  BuildResult->Error.Msg = Ex.what();
239  BuildResult->Error.Code = Ex.get_cl_code();
240 
241  BuildResult->State.store(BS_Failed);
242 
243  KPCache.notifyAllBuild();
244 
245  std::rethrow_exception(std::current_exception());
246  } catch (...) {
247  BuildResult->State.store(BS_Failed);
248 
249  KPCache.notifyAllBuild();
250 
251  std::rethrow_exception(std::current_exception());
252  }
253 }

255 static bool isDeviceBinaryTypeSupported(const context &C,
256  RT::PiDeviceBinaryType Format) {
257  // All formats except PI_DEVICE_BINARY_TYPE_SPIRV are supported.
258  if (Format != PI_DEVICE_BINARY_TYPE_SPIRV)
259  return true;
260 
261  vector_class<device> Devices = C.get_devices();
262 
263  // Program type is SPIR-V, so we need a device compiler to do JIT.
264  for (const device &D : Devices) {
265  if (!D.get_info<info::device::is_compiler_available>())
266  return false;
267  }
268 
269  // OpenCL 2.1 and greater require clCreateProgramWithIL
270  if (pi::useBackend(pi::SYCL_BE_PI_OPENCL) &&
271  C.get_platform().get_info<info::platform::version>() >= "2.1")
272  return true;
273 
274  for (const device &D : Devices) {
275  // We need cl_khr_il_program extension to be present
276  // and we can call clCreateProgramWithILKHR using the extension
277  vector_class<string_class> Extensions =
278  D.get_info<info::device::extensions>();
279  if (Extensions.end() ==
280  std::find(Extensions.begin(), Extensions.end(), "cl_khr_il_program"))
281  return false;
282  }
283 
284  return true;
285 }

302 RT::PiProgram ProgramManager::createPIProgram(const DeviceImage &Img,
303  const context &Context) {
304  if (DbgProgMgr > 0)
305  std::cerr << ">>> ProgramManager::createPIProgram(" << &Img << ")\n";
306 
307  // perform minimal sanity checks on the device image and the descriptor
308  if (Img.BinaryEnd < Img.BinaryStart) {
309  throw runtime_error("Malformed device program image descriptor",
310  PI_INVALID_VALUE);
311  }
312  if (Img.BinaryEnd == Img.BinaryStart) {
313  throw runtime_error("Invalid device program image: size is zero",
314  PI_INVALID_VALUE);
315  }
316  size_t ImgSize = static_cast<size_t>(Img.BinaryEnd - Img.BinaryStart);
317 
318  // TODO if the binary image is a part of the fat binary, the clang
319  // driver should have set proper format option to the
320  // clang-offload-wrapper. The fix depends on AOT compilation
321  // implementation, so will be implemented together with it.
322  // Img->Format can't be updated as it is inside of the in-memory
323  // OS module binary.
324  RT::PiDeviceBinaryType Format = getFormat(Img);
325  // RT::PiDeviceBinaryType Format = Img->Format;
326  // assert(Format != PI_DEVICE_BINARY_TYPE_NONE && "Image format not set");
327 
328  if (!isDeviceBinaryTypeSupported(Context, Format))
329  throw feature_not_supported(
330  "Online compilation is not supported in this context",
331  PI_INVALID_OPERATION);
332 
333  // Load the image
334  const ContextImplPtr Ctx = getSyclObjImpl(Context);
335  RT::PiProgram Res = Format == PI_DEVICE_BINARY_TYPE_SPIRV
336  ? createSpirvProgram(Ctx, Img.BinaryStart, ImgSize)
337  : createBinaryProgram(Ctx, Img.BinaryStart, ImgSize);
338 
339  if (DbgProgMgr > 1)
340  std::cerr << "created program: " << Res
341  << "; image format: " << getFormatStr(Format) << "\n";
342 
343  return Res;
344 }

346 RT::PiProgram
347 ProgramManager::getBuiltPIProgram(OSModuleHandle M, const context &Context,
348  const string_class &KernelName) {
349  KernelSetId KSId = getKernelSetId(M, KernelName);
350 
351  const ContextImplPtr Ctx = getSyclObjImpl(Context);
352 
353  using PiProgramT = KernelProgramCache::PiProgramT;
354  using ProgramCacheT = KernelProgramCache::ProgramCacheT;
355 
356  KernelProgramCache &Cache = Ctx->getKernelProgramCache();
357 
358  auto AcquireF = [](KernelProgramCache &Cache) {
359  return Cache.acquireCachedPrograms();
360  };
361  auto GetF = [](const Locked<ProgramCacheT> &LockedCache) -> ProgramCacheT& {
362  return LockedCache.get();
363  };
364  auto BuildF = [this, &M, &KSId, &Context] {
365  const DeviceImage &Img = getDeviceImage(M, KSId, Context);
366 
367  ContextImplPtr ContextImpl = getSyclObjImpl(Context);
368  const detail::plugin &Plugin = ContextImpl->getPlugin();
369  RT::PiProgram Prg = createPIProgram(Img, Context);
370  ProgramPtr ProgramManaged(
371  Prg, Plugin.getPiPlugin().PiFunctionTable.piProgramRelease);
372 
373  // Link a fallback implementation of device libraries if they are not
374  // supported by a device compiler.
375  // Pre-compiled programs are supposed to be already linked.
376  const bool LinkDeviceLibs = getFormat(Img) == PI_DEVICE_BINARY_TYPE_SPIRV;
377 
378  const std::vector<device> &Devices = ContextImpl->getDevices();
379  std::vector<RT::PiDevice> PiDevices(Devices.size());
380  std::transform(
381  Devices.begin(), Devices.end(), PiDevices.begin(),
382  [](const device Dev) { return getRawSyclObjImpl(Dev)->getHandleRef(); });
383 
384  ProgramPtr BuiltProgram =
385  build(std::move(ProgramManaged), ContextImpl, Img.CompileOptions,
386  Img.LinkOptions, PiDevices, ContextImpl->getCachedLibPrograms(),
387  LinkDeviceLibs);
388 
389  return BuiltProgram.release();
390  };
391 
392  return getOrBuild<PiProgramT, compile_program_error>(Cache, KSId, AcquireF,
393  GetF, BuildF);
394 }

445 string_class ProgramManager::getProgramBuildLog(const RT::PiProgram &Program,
446  const ContextImplPtr Context) {
447  size_t Size = 0;
448  const detail::plugin &Plugin = Context->getPlugin();
449  Plugin.call<PiApiKind::piProgramGetInfo>(Program, PI_PROGRAM_INFO_DEVICES, 0,
450  nullptr, &Size);
451  vector_class<RT::PiDevice> PIDevices(Size / sizeof(RT::PiDevice));
452  Plugin.call<PiApiKind::piProgramGetInfo>(Program, PI_PROGRAM_INFO_DEVICES,

454  string_class Log = "The program was built for " +
455  std::to_string(PIDevices.size()) + " devices";
456  for (RT::PiDevice &Device : PIDevices) {
457  Plugin.call<PiApiKind::piProgramGetBuildInfo>(
458  Program, Device, CL_PROGRAM_BUILD_LOG, 0, nullptr, &Size);
459  vector_class<char> DeviceBuildInfo(Size);
460  Plugin.call<PiApiKind::piProgramGetBuildInfo>(
461  Program, Device, CL_PROGRAM_BUILD_LOG, Size, DeviceBuildInfo.data(),
462  nullptr);
463  Plugin.call<PiApiKind::piDeviceGetInfo>(Device, PI_DEVICE_INFO_NAME, 0,
464  nullptr, &Size);
465  vector_class<char> DeviceName(Size);
466  Plugin.call<PiApiKind::piDeviceGetInfo>(Device, PI_DEVICE_INFO_NAME, Size,
467  DeviceName.data(), nullptr);
468 
469  Log += "\nBuild program log for '" + string_class(DeviceName.data()) +
470  "':\n" + string_class(DeviceBuildInfo.data());
471  }
472  return Log;

475 static bool loadDeviceLib(const ContextImplPtr Context, const char *Name,
476  RT::PiProgram &Prog) {
477  std::string LibSyclDir = OSUtil::getCurrentDSODir();
478  std::ifstream File(LibSyclDir + OSUtil::DirSep + Name,
479  std::ifstream::in | std::ifstream::binary);
480  if (!File.good()) {
481  return false;
482  }
483 
484  File.seekg(0, std::ios::end);
485  size_t FileSize = File.tellg();
486  File.seekg(0, std::ios::beg);
487  std::vector<char> FileContent(FileSize);
488  File.read(&FileContent[0], FileSize);
489  File.close();
490 
491  Prog =
492  createSpirvProgram(Context, (unsigned char *)&FileContent[0], FileSize);
493  return Prog != nullptr;
494 }

530 static RT::PiProgram loadDeviceLibFallback(
531  const ContextImplPtr Context, DeviceLibExt Extension,
532  const std::vector<RT::PiDevice> &Devices,
533  std::map<DeviceLibExt, RT::PiProgram> &CachedLibPrograms) {
534 
535  const char *LibFileName = getDeviceLibFilename(Extension);
536  std::map<DeviceLibExt, RT::PiProgram>::iterator LibProgIt;
537  bool NotExists = false;
538  std::tie(LibProgIt, NotExists) =
539  CachedLibPrograms.insert({Extension, nullptr});
540  RT::PiProgram &LibProg = LibProgIt->second;
541 
542  if (!NotExists) {
543  return LibProg;
544  }
545 
546  if (!loadDeviceLib(Context, LibFileName, LibProg)) {
547  CachedLibPrograms.erase(LibProgIt);
548  throw compile_program_error(std::string("Failed to load ") + LibFileName,
549  PI_INVALID_VALUE);
550  }
551 
552  const detail::plugin &Plugin = Context->getPlugin();
553  RT::PiResult Error = Plugin.call_nocheck<PiApiKind::piProgramCompile>(
554  LibProg,
555  // Assume that Devices contains all devices from Context.
556  Devices.size(), Devices.data(),
557  // Do not use compile options for library programs: it is not clear
558  // if user options (image options) are supposed to be applied to
559  // library program as well, and what actually happens to a SPIR-V
560  // program if we apply them.
561  "", 0, nullptr, nullptr, nullptr, nullptr);
562  if (Error != PI_SUCCESS) {
563  CachedLibPrograms.erase(LibProgIt);
564  throw compile_program_error(
565  ProgramManager::getProgramBuildLog(LibProg, Context), Error);
566  }
567 
568  return LibProg;
569 }

578 ProgramManager::ProgramManager() {
579  const char *SpvFile = std::getenv(UseSpvEnv);
580  // If a SPIRV file is specified with an environment variable,
581  // register the corresponding image
582  if (SpvFile) {
583  m_UseSpvFile = true;
584  // The env var requests that the program is loaded from a SPIRV file on disk
585  std::ifstream File(SpvFile, std::ios::binary);
586 
587  if (!File.is_open())
588  throw runtime_error(std::string("Can't open file specified via ") +
589  UseSpvEnv + ": " + SpvFile,
590  PI_INVALID_VALUE);
591  File.seekg(0, std::ios::end);
592  size_t Size = File.tellg();
593  std::unique_ptr<unsigned char[]> Data(new unsigned char[Size]);
594  File.seekg(0);
595  File.read(reinterpret_cast<char *>(Data.get()), Size);
596  File.close();
597  if (!File.good())
598  throw runtime_error(std::string("read from ") + SpvFile +
599  std::string(" failed"),
600  PI_INVALID_VALUE);
601 
602  std::unique_ptr<DeviceImage, ImageDeleter> ImgPtr(new DeviceImage(),
603  ImageDeleter());
604  ImgPtr->Version = PI_DEVICE_BINARY_VERSION;
605  ImgPtr->Kind = PI_DEVICE_BINARY_OFFLOAD_KIND_SYCL;
606  ImgPtr->DeviceTargetSpec = PI_DEVICE_BINARY_TARGET_UNKNOWN;
607  ImgPtr->CompileOptions = "";
608  ImgPtr->LinkOptions = "";
609  ImgPtr->ManifestStart = nullptr;
610  ImgPtr->ManifestEnd = nullptr;
611  ImgPtr->BinaryStart = Data.release();
612  ImgPtr->BinaryEnd = ImgPtr->BinaryStart + Size;
613  ImgPtr->EntriesBegin = nullptr;
614  ImgPtr->EntriesEnd = nullptr;
615  // TODO the environment variable name implies that the only binary format
616  // it accepts is SPIRV but that is not the case, should be aligned?
617  ImgPtr->Format = getFormat(*ImgPtr);
618 
619  // No need for a mutex here since all access to these private fields is
620  // blocked until the construction of the ProgramManager singleton is
621  // finished.
622  m_DeviceImages[SpvFileKSId].reset(
623  new std::vector<DeviceImage *>({ImgPtr.get()}));
624 
625  m_OrphanDeviceImages.emplace_back(std::move(ImgPtr));
626 
627  if (DbgProgMgr > 0)
628  std::cerr << "loaded device image from " << SpvFile << "\n";
629  }
630 }

666 // TODO: getDeviceLibPrograms should also support Windows but
667 // current implementation doesn't work on Windows when multiple
668 // device libraries exist and this problem should be fixed.
669 static std::vector<RT::PiProgram>
670 getDeviceLibPrograms(const ContextImplPtr Context,
671  const std::vector<RT::PiDevice> &Devices,
672  std::map<DeviceLibExt, RT::PiProgram> &CachedLibPrograms) {
673  std::vector<RT::PiProgram> Programs;
674 
675  // TODO: SYCL compiler should generate a list of required extensions for a
676  // particular program in order to allow us do a more fine-grained check here.
677  // Require *all* possible devicelib extensions for now.
678  std::pair<DeviceLibExt, bool> RequiredDeviceLibExt[] = {
679  {cl_intel_devicelib_assert, /* is fallback loaded? */ false},
680  {cl_intel_devicelib_math, false},
681  {cl_intel_devicelib_math_fp64, false},
682  {cl_intel_devicelib_complex, false},
683  {cl_intel_devicelib_complex_fp64, false}
684  };
685 
686  // Disable all devicelib extensions requiring fp64 support if at least
687  // one underlying device doesn't support cl_khr_fp64.
688  bool fp64Support = true;
689  for (RT::PiDevice Dev : Devices) {
690  std::string DevExtList =
691  get_device_info<std::string, info::device::extensions>::get(
692  Dev, Context->getPlugin());
693  fp64Support = fp64Support &&
694  (DevExtList.npos != DevExtList.find("cl_khr_fp64"));
695  }
696 
697  // Load a fallback library for an extension if at least one device does not
698  // support it.
699  for (RT::PiDevice Dev : Devices) {
700  std::string DevExtList =
701  get_device_info<std::string, info::device::extensions>::get(
702  Dev, Context->getPlugin());
703  for (auto &Pair : RequiredDeviceLibExt) {
704  DeviceLibExt Ext = Pair.first;
705  bool &FallbackIsLoaded = Pair.second;
706 
707  if (FallbackIsLoaded) {
708  continue;
709  }
710 
711  if ((Ext == cl_intel_devicelib_math_fp64 ||
712  Ext == cl_intel_devicelib_complex_fp64) && !fp64Support) {
713  continue;
714  }
715 
716  const char* ExtStr = getDeviceLibExtensionStr(Ext);
717 
718  bool InhibitNativeImpl = false;
719  if (const char *Env = getenv("SYCL_DEVICELIB_INHIBIT_NATIVE")) {
720  InhibitNativeImpl = strstr(Env, ExtStr) != nullptr;
721  }
722 
723  bool DeviceSupports = DevExtList.npos != DevExtList.find(ExtStr);
724 
725  if (!DeviceSupports || InhibitNativeImpl) {
726  Programs.push_back(
727  loadDeviceLibFallback(Context, Ext, Devices, CachedLibPrograms));
728  FallbackIsLoaded = true;
729  }
730  }
731  }
732  return Programs;
733 }

735 ProgramManager::ProgramPtr
736 ProgramManager::build(ProgramPtr Program, const ContextImplPtr Context,
737  const string_class &CompileOptions,
738  const string_class &LinkOptions,
739  const std::vector<RT::PiDevice> &Devices,
740  std::map<DeviceLibExt, RT::PiProgram> &CachedLibPrograms,
741  bool LinkDeviceLibs) {
742 
743  if (DbgProgMgr > 0) {
744  std::cerr << ">>> ProgramManager::build(" << Program.get() << ", "
745  << CompileOptions << ", " << LinkOptions << ", ... "
746  << Devices.size() << ")\n";
747  }
748 
749  const char *CompileOpts = std::getenv("SYCL_PROGRAM_COMPILE_OPTIONS");
750  if (!CompileOpts) {
751  CompileOpts = CompileOptions.c_str();
752  }
753  const char *LinkOpts = std::getenv("SYCL_PROGRAM_LINK_OPTIONS");
754  if (!LinkOpts) {
755  LinkOpts = LinkOptions.c_str();
756  }
757 
758  std::vector<RT::PiProgram> LinkPrograms;
759  if (LinkDeviceLibs) {
760  LinkPrograms = getDeviceLibPrograms(Context, Devices, CachedLibPrograms);
761  }
762 
763  const detail::plugin &Plugin = Context->getPlugin();
764  if (LinkPrograms.empty()) {
765  std::string Opts(CompileOpts);
766  Opts += " ";
767  Opts += LinkOpts;
768 
769  RT::PiResult Error = Plugin.call_nocheck<PiApiKind::piProgramBuild>(
770  Program.get(), Devices.size(), Devices.data(), Opts.c_str(), nullptr,
771  nullptr);
772  if (Error != PI_SUCCESS)
773  throw compile_program_error(getProgramBuildLog(Program.get(), Context),
774  Error);
775  return Program;
776  }
777 
778  // Include the main program and compile/link everything together
779  Plugin.call<PiApiKind::piProgramCompile>(Program.get(), Devices.size(),
780  Devices.data(), CompileOpts, 0,
781  nullptr, nullptr, nullptr, nullptr);
782  LinkPrograms.push_back(Program.get());
783 
784  RT::PiProgram LinkedProg = nullptr;
785  RT::PiResult Error = Plugin.call_nocheck<PiApiKind::piProgramLink>(
786  Context->getHandleRef(), Devices.size(), Devices.data(), LinkOpts,
787  LinkPrograms.size(), LinkPrograms.data(), nullptr, nullptr, &LinkedProg);
788 
789  // Link program call returns a new program object if all parameters are valid,
790  // or NULL otherwise. Release the original (user) program.
791  Program.reset(LinkedProg);
792  if (Error != PI_SUCCESS) {
793  if (LinkedProg) {
794  // A non-trivial error occurred during linkage: get a build log, release
795  // an incomplete (but valid) LinkedProg, and throw.
796  throw compile_program_error(getProgramBuildLog(LinkedProg, Context),
797  Error);
798  }
799  Plugin.checkPiResult(Error);
800  }
801  return Program;
802 }

804 void ProgramManager::addImages(pi_device_binaries DeviceBinary) {
805  std::lock_guard<std::mutex> Guard(Sync::getGlobalLock());
806 
807  for (int I = 0; I < DeviceBinary->NumDeviceBinaries; I++) {
808  pi_device_binary Img = &(DeviceBinary->DeviceBinaries[I]);
809  OSModuleHandle M = OSUtil::getOSModuleHandle(Img);
810  const _pi_offload_entry EntriesB = Img->EntriesBegin;
811  const _pi_offload_entry EntriesE = Img->EntriesEnd;
812  // Use the entry information if it's available
813  if (EntriesB != EntriesE) {
814  // The kernel sets for any pair of images are either disjoint or
815  // identical, look up the kernel set using the first kernel name...
816  StrToKSIdMap &KSIdMap = m_KernelSets[M];
817  auto KSIdIt = KSIdMap.find(EntriesB->name);
818  if (KSIdIt != KSIdMap.end()) {
819  for (_pi_offload_entry EntriesIt = EntriesB + 1; EntriesIt != EntriesE;
820  ++EntriesIt)
821  assert(KSIdMap[EntriesIt->name] == KSIdIt->second &&
822  "Kernel sets are not disjoint");
823  auto &Imgs = m_DeviceImages[KSIdIt->second];
824  assert(Imgs && "Device image vector should have been already created");
825  Imgs->push_back(Img);
826  continue;
827  }
828  // ... or create the set first if it hasn't been
829  KernelSetId KSId = getNextKernelSetId();
830  for (_pi_offload_entry EntriesIt = EntriesB; EntriesIt != EntriesE;
831  ++EntriesIt) {
832  auto Result =
833  KSIdMap.insert(std::make_pair(EntriesIt->name, KSId));
834  (void)Result;
835  assert(Result.second && "Kernel sets are not disjoint");
836  }
837  m_DeviceImages[KSId].reset(new std::vector<DeviceImage *>({Img}));
838  continue;
839  }
840  // Otherwise assume that the image contains all kernels associated with the
841  // module
842  KernelSetId &KSId = m_OSModuleKernelSets[M];
843  if (KSId == 0)
844  KSId = getNextKernelSetId();
845 
846  auto &Imgs = m_DeviceImages[KSId];
847  if (!Imgs)
848  Imgs.reset(new std::vector<DeviceImage *>({Img}));
849  else
850  Imgs->push_back(Img);
851  }
852 }

915 RT::PiDeviceBinaryType ProgramManager::getFormat(const DeviceImage &Img) const {
916  if (Img.Format != PI_DEVICE_BINARY_TYPE_NONE)
917  return Img.Format;
918 
919  struct {
920  RT::PiDeviceBinaryType Fmt;
921  const uint32_t Magic;
922  } Fmts[] = {{PI_DEVICE_BINARY_TYPE_SPIRV, 0x07230203},
923  {PI_DEVICE_BINARY_TYPE_LLVMIR_BITCODE, 0xDEC04342}};
924 
925  size_t ImgSize = static_cast<size_t>(Img.BinaryEnd - Img.BinaryStart);
926  if (ImgSize >= sizeof(Fmts[0].Magic)) {
927  std::remove_const<decltype(Fmts[0].Magic)>::type Hdr = 0;
928  std::copy(Img.BinaryStart, Img.BinaryStart + sizeof(Hdr),
929  reinterpret_cast<char *>(&Hdr));
930 
931  for (const auto &Fmt : Fmts) {
932  if (Hdr == Fmt.Magic) {
933  if (DbgProgMgr > 1)
934  std::cerr << "determined image format: " << getFormatStr(Fmt.Fmt)
935  << "\n";
936  return Fmt.Fmt;
937  }
938  }
939  }
940 
941  return PI_DEVICE_BINARY_TYPE_NONE;
942 }

966  F.write(reinterpret_cast<const char *>(Img.BinaryStart), ImgSize);
