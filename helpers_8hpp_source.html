172 inline constexpr uint32_t
173 getSPIRVMemorySemanticsMask(const access::fence_space AccessSpace,
174  const __spv::MemorySemanticsMask LocalScopeMask =
175  __spv::MemorySemanticsMask::WorkgroupMemory) {
176  // Huge ternary operator below is a workaround for constexpr function
177  // requirement that such function can only contain return statement and
178  // nothing more
179  //
180  // It is equivalent to the following code:
181  //
182  // uint32_t Flags =
183  // static_cast<uint32_t>(__spv::MemorySemanticsMask::SequentiallyConsistent);
184  // switch (AccessSpace) {
185  // case access::fence_space::global_space:
186  // Flags |=
187  // static_cast<uint32_t>(__spv::MemorySemanticsMask::CrossWorkgroupMemory);
188  // break;
189  // case access::fence_space::local_space:
190  // Flags |= static_cast<uint32_t>(LocalScopeMask);
191  // break;
192  // case access::fence_space::global_and_local:
193  // default:
194  // Flags |= static_cast<uint32_t>(
195  // __spv::MemorySemanticsMask::CrossWorkgroupMemory) |
196  // static_cast<uint32_t>(LocalScopeMask);
197  // break;
198  // }
199  // return Flags;
200 
201  return (AccessSpace == access::fence_space::global_space)
202  ? static_cast<uint32_t>(
203  __spv::MemorySemanticsMask::SequentiallyConsistent |
204  __spv::MemorySemanticsMask::CrossWorkgroupMemory)
205  : (AccessSpace == access::fence_space::local_space)
206  ? static_cast<uint32_t>(
207  __spv::MemorySemanticsMask::SequentiallyConsistent |
208  LocalScopeMask)
209  : /* default: (AccessSpace ==
210  access::fence_space::global_and_local) */
211  static_cast<uint32_t>(
212  __spv::MemorySemanticsMask::SequentiallyConsistent |
213  __spv::MemorySemanticsMask::CrossWorkgroupMemory |
214  LocalScopeMask);
215 }
