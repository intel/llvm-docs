15 // Truncates a wide (16 or 32 bit) string (wstr) into an ASCII string (str).
16 // Any non-ASCII characters are replaced by question mark '?'.
17 static void __truncate_wchar_char_str(const wchar_t *wstr, char *str,
18  size_t str_size) {
19  str_size -= 1; // reserve for NULL terminator
20  while (str_size > 0 && *wstr != L'\0') {
21  wchar_t w = *wstr++;
22  *str++ = (w > 0 && w < 127) ? (char)w : '?';
23  str_size--;
24  }
25  *str = '\0';
26 }
